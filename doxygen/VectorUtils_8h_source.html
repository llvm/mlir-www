50 /// Return true if `vectorType` is a contiguous slice of `memrefType`,
51 /// in the sense that it can be read/written from/to a contiguous area
52 /// of the memref.
53 ///
54 /// The leading unit dimensions of the vector type are ignored as they
55 /// are not relevant to the result. Let N be the number of the vector
56 /// dimensions after ignoring a leading sequence of unit ones.
57 ///
58 /// For `vectorType` to be a contiguous slice of `memrefType`
59 /// a) the N trailing dimensions of `memrefType` must be contiguous, and
60 /// b) the N-1 trailing dimensions of `vectorType` and `memrefType`
61 /// must match.
62 ///
63 /// Examples:
64 ///
65 /// Ex.1 contiguous slice, perfect match
66 /// vector<4x3x2xi32> from memref<5x4x3x2xi32>
67 /// Ex.2 contiguous slice, the leading dim does not match (2 != 4)
68 /// vector<2x3x2xi32> from memref<5x4x3x2xi32>
69 /// Ex.3 non-contiguous slice, 2 != 3
70 /// vector<2x2x2xi32> from memref<5x4x3x2xi32>
71 /// Ex.4 contiguous slice, leading unit dimension of the vector ignored,
72 /// 2 != 3 (allowed)
73 /// vector<1x2x2xi32> from memref<5x4x3x2xi32>
74 /// Ex.5. contiguous slice, leading two unit dims of the vector ignored,
75 /// 2 != 3 (allowed)
76 /// vector<1x1x2x2xi32> from memref<5x4x3x2xi32>
77 /// Ex.6. non-contiguous slice, 2 != 3, no leading sequence of unit dims
78 /// vector<2x1x2x2xi32> from memref<5x4x3x2xi32>)
79 /// Ex.7 contiguous slice, memref needs to be contiguous only in the last
80 /// dimension
81 /// vector<1x1x2xi32> from memref<2x2x2xi32, strided<[8, 4, 1]>>
82 /// Ex.8 non-contiguous slice, memref needs to be contiguous in the last
83 /// two dimensions, and it isn't
84 /// vector<1x2x2xi32> from memref<2x2x2xi32, strided<[8, 4, 1]>>
85 bool isContiguousSlice(MemRefType memrefType, VectorType vectorType);

144 /// A pattern for ops that implement `MaskableOpInterface` and that _might_ be
145 /// masked (i.e. inside `vector.mask` Op region). In particular:
146 /// 1. Matches `SourceOp` operation, Op.
147 /// 2.1. If Op is masked, retrieves the masking Op, maskOp, and updates the
148 /// insertion point to avoid inserting new ops into the `vector.mask` Op
149 /// region (which only allows one Op).
150 /// 2.2 If Op is not masked, this step is skipped.
151 /// 3. Invokes `matchAndRewriteMaskableOp` on Op and optionally maskOp if
152 /// found in step 2.1.
153 ///
154 /// This wrapper frees patterns from re-implementing the logic to update the
155 /// insertion point when a maskable Op is masked. Such patterns are still
156 /// responsible for providing an updated ("rewritten") version of:
157 /// a. the source Op when mask _is not_ present,
158 /// b. the source Op and the masking Op when mask _is_ present.
159 /// To use this pattern, implement `matchAndRewriteMaskableOp`. Note that
160 /// the return value will depend on the case above.
161 template <class SourceOp>
162 struct MaskableOpRewritePattern : OpRewritePattern<SourceOp> {
163  using OpRewritePattern<SourceOp>::OpRewritePattern;
164 
165 private:
166  LogicalResult matchAndRewrite(SourceOp sourceOp,
167  PatternRewriter &rewriter) const final {
168  auto maskableOp = dyn_cast<MaskableOpInterface>(sourceOp.getOperation());
169  if (!maskableOp)
170  return failure();
171 
172  Operation *rootOp = sourceOp;
173 
174  // If this Op is masked, update the insertion point to avoid inserting into
175  // the vector.mask Op region.
176  OpBuilder::InsertionGuard guard(rewriter);
177  MaskingOpInterface maskOp;
178  if (maskableOp.isMasked()) {
179  maskOp = maskableOp.getMaskingOp();
180  rewriter.setInsertionPoint(maskOp);
181  rootOp = maskOp;
182  }
183 
184  FailureOr<Value> newOp =
185  matchAndRewriteMaskableOp(sourceOp, maskOp, rewriter);
186  if (failed(newOp))
187  return failure();
188 
189  // Rewriting succeeded but there are no values to replace.
190  if (rootOp->getNumResults() == 0) {
191  rewriter.eraseOp(rootOp);
192  } else {
193  assert(*newOp != Value() &&
194  "Cannot replace an op's use with an empty value.");
195  rewriter.replaceOp(rootOp, *newOp);
196  }
197  return success();
198  }
199 
200 public:
201  // Matches `sourceOp` that can potentially be masked with `maskingOp`. If the
202  // latter is present, returns a replacement for `maskingOp`. Otherwise,
203  // returns a replacement for `sourceOp`.
204  virtual FailureOr<Value>
205  matchAndRewriteMaskableOp(SourceOp sourceOp, MaskingOpInterface maskingOp,
206  PatternRewriter &rewriter) const = 0;
207 };

243 /// Constructs a permutation map of invariant memref indices to vector
244 /// dimension.
245 ///
246 /// If no index is found to be invariant, 0 is added to the permutation_map and
247 /// corresponds to a vector broadcast along that dimension.
248 ///
249 /// The implementation uses the knowledge of the mapping of loops to
250 /// vector dimension. `loopToVectorDim` carries this information as a map with:
251 /// - keys representing "vectorized enclosing loops";
252 /// - values representing the corresponding vector dimension.
253 /// Note that loopToVectorDim is a whole function map from which only enclosing
254 /// loop information is extracted.
255 ///
256 /// Prerequisites: `indices` belong to a vectorizable load or store operation
257 /// (i.e. at most one invariant index along each AffineForOp of
258 /// `loopToVectorDim`). `insertPoint` is the insertion point for the vectorized
259 /// load or store operation.
260 ///
261 /// Example 1:
262 /// The following MLIR snippet:
263 ///
264 /// ```mlir
265 /// affine.for %i3 = 0 to %0 {
266 /// affine.for %i4 = 0 to %1 {
267 /// affine.for %i5 = 0 to %2 {
268 /// %a5 = load %arg0[%i4, %i5, %i3] : memref<?x?x?xf32>
269 /// }}}
270 /// ```
271 ///
272 /// may vectorize with {permutation_map: (d0, d1, d2) -> (d2, d1)} into:
273 ///
274 /// ```mlir
275 /// affine.for %i3 = 0 to %0 step 32 {
276 /// affine.for %i4 = 0 to %1 {
277 /// affine.for %i5 = 0 to %2 step 256 {
278 /// %4 = vector.transfer_read %arg0, %i4, %i5, %i3
279 /// {permutation_map: (d0, d1, d2) -> (d2, d1)} :
280 /// (memref<?x?x?xf32>, index, index) -> vector<32x256xf32>
281 /// }}}
282 /// ```
283 ///
284 /// Meaning that vector.transfer_read will be responsible for reading the slice:
285 /// `%arg0[%i4, %i5:%15+256, %i3:%i3+32]` into vector<32x256xf32>.
286 ///
287 /// Example 2:
288 /// The following MLIR snippet:
289 ///
290 /// ```mlir
291 /// %cst0 = arith.constant 0 : index
292 /// affine.for %i0 = 0 to %0 {
293 /// %a0 = load %arg0[%cst0, %cst0] : memref<?x?xf32>
294 /// }
295 /// ```
296 ///
297 /// may vectorize with {permutation_map: (d0) -> (0)} into:
298 ///
299 /// ```mlir
300 /// affine.for %i0 = 0 to %0 step 128 {
301 /// %3 = vector.transfer_read %arg0, %c0_0, %c0_0
302 /// {permutation_map: (d0, d1) -> (0)} :
303 /// (memref<?x?xf32>, index, index) -> vector<128xf32>
304 /// }
305 /// ````
306 ///
307 /// Meaning that vector.transfer_read will be responsible of reading the slice
308 /// `%arg0[%c0, %c0]` into vector<128xf32> which needs a 1-D vector broadcast.
309 ///
310 AffineMap
311 makePermutationMap(Block *insertPoint, ArrayRef<Value> indices,
312  const DenseMap<Operation *, unsigned> &loopToVectorDim);
