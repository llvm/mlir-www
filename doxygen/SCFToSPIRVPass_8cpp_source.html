36void SCFToSPIRVPass::runOnOperation() {
37 MLIRContext *context = &getContext();
38 Operation *op = getOperation();
39
40 auto targetAttr = spirv::lookupTargetEnvOrDefault(op);
41 std::unique_ptr<ConversionTarget> target =
42 SPIRVConversionTarget::get(targetAttr);
43
44 SPIRVTypeConverter typeConverter(targetAttr);
45 ScfToSPIRVContext scfContext;
46 RewritePatternSet patterns(context);
47 populateSCFToSPIRVPatterns(typeConverter, scfContext, patterns);
48
49 // TODO: Change SPIR-V conversion to be progressive and remove the following
50 // patterns.
51 mlir::arith::populateArithToSPIRVPatterns(typeConverter, patterns);
52 populateFuncToSPIRVPatterns(typeConverter, patterns);
53 populateMemRefToSPIRVPatterns(typeConverter, patterns);
54 populateBuiltinFuncToSPIRVPatterns(typeConverter, patterns);
55 index::populateIndexToSPIRVPatterns(typeConverter, patterns);
56
57 if (failed(applyPartialConversion(op, *target, std::move(patterns))))
58 return signalPassFailure();
59}
