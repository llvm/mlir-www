47// Helper visitor to determine whether an AffineExpr is tiled.
48// This is achieved by traversing every AffineDimExpr with position `pos` and
49// checking whether the corresponding `tileSizes[pos]` is non-zero.
50// This also enforces only positive coefficients occur in multiplications.
51//
52// Example:
53// `d0 + 2 * d1 + d3` is tiled by [0, 0, 0, 2] but not by [0, 0, 2, 0]
54//
55struct TileCheck : public AffineExprVisitor<TileCheck> {
56 TileCheck(ArrayRef<OpFoldResult> tileSizes) : tileSizes(tileSizes) {}
57
58 void visitDimExpr(AffineDimExpr expr) {
59 isTiled |= !isZeroInteger(tileSizes[expr.getPosition()]);
60 }
61 void visitAffineBinaryOpExpr(AffineBinaryOpExpr expr) {
62 visit(expr.getLHS());
63 visit(expr.getRHS());
64 if (expr.getKind() == mlir::AffineExprKind::Mul)
65 assert(cast<AffineConstantExpr>(expr.getRHS()).getValue() > 0 &&
66 "nonpositive multiplying coefficient");
67 }
68 bool isTiled = false;
69 ArrayRef<OpFoldResult> tileSizes;
70};
71
72} // namespace

74static bool isTiled(AffineExpr expr, ArrayRef<OpFoldResult> tileSizes) {
75 if (!expr)
76 return false;
77 TileCheck t(tileSizes);
78 t.visit(expr);
79 return t.isTiled;
80}

82// Checks whether the `map varies with respect to a non-zero `tileSize`.
83static bool isTiled(AffineMap map, ArrayRef<OpFoldResult> tileSizes) {
84 if (!map)
85 return false;

92std::optional<RegionMatcher::BinaryOpKind>
93RegionMatcher::matchAsScalarBinaryOp(GenericOp op) {
94 auto &region = op.getRegion();
95 if (!region.hasOneBlock())
96 return std::nullopt;
97
98 Block &block = region.front();
99 if (block.getNumArguments() != 2 ||
100 !block.getArgument(0).getType().isSignlessIntOrFloat() ||
101 !block.getArgument(1).getType().isSignlessIntOrFloat())
102 return std::nullopt;
103
104 auto &ops = block.getOperations();
105 if (!llvm::hasSingleElement(block.without_terminator()))
106 return std::nullopt;
107
108 using mlir::matchers::m_Val;
109 auto a = m_Val(block.getArgument(0));
110 auto b = m_Val(block.getArgument(1));
111
112 auto addPattern = m_Op<linalg::YieldOp>(m_Op<arith::AddIOp>(a, b));
113 if (addPattern.match(&ops.back()))
114 return BinaryOpKind::IAdd;
115
116 return std::nullopt;
117}

143/// The permutation can be obtained from two permutations:
144/// a) Compute the permutation vector to move the last `numPackedDims` into
145/// the `innerPosDims` of a shape of rank `rank`.
146/// b) Compute the permutation vector to move outer dims if the
147/// `outerPerm` parameter is not empty.
148/// Apply (b) permutation on (a) permutation to get the final permutation.
149static SmallVector<int64_t>
150computePackUnPackPerm(int64_t rank, ArrayRef<int64_t> &innerDimsPos,
151 ArrayRef<int64_t> &outerPerm,
152 PackingMetadata &packingMetadata) {
153 int64_t numPackedDims = innerDimsPos.size();
154 auto lastDims =
155 llvm::to_vector(llvm::seq<int64_t>(rank - numPackedDims, rank));
156 packingMetadata = computePackingMetadata(rank, innerDimsPos);
157 SmallVector<int64_t> innerPositionsPerm =
158 computePermutationVector(rank, lastDims, packingMetadata.insertPositions);
159
160 SmallVector<int64_t> outerPos = packingMetadata.outerPositions;
161 if (!outerPerm.empty())
162 applyPermutationToVector(outerPos, outerPerm);
163 SmallVector<int64_t> outerPositionPerm =
164 computePermutationVector(rank, packingMetadata.outerPositions, outerPos);
165
166 SmallVector<int64_t> packInverseDestPermutation = innerPositionsPerm;
167 applyPermutationToVector(packInverseDestPermutation, outerPositionPerm);
168 return packInverseDestPermutation;
169}

202 if (!r.hasOneBlock())

204 for (Operation &op : r.front()) {

259/// Utility function to match the zero point offset body of quantized
260/// convolution ops.
261///
262/// Quantized convolutions have a body of the form:
263/// %out + ((%input - %inputZp) * (%filter - %filterZp))
264/// where:
265/// - %input is the input tensor element (block arg 0)
266/// - %filter is the filter tensor element (block arg 1)
267/// - %inputZp is the input zero-point scalar (block arg 2)
268/// - %filterZp is the filter zero-point scalar (block arg 3)
269/// - %out is the output accumulator (block arg 4)
270///
271/// This function verifies that the multiplication operands are subtraction
272/// operations matching this pattern.
273static bool bodyMatcherForZeroPointOffsets(Operation *addOp, Operation *mulOp,
274 Block *body) {
275 // The multiplication should have two subtraction operands:
276 // one for (input - inputZp) and one for (filter - filterZp).
277 Operation *inputSubOp = mulOp->getOperand(0).getDefiningOp();
278 if (!isa_and_present<arith::SubIOp, arith::SubFOp>(inputSubOp))
279 return false;
280
281 Operation *filterSubOp = mulOp->getOperand(1).getDefiningOp();
282 if (!isa_and_present<arith::SubIOp, arith::SubFOp>(filterSubOp))
283 return false;
284
285 // Extract block arguments from subtraction operands.
286 BlockArgument inputBlockArg =
287 getBlockArgumentWithOptionalCastOps(inputSubOp->getOperand(0));
288 BlockArgument inputZpBlockArg =
289 getBlockArgumentWithOptionalCastOps(inputSubOp->getOperand(1));
290 BlockArgument filterBlockArg =
291 getBlockArgumentWithOptionalCastOps(filterSubOp->getOperand(0));
292 BlockArgument filterZpBlockArg =
293 getBlockArgumentWithOptionalCastOps(filterSubOp->getOperand(1));
294 BlockArgument outBlockArg =
295 getBlockArgumentWithOptionalCastOps(addOp->getOperand(0));
296
297 // Verify all block arguments are valid.
298 if (!inputBlockArg || !inputZpBlockArg || !filterBlockArg ||
299 !filterZpBlockArg || !outBlockArg)
300 return false;
301
302 // Verify all block arguments belong to the convolution body.
303 if (inputBlockArg.getOwner() != body || inputZpBlockArg.getOwner() != body ||
304 filterBlockArg.getOwner() != body ||
305 filterZpBlockArg.getOwner() != body || outBlockArg.getOwner() != body)
306 return false;
307
308 // Verify block arguments have expected indices:
309 // arg0: input, arg1: filter, arg2: inputZp, arg3: filterZp, arg4: output
310 if (inputBlockArg.getArgNumber() != 0 || filterBlockArg.getArgNumber() != 1 ||
311 inputZpBlockArg.getArgNumber() != 2 ||
312 filterZpBlockArg.getArgNumber() != 3 || outBlockArg.getArgNumber() != 4)
313 return false;
314
315 return true;
316}

318/// Utility to match block body for convolution ops.
319/// The body is thus expected to yield :-
320/// %out + (%lhs * %rhs)
321/// where: %lhs, %rhs and %out are block arguments and
322/// %lhs and %rhs can have optional upcast operation.
323/// NOTE: In case of zero point offset convolution ops %lhs and %rhs would be :-
324/// %input - %input_scalar
325/// where, %input_scalar can have optional upcast operation.
326static bool bodyMatcherForConvolutionOps(Value yieldVal, Block *body,
327 bool containsZeroPointOffset = false) {
328 Operation *addOp = yieldVal.getDefiningOp();
329 if (!isa_and_present<arith::AddIOp, arith::AddFOp>(addOp))
330 return false;
331
332 Operation *mulOp = addOp->getOperand(1).getDefiningOp();
333 if (!isa_and_present<arith::MulIOp, arith::MulFOp>(mulOp))
334 return false;
335
336 if (containsZeroPointOffset) {
337 return bodyMatcherForZeroPointOffsets(addOp, mulOp, body);
338 }
339 BlockArgument lhsBlockArg =
340 getBlockArgumentWithOptionalCastOps(mulOp->getOperand(0));
341 BlockArgument rhsBlockArg =
342 getBlockArgumentWithOptionalCastOps(mulOp->getOperand(1));
343 BlockArgument outBlockArg =
344 getBlockArgumentWithOptionalCastOps(addOp->getOperand(0));
345 if (!lhsBlockArg || !rhsBlockArg || !outBlockArg ||
346 lhsBlockArg.getOwner() != body || rhsBlockArg.getOwner() != body ||
347 outBlockArg.getOwner() != body || lhsBlockArg.getArgNumber() != 0 ||
348 rhsBlockArg.getArgNumber() != 1 || outBlockArg.getArgNumber() != 2)
349 return false;
350 return true;
351}

353/// Utility to match block body for linalg.pool* ops.
354template <typename... OpTypes>
355static bool bodyMatcherForPoolOps(Value yieldVal, Block *body) {
356 Operation *defOp = yieldVal.getDefiningOp();
357 if (!(isa_and_present<OpTypes>(defOp) || ...))
358 return false;
359
360 BlockArgument lhsArg =
361 getBlockArgumentWithOptionalCastOps(defOp->getOperand(0));
362 BlockArgument rhsArg =
363 getBlockArgumentWithOptionalCastOps(defOp->getOperand(1));
364 if (!lhsArg || !rhsArg || lhsArg.getOwner() != body ||
365 rhsArg.getOwner() != body || lhsArg.getArgNumber() != 2 ||
366 rhsArg.getArgNumber() != 0)
367 return false;
368 return true;
369}

407/// Check if `expr` is either:
408/// - a dimension expr alone (implying multiplication by 1), or
409/// - a multiplication of dimension expr by any positive constant != 1
410/// In both cases we will capture the dimension expression into `dim` and
411/// return the constant multiplier. Returns -1 in case of a match failure.
412static int64_t isDimTimesConstantOrDimOnly(AffineExpr expr, AffineExpr &dim) {
413 if ((dim = dyn_cast<AffineDimExpr>(expr)))
414 return 1;
415
416 auto mulExpr = dyn_cast<AffineBinaryOpExpr>(expr);
417 if (!mulExpr || mulExpr.getKind() != AffineExprKind::Mul)
418 return -1;
419
420 AffineExpr lhs = mulExpr.getLHS();
421 AffineExpr rhs = mulExpr.getRHS();
422
423 AffineConstantExpr cst = nullptr;
424 if (((dim = dyn_cast<AffineDimExpr>(lhs)) &&
425 (cst = dyn_cast<AffineConstantExpr>(rhs))) ||
426 ((dim = dyn_cast<AffineDimExpr>(rhs)) &&
427 (cst = dyn_cast<AffineConstantExpr>(lhs))))
428 return cst.getValue();
429 return -1;
430}

432/// Given an array of AffineMaps `indexingMaps` verify the following
433/// commutatively:-
434/// indexingMaps[0].getResult(iDim) ==
435/// indexingMaps[1].getResult(fDim) * <c0> +
436/// indexingMaps[n-1].getResult(oDim) * <c1>
437/// where,
438/// - c0 and c1 can be any constant,
439/// - n is the size of the indexingMaps' array,
440/// - 0, 1 and n-1 are input, filter and output map indices respectively,
441/// - iDim, fDim and oDim are the input, filter and output dimension
442/// indices in their respective indexing maps
443/// Example:
444/// #inputMap = affine_map<(d0, d1, d2, d3, d4, d5, d6)
445/// -> (d0, d1 * 2 + d4 * 3, d2 + d5, d6)>
446/// #filterMap = affine_map<(d0, d1, d2, d3, d4, d5, d6) -> (d4, d5, d6, d3)>
447/// #outputMap = affine_map<(d0, d1, d2, d3, d4, d5, d6) -> (d0, d1, d2, d3)>
448///
449/// Here,
450/// #inputMap[1] = #outputMap[1] * 2 + #filterMap[0] * 3
451/// Therefore,
452/// matchConvDimAddExprPattern(indexingMaps, 1, 0, 1, dilation, stride)
453/// would return true and update dilation = 3 and stride = 2
454static bool matchConvDimAddExprPattern(ArrayAttr indexingMaps, unsigned iDim,
455 unsigned fDim, unsigned oDim,
456 int64_t &dilation, int64_t &stride) {
457 unsigned inputMapIdx = 0, filterMapIdx = 1,
458 outputMapIdx = indexingMaps.size() - 1;
459 AffineExpr inpExpr = getAffineMapDim(indexingMaps, inputMapIdx, iDim);
460 auto addExpr = dyn_cast_or_null<AffineBinaryOpExpr>(inpExpr);
461 if (!addExpr || addExpr.getKind() != AffineExprKind::Add)
462 return false;
463
464 AffineExpr dim0, dim1;
465 int64_t c0 = isDimTimesConstantOrDimOnly(addExpr.getLHS(), dim0);
466 int64_t c1 = isDimTimesConstantOrDimOnly(addExpr.getRHS(), dim1);
467
468 if (c0 == -1 || c1 == -1)
469 return false;
470 // Pattern matched with dims and constants extracted.
471 AffineExpr fExpr = getAffineMapDim(indexingMaps, filterMapIdx, fDim);
472 AffineExpr oExpr = getAffineMapDim(indexingMaps, outputMapIdx, oDim);
473 if (dim0 == fExpr && dim1 == oExpr) {
474 dilation = c0;
475 stride = c1;
476 return true;
477 }
478 if (dim1 == fExpr && dim0 == oExpr) {
479 dilation = c1;
480 stride = c0;
481 return true;
482 }
483 return false;
484}

510/// Helper class for building convolution op matchers with minimal boilerplate.
511/// Reduces repetitive code across Conv1D/2D/3D and Depthwise variants as well
512/// as Pooling ops.
513///
514/// Usage: Create an instance with the op, spatial rank, and output pointers for
515/// extracted dilations/strides. Then chain matchStride() calls for each spatial
516/// dimension, followed by matchMaps() to verify indexing maps, and finally
517/// matchBody() to verify the operation body pattern.
518///
519/// The `matched` flag starts as `true` and is set to `false` if any match step
520/// fails. This allows chaining multiple match calls; once any match fails, all
521/// subsequent calls become no-ops and the final result is `false`.
522///
523/// The `dilations` and `strides` pointers are output parameters that get
524/// populated with the extracted dilation and stride values from the operation's
525/// indexing maps during matchStride() calls. These values are initially set to
526/// 1 for each spatial dimension and updated as patterns are matched.
527class ConvMatcherBuilder {
528 LinalgOp op;
529 MLIRContext *ctx;
530 SmallVector<int64_t> *dilations, *strides;
531 ArrayAttr indexingMaps;
532 PoolingType poolingType;
533 bool matched = true;
534
535public:
536 ConvMatcherBuilder(LinalgOp op, unsigned spatialRank, SmallVector<int64_t> *d,
537 SmallVector<int64_t> *s,
538 PoolingType poolingType = PoolingType::None)
539 : op(op), ctx(op->getContext()), dilations(d), strides(s),
540 indexingMaps(op.getIndexingMaps()), poolingType(poolingType) {
541 *dilations = SmallVector<int64_t>(spatialRank, 1);
542 *strides = SmallVector<int64_t>(spatialRank, 1);
543 }
544
545 /// Get affine dimension expression for dimension `i`.
546 AffineExpr dim(unsigned i) { return getAffineDimExpr(i, ctx); }
547
548 /// Build strided expression: base * stride[idx] + kernel * dilation[idx].
549 AffineExpr strided(AffineExpr base, AffineExpr kernel, unsigned idx) {
550 return base * (*strides)[idx] + kernel * (*dilations)[idx];
551 }
552
553 /// Match stride/dilation pattern for a spatial dimension.
554 /// Returns *this for method chaining.
555 ConvMatcherBuilder &matchStride(unsigned iDim, unsigned fDim, unsigned oDim,
556 unsigned idx) {
557 if (matched) {
558 matched &= matchConvDimAddExprPattern(indexingMaps, iDim, fDim, oDim,
559 (*dilations)[idx], (*strides)[idx]);
560 }
561 return *this;
562 }
563
564 /// Match expected indexing maps layout. Returns *this for method chaining.
565 ConvMatcherBuilder &matchMaps(ArrayRef<ArrayRef<AffineExpr>> maps) {
566 if (matched)
567 matched &= convLayoutMatches(maps, indexingMaps, ctx);
568 return *this;
569 }
570
571 /// Match body pattern. This should be called last.
572 bool matchBody(bool zeroPointOffset = false) {
573 if (!matched)
574 return false;
575 Block *body = op.getBlock();
576 auto yieldOp = cast<linalg::YieldOp>(body->getTerminator());
577 switch (poolingType) {
578 case PoolingType::None:
579 return bodyMatcherForConvolutionOps(yieldOp.getOperand(0), body,
580 zeroPointOffset);
581 case PoolingType::MaxSigned:
582 return bodyMatcherForMaxSignedPoolOps(yieldOp.getOperand(0), body);
583 case PoolingType::MaxUnsigned:
584 return bodyMatcherForMaxUnsignedPoolOps(yieldOp.getOperand(0), body);
585 case PoolingType::MinSigned:
586 return bodyMatcherForMinSignedPoolOps(yieldOp.getOperand(0), body);
587 case PoolingType::MinUnsigned:
588 return bodyMatcherForMinUnsignedPoolOps(yieldOp.getOperand(0), body);
589 case PoolingType::Sum:
590 return bodyMatcherForSumPoolOps(yieldOp.getOperand(0), body);
591 }
592 return false;
593 }
594};

1515 if (!sliceOp)

1517 typeDynDims);
1518
1519 // Search the `source` use-def chain for padded LinalgOps.
1520 Value current = sliceOp.getSource();
1521 while (current) {
1522 auto linalgOp = current.getDefiningOp<LinalgOp>();
1523 if (!linalgOp)
1524 break;
1525 OpResult opResult = cast<OpResult>(current);
1526 current = linalgOp.getDpsInitOperand(opResult.getResultNumber())->get();
1527 }
1528 auto padOp = current ? current.getDefiningOp<tensor::PadOp>() : nullptr;
1529
1530 // Exit if the search fails to match a tensor::PadOp at the end of the matched
1531 // LinalgOp sequence.
1532 if (!padOp)
1533 return tensor::createPadHighOp(type, source, pad, nofold, loc, b,
1534 typeDynDims);
1535
1536 // Exit if the padded result type does not match.
1537 if (sliceOp.getSource().getType() != type)
1538 return tensor::createPadHighOp(type, source, pad, nofold, loc, b,
1539 typeDynDims);
1540
1541 // Exit if the LinalgOps are not high padded.
1542 if (llvm::any_of(padOp.getMixedLowPad(), [](OpFoldResult ofr) {
1543 return getConstantIntValue(ofr) != static_cast<int64_t>(0);
1544 }))
1545 return tensor::createPadHighOp(type, source, pad, nofold, loc, b,
1546 typeDynDims);
1547
1548 // Exit if `padOpSliceOp`, which defines the slice used by
1549 // `padOp`, is rank-reducing.
1550 auto padOpSliceOp = padOp.getSource().getDefiningOp<tensor::ExtractSliceOp>();
1551 if (!padOpSliceOp ||
1552 sliceOp.getMixedSizes().size() != padOpSliceOp.getMixedSizes().size())
1553 return tensor::createPadHighOp(type, source, pad, nofold, loc, b,
1554 typeDynDims);
1555
1556 // Exit if the sizes of the dynamic sizes of `sliceOp` do not match the size
1557 // of the slice padded by `padOp`.
1558 if (llvm::any_of(
1559 llvm::zip(sliceOp.getMixedSizes(), padOpSliceOp.getMixedSizes()),
1560 [](std::tuple<OpFoldResult, OpFoldResult> it) {
1561 return !isEqualConstantIntOrValue(std::get<0>(it), std::get<1>(it));
1562 }))
1563 return tensor::createPadHighOp(type, source, pad, nofold, loc, b,
1564 typeDynDims);
1565
1566 // Exit if the padding values do not match.
1567 Attribute padOpPadAttr, padAttr;
1568 Value padOpPad = padOp.getConstantPaddingValue();
1569 if (!padOpPad || !matchPattern(padOpPad, m_Constant(&padOpPadAttr)) ||
1570 !matchPattern(pad, m_Constant(&padAttr)) || padOpPadAttr != padAttr)
1571 return tensor::createPadHighOp(type, source, pad, nofold, loc, b,
1572 typeDynDims);
1573
1574 // Return the padded result if the padding values and sizes match.
1575 return sliceOp.getSource();
1576}

1688/// Generates a loop nest consisting of scf.parallel and scf.for, depending
1689/// on the `iteratorTypes.` Consecutive parallel loops create a single
1690/// scf.parallel operation; each sequential loop creates a new scf.for
1691/// operation. The body of the innermost loop is populated by
1692/// `bodyBuilderFn` that accepts a range of induction variables for all
1693/// loops. `ivStorage` is used to store the partial list of induction
1694/// variables.
1695// TODO: this function can be made iterative instead. However, it
1696// will have at most as many recursive calls as nested loops, which rarely
1697// exceeds 10.
1698static void generateParallelLoopNest(
1699 OpBuilder &b, Location loc, ValueRange lbs, ValueRange ubs,
1700 ValueRange steps, ArrayRef<utils::IteratorType> iteratorTypes,
1701 ArrayRef<linalg::ProcInfo> procInfo,
1702 function_ref<void(OpBuilder &, Location, ValueRange)> bodyBuilderFn,
1703 SmallVectorImpl<Value> &ivStorage) {
1704 assert(lbs.size() == ubs.size());
1705 assert(lbs.size() == steps.size());
1706 assert(lbs.size() == iteratorTypes.size());
1707 assert(procInfo.empty() || (lbs.size() == procInfo.size()));
1708
1709 // If there are no (more) loops to be generated, generate the body and be
1710 // done with it.
1711 if (iteratorTypes.empty()) {
1712 bodyBuilderFn(b, loc, ivStorage);
1713 return;
1714 }
1715
1716 // If there are no outer parallel loops, generate one sequential loop and
1717 // recurse.
1718 if (!isParallelIterator(iteratorTypes.front())) {
1719 LoopNest singleLoop = buildLoopNest(
1720 b, loc, lbs.take_front(), ubs.take_front(), steps.take_front(),
1721 [&](OpBuilder &b, Location loc, ValueRange ivs) {
1722 ivStorage.append(ivs.begin(), ivs.end());
1723 generateParallelLoopNest(
1724 b, loc, lbs.drop_front(), ubs.drop_front(), steps.drop_front(),
1725 iteratorTypes.drop_front(),
1726 procInfo.empty() ? procInfo : procInfo.drop_front(),
1727 bodyBuilderFn, ivStorage);
1728 });
1729 return;
1730 }
1731
1732 unsigned nLoops = iteratorTypes.size();
1733 unsigned numProcessed = 0;
1734 DistributionMethod distributionMethod = DistributionMethod::None;
1735 if (procInfo.empty()) {
1736 numProcessed = nLoops - iteratorTypes.drop_while(isParallelIterator).size();
1737 } else {
1738 distributionMethod = procInfo.front().distributionMethod;
1739 numProcessed =
1740 nLoops - procInfo
1741 .drop_while([&](linalg::ProcInfo p) {
1742 return p.distributionMethod == distributionMethod;
1743 })
1744 .size();
1745 }
1746
1747 auto remainderProcInfo =
1748 procInfo.empty() ? procInfo : procInfo.drop_front(numProcessed);
1749 switch (distributionMethod) {
1750 case DistributionMethod::None: {
1751 // Generate a single parallel loop-nest operation for all outermost
1752 // parallel loops and recurse.
1753 scf::ParallelOp::create(
1754 b, loc, lbs.take_front(numProcessed), ubs.take_front(numProcessed),
1755 steps.take_front(numProcessed),
1756 [&](OpBuilder &nestedBuilder, Location nestedLoc, ValueRange localIvs) {
1757 ivStorage.append(localIvs.begin(), localIvs.end());
1758 generateParallelLoopNest(
1759 nestedBuilder, nestedLoc, lbs.drop_front(numProcessed),
1760 ubs.drop_front(numProcessed), steps.drop_front(numProcessed),
1761 iteratorTypes.drop_front(numProcessed), remainderProcInfo,
1762 bodyBuilderFn, ivStorage);
1763 });
1764 return;
1765 }
1766 case DistributionMethod::Cyclic: {
1767 // Generate a single parallel loop-nest operation for all outermost
1768 // parallel loops and recurse.
1769 scf::ParallelOp::create(
1770 b, loc, lbs.take_front(numProcessed), ubs.take_front(numProcessed),
1771 steps.take_front(numProcessed),
1772 [&](OpBuilder &nestedBuilder, Location nestedLoc, ValueRange localIvs) {
1773 ivStorage.append(localIvs.begin(), localIvs.end());
1774 generateParallelLoopNest(
1775 nestedBuilder, nestedLoc, lbs.drop_front(numProcessed),
1776 ubs.drop_front(numProcessed), steps.drop_front(numProcessed),
1777 iteratorTypes.drop_front(numProcessed), remainderProcInfo,
1778 bodyBuilderFn, ivStorage);
1779 });
1780 return;
1781 }
1782 case DistributionMethod::CyclicNumProcsGeNumIters: {
1783 // Check (for the processed loops) that the iteration is in-bounds.
1784 ArithBuilder ab(b, loc);
1785 Value cond = ab.slt(lbs[0], ubs[0]);
1786 for (unsigned i = 1; i < numProcessed; ++i)
1787 cond = ab._and(cond, ab.slt(lbs[i], ubs[i]));
1788 ivStorage.append(lbs.begin(), std::next(lbs.begin(), numProcessed));
1789 scf::IfOp::create(b, loc, cond, [&](OpBuilder &b, Location loc) {
1790 generateParallelLoopNest(b, loc, lbs.drop_front(numProcessed),
1791 ubs.drop_front(numProcessed),
1792 steps.drop_front(numProcessed),
1793 iteratorTypes.drop_front(numProcessed),
1794 remainderProcInfo, bodyBuilderFn, ivStorage);
1795 scf::YieldOp::create(b, loc, ValueRange{});
1796 });
1797 return;
1798 }
1799 case DistributionMethod::CyclicNumProcsEqNumIters:
1800 // No check/loops needed here. Set the `%iv` to be the `%lb` and proceed
1801 // with inner loop generation.
1802 ivStorage.append(lbs.begin(), std::next(lbs.begin(), numProcessed));
1803 generateParallelLoopNest(
1804 b, loc, lbs.drop_front(numProcessed), ubs.drop_front(numProcessed),
1805 steps.drop_front(numProcessed), iteratorTypes.drop_front(numProcessed),
1806 remainderProcInfo, bodyBuilderFn, ivStorage);
1807 return;
1808 }
1809}

1811/// Specialization for generating a mix of parallel and sequential scf loops.
1812template <>
1813void GenerateLoopNest<scf::ParallelOp>::doit(
1814 OpBuilder &b, Location loc, ArrayRef<Range> loopRanges, LinalgOp linalgOp,
1815 ArrayRef<utils::IteratorType> iteratorTypes,
1816 function_ref<scf::ValueVector(OpBuilder &, Location, ValueRange,
1817 ValueRange)>
1818 bodyBuilderFn,
1819 ArrayRef<linalg::ProcInfo> procInfo) {
1820 SmallVector<Value> iterArgInitValues;
1821 if (!linalgOp.hasPureBufferSemantics())
1822 llvm::append_range(iterArgInitValues, linalgOp.getDpsInits());
1823 assert(iterArgInitValues.empty() && "unexpected ParallelOp init values");
1824 // This function may be passed more iterator types than ranges.
1825 assert(iteratorTypes.size() >= loopRanges.size() &&
1826 "expected iterator type for all ranges");
1827 assert((procInfo.empty() || (procInfo.size() == loopRanges.size())) &&
1828 "expected proc information for all loops when present");
1829 iteratorTypes = iteratorTypes.take_front(loopRanges.size());
1830 SmallVector<Value, 8> lbsStorage, ubsStorage, stepsStorage, ivs;
1831 unsigned numLoops = iteratorTypes.size();
1832 ivs.reserve(numLoops);
1833 lbsStorage.reserve(numLoops);
1834 ubsStorage.reserve(numLoops);
1835 stepsStorage.reserve(numLoops);
1836
1837 // Get the loop lb, ub, and step.
1838 unpackRanges(b, loc, loopRanges, lbsStorage, ubsStorage, stepsStorage);
1839
1840 // Modify the lb, ub, and step based on the distribution options.
1841 for (const auto &it : llvm::enumerate(procInfo)) {
1842 if (it.value().distributionMethod != linalg::DistributionMethod::None) {
1843 updateBoundsForCyclicDistribution(
1844 b, loc, it.value().procId, it.value().nprocs, lbsStorage[it.index()],
1845 ubsStorage[it.index()], stepsStorage[it.index()]);
1846 }
1847 }
1848 ValueRange lbs(lbsStorage), ubs(ubsStorage), steps(stepsStorage);
1849 generateParallelLoopNest(
1850 b, loc, lbs, ubs, steps, iteratorTypes, procInfo,
1851 [&](OpBuilder &b, Location loc, ValueRange ivs) {
1852 bodyBuilderFn(b, loc, ivs, linalgOp->getOperands());
1853 },
1854 ivs);
1855
1856 assert(ivs.size() == iteratorTypes.size() && "did not generate enough loops");
1857}

1878Operation *makeTiledShape(OpBuilder &builder, Location loc, Value valueToTile,
1879 ArrayRef<OpFoldResult> tileSizes, AffineMap map,
1880 ArrayRef<OpFoldResult> lbs,
1881 ArrayRef<OpFoldResult> ubs,

1890SliceParameters
1891computeSliceParameters(OpBuilder &builder, Location loc, Value valueToTile,
1892 ArrayRef<OpFoldResult> tileSizes, AffineMap map,
1893 ArrayRef<OpFoldResult> lbs, ArrayRef<OpFoldResult> ubs,
1894 ArrayRef<OpFoldResult> subShapeSizes,
1895 bool omitPartialTileCheck) {
1896 auto shapedType = dyn_cast<ShapedType>(valueToTile.getType());
1897 assert(shapedType && "only shaped types can be tiled");
1898 ArrayRef<int64_t> shape = shapedType.getShape();
1899 int64_t rank = shapedType.getRank();
1900
1901 // Compute offsets/sizes/strides for the tile.
1902 SliceParameters sliceParams;
1903 sliceParams.offsets.reserve(rank);
1904 sliceParams.sizes.reserve(rank);
1905 sliceParams.strides.reserve(rank);
1906 for (unsigned r = 0; r < rank; ++r) {
1907 LLVM_DEBUG(llvm::dbgs() << "computeSliceParameters: for dim#" << r);
1908 if (!isTiled(map.getSubMap({r}), tileSizes)) {
1909 sliceParams.offsets.push_back(builder.getIndexAttr(0));
1910 OpFoldResult dim = createFoldedDimOp(builder, loc, valueToTile, r);
1911 sliceParams.sizes.push_back(dim);
1912 sliceParams.strides.push_back(builder.getIndexAttr(1));
1913 LLVM_DEBUG(llvm::dbgs() << ": not tiled: use size: " << dim << "\n");
1914 continue;
1915 }
1916 LLVM_DEBUG(llvm::dbgs() << ": tiled: figure out subsize...\n");
1917
1918 // Tiling creates a new slice at the proper index, the slice step is 1
1919 // (i.e. the op does not subsample, stepping occurs in the loop).
1920 auto m = map.getSubMap({r});
1921 LLVM_DEBUG(llvm::dbgs() << "computeSliceParameters: submap: " << m << "\n");
1922 IRRewriter rewriter(builder);
1923 // The offset of the slice is m(lbs) - m(0).
1924 SmallVector<Attribute> zeros(lbs.size(), rewriter.getIndexAttr(0));
1925 SmallVector<Attribute> mAtZero;
1926 [[maybe_unused]] auto res = m.constantFold(zeros, mAtZero);
1927 assert(succeeded(res) && "affine_map must be evaluatable (not symbols)");
1928 int64_t mAtZeroInt =
1929 cast<IntegerAttr>(mAtZero[0]).getValue().getSExtValue();
1930 OpFoldResult offset = makeComposedFoldedAffineApply(
1931 rewriter, loc, m.getResult(0) - mAtZeroInt, lbs);
1932 sliceParams.offsets.push_back(offset);
1933
1934 OpFoldResult closedIntSize =
1935 makeComposedFoldedAffineApply(rewriter, loc, m, subShapeSizes);
1936 // Resulting size needs to be made half open interval again.
1937 AffineExpr s0 = getAffineSymbolExpr(0, builder.getContext());
1938 OpFoldResult size =
1939 makeComposedFoldedAffineApply(rewriter, loc, s0 + 1, closedIntSize);
1940 LLVM_DEBUG(llvm::dbgs()
1941 << "computeSliceParameters: raw size: " << size << "\n");
1942 LLVM_DEBUG(llvm::dbgs()
1943 << "computeSliceParameters: new offset: " << offset << "\n");
1944 sliceParams.strides.push_back(builder.getIndexAttr(1));
1945
1946 if (omitPartialTileCheck) {
1947 // We statically know that the partial/boundary tile condition is
1948 // unnecessary.
1949 LLVM_DEBUG(llvm::dbgs() << "makeTiledShape: new size: " << size << "\n");
1950 sliceParams.sizes.push_back(size);
1951 continue;
1952 }
1953
1954 // The size of the subview / extract_slice should be trimmed to avoid
1955 // out-of-bounds accesses, unless:
1956 // a. We statically know the subshape size divides the shape size evenly.
1957 // b. The subshape size is 1. According to the way the loops are set up,
1958 // tensors with "0" dimensions would never be constructed.
1959 int64_t shapeSize = shape[r];
1960 std::optional<int64_t> sizeCst = getConstantIntValue(size);
1961 auto hasTileSizeOne = sizeCst == 1;
1962 auto dividesEvenly = sizeCst && ShapedType::isStatic(shapeSize) &&
1963 ((shapeSize % *sizeCst) == 0);
1964 if (!hasTileSizeOne && !dividesEvenly) {
1965 LLVM_DEBUG(llvm::dbgs() << "makeTiledShape: shapeSize=" << shapeSize
1966 << ", size: " << size
1967 << ": make sure in bound with affine.min\n");
1968
1969 AffineExpr dim0, dim1, dim2;
1970 MLIRContext *context = builder.getContext();
1971 bindDims(context, dim0, dim1, dim2);
1972
1973 // Get the dimension size for this dimension. We need to first calculate
1974 // the max index and then plus one. This is important because for
1975 // convolution ops, we have its input window dimension's affine map of the
1976 // form `(d0 * s0 + d1)`, where `d0`/`d1 is an output/filter window
1977 // dimension and `s0` is stride. Directly use the dimension size of
1978 // output/filer window dimensions will cause incorrect calculation.
1979 AffineMap minusOneMap = AffineMap::inferFromExprList(
1980 {ArrayRef<AffineExpr>{dim0 - 1}}, context)
1981 .front();
1982 AffineMap plusOneMap = AffineMap::inferFromExprList(
1983 {ArrayRef<AffineExpr>{dim0 + 1}}, context)
1984 .front();
1985 SmallVector<OpFoldResult> maxIndices =
1986 llvm::to_vector(llvm::map_range(ubs, [&](OpFoldResult ub) {
1987 return makeComposedFoldedAffineApply(rewriter, loc, minusOneMap,
1988 {ub});
1989 }));
1990 OpFoldResult maxIndex =
1991 makeComposedFoldedAffineApply(rewriter, loc, m, maxIndices);
1992 OpFoldResult d =
1993 makeComposedFoldedAffineApply(rewriter, loc, plusOneMap, {maxIndex});
1994
1995 // Compute min(dim - offset, size) to avoid out-of-bounds accesses.
1996 AffineMap minMap = AffineMap::inferFromExprList(
1997 {ArrayRef<AffineExpr>{dim1 - dim2, dim0}}, context)
1998 .front();
1999 size =
2000 makeComposedFoldedAffineMin(rewriter, loc, minMap, {size, d, offset});
2001 }
2002 LLVM_DEBUG(llvm::dbgs() << "makeTiledShape: new size: " << size << "\n");
2003 sliceParams.sizes.push_back(size);
2004 }
2005 return sliceParams;
2006}
