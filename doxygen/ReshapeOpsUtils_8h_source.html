82 /// linalg::(Tensor)CollapseShapeOp.
83 template <typename ReshapeLikeOp>
84 void printReshapeOp(OpAsmPrinter &p, ReshapeLikeOp op) {
85  p << op.getOperationName() << ' ' << op.src() << " [";
86 
87  llvm::interleaveComma(op.reassociation(), p, [&](const Attribute &attr) {
88  p << '[';
89  auto arrayAttr = attr.template cast<ArrayAttr>();
90  llvm::interleaveComma(arrayAttr, p, [&](const Attribute &attr) {
91  p << attr.cast<IntegerAttr>().getInt();
92  });
93  p << ']';
94  });
95 
96  p << "] ";
97  p.printOptionalAttrDict(op->getAttrs(),
98  /*elidedAttrs=*/{getReassociationAttrName()});
99  p << ": " << op.src().getType() << " into " << op.getType();
100 }

102 template <typename ReshapeOpTy, typename InverseReshapeOpTy>
103 static OpFoldResult foldReshapeOp(ReshapeOpTy reshapeOp,
104  ArrayRef<Attribute> operands) {
105  // Fold producer-consumer reshape ops that where the operand type of the
106  // producer is same as the return type of the consumer.
107  auto reshapeSrcOp =
108  reshapeOp.src().template getDefiningOp<InverseReshapeOpTy>();
109  if (reshapeSrcOp && reshapeSrcOp.getSrcType() == reshapeOp.getResultType())
110  return reshapeSrcOp.src();
111  // Reshape of a constant can be replaced with a new constant.
112  if (auto elements = operands.front().dyn_cast_or_null<DenseElementsAttr>()) {
113  return elements.reshape(
114  reshapeOp.getResult().getType().template cast<ShapedType>());
115  }
116  return nullptr;
117 }

119 /// Common verifier for reshape-like types. Fills `expandedType` and
120 ///`collapsedType` with the proper `src` or `result` type.
121 template <typename Op, typename T>
122 static LogicalResult verifyReshapeLikeTypes(Op op, T expandedType,
123  T collapsedType, bool isExpansion) {
124  unsigned expandedRank = expandedType.getRank();
125  unsigned collapsedRank = collapsedType.getRank();
126  if (expandedRank < collapsedRank)
127  return op.emitOpError("expected the type ")
128  << expandedType
129  << " to have higher rank than the type = " << collapsedType;
130  if (expandedRank == 0)
131  return op.emitOpError("expected non-zero memref ranks");
132  if (expandedRank == collapsedRank)
133  return op.emitOpError("expected to collapse or expand dims");
134 
135  if (collapsedRank == 0) {
136  // If collapsed rank is 0, then expanded type must be static shaped and of
137  // sizes 1.
138  if (llvm::any_of(expandedType.getShape(),
139  [](int64_t dim) -> bool { return dim != 1; }))
140  return op.emitOpError("invalid to reshape tensor/memref with non-unit "
141  "extent dimensions to zero-rank tensor/memref");
142  return success();
143  }
144  if (collapsedRank != op.reassociation().size())
145  return op.emitOpError("expected rank of the collapsed type(")
146  << collapsedRank << ") to be the number of reassociation maps("
147  << op.reassociation().size() << ")";
148  auto maps = op.getReassociationMaps();
149  for (auto it : llvm::enumerate(maps))
150  if (it.value().getNumDims() != expandedRank)
151  return op.emitOpError("expected reassociation map #")
152  << it.index() << " of same rank as expanded memref("
153  << expandedRank << "), but got " << it.value().getNumDims();
154  int invalidIdx = 0;
155  if (!isReassociationValid(maps, &invalidIdx))
156  return op.emitOpError("expected reassociation map #")
157  << invalidIdx << " to be valid and contiguous";
158  return verifyReshapeLikeShapes(op, collapsedType, expandedType, isExpansion);
159 }

161 /// Verify that shapes of the reshaped types using following rules
162 /// 1) if a dimension in the collapsed type is static, then the corresponding
163 /// dimensions in the expanded shape should be
164 /// a) static
165 /// b) the product should be same as the collaped shape.
166 /// 2) if a dimension in the collaped type is dynamic, one and only one of the
167 /// corresponding dimensions in the expanded type should be dynamic. This
168 /// rule is only needed with reshape operations that are expanding.
169 template <typename OpTy>
170 static LogicalResult verifyReshapeLikeShapes(OpTy op, ShapedType collapsedType,
171  ShapedType expandedType,
172  bool isExpandingReshape) {
173  ArrayRef<int64_t> collapsedShape = collapsedType.getShape();
174  ArrayRef<int64_t> expandedShape = expandedType.getShape();
175  unsigned expandedDimStart = 0;
176  for (auto map : llvm::enumerate(op.getReassociationMaps())) {
177  Optional<int64_t> dynamicShape;
178  int64_t linearizedStaticShape = 1;
179  for (auto dim : llvm::enumerate(expandedShape.slice(
180  expandedDimStart, map.value().getNumResults()))) {
181  if (ShapedType::isDynamic(dim.value())) {
182  if (isExpandingReshape && dynamicShape) {
183  return op->emitOpError("invalid to have a single dimension (")
184  << map.index() << ") expanded into multiple dynamic dims ("
185  << expandedDimStart + dynamicShape.getValue() << ","
186  << expandedDimStart + dim.index() << ")";
187  }
188  dynamicShape = dim.index();
189  } else {
190  linearizedStaticShape *= dim.value();
191  }
192  }
193  if (dynamicShape) {
194  if (!ShapedType::isDynamic(collapsedShape[map.index()])) {
195  return op->emitOpError("expected dimension ")
196  << map.index()
197  << " of collapsed type to be dynamic since one or more of the "
198  "corresponding dimensions in the expanded type is dynamic";
199  }
200  } else {
201  if (collapsedShape[map.index()] != linearizedStaticShape) {
202  return op->emitOpError("expected dimension ")
203  << map.index() << " of collapsed type to be static value of "
204  << linearizedStaticShape << " ";
205  }
206  }
207  expandedDimStart += map.value().getNumResults();
208  }
209  return success();
210 }

212 /// Pattern to collapse producer/consumer reshape ops that are both collapsing
213 /// dimensions or are both expanding dimensions.
214 template <typename ReshapeOpTy>
215 struct CollapseReshapeOps : public OpRewritePattern<ReshapeOpTy> {
216  using OpRewritePattern<ReshapeOpTy>::OpRewritePattern;
217  LogicalResult matchAndRewrite(ReshapeOpTy reshapeOp,
218  PatternRewriter &rewriter) const override {
219  auto srcReshapeOp = reshapeOp.src().template getDefiningOp<ReshapeOpTy>();
220  if (!srcReshapeOp)
221  return failure();
222 
223  ShapedType resultType = reshapeOp.getResultType();
224  Optional<SmallVector<ReassociationIndices>> reassociationIndices =
225  composeReassociationIndices(srcReshapeOp.getReassociationIndices(),
226  reshapeOp.getReassociationIndices(),
227  rewriter.getContext());
228  if (!reassociationIndices)
229  return failure();
230  rewriter.replaceOpWithNewOp<ReshapeOpTy>(
231  reshapeOp, resultType, srcReshapeOp.src(), *reassociationIndices);
232  return success();
233  }
234 };

236 /// Pattern to collapse producer/consumer reshape ops that are both collapsing
237 /// dimensions or are both expanding dimensions.
238 template <typename ReshapeOpTy, typename InverseReshapeOpTy>
239 struct CollapseMixedReshapeOps : public OpRewritePattern<ReshapeOpTy> {
240  using OpRewritePattern<ReshapeOpTy>::OpRewritePattern;
241  LogicalResult matchAndRewrite(ReshapeOpTy reshapeOp,
242  PatternRewriter &rewriter) const override {
243  auto srcReshapeOp =
244  reshapeOp.src().template getDefiningOp<InverseReshapeOpTy>();
245  if (!srcReshapeOp)
246  return failure();
247 
248  ShapedType srcReshapeSrcType = srcReshapeOp.getSrcType();
249  ShapedType intermediateType = reshapeOp.getSrcType();
250  ShapedType resultType = reshapeOp.getResultType();
251 
252  // If the source reshape can be collapsed/expanded into the target reshape
253  // they can still be folded. This can only be reasoned about statically
254  // for cases where
255  // - either all shapes are static, or
256  // - The number of dynamic dimensions matches in the source of source and
257  // result with all other dimensions being 1.
258  Optional<SmallVector<ReassociationIndices>> reassociationIndices =
259  getReassociationIndicesForReshape(srcReshapeSrcType, resultType);
260  if (!reassociationIndices)
261  return failure();
262  bool originalOpExpands =
263  intermediateType.getRank() > srcReshapeSrcType.getRank();
264  bool resultingOpExpands =
265  resultType.getRank() > srcReshapeSrcType.getRank();
266  if (!(resultingOpExpands ^ originalOpExpands))
267  rewriter.replaceOpWithNewOp<InverseReshapeOpTy>(
268  reshapeOp, resultType, srcReshapeOp.src(), *reassociationIndices);
269  else
270  rewriter.replaceOpWithNewOp<ReshapeOpTy>(
271  reshapeOp, resultType, srcReshapeOp.src(), *reassociationIndices);
272  return success();
273  }
274 };
