30 static nb::object objectFromPDLValue(MlirPDLValue value) {
31  if (MlirValue v = mlirPDLValueAsValue(value); !mlirValueIsNull(v))
32  return nb::cast(v);
33  if (MlirOperation v = mlirPDLValueAsOperation(value); !mlirOperationIsNull(v))
34  return nb::cast(v);
35  if (MlirAttribute v = mlirPDLValueAsAttribute(value); !mlirAttributeIsNull(v))
36  return nb::cast(v);
37  if (MlirType v = mlirPDLValueAsType(value); !mlirTypeIsNull(v))
38  return nb::cast(v);
39 
40  throw std::runtime_error("unsupported PDL value type");
41 }

43 static std::vector<nb::object> objectsFromPDLValues(size_t nValues,
44  MlirPDLValue *values) {
45  std::vector<nb::object> args;
46  args.reserve(nValues);
47  for (size_t i = 0; i < nValues; ++i)
48  args.push_back(objectFromPDLValue(values[i]));
49  return args;
50 }

52 // Convert the Python object to a boolean.
53 // If it evaluates to False, treat it as success;
54 // otherwise, treat it as failure.
55 // Note that None is considered success.
56 static MlirLogicalResult logicalResultFromObject(const nb::object &obj) {
57  if (obj.is_none())
58  return mlirLogicalResultSuccess();
59 
60  return nb::cast<bool>(obj) ? mlirLogicalResultFailure()
61  : mlirLogicalResultSuccess();
62 }

78  void registerRewriteFunction(const std::string &name,
79  const nb::callable &fn) {
80  mlirPDLPatternModuleRegisterRewriteFunction(
81  get(), mlirStringRefCreate(name.data(), name.size()),
82  [](MlirPatternRewriter rewriter, MlirPDLResultList results,
83  size_t nValues, MlirPDLValue *values,
84  void *userData) -> MlirLogicalResult {
85  nb::handle f = nb::handle(static_cast<PyObject *>(userData));
86  return logicalResultFromObject(
87  f(rewriter, results, objectsFromPDLValues(nValues, values)));

90  }
