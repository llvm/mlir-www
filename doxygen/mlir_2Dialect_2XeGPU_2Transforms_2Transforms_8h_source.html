25 /// Options to control the XeGPU unrolling. Its main purpose is to
26 /// provide a way to customize the native shape of the operation.
27 struct UnrollOptions {
28  /// Callback function that indicates whether vector unrolling should be
29  /// attempted on the operation.
30  using FilterConstraintFnType = std::function<LogicalResult(Operation *op)>;
31  FilterConstraintFnType filterConstraint = nullptr;
32  UnrollOptions &setFilterConstraint(FilterConstraintFnType constraint) {
33  filterConstraint = std::move(constraint);
34  return *this;
35  }
36 
37  /// Function that computes the target shape for unrolling. It returns an
38  /// optional vector of integers representing the shape. If it returns
39  /// `std::nullopt`, unrolling is aborted for the given operation.
40  using NativeShapeFnType =
41  std::function<std::optional<SmallVector<int64_t>>(Operation *op)>;
42  NativeShapeFnType nativeShape = nullptr;
43  UnrollOptions &setNativeShapeFn(NativeShapeFnType fn) {
44  nativeShape = std::move(fn);
45  return *this;
46  }
47 
48  /// Function that converts a ShapedType (TensorDescType or VectorType)
49  /// into the unrolled type based on the tileShape. It returns a vector of
50  /// types representing the unrolled types for simplicity. When
51  /// `returnSingleType` is true, it returns a vector containing only one single
52  /// unrolled type.
53  using UnrolledTypeFnType = std::function<SmallVector<Type>(
54  ShapedType type, ArrayRef<int64_t> tileShape, bool returnSingleType)>;
55  UnrolledTypeFnType getUnrolledTypes = nullptr;
56  UnrollOptions &setUnrolledTypesFn(UnrolledTypeFnType fn) {
57  getUnrolledTypes = std::move(fn);
58  return *this;
59  }
60 };

69 /// Collect a set of patterns to unroll xegpu operations to a smaller shapes.
70 /// Users can control whether an operation to be unrolled or not, as well as
71 /// its target shape via `options` structure. (via setting filterConstraint
72 /// and nativeShape respectively, both of them are function refs taking `op` as
73 /// input).
74 /// An `op` is unrolled to the `targetShape` as follows, for each of its
75 /// operands:
76 /// 1. the unrolled type `unrolledType` and number of unrolled instances
77 /// `numUnrolledInstances` are computed from the `targetShape`.
78 /// 2. pack each operand. ExtractStridedSlice are created to break-up the
79 /// vector operands. And BuiltinUnrealizedCastop are created to break-up
80 /// the TensorDesc operands.
81 /// 3. the original op is cloned `numUnrolledInstances` times, once for each
82 /// result.
83 /// 4. unpack the results. InsertStridedSlice are inserted for VectorType
84 /// result, and BuiltinUnrealizedCastOp are inserted for TensorDescType result
85 /// to re-assemble the slices into the original shape.
86 void populateXeGPUUnrollPatterns(RewritePatternSet &patterns,
87  const UnrollOptions &options);
