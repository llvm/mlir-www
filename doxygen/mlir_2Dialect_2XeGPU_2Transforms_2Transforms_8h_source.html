25 /// Options to control the XeGPU unrolling. Its main purpose is to
26 /// provide a way to customize the native shape of the operation.
27 struct UnrollOptions {
28  /// Callback function that indicates whether vector unrolling should be
29  /// attempted on the operation.
30  using FilterConstraintFnType = std::function<LogicalResult(Operation *op)>;
31  FilterConstraintFnType filterConstraint = nullptr;
32  UnrollOptions &setFilterConstraint(FilterConstraintFnType constraint) {
33  filterConstraint = std::move(constraint);
34  return *this;
35  }
36 
37  /// Function that computes the target shape for unrolling. It returns an
38  /// optional vector of integers representing the shape. If it returns
39  /// `std::nullopt`, unrolling is aborted for the given operation.
40  using NativeShapeFnType =
41  std::function<std::optional<SmallVector<int64_t>>(Operation *op)>;
42  NativeShapeFnType nativeShape = nullptr;
43  UnrollOptions &setNativeShapeFn(NativeShapeFnType fn) {
44  nativeShape = std::move(fn);
45  return *this;
46  }
47 
48  /// Function that converts a ShapedType (TensorDescType or VectorType)
49  /// into the unrolled type based on the tileShape. It returns a vector of
50  /// types representing the unrolled types for simplicity.
51  using UnrolledTypeFnType = std::function<SmallVector<Type>(
52  ShapedType type, ArrayRef<int64_t> tileShape)>;
53  UnrolledTypeFnType getUnrolledTypes = nullptr;
54  UnrollOptions &setUnrolledTypesFn(UnrolledTypeFnType fn) {
55  getUnrolledTypes = std::move(fn);
56  return *this;
57  }
58 };

67 /// Collect a set of patterns to unroll xegpu operations to a smaller shapes.
68 /// Users can control whether an operation to be unrolled or not, as well as
69 /// its target shape via `options` structure. (via setting filterConstraint
70 /// and nativeShape respectively, both of them are function refs taking `op` as
71 /// input).
72 /// An `op` is unrolled to the `targetShape` as follows, for each of its
73 /// operands:
74 /// 1. the unrolled type `unrolledType` and number of unrolled instances
75 /// `numUnrolledInstances` are computed from the `targetShape`.
76 /// 2. pack each operand. ExtractStridedSlice are created to break-up the
77 /// vector operands. And BuiltinUnrealizedCastop are created to break-up
78 /// the TensorDesc operands.
79 /// 3. the original op is cloned `numUnrolledInstances` times, once for each
80 /// result.
81 /// 4. unpack the results. InsertStridedSlice are inserted for VectorType
82 /// result, and BuiltinUnrealizedCastOp are inserted for TensorDescType result
83 /// to re-assemble the slices into the original shape.
84 void populateXeGPUUnrollPatterns(RewritePatternSet &patterns,
85  const UnrollOptions &options);
