27/// Options to control the XeGPU unrolling. Its main purpose is to
28/// provide a way to customize the native shape of the operation.
29struct UnrollOptions {
30 /// Callback function that indicates whether vector unrolling should be
31 /// attempted on the operation.
32 using FilterConstraintFnType = std::function<LogicalResult(Operation *op)>;
33 FilterConstraintFnType filterConstraint = nullptr;
34 UnrollOptions &setFilterConstraint(FilterConstraintFnType constraint) {
35 filterConstraint = std::move(constraint);
36 return *this;
37 }
38
39 /// Function that computes the target shape for unrolling. It returns an
40 /// optional vector of integers representing the shape. If it returns
41 /// `std::nullopt`, unrolling is aborted for the given operation.
42 using NativeShapeFnType =
43 std::function<std::optional<SmallVector<int64_t>>(Operation *op)>;
44 NativeShapeFnType nativeShape = nullptr;
45 UnrollOptions &setNativeShapeFn(NativeShapeFnType fn) {
46 nativeShape = std::move(fn);
47 return *this;
48 }
49
50 /// Function that converts a ShapedType (TensorDescType or VectorType)
51 /// into the unrolled type based on the tileShape. It returns a vector of
52 /// types representing the unrolled types for simplicity.
53 using UnrolledTypeFnType = std::function<SmallVector<Type>(
54 ShapedType type, ArrayRef<int64_t> tileShape)>;
55 UnrolledTypeFnType getUnrolledTypes = nullptr;
56 UnrollOptions &setUnrolledTypesFn(UnrolledTypeFnType fn) {
57 getUnrolledTypes = std::move(fn);
58 return *this;
59 }
60};

95/// Run the AxisInfo-based contiguity analysis over `root` and stamp a
96/// `contiguity` attribute on every `xegpu.load` / `xegpu.store` whose
97/// offsets are contiguous (in runs of >= 2) along the innermost dimension.
98/// The stamped value is the inner-dim contiguity; it is a target-independent
99/// property consumed downstream (e.g. to derive a `lane_data` split). Ops that
100/// already carry a `contiguity` attribute are left untouched.
101void runContiguityAnalysis(Operation *root);

103/// Collect a set of patterns to unroll xegpu operations to a smaller shapes.
104/// Users can control whether an operation to be unrolled or not, as well as
105/// its target shape via `options` structure. (via setting filterConstraint
106/// and nativeShape respectively, both of them are function refs taking `op` as
107/// input).
108/// An `op` is unrolled to the `targetShape` as follows, for each of its
109/// operands:
110/// 1. the unrolled type `unrolledType` and number of unrolled instances
111/// `numUnrolledInstances` are computed from the `targetShape`.
112/// 2. pack each operand. ExtractStridedSlice are created to break-up the
113/// vector operands. And BuiltinUnrealizedCastOp are created to break-up
114/// the TensorDesc operands.
115/// 3. the original op is cloned `numUnrolledInstances` times, once for each
116/// result.
117/// 4. unpack the results. InsertStridedSlice are inserted for VectorType
118/// result, and BuiltinUnrealizedCastOp are inserted for TensorDescType result
119/// to re-assemble the slices into the original shape.
120void populateXeGPUUnrollPatterns(RewritePatternSet &patterns,
121 const UnrollOptions &options);
