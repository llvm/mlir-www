36 func.walk<WalkOrder::PreOrder>([](Operation *op) {
37 if (auto scfForOp = dyn_cast<scf::ForOp>(op))
38 (void)coalescePerfectlyNestedSCFForLoops(scfForOp);
39 else if (auto affineForOp = dyn_cast<AffineForOp>(op))

41 });
42 }
