430 Block *OpBuilder::createBlock(Region *parent, Region::iterator insertPt,
431  TypeRange argTypes, ArrayRef<Location> locs) {
432  assert(parent && "expected valid parent region");
433  assert(argTypes.size() == locs.size() && "argument location mismatch");
434  if (insertPt == Region::iterator())
435  insertPt = parent->end();

437  Block *b = new Block();

442  if (listener)

472 LogicalResult
473 OpBuilder::tryFold(Operation *op, SmallVectorImpl<Value> &results,
474  SmallVectorImpl<Operation *> *materializedConstants) {
475  assert(results.empty() && "expected empty results");
476  ResultRange opResults = op->getResults();
477 
478  results.reserve(opResults.size());
479  auto cleanupFailure = [&] {
480  results.clear();
481  return failure();
482  };
483 
484  // If this operation is already a constant, there is nothing to do.
485  if (matchPattern(op, m_Constant()))
486  return cleanupFailure();
487 
488  // Try to fold the operation.
489  SmallVector<OpFoldResult, 4> foldResults;
490  LDBG() << "Trying to fold: "
491  << OpWithFlags(op, OpPrintingFlags().skipRegions());
492  if (op->getName().getStringRef() == "vector.extract") {
493  Operation *parent = op->getParentOp();
494  while (parent && parent->getName().getStringRef() != "spirv.func")
495  parent = parent->getParentOp();
496  if (parent)
497  parent->dump();
498  }
499  if (failed(op->fold(foldResults)))
500  return cleanupFailure();
501 
502  int count = 0;
503  do {
504  LDBG() << "Folded in place #" << count
505  << " times: " << OpWithFlags(op, OpPrintingFlags().skipRegions());
506  count++;
507  } while (foldResults.empty() && succeeded(op->fold(foldResults)));
508 
509  // An in-place fold does not require generation of any constants.
510  if (foldResults.empty())
511  return success();
512 
513  // A temporary builder used for creating constants during folding.
514  OpBuilder cstBuilder(context);
515  SmallVector<Operation *, 1> generatedConstants;
516 
517  // Populate the results with the folded results.
518  Dialect *dialect = op->getDialect();
519  for (auto [foldResult, expectedType] :
520  llvm::zip_equal(foldResults, opResults.getTypes())) {
521 
522  // Normal values get pushed back directly.
523  if (auto value = llvm::dyn_cast_if_present<Value>(foldResult)) {
524  results.push_back(value);
525  continue;
526  }
527 
528  // Otherwise, try to materialize a constant operation.
529  if (!dialect)
530  return cleanupFailure();
531 
532  // Ask the dialect to materialize a constant operation for this value.
533  Attribute attr = cast<Attribute>(foldResult);
534  auto *constOp = dialect->materializeConstant(cstBuilder, attr, expectedType,
535  op->getLoc());
536  if (!constOp) {
537  // Erase any generated constants.
538  for (Operation *cst : generatedConstants)
539  cst->erase();
540  return cleanupFailure();
541  }
542  assert(matchPattern(constOp, m_Constant()));
543 
544  generatedConstants.push_back(constOp);
545  results.push_back(constOp->getResult(0));
546  }
547 
548  // If we were successful, insert any generated constants.
549  for (Operation *cst : generatedConstants)
550  insert(cst);
551 
552  // Return materialized constant operations.
553  if (materializedConstants)
554  *materializedConstants = std::move(generatedConstants);
555 
556  return success();
557 }

559 /// Helper function that sends block insertion notifications for every block
560 /// that is directly nested in the given op.
561 static void notifyBlockInsertions(Operation *op,
562  OpBuilder::Listener *listener) {
563  for (Region &r : op->getRegions())
564  for (Block &b : r.getBlocks())
565  listener->notifyBlockInserted(&b, /*previous=*/nullptr,
566  /*previousIt=*/{});

569 Operation *OpBuilder::clone(Operation &op, IRMapping &mapper) {
570  Operation *newOp = op.clone(mapper);
571  newOp = insert(newOp);
572 
573  // The `insert` call above handles the notification for inserting `newOp`
574  // itself. But if `newOp` has any regions, we need to notify the listener
575  // about any ops that got inserted inside those regions as part of cloning.
576  if (listener) {
577  // The `insert` call above notifies about op insertion, but not about block
578  // insertion.
579  notifyBlockInsertions(newOp, listener);
580  auto walkFn = [&](Operation *walkedOp) {
581  listener->notifyOperationInserted(walkedOp, /*previous=*/{});
582  notifyBlockInsertions(walkedOp, listener);
583  };
584  for (Region &region : newOp->getRegions())
585  region.walk<WalkOrder::PreOrder>(walkFn);
586  }
587 
588  return newOp;
589 }

596 void OpBuilder::cloneRegionBefore(Region &region, Region &parent,
597  Region::iterator before, IRMapping &mapping) {
598  region.cloneInto(&parent, before, mapping);
599 
600  // Fast path: If no listener is attached, there is no more work to do.
601  if (!listener)
602  return;
603 
604  // Notify about op/block insertion.
605  for (auto it = mapping.lookup(&region.front())->getIterator(); it != before;
606  ++it) {
607  listener->notifyBlockInserted(&*it, /*previous=*/nullptr,
608  /*previousIt=*/{});
609  it->walk<WalkOrder::PreOrder>([&](Operation *walkedOp) {
610  listener->notifyOperationInserted(walkedOp, /*previous=*/{});
611  notifyBlockInsertions(walkedOp, listener);
612  });
613  }
614 }
