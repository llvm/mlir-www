63  // memory allocations, so drop private annotations. NVVM uses address
64  // space 3 for shared memory. NVVM uses the default address space to
65  // represent global memory.

76  NVVM::NVVMMemorySpace::kSharedMemorySpace);

88  populateGpuToNVVMConversionPatterns(llvmTypeConverter, patterns,
89  getBenefit());

180 /// Pick an unrolling order that will allow tensorcore operation to reuse LHS
181 /// register.
182 static std::optional<SmallVector<int64_t>>
183 gpuMmaUnrollOrder(vector::ContractionOp contract) {
184  SmallVector<int64_t> order;
185  // First make reduction the outer dimensions.
186  for (auto [index, iter] : llvm::enumerate(contract.getIteratorTypes())) {
187  if (vector::isReductionIterator(iter)) {
188  order.push_back(index);
189  }
190  }
191 
192  llvm::SmallDenseSet<int64_t> dims;
193  for (AffineExpr expr : contract.getIndexingMapsArray()[0].getResults()) {
194  dims.insert(cast<AffineDimExpr>(expr).getPosition());
195  }
196  // Then parallel dimensions that are part of Lhs as we want to re-use Lhs.
197  for (auto [index, iter] : llvm::enumerate(contract.getIteratorTypes())) {
198  if (vector::isParallelIterator(iter) && dims.count(index)) {
199  order.push_back(index);
200  }
201  }
202  // Then the remaining parallel loops.
203  for (auto [index, iter] : llvm::enumerate(contract.getIteratorTypes())) {
204  if (vector::isParallelIterator(iter) && !dims.count(index)) {
205  order.push_back(index);
206  }
207  }
208  return order;
209 }

211 /// Returns the target vector size for the target operation based on the native
212 /// vector size specified with `m`, `n`, and `k`.
213 static std::optional<SmallVector<int64_t>>
214 getSubgroupMmaNativeVectorSize(Operation *op, int64_t m, int64_t n, int64_t k) {
215  if (auto contract = dyn_cast<vector::ContractionOp>(op)) {
216  int64_t contractRank = contract.getIteratorTypes().size();
217  if (contractRank < 3)
218  return std::nullopt;
219  SmallVector<int64_t> nativeSize(contractRank - 3, 1);
220  nativeSize.append({m, n, k});
221  return nativeSize;
222  }
223  if (auto writeOp = dyn_cast<vector::TransferWriteOp>(op)) {
224  int64_t writeRank = writeOp.getVectorType().getRank();
225  if (writeRank < 2)
226  return std::nullopt;
227  SmallVector<int64_t> nativeSize(writeRank - 2, 1);
228  nativeSize.append({m, n});
229  return nativeSize;
230  }
231  if (auto readOp = dyn_cast<vector::TransferReadOp>(op)) {
232  // Transfer read ops may need different shapes based on how they are being
233  // used. For simplicity just match the shape used by the extract strided op.
234  VectorType sliceType;
235  for (Operation *users : op->getUsers()) {
236  auto extract = dyn_cast<vector::ExtractStridedSliceOp>(users);
237  if (!extract)
238  return std::nullopt;
239  auto vecType = cast<VectorType>(extract.getResult().getType());
240  if (sliceType && sliceType != vecType)
241  return std::nullopt;
242  sliceType = vecType;
243  }
244  return llvm::to_vector(sliceType.getShape());
245  }
246  if ((OpTrait::hasElementwiseMappableTraits(op) && op->getNumResults() == 1)) {
247  if (auto vecType = dyn_cast<VectorType>(op->getResultTypes()[0])) {
248  // TODO: The condition for unrolling elementwise should be restricted
249  // only to operations that need unrolling (connected to the contract).
250  if (vecType.getRank() < 2)
251  return std::nullopt;
252 
253  // First check whether there is a slice to infer the shape from. This is
254  // required for cases where the accumulator type differs from the input
255  // types, in which case we will see an `arith.ext_` between the contract
256  // and transfer_read which needs to be unrolled.
257  VectorType sliceType;
258  for (Operation *users : op->getUsers()) {
259  auto extract = dyn_cast<vector::ExtractStridedSliceOp>(users);
260  if (!extract)
261  return std::nullopt;
262  auto vecType = cast<VectorType>(extract.getResult().getType());
263  if (sliceType && sliceType != vecType)
264  return std::nullopt;
265  sliceType = vecType;
266  }
267  if (sliceType)
268  return llvm::to_vector(sliceType.getShape());
269 
270  // Else unroll for trailing elementwise.
271  SmallVector<int64_t> nativeSize(vecType.getRank() - 2, 1);
272  // Map elementwise ops to the output shape.
273  nativeSize.append({m, n});
274  return nativeSize;
275  }
276  }
277  return std::nullopt;
278 }

280 void transform::ApplyUnrollVectorsSubgroupMmaOp::populatePatterns(
281  RewritePatternSet &patterns) {
282  auto unrollOrder = [](Operation *op) -> std::optional<SmallVector<int64_t>> {
283  auto contract = dyn_cast<vector::ContractionOp>(op);
284  if (!contract)
285  return std::nullopt;
286  return gpuMmaUnrollOrder(contract);
287  };
288 
289  int64_t m = getM();
290  int64_t n = getN();
291  int64_t k = getK();
292  auto nativeShapeFn =
293  [m, n, k](Operation *op) -> std::optional<SmallVector<int64_t>> {
294  return getSubgroupMmaNativeVectorSize(op, m, n, k);
295  };
296  vector::populateVectorUnrollPatterns(
297  patterns, vector::UnrollVectorOptions()
298  .setNativeShapeFn(nativeShapeFn)
299  .setUnrollTraversalOrderFn(unrollOrder));

329 /// Check if given mapping attributes are one of the desired attributes
330 template <typename MappingKindType>
331 static DiagnosedSilenceableFailure
332 checkMappingAttributeTypes(std::optional<TransformOpInterface> transformOp,
333  scf::ForallOp forallOp) {
334  if (!forallOp.getMapping().has_value()) {
335  return definiteFailureHelper(transformOp, forallOp,
336  "scf.forall op requires a mapping attribute");
337  }
338 
339  bool hasBlockMapping = llvm::any_of(forallOp.getMapping().value(),
340  llvm::IsaPred<GPUBlockMappingAttr>);
341  bool hasWarpgroupMapping = llvm::any_of(
342  forallOp.getMapping().value(), llvm::IsaPred<GPUWarpgroupMappingAttr>);
343  bool hasWarpMapping = llvm::any_of(forallOp.getMapping().value(),
344  llvm::IsaPred<GPUWarpMappingAttr>);
345  bool hasThreadMapping = llvm::any_of(forallOp.getMapping().value(),
346  llvm::IsaPred<GPUThreadMappingAttr>);
347  bool hasLaneMapping = llvm::any_of(forallOp.getMapping().value(),
348  llvm::IsaPred<GPULaneMappingAttr>);
349  int64_t countMappingTypes = 0;
350  countMappingTypes += hasBlockMapping ? 1 : 0;
351  countMappingTypes += hasWarpgroupMapping ? 1 : 0;
352  countMappingTypes += hasWarpMapping ? 1 : 0;
353  countMappingTypes += hasThreadMapping ? 1 : 0;
354  countMappingTypes += hasLaneMapping ? 1 : 0;
355  if (countMappingTypes > 1) {
356  return definiteFailureHelper(
357  transformOp, forallOp,
358  "cannot mix different mapping types, use nesting");
359  }
360  if (std::is_same<MappingKindType, BlockMappingKind>::value &&
361  !hasBlockMapping) {
362  return definiteFailureHelper(
363  transformOp, forallOp,
364  "scf.forall op requires a mapping attribute of kind 'block'");
365  }
366  if (std::is_same<MappingKindType, ThreadMappingKind>::value &&
367  !hasLaneMapping && !hasThreadMapping && !hasWarpMapping &&
368  !hasWarpgroupMapping) {
369  return definiteFailureHelper(transformOp, forallOp,
370  "scf.forall op requires a mapping attribute "
371  "of kind 'thread' or 'warp'");
372  }
373 
374  DenseSet<Attribute> seen;
375  for (Attribute map : forallOp.getMapping()->getValue()) {
376  if (seen.contains(map)) {
377  return definiteFailureHelper(
378  transformOp, forallOp,
379  "duplicate attribute, cannot map different loops "
380  "to the same mapping id");
381  }
382  seen.insert(map);
383  }
384 
385  auto isLinear = [](DeviceMappingAttrInterface attr) {
386  return attr.isLinearMapping();
387  };
388  if (llvm::any_of(forallOp.getDeviceMappingAttrs(), isLinear) &&
389  !llvm::all_of(forallOp.getDeviceMappingAttrs(), isLinear)) {
390  return definiteFailureHelper(
391  transformOp, forallOp,
392  "cannot mix linear and non-linear mapping modes");
393  }
394 
395  FailureOr<DeviceMaskingAttrInterface> maybeMaskingAttr =
396  forallOp.getDeviceMaskingAttr();
397  if (succeeded(maybeMaskingAttr) && *maybeMaskingAttr &&
398  !forallOp.usesLinearMapping()) {
399  return definiteFailureHelper(
400  transformOp, forallOp,
401  "device masking is only available in linear mapping mode");
402  }
403 
404  return DiagnosedSilenceableFailure::success();
405 }

407 template <typename MappingKindType>
408 static DiagnosedSilenceableFailure
409 verifyGpuMapping(std::optional<TransformOpInterface> transformOp,
410  scf::ForallOp forallOp) {
411  // Check the types of the mapping attributes match.
412  DiagnosedSilenceableFailure typeRes =
413  checkMappingAttributeTypes<MappingKindType>(transformOp, forallOp);
414  if (!typeRes.succeeded())
415  return typeRes;
416 
417  // Perform other non-types verifications.
418  if (!forallOp.isNormalized())
419  return definiteFailureHelper(transformOp, forallOp,
420  "unsupported non-normalized loops");
421  if (forallOp.getNumResults() > 0)
422  return definiteFailureHelper(transformOp, forallOp,
423  "only bufferized scf.forall can be mapped");
424  bool useLinearMapping = forallOp.usesLinearMapping();
425  // TODO: This would be more natural with support for Optional<EnumParameter>
426  // in GPUDeviceMappingAttr.
427  int64_t maxNumMappingsSupported =
428  useLinearMapping ? (getMaxEnumValForMappingId() -
429  static_cast<uint64_t>(MappingId::DimZ))
430  : 3;
431  if (forallOp.getRank() > maxNumMappingsSupported) {
432  return definiteFailureHelper(transformOp, forallOp,
433  "scf.forall with rank > ")
434  << maxNumMappingsSupported
435  << " does not lower for the specified mapping attribute type";
436  }
437  auto numParallelIterations =
438  getConstantIntValues(forallOp.getMixedUpperBound());
439  if (!forallOp.isNormalized() || !numParallelIterations.has_value()) {
440  return definiteFailureHelper(
441  transformOp, forallOp,
442  "requires statically sized, normalized forall op");
443  }
444  return DiagnosedSilenceableFailure::success();
445 }

465 static DiagnosedSilenceableFailure rewriteOneForallCommonImpl(
466  RewriterBase &rewriter, std::optional<TransformOpInterface> transformOp,
467  scf::ForallOp forallOp, ArrayRef<int64_t> availableMappingSizes,
468  ForallRewriteResult &result, const GpuIdBuilder &gpuIdBuilder) {
469  LDBG() << "--start rewriteOneForallCommonImpl";
470 
471  // Step 1. Complete the mapping to a full mapping (with 1s) if necessary.
472  auto numParallelIterations =
473  getConstantIntValues(forallOp.getMixedUpperBound());
474  assert(forallOp.isNormalized() && numParallelIterations.has_value() &&
475  "requires statically sized, normalized forall op");
476  SmallVector<int64_t> tmpMappingSizes = numParallelIterations.value();
477  SmallVector<DeviceMappingAttrInterface> forallMappingAttrsVec =
478  forallOp.getDeviceMappingAttrs();
479  SetVector<Attribute> forallMappingAttrs;
480  forallMappingAttrs.insert_range(forallMappingAttrsVec);
481  auto comparator = [](Attribute a, Attribute b) -> bool {
482  return cast<DeviceMappingAttrInterface>(a).getMappingId() <
483  cast<DeviceMappingAttrInterface>(b).getMappingId();
484  };
485 
486  // Step 1.b. In the linear case, compute the max mapping to avoid needlessly
487  // mapping all dimensions. In the 3-D mapping case we need to map all
488  // dimensions.
489  DeviceMappingAttrInterface maxMapping = cast<DeviceMappingAttrInterface>(
490  *llvm::max_element(forallMappingAttrs, comparator));
491  DeviceMappingAttrInterface maxLinearMapping;
492  if (maxMapping.isLinearMapping())
493  maxLinearMapping = maxMapping;
494  for (auto attr : gpuIdBuilder.mappingAttributes) {
495  // If attr overflows, just skip.
496  if (maxLinearMapping && comparator(maxLinearMapping, attr))
497  continue;
498  // Try to insert. If element was already present, just continue.
499  if (!forallMappingAttrs.insert(attr))
500  continue;
501  // Otherwise, we have a new insertion without a size -> use size 1.
502  tmpMappingSizes.push_back(1);
503  }
504  LDBG() << "----tmpMappingSizes extracted from scf.forall op: "
505  << llvm::interleaved(tmpMappingSizes);
506 
507  // Step 2. sort the values by the corresponding DeviceMappingAttrInterface.
508  SmallVector<int64_t> forallMappingSizes = getValuesSortedByKey(
509  forallMappingAttrs.getArrayRef(), tmpMappingSizes, comparator);
510  LDBG() << "----forallMappingSizes: " << llvm::interleaved(forallMappingSizes);
511  LDBG() << "----forallMappingAttrs: " << llvm::interleaved(forallMappingAttrs);
512 
513  // Step 3. Generate the mappingIdOps using the provided generator.
514  Location loc = forallOp.getLoc();
515  OpBuilder::InsertionGuard guard(rewriter);
516  rewriter.setInsertionPoint(forallOp);
517  SmallVector<int64_t> originalBasis(availableMappingSizes);
518  bool originalBasisWasProvided = !originalBasis.empty();
519  if (!originalBasisWasProvided) {
520  LDBG() << "----originalBasis was not provided, deriving it and there will "
521  "be no "
522  "predication";
523  originalBasis = forallMappingSizes;
524  while (originalBasis.size() < 3)
525  originalBasis.push_back(1);
526  } else {
527  LDBG() << "----originalBasis was provided, using it, there will be "
528  "predication";
529  }
530  LDBG() << "------originalBasis: " << llvm::interleaved(originalBasis);
531 
532  IdBuilderResult builderResult =
533  gpuIdBuilder.idBuilder(rewriter, loc, forallMappingSizes, originalBasis);
534  if (!builderResult.errorMsg.empty())
535  return definiteFailureHelper(transformOp, forallOp, builderResult.errorMsg);
536 
537  LDBG() << builderResult;
538 
539  // Step 4. Map the induction variables to the mappingIdOps, this may involve
540  // a permutation.
541  SmallVector<Value> mappingIdOps = builderResult.mappingIdOps;
542  IRMapping bvm;
543  for (auto [iv, dim] : llvm::zip_equal(
544  forallOp.getInductionVars(),
545  forallMappingAttrs.getArrayRef().take_front(forallOp.getRank()))) {
546  auto mappingAttr = cast<DeviceMappingAttrInterface>(dim);
547  Value peIdOp = mappingIdOps[mappingAttr.getRelativeIndex()];
548  LDBG() << "----map: " << iv << " to " << peIdOp;
549  bvm.map(iv, peIdOp);
550  }
551 
552  // Step 5. If the originalBasis is already known, create conditionals to
553  // predicate the region. Otherwise, the current forall determines the
554  // originalBasis and no predication occurs.
555  Value predicate;
556  if (originalBasisWasProvided) {
557  for (Value tmpPredicate : builderResult.predicateOps) {
558  predicate = predicate ? arith::AndIOp::create(rewriter, loc, predicate,
559  tmpPredicate)
560  : tmpPredicate;
561  }
562  }
563 
564  // Step 6. Move the body of forallOp.
565  // Erase the terminator first, it will not be used.
566  rewriter.eraseOp(forallOp.getTerminator());
567  Block *targetBlock;
568  Block::iterator insertionPoint;
569  if (predicate) {
570  // Step 6.a. If predicated, move at the beginning.
571  auto ifOp = scf::IfOp::create(rewriter, loc, predicate,
572  /*withElseRegion=*/false);
573  targetBlock = ifOp.thenBlock();
574  insertionPoint = ifOp.thenBlock()->begin();
575  } else {

763 static DiagnosedSilenceableFailure checkMappingSpec(
764  std::optional<TransformOpInterface> transformOp, scf::ForallOp forallOp,
765  ArrayRef<int64_t> numParallelIterations, ArrayRef<int64_t> blockOrGridSizes,
766  int factor, bool useLinearMapping = false) {
767  if (!useLinearMapping && blockOrGridSizes.front() % factor != 0) {
768  auto diag = definiteFailureHelper(
769  transformOp, forallOp,
770  Twine("3-D mapping: size of threadIdx.x must be a multiple of ") +
771  Twine(factor));
772  return diag;
773  }
774  if (computeProduct(numParallelIterations) * factor >
775  computeProduct(blockOrGridSizes)) {
776  auto diag = definiteFailureHelper(
777  transformOp, forallOp,
778  Twine("the number of required parallel resources (blocks or "
779  "threads) ") +
780  Twine(computeProduct(numParallelIterations) * factor) +
781  " overflows the number of available resources " +
782  Twine(computeProduct(blockOrGridSizes)));
783  return diag;
784  }
785  return DiagnosedSilenceableFailure::success();
786 }

788 static DiagnosedSilenceableFailure
789 getThreadIdBuilder(std::optional<TransformOpInterface> transformOp,
790  scf::ForallOp forallOp, ArrayRef<int64_t> blockSizes,
791  int64_t warpSize, GpuIdBuilder &gpuIdBuilder) {
792  DeviceMappingAttrInterface mappingAttr =
793  forallOp.getDeviceMappingAttrs().front();
794  bool useLinearMapping = mappingAttr.isLinearMapping();
795 
796  // Sanity checks that may result in runtime verification errors.
797  auto numParallelIterations =
798  getConstantIntValues((forallOp.getMixedUpperBound()));
799  if (!forallOp.isNormalized() || !numParallelIterations.has_value()) {
800  return definiteFailureHelper(
801  transformOp, forallOp,
802  "requires statically sized, normalized forall op");
803  }
804  int64_t factor = 1;
805  if (isa<GPUWarpgroupMappingAttr>(mappingAttr)) {
806  factor = GpuWarpgroupIdBuilder::kNumWarpsPerGroup * warpSize;
807  } else if (isa<GPUWarpMappingAttr>(mappingAttr)) {
808  factor = warpSize;
809  }
810  DiagnosedSilenceableFailure diag =
811  checkMappingSpec(transformOp, forallOp, numParallelIterations.value(),
812  blockSizes, factor, useLinearMapping);
813  if (!diag.succeeded())
814  return diag;
815 
816  FailureOr<DeviceMaskingAttrInterface> maybeMaskingAttr =
817  forallOp.getDeviceMaskingAttr();
818  assert(succeeded(maybeMaskingAttr) && "unexpected failed maybeMaskingAttr");
819  assert((!*maybeMaskingAttr || useLinearMapping) &&
820  "masking requires linear mapping");
821 
822  // Start mapping.
823  MLIRContext *ctx = forallOp.getContext();
824  gpuIdBuilder =
825  TypeSwitch<DeviceMappingAttrInterface, GpuIdBuilder>(mappingAttr)
826  .Case([&](GPUWarpgroupMappingAttr) {
827  return GpuWarpgroupIdBuilder(ctx, warpSize, useLinearMapping,
828  *maybeMaskingAttr);
829  })
830  .Case([&](GPUWarpMappingAttr) {
831  return GpuWarpIdBuilder(ctx, warpSize, useLinearMapping,
832  *maybeMaskingAttr);
833  })
834  .Case([&](GPUThreadMappingAttr) {
835  return GpuThreadIdBuilder(ctx, useLinearMapping, *maybeMaskingAttr);
836  })
837  .Case([&](GPULaneMappingAttr) {
838  return GpuLaneIdBuilder(ctx, warpSize, useLinearMapping,
839  *maybeMaskingAttr);
840  })
841  .Default([&](DeviceMappingAttrInterface) -> GpuIdBuilder {
842  llvm_unreachable("unknown mapping attribute");
843  });
844  return DiagnosedSilenceableFailure::success();
845 }
