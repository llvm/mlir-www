62void transform::ApplyGPUToNVVMConversionPatternsOp::populatePatterns(
63 TypeConverter &typeConverter, RewritePatternSet &patterns) {
64 auto &llvmTypeConverter = static_cast<LLVMTypeConverter &>(typeConverter);
65 // NVVM uses alloca in the default address space to represent private
66 // memory allocations, so drop private annotations. NVVM uses address
67 // space 3 for shared memory. NVVM uses the default address space to
68 // represent global memory.
69 // Used in populateGpuToNVVMConversionPatternsso attaching here for now.
70 // TODO: We should have a single to_nvvm_type_converter.
71 populateGpuMemorySpaceAttributeConversions(
72 llvmTypeConverter, [](AddressSpace space) -> unsigned {
73 switch (space) {
74 case AddressSpace::Global:
75 return static_cast<unsigned>(NVVM::NVVMMemorySpace::Global);
76 case AddressSpace::Workgroup:
77 return static_cast<unsigned>(NVVM::NVVMMemorySpace::Shared);
78 case AddressSpace::Private:
79 return 0;
80 }
81 llvm_unreachable("unknown address space enum value");
82 return static_cast<unsigned>(NVVM::NVVMMemorySpace::Generic);
83 });
84 // Used in GPUToNVVM/WmmaOpsToNvvm.cpp so attaching here for now.
85 // TODO: We should have a single to_nvvm_type_converter.
86 llvmTypeConverter.addConversion(
87 [&](MMAMatrixType type) -> Type { return convertMMAToLLVMType(type); });
88 // Set higher benefit, so patterns will run before generic LLVM lowering.
89 populateGpuToNVVMConversionPatterns(llvmTypeConverter, patterns,
90 getBenefit());
91}

179/// Pick an unrolling order that will allow tensorcore operation to reuse LHS
180/// register.
181static std::optional<SmallVector<int64_t>>
182gpuMmaUnrollOrder(vector::ContractionOp contract) {
183 SmallVector<int64_t> order;
184 // First make reduction the outer dimensions.
185 for (auto [index, iter] : llvm::enumerate(contract.getIteratorTypes())) {
186 if (vector::isReductionIterator(iter)) {
187 order.push_back(index);
188 }
189 }
190
191 llvm::SmallDenseSet<int64_t> dims;
192 for (AffineExpr expr : contract.getIndexingMapsArray()[0].getResults()) {
193 dims.insert(cast<AffineDimExpr>(expr).getPosition());
194 }
195 // Then parallel dimensions that are part of Lhs as we want to re-use Lhs.
196 for (auto [index, iter] : llvm::enumerate(contract.getIteratorTypes())) {
197 if (vector::isParallelIterator(iter) && dims.count(index)) {
198 order.push_back(index);
199 }
200 }
201 // Then the remaining parallel loops.
202 for (auto [index, iter] : llvm::enumerate(contract.getIteratorTypes())) {
203 if (vector::isParallelIterator(iter) && !dims.count(index)) {
204 order.push_back(index);
205 }
206 }
207 return order;
208}

210/// Returns the target vector size for the target operation based on the native
211/// vector size specified with `m`, `n`, and `k`.
212static std::optional<SmallVector<int64_t>>
213getSubgroupMmaNativeVectorSize(Operation *op, int64_t m, int64_t n, int64_t k) {
214 if (auto contract = dyn_cast<vector::ContractionOp>(op)) {
215 int64_t contractRank = contract.getIteratorTypes().size();
216 if (contractRank < 3)
217 return std::nullopt;
218 SmallVector<int64_t> nativeSize(contractRank - 3, 1);
219 nativeSize.append({m, n, k});
220 return nativeSize;
221 }
222 if (auto writeOp = dyn_cast<vector::TransferWriteOp>(op)) {
223 int64_t writeRank = writeOp.getVectorType().getRank();
224 if (writeRank < 2)
225 return std::nullopt;
226 SmallVector<int64_t> nativeSize(writeRank - 2, 1);
227 nativeSize.append({m, n});
228 return nativeSize;
229 }
230 if (auto readOp = dyn_cast<vector::TransferReadOp>(op)) {
231 // Transfer read ops may need different shapes based on how they are being
232 // used. For simplicity just match the shape used by the extract strided op.
233 VectorType sliceType;
234 for (Operation *users : op->getUsers()) {
235 auto extract = dyn_cast<vector::ExtractStridedSliceOp>(users);
236 if (!extract)
237 return std::nullopt;
238 auto vecType = cast<VectorType>(extract.getResult().getType());
239 if (sliceType && sliceType != vecType)
240 return std::nullopt;
241 sliceType = vecType;
242 }
243 return llvm::to_vector(sliceType.getShape());
244 }
245 if ((OpTrait::hasElementwiseMappableTraits(op) && op->getNumResults() == 1)) {
246 if (auto vecType = dyn_cast<VectorType>(op->getResultTypes()[0])) {
247 // TODO: The condition for unrolling elementwise should be restricted
248 // only to operations that need unrolling (connected to the contract).
249 if (vecType.getRank() < 2)
250 return std::nullopt;
251
252 // First check whether there is a slice to infer the shape from. This is
253 // required for cases where the accumulator type differs from the input
254 // types, in which case we will see an `arith.ext_` between the contract
255 // and transfer_read which needs to be unrolled.
256 VectorType sliceType;
257 for (Operation *users : op->getUsers()) {
258 auto extract = dyn_cast<vector::ExtractStridedSliceOp>(users);
259 if (!extract)
260 return std::nullopt;
261 auto vecType = cast<VectorType>(extract.getResult().getType());
262 if (sliceType && sliceType != vecType)
263 return std::nullopt;
264 sliceType = vecType;
265 }
266 if (sliceType)
267 return llvm::to_vector(sliceType.getShape());
268
269 // Else unroll for trailing elementwise.
270 SmallVector<int64_t> nativeSize(vecType.getRank() - 2, 1);
271 // Map elementwise ops to the output shape.
272 nativeSize.append({m, n});
273 return nativeSize;
274 }
275 }
276 return std::nullopt;
277}

279void transform::ApplyUnrollVectorsSubgroupMmaOp::populatePatterns(
280 RewritePatternSet &patterns) {
281 auto unrollOrder = [](Operation *op) -> std::optional<SmallVector<int64_t>> {
282 auto contract = dyn_cast<vector::ContractionOp>(op);
283 if (!contract)
284 return std::nullopt;
285 return gpuMmaUnrollOrder(contract);
286 };
287
288 int64_t m = getM();
289 int64_t n = getN();
290 int64_t k = getK();
291 auto nativeShapeFn =
292 [m, n, k](Operation *op) -> std::optional<SmallVector<int64_t>> {
293 return getSubgroupMmaNativeVectorSize(op, m, n, k);
294 };
295 vector::populateVectorUnrollPatterns(
296 patterns, vector::UnrollVectorOptions()
297 .setNativeShapeFn(nativeShapeFn)
298 .setUnrollTraversalOrderFn(unrollOrder));
299}

328/// Check if given mapping attributes are one of the desired attributes
329template <typename MappingKindType>
330static DiagnosedSilenceableFailure
331checkMappingAttributeTypes(std::optional<TransformOpInterface> transformOp,
332 scf::ForallOp forallOp) {
333 if (!forallOp.getMapping().has_value()) {
334 return definiteFailureHelper(transformOp, forallOp,
335 "scf.forall op requires a mapping attribute");
336 }
337
338 bool hasBlockMapping = llvm::any_of(forallOp.getMapping().value(),
339 llvm::IsaPred<GPUBlockMappingAttr>);
340 bool hasWarpgroupMapping = llvm::any_of(
341 forallOp.getMapping().value(), llvm::IsaPred<GPUWarpgroupMappingAttr>);
342 bool hasWarpMapping = llvm::any_of(forallOp.getMapping().value(),
343 llvm::IsaPred<GPUWarpMappingAttr>);
344 bool hasThreadMapping = llvm::any_of(forallOp.getMapping().value(),
345 llvm::IsaPred<GPUThreadMappingAttr>);
346 bool hasLaneMapping = llvm::any_of(forallOp.getMapping().value(),
347 llvm::IsaPred<GPULaneMappingAttr>);
348 int64_t countMappingTypes = 0;
349 countMappingTypes += hasBlockMapping ? 1 : 0;
350 countMappingTypes += hasWarpgroupMapping ? 1 : 0;
351 countMappingTypes += hasWarpMapping ? 1 : 0;
352 countMappingTypes += hasThreadMapping ? 1 : 0;
353 countMappingTypes += hasLaneMapping ? 1 : 0;
354 if (countMappingTypes > 1) {
355 return definiteFailureHelper(
356 transformOp, forallOp,
357 "cannot mix different mapping types, use nesting");
358 }
359 if (std::is_same<MappingKindType, BlockMappingKind>::value &&
360 !hasBlockMapping) {
361 return definiteFailureHelper(
362 transformOp, forallOp,
363 "scf.forall op requires a mapping attribute of kind 'block'");
364 }
365 if (std::is_same<MappingKindType, ThreadMappingKind>::value &&
366 !hasLaneMapping && !hasThreadMapping && !hasWarpMapping &&
367 !hasWarpgroupMapping) {
368 return definiteFailureHelper(transformOp, forallOp,
369 "scf.forall op requires a mapping attribute "
370 "of kind 'thread' or 'warp'");
371 }
372
373 DenseSet<Attribute> seen;
374 for (Attribute map : forallOp.getMapping()->getValue()) {
375 if (seen.contains(map)) {
376 return definiteFailureHelper(
377 transformOp, forallOp,
378 "duplicate attribute, cannot map different loops "
379 "to the same mapping id");
380 }
381 seen.insert(map);
382 }
383
384 auto isLinear = [](DeviceMappingAttrInterface attr) {
385 return attr.isLinearMapping();
386 };
387 if (llvm::any_of(forallOp.getDeviceMappingAttrs(), isLinear) &&
388 !llvm::all_of(forallOp.getDeviceMappingAttrs(), isLinear)) {
389 return definiteFailureHelper(
390 transformOp, forallOp,
391 "cannot mix linear and non-linear mapping modes");
392 }
393
394 FailureOr<DeviceMaskingAttrInterface> maybeMaskingAttr =
395 forallOp.getDeviceMaskingAttr();
396 if (succeeded(maybeMaskingAttr) && *maybeMaskingAttr &&
397 !forallOp.usesLinearMapping()) {
398 return definiteFailureHelper(
399 transformOp, forallOp,
400 "device masking is only available in linear mapping mode");
401 }
402
403 return DiagnosedSilenceableFailure::success();
404}

406template <typename MappingKindType>
407static DiagnosedSilenceableFailure
408verifyGpuMapping(std::optional<TransformOpInterface> transformOp,
409 scf::ForallOp forallOp) {
410 // Check the types of the mapping attributes match.
411 DiagnosedSilenceableFailure typeRes =
412 checkMappingAttributeTypes<MappingKindType>(transformOp, forallOp);
413 if (!typeRes.succeeded())
414 return typeRes;
415
416 // Perform other non-types verifications.
417 if (!forallOp.isNormalized())
418 return definiteFailureHelper(transformOp, forallOp,
419 "unsupported non-normalized loops");
420 if (forallOp.getNumResults() > 0)
421 return definiteFailureHelper(transformOp, forallOp,
422 "only bufferized scf.forall can be mapped");
423 bool useLinearMapping = forallOp.usesLinearMapping();
424 // TODO: This would be more natural with support for Optional<EnumParameter>
425 // in GPUDeviceMappingAttr.
426 int64_t maxNumMappingsSupported =
427 useLinearMapping ? (getMaxEnumValForMappingId() -
428 static_cast<uint64_t>(MappingId::DimZ))
429 : 3;
430 if (forallOp.getRank() > maxNumMappingsSupported) {
431 return definiteFailureHelper(transformOp, forallOp,
432 "scf.forall with rank > ")
433 << maxNumMappingsSupported
434 << " does not lower for the specified mapping attribute type";
435 }
436 auto numParallelIterations =
437 getConstantIntValues(forallOp.getMixedUpperBound());
438 if (!forallOp.isNormalized() || !numParallelIterations.has_value()) {
439 return definiteFailureHelper(
440 transformOp, forallOp,
441 "requires statically sized, normalized forall op");
442 }
443 return DiagnosedSilenceableFailure::success();
444}

452/// Helper to replace ids of dimensions known to be 1 by 0 to simplify the IR.
453template <typename OpTy, typename OperationOrBlock>
454static void
455replaceUnitMappingIdsHelper(RewriterBase &rewriter, Location loc,
456 OperationOrBlock *parent, Value replacement,
457 ArrayRef<int64_t> availableMappingSizes) {
458 parent->walk([&](OpTy idOp) {

461 });

464static DiagnosedSilenceableFailure rewriteOneForallCommonImpl(
465 RewriterBase &rewriter, std::optional<TransformOpInterface> transformOp,
466 scf::ForallOp forallOp, ArrayRef<int64_t> availableMappingSizes,
467 ForallRewriteResult &result, const GpuIdBuilder &gpuIdBuilder) {
468 LDBG() << "--start rewriteOneForallCommonImpl";
469
470 // Step 1. Complete the mapping to a full mapping (with 1s) if necessary.
471 auto numParallelIterations =
472 getConstantIntValues(forallOp.getMixedUpperBound());
473 assert(forallOp.isNormalized() && numParallelIterations.has_value() &&
474 "requires statically sized, normalized forall op");
475 SmallVector<int64_t> tmpMappingSizes = numParallelIterations.value();
476 SmallVector<DeviceMappingAttrInterface> forallMappingAttrsVec =
477 forallOp.getDeviceMappingAttrs();
478 SetVector<Attribute> forallMappingAttrs;
479 forallMappingAttrs.insert_range(forallMappingAttrsVec);
480 auto comparator = [](Attribute a, Attribute b) -> bool {
481 return cast<DeviceMappingAttrInterface>(a).getMappingId() <
482 cast<DeviceMappingAttrInterface>(b).getMappingId();
483 };
484
485 // Step 1.b. In the linear case, compute the max mapping to avoid needlessly
486 // mapping all dimensions. In the 3-D mapping case we need to map all
487 // dimensions.
488 DeviceMappingAttrInterface maxMapping = cast<DeviceMappingAttrInterface>(
489 *llvm::max_element(forallMappingAttrs, comparator));
490 DeviceMappingAttrInterface maxLinearMapping;
491 if (maxMapping.isLinearMapping())
492 maxLinearMapping = maxMapping;
493 for (auto attr : gpuIdBuilder.mappingAttributes) {
494 // If attr overflows, just skip.
495 if (maxLinearMapping && comparator(maxLinearMapping, attr))
496 continue;
497 // Try to insert. If element was already present, just continue.
498 if (!forallMappingAttrs.insert(attr))
499 continue;
500 // Otherwise, we have a new insertion without a size -> use size 1.
501 tmpMappingSizes.push_back(1);
502 }
503 LDBG() << "----tmpMappingSizes extracted from scf.forall op: "
504 << llvm::interleaved(tmpMappingSizes);
505
506 // Step 2. sort the values by the corresponding DeviceMappingAttrInterface.
507 SmallVector<int64_t> forallMappingSizes = getValuesSortedByKey(
508 forallMappingAttrs.getArrayRef(), tmpMappingSizes, comparator);
509 LDBG() << "----forallMappingSizes: " << llvm::interleaved(forallMappingSizes);
510 LDBG() << "----forallMappingAttrs: " << llvm::interleaved(forallMappingAttrs);
511
512 // Step 3. Generate the mappingIdOps using the provided generator.
513 Location loc = forallOp.getLoc();
514 OpBuilder::InsertionGuard guard(rewriter);
515 rewriter.setInsertionPoint(forallOp);
516 SmallVector<int64_t> originalBasis(availableMappingSizes);
517 bool originalBasisWasProvided = !originalBasis.empty();
518 if (!originalBasisWasProvided) {
519 LDBG() << "----originalBasis was not provided, deriving it and there will "
520 "be no "
521 "predication";
522 originalBasis = forallMappingSizes;
523 while (originalBasis.size() < 3)
524 originalBasis.push_back(1);
525 } else {
526 LDBG() << "----originalBasis was provided, using it, there will be "
527 "predication";
528 }
529 LDBG() << "------originalBasis: " << llvm::interleaved(originalBasis);
530
531 IdBuilderResult builderResult =
532 gpuIdBuilder.idBuilder(rewriter, loc, forallMappingSizes, originalBasis);
533 if (!builderResult.errorMsg.empty())
534 return definiteFailureHelper(transformOp, forallOp, builderResult.errorMsg);
535
536 LDBG() << builderResult;
537
538 // Step 4. Map the induction variables to the mappingIdOps, this may involve
539 // a permutation.
540 SmallVector<Value> mappingIdOps = builderResult.mappingIdOps;
541 IRMapping bvm;
542 for (auto [iv, dim] : llvm::zip_equal(
543 forallOp.getInductionVars(),
544 forallMappingAttrs.getArrayRef().take_front(forallOp.getRank()))) {
545 auto mappingAttr = cast<DeviceMappingAttrInterface>(dim);
546 Value peIdOp = mappingIdOps[mappingAttr.getRelativeIndex()];
547 LDBG() << "----map: " << iv << " to " << peIdOp;
548 bvm.map(iv, peIdOp);
549 }
550
551 // Step 5. If the originalBasis is already known, create conditionals to
552 // predicate the region. Otherwise, the current forall determines the
553 // originalBasis and no predication occurs.
554 Value predicate;
555 if (originalBasisWasProvided) {
556 for (Value tmpPredicate : builderResult.predicateOps) {
557 predicate = predicate ? arith::AndIOp::create(rewriter, loc, predicate,
558 tmpPredicate)
559 : tmpPredicate;
560 }
561 }
562
563 // Step 6. Move the body of forallOp.
564 // Erase the terminator first, it will not be used.
565 rewriter.eraseOp(forallOp.getTerminator());
566 Block *targetBlock;
567 Block::iterator insertionPoint;
568 if (predicate) {
569 // Step 6.a. If predicated, move at the beginning.
570 auto ifOp = scf::IfOp::create(rewriter, loc, predicate,
571 /*withElseRegion=*/false);
572 targetBlock = ifOp.thenBlock();
573 insertionPoint = ifOp.thenBlock()->begin();
574 } else {
575 // Step 6.b. Otherwise, move inline just at the rewriter insertion
576 // point.
577 targetBlock = forallOp->getBlock();
578 insertionPoint = rewriter.getInsertionPoint();
579 }
580 Block &sourceBlock = forallOp.getRegion().front();
581 targetBlock->getOperations().splice(insertionPoint,
582 sourceBlock.getOperations());
583
584 // Step 7. RAUW indices.
585 for (Value loopIndex : forallOp.getInductionVars()) {
586 Value threadIdx = bvm.lookup(loopIndex);
587 rewriter.replaceAllUsesWith(loopIndex, threadIdx);
588 }
589
590 // Step 8. Erase old op.
591 rewriter.eraseOp(forallOp);
592
593 LDBG() << "----result forallMappingSizes: "
594 << llvm::interleaved(forallMappingSizes);
595 LDBG() << "----result mappingIdOps: " << llvm::interleaved(mappingIdOps);
596
597 result = ForallRewriteResult{forallMappingSizes, mappingIdOps};
598 return DiagnosedSilenceableFailure::success();
599}

718 rewriter.eraseOp(topLevelForallOp);

723 bool useLinearMapping = false;

742 // subject to IR inspection.

762static DiagnosedSilenceableFailure checkMappingSpec(
763 std::optional<TransformOpInterface> transformOp, scf::ForallOp forallOp,
764 ArrayRef<int64_t> numParallelIterations, ArrayRef<int64_t> blockOrGridSizes,
765 int factor, bool useLinearMapping = false) {
766 if (!useLinearMapping && blockOrGridSizes.front() % factor != 0) {
767 auto diag = definiteFailureHelper(
768 transformOp, forallOp,
769 Twine("3-D mapping: size of threadIdx.x must be a multiple of ") +
770 Twine(factor));
771 return diag;
772 }
773 if (computeProduct(numParallelIterations) * factor >
774 computeProduct(blockOrGridSizes)) {
775 auto diag = definiteFailureHelper(
776 transformOp, forallOp,
777 Twine("the number of required parallel resources (blocks or "
778 "threads) ") +
779 Twine(computeProduct(numParallelIterations) * factor) +
780 " overflows the number of available resources " +
781 Twine(computeProduct(blockOrGridSizes)));
782 return diag;
783 }
784 return DiagnosedSilenceableFailure::success();

787static DiagnosedSilenceableFailure
788getThreadIdBuilder(std::optional<TransformOpInterface> transformOp,
789 scf::ForallOp forallOp, ArrayRef<int64_t> blockSizes,
790 int64_t warpSize, GpuIdBuilder &gpuIdBuilder) {
791 DeviceMappingAttrInterface mappingAttr =
792 forallOp.getDeviceMappingAttrs().front();
793 bool useLinearMapping = mappingAttr.isLinearMapping();
794
795 // Sanity checks that may result in runtime verification errors.
796 auto numParallelIterations =
797 getConstantIntValues((forallOp.getMixedUpperBound()));
798 if (!forallOp.isNormalized() || !numParallelIterations.has_value()) {
799 return definiteFailureHelper(
800 transformOp, forallOp,
801 "requires statically sized, normalized forall op");
802 }
803 int64_t factor = 1;
804 if (isa<GPUWarpgroupMappingAttr>(mappingAttr)) {
805 factor = GpuWarpgroupIdBuilder::kNumWarpsPerGroup * warpSize;
806 } else if (isa<GPUWarpMappingAttr>(mappingAttr)) {
807 factor = warpSize;
808 }
809 DiagnosedSilenceableFailure diag =
810 checkMappingSpec(transformOp, forallOp, numParallelIterations.value(),
811 blockSizes, factor, useLinearMapping);
812 if (!diag.succeeded())
813 return diag;
814
815 FailureOr<DeviceMaskingAttrInterface> maybeMaskingAttr =
816 forallOp.getDeviceMaskingAttr();
817 assert(succeeded(maybeMaskingAttr) && "unexpected failed maybeMaskingAttr");
818 assert((!*maybeMaskingAttr || useLinearMapping) &&
819 "masking requires linear mapping");
820
821 // Start mapping.
822 MLIRContext *ctx = forallOp.getContext();
823 gpuIdBuilder =
824 TypeSwitch<DeviceMappingAttrInterface, GpuIdBuilder>(mappingAttr)
825 .Case([&](GPUWarpgroupMappingAttr) {
826 return GpuWarpgroupIdBuilder(ctx, warpSize, useLinearMapping,
827 *maybeMaskingAttr);
828 })
829 .Case([&](GPUWarpMappingAttr) {
830 return GpuWarpIdBuilder(ctx, warpSize, useLinearMapping,
831 *maybeMaskingAttr);
832 })
833 .Case([&](GPUThreadMappingAttr) {
834 return GpuThreadIdBuilder(ctx, useLinearMapping, *maybeMaskingAttr);
835 })
836 .Case([&](GPULaneMappingAttr) {
837 return GpuLaneIdBuilder(ctx, warpSize, useLinearMapping,
838 *maybeMaskingAttr);
839 })
840 .DefaultUnreachable("unknown mapping attribute");
841 return DiagnosedSilenceableFailure::success();
842}
