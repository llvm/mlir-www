22 /// Convert vector.maskedload
23 ///
24 /// Before:
25 ///
26 /// vector.maskedload %base[%idx_0, %idx_1], %mask, %pass_thru
27 ///
28 /// After:
29 ///
30 /// %ivalue = %pass_thru
31 /// %m = vector.extract %mask[0]
32 /// %result0 = scf.if %m {
33 /// %v = memref.load %base[%idx_0, %idx_1]
34 /// %combined = vector.insert %v, %ivalue[0]
35 /// scf.yield %combined
36 /// } else {
37 /// scf.yield %ivalue
38 /// }
39 /// %m = vector.extract %mask[1]
40 /// %result1 = scf.if %m {
41 /// %v = memref.load %base[%idx_0, %idx_1 + 1]
42 /// %combined = vector.insert %v, %result0[1]
43 /// scf.yield %combined
44 /// } else {
45 /// scf.yield %result0
46 /// }
47 /// ...
48 ///
49 struct VectorMaskedLoadOpConverter final
50  : OpRewritePattern<vector::MaskedLoadOp> {
51  using OpRewritePattern::OpRewritePattern;
52 
53  LogicalResult matchAndRewrite(vector::MaskedLoadOp maskedLoadOp,
54  PatternRewriter &rewriter) const override {
55  VectorType maskVType = maskedLoadOp.getMaskVectorType();
56  if (maskVType.getShape().size() != 1)
57  return rewriter.notifyMatchFailure(
58  maskedLoadOp, "expected vector.maskedstore with 1-D mask");
59 
60  Location loc = maskedLoadOp.getLoc();
61  int64_t maskLength = maskVType.getShape()[0];
62 
63  Type indexType = rewriter.getIndexType();
64  Value mask = maskedLoadOp.getMask();
65  Value base = maskedLoadOp.getBase();
66  Value iValue = maskedLoadOp.getPassThru();
67  auto indices = llvm::to_vector_of<Value>(maskedLoadOp.getIndices());
68  Value one = arith::ConstantOp::create(rewriter, loc, indexType,
69  IntegerAttr::get(indexType, 1));
70  for (int64_t i = 0; i < maskLength; ++i) {
71  auto maskBit = vector::ExtractOp::create(rewriter, loc, mask, i);
72 
73  auto ifOp = scf::IfOp::create(
74  rewriter, loc, maskBit,
75  [&](OpBuilder &builder, Location loc) {
76  auto loadedValue =
77  memref::LoadOp::create(builder, loc, base, indices);
78  auto combinedValue =
79  vector::InsertOp::create(builder, loc, loadedValue, iValue, i);
80  scf::YieldOp::create(builder, loc, combinedValue.getResult());
81  },
82  [&](OpBuilder &builder, Location loc) {
83  scf::YieldOp::create(builder, loc, iValue);
84  });
85  iValue = ifOp.getResult(0);
86 
87  indices.back() =
88  arith::AddIOp::create(rewriter, loc, indices.back(), one);
89  }
90 
91  rewriter.replaceOp(maskedLoadOp, iValue);
92 
93  return success();
94  }
95 };

97 /// Convert vector.maskedstore
98 ///
99 /// Before:
100 ///
101 /// vector.maskedstore %base[%idx_0, %idx_1], %mask, %value
102 ///
103 /// After:
104 ///
105 /// %m = vector.extract %mask[0]
106 /// scf.if %m {
107 /// %extracted = vector.extract %value[0]
108 /// memref.store %extracted, %base[%idx_0, %idx_1]
109 /// }
110 /// %m = vector.extract %mask[1]
111 /// scf.if %m {
112 /// %extracted = vector.extract %value[1]
113 /// memref.store %extracted, %base[%idx_0, %idx_1 + 1]
114 /// }
115 /// ...
116 ///
117 struct VectorMaskedStoreOpConverter final
118  : OpRewritePattern<vector::MaskedStoreOp> {
119  using OpRewritePattern::OpRewritePattern;
120 
121  LogicalResult matchAndRewrite(vector::MaskedStoreOp maskedStoreOp,
122  PatternRewriter &rewriter) const override {
123  VectorType maskVType = maskedStoreOp.getMaskVectorType();
124  if (maskVType.getShape().size() != 1)
125  return rewriter.notifyMatchFailure(
126  maskedStoreOp, "expected vector.maskedstore with 1-D mask");
127 
128  Location loc = maskedStoreOp.getLoc();
129  int64_t maskLength = maskVType.getShape()[0];
130 
131  Type indexType = rewriter.getIndexType();
132  Value mask = maskedStoreOp.getMask();
133  Value base = maskedStoreOp.getBase();
134  Value value = maskedStoreOp.getValueToStore();
135  auto indices = llvm::to_vector_of<Value>(maskedStoreOp.getIndices());
136  Value one = arith::ConstantOp::create(rewriter, loc, indexType,
137  IntegerAttr::get(indexType, 1));
138  for (int64_t i = 0; i < maskLength; ++i) {
139  auto maskBit = vector::ExtractOp::create(rewriter, loc, mask, i);
140 
141  auto ifOp = scf::IfOp::create(rewriter, loc, maskBit, /*else=*/false);
142  rewriter.setInsertionPointToStart(&ifOp.getThenRegion().front());
143  auto extractedValue = vector::ExtractOp::create(rewriter, loc, value, i);
144  memref::StoreOp::create(rewriter, loc, extractedValue, base, indices);
145 
146  rewriter.setInsertionPointAfter(ifOp);
147  indices.back() =
148  arith::AddIOp::create(rewriter, loc, indices.back(), one);
149  }
150 
151  rewriter.eraseOp(maskedStoreOp);
152 
153  return success();
154  }
155 };
