134 /// For NVIDIA GPUs there are 3 compilation format options:
135 /// 1. `isa`: the compiler generates PTX and the driver JITs the PTX.
136 /// 2. `bin`: generates a CUBIN object for `chip=gpuChip`.
137 /// 3. `fatbin`: generates a fat binary with a CUBIN object for `gpuChip` and
138 /// also embeds the PTX in the fat binary.
139 /// Notes:
140 /// Option 1 adds a significant runtime performance hit, however, tests are
141 /// more likely to pass with this option.
142 /// Option 2 is better for execution time as there is no JIT; however, the
143 /// program will fail if there's an architecture mismatch between `gpuChip`
144 /// and the GPU running the program.
145 /// Option 3 is the best compromise between options 1 and 2 as it can JIT in
146 /// case of an architecture mismatch between `gpuChip` and the running
147 /// architecture. However, it's only possible to JIT to a higher CC than
148 /// `gpuChip`.
149 PassOptions::Option<std::string> gpuFormat{
150 *this, "gpu-format", desc("GPU compilation format"), init("fatbin")};
151
152 /// This option is used to enable GPU library generation.
153 PassOptions::Option<bool> enableGPULibgen{
154 *this, "enable-gpu-libgen",
155 desc("Enables GPU acceleration by means of direct library calls (like "
156 "cuSPARSE)")};

173 return opts;
174 }
