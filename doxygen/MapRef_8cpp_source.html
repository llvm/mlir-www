22bool mlir::sparse_tensor::MapRef::isPermutationMap() const {
23 if (dimRank != lvlRank)
24 return false;
25 std::vector<bool> seen(dimRank, false);
26 for (uint64_t l = 0; l < lvlRank; l++) {
27 const uint64_t d = dim2lvl[l];
28 if (d >= dimRank || seen[d])
29 return false;
30 seen[d] = true;
31 }
32 return true;
33}
