28  Operation *op, const OneShotBufferizationOptions &options,
29  const BufferizationState &bufferizationState,
30  BufferizationStatistics *statistics) {

32  // Run normal One-Shot Bufferize analysis or One-Shot Module Bufferize
33  // analysis depending on whether function boundary bufferization is enabled or
34  // not.
35  if (options.bufferizeFunctionBoundaries) {
36  if (failed(analyzeModuleOp(op, analysisState, statistics)))
37  return failure();

39  if (failed(analyzeOp(op, analysisState, statistics)))
40  return failure();
41  }
42 
43  if (options.testAnalysisOnly)
44  return success();
45 
46  return insertTensorCopies(op, analysisState, bufferizationState);

49 LogicalResult mlir::bufferization::insertTensorCopies(
50  Operation *op, const AnalysisState &analysisState,
51  const BufferizationState &bufferizationState) {
52  IRRewriter rewriter(op->getContext());
53 
54  // It may be more efficient to walk in pre-order here, but the current
55  // implementation visits regions of ops even if they are not allowed or
56  // bufferizable, and existing tests rely on this behavior.
57  // For now, only exclude nested operations if they are in a different symbol
58  // table scope.
59  WalkResult result = op->walk([&](Operation *nestedOp) {
60  if (op->hasTrait<OpTrait::SymbolTable>() &&
61  nestedOp->getParentWithTrait<OpTrait::SymbolTable>() != op)
62  return WalkResult::skip();
63 
64  auto bufferizableOp =
65  analysisState.getOptions().dynCastBufferizableOp(nestedOp);
66  if (!bufferizableOp)
67  return WalkResult::skip();
68 
69  // Find inplacability conflicts and resolve them. (Typically with explicit
70  // tensor copies in the form of AllocTensorOps.)
71  rewriter.setInsertionPoint(nestedOp);
72  if (failed(bufferizableOp.resolveConflicts(rewriter, analysisState,
73  bufferizationState)))
74  return WalkResult::interrupt();
75 
76  return WalkResult::advance();
77  });
78 
79  return failure(result.wasInterrupted());
80 }
