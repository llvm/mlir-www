90 /// Verifies that the first block of the given `region` is terminated by a
91 /// TerminatorTy. Reports errors on the given operation if it is not the case.
92 template <typename TerminatorTy>
93 static TerminatorTy verifyAndGetTerminator(Operation *op, Region &region,
94  StringRef errorMessage) {
95  Operation *terminatorOperation = nullptr;
96  if (!region.empty() && !region.front().empty()) {
97  terminatorOperation = &region.front().back();
98  if (auto yield = dyn_cast_or_null<TerminatorTy>(terminatorOperation))
99  return yield;
100  }
101  auto diag = op->emitOpError(errorMessage);
102  if (terminatorOperation)
103  diag.attachNote(terminatorOperation->getLoc()) << "terminator here";
104  return nullptr;
105 }

111 /// Replaces the given op with the contents of the given single-block region,
112 /// using the operands of the block terminator to replace operation results.
113 static void replaceOpWithRegion(PatternRewriter &rewriter, Operation *op,
114  Region &region, ValueRange blockArgs = {}) {
115  assert(region.hasOneBlock() && "expected single-block region");
116  Block *block = &region.front();
117  Operation *terminator = block->getTerminator();
118  ValueRange results = terminator->getOperands();
119  rewriter.inlineBlockBefore(block, op, blockArgs);
120  rewriter.replaceOp(op, results);
121  rewriter.eraseOp(terminator);
122 }

198 // Inline an ExecuteRegionOp if its parent can contain multiple blocks.
199 // TODO generalize the conditions for operations which can be inlined into.
200 // func @func_execute_region_elim() {
201 // "test.foo"() : () -> ()
202 // %v = scf.execute_region -> i64 {
203 // %c = "test.cmp"() : () -> i1
204 // cf.cond_br %c, ^bb2, ^bb3
205 // ^bb2:
206 // %x = "test.val1"() : () -> i64
207 // cf.br ^bb4(%x : i64)
208 // ^bb3:
209 // %y = "test.val2"() : () -> i64
210 // cf.br ^bb4(%y : i64)
211 // ^bb4(%z : i64):
212 // scf.yield %z : i64
213 // }
214 // "test.bar"(%v) : (i64) -> ()
215 // return
216 // }
217 //
218 // becomes
219 //
220 // func @func_execute_region_elim() {
221 // "test.foo"() : () -> ()
222 // %c = "test.cmp"() : () -> i1
223 // cf.cond_br %c, ^bb1, ^bb2
224 // ^bb1: // pred: ^bb0
225 // %x = "test.val1"() : () -> i64
226 // cf.br ^bb3(%x : i64)
227 // ^bb2: // pred: ^bb0
228 // %y = "test.val2"() : () -> i64
229 // cf.br ^bb3(%y : i64)
230 // ^bb3(%z: i64): // 2 preds: ^bb1, ^bb2
231 // "test.bar"(%z) : (i64) -> ()
232 // return
233 // }
234 //
235 struct MultiBlockExecuteInliner : public OpRewritePattern<ExecuteRegionOp> {
236  using OpRewritePattern<ExecuteRegionOp>::OpRewritePattern;
237 
238  LogicalResult matchAndRewrite(ExecuteRegionOp op,
239  PatternRewriter &rewriter) const override {
240  if (!isa<FunctionOpInterface, ExecuteRegionOp>(op->getParentOp()))
241  return failure();
242 
243  Block *prevBlock = op->getBlock();
244  Block *postBlock = rewriter.splitBlock(prevBlock, op->getIterator());
245  rewriter.setInsertionPointToEnd(prevBlock);
246 
247  cf::BranchOp::create(rewriter, op.getLoc(), &op.getRegion().front());
248 
249  for (Block &blk : op.getRegion()) {
250  if (YieldOp yieldOp = dyn_cast<YieldOp>(blk.getTerminator())) {
251  rewriter.setInsertionPoint(yieldOp);
252  cf::BranchOp::create(rewriter, yieldOp.getLoc(), postBlock,
253  yieldOp.getResults());
254  rewriter.eraseOp(yieldOp);
255  }
256  }
257 
258  rewriter.inlineRegionBefore(op.getRegion(), postBlock);
259  SmallVector<Value> blockArgs;
260 
261  for (auto res : op.getResults())
262  blockArgs.push_back(postBlock->addArgument(res.getType(), res.getLoc()));
263 
264  rewriter.replaceOp(op, blockArgs);
265  return success();
266  }
267 };

319 void ForOp::build(OpBuilder &builder, OperationState &result, Value lb,
320  Value ub, Value step, ValueRange initArgs,
321  BodyBuilderFn bodyBuilder) {
322  OpBuilder::InsertionGuard guard(builder);
323 
324  result.addOperands({lb, ub, step});
325  result.addOperands(initArgs);
326  for (Value v : initArgs)
327  result.addTypes(v.getType());
328  Type t = lb.getType();
329  Region *bodyRegion = result.addRegion();
330  Block *bodyBlock = builder.createBlock(bodyRegion);
331  bodyBlock->addArgument(t, result.location);
332  for (Value v : initArgs)
333  bodyBlock->addArgument(v.getType(), v.getLoc());
334 
335  // Create the default terminator if the builder is not provided and if the
336  // iteration arguments are not provided. Otherwise, leave this to the caller
337  // because we don't know which values to return from the loop.
338  if (initArgs.empty() && !bodyBuilder) {
339  ForOp::ensureTerminator(*bodyRegion, builder, result.location);
340  } else if (bodyBuilder) {
341  OpBuilder::InsertionGuard guard(builder);
342  builder.setInsertionPointToStart(bodyBlock);
343  bodyBuilder(builder, result.location, bodyBlock->getArgument(0),
344  bodyBlock->getArguments().drop_front());
345  }
346 }

403 /// Promotes the loop body of a forOp to its containing block if the forOp
404 /// it can be determined that the loop has a single iteration.
405 LogicalResult ForOp::promoteIfSingleIteration(RewriterBase &rewriter) {
406  std::optional<int64_t> tripCount =
407  constantTripCount(getLowerBound(), getUpperBound(), getStep());
408  if (!tripCount.has_value() || tripCount != 1)
409  return failure();
410 
411  // Replace all results with the yielded values.
412  auto yieldOp = cast<scf::YieldOp>(getBody()->getTerminator());
413  rewriter.replaceAllUsesWith(getResults(), getYieldedValues());
414 
415  // Replace block arguments with lower bound (replacement for IV) and
416  // iter_args.
417  SmallVector<Value> bbArgReplacements;
418  bbArgReplacements.push_back(getLowerBound());
419  llvm::append_range(bbArgReplacements, getInitArgs());
420 
421  // Move the loop body operations to the loop's containing block.
422  rewriter.inlineBlockBefore(getBody(), getOperation()->getBlock(),
423  getOperation()->getIterator(), bbArgReplacements);
424 
425  // Erase the old terminator and the loop.
426  rewriter.eraseOp(yieldOp);
427  rewriter.eraseOp(*this);
428 
429  return success();
430 }

468 ParseResult ForOp::parse(OpAsmParser &parser, OperationState &result) {
469  auto &builder = parser.getBuilder();
470  Type type;
471 
472  OpAsmParser::Argument inductionVariable;
473  OpAsmParser::UnresolvedOperand lb, ub, step;
474 
475  // Parse the induction variable followed by '='.
476  if (parser.parseOperand(inductionVariable.ssaName) || parser.parseEqual() ||
477  // Parse loop bounds.
478  parser.parseOperand(lb) || parser.parseKeyword("to") ||
479  parser.parseOperand(ub) || parser.parseKeyword("step") ||
480  parser.parseOperand(step))
481  return failure();
482 
483  // Parse the optional initial iteration arguments.
484  SmallVector<OpAsmParser::Argument, 4> regionArgs;
485  SmallVector<OpAsmParser::UnresolvedOperand, 4> operands;
486  regionArgs.push_back(inductionVariable);
487 
488  bool hasIterArgs = succeeded(parser.parseOptionalKeyword("iter_args"));
489  if (hasIterArgs) {
490  // Parse assignment list and results type list.
491  if (parser.parseAssignmentList(regionArgs, operands) ||
492  parser.parseArrowTypeList(result.types))
493  return failure();
494  }
495 
496  if (regionArgs.size() != result.types.size() + 1)
497  return parser.emitError(
498  parser.getNameLoc(),
499  "mismatch in number of loop-carried values and defined values");
500 
501  // Parse optional type, else assume Index.
502  if (parser.parseOptionalColon())
503  type = builder.getIndexType();
504  else if (parser.parseType(type))
505  return failure();
506 
507  // Set block argument types, so that they are known when parsing the region.
508  regionArgs.front().type = type;
509  for (auto [iterArg, type] :
510  llvm::zip_equal(llvm::drop_begin(regionArgs), result.types))
511  iterArg.type = type;
512 
513  // Parse the body region.
514  Region *body = result.addRegion();
515  if (parser.parseRegion(*body, regionArgs))
516  return failure();
517  ForOp::ensureTerminator(*body, builder, result.location);
518 
519  // Resolve input operands. This should be done after parsing the region to
520  // catch invalid IR where operands were defined inside of the region.
521  if (parser.resolveOperand(lb, type, result.operands) ||
522  parser.resolveOperand(ub, type, result.operands) ||
523  parser.resolveOperand(step, type, result.operands))
524  return failure();
525  if (hasIterArgs) {
526  for (auto argOperandType : llvm::zip_equal(llvm::drop_begin(regionArgs),
527  operands, result.types)) {
528  Type type = std::get<2>(argOperandType);
529  std::get<0>(argOperandType).type = type;
530  if (parser.resolveOperand(std::get<1>(argOperandType), type,
531  result.operands))
532  return failure();
533  }
534  }
535 
536  // Parse the optional attribute list.
537  if (parser.parseOptionalAttrDict(result.attributes))
538  return failure();
539 
540  return success();
541 }

553 FailureOr<LoopLikeOpInterface>
554 ForOp::replaceWithAdditionalYields(RewriterBase &rewriter,
555  ValueRange newInitOperands,
556  bool replaceInitOperandUsesInLoop,
557  const NewYieldValuesFn &newYieldValuesFn) {
558  // Create a new loop before the existing one, with the extra operands.
559  OpBuilder::InsertionGuard g(rewriter);
560  rewriter.setInsertionPoint(getOperation());
561  auto inits = llvm::to_vector(getInitArgs());
562  inits.append(newInitOperands.begin(), newInitOperands.end());
563  scf::ForOp newLoop = scf::ForOp::create(
564  rewriter, getLoc(), getLowerBound(), getUpperBound(), getStep(), inits,
565  [](OpBuilder &, Location, Value, ValueRange) {});
566  newLoop->setAttrs(getPrunedAttributeList(getOperation(), {}));
567 
568  // Generate the new yield values and append them to the scf.yield operation.
569  auto yieldOp = cast<scf::YieldOp>(getBody()->getTerminator());
570  ArrayRef<BlockArgument> newIterArgs =
571  newLoop.getBody()->getArguments().take_back(newInitOperands.size());
572  {
573  OpBuilder::InsertionGuard g(rewriter);
574  rewriter.setInsertionPoint(yieldOp);
575  SmallVector<Value> newYieldedValues =
576  newYieldValuesFn(rewriter, getLoc(), newIterArgs);
577  assert(newInitOperands.size() == newYieldedValues.size() &&
578  "expected as many new yield values as new iter operands");
579  rewriter.modifyOpInPlace(yieldOp, [&]() {
580  yieldOp.getResultsMutable().append(newYieldedValues);
581  });
582  }
583 
584  // Move the loop body to the new op.
585  rewriter.mergeBlocks(getBody(), newLoop.getBody(),
586  newLoop.getBody()->getArguments().take_front(
587  getBody()->getNumArguments()));
588 
589  if (replaceInitOperandUsesInLoop) {
590  // Replace all uses of `newInitOperands` with the corresponding basic block
591  // arguments.
592  for (auto it : llvm::zip(newInitOperands, newIterArgs)) {
593  rewriter.replaceUsesWithIf(std::get<0>(it), std::get<1>(it),
594  [&](OpOperand &use) {
595  Operation *user = use.getOwner();
596  return newLoop->isProperAncestor(user);
597  });
598  }
599  }
600 
601  // Replace the old loop.
602  rewriter.replaceOp(getOperation(),
603  newLoop->getResults().take_front(getNumResults()));
604  return cast<LoopLikeOpInterface>(newLoop.getOperation());
605 }

608  auto ivArg = llvm::dyn_cast<BlockArgument>(val);

654 void mlir::scf::promote(RewriterBase &rewriter, scf::ForallOp forallOp) {
655  OpBuilder::InsertionGuard g(rewriter);
656  scf::InParallelOp terminator = forallOp.getTerminator();
657 
658  // Replace block arguments with lower bounds (replacements for IVs) and
659  // outputs.
660  SmallVector<Value> bbArgReplacements = forallOp.getLowerBound(rewriter);
661  bbArgReplacements.append(forallOp.getOutputs().begin(),
662  forallOp.getOutputs().end());
663 
664  // Move the loop body operations to the loop's containing block.
665  rewriter.inlineBlockBefore(forallOp.getBody(), forallOp->getBlock(),
666  forallOp->getIterator(), bbArgReplacements);
667 
668  // Replace the terminator with tensor.insert_slice ops.
669  rewriter.setInsertionPointAfter(forallOp);
670  SmallVector<Value> results;
671  results.reserve(forallOp.getResults().size());
672  for (auto &yieldingOp : terminator.getYieldingOps()) {
673  auto parallelInsertSliceOp =
674  cast<tensor::ParallelInsertSliceOp>(yieldingOp);
675 
676  Value dst = parallelInsertSliceOp.getDest();
677  Value src = parallelInsertSliceOp.getSource();
678  if (llvm::isa<TensorType>(src.getType())) {
679  results.push_back(tensor::InsertSliceOp::create(
680  rewriter, forallOp.getLoc(), dst.getType(), src, dst,
681  parallelInsertSliceOp.getOffsets(), parallelInsertSliceOp.getSizes(),
682  parallelInsertSliceOp.getStrides(),
683  parallelInsertSliceOp.getStaticOffsets(),
684  parallelInsertSliceOp.getStaticSizes(),
685  parallelInsertSliceOp.getStaticStrides()));
686  } else {
687  llvm_unreachable("unsupported terminator");
688  }
689  }
690  rewriter.replaceAllUsesWith(forallOp.getResults(), results);
691 
692  // Erase the old terminator and the loop.
693  rewriter.eraseOp(terminator);
694  rewriter.eraseOp(forallOp);

697 LoopNest mlir::scf::buildLoopNest(
698  OpBuilder &builder, Location loc, ValueRange lbs, ValueRange ubs,
699  ValueRange steps, ValueRange iterArgs,
700  function_ref<ValueVector(OpBuilder &, Location, ValueRange, ValueRange)>
701  bodyBuilder) {
702  assert(lbs.size() == ubs.size() &&
703  "expected the same number of lower and upper bounds");
704  assert(lbs.size() == steps.size() &&
705  "expected the same number of lower bounds and steps");
706 
707  // If there are no bounds, call the body-building function and return early.
708  if (lbs.empty()) {
709  ValueVector results =
710  bodyBuilder ? bodyBuilder(builder, loc, ValueRange(), iterArgs)
711  : ValueVector();
712  assert(results.size() == iterArgs.size() &&
713  "loop nest body must return as many values as loop has iteration "
714  "arguments");
715  return LoopNest{{}, std::move(results)};
716  }
717 
718  // First, create the loop structure iteratively using the body-builder
719  // callback of `ForOp::build`. Do not create `YieldOp`s yet.
720  OpBuilder::InsertionGuard guard(builder);
721  SmallVector<scf::ForOp, 4> loops;
722  SmallVector<Value, 4> ivs;
723  loops.reserve(lbs.size());
724  ivs.reserve(lbs.size());
725  ValueRange currentIterArgs = iterArgs;
726  Location currentLoc = loc;
727  for (unsigned i = 0, e = lbs.size(); i < e; ++i) {
728  auto loop = scf::ForOp::create(
729  builder, currentLoc, lbs[i], ubs[i], steps[i], currentIterArgs,
730  [&](OpBuilder &nestedBuilder, Location nestedLoc, Value iv,
731  ValueRange args) {
732  ivs.push_back(iv);
733  // It is safe to store ValueRange args because it points to block
734  // arguments of a loop operation that we also own.
735  currentIterArgs = args;
736  currentLoc = nestedLoc;
737  });
738  // Set the builder to point to the body of the newly created loop. We don't
739  // do this in the callback because the builder is reset when the callback
740  // returns.
741  builder.setInsertionPointToStart(loop.getBody());
742  loops.push_back(loop);
743  }
744 
745  // For all loops but the innermost, yield the results of the nested loop.
746  for (unsigned i = 0, e = loops.size() - 1; i < e; ++i) {
747  builder.setInsertionPointToEnd(loops[i].getBody());
748  scf::YieldOp::create(builder, loc, loops[i + 1].getResults());
749  }
750 
751  // In the body of the innermost loop, call the body building function if any
752  // and yield its results.
753  builder.setInsertionPointToStart(loops.back().getBody());
754  ValueVector results = bodyBuilder
755  ? bodyBuilder(builder, currentLoc, ivs,
756  loops.back().getRegionIterArgs())
757  : ValueVector();
758  assert(results.size() == iterArgs.size() &&
759  "loop nest body must return as many values as loop has iteration "
760  "arguments");
761  builder.setInsertionPointToEnd(loops.back().getBody());
762  scf::YieldOp::create(builder, loc, results);
763 
764  // Return the loops.
765  ValueVector nestResults;
766  llvm::append_range(nestResults, loops.front().getResults());
767  return LoopNest{std::move(loops), std::move(nestResults)};
768 }

785 SmallVector<Value>
786 mlir::scf::replaceAndCastForOpIterArg(RewriterBase &rewriter, scf::ForOp forOp,
787  OpOperand &operand, Value replacement,
788  const ValueTypeCastFnTy &castFn) {
789  assert(operand.getOwner() == forOp);
790  Type oldType = operand.get().getType(), newType = replacement.getType();
791 
792  // 1. Create new iter operands, exactly 1 is replaced.
793  assert(operand.getOperandNumber() >= forOp.getNumControlOperands() &&
794  "expected an iter OpOperand");
795  assert(operand.get().getType() != replacement.getType() &&
796  "Expected a different type");
797  SmallVector<Value> newIterOperands;
798  for (OpOperand &opOperand : forOp.getInitArgsMutable()) {
799  if (opOperand.getOperandNumber() == operand.getOperandNumber()) {
800  newIterOperands.push_back(replacement);
801  continue;
802  }
803  newIterOperands.push_back(opOperand.get());
804  }
805 
806  // 2. Create the new forOp shell.
807  scf::ForOp newForOp = scf::ForOp::create(
808  rewriter, forOp.getLoc(), forOp.getLowerBound(), forOp.getUpperBound(),
809  forOp.getStep(), newIterOperands);
810  newForOp->setAttrs(forOp->getAttrs());
811  Block &newBlock = newForOp.getRegion().front();
812  SmallVector<Value, 4> newBlockTransferArgs(newBlock.getArguments().begin(),
813  newBlock.getArguments().end());
814 
815  // 3. Inject an incoming cast op at the beginning of the block for the bbArg
816  // corresponding to the `replacement` value.
817  OpBuilder::InsertionGuard g(rewriter);
818  rewriter.setInsertionPointToStart(&newBlock);
819  BlockArgument newRegionIterArg = newForOp.getTiedLoopRegionIterArg(
820  &newForOp->getOpOperand(operand.getOperandNumber()));
821  Value castIn = castFn(rewriter, newForOp.getLoc(), oldType, newRegionIterArg);
822  newBlockTransferArgs[newRegionIterArg.getArgNumber()] = castIn;
823 
824  // 4. Steal the old block ops, mapping to the newBlockTransferArgs.
825  Block &oldBlock = forOp.getRegion().front();
826  rewriter.mergeBlocks(&oldBlock, &newBlock, newBlockTransferArgs);
827 
828  // 5. Inject an outgoing cast op at the end of the block and yield it instead.
829  auto clonedYieldOp = cast<scf::YieldOp>(newBlock.getTerminator());
830  rewriter.setInsertionPoint(clonedYieldOp);
831  unsigned yieldIdx =
832  newRegionIterArg.getArgNumber() - forOp.getNumInductionVars();
833  Value castOut = castFn(rewriter, newForOp.getLoc(), newType,
834  clonedYieldOp.getOperand(yieldIdx));
835  SmallVector<Value> newYieldOperands = clonedYieldOp.getOperands();
836  newYieldOperands[yieldIdx] = castOut;
837  scf::YieldOp::create(rewriter, newForOp.getLoc(), newYieldOperands);
838  rewriter.eraseOp(clonedYieldOp);
839 
840  // 6. Inject an outgoing cast op after the forOp.
841  rewriter.setInsertionPointAfter(newForOp);
842  SmallVector<Value> newResults = newForOp.getResults();
843  newResults[yieldIdx] =
844  castFn(rewriter, newForOp.getLoc(), oldType, newResults[yieldIdx]);
845 
846  return newResults;
847 }

850 // Fold away ForOp iter arguments when:
851 // 1) The op yields the iter arguments.
852 // 2) The argument's corresponding outer region iterators (inputs) are yielded.
853 // 3) The iter arguments have no use and the corresponding (operation) results
854 // have no use.
855 //
856 // These arguments must be defined outside of the ForOp region and can just be
857 // forwarded after simplifying the op inits, yields and returns.
858 //
859 // The implementation uses `inlineBlockBefore` to steal the content of the
860 // original ForOp and avoid cloning.
861 struct ForOpIterArgsFolder : public OpRewritePattern<scf::ForOp> {
862  using OpRewritePattern<scf::ForOp>::OpRewritePattern;
863 
864  LogicalResult matchAndRewrite(scf::ForOp forOp,
865  PatternRewriter &rewriter) const final {
866  bool canonicalize = false;
867 
868  // An internal flat vector of block transfer
869  // arguments `newBlockTransferArgs` keeps the 1-1 mapping of original to
870  // transformed block argument mappings. This plays the role of a
871  // IRMapping for the particular use case of calling into
872  // `inlineBlockBefore`.
873  int64_t numResults = forOp.getNumResults();
874  SmallVector<bool, 4> keepMask;
875  keepMask.reserve(numResults);
876  SmallVector<Value, 4> newBlockTransferArgs, newIterArgs, newYieldValues,
877  newResultValues;
878  newBlockTransferArgs.reserve(1 + numResults);
879  newBlockTransferArgs.push_back(Value()); // iv placeholder with null value
880  newIterArgs.reserve(forOp.getInitArgs().size());
881  newYieldValues.reserve(numResults);
882  newResultValues.reserve(numResults);
883  DenseMap<std::pair<Value, Value>, std::pair<Value, Value>> initYieldToArg;
884  for (auto [init, arg, result, yielded] :
885  llvm::zip(forOp.getInitArgs(), // iter from outside
886  forOp.getRegionIterArgs(), // iter inside region
887  forOp.getResults(), // op results
888  forOp.getYieldedValues() // iter yield
889  )) {
890  // Forwarded is `true` when:
891  // 1) The region `iter` argument is yielded.
892  // 2) The region `iter` argument the corresponding input is yielded.
893  // 3) The region `iter` argument has no use, and the corresponding op
894  // result has no use.
895  bool forwarded = (arg == yielded) || (init == yielded) ||
896  (arg.use_empty() && result.use_empty());
897  if (forwarded) {
898  canonicalize = true;
899  keepMask.push_back(false);
900  newBlockTransferArgs.push_back(init);
901  newResultValues.push_back(init);
902  continue;
903  }
904 
905  // Check if a previous kept argument always has the same values for init
906  // and yielded values.
907  if (auto it = initYieldToArg.find({init, yielded});
908  it != initYieldToArg.end()) {
909  canonicalize = true;
910  keepMask.push_back(false);
911  auto [sameArg, sameResult] = it->second;
912  rewriter.replaceAllUsesWith(arg, sameArg);
913  rewriter.replaceAllUsesWith(result, sameResult);
914  // The replacement value doesn't matter because there are no uses.
915  newBlockTransferArgs.push_back(init);
916  newResultValues.push_back(init);
917  continue;
918  }
919 
920  // This value is kept.
921  initYieldToArg.insert({{init, yielded}, {arg, result}});
922  keepMask.push_back(true);
923  newIterArgs.push_back(init);
924  newYieldValues.push_back(yielded);
925  newBlockTransferArgs.push_back(Value()); // placeholder with null value
926  newResultValues.push_back(Value()); // placeholder with null value
927  }
928 
929  if (!canonicalize)
930  return failure();
931 
932  scf::ForOp newForOp =
933  scf::ForOp::create(rewriter, forOp.getLoc(), forOp.getLowerBound(),
934  forOp.getUpperBound(), forOp.getStep(), newIterArgs);
935  newForOp->setAttrs(forOp->getAttrs());
936  Block &newBlock = newForOp.getRegion().front();
937 
938  // Replace the null placeholders with newly constructed values.
939  newBlockTransferArgs[0] = newBlock.getArgument(0); // iv
940  for (unsigned idx = 0, collapsedIdx = 0, e = newResultValues.size();
941  idx != e; ++idx) {
942  Value &blockTransferArg = newBlockTransferArgs[1 + idx];
943  Value &newResultVal = newResultValues[idx];
944  assert((blockTransferArg && newResultVal) ||
945  (!blockTransferArg && !newResultVal));
946  if (!blockTransferArg) {
947  blockTransferArg = newForOp.getRegionIterArgs()[collapsedIdx];
948  newResultVal = newForOp.getResult(collapsedIdx++);
949  }
950  }
951 
952  Block &oldBlock = forOp.getRegion().front();
953  assert(oldBlock.getNumArguments() == newBlockTransferArgs.size() &&
954  "unexpected argument size mismatch");
955 
956  // No results case: the scf::ForOp builder already created a zero
957  // result terminator. Merge before this terminator and just get rid of the
958  // original terminator that has been merged in.
959  if (newIterArgs.empty()) {
960  auto newYieldOp = cast<scf::YieldOp>(newBlock.getTerminator());
961  rewriter.inlineBlockBefore(&oldBlock, newYieldOp, newBlockTransferArgs);
962  rewriter.eraseOp(newBlock.getTerminator()->getPrevNode());
963  rewriter.replaceOp(forOp, newResultValues);
964  return success();
965  }
966 
967  // No terminator case: merge and rewrite the merged terminator.
968  auto cloneFilteredTerminator = [&](scf::YieldOp mergedTerminator) {
969  OpBuilder::InsertionGuard g(rewriter);
970  rewriter.setInsertionPoint(mergedTerminator);
971  SmallVector<Value, 4> filteredOperands;
972  filteredOperands.reserve(newResultValues.size());
973  for (unsigned idx = 0, e = keepMask.size(); idx < e; ++idx)
974  if (keepMask[idx])
975  filteredOperands.push_back(mergedTerminator.getOperand(idx));
976  scf::YieldOp::create(rewriter, mergedTerminator.getLoc(),
977  filteredOperands);
978  };
979 
980  rewriter.mergeBlocks(&oldBlock, &newBlock, newBlockTransferArgs);
981  auto mergedYieldOp = cast<scf::YieldOp>(newBlock.getTerminator());
982  cloneFilteredTerminator(mergedYieldOp);
983  rewriter.eraseOp(mergedYieldOp);
984  rewriter.replaceOp(forOp, newResultValues);
985  return success();
986  }
987 };

990 /// Returns std::nullopt when the difference between two AffineValueMap is
991 /// dynamic.
992 static std::optional<int64_t> computeConstDiff(Value l, Value u) {
993  IntegerAttr clb, cub;
994  if (matchPattern(l, m_Constant(&clb)) && matchPattern(u, m_Constant(&cub))) {
995  llvm::APInt lbValue = clb.getValue();

1003  u, m_Op<arith::AddIOp>(matchers::m_Val(l), m_ConstantInt(&diff))) ||
1004  matchPattern(

1010 /// Rewriting pattern that erases loops that are known not to iterate, replaces
1011 /// single-iteration loops with their bodies, and removes empty loops that
1012 /// iterate at least once and only return values defined outside of the loop.
1013 struct SimplifyTrivialLoops : public OpRewritePattern<ForOp> {
1014  using OpRewritePattern<ForOp>::OpRewritePattern;
1015 
1016  LogicalResult matchAndRewrite(ForOp op,
1017  PatternRewriter &rewriter) const override {
1018  // If the upper bound is the same as the lower bound, the loop does not
1019  // iterate, just remove it.
1020  if (op.getLowerBound() == op.getUpperBound()) {
1021  rewriter.replaceOp(op, op.getInitArgs());
1022  return success();
1023  }
1024 
1025  std::optional<int64_t> diff =
1026  computeConstDiff(op.getLowerBound(), op.getUpperBound());
1027  if (!diff)
1028  return failure();
1029 
1030  // If the loop is known to have 0 iterations, remove it.
1031  if (*diff <= 0) {
1032  rewriter.replaceOp(op, op.getInitArgs());
1033  return success();
1034  }
1035 
1036  std::optional<llvm::APInt> maybeStepValue = op.getConstantStep();
1037  if (!maybeStepValue)
1038  return failure();
1039 
1040  // If the loop is known to have 1 iteration, inline its body and remove the
1041  // loop.
1042  llvm::APInt stepValue = *maybeStepValue;
1043  if (stepValue.sge(*diff)) {
1044  SmallVector<Value, 4> blockArgs;
1045  blockArgs.reserve(op.getInitArgs().size() + 1);
1046  blockArgs.push_back(op.getLowerBound());
1047  llvm::append_range(blockArgs, op.getInitArgs());
1048  replaceOpWithRegion(rewriter, op, op.getRegion(), blockArgs);
1049  return success();
1050  }
1051 
1052  // Now we are left with loops that have more than 1 iterations.
1053  Block &block = op.getRegion().front();
1054  if (!llvm::hasSingleElement(block))
1055  return failure();
1056  // If the loop is empty, iterates at least once, and only returns values
1057  // defined outside of the loop, remove it and replace it with yield values.
1058  if (llvm::any_of(op.getYieldedValues(),
1059  [&](Value v) { return !op.isDefinedOutsideOfLoop(v); }))
1060  return failure();
1061  rewriter.replaceOp(op, op.getYieldedValues());
1062  return success();
1063  }
1064 };

1068 ///

1070 /// %0 = tensor.cast %t0 : tensor<32x1024xf32> to tensor<?x?xf32>
1071 /// %1 = scf.for %i = %c0 to %c1024 step %c32 iter_args(%iter_t0 = %0)
1072 /// -> (tensor<?x?xf32>) {
1073 /// %2 = call @do(%iter_t0) : (tensor<?x?xf32>) -> tensor<?x?xf32>
1074 /// scf.yield %2 : tensor<?x?xf32>
1075 /// }
1076 /// use_of(%1)
1077 /// ```
1078 ///
1079 /// folds into:
1080 ///
1081 /// ```
1082 /// %0 = scf.for %arg2 = %c0 to %c1024 step %c32 iter_args(%arg3 = %arg0)
1083 /// -> (tensor<32x1024xf32>) {
1084 /// %2 = tensor.cast %arg3 : tensor<32x1024xf32> to tensor<?x?xf32>
1085 /// %3 = call @do(%2) : (tensor<?x?xf32>) -> tensor<?x?xf32>
1086 /// %4 = tensor.cast %3 : tensor<?x?xf32> to tensor<32x1024xf32>
1087 /// scf.yield %4 : tensor<32x1024xf32>
1088 /// }
1089 /// %1 = tensor.cast %0 : tensor<32x1024xf32> to tensor<?x?xf32>
1090 /// use_of(%1)
1091 /// ```

1098  OpOperand &iterOpOperand = std::get<0>(it);
1099  auto incomingCast = iterOpOperand.get().getDefiningOp<tensor::CastOp>();

1101  incomingCast.getSource().getType() == incomingCast.getType())
1102  continue;

1104  // the source type.

1107  incomingCast.getSource().getType()))

1117  return tensor::CastOp::create(b, loc, type, source);
1118  }));
1119  return success();
1120  }
1121  return failure();
1122  }

1177  if (body->getArgument(i + numLoops).getType() != getOutputs()[i].getType())

1196  getDynamicUpperBound())))
1197  return failure();

1240 ParseResult ForallOp::parse(OpAsmParser &parser, OperationState &result) {
1241  OpBuilder b(parser.getContext());
1242  auto indexType = b.getIndexType();
1243 
1244  // Parse an opening `(` followed by thread index variables followed by `)`
1245  // TODO: when we can refer to such "induction variable"-like handles from the
1246  // declarative assembly format, we can implement the parser as a custom hook.
1247  SmallVector<OpAsmParser::Argument, 4> ivs;
1248  if (parser.parseArgumentList(ivs, OpAsmParser::Delimiter::Paren))
1249  return failure();
1250 
1251  DenseI64ArrayAttr staticLbs, staticUbs, staticSteps;
1252  SmallVector<OpAsmParser::UnresolvedOperand> dynamicLbs, dynamicUbs,
1253  dynamicSteps;
1254  if (succeeded(parser.parseOptionalKeyword("in"))) {
1255  // Parse upper bounds.
1256  if (parseDynamicIndexList(parser, dynamicUbs, staticUbs,
1257  /*valueTypes=*/nullptr,
1258  OpAsmParser::Delimiter::Paren) ||
1259  parser.resolveOperands(dynamicUbs, indexType, result.operands))
1260  return failure();
1261 
1262  unsigned numLoops = ivs.size();
1263  staticLbs = b.getDenseI64ArrayAttr(SmallVector<int64_t>(numLoops, 0));
1264  staticSteps = b.getDenseI64ArrayAttr(SmallVector<int64_t>(numLoops, 1));
1265  } else {
1266  // Parse lower bounds.
1267  if (parser.parseEqual() ||
1268  parseDynamicIndexList(parser, dynamicLbs, staticLbs,
1269  /*valueTypes=*/nullptr,
1270  OpAsmParser::Delimiter::Paren) ||
1271 
1272  parser.resolveOperands(dynamicLbs, indexType, result.operands))
1273  return failure();
1274 
1275  // Parse upper bounds.
1276  if (parser.parseKeyword("to") ||
1277  parseDynamicIndexList(parser, dynamicUbs, staticUbs,
1278  /*valueTypes=*/nullptr,
1279  OpAsmParser::Delimiter::Paren) ||
1280  parser.resolveOperands(dynamicUbs, indexType, result.operands))
1281  return failure();
1282 
1283  // Parse step values.
1284  if (parser.parseKeyword("step") ||
1285  parseDynamicIndexList(parser, dynamicSteps, staticSteps,
1286  /*valueTypes=*/nullptr,
1287  OpAsmParser::Delimiter::Paren) ||
1288  parser.resolveOperands(dynamicSteps, indexType, result.operands))
1289  return failure();
1290  }
1291 
1292  // Parse out operands and results.
1293  SmallVector<OpAsmParser::Argument, 4> regionOutArgs;
1294  SmallVector<OpAsmParser::UnresolvedOperand, 4> outOperands;
1295  SMLoc outOperandsLoc = parser.getCurrentLocation();
1296  if (succeeded(parser.parseOptionalKeyword("shared_outs"))) {
1297  if (outOperands.size() != result.types.size())
1298  return parser.emitError(outOperandsLoc,
1299  "mismatch between out operands and types");
1300  if (parser.parseAssignmentList(regionOutArgs, outOperands) ||
1301  parser.parseOptionalArrowTypeList(result.types) ||
1302  parser.resolveOperands(outOperands, result.types, outOperandsLoc,
1303  result.operands))
1304  return failure();
1305  }
1306 
1307  // Parse region.
1308  SmallVector<OpAsmParser::Argument, 4> regionArgs;
1309  std::unique_ptr<Region> region = std::make_unique<Region>();
1310  for (auto &iv : ivs) {
1311  iv.type = b.getIndexType();
1312  regionArgs.push_back(iv);
1313  }
1314  for (const auto &it : llvm::enumerate(regionOutArgs)) {
1315  auto &out = it.value();
1316  out.type = result.types[it.index()];
1317  regionArgs.push_back(out);
1318  }
1319  if (parser.parseRegion(*region, regionArgs))
1320  return failure();
1321 
1322  // Ensure terminator and move region.
1323  ForallOp::ensureTerminator(*region, b, result.location);
1324  result.addRegion(std::move(region));
1325 
1326  // Parse the optional attribute list.
1327  if (parser.parseOptionalAttrDict(result.attributes))
1328  return failure();
1329 
1330  result.addAttribute("staticLowerBound", staticLbs);
1331  result.addAttribute("staticUpperBound", staticUbs);
1332  result.addAttribute("staticStep", staticSteps);
1333  result.addAttribute("operandSegmentSizes",
1334  parser.getBuilder().getDenseI32ArrayAttr(
1335  {static_cast<int32_t>(dynamicLbs.size()),
1336  static_cast<int32_t>(dynamicUbs.size()),
1337  static_cast<int32_t>(dynamicSteps.size()),
1338  static_cast<int32_t>(outOperands.size())}));
1339  return success();

1564 /// The following canonicalization pattern folds the iter arguments of
1565 /// scf.forall op if :-
1566 /// 1. The corresponding result has zero uses.
1567 /// 2. The iter argument is NOT being modified within the loop body.
1568 /// uses.
1569 ///
1570 /// Example of first case :-
1571 /// INPUT:
1572 /// %res:3 = scf.forall ... shared_outs(%arg0 = %a, %arg1 = %b, %arg2 = %c)
1573 /// {
1574 /// ...
1575 /// <SOME USE OF %arg0>
1576 /// <SOME USE OF %arg1>
1577 /// <SOME USE OF %arg2>
1578 /// ...
1579 /// scf.forall.in_parallel {
1580 /// <STORE OP WITH DESTINATION %arg1>
1581 /// <STORE OP WITH DESTINATION %arg0>
1582 /// <STORE OP WITH DESTINATION %arg2>
1583 /// }
1584 /// }
1585 /// return %res#1
1586 ///
1587 /// OUTPUT:
1588 /// %res:3 = scf.forall ... shared_outs(%new_arg0 = %b)
1589 /// {
1590 /// ...
1591 /// <SOME USE OF %a>
1592 /// <SOME USE OF %new_arg0>
1593 /// <SOME USE OF %c>
1594 /// ...
1595 /// scf.forall.in_parallel {
1596 /// <STORE OP WITH DESTINATION %new_arg0>
1597 /// }
1598 /// }
1599 /// return %res
1600 ///
1601 /// NOTE: 1. All uses of the folded shared_outs (iter argument) within the
1602 /// scf.forall is replaced by their corresponding operands.
1603 /// 2. Even if there are <STORE OP WITH DESTINATION *> ops within the body
1604 /// of the scf.forall besides within scf.forall.in_parallel terminator,
1605 /// this canonicalization remains valid. For more details, please refer
1606 /// to :
1607 /// https://github.com/llvm/llvm-project/pull/90189#discussion_r1589011124
1608 /// 3. TODO(avarma): Generalize it for other store ops. Currently it
1609 /// handles tensor.parallel_insert_slice ops only.
1610 ///
1611 /// Example of second case :-
1612 /// INPUT:
1613 /// %res:2 = scf.forall ... shared_outs(%arg0 = %a, %arg1 = %b)
1614 /// {
1615 /// ...
1616 /// <SOME USE OF %arg0>
1617 /// <SOME USE OF %arg1>
1618 /// ...
1619 /// scf.forall.in_parallel {
1620 /// <STORE OP WITH DESTINATION %arg1>
1621 /// }
1622 /// }
1623 /// return %res#0, %res#1
1624 ///
1625 /// OUTPUT:
1626 /// %res = scf.forall ... shared_outs(%new_arg0 = %b)
1627 /// {
1628 /// ...
1629 /// <SOME USE OF %a>
1630 /// <SOME USE OF %new_arg0>
1631 /// ...
1632 /// scf.forall.in_parallel {
1633 /// <STORE OP WITH DESTINATION %new_arg0>
1634 /// }
1635 /// }
1636 /// return %a, %res
1637 struct ForallOpIterArgsFolder : public OpRewritePattern<ForallOp> {
1638  using OpRewritePattern<ForallOp>::OpRewritePattern;
1639 
1640  LogicalResult matchAndRewrite(ForallOp forallOp,
1641  PatternRewriter &rewriter) const final {
1642  // Step 1: For a given i-th result of scf.forall, check the following :-
1643  // a. If it has any use.
1644  // b. If the corresponding iter argument is being modified within
1645  // the loop, i.e. has at least one store op with the iter arg as
1646  // its destination operand. For this we use
1647  // ForallOp::getCombiningOps(iter_arg).
1648  //
1649  // Based on the check we maintain the following :-
1650  // a. `resultToDelete` - i-th result of scf.forall that'll be
1651  // deleted.
1652  // b. `resultToReplace` - i-th result of the old scf.forall
1653  // whose uses will be replaced by the new scf.forall.
1654  // c. `newOuts` - the shared_outs' operand of the new scf.forall
1655  // corresponding to the i-th result with at least one use.
1656  SetVector<OpResult> resultToDelete;
1657  SmallVector<Value> resultToReplace;
1658  SmallVector<Value> newOuts;
1659  for (OpResult result : forallOp.getResults()) {
1660  OpOperand *opOperand = forallOp.getTiedOpOperand(result);
1661  BlockArgument blockArg = forallOp.getTiedBlockArgument(opOperand);
1662  if (result.use_empty() || forallOp.getCombiningOps(blockArg).empty()) {
1663  resultToDelete.insert(result);
1664  } else {
1665  resultToReplace.push_back(result);
1666  newOuts.push_back(opOperand->get());
1667  }
1668  }
1669 
1670  // Return early if all results of scf.forall have at least one use and being
1671  // modified within the loop.
1672  if (resultToDelete.empty())
1673  return failure();
1674 
1675  // Step 2: For the the i-th result, do the following :-
1676  // a. Fetch the corresponding BlockArgument.
1677  // b. Look for store ops (currently tensor.parallel_insert_slice)
1678  // with the BlockArgument as its destination operand.
1679  // c. Remove the operations fetched in b.
1680  for (OpResult result : resultToDelete) {
1681  OpOperand *opOperand = forallOp.getTiedOpOperand(result);
1682  BlockArgument blockArg = forallOp.getTiedBlockArgument(opOperand);
1683  SmallVector<Operation *> combiningOps =
1684  forallOp.getCombiningOps(blockArg);
1685  for (Operation *combiningOp : combiningOps)
1686  rewriter.eraseOp(combiningOp);
1687  }
1688 
1689  // Step 3. Create a new scf.forall op with the new shared_outs' operands
1690  // fetched earlier
1691  auto newForallOp = scf::ForallOp::create(
1692  rewriter, forallOp.getLoc(), forallOp.getMixedLowerBound(),
1693  forallOp.getMixedUpperBound(), forallOp.getMixedStep(), newOuts,
1694  forallOp.getMapping(),
1695  /*bodyBuilderFn =*/[](OpBuilder &, Location, ValueRange) {});
1696 
1697  // Step 4. Merge the block of the old scf.forall into the newly created
1698  // scf.forall using the new set of arguments.
1699  Block *loopBody = forallOp.getBody();
1700  Block *newLoopBody = newForallOp.getBody();
1701  ArrayRef<BlockArgument> newBbArgs = newLoopBody->getArguments();
1702  // Form initial new bbArg list with just the control operands of the new
1703  // scf.forall op.
1704  SmallVector<Value> newBlockArgs =
1705  llvm::map_to_vector(newBbArgs.take_front(forallOp.getRank()),
1706  [](BlockArgument b) -> Value { return b; });
1707  Block::BlockArgListType newSharedOutsArgs = newForallOp.getRegionOutArgs();
1708  unsigned index = 0;
1709  // Take the new corresponding bbArg if the old bbArg was used as a
1710  // destination in the in_parallel op. For all other bbArgs, use the
1711  // corresponding init_arg from the old scf.forall op.
1712  for (OpResult result : forallOp.getResults()) {
1713  if (resultToDelete.count(result)) {
1714  newBlockArgs.push_back(forallOp.getTiedOpOperand(result)->get());
1715  } else {

1771  newMixedSteps.push_back(step);

1776  promote(rewriter, op);
1777  return success();

1789  newMixedUpperBounds, newMixedSteps,

1793  // "operandSegmentSizes" and static loop bound attributes which capture

1891  bbArgs.front().getParentBlock(), ivsBlockArgs);

2691 /// Convert nested `if`s into `arith.andi` + single `if`.
2692 ///
2693 /// scf.if %arg0 {
2694 /// scf.if %arg1 {
2695 /// ...
2696 /// scf.yield
2697 /// }
2698 /// scf.yield
2699 /// }
2700 /// becomes
2701 ///
2702 /// %0 = arith.andi %arg0, %arg1
2703 /// scf.if %0 {
2704 /// ...
2705 /// scf.yield
2706 /// }
2707 struct CombineNestedIfs : public OpRewritePattern<IfOp> {
2708  using OpRewritePattern<IfOp>::OpRewritePattern;
2709 
2710  LogicalResult matchAndRewrite(IfOp op,
2711  PatternRewriter &rewriter) const override {
2712  auto nestedOps = op.thenBlock()->without_terminator();
2713  // Nested `if` must be the only op in block.
2714  if (!llvm::hasSingleElement(nestedOps))
2715  return failure();
2716 
2717  // If there is an else block, it can only yield
2718  if (op.elseBlock() && !llvm::hasSingleElement(*op.elseBlock()))
2719  return failure();
2720 
2721  auto nestedIf = dyn_cast<IfOp>(*nestedOps.begin());
2722  if (!nestedIf)
2723  return failure();
2724 
2725  if (nestedIf.elseBlock() && !llvm::hasSingleElement(*nestedIf.elseBlock()))
2726  return failure();
2727 
2728  SmallVector<Value> thenYield(op.thenYield().getOperands());
2729  SmallVector<Value> elseYield;
2730  if (op.elseBlock())
2731  llvm::append_range(elseYield, op.elseYield().getOperands());
2732 
2733  // A list of indices for which we should upgrade the value yielded
2734  // in the else to a select.
2735  SmallVector<unsigned> elseYieldsToUpgradeToSelect;
2736 
2737  // If the outer scf.if yields a value produced by the inner scf.if,
2738  // only permit combining if the value yielded when the condition
2739  // is false in the outer scf.if is the same value yielded when the
2740  // inner scf.if condition is false.
2741  // Note that the array access to elseYield will not go out of bounds
2742  // since it must have the same length as thenYield, since they both
2743  // come from the same scf.if.
2744  for (const auto &tup : llvm::enumerate(thenYield)) {
2745  if (tup.value().getDefiningOp() == nestedIf) {
2746  auto nestedIdx = llvm::cast<OpResult>(tup.value()).getResultNumber();
2747  if (nestedIf.elseYield().getOperand(nestedIdx) !=
2748  elseYield[tup.index()]) {
2749  return failure();
2750  }
2751  // If the correctness test passes, we will yield
2752  // corresponding value from the inner scf.if
2753  thenYield[tup.index()] = nestedIf.thenYield().getOperand(nestedIdx);
2754  continue;
2755  }
2756 
2757  // Otherwise, we need to ensure the else block of the combined
2758  // condition still returns the same value when the outer condition is
2759  // true and the inner condition is false. This can be accomplished if
2760  // the then value is defined outside the outer scf.if and we replace the
2761  // value with a select that considers just the outer condition. Since
2762  // the else region contains just the yield, its yielded value is
2763  // defined outside the scf.if, by definition.
2764 
2765  // If the then value is defined within the scf.if, bail.
2766  if (tup.value().getParentRegion() == &op.getThenRegion()) {
2767  return failure();
2768  }
2769  elseYieldsToUpgradeToSelect.push_back(tup.index());
2770  }
2771 
2772  Location loc = op.getLoc();
2773  Value newCondition = arith::AndIOp::create(rewriter, loc, op.getCondition(),
2774  nestedIf.getCondition());
2775  auto newIf = IfOp::create(rewriter, loc, op.getResultTypes(), newCondition);
2776  Block *newIfBlock = rewriter.createBlock(&newIf.getThenRegion());
2777 
2778  SmallVector<Value> results;
2779  llvm::append_range(results, newIf.getResults());
2780  rewriter.setInsertionPoint(newIf);
2781 
2782  for (auto idx : elseYieldsToUpgradeToSelect)
2783  results[idx] =
2784  arith::SelectOp::create(rewriter, op.getLoc(), op.getCondition(),
2785  thenYield[idx], elseYield[idx]);
2786 
2787  rewriter.mergeBlocks(nestedIf.thenBlock(), newIfBlock);
2788  rewriter.setInsertionPointToEnd(newIf.thenBlock());
2789  rewriter.replaceOpWithNewOp<YieldOp>(newIf.thenYield(), thenYield);
2790  if (!elseYield.empty()) {
2791  rewriter.createBlock(&newIf.getElseRegion());
2792  rewriter.setInsertionPointToEnd(newIf.elseBlock());
2793  YieldOp::create(rewriter, loc, elseYield);
2794  }
2795  rewriter.replaceOp(op, results);
2796  return success();
2797  }
