102 // Helper to resolve the alignment for vector load/store, gather and scatter
103 // ops. If useVectorAlignment is true, get the preferred alignment for the
104 // vector type in the operation. This option is used for hardware backends with
105 // vectorization. Otherwise, use the preferred alignment of the element type of
106 // the memref. Note that if you choose to use vector alignment, the shape of the
107 // vector type must be resolved before the ConvertVectorToLLVM pass is run.
108 LogicalResult getVectorToLLVMAlignment(const LLVMTypeConverter &typeConverter,
109  VectorType vectorType,
110  MemRefType memrefType, unsigned &align,
111  bool useVectorAlignment) {
112  if (useVectorAlignment) {
113  if (failed(getVectorAlignment(typeConverter, vectorType, align))) {
114  return failure();
115  }
116  } else {
117  if (failed(getMemRefAlignment(typeConverter, memrefType, align))) {
118  return failure();
119  }
120  }
121  return success();
122 }

226 /// Conversion pattern for a vector.load, vector.store, vector.maskedload, and
227 /// vector.maskedstore.
228 template <class LoadOrStoreOp>
229 class VectorLoadStoreConversion : public ConvertOpToLLVMPattern<LoadOrStoreOp> {
230 public:
231  explicit VectorLoadStoreConversion(const LLVMTypeConverter &typeConv,
232  bool useVectorAlign)
233  : ConvertOpToLLVMPattern<LoadOrStoreOp>(typeConv),
234  useVectorAlignment(useVectorAlign) {}
235  using ConvertOpToLLVMPattern<LoadOrStoreOp>::ConvertOpToLLVMPattern;
236 
237  LogicalResult
238  matchAndRewrite(LoadOrStoreOp loadOrStoreOp,
239  typename LoadOrStoreOp::Adaptor adaptor,
240  ConversionPatternRewriter &rewriter) const override {
241  // Only 1-D vectors can be lowered to LLVM.
242  VectorType vectorTy = loadOrStoreOp.getVectorType();
243  if (vectorTy.getRank() > 1)
244  return failure();
245 
246  auto loc = loadOrStoreOp->getLoc();
247  MemRefType memRefTy = loadOrStoreOp.getMemRefType();
248 
249  // Resolve alignment.
250  unsigned align;
251  if (failed(getVectorToLLVMAlignment(*this->getTypeConverter(), vectorTy,
252  memRefTy, align, useVectorAlignment)))
253  return rewriter.notifyMatchFailure(loadOrStoreOp,
254  "could not resolve alignment");
255 
256  // Resolve address.
257  auto vtype = cast<VectorType>(
258  this->typeConverter->convertType(loadOrStoreOp.getVectorType()));
259  Value dataPtr = this->getStridedElementPtr(
260  rewriter, loc, memRefTy, adaptor.getBase(), adaptor.getIndices());
261  replaceLoadOrStoreOp(loadOrStoreOp, adaptor, vtype, dataPtr, align,
262  rewriter);
263  return success();
264  }
265 
266 private:
267  // If true, use the preferred alignment of the vector type.
268  // If false, use the preferred alignment of the element type
269  // of the memref. This flag is intended for use with hardware
270  // backends that require alignment of vector operations.
271  const bool useVectorAlignment;
272 };

275 class VectorGatherOpConversion
276  : public ConvertOpToLLVMPattern<vector::GatherOp> {
277 public:
278  explicit VectorGatherOpConversion(const LLVMTypeConverter &typeConv,
279  bool useVectorAlign)
280  : ConvertOpToLLVMPattern<vector::GatherOp>(typeConv),
281  useVectorAlignment(useVectorAlign) {}
282  using ConvertOpToLLVMPattern<vector::GatherOp>::ConvertOpToLLVMPattern;
283 
284  LogicalResult
285  matchAndRewrite(vector::GatherOp gather, OpAdaptor adaptor,
286  ConversionPatternRewriter &rewriter) const override {
287  Location loc = gather->getLoc();
288  MemRefType memRefType = dyn_cast<MemRefType>(gather.getBaseType());
289  assert(memRefType && "The base should be bufferized");
290 
291  if (failed(isMemRefTypeSupported(memRefType, *this->getTypeConverter())))
292  return rewriter.notifyMatchFailure(gather, "memref type not supported");
293 
294  VectorType vType = gather.getVectorType();
295  if (vType.getRank() > 1) {
296  return rewriter.notifyMatchFailure(
297  gather, "only 1-D vectors can be lowered to LLVM");
298  }
299 
300  // Resolve alignment.
301  unsigned align;
302  if (failed(getVectorToLLVMAlignment(*this->getTypeConverter(), vType,
303  memRefType, align, useVectorAlignment)))
304  return rewriter.notifyMatchFailure(gather, "could not resolve alignment");
305 
306  // Resolve address.
307  Value ptr = getStridedElementPtr(rewriter, loc, memRefType,
308  adaptor.getBase(), adaptor.getIndices());
309  Value base = adaptor.getBase();
310  Value ptrs =
311  getIndexedPtrs(rewriter, loc, *this->getTypeConverter(), memRefType,
312  base, ptr, adaptor.getIndexVec(), vType);
313 
314  // Replace with the gather intrinsic.
315  rewriter.replaceOpWithNewOp<LLVM::masked_gather>(
316  gather, typeConverter->convertType(vType), ptrs, adaptor.getMask(),
317  adaptor.getPassThru(), rewriter.getI32IntegerAttr(align));
318  return success();
319  }

322  // If true, use the preferred alignment of the vector type.
323  // If false, use the preferred alignment of the element type
324  // of the memref. This flag is intended for use with hardware
325  // backends that require alignment of vector operations.
326  const bool useVectorAlignment;

749 /// Conversion pattern for all vector reductions.
750 class VectorReductionOpConversion
751  : public ConvertOpToLLVMPattern<vector::ReductionOp> {
752 public:
753  explicit VectorReductionOpConversion(const LLVMTypeConverter &typeConv,
754  bool reassociateFPRed)
755  : ConvertOpToLLVMPattern<vector::ReductionOp>(typeConv),
756  reassociateFPReductions(reassociateFPRed) {}
757 
758  LogicalResult
759  matchAndRewrite(vector::ReductionOp reductionOp, OpAdaptor adaptor,
760  ConversionPatternRewriter &rewriter) const override {
761  auto kind = reductionOp.getKind();
762  Type eltType = reductionOp.getDest().getType();
763  Type llvmType = typeConverter->convertType(eltType);
764  Value operand = adaptor.getVector();
765  Value acc = adaptor.getAcc();
766  Location loc = reductionOp.getLoc();
767 
768  if (eltType.isIntOrIndex()) {
769  // Integer reductions: add/mul/min/max/and/or/xor.
770  Value result;
771  switch (kind) {
772  case vector::CombiningKind::ADD:
773  result =
774  createIntegerReductionArithmeticOpLowering<LLVM::vector_reduce_add,
775  LLVM::AddOp>(
776  rewriter, loc, llvmType, operand, acc);
777  break;
778  case vector::CombiningKind::MUL:
779  result =
780  createIntegerReductionArithmeticOpLowering<LLVM::vector_reduce_mul,
781  LLVM::MulOp>(
782  rewriter, loc, llvmType, operand, acc);
783  break;
784  case vector::CombiningKind::MINUI:
785  result = createIntegerReductionComparisonOpLowering<
786  LLVM::vector_reduce_umin>(rewriter, loc, llvmType, operand, acc,
787  LLVM::ICmpPredicate::ule);
788  break;
789  case vector::CombiningKind::MINSI:
790  result = createIntegerReductionComparisonOpLowering<
791  LLVM::vector_reduce_smin>(rewriter, loc, llvmType, operand, acc,
792  LLVM::ICmpPredicate::sle);
793  break;
794  case vector::CombiningKind::MAXUI:
795  result = createIntegerReductionComparisonOpLowering<
796  LLVM::vector_reduce_umax>(rewriter, loc, llvmType, operand, acc,
797  LLVM::ICmpPredicate::uge);
798  break;
799  case vector::CombiningKind::MAXSI:
800  result = createIntegerReductionComparisonOpLowering<
801  LLVM::vector_reduce_smax>(rewriter, loc, llvmType, operand, acc,
802  LLVM::ICmpPredicate::sge);
803  break;
804  case vector::CombiningKind::AND:
805  result =
806  createIntegerReductionArithmeticOpLowering<LLVM::vector_reduce_and,
807  LLVM::AndOp>(
808  rewriter, loc, llvmType, operand, acc);
809  break;
810  case vector::CombiningKind::OR:
811  result =
812  createIntegerReductionArithmeticOpLowering<LLVM::vector_reduce_or,
813  LLVM::OrOp>(
814  rewriter, loc, llvmType, operand, acc);
815  break;
816  case vector::CombiningKind::XOR:
817  result =
818  createIntegerReductionArithmeticOpLowering<LLVM::vector_reduce_xor,
819  LLVM::XOrOp>(
820  rewriter, loc, llvmType, operand, acc);
821  break;
822  default:
823  return failure();
824  }
825  rewriter.replaceOp(reductionOp, result);
826 
827  return success();
828  }
829 
830  if (!isa<FloatType>(eltType))
831  return failure();
832 
833  arith::FastMathFlagsAttr fMFAttr = reductionOp.getFastMathFlagsAttr();
834  LLVM::FastmathFlagsAttr fmf = LLVM::FastmathFlagsAttr::get(
835  reductionOp.getContext(),
836  convertArithFastMathFlagsToLLVM(fMFAttr.getValue()));
837  fmf = LLVM::FastmathFlagsAttr::get(
838  reductionOp.getContext(),
839  fmf.getValue() | (reassociateFPReductions ? LLVM::FastmathFlags::reassoc
840  : LLVM::FastmathFlags::none));
841 
842  // Floating-point reductions: add/mul/min/max
843  Value result;
844  if (kind == vector::CombiningKind::ADD) {
845  result = lowerReductionWithStartValue<LLVM::vector_reduce_fadd,
846  ReductionNeutralZero>(
847  rewriter, loc, llvmType, operand, acc, fmf);
848  } else if (kind == vector::CombiningKind::MUL) {
849  result = lowerReductionWithStartValue<LLVM::vector_reduce_fmul,
850  ReductionNeutralFPOne>(
851  rewriter, loc, llvmType, operand, acc, fmf);
852  } else if (kind == vector::CombiningKind::MINIMUMF) {
853  result =
854  createFPReductionComparisonOpLowering<LLVM::vector_reduce_fminimum>(
855  rewriter, loc, llvmType, operand, acc, fmf);
856  } else if (kind == vector::CombiningKind::MAXIMUMF) {
857  result =
858  createFPReductionComparisonOpLowering<LLVM::vector_reduce_fmaximum>(
859  rewriter, loc, llvmType, operand, acc, fmf);
860  } else if (kind == vector::CombiningKind::MINNUMF) {
861  result = createFPReductionComparisonOpLowering<LLVM::vector_reduce_fmin>(
862  rewriter, loc, llvmType, operand, acc, fmf);
863  } else if (kind == vector::CombiningKind::MAXNUMF) {
864  result = createFPReductionComparisonOpLowering<LLVM::vector_reduce_fmax>(
865  rewriter, loc, llvmType, operand, acc, fmf);
866  } else {
867  return failure();
868  }
869 
870  rewriter.replaceOp(reductionOp, result);
871  return success();
872  }

874 private:
875  const bool reassociateFPReductions;
876 };

1013 
1014  LogicalResult
1015  matchAndRewrite(vector::ShuffleOp shuffleOp, OpAdaptor adaptor,
1016  ConversionPatternRewriter &rewriter) const override {
1017  auto loc = shuffleOp->getLoc();
1018  auto v1Type = shuffleOp.getV1VectorType();
1019  auto v2Type = shuffleOp.getV2VectorType();
1020  auto vectorType = shuffleOp.getResultVectorType();
1021  Type llvmType = typeConverter->convertType(vectorType);
1022  ArrayRef<int64_t> mask = shuffleOp.getMask();
1023 
1024  // Bail if result type cannot be lowered.
1025  if (!llvmType)
1026  return failure();
1027 
1028  // Get rank and dimension sizes.
1029  int64_t rank = vectorType.getRank();
1030 #ifndef NDEBUG
1031  bool wellFormed0DCase =
1032  v1Type.getRank() == 0 && v2Type.getRank() == 0 && rank == 1;
1033  bool wellFormedNDCase =
1034  v1Type.getRank() == rank && v2Type.getRank() == rank;
1035  assert((wellFormed0DCase || wellFormedNDCase) && "op is not well-formed");
1036 #endif
1037 
1038  // For rank 0 and 1, where both operands have *exactly* the same vector
1039  // type, there is direct shuffle support in LLVM. Use it!
1040  if (rank <= 1 && v1Type == v2Type) {
1041  Value llvmShuffleOp = LLVM::ShuffleVectorOp::create(
1042  rewriter, loc, adaptor.getV1(), adaptor.getV2(),
1043  llvm::to_vector_of<int32_t>(mask));
1044  rewriter.replaceOp(shuffleOp, llvmShuffleOp);
1045  return success();
1046  }
1047 
1048  // For all other cases, insert the individual values individually.
1049  int64_t v1Dim = v1Type.getDimSize(0);
1050  Type eltType;
1051  if (auto arrayType = dyn_cast<LLVM::LLVMArrayType>(llvmType))
1052  eltType = arrayType.getElementType();
1053  else
1054  eltType = cast<VectorType>(llvmType).getElementType();
1055  Value insert = LLVM::PoisonOp::create(rewriter, loc, llvmType);
1056  int64_t insPos = 0;
1057  for (int64_t extPos : mask) {
1058  Value value = adaptor.getV1();
1059  if (extPos >= v1Dim) {
1060  extPos -= v1Dim;
1061  value = adaptor.getV2();
1062  }
1063  Value extract = extractOne(rewriter, *getTypeConverter(), loc, value,
1064  eltType, rank, extPos);
1065  insert = insertOne(rewriter, *getTypeConverter(), loc, insert, extract,
1066  llvmType, rank, insPos++);
1067  }
1068  rewriter.replaceOp(shuffleOp, insert);
1069  return success();
1070  }

1073 class VectorExtractOpConversion
1074  : public ConvertOpToLLVMPattern<vector::ExtractOp> {
1075 public:
1076  using ConvertOpToLLVMPattern<vector::ExtractOp>::ConvertOpToLLVMPattern;
1077 
1078  LogicalResult
1079  matchAndRewrite(vector::ExtractOp extractOp, OpAdaptor adaptor,
1080  ConversionPatternRewriter &rewriter) const override {
1081  auto loc = extractOp->getLoc();
1082  auto resultType = extractOp.getResult().getType();
1083  auto llvmResultType = typeConverter->convertType(resultType);
1084  // Bail if result type cannot be lowered.
1085  if (!llvmResultType)
1086  return failure();
1087 
1088  SmallVector<OpFoldResult> positionVec = getMixedValues(
1089  adaptor.getStaticPosition(), adaptor.getDynamicPosition(), rewriter);
1090 
1091  // The Vector -> LLVM lowering models N-D vectors as nested aggregates of
1092  // 1-d vectors. This nesting is modeled using arrays. We do this conversion
1093  // from a N-d vector extract to a nested aggregate vector extract in two
1094  // steps:
1095  // - Extract a member from the nested aggregate. The result can be
1096  // a lower rank nested aggregate or a vector (1-D). This is done using
1097  // `llvm.extractvalue`.
1098  // - Extract a scalar out of the vector if needed. This is done using
1099  // `llvm.extractelement`.
1100 
1101  // Determine if we need to extract a member out of the aggregate. We
1102  // always need to extract a member if the input rank >= 2.
1103  bool extractsAggregate = extractOp.getSourceVectorType().getRank() >= 2;
1104  // Determine if we need to extract a scalar as the result. We extract
1105  // a scalar if the extract is full rank, i.e., the number of indices is
1106  // equal to source vector rank.
1107  bool extractsScalar = static_cast<int64_t>(positionVec.size()) ==
1108  extractOp.getSourceVectorType().getRank();
1109 
1110  // Since the LLVM type converter converts 0-d vectors to 1-d vectors, we
1111  // need to add a position for this change.
1112  if (extractOp.getSourceVectorType().getRank() == 0) {
1113  Type idxType = typeConverter->convertType(rewriter.getIndexType());
1114  positionVec.push_back(rewriter.getZeroAttr(idxType));
1115  }
1116 
1117  Value extracted = adaptor.getVector();
1118  if (extractsAggregate) {
1119  ArrayRef<OpFoldResult> position(positionVec);
1120  if (extractsScalar) {
1121  // If we are extracting a scalar from the extracted member, we drop
1122  // the last index, which will be used to extract the scalar out of the
1123  // vector.
1124  position = position.drop_back();
1125  }
1126  // llvm.extractvalue does not support dynamic dimensions.
1127  if (!llvm::all_of(position, llvm::IsaPred<Attribute>)) {
1128  return failure();
1129  }
1130  extracted = LLVM::ExtractValueOp::create(rewriter, loc, extracted,
1131  getAsIntegers(position));
1132  }
1133 
1134  if (extractsScalar) {
1135  extracted = LLVM::ExtractElementOp::create(
1136  rewriter, loc, extracted,
1137  getAsLLVMValue(rewriter, loc, positionVec.back()));
1138  }
1139 
1140  rewriter.replaceOp(extractOp, extracted);
1141  return success();
1142  }

1177  : public ConvertOpToLLVMPattern<vector::InsertOp> {
1178 public:
1179  using ConvertOpToLLVMPattern<vector::InsertOp>::ConvertOpToLLVMPattern;
1180 
1181  LogicalResult
1182  matchAndRewrite(vector::InsertOp insertOp, OpAdaptor adaptor,
1183  ConversionPatternRewriter &rewriter) const override {
1184  auto loc = insertOp->getLoc();
1185  auto destVectorType = insertOp.getDestVectorType();
1186  auto llvmResultType = typeConverter->convertType(destVectorType);
1187  // Bail if result type cannot be lowered.
1188  if (!llvmResultType)
1189  return failure();
1190 
1191  SmallVector<OpFoldResult> positionVec = getMixedValues(
1192  adaptor.getStaticPosition(), adaptor.getDynamicPosition(), rewriter);
1193 
1194  // The logic in this pattern mirrors VectorExtractOpConversion. Refer to
1195  // its explanatory comment about how N-D vectors are converted as nested
1196  // aggregates (llvm.array's) of 1D vectors.
1197  //
1198  // The innermost dimension of the destination vector, when converted to a
1199  // nested aggregate form, will always be a 1D vector.
1200  //
1201  // * If the insertion is happening into the innermost dimension of the
1202  // destination vector:
1203  // - If the destination is a nested aggregate, extract a 1D vector out of
1204  // the aggregate. This can be done using llvm.extractvalue. The
1205  // destination is now guaranteed to be a 1D vector, to which we are
1206  // inserting.
1207  // - Do the insertion into the 1D destination vector, and make the result
1208  // the new source nested aggregate. This can be done using
1209  // llvm.insertelement.
1210  // * Insert the source nested aggregate into the destination nested
1211  // aggregate.
1212 
1213  // Determine if we need to extract/insert a 1D vector out of the aggregate.
1214  bool isNestedAggregate = isa<LLVM::LLVMArrayType>(llvmResultType);
1215  // Determine if we need to insert a scalar into the 1D vector.
1216  bool insertIntoInnermostDim =
1217  static_cast<int64_t>(positionVec.size()) == destVectorType.getRank();
1218 
1219  ArrayRef<OpFoldResult> positionOf1DVectorWithinAggregate(
1220  positionVec.begin(),
1221  insertIntoInnermostDim ? positionVec.size() - 1 : positionVec.size());
1222  OpFoldResult positionOfScalarWithin1DVector;
1223  if (destVectorType.getRank() == 0) {
1224  // Since the LLVM type converter converts 0D vectors to 1D vectors, we
1225  // need to create a 0 here as the position into the 1D vector.
1226  Type idxType = typeConverter->convertType(rewriter.getIndexType());
1227  positionOfScalarWithin1DVector = rewriter.getZeroAttr(idxType);
1228  } else if (insertIntoInnermostDim) {
1229  positionOfScalarWithin1DVector = positionVec.back();
1230  }
1231 
1232  // We are going to mutate this 1D vector until it is either the final
1233  // result (in the non-aggregate case) or the value that needs to be
1234  // inserted into the aggregate result.
1235  Value sourceAggregate = adaptor.getValueToStore();
1236  if (insertIntoInnermostDim) {
1237  // Scalar-into-1D-vector case, so we know we will have to create a
1238  // InsertElementOp. The question is into what destination.
1239  if (isNestedAggregate) {
1240  // Aggregate case: the destination for the InsertElementOp needs to be
1241  // extracted from the aggregate.
1242  if (!llvm::all_of(positionOf1DVectorWithinAggregate,
1243  llvm::IsaPred<Attribute>)) {
1244  // llvm.extractvalue does not support dynamic dimensions.
1245  return failure();
1246  }
1247  sourceAggregate = LLVM::ExtractValueOp::create(
1248  rewriter, loc, adaptor.getDest(),
1249  getAsIntegers(positionOf1DVectorWithinAggregate));
1250  } else {
1251  // No-aggregate case. The destination for the InsertElementOp is just
1252  // the insertOp's destination.
1253  sourceAggregate = adaptor.getDest();
1254  }
1255  // Insert the scalar into the 1D vector.
1256  sourceAggregate = LLVM::InsertElementOp::create(
1257  rewriter, loc, sourceAggregate.getType(), sourceAggregate,
1258  adaptor.getValueToStore(),
1259  getAsLLVMValue(rewriter, loc, positionOfScalarWithin1DVector));
1260  }
1261 
1262  Value result = sourceAggregate;
1263  if (isNestedAggregate) {
1264  result = LLVM::InsertValueOp::create(
1265  rewriter, loc, adaptor.getDest(), sourceAggregate,
1266  getAsIntegers(positionOf1DVectorWithinAggregate));
1267  }
1268 
1269  rewriter.replaceOp(insertOp, result);
1270  return success();
1271  }

1359 /// Returns the strides if the memory underlying `memRefType` has a contiguous
1360 /// static layout.
1361 static std::optional<SmallVector<int64_t, 4>>
1362 computeContiguousStrides(MemRefType memRefType) {
1363  int64_t offset;
1364  SmallVector<int64_t, 4> strides;
1365  if (failed(memRefType.getStridesAndOffset(strides, offset)))
1366  return std::nullopt;
1367  if (!strides.empty() && strides.back() != 1)
1368  return std::nullopt;
1369  // If no layout or identity layout, this is contiguous by definition.
1370  if (memRefType.getLayout().isIdentity())
1371  return strides;
1372 
1373  // Otherwise, we must determine contiguity form shapes. This can only ever
1374  // work in static cases because MemRefType is underspecified to represent
1375  // contiguous dynamic shapes in other ways than with just empty/identity
1376  // layout.
1377  auto sizes = memRefType.getShape();
1378  for (int index = 0, e = strides.size() - 1; index < e; ++index) {
1379  if (ShapedType::isDynamic(sizes[index + 1]) ||
1380  ShapedType::isDynamic(strides[index]) ||
1381  ShapedType::isDynamic(strides[index + 1]))
1382  return std::nullopt;
1383  if (strides[index] != strides[index + 1] * sizes[index + 1])
1384  return std::nullopt;
1385  }
1386  return strides;
1387 }

1995 /// %e = add %c, %d
1996 /// ```

1999 /// vector.transpose operations are inserted if the vector.contract op is not a
2000 /// row-major matrix multiply.
2001 ///
2002 /// Scalable vectors are not supported.
2003 FailureOr<Value> ContractionOpToMatmulOpLowering::matchAndRewriteMaskableOp(
2004  vector::ContractionOp op, MaskingOpInterface maskOp,
2005  PatternRewriter &rew) const {
2006  // TODO: Support vector.mask.
2007  if (maskOp)
2008  return failure();
2009 
2010  auto iteratorTypes = op.getIteratorTypes().getValue();
2011  if (!isParallelIterator(iteratorTypes[0]) ||
2012  !isParallelIterator(iteratorTypes[1]) ||
2013  !isReductionIterator(iteratorTypes[2]))
2014  return failure();
2015 
2016  Type opResType = op.getType();
2017  VectorType vecType = dyn_cast<VectorType>(opResType);
2018  if (vecType && vecType.isScalable()) {
2019  // Note - this is sufficient to reject all cases with scalable vectors.
2020  return failure();
2021  }
2022 
2023  Type elementType = op.getLhsType().getElementType();
2024  if (!elementType.isIntOrFloat())
2025  return failure();
2026 
2027  Type dstElementType = vecType ? vecType.getElementType() : opResType;
2028  if (elementType != dstElementType)
2029  return failure();

2032  // Bail out if the contraction cannot be put in this form.
2033  MLIRContext *ctx = op.getContext();
2034  Location loc = op.getLoc();
2035  AffineExpr m, n, k;
2036  bindDims(rew.getContext(), m, n, k);
2037  // LHS must be A(m, k) or A(k, m).
2038  Value lhs = op.getLhs();
2039  auto lhsMap = op.getIndexingMapsArray()[0];
2040  if (lhsMap == AffineMap::get(3, 0, {k, m}, ctx))
2041  lhs = vector::TransposeOp::create(rew, loc, lhs, ArrayRef<int64_t>{1, 0});
2042  else if (lhsMap != AffineMap::get(3, 0, {m, k}, ctx))
2043  return failure();
2044 
2045  // RHS must be B(k, n) or B(n, k).
2046  Value rhs = op.getRhs();
2047  auto rhsMap = op.getIndexingMapsArray()[1];
2048  if (rhsMap == AffineMap::get(3, 0, {n, k}, ctx))
2049  rhs = vector::TransposeOp::create(rew, loc, rhs, ArrayRef<int64_t>{1, 0});
2050  else if (rhsMap != AffineMap::get(3, 0, {k, n}, ctx))
2051  return failure();

2053  // At this point lhs and rhs are in row-major.
2054  VectorType lhsType = cast<VectorType>(lhs.getType());
2055  VectorType rhsType = cast<VectorType>(rhs.getType());
2056  int64_t lhsRows = lhsType.getDimSize(0);
2057  int64_t lhsColumns = lhsType.getDimSize(1);
2058  int64_t rhsColumns = rhsType.getDimSize(1);
2059 
2060  Type flattenedLHSType =
2061  VectorType::get(lhsType.getNumElements(), lhsType.getElementType());
2062  lhs = vector::ShapeCastOp::create(rew, loc, flattenedLHSType, lhs);
2063 
2064  Type flattenedRHSType =
2065  VectorType::get(rhsType.getNumElements(), rhsType.getElementType());

2067

2069  rew, loc,
2070  VectorType::get(lhsRows * rhsColumns,
2071  cast<VectorType>(lhs.getType()).getElementType()),
2072  lhs, rhs, lhsRows, lhsColumns, rhsColumns);
2073 
2074  mul = vector::ShapeCastOp::create(
2075  rew, loc,
2076  VectorType::get({lhsRows, rhsColumns},
2077  getElementTypeOrSelf(op.getAcc().getType())),
2078  mul);
2079 
2080  // ACC must be C(m, n) or C(n, m).
2081  auto accMap = op.getIndexingMapsArray()[2];
2082  if (accMap == AffineMap::get(3, 0, {n, m}, ctx))
2083  mul = vector::TransposeOp::create(rew, loc, mul, ArrayRef<int64_t>{1, 0});
2084  else if (accMap != AffineMap::get(3, 0, {m, n}, ctx))
2085  llvm_unreachable("invalid contraction semantics");
2086

2088  ? static_cast<Value>(
2089  arith::AddIOp::create(rew, loc, op.getAcc(), mul))
2090  : static_cast<Value>(
2091  arith::AddFOp::create(rew, loc, op.getAcc(), mul));
