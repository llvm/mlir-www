210 template <typename OpTy, typename ExtCastTy>
211 LogicalResult IndexCastOpLowering<OpTy, ExtCastTy>::matchAndRewrite(
212  OpTy op, typename OpTy::Adaptor adaptor,
213  ConversionPatternRewriter &rewriter) const {
214  Type resultType = op.getResult().getType();
215  Type targetElementType =
216  this->typeConverter->convertType(getElementTypeOrSelf(resultType));
217  Type sourceElementType =
218  this->typeConverter->convertType(getElementTypeOrSelf(op.getIn()));
219  unsigned targetBits = targetElementType.getIntOrFloatBitWidth();
220  unsigned sourceBits = sourceElementType.getIntOrFloatBitWidth();
221 
222  if (targetBits == sourceBits) {
223  rewriter.replaceOp(op, adaptor.getIn());
224  return success();
225  }
226 
227  // Handle the scalar and 1D vector cases.
228  Type operandType = adaptor.getIn().getType();
229  if (!isa<LLVM::LLVMArrayType>(operandType)) {
230  Type targetType = this->typeConverter->convertType(resultType);
231  if (targetBits < sourceBits)
232  rewriter.replaceOpWithNewOp<LLVM::TruncOp>(op, targetType,
233  adaptor.getIn());
234  else
235  rewriter.replaceOpWithNewOp<ExtCastTy>(op, targetType, adaptor.getIn());
236  return success();
237  }
238 
239  if (!isa<VectorType>(resultType))
240  return rewriter.notifyMatchFailure(op, "expected vector result type");
241 
242  return LLVM::detail::handleMultidimensionalVectors(
243  op.getOperation(), adaptor.getOperands(), *(this->getTypeConverter()),
244  [&](Type llvm1DVectorTy, ValueRange operands) -> Value {
245  typename OpTy::Adaptor adaptor(operands);
246  if (targetBits < sourceBits) {
247  return rewriter.create<LLVM::TruncOp>(op.getLoc(), llvm1DVectorTy,
248  adaptor.getIn());
249  }
250  return rewriter.create<ExtCastTy>(op.getLoc(), llvm1DVectorTy,
251  adaptor.getIn());
252  },
253  rewriter);
254 }

260 LogicalResult AddUIExtendedOpLowering::matchAndRewrite(
261  arith::AddUIExtendedOp op, OpAdaptor adaptor,
262  ConversionPatternRewriter &rewriter) const {
263  Type operandType = adaptor.getLhs().getType();
264  Type sumResultType = op.getSum().getType();
265  Type overflowResultType = op.getOverflow().getType();
266 
267  if (!LLVM::isCompatibleType(operandType))
268  return failure();
269 
270  MLIRContext *ctx = rewriter.getContext();
271  Location loc = op.getLoc();
272 
273  // Handle the scalar and 1D vector cases.
274  if (!isa<LLVM::LLVMArrayType>(operandType)) {
275  Type newOverflowType = typeConverter->convertType(overflowResultType);
276  Type structType =
277  LLVM::LLVMStructType::getLiteral(ctx, {sumResultType, newOverflowType});
278  Value addOverflow = rewriter.create<LLVM::UAddWithOverflowOp>(
279  loc, structType, adaptor.getLhs(), adaptor.getRhs());
280  Value sumExtracted =
281  rewriter.create<LLVM::ExtractValueOp>(loc, addOverflow, 0);
282  Value overflowExtracted =
283  rewriter.create<LLVM::ExtractValueOp>(loc, addOverflow, 1);
284  rewriter.replaceOp(op, {sumExtracted, overflowExtracted});
285  return success();
286  }
287 
288  if (!isa<VectorType>(sumResultType))
289  return rewriter.notifyMatchFailure(loc, "expected vector result types");
290 
291  return rewriter.notifyMatchFailure(loc,
292  "ND vector types are not supported yet");
293 }

299 template <typename ArithMulOp, bool IsSigned>
300 LogicalResult MulIExtendedOpLowering<ArithMulOp, IsSigned>::matchAndRewrite(
301  ArithMulOp op, typename ArithMulOp::Adaptor adaptor,
302  ConversionPatternRewriter &rewriter) const {
303  Type resultType = adaptor.getLhs().getType();
304 
305  if (!LLVM::isCompatibleType(resultType))
306  return failure();
307 
308  Location loc = op.getLoc();
309 
310  // Handle the scalar and 1D vector cases. Because LLVM does not have a
311  // matching extended multiplication intrinsic, perform regular multiplication
312  // on operands zero-extended to i(2*N) bits, and truncate the results back to
313  // iN types.
314  if (!isa<LLVM::LLVMArrayType>(resultType)) {
315  // Shift amount necessary to extract the high bits from widened result.
316  TypedAttr shiftValAttr;
317 
318  if (auto intTy = dyn_cast<IntegerType>(resultType)) {
319  unsigned resultBitwidth = intTy.getWidth();
320  auto attrTy = rewriter.getIntegerType(resultBitwidth * 2);
321  shiftValAttr = rewriter.getIntegerAttr(attrTy, resultBitwidth);
322  } else {
323  auto vecTy = cast<VectorType>(resultType);
324  unsigned resultBitwidth = vecTy.getElementTypeBitWidth();
325  auto attrTy = VectorType::get(
326  vecTy.getShape(), rewriter.getIntegerType(resultBitwidth * 2));
327  shiftValAttr = SplatElementsAttr::get(
328  attrTy, APInt(resultBitwidth * 2, resultBitwidth));
329  }
330  Type wideType = shiftValAttr.getType();
331  assert(LLVM::isCompatibleType(wideType) &&
332  "LLVM dialect should support all signless integer types");
333 
334  using LLVMExtOp = std::conditional_t<IsSigned, LLVM::SExtOp, LLVM::ZExtOp>;
335  Value lhsExt = rewriter.create<LLVMExtOp>(loc, wideType, adaptor.getLhs());
336  Value rhsExt = rewriter.create<LLVMExtOp>(loc, wideType, adaptor.getRhs());
337  Value mulExt = rewriter.create<LLVM::MulOp>(loc, wideType, lhsExt, rhsExt);
338 
339  // Split the 2*N-bit wide result into two N-bit values.
340  Value low = rewriter.create<LLVM::TruncOp>(loc, resultType, mulExt);
341  Value shiftVal = rewriter.create<LLVM::ConstantOp>(loc, shiftValAttr);
342  Value highExt = rewriter.create<LLVM::LShrOp>(loc, mulExt, shiftVal);
343  Value high = rewriter.create<LLVM::TruncOp>(loc, resultType, highExt);
344 
345  rewriter.replaceOp(op, {low, high});
346  return success();
347  }
348 
349  if (!isa<VectorType>(resultType))
350  return rewriter.notifyMatchFailure(op, "expected vector result type");
351 
352  return rewriter.notifyMatchFailure(op,
353  "ND vector types are not supported yet");
354 }

360 // Convert arith.cmp predicate into the LLVM dialect CmpPredicate. The two enums
361 // share numerical values so just cast.
362 template <typename LLVMPredType, typename PredType>
363 static LLVMPredType convertCmpPredicate(PredType pred) {
364  return static_cast<LLVMPredType>(pred);
365 }

367 LogicalResult
368 CmpIOpLowering::matchAndRewrite(arith::CmpIOp op, OpAdaptor adaptor,
369  ConversionPatternRewriter &rewriter) const {
370  Type operandType = adaptor.getLhs().getType();
371  Type resultType = op.getResult().getType();
372 
373  // Handle the scalar and 1D vector cases.
374  if (!isa<LLVM::LLVMArrayType>(operandType)) {
375  rewriter.replaceOpWithNewOp<LLVM::ICmpOp>(
376  op, typeConverter->convertType(resultType),
377  convertCmpPredicate<LLVM::ICmpPredicate>(op.getPredicate()),
378  adaptor.getLhs(), adaptor.getRhs());
379  return success();
380  }
381 
382  if (!isa<VectorType>(resultType))
383  return rewriter.notifyMatchFailure(op, "expected vector result type");
384 
385  return LLVM::detail::handleMultidimensionalVectors(
386  op.getOperation(), adaptor.getOperands(), *getTypeConverter(),
387  [&](Type llvm1DVectorTy, ValueRange operands) {
388  OpAdaptor adaptor(operands);
389  return rewriter.create<LLVM::ICmpOp>(
390  op.getLoc(), llvm1DVectorTy,
391  convertCmpPredicate<LLVM::ICmpPredicate>(op.getPredicate()),
392  adaptor.getLhs(), adaptor.getRhs());
393  },
394  rewriter);
395 }
