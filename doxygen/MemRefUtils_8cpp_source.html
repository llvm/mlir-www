23 bool isStaticShapeAndContiguousRowMajor(MemRefType type) {
24  if (!type.hasStaticShape())
25  return false;
26 
27  SmallVector<int64_t> strides;
28  int64_t offset;
29  if (failed(type.getStridesAndOffset(strides, offset)))
30  return false;
31 
32  // MemRef is contiguous if outer dimensions are size-1 and inner
33  // dimensions have unit strides.
34  int64_t runningStride = 1;
35  int64_t curDim = strides.size() - 1;
36  // Finds all inner dimensions with unit strides.
37  while (curDim >= 0 && strides[curDim] == runningStride) {
38  runningStride *= type.getDimSize(curDim);
39  --curDim;
40  }
41 
42  // Check if other dimensions are size-1.
43  while (curDim >= 0 && type.getDimSize(curDim) == 1) {
44  --curDim;
45  }
46 
47  // All dims are unit-strided or size-1.
48  return curDim < 0;
49 }

51 std::pair<LinearizedMemRefInfo, OpFoldResult> getLinearizedMemRefOffsetAndSize(
52  OpBuilder &builder, Location loc, int srcBits, int dstBits,
53  OpFoldResult offset, ArrayRef<OpFoldResult> sizes,
54  ArrayRef<OpFoldResult> strides, ArrayRef<OpFoldResult> indices) {
55  unsigned sourceRank = sizes.size();
56  assert(sizes.size() == strides.size() &&
57  "expected as many sizes as strides for a memref");
58  SmallVector<OpFoldResult> indicesVec = llvm::to_vector(indices);
59  if (indices.empty())
60  indicesVec.resize(sourceRank, builder.getIndexAttr(0));
61  assert(indicesVec.size() == strides.size() &&
62  "expected as many indices as rank of memref");
63 
64  // Create the affine symbols and values for linearization.
65  SmallVector<AffineExpr> symbols(2 * sourceRank);
66  bindSymbolsList(builder.getContext(), MutableArrayRef{symbols});
67  AffineExpr addMulMap = builder.getAffineConstantExpr(0);
68 
69  SmallVector<OpFoldResult> offsetValues(2 * sourceRank);
70 
71  for (unsigned i = 0; i < sourceRank; ++i) {
72  unsigned offsetIdx = 2 * i;
73  addMulMap = addMulMap + symbols[offsetIdx] * symbols[offsetIdx + 1];
74  offsetValues[offsetIdx] = indicesVec[i];
75  offsetValues[offsetIdx + 1] = strides[i];
76  }
77  // Adjust linearizedIndices and size by the scale factor (dstBits / srcBits).
78  int64_t scaler = dstBits / srcBits;
79  OpFoldResult linearizedIndices = affine::makeComposedFoldedAffineApply(
80  builder, loc, addMulMap.floorDiv(scaler), offsetValues);
81 
82  size_t symbolIndex = 0;
83  SmallVector<OpFoldResult> values;
84  SmallVector<AffineExpr> productExpressions;
85  for (unsigned i = 0; i < sourceRank; ++i) {
86  AffineExpr strideExpr = symbols[symbolIndex++];
87  values.push_back(strides[i]);
88  AffineExpr sizeExpr = symbols[symbolIndex++];
89  values.push_back(sizes[i]);
90 
91  productExpressions.push_back((strideExpr * sizeExpr).floorDiv(scaler));
92  }
93  AffineMap maxMap = AffineMap::get(
94  /*dimCount=*/0, /*symbolCount=*/symbolIndex, productExpressions,
95  builder.getContext());
96  OpFoldResult linearizedSize =
97  affine::makeComposedFoldedAffineMax(builder, loc, maxMap, values);
98 
99  // Adjust baseOffset by the scale factor (dstBits / srcBits).
100  AffineExpr s0;
101  bindSymbols(builder.getContext(), s0);
102  OpFoldResult adjustBaseOffset = affine::makeComposedFoldedAffineApply(
103  builder, loc, s0.floorDiv(scaler), {offset});
104 
105  OpFoldResult intraVectorOffset = affine::makeComposedFoldedAffineApply(
106  builder, loc, addMulMap % scaler, offsetValues);
107 
108  return {{adjustBaseOffset, linearizedSize, intraVectorOffset},
109  linearizedIndices};
110 }

112 LinearizedMemRefInfo
113 getLinearizedMemRefOffsetAndSize(OpBuilder &builder, Location loc, int srcBits,
114  int dstBits, OpFoldResult offset,
115  ArrayRef<OpFoldResult> sizes) {
116  SmallVector<OpFoldResult> strides(sizes.size());
117  if (!sizes.empty()) {
118  strides.back() = builder.getIndexAttr(1);
119  AffineExpr s0, s1;
120  bindSymbols(builder.getContext(), s0, s1);
121  for (int index = sizes.size() - 1; index > 0; --index) {
122  strides[index - 1] = affine::makeComposedFoldedAffineApply(
123  builder, loc, s0 * s1,
124  ArrayRef<OpFoldResult>{strides[index], sizes[index]});
125  }
126  }
127 
128  LinearizedMemRefInfo linearizedMemRefInfo;
129  std::tie(linearizedMemRefInfo, std::ignore) =
130  getLinearizedMemRefOffsetAndSize(builder, loc, srcBits, dstBits, offset,
131  sizes, strides);
132  return linearizedMemRefInfo;
133 }

135 /// Returns true if all the uses of op are not read/load.
136 /// There can be SubviewOp users as long as all its users are also
137 /// StoreOp/transfer_write. If return true it also fills out the uses, if it
138 /// returns false uses is unchanged.
139 static bool resultIsNotRead(Operation *op, std::vector<Operation *> &uses) {
140  std::vector<Operation *> opUses;
141  for (OpOperand &use : op->getUses()) {
142  Operation *useOp = use.getOwner();
143  if (isa<memref::DeallocOp>(useOp) ||
144  (useOp->getNumResults() == 0 && useOp->getNumRegions() == 0 &&
145  !mlir::hasEffect<MemoryEffects::Read>(useOp)) ||
146  (isa<memref::SubViewOp>(useOp) && resultIsNotRead(useOp, opUses))) {
147  opUses.push_back(useOp);
148  continue;
149  }
150  return false;
151  }
152  llvm::append_range(uses, opUses);
153  return true;
154 }

156 void eraseDeadAllocAndStores(RewriterBase &rewriter, Operation *parentOp) {
157  std::vector<Operation *> opToErase;
158  parentOp->walk([&](Operation *op) {
159  std::vector<Operation *> candidates;
160  if (isa<memref::AllocOp, memref::AllocaOp>(op) &&
161  resultIsNotRead(op, candidates)) {
162  llvm::append_range(opToErase, candidates);
163  opToErase.push_back(op);
164  }
165  });
166 
167  for (Operation *op : opToErase)
168  rewriter.eraseOp(op);
169 }

171 static SmallVector<OpFoldResult>
172 computeSuffixProductIRBlockImpl(Location loc, OpBuilder &builder,
173  ArrayRef<OpFoldResult> sizes,
174  OpFoldResult unit) {
175  SmallVector<OpFoldResult> strides(sizes.size(), unit);
176  AffineExpr s0, s1;
177  bindSymbols(builder.getContext(), s0, s1);
178 
179  for (int64_t r = strides.size() - 1; r > 0; --r) {
180  strides[r - 1] = affine::makeComposedFoldedAffineApply(
181  builder, loc, s0 * s1, {strides[r], sizes[r]});
182  }
183  return strides;
184 }

193 MemrefValue skipFullyAliasingOperations(MemrefValue source) {
194  while (auto op = source.getDefiningOp()) {
195  if (auto subViewOp = dyn_cast<memref::SubViewOp>(op);
196  subViewOp && subViewOp.hasZeroOffset() && subViewOp.hasUnitStride()) {
197  // A `memref.subview` with an all zero offset, and all unit strides, still
198  // points to the same memory.
199  source = cast<MemrefValue>(subViewOp.getSource());
200  } else if (auto castOp = dyn_cast<memref::CastOp>(op)) {
201  // A `memref.cast` still points to the same memory.
202  source = castOp.getSource();
203  } else {
204  return source;
205  }
206  }
207  return source;
208 }

210 MemrefValue skipViewLikeOps(MemrefValue source) {
211  while (auto op = source.getDefiningOp()) {
212  if (auto viewLike = dyn_cast<ViewLikeOpInterface>(op)) {
213  if (source == viewLike.getViewDest()) {
214  source = cast<MemrefValue>(viewLike.getViewSource());
215  continue;
216  }
217  }
218  return source;
219  }
220  return source;
221 }

223 LogicalResult resolveSourceIndicesExpandShape(
224  Location loc, PatternRewriter &rewriter,
225  memref::ExpandShapeOp expandShapeOp, ValueRange indices,
226  SmallVectorImpl<Value> &sourceIndices, bool startsInbounds) {
227  SmallVector<OpFoldResult> destShape = expandShapeOp.getMixedOutputShape();
228 
229  // Traverse all reassociation groups to determine the appropriate indices
230  // corresponding to each one of them post op folding.
231  for (ArrayRef<int64_t> group : expandShapeOp.getReassociationIndices()) {
232  assert(!group.empty() && "association indices groups cannot be empty");
233  int64_t groupSize = group.size();
234  if (groupSize == 1) {
235  sourceIndices.push_back(indices[group[0]]);
236  continue;
237  }
238  SmallVector<OpFoldResult> groupBasis =
239  llvm::map_to_vector(group, [&](int64_t d) { return destShape[d]; });
240  SmallVector<Value> groupIndices =
241  llvm::map_to_vector(group, [&](int64_t d) { return indices[d]; });
242  Value collapsedIndex = affine::AffineLinearizeIndexOp::create(
243  rewriter, loc, groupIndices, groupBasis, /*disjoint=*/startsInbounds);
244  sourceIndices.push_back(collapsedIndex);
245  }
246  return success();
247 }

249 LogicalResult
250 resolveSourceIndicesCollapseShape(Location loc, PatternRewriter &rewriter,
251  memref::CollapseShapeOp collapseShapeOp,
252  ValueRange indices,
253  SmallVectorImpl<Value> &sourceIndices) {
254  // Note: collapse_shape requires a strided memref, we can do this.
255  auto metadata = memref::ExtractStridedMetadataOp::create(
256  rewriter, loc, collapseShapeOp.getSrc());
257  SmallVector<OpFoldResult> sourceSizes = metadata.getConstifiedMixedSizes();
258  for (auto [index, group] :
259  llvm::zip(indices, collapseShapeOp.getReassociationIndices())) {
260  assert(!group.empty() && "association indices groups cannot be empty");
261  int64_t groupSize = group.size();
262 
263  if (groupSize == 1) {
264  sourceIndices.push_back(index);
265  continue;
266  }
267 
268  SmallVector<OpFoldResult> basis =
269  llvm::map_to_vector(group, [&](int64_t d) { return sourceSizes[d]; });
270  auto delinearize = affine::AffineDelinearizeIndexOp::create(
271  rewriter, loc, index, basis, /*hasOuterBound=*/true);
272  llvm::append_range(sourceIndices, delinearize.getResults());
273  }
274  if (collapseShapeOp.getReassociationIndices().empty()) {
275  auto zeroAffineMap = rewriter.getConstantAffineMap(0);
276  int64_t srcRank =
277  cast<MemRefType>(collapseShapeOp.getViewSource().getType()).getRank();
278  OpFoldResult ofr = affine::makeComposedFoldedAffineApply(
279  rewriter, loc, zeroAffineMap, ArrayRef<OpFoldResult>{});
280  for (int64_t i = 0; i < srcRank; i++) {
281  sourceIndices.push_back(
282  getValueOrCreateConstantIndexOp(rewriter, loc, ofr));
283  }
284  }
285  return success();
286 }
