22 static bool isLessThanTargetBitWidth(Operation *op, unsigned targetBitWidth) {
23  auto resultTypes = op->getResultTypes();
24  for (auto resType : resultTypes) {
25  VectorType vecType = dyn_cast<VectorType>(resType);
26  // Reject index since getElementTypeBitWidth will abort for Index types.
27  if (!vecType || vecType.getElementType().isIndex())
28  return false;
29  unsigned trailingVecDimBitWidth =
30  vecType.getShape().back() * vecType.getElementTypeBitWidth();
31  if (trailingVecDimBitWidth >= targetBitWidth)
32  return false;
33  }
34  return true;
35 }

38 struct LinearizeConstant final : OpConversionPattern<arith::ConstantOp> {
39  using OpConversionPattern::OpConversionPattern;
40  LinearizeConstant(
41  const TypeConverter &typeConverter, MLIRContext *context,
42  unsigned targetVectBitWidth = std::numeric_limits<unsigned>::max(),
43  PatternBenefit benefit = 1)
44  : OpConversionPattern(typeConverter, context, benefit),
45  targetVectorBitWidth(targetVectBitWidth) {}
46  LogicalResult
47  matchAndRewrite(arith::ConstantOp constOp, OpAdaptor adaptor,
48  ConversionPatternRewriter &rewriter) const override {
49  Location loc = constOp.getLoc();
50  auto resType =
51  getTypeConverter()->convertType<VectorType>(constOp.getType());
52 
53  if (resType.isScalable() && !isa<SplatElementsAttr>(constOp.getValue()))
54  return rewriter.notifyMatchFailure(
55  loc,
56  "Cannot linearize a constant scalable vector that's not a splat");
57 
58  if (!resType)
59  return rewriter.notifyMatchFailure(loc, "can't convert return type");
60  if (!isLessThanTargetBitWidth(constOp, targetVectorBitWidth))
61  return rewriter.notifyMatchFailure(
62  loc, "Can't flatten since targetBitWidth <= OpSize");
63  auto dstElementsAttr = dyn_cast<DenseElementsAttr>(constOp.getValue());
64  if (!dstElementsAttr)
65  return rewriter.notifyMatchFailure(loc, "unsupported attr type");
66 
67  dstElementsAttr = dstElementsAttr.reshape(resType);
68  rewriter.replaceOpWithNewOp<arith::ConstantOp>(constOp, resType,
69  dstElementsAttr);
70  return success();
71  }
72 
73 private:
74  unsigned targetVectorBitWidth;
75 };

77 struct LinearizeVectorizable final
78  : OpTraitConversionPattern<OpTrait::Vectorizable> {
79  using OpTraitConversionPattern::OpTraitConversionPattern;
80 
81 public:
82  LinearizeVectorizable(
83  const TypeConverter &typeConverter, MLIRContext *context,
84  unsigned targetVectBitWidth = std::numeric_limits<unsigned>::max(),
85  PatternBenefit benefit = 1)
86  : OpTraitConversionPattern(typeConverter, context, benefit),
87  targetVectorBitWidth(targetVectBitWidth) {}
88  LogicalResult
89  matchAndRewrite(Operation *op, ArrayRef<Value> operands,
90  ConversionPatternRewriter &rewriter) const override {
91  if (!isLessThanTargetBitWidth(op, targetVectorBitWidth))
92  return rewriter.notifyMatchFailure(
93  op->getLoc(), "Can't flatten since targetBitWidth <= OpSize");
94  FailureOr<Operation *> newOp =
95  convertOpResultTypes(op, operands, *getTypeConverter(), rewriter);
96  if (failed(newOp))
97  return failure();
98 
99  rewriter.replaceOp(op, (*newOp)->getResults());
100  return success();
101  }
102 
103 private:
104  unsigned targetVectorBitWidth;
105 };

108 void mlir::vector::populateVectorLinearizeTypeConversionsAndLegality(
109  TypeConverter &typeConverter, RewritePatternSet &patterns,
110  ConversionTarget &target, unsigned targetBitWidth) {
111 
112  typeConverter.addConversion([](VectorType type) -> std::optional<Type> {
113  if (!isLinearizableVector(type))
114  return type;
115 
116  return VectorType::get(type.getNumElements(), type.getElementType(),
117  type.isScalable());
118  });
119 
120  auto materializeCast = [](OpBuilder &builder, Type type, ValueRange inputs,
121  Location loc) -> Value {
122  if (inputs.size() != 1 || !isa<VectorType>(inputs.front().getType()) ||
123  !isa<VectorType>(type))
124  return nullptr;
125 
126  return builder.create<vector::ShapeCastOp>(loc, type, inputs.front());
127  };
128  typeConverter.addArgumentMaterialization(materializeCast);
129  typeConverter.addSourceMaterialization(materializeCast);
130  typeConverter.addTargetMaterialization(materializeCast);
131  target.markUnknownOpDynamicallyLegal(
132  [=](Operation *op) -> std::optional<bool> {
133  if ((isa<arith::ConstantOp>(op) ||
134  op->hasTrait<OpTrait::Vectorizable>())) {
135  return (isLessThanTargetBitWidth(op, targetBitWidth)
136  ? typeConverter.isLegal(op)
137  : true);
138  }
139  return std::nullopt;
140  });
141 
142  patterns.add<LinearizeConstant, LinearizeVectorizable>(
143  typeConverter, patterns.getContext(), targetBitWidth);
144 }
