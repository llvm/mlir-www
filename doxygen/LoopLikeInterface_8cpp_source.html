21 bool LoopLikeOpInterface::blockIsInLoop(Block *block) {
22  Operation *parent = block->getParentOp();
23 
24  // The block could be inside a loop-like operation
25  if (isa<LoopLikeOpInterface>(parent) ||
26  parent->getParentOfType<LoopLikeOpInterface>())
27  return true;
28 
29  // This block might be nested inside another block, which is in a loop
30  if (!isa<FunctionOpInterface>(parent))
31  if (mlir::Block *parentBlock = parent->getBlock())
32  if (blockIsInLoop(parentBlock))
33  return true;
34 
35  // Or the block could be inside a control flow graph loop:
36  // A block is in a control flow graph loop if it can reach itself in a graph
37  // traversal
38  DenseSet<Block *> visited;
39  SmallVector<Block *> stack;
40  stack.push_back(block);
41  while (!stack.empty()) {
42  Block *current = stack.pop_back_val();
43  auto [it, inserted] = visited.insert(current);
44  if (!inserted) {
45  // loop detected
46  if (current == block)
47  return true;
48  continue;
49  }
50 
51  stack.reserve(stack.size() + current->getNumSuccessors());
52  for (Block *successor : current->getSuccessors())
53  stack.push_back(successor);
54  }
55  return false;
56 }

58 LogicalResult detail::verifyLoopLikeOpInterface(Operation *op) {
59  // Note: These invariants are also verified by the RegionBranchOpInterface,
60  // but the LoopLikeOpInterface provides better error messages.
61  auto loopLikeOp = cast<LoopLikeOpInterface>(op);
62 
63  // Verify number of inits/iter_args/yielded values/loop results.
64  if (loopLikeOp.getInits().size() != loopLikeOp.getRegionIterArgs().size())
65  return op->emitOpError("different number of inits and region iter_args: ")
66  << loopLikeOp.getInits().size()
67  << " != " << loopLikeOp.getRegionIterArgs().size();
68  if (!loopLikeOp.getYieldedValues().empty() &&
69  loopLikeOp.getRegionIterArgs().size() !=
70  loopLikeOp.getYieldedValues().size())
71  return op->emitOpError(
72  "different number of region iter_args and yielded values: ")
73  << loopLikeOp.getRegionIterArgs().size()
74  << " != " << loopLikeOp.getYieldedValues().size();
75  if (loopLikeOp.getLoopResults() && loopLikeOp.getLoopResults()->size() !=
76  loopLikeOp.getRegionIterArgs().size())
77  return op->emitOpError(
78  "different number of loop results and region iter_args: ")
79  << loopLikeOp.getLoopResults()->size()
80  << " != " << loopLikeOp.getRegionIterArgs().size();
81 
82  // Verify types of inits/iter_args/yielded values/loop results.
83  int64_t i = 0;
84  auto yieldedValues = loopLikeOp.getYieldedValues();
85  for (const auto [index, init, regionIterArg] :
86  llvm::enumerate(loopLikeOp.getInits(), loopLikeOp.getRegionIterArgs())) {
87  if (init.getType() != regionIterArg.getType())
88  return op->emitOpError(std::to_string(index))
89  << "-th init and " << index
90  << "-th region iter_arg have different type: " << init.getType()
91  << " != " << regionIterArg.getType();
92  if (!yieldedValues.empty()) {
93  if (regionIterArg.getType() != yieldedValues[index].getType())
94  return op->emitOpError(std::to_string(index))
95  << "-th region iter_arg and " << index
96  << "-th yielded value have different type: "
97  << regionIterArg.getType()
98  << " != " << yieldedValues[index].getType();
99  }
100  ++i;
101  }
102  i = 0;
103  if (loopLikeOp.getLoopResults()) {
104  for (const auto it : llvm::zip_equal(loopLikeOp.getRegionIterArgs(),
105  *loopLikeOp.getLoopResults())) {
106  if (std::get<0>(it).getType() != std::get<1>(it).getType())
107  return op->emitOpError(std::to_string(i))
108  << "-th region iter_arg and " << i
109  << "-th loop result have different type: "
110  << std::get<0>(it).getType()
111  << " != " << std::get<1>(it).getType();
112  }
113  ++i;
114  }
115 
116  return success();
117 }

119 LoopLikeOpInterface mlir::createFused(LoopLikeOpInterface target,
120  LoopLikeOpInterface source,
121  RewriterBase &rewriter,
122  NewYieldValuesFn newYieldValuesFn,
123  FuseTerminatorFn fuseTerminatorFn) {
124  auto targetIterArgs = target.getRegionIterArgs();
125  std::optional<SmallVector<Value>> targetInductionVar =
126  target.getLoopInductionVars();
127  SmallVector<Value> targetYieldOperands(target.getYieldedValues());
128  auto sourceIterArgs = source.getRegionIterArgs();
129  std::optional<SmallVector<Value>> sourceInductionVar =
130  *source.getLoopInductionVars();
131  SmallVector<Value> sourceYieldOperands(source.getYieldedValues());
132  auto sourceRegion = source.getLoopRegions().front();
133 
134  FailureOr<LoopLikeOpInterface> maybeFusedLoop =
135  target.replaceWithAdditionalYields(rewriter, source.getInits(),
136  /*replaceInitOperandUsesInLoop=*/false,
137  newYieldValuesFn);
138  if (failed(maybeFusedLoop))
139  llvm_unreachable("failed to replace loop");
140  LoopLikeOpInterface fusedLoop = *maybeFusedLoop;
141 
142  // Map control operands.
143  IRMapping mapping;
144  std::optional<SmallVector<Value>> fusedInductionVar =
145  fusedLoop.getLoopInductionVars();
146  if (fusedInductionVar) {
147  if (!targetInductionVar || !sourceInductionVar)
148  llvm_unreachable("expected target and source loops to have induction vars");
149  mapping.map(*targetInductionVar, *fusedInductionVar);
150  mapping.map(*sourceInductionVar, *fusedInductionVar);
151  }
152  mapping.map(targetIterArgs,
153  fusedLoop.getRegionIterArgs().take_front(targetIterArgs.size()));
154  mapping.map(targetYieldOperands,
155  fusedLoop.getYieldedValues().take_front(targetIterArgs.size()));
156  mapping.map(sourceIterArgs,
157  fusedLoop.getRegionIterArgs().take_back(sourceIterArgs.size()));
158  mapping.map(sourceYieldOperands,
159  fusedLoop.getYieldedValues().take_back(sourceIterArgs.size()));
160  // Append everything except the terminator into the fused operation.
161  rewriter.setInsertionPoint(
162  fusedLoop.getLoopRegions().front()->front().getTerminator());
163  for (Operation &op : sourceRegion->front().without_terminator())
164  rewriter.clone(op, mapping);
165 
166  // TODO: Replace with corresponding interface method if added
167  fuseTerminatorFn(rewriter, source, fusedLoop, mapping);
168 
169  return fusedLoop;
170 }
