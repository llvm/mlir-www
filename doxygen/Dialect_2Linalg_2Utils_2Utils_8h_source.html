32 /// Apply the permutation defined by `permutation` to `inVec`.
33 /// Element `i` in `inVec` is mapped to location `j = permutation[i]`.
34 /// E.g.: for an input vector `inVec = ['a', 'b', 'c']` and a permutation vector
35 /// `permutation = [2, 0, 1]`, this function leaves `inVec = ['c', 'a', 'b']`.
36 template <typename T, unsigned N>
37 void applyPermutationToVector(SmallVector<T, N> &inVec,
38  ArrayRef<unsigned> permutation) {
39  SmallVector<T, N> auxVec(inVec.size());
40  for (unsigned i = 0; i < permutation.size(); ++i)
41  auxVec[i] = inVec[permutation[i]];
42  inVec = auxVec;
43 }

168 /// Scheme used to distribute loops to processors.
169 enum class DistributionMethod {
170  /// Cyclic distribution where no assumption is made about the dynamic
171  /// relationship between number of processors and number of iterations of the
172  /// distributed loop. Distributes the following loop
173  ///
174  /// scf.parallel (%iv) = (%lb) to (%ub) step (%step)
175  ///
176  /// to
177  ///
178  /// scf.parallel(%iv)= (%lb + %procId * %step) to (%ub) step (%step * %nprocs)
179  Cyclic = 0,
180 
181  /// Cyclic distribution where the number of processors can be assumed to be
182  /// more than or equal to the number of iterations of the distributed loop. In
183  /// such cases, a simple in-bounds check is enough (instead of materializing a
184  /// loop). Distributes the following loop
185  ///
186  /// scf.parallel (%iv) = (%lb) to (%ub) step (%step)
187  ///
188  /// to
189  ///
190  /// %iv = %lb + %procId * %step
191  /// %cond = cmpi "slt", %iv, %ub
192  /// scf.if %cond {
193  /// ...
194  /// }
195  CyclicNumProcsGeNumIters = 1,
196 
197  /// Cyclic distribution where the number of processors can be assumed to be
198  /// equal to the number of iterations of the distributed loop. In such cases,
199  /// no bounds check is needed. Distributes the following loop
200  ///
201  /// scf.parallel (%iv) = (%lb) to (%ub) step (%step)
202  ///
203  /// to
204  ///
205  /// %iv = %lb + %procId * %step
206  CyclicNumProcsEqNumIters = 2
207 };

225  /// number of `{procId, nprocs}` pairs returned must be equal to the number of
