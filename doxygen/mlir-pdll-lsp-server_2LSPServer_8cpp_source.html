117 void LSPServer::onInitialize(const InitializeParams &params,
118  Callback<llvm::json::Value> reply) {
119  // Send a response with the capabilities of this server.
120  llvm::json::Object serverCaps{
121  {"textDocumentSync",
122  llvm::json::Object{
123  {"openClose", true},
124  {"change", (int)TextDocumentSyncKind::Incremental},
125  {"save", true},
126  }},
127  {"completionProvider",
128  llvm::json::Object{
129  {"allCommitCharacters",
130  {"\t", "(", ")", "[", "]", "{", "}", "<", ">",
131  ":", ";", ",", "+", "-", "/", "*", "%", "^",
132  "&", "#", "?", ".", "=", "\"", "'", "|"}},
133  {"resolveProvider", false},
134  {"triggerCharacters",
135  {".", ">", "(", "{", ",", "<", ":", "[", " ", "\"", "/"}},
136  }},
137  {"signatureHelpProvider",
138  llvm::json::Object{
139  {"triggerCharacters", {"(", ","}},
140  }},
141  {"definitionProvider", true},
142  {"referencesProvider", true},
143  {"documentLinkProvider",
144  llvm::json::Object{
145  {"resolveProvider", false},
146  }},
147  {"hoverProvider", true},
148  {"documentSymbolProvider", true},
149  {"inlayHintProvider", true},
150  };
151 
152  llvm::json::Object result{
153  {{"serverInfo", llvm::json::Object{{"name", "mlir-pdll-lsp-server"},
154  {"version", "0.0.1"}}},
155  {"capabilities", std::move(serverCaps)}}};
156  reply(std::move(result));
157 }

168 void LSPServer::onDocumentDidOpen(const DidOpenTextDocumentParams &params) {
169  PublishDiagnosticsParams diagParams(params.textDocument.uri,
170  params.textDocument.version);
171  server.addDocument(params.textDocument.uri, params.textDocument.text,
172  params.textDocument.version, diagParams.diagnostics);
173 
174  // Publish any recorded diagnostics.
175  publishDiagnostics(diagParams);
176 }
177 void LSPServer::onDocumentDidClose(const DidCloseTextDocumentParams &params) {
178  std::optional<int64_t> version =
179  server.removeDocument(params.textDocument.uri);
180  if (!version)
181  return;
182 
183  // Empty out the diagnostics shown for this document. This will clear out
184  // anything currently displayed by the client for this document (e.g. in the
185  // "Problems" pane of VSCode).
186  publishDiagnostics(
187  PublishDiagnosticsParams(params.textDocument.uri, *version));
188 }
189 void LSPServer::onDocumentDidChange(const DidChangeTextDocumentParams &params) {
190  PublishDiagnosticsParams diagParams(params.textDocument.uri,
191  params.textDocument.version);
192  server.updateDocument(params.textDocument.uri, params.contentChanges,
193  params.textDocument.version, diagParams.diagnostics);
194 
195  // Publish any recorded diagnostics.
196  publishDiagnostics(diagParams);
197 }

291 LogicalResult mlir::lsp::runPdllLSPServer(PDLLServer &server,
292  JSONTransport &transport) {
293  LSPServer lspServer(server, transport);
294  MessageHandler messageHandler(transport);
295 
296  // Initialization
297  messageHandler.method("initialize", &lspServer, &LSPServer::onInitialize);
298  messageHandler.notification("initialized", &lspServer,
299  &LSPServer::onInitialized);
300  messageHandler.method("shutdown", &lspServer, &LSPServer::onShutdown);
301 
302  // Document Changes
303  messageHandler.notification("textDocument/didOpen", &lspServer,
304  &LSPServer::onDocumentDidOpen);
305  messageHandler.notification("textDocument/didClose", &lspServer,
306  &LSPServer::onDocumentDidClose);
307  messageHandler.notification("textDocument/didChange", &lspServer,
308  &LSPServer::onDocumentDidChange);
309 
310  // Definitions and References
311  messageHandler.method("textDocument/definition", &lspServer,
312  &LSPServer::onGoToDefinition);
313  messageHandler.method("textDocument/references", &lspServer,
314  &LSPServer::onReference);
315 
316  // Document Link
317  messageHandler.method("textDocument/documentLink", &lspServer,
318  &LSPServer::onDocumentLink);
319 
320  // Hover
321  messageHandler.method("textDocument/hover", &lspServer, &LSPServer::onHover);
322 
323  // Document Symbols
324  messageHandler.method("textDocument/documentSymbol", &lspServer,
325  &LSPServer::onDocumentSymbol);
326 
327  // Code Completion
328  messageHandler.method("textDocument/completion", &lspServer,
329  &LSPServer::onCompletion);
330 
331  // Signature Help
332  messageHandler.method("textDocument/signatureHelp", &lspServer,
333  &LSPServer::onSignatureHelp);
334 
335  // Inlay Hints
336  messageHandler.method("textDocument/inlayHint", &lspServer,
337  &LSPServer::onInlayHint);
338 
339  // PDLL ViewOutput
340  messageHandler.method("pdll/viewOutput", &lspServer,
341  &LSPServer::onPDLLViewOutput);
342 
343  // Diagnostics
344  lspServer.publishDiagnostics =
345  messageHandler.outgoingNotification<PublishDiagnosticsParams>(
346  "textDocument/publishDiagnostics");
347 
348  // Run the main loop of the transport.
349  if (llvm::Error error = transport.run(messageHandler)) {
350  Logger::error("Transport error: {0}", error);
351  llvm::consumeError(std::move(error));
352  return failure();
353  }
354  return success(lspServer.shutdownRequestReceived);
355 }
