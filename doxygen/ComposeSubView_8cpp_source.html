28 // Replaces a subview of a subview with a single subview. Only supports subview
29 // ops with static sizes and static strides of 1 (both static and dynamic
30 // offsets are supported).
31 struct ComposeSubViewOpPattern : public OpRewritePattern<memref::SubViewOp> {
32  using OpRewritePattern::OpRewritePattern;
33 
34  LogicalResult matchAndRewrite(memref::SubViewOp op,
35  PatternRewriter &rewriter) const override {
36  // 'op' is the 'SubViewOp' we're rewriting. 'sourceOp' is the op that
37  // produces the input of the op we're rewriting (for 'SubViewOp' the input
38  // is called the "source" value). We can only combine them if both 'op' and
39  // 'sourceOp' are 'SubViewOp'.
40  auto sourceOp = op.source().getDefiningOp<memref::SubViewOp>();
41  if (!sourceOp)
42  return failure();
43 
44  // A 'SubViewOp' can be "rank-reducing" by eliminating dimensions of the
45  // output memref that are statically known to be equal to 1. We do not
46  // allow 'sourceOp' to be a rank-reducing subview because then our two
47  // 'SubViewOp's would have different numbers of offset/size/stride
48  // parameters (just difficult to deal with, not impossible if we end up
49  // needing it).
50  if (sourceOp.getSourceType().getRank() != sourceOp.getType().getRank()) {
51  return failure();
52  }
53 
54  // Offsets, sizes and strides OpFoldResult for the combined 'SubViewOp'.
55  SmallVector<OpFoldResult> offsets, sizes, strides;
56 
57  // Because we only support input strides of 1, the output stride is also
58  // always 1.
59  if (llvm::all_of(strides, [](OpFoldResult &valueOrAttr) {
60  Attribute attr = valueOrAttr.dyn_cast<Attribute>();
61  return attr && attr.cast<IntegerAttr>().getInt() == 1;
62  })) {
63  strides = SmallVector<OpFoldResult>(sourceOp.getMixedStrides().size(),
64  rewriter.getI64IntegerAttr(1));
65  } else {
66  return failure();
67  }
68 
69  // The rules for calculating the new offsets and sizes are:
70  // * Multiple subview offsets for a given dimension compose additively.
71  // ("Offset by m" followed by "Offset by n" == "Offset by m + n")
72  // * Multiple sizes for a given dimension compose by taking the size of the
73  // final subview and ignoring the rest. ("Take m values" followed by "Take
74  // n values" == "Take n values") This size must also be the smallest one
75  // by definition (a subview needs to be the same size as or smaller than
76  // its source along each dimension; presumably subviews that are larger
77  // than their sources are disallowed by validation).
78  for (auto it : llvm::zip(op.getMixedOffsets(), sourceOp.getMixedOffsets(),
79  op.getMixedSizes())) {
80  auto opOffset = std::get<0>(it);
81  auto sourceOffset = std::get<1>(it);
82  auto opSize = std::get<2>(it);
83 
84  // We only support static sizes.
85  if (opSize.is<Value>()) {
86  return failure();
87  }
88 
89  sizes.push_back(opSize);
90  Attribute opOffsetAttr = opOffset.dyn_cast<Attribute>(),
91  sourceOffsetAttr = sourceOffset.dyn_cast<Attribute>();
92 
93  if (opOffsetAttr && sourceOffsetAttr) {
94  // If both offsets are static we can simply calculate the combined
95  // offset statically.
96  offsets.push_back(rewriter.getI64IntegerAttr(
97  opOffsetAttr.cast<IntegerAttr>().getInt() +
98  sourceOffsetAttr.cast<IntegerAttr>().getInt()));
99  } else {
100  // When either offset is dynamic, we must emit an additional affine
101  // transformation to add the two offsets together dynamically.
102  AffineExpr expr = rewriter.getAffineConstantExpr(0);
103  SmallVector<Value> affineApplyOperands;
104  for (auto valueOrAttr : {opOffset, sourceOffset}) {
105  if (auto attr = valueOrAttr.dyn_cast<Attribute>()) {
106  expr = expr + attr.cast<IntegerAttr>().getInt();
107  } else {
108  expr =
109  expr + rewriter.getAffineSymbolExpr(affineApplyOperands.size());
110  affineApplyOperands.push_back(valueOrAttr.get<Value>());
111  }
112  }
113 
114  AffineMap map = AffineMap::get(0, affineApplyOperands.size(), expr);
115  Value result = rewriter.create<AffineApplyOp>(op.getLoc(), map,
116  affineApplyOperands);
117  offsets.push_back(result);
118  }
119  }
120 
121  // This replaces 'op' but leaves 'sourceOp' alone; if it no longer has any
122  // uses it can be removed by a (separate) dead code elimination pass.
123  rewriter.replaceOpWithNewOp<memref::SubViewOp>(op, sourceOp.source(),
124  offsets, sizes, strides);
125  return success();
126  }
127 };
