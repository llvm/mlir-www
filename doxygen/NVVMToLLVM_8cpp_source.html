43 struct PtxLowering
44  : public OpInterfaceRewritePattern<BasicPtxBuilderInterface> {
45  using OpInterfaceRewritePattern<
46  BasicPtxBuilderInterface>::OpInterfaceRewritePattern;
47 
48  PtxLowering(MLIRContext *context, PatternBenefit benefit = 2)
49  : OpInterfaceRewritePattern(context, benefit) {}
50 
51  LogicalResult matchAndRewrite(BasicPtxBuilderInterface op,
52  PatternRewriter &rewriter) const override {
53  if (op.hasIntrinsic()) {
54  LDBG() << "Ptx Builder does not lower \n\t" << op;
55  return failure();
56  }
57 
58  SmallVector<std::pair<Value, PTXRegisterMod>> asmValues;
59  LDBG() << op.getPtx();
60  PtxBuilder generator(op, rewriter);
61 
62  op.getAsmValues(rewriter, asmValues);
63  for (auto &[asmValue, modifier] : asmValues) {
64  LDBG() << asmValue << "\t Modifier : " << &modifier;
65  generator.insertValue(asmValue, modifier);
66  }
67 
68  generator.buildAndReplaceOp();
69  return success();
70  }
71 };
