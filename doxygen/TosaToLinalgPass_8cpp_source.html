31 struct TosaToLinalgOnTensors
32  : public TosaToLinalgOnTensorsBase<TosaToLinalgOnTensors> {
33 public:
34  void getDependentDialects(DialectRegistry &registry) const override {
35  registry.insert<linalg::LinalgDialect, math::MathDialect,
36  StandardOpsDialect, tensor::TensorDialect>();
37  }
38 
39  void runOnFunction() override {
40  RewritePatternSet patterns(&getContext());
41  ConversionTarget target(getContext());
42  target.addLegalDialect<linalg::LinalgDialect, StandardOpsDialect,
43  tensor::TensorDialect>();
44  target.addIllegalDialect<tosa::TosaDialect>();
45 
46  // Not every TOSA op can be legalized to linalg.
47  target.addLegalOp<tosa::ApplyScaleOp>();
48  target.addLegalOp<tosa::IfOp>();
49  target.addLegalOp<tosa::ConstOp>();
50  target.addLegalOp<tosa::WhileOp>();
51 
52  target.markUnknownOpDynamicallyLegal([](Operation *) { return true; });
53 
54  FuncOp func = getFunction();
55  mlir::tosa::populateTosaToLinalgOnTensorsConversionPatterns(&patterns);
56  if (failed(applyFullConversion(func, target, std::move(patterns))))
57  signalPassFailure();
58  }
59 };
60 } // namespace
