106 LogicalResult DataFlowSolver::initializeAndRun(Operation *top) {
107  // Enable enqueue to the worklist.
108  isRunning = true;
109  auto guard = llvm::make_scope_exit([&]() { isRunning = false; });
110 
111  // Initialize equivalent lattice anchors.
112  for (DataFlowAnalysis &analysis : llvm::make_pointee_range(childAnalyses)) {
113  analysis.initializeEquivalentLatticeAnchor(top);
114  }
115 
116  // Initialize the analyses.
117  for (DataFlowAnalysis &analysis : llvm::make_pointee_range(childAnalyses)) {
118  DATAFLOW_DEBUG(LDBG() << "Priming analysis: " << analysis.debugName);
119  if (failed(analysis.initialize(top)))
120  return failure();
121  }
122 
123  // Run the analysis until fixpoint.
124  // Iterate until all states are in some initialized state and the worklist
125  // is exhausted.
126  while (!worklist.empty()) {
127  auto [point, analysis] = worklist.front();
128  worklist.pop();
129 
130  DATAFLOW_DEBUG(LDBG() << "Invoking '" << analysis->debugName
131  << "' on: " << point);
132  if (failed(analysis->visit(point)))
133  return failure();
134  }
135 
136  return success();
137 }
