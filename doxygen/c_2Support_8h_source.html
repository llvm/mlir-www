49 /// A pointer to a sized fragment of a string, not necessarily null-terminated.
50 /// Does not own the underlying string. This is equivalent to llvm::StringRef.
51 
52 struct MlirStringRef {
53  const char *data; ///< Pointer to the first symbol.
54  size_t length; ///< Length of the fragment.
55 };
56 typedef struct MlirStringRef MlirStringRef;
57 
58 /// Constructs a string reference from the pointer and length. The pointer need
59 /// not reference to a null-terminated string.
60 
61 inline static MlirStringRef mlirStringRefCreate(const char *str,
62  size_t length) {
63  MlirStringRef result;
64  result.data = str;
65  result.length = length;
66  return result;
67 }
68 
69 /// Constructs a string reference from a null-terminated C string. Prefer
70 /// mlirStringRefCreate if the length of the string is known.
71 MLIR_CAPI_EXPORTED MlirStringRef
72 mlirStringRefCreateFromCString(const char *str);

86 /// A logical result value, essentially a boolean with named states. LLVM
87 /// convention for using boolean values to designate success or failure of an
88 /// operation is a moving target, so MLIR opted for an explicit class.
89 /// Instances of MlirLogicalResult must only be inspected using the associated
90 /// functions.
91 struct MlirLogicalResult {
92  int8_t value;
93 };
94 typedef struct MlirLogicalResult MlirLogicalResult;
95 
96 /// Checks if the given logical result represents a success.
97 inline static bool mlirLogicalResultIsSuccess(MlirLogicalResult res) {
98  return res.value != 0;
99 }
100 
101 /// Checks if the given logical result represents a failure.
102 inline static bool mlirLogicalResultIsFailure(MlirLogicalResult res) {
103  return res.value == 0;
104 }
105 
106 /// Creates a logical result representing a success.
107 inline static MlirLogicalResult mlirLogicalResultSuccess() {
108  MlirLogicalResult res = {1};
109  return res;
110 }
111 
112 /// Creates a logical result representing a failure.
113 inline static MlirLogicalResult mlirLogicalResultFailure() {
114  MlirLogicalResult res = {0};
115  return res;
116 }
