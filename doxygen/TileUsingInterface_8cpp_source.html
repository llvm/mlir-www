61  size_t iterationDomainSize) {
62  SmallVector<int64_t> filledVector = llvm::to_vector(interchangeVector);
63  if (filledVector.size() < iterationDomainSize) {
64  auto range = llvm::seq<int64_t>(filledVector.size(), iterationDomainSize);

77 static LogicalResult verifyOptions(RewriterBase &rewriter, Location loc,
78  const scf::SCFTilingOptions &options) {
79  // Specifying number of threads is only supported on `scf.forall` op.
80  if (options.numThreadsComputationFunction &&
81  options.loopType != scf::SCFTilingOptions::LoopType::ForallOp) {
82  return rewriter.notifyMatchFailure(
83  loc, "number of threads can only by specified when loop type is "
84  "set to use `scf.forall`");
85  }
86 
87  // If specified, check that the interchange vector is a permutation.
88  if (!options.interchangeVector.empty()) {
89  if (!isPermutationVector(options.interchangeVector)) {
90  return rewriter.notifyMatchFailure(
91  loc, "invalid interchange vector, not a permutation of the entire "
92  "iteration space");
93  }
94  }
95  return success();

98 /// Method to instantiate the tile sizes and/or number of threads specified
99 /// by the user.
100 static std::tuple<SmallVector<OpFoldResult>, SmallVector<OpFoldResult>>
101 getUserTileSizesAndNumThreads(RewriterBase &rewriter, TilingInterface op,
102  ArrayRef<Range> iterationDomain,
103  const scf::SCFTilingOptions &options) {
104  OpFoldResult zero = rewriter.getIndexAttr(0);
105  SmallVector<OpFoldResult> tileSizes, numThreads;
106  size_t numLoops = iterationDomain.size();
107 
108  // Check whether the number of tiles to use is specified.
109  if (options.numThreadsComputationFunction) {
110  numThreads = options.numThreadsComputationFunction(rewriter, op);
111  numThreads.resize(numLoops, zero);
112 
113  // If the number of tiles is also specified, use that.
114  if (options.tileSizeComputationFunction) {
115  tileSizes = options.tileSizeComputationFunction(rewriter, op);
116  tileSizes.resize(numLoops, zero);
117  return {tileSizes, numThreads};
118  }
119 
120  // Compute the tile sizes from the iteration domain and number
121  // of tiles as follows
122  // - niters = ceilDiv(ub - lb, step)
123  // - tileSize = ceilDiv(niters, numThreads)
124  AffineExpr s0, s1, s2;
125  bindSymbols(rewriter.getContext(), s0, s1, s2);
126  // TODO: The step here is assumed to be 1.
127  AffineExpr numItersExpr = (s1 - s0);
128  AffineExpr tileSizeExpr = numItersExpr.ceilDiv(s2);
129  tileSizes.resize(numLoops, zero);
130  for (auto [index, range, nt] :
131  llvm::enumerate(iterationDomain, numThreads)) {
132  if (isZeroInteger(nt))
133  continue;
134 
135  tileSizes[index] = affine::makeComposedFoldedAffineApply(
136  rewriter, op.getLoc(), tileSizeExpr, {range.offset, range.size, nt});
137  }
138  tileSizes.resize(numLoops, zero);
139  return {tileSizes, numThreads};
140  }
141 
142  // Enforce the convention that "tiling by zero"
143  // skips tiling a particular dimension. This convention is significantly
144  // simpler to handle instead of adjusting affine maps to account for missing
145  // dimensions.
146  assert(options.tileSizeComputationFunction &&
147  "expected tile sizes to be specified");
148  tileSizes = options.tileSizeComputationFunction(rewriter, op);
149  tileSizes.resize(numLoops, zero);
150 
151  return {tileSizes, numThreads};
152 }

154 /// Checks if any of the tiled loops are not parallel.
155 static LogicalResult checkTileSizes(TilingInterface op,
156  scf::SCFTilingOptions::LoopType loopType,
157  ReductionTilingStrategy reductionStrategy,
158  ArrayRef<OpFoldResult> tileSizes,
159  ArrayRef<OpFoldResult> numThreads) {
160  auto iterators = op.getLoopIteratorTypes();
161  assert(iterators.size() == tileSizes.size() &&
162  "expected as many tile size values as number of loops");
163  assert((numThreads.empty() || (numThreads.size() == iterators.size())) &&
164  "when specified, expected number of threads to use for each loop");
165 
166  bool isParallelTiling = false;
167  for (auto [index, iterator, tileSize] :
168  llvm::enumerate(iterators, tileSizes)) {
169  if (!isConstantIntValue(tileSize, 0)) {
170  isParallelTiling |= iterator == utils::IteratorType::parallel;
171  }
172 
173  if (loopType == scf::SCFTilingOptions::LoopType::ForallOp &&
174  reductionStrategy == ReductionTilingStrategy::FullReduction) {
175  // If num threads is specified, check that it is greater than one only for
176  // parallel dimensions.
177  if (!numThreads.empty()) {
178  if (std::optional<int64_t> constNumThreads =
179  getConstantIntValue(numThreads[index])) {
180  if (constNumThreads.value() > 1 &&
181  iterator != utils::IteratorType::parallel) {
182  op.emitWarning() << "tiling is not thread safe at axis #" << index;
183  }
184  }
185  continue;
186  }
187 
188  if (std::optional<int64_t> constTileSize =
189  getConstantIntValue(tileSize)) {
190  if (constTileSize.value() > 0 &&
191  iterator != utils::IteratorType::parallel) {
192  op.emitWarning() << "tiling is not thread safe at axis #" << index;
193  }
194  }
195  }
196  }
197 
198  if (reductionStrategy != ReductionTilingStrategy::FullReduction) {
199  if (isParallelTiling) {
200  return op->emitOpError("tiling parallel dimensions is not supported with "
201  "partial reduction tiling strategies");
202  }
203  }
204  return success();
205 }

221 /// Check if `stride` evenly divides the trip count `size - offset`.
222 static bool tileDividesIterationDomain(Range loopRange) {
223  std::optional<int64_t> offsetAsInt = getConstantIntValue(loopRange.offset);
224  if (!offsetAsInt)
225  return false;
226  std::optional<int64_t> sizeAsInt = getConstantIntValue(loopRange.size);
227  if (!sizeAsInt)
228  return false;
229  std::optional<int64_t> strideAsInt = getConstantIntValue(loopRange.stride);
230  if (!strideAsInt)
231  return false;
232  return ((sizeAsInt.value() - offsetAsInt.value()) % strideAsInt.value() == 0);
233 }

235 /// Returns the bounded tile size given the current `offset`, `loopRange` and
236 /// `tileSize`, i.e., `min(tileSize, range.end() - offset)`.
237 static OpFoldResult getBoundedTileSize(OpBuilder &b, Location loc,
238  Range loopRange, OpFoldResult offset,
239  OpFoldResult tileSize) {
240  std::optional<int64_t> ts = getConstantIntValue(tileSize);
241  if (ts && ts.value() == 1)
242  return tileSize;
243 
244  if (tileDividesIterationDomain(
245  Range{loopRange.offset, loopRange.size, tileSize}))
246  return tileSize;
247 
248  // The tile size to use (to avoid out of bounds access) is minimum of
249  // `tileSize` and `ub - iv`, where `iv` is the induction variable of the tiled
250  // loop.
251  AffineExpr s0, s1, d0;
252  bindDims(b.getContext(), d0);
253  bindSymbols(b.getContext(), s0, s1);
254  AffineMap minMap = AffineMap::get(1, 2, {s0 - d0, s1}, b.getContext());
255  Value size = getValueOrCreateConstantIndexOp(b, loc, loopRange.size);
256  return affine::makeComposedFoldedAffineMin(
257  b, loc, minMap, SmallVector<OpFoldResult>{offset, size, tileSize});
258 }

260 /// Returns true if the maximum tile offset `tileSize * numThreads-1` is less
261 /// than `iterationSize`.
262 static bool canOmitTileOffsetInBoundsCheck(OpFoldResult tileSize,
263  OpFoldResult numThreads,
264  OpFoldResult iterationSize) {
265  std::optional<int64_t> tileSizeConst = getConstantIntValue(tileSize);
266  std::optional<int64_t> numThreadsConst = getConstantIntValue(numThreads);
267  std::optional<int64_t> iterSizeConst = getConstantIntValue(iterationSize);
268  if (!tileSizeConst || !numThreadsConst || !iterSizeConst)
269  return false;
270  return *tileSizeConst * (*numThreadsConst - 1) < *iterSizeConst;
271 }

273 /// Compute the `OpFoldResult`s that represents the multi-dimensional
274 /// `offset`s and `size`s of the tile of the iteration space that the
275 /// innermost loop body of the generated tiled loops corresponds to.
276 static std::tuple<SmallVector<OpFoldResult>, SmallVector<OpFoldResult>>
277 getTileOffsetAndSizes(RewriterBase &rewriter, Location loc,
278  ReductionTilingStrategy strategy, ValueRange ivs,
279  ArrayRef<Range> iterationDomain,
280  ArrayRef<OpFoldResult> tileSizes,
281  ArrayRef<OpFoldResult> numThreads,
282  const llvm::SetVector<unsigned> &reductionDims) {
283  SmallVector<OpFoldResult> offsets, sizes;
284  int materializedLoopNum = 0;
285 
286  if (!numThreads.empty()) {
287  AffineExpr d0, d1, s0, s1;
288  AffineExpr offsetExpr, residualTileSizeExpr;
289  bindDims(rewriter.getContext(), d0, d1);
290  bindSymbols(rewriter.getContext(), s0, s1);
291  offsetExpr = d0 + d1 * s0;
292  residualTileSizeExpr = s1 - (d0 + d1 * s0);
293 
294  for (auto [index, nt, tileSize, loopRange] :
295  llvm::enumerate(numThreads, tileSizes, iterationDomain)) {
296 
297  // Non-tiled cases, set the offset and size to the
298  // `loopRange.offset/size`.
299  if (isZeroInteger(nt)) {
300  offsets.push_back(loopRange.offset);
301  sizes.push_back(loopRange.size);
302  continue;
303  }
304 
305  Value iv = ivs[materializedLoopNum++];
306  OpFoldResult offset = affine::makeComposedFoldedAffineApply(
307  rewriter, loc, offsetExpr,
308  ArrayRef<OpFoldResult>{loopRange.offset, iv, tileSize});
309  OpFoldResult residualTileSize = affine::makeComposedFoldedAffineApply(
310  rewriter, loc, residualTileSizeExpr,
311  {loopRange.offset, nt, tileSize, loopRange.size});
312 
313  OpFoldResult size = tileSize;
314  if (!isZeroInteger(residualTileSize)) {
315  OpFoldResult sizeMinusOffsetPerThread =
316  affine::makeComposedFoldedAffineApply(rewriter, loc, s0 - d0,
317  {offset, loopRange.size});
318  size = affine::makeComposedFoldedAffineMin(
319  rewriter, loc,
320  AffineMap::getMultiDimIdentityMap(2, rewriter.getContext()),
321  {sizeMinusOffsetPerThread, tileSize});
322  }
323 
324  // Consider the case where the original loop was `[0, 100)`.
325  // If number of threads are `7`, the tile size would be computed as
326  // `ceilDiv(100, 7) = 15`. For the last thread (thread_id = 6)
327  // - `offset = 0 + 6 * 15 = 105`
328  // - `tileSize = min(15, 100 - 105) = -5`
329  // To avoid negative tile sizes, we need to do a further
330  // `nonNegativeTileSize = affine.max(0, tileSize)`.
331  // This `max` can be avoided if
332  // `offset + tileSize * (numThreads - 1) < (ub - lb)`
333  if (!canOmitTileOffsetInBoundsCheck(tileSize, nt, loopRange.size)) {
334  AffineMap maxMap =
335  AffineMap::getMultiDimIdentityMap(2, rewriter.getContext());
336  size = affine::makeComposedFoldedAffineMax(
337  rewriter, loc, maxMap, {rewriter.getIndexAttr(0), size});
338  }
339 
340  offsets.push_back(offset);
341  sizes.push_back(size);
342  }
343  return {offsets, sizes};
344  } else {
345  for (auto [tileSize, loopRange] :
346  llvm::zip_equal(tileSizes, iterationDomain)) {
347 
348  // Non-tiled cases, set the offset and size to the
349  // `loopRange.offset/size`.
350  if (isZeroInteger(tileSize)) {
351  offsets.push_back(loopRange.offset);
352  sizes.push_back(loopRange.size);
353  continue;
354  }
355 
356  Value iv = ivs[materializedLoopNum++];
357  OpFoldResult offset = getAsOpFoldResult(iv);
358  offsets.push_back(offset);
359  OpFoldResult size =
360  getBoundedTileSize(rewriter, loc, loopRange, offset, tileSize);
361  sizes.push_back(size);
362  }
363  return {offsets, sizes};
364  }
365 }

418 /// Generate the tile-loop nest using `scf.for` operation.
419 /// - `loopRanges` specifies the lb, ub and step of the untiled iteration space.
420 /// - `tileSizes` is the tile sizes to use. Zero represent untiled loops.
421 /// - `destinationTensors` are the init values to use for the outer most loop.
422 /// - `yieldTiledValuesFn` is called to generated the loop body of the inner
423 /// most
424 /// loop.
425 /// - `loops` is an in-out parameter into which the generated loops are
426 /// populated.
427 static LogicalResult generateLoopNestUsingForOp(
428  RewriterBase &rewriter, Location loc, ArrayRef<Range> loopRanges,
429  ArrayRef<OpFoldResult> tileSizes, ValueRange destinationTensors,
430  YieldTiledValuesFn yieldTiledValuesFn,
431  SmallVector<LoopLikeOpInterface> &loops) {
432  assert(!loopRanges.empty() && "unexpected empty loop ranges");
433  assert(loopRanges.size() == tileSizes.size() &&
434  "expected as many tile sizes as loop ranges");
435  OpBuilder::InsertionGuard guard(rewriter);
436 
437  SmallVector<OpFoldResult> lbs, ubs, steps;
438  std::tie(lbs, ubs, steps) =
439  getLoopBounds(rewriter, loc, loopRanges, tileSizes);
440  SmallVector<Value> lbVals =
441  getValueOrCreateConstantIndexOp(rewriter, loc, lbs);
442  SmallVector<Value> ubVals =
443  getValueOrCreateConstantIndexOp(rewriter, loc, ubs);
444  SmallVector<Value> stepVals =
445  getValueOrCreateConstantIndexOp(rewriter, loc, steps);
446 
447  SmallVector<Value> ivs;
448  for (auto [lb, ub, step] : llvm::zip_equal(lbVals, ubVals, stepVals)) {
449  auto loop =
450  scf::ForOp::create(rewriter, loc, lb, ub, step, destinationTensors,
451  [](OpBuilder &bodyBuilder, Location bodyLoc,
452  Value iv, ValueRange /*iterArgs*/) {});
453  loops.push_back(loop);
454  ivs.push_back(loop.getInductionVar());
455  rewriter.setInsertionPointToEnd(loop.getBody());
456  destinationTensors = loop.getRegionIterArgs();
457  }
458 
459  SmallVector<Value> tiledResults;
460  SmallVector<SmallVector<OpFoldResult>> resultOffsets, resultSizes;
461  if (failed(yieldTiledValuesFn(rewriter, loc, ivs, destinationTensors,
462  tiledResults, resultOffsets, resultSizes))) {
463  return rewriter.notifyMatchFailure(
464  loc, "failed to generate inner tile loop body");
465  }
466  if (loops.empty())
467  return success();
468 
469  assert(tiledResults.size() == destinationTensors.size() &&
470  "Number of results of body should be equal to number of iter args");
471 
472  // 6. Yield all the results of the tiled operation.
473  SmallVector<Value> yieldedValues;
474  for (auto [tiledValue, destinationTensor, resultOffset, resultSize] :
475  llvm::zip_equal(tiledResults, destinationTensors, resultOffsets,
476  resultSizes)) {
477  SmallVector<OpFoldResult> resultStride(resultOffset.size(),
478  rewriter.getIndexAttr(1));
479  auto insertSlice = tensor::InsertSliceOp::create(
480  rewriter, loc, tiledValue, destinationTensor, resultOffset, resultSize,
481  resultStride);
482  yieldedValues.push_back(insertSlice);
483  }
484  scf::YieldOp::create(rewriter, loc, yieldedValues);
485 
486  // Add the scf.yield operations for all the outer loops.
487  for (auto [outerLoop, innerLoop] :
488  llvm::zip_equal(MutableArrayRef(loops).drop_back(),
489  MutableArrayRef(loops).drop_front())) {
490  rewriter.setInsertionPointToEnd(
491  cast<scf::ForOp>(outerLoop.getOperation()).getBody());
492  scf::YieldOp::create(rewriter, outerLoop.getLoc(), innerLoop->getResults());
493  }
494  return success();
495 }

497 /// Generate the tile-loop nest using `scf.forall` operation.
498 /// - `loopRanges` specifies the lb, ub and step of the untiled iteration space.
499 /// - `tileSizes` is the tile sizes to use. Zero represent untiled loops.
500 /// - `destinationTensors` are the init values to use for the outer most loop.
501 /// - `mappingVector` is the mapping attributes to use for loop construction.
502 /// Can be empty.
503 /// - `yieldTiledValuesFn` is called to generated the loop body of the inner
504 /// most
505 /// loop.
506 /// - `loops` is an in-out parameter into which the generated loops are
507 /// populated.
508 static LogicalResult generateLoopNestUsingForallOp(
509  RewriterBase &rewriter, Location loc, ArrayRef<Range> loopRanges,
510  ArrayRef<OpFoldResult> tileSizes, ArrayRef<OpFoldResult> numThreads,
511  ArrayRef<Attribute> mappingVector, ValueRange destinationTensors,
512  YieldTiledValuesFn tiledBodyFn, SmallVector<LoopLikeOpInterface> &loops) {
513  assert(!loopRanges.empty() && "unexpected empty loop ranges");
514  assert(loopRanges.size() == tileSizes.size() &&
515  "expected as many tile sizes as loop ranges");
516  OpBuilder::InsertionGuard guard(rewriter);
517 
518  std::optional<ArrayAttr> mappingAttr;
519  if (!mappingVector.empty())
520  mappingAttr = rewriter.getArrayAttr(mappingVector);
521 
522  scf::ForallOp forallOp;
523  bool useNumThreads = !numThreads.empty();
524 
525  if (useNumThreads) {
526  // Prune the zero numthreads.
527  SmallVector<OpFoldResult> nonZeroNumThreads;
528  for (auto nt : numThreads) {
529  if (isZeroInteger(nt))
530  continue;
531  nonZeroNumThreads.push_back(nt);
532  }
533  forallOp = scf::ForallOp::create(rewriter, loc, nonZeroNumThreads,
534  destinationTensors, mappingAttr);
535  } else {
536  SmallVector<OpFoldResult> lbs, ubs, steps;
537  std::tie(lbs, ubs, steps) =
538  getLoopBounds(rewriter, loc, loopRanges, tileSizes);
539  forallOp = scf::ForallOp::create(rewriter, loc, lbs, ubs, steps,
540  destinationTensors, mappingAttr);
541  }
542  loops.push_back(forallOp);
543 
544  rewriter.setInsertionPoint(forallOp.getTerminator());
545  destinationTensors = forallOp.getRegionOutArgs();
546 
547  SmallVector<Value> tiledResults;
548  SmallVector<SmallVector<OpFoldResult>> resultOffsets, resultSizes;
549  if (failed(tiledBodyFn(rewriter, loc, forallOp.getInductionVars(),
550  destinationTensors, tiledResults, resultOffsets,
551  resultSizes)))
552  return rewriter.notifyMatchFailure(loc, "failed to generate loop body");
553 
554  rewriter.setInsertionPointToEnd(forallOp.getTerminator().getBody());
555  for (auto [tiledValue, destinationTensor, resultOffset, resultSize] :
556  llvm::zip_equal(tiledResults, destinationTensors, resultOffsets,
557  resultSizes)) {
558  SmallVector<OpFoldResult> resultStride(resultOffset.size(),

604 static FailureOr<SmallVector<Value>> createInitialTensorsForTiling(
605  RewriterBase &rewriter, TilingInterface op,
606  ReductionTilingStrategy reductionStrategy, ArrayRef<Range> iterationDomain,
607  ArrayRef<OpFoldResult> numThreads, ArrayRef<OpFoldResult> tileSizes,
608  const SetVector<unsigned> &reductionDims) {
609  SmallVector<Value> initTensors;
610  Location loc = op->getLoc();
611  if (reductionStrategy == ReductionTilingStrategy::FullReduction) {
612  if (failed(tensor::getOrCreateDestinations(rewriter, loc, op, initTensors)))
613  return failure();
614  return initTensors;
615  }
616 
617  auto redOp = dyn_cast<PartialReductionOpInterface>(op.getOperation());
618  if (!redOp) {
619  return op->emitOpError(
620  "PartialReductionOuterReduction tiling strategy is only supported for "
621  "operations implementing PartialReductionOpInterface");
622  }
623  SmallVector<OpFoldResult> sizes(iterationDomain.size());
624  AffineExpr s0, s1, s2;
625  bindSymbols(rewriter.getContext(), s0, s1, s2);
626  AffineExpr sizeExpr = ((s0 - s1).ceilDiv(s2));
627  AffineExpr divExpr = s0.ceilDiv(s1);
628  for (auto [index, domain, tileSize] :
629  llvm::enumerate(iterationDomain, tileSizes)) {
630  if (!numThreads.empty()) {
631  // Untiled case.
632  if (isConstantIntValue(numThreads[index], 0)) {
633  sizes[index] = affine::makeComposedFoldedAffineApply(
634  rewriter, op.getLoc(), sizeExpr,
635  {domain.size, domain.offset, domain.stride});
636  continue;
637  }
638  sizes[index] = numThreads[index];
639  continue;
640  }
641 
642  // Non reduction dimensions/non-tiled dimensions.
643  if (!reductionDims.contains(index) || isConstantIntValue(tileSize, 0)) {
644  sizes[index] = affine::makeComposedFoldedAffineApply(
645  rewriter, op.getLoc(), sizeExpr,
646  {domain.size, domain.offset, domain.stride});
647  continue;
648  }
649 
650  if (reductionStrategy ==
651  ReductionTilingStrategy::PartialReductionOuterReduction) {
652  sizes[index] = tileSize;
653  continue;
654  }
655 
656  assert(reductionStrategy ==
657  ReductionTilingStrategy::PartialReductionOuterParallel);
658  OpFoldResult normalizedRange = affine::makeComposedFoldedAffineApply(
659  rewriter, op.getLoc(), sizeExpr,
660  {domain.size, domain.offset, domain.stride});
661  sizes[index] = affine::makeComposedFoldedAffineApply(
662  rewriter, op.getLoc(), divExpr, {normalizedRange, tileSize});
663  }
664  return redOp.generateInitialTensorForPartialReduction(rewriter, loc, sizes,
665  reductionDims);
666 }

668 /// For the case of `ReductionTilingStrategy::PartialReductionOuterParallel`
669 /// the `PartialReductionOpInterface` methods need the index of the parallel
670 /// split reduction being executed.
671 static SmallVector<OpFoldResult>
672 getSplitReductionIvs(RewriterBase &rewriter, Location loc,
673  ReductionTilingStrategy reductionStrategy, ValueRange ivs,
674  ArrayRef<OpFoldResult> numThreads,
675  ArrayRef<OpFoldResult> tileSizes,
676  const SetVector<unsigned> &reductionDims) {
677  SmallVector<OpFoldResult> splitReductionIvs;
678  splitReductionIvs.resize(reductionDims.size(), rewriter.getIndexAttr(0));
679  AffineExpr s0, s1;
680  bindSymbols(rewriter.getContext(), s0, s1);
681  AffineExpr divExpr = s0.floorDiv(s1);
682  int ivIndex = 0;
683  if (reductionStrategy ==
684  ReductionTilingStrategy::PartialReductionOuterParallel) {

687  splitReductionIvs[index] = ivs[ivIndex++];
688  continue;
689  }
690  splitReductionIvs[index] = affine::makeComposedFoldedAffineApply(
691  rewriter, loc, divExpr,
692  ArrayRef<OpFoldResult>{ivs[ivIndex++], tileSizes[reductionDim]});
693  }
694  }
695  return splitReductionIvs;

787 /// Implementation of `yieldTiledValuesAndReplaceLoop` for `scf.for`.
788 template <>
789 FailureOr<LoopLikeOpInterface> yieldTiledValuesAndReplaceLoop<scf::ForOp>(
790  scf::ForOp loopOp, RewriterBase &rewriter, ValueRange newInitOperands,
791  YieldTiledValuesFn yieldTiledValuesFn) {
792  OpBuilder::InsertionGuard g(rewriter);
793  Location loc = loopOp.getLoc();
794  rewriter.setInsertionPoint(loopOp);
795 
796  auto inits = llvm::to_vector(loopOp.getInitArgs());
797  inits.append(newInitOperands.begin(), newInitOperands.end());
798  auto newLoop = scf::ForOp::create(
799  rewriter, loc, loopOp.getLowerBound(), loopOp.getUpperBound(),
800  loopOp.getStep(), inits, [](OpBuilder &, Location, Value, ValueRange) {});
801 
802  // Move the loop body to the new op.
803  Block *loopBody = loopOp.getBody();
804  Block *newLoopBody = newLoop.getBody();
805  rewriter.mergeBlocks(
806  loopBody, newLoopBody,
807  newLoopBody->getArguments().take_front(loopBody->getNumArguments()));
808 
809  auto yieldOp = cast<scf::YieldOp>(newLoopBody->getTerminator());
810  rewriter.setInsertionPoint(yieldOp);
811 
812  SmallVector<Value> tiledValues;
813  SmallVector<SmallVector<OpFoldResult>> resultOffsets, resultSizes;
814  ValueRange newRegionIterArgs =
815  newLoop.getRegionIterArgs().take_back(newInitOperands.size());
816  if (failed(yieldTiledValuesFn(rewriter, loc, newLoop.getInductionVar(),
817  newRegionIterArgs, tiledValues, resultOffsets,
818  resultSizes))) {
819  rewriter.eraseOp(newLoop);
820  return rewriter.notifyMatchFailure(loopOp, "failed to get tiled values");
821  }
822 
823  SmallVector<Value> newYieldValues = llvm::to_vector(yieldOp.getOperands());
824  for (auto [tiledValue, regionIterArg, resultOffset, resultSize] :
825  llvm::zip_equal(tiledValues, newRegionIterArgs, resultOffsets,
826  resultSizes)) {
827  SmallVector<OpFoldResult> resultStride(resultOffset.size(),
828  rewriter.getIndexAttr(1));
829  Value insert = tensor::InsertSliceOp::create(
830  rewriter, yieldOp->getLoc(), tiledValue, regionIterArg, resultOffset,
831  resultSize, resultStride);
832  newYieldValues.push_back(insert);
833  }
834 
835  rewriter.replaceOpWithNewOp<scf::YieldOp>(yieldOp, newYieldValues);
836  rewriter.replaceOp(loopOp,
837  newLoop->getResults().take_front(loopOp.getNumResults()));
838  return cast<LoopLikeOpInterface>(newLoop.getOperation());
839 }

912 /// Method to add new init values to a loop nest. Updates `loops` in-place
913 /// with new loops that use the `newInitValues`. The outer-loops are updated
914 /// to yield the new result values of the inner loop. For the innermost loop,
915 /// the call back `getNewYields` is invoked to get the additional values to
916 /// yield form the innermost loop.
917 static LogicalResult addInitOperandsToLoopNest(
918  RewriterBase &rewriter, MutableArrayRef<LoopLikeOpInterface> loops,
919  ValueRange newInitValues, YieldTiledValuesFn getNewTiledYieldsFn) {
920  if (loops.empty())
921  return success();
922  OpBuilder::InsertionGuard g(rewriter);
923  rewriter.setInsertionPoint(loops.front());
924 
925  SmallVector<Value> ivs;
926  for (auto &loop : loops.drop_back()) {
927  rewriter.setInsertionPoint(loop);
928 
929  // if loops.size() > 1 we assume that scf.for is used for the loops.
930  auto forLoop = cast<scf::ForOp>(loop.getOperation());
931 
932  // Create a new loop with the new init values for this loop.
933  SmallVector<Value> newInits = llvm::to_vector(forLoop.getInitArgs());
934  newInits.append(newInitValues.begin(), newInitValues.end());
935  auto newLoop = scf::ForOp::create(
936  rewriter, forLoop.getLoc(), forLoop.getLowerBound(),
937  forLoop.getUpperBound(), forLoop.getStep(), newInits,
938  [&](OpBuilder &b, Location loc, Value iv, ValueRange iterArgs) {});
939 
940  // Merge the body of the new loop with the body of the old loops.
941  SmallVector<Value> sourceBlockArgs;
942  sourceBlockArgs.push_back(newLoop.getInductionVar());
943  auto newRegionIterArgs = newLoop.getRegionIterArgs();
944  sourceBlockArgs.append(
945  newRegionIterArgs.begin(),
946  std::next(newRegionIterArgs.begin(), forLoop.getNumResults()));
947  rewriter.mergeBlocks(forLoop.getBody(), newLoop.getBody(), sourceBlockArgs);
948  rewriter.replaceOp(
949  forLoop, newLoop.getResults().take_front(forLoop.getNumResults()));
950  loop = newLoop;
951  ivs.push_back(newLoop.getInductionVar());
952  newInitValues = newLoop.getRegionIterArgs().take_back(newInitValues.size());
953  }
954 
955  // Update the loop body of the innermost loop to get new yield values.
956  LoopLikeOpInterface innerMostLoop = loops.back();
957  FailureOr<LoopLikeOpInterface> newInnerMostLoop =
958  yieldTiledValuesAndReplaceLoop(innerMostLoop, rewriter, newInitValues,
959  getNewTiledYieldsFn);
960 
961  if (failed(newInnerMostLoop))
962  return innerMostLoop.emitOpError("failed to return additional yields");
963  loops.back() = newInnerMostLoop.value();
964 
965  // Make all other loops except the innermost loops yield the values returned
966  // by the inner loop.
967  for (auto [outerLoop, innerLoop] :
968  llvm::zip_equal(loops.drop_back(), loops.drop_front())) {
969  // Again assume that all the outer loops are scf.for operations.
970  auto outerForLoop = cast<scf::ForOp>(outerLoop);
971  auto outerLoopYield =
972  cast<scf::YieldOp>(outerForLoop.getBody()->getTerminator());
973  SmallVector<Value> newYields =
974  llvm::to_vector(outerLoopYield.getOperands());
975  ValueRange additionalYields =
976  innerLoop->getResults().take_back(newInitValues.size());
977  newYields.append(additionalYields.begin(), additionalYields.end());
978  rewriter.setInsertionPoint(outerLoopYield);
979  rewriter.replaceOpWithNewOp<scf::YieldOp>(outerLoopYield, newYields);
980  }
981  return success();
982 }

984 /// Implementation of tiling transformation of `op` that implements the
985 /// `TilingInterface` using `scf.for` to iterate over the tiles.
986 FailureOr<scf::SCFTilingResult>
987 mlir::scf::tileUsingSCF(RewriterBase &rewriter, TilingInterface op,
988  const scf::SCFTilingOptions &options) {
989  if (failed(verifyOptions(rewriter, op.getLoc(), options))) {
990  return failure();
991  }
992 
993  OpBuilder::InsertionGuard guard(rewriter);
994  rewriter.setInsertionPointAfter(op);
995 
996  // 1. Get the range of the loops that are represented by the operation.
997  SmallVector<Range> iterationDomain = op.getIterationDomain(rewriter);
998 
999  // 2. Materialize the tile sizes and/or number of threads;
1000  SmallVector<OpFoldResult> tileSizes, numThreads;
1001  std::tie(tileSizes, numThreads) =
1002  getUserTileSizesAndNumThreads(rewriter, op, iterationDomain, options);
1003 
1004  // Check if it is safe to tile. This is hold over from previous iterations
1005  // of tile to for-all. Consider dropping it.
1006  if (failed(checkTileSizes(op, options.loopType, options.reductionStrategy,
1007  tileSizes, numThreads))) {
1008  return failure();
1009  }
1010 
1011  // Get the reduction dims
1012  SetVector<unsigned> reductionDims =
1013  getSanitizedReductionDims(tileSizes, options);
1014 
1015  // 3. If there is an interchange specified, permute the iteration domain and
1016  // the tile sizes.
1017  SmallVector<int64_t> interchangeVector;
1018  if (!options.interchangeVector.empty()) {
1019  interchangeVector = fillInterchangeVector(options.interchangeVector,
1020  iterationDomain.size());
1021  assert(isPermutationVector(interchangeVector) &&
1022  "expected interchange vector to be a permutation");
1023 
1024  applyPermutationToVector(iterationDomain, interchangeVector);
1025  applyPermutationToVector(tileSizes, interchangeVector);
1026  if (!numThreads.empty())
1027  applyPermutationToVector(numThreads, interchangeVector);
1028  }
1029 
1030  FailureOr<TilingResult> tilingResult;
1031  // 4. Define the lambda function used later to generate the body of the
1032  // innermost tiled loop.
1033  YieldTiledValuesFn innerYieldTiledValuesFn =
1034  [&](RewriterBase &rewriter, Location loc, ValueRange ivs,
1035  ValueRange regionIterArgs, SmallVector<Value> &tiledResults,
1036  SmallVector<SmallVector<OpFoldResult>> &resultOffsets,
1037  SmallVector<SmallVector<OpFoldResult>> &resultSizes)
1038  -> LogicalResult {
1039  // 4a. Compute the `offsets` and `sizes` to use for tiling.
1040  SmallVector<OpFoldResult> offsets, sizes;
1041  std::tie(offsets, sizes) = getTileOffsetAndSizes(
1042  rewriter, loc, options.reductionStrategy, ivs, iterationDomain,
1043  tileSizes, numThreads, reductionDims);
1044 
1045  // 4b. If interchange was provided, apply inverse of the interchange
1046  // to get back the offsets/sizes in the order to be specified.
1047  if (!interchangeVector.empty()) {
1048  auto inversePermutation = invertPermutationVector(interchangeVector);
1049  applyPermutationToVector(offsets, inversePermutation);
1050  applyPermutationToVector(sizes, inversePermutation);
1051  }
1052 
1053  // 5. Generate the tiled implementation within the inner most loop.
1054 
1055  // 5a. Clone the operation within the loop body.
1056  auto clonedOp = cast<TilingInterface>(
1057  cloneOpAndUpdateDestinationArgs(rewriter, op, regionIterArgs));
1058 
1059  // 5b. Early return cloned op if tiling is not happening. We can not
1060  // return the original op because it could lead to `rewriter.replaceOp(op,
1061  // op->getResults())` and users would get crash.
1062  if (llvm::all_of(tileSizes, isZeroInteger)) {
1063  tiledResults.append(clonedOp->result_begin(), clonedOp->result_end());
1064  tilingResult =
1065  TilingResult{/*tiledOps=*/{clonedOp}, clonedOp->getResults(),
1066  /*generatedSlices=*/{}};
1067  return success();
1068  }
1069 
1070  // 5c. Tile the cloned operation.
1071  tilingResult = getTiledImplementation(
1072  rewriter, clonedOp, options.reductionStrategy, regionIterArgs, offsets,
1073  sizes, ivs, numThreads, tileSizes, reductionDims);
1074  if (failed(tilingResult)) {
1075  rewriter.eraseOp(clonedOp);
1076  return op.emitOpError("faild to tile operation");
1077  }
1078 
1079  // 5d. Delete the cloned operation.
1080  rewriter.eraseOp(clonedOp);
1081 
1082  // 5e. Compute the offsets at which the result values are to be inserted
1083  // back into its destinations.
1084  for (auto [index, tiledValue] :
1085  llvm::enumerate(tilingResult->tiledValues)) {
1086  tiledResults.push_back(tiledValue);
1087  SmallVector<OpFoldResult> resultOffset, resultSize;
1088  if (failed(getResultTilePosition(
1089  rewriter, options.reductionStrategy, index, tiledValue, op,
1090  offsets, sizes, ivs, numThreads, tileSizes, reductionDims,
1091  resultOffset, resultSize))) {
1092  for (auto op : tilingResult->tiledOps) {
1093  rewriter.eraseOp(op);
1094  }
1095  return rewriter.notifyMatchFailure(
1096  op, "failed to get slice of result produced");
1097  }
1098  resultOffsets.emplace_back(std::move(resultOffset));
1099  resultSizes.emplace_back(std::move(resultSize));
1100  }
1101 
1102  return success();
1103  };
1104 
1105  // 6. Find the destination tensors to use for the operation.
1106  FailureOr<SmallVector<Value>> maybeInits = createInitialTensorsForTiling(
1107  rewriter, op, options.reductionStrategy, iterationDomain, numThreads,
1108  tileSizes, reductionDims);
1109  if (failed(maybeInits)) {
1110  return rewriter.notifyMatchFailure(
1111  op, "unable to create initial tensors for tiling");
1112  }
1113  SmallVector<Value> &initTensors = maybeInits.value();
1114 
1115  // 7. Generate the tiled loops nest using the callback defined above.
1116  SmallVector<LoopLikeOpInterface> loops;
1117  if (failed(generateLoopNest(rewriter, op.getLoc(), options.loopType,
1118  iterationDomain, tileSizes, numThreads,
1119  initTensors, options.mappingVector,
1120  innerYieldTiledValuesFn, loops)))
1121  return op.emitOpError("failed to generate tiling loops");
1122  assert(succeeded(tilingResult) &&
1123  "expected tiling result to be computed after loop generation");
1124 
1125  if (loops.empty()) {
1126  // If loops are empty, the tiled op is used as the replacement for the
1127  // untiled op.
1128  return scf::SCFTilingResult{tilingResult->tiledOps,
1129  initTensors,
1130  loops,
1131  tilingResult->tiledValues,
1132  tilingResult->generatedSlices,
1133  {}};
1134  }
1135 
1136  auto loopResults = llvm::map_to_vector(loops.front()->getResults(),
1137  [](OpResult r) -> Value { return r; });
1138 
1139  // For the full reduction case, there is nothing more to do.
1140  if (options.reductionStrategy == ReductionTilingStrategy::FullReduction) {
1141  return scf::SCFTilingResult{
1142  tilingResult->tiledOps, initTensors, loops, loopResults,
1143  tilingResult->generatedSlices, {}};
1144  }
1145 
1146  // The results of the loop needs to be merged.
1147  FailureOr<MergeResult> mergeResult = mergeTilingResults(
1148  rewriter, op, options.reductionStrategy, reductionDims, loopResults);
1149  if (failed(mergeResult)) {
1150  return rewriter.notifyMatchFailure(
1151  op, "Failed to merge partial results from tiling");
1152  }
1153  return scf::SCFTilingResult{tilingResult->tiledOps,
1154  initTensors,
1155  loops,
1156  mergeResult->replacements,
1157  tilingResult->generatedSlices,
1158  mergeResult->mergeOps};
1159 }

1182 /// Return the untiled producer whose slice is used in a tiled consumer. The
1183 /// method traverses the tile loop nest (`loops`) if needed, and returns the
1184 /// `iter_args` of the outer most that is encountered. Traversing the
1185 /// iter_args indicates that this is a destination operand of the consumer. If
1186 /// there was no loop traversal needed, the second value of the returned tuple
1187 /// is empty.
1188 static std::tuple<OpResult, std::optional<OpOperand *>>
1189 getUntiledProducerFromSliceSource(OpOperand *source,
1190  ArrayRef<LoopLikeOpInterface> loops) {
1191  std::optional<OpOperand *> destinationIterArg;
1192  assert(!loops.empty() && "expected non empty loops container");
1193  auto loopIt = loops.rbegin();
1194  while (loopIt != loops.rend() && isa<BlockArgument>(source->get())) {
1195  auto iterArg = cast<BlockArgument>(source->get());
1196  auto loop = *loopIt;
1197  if (iterArg.getOwner()->getParentOp() != loop)
1198  break;
1199  source = loop.getTiedLoopInit(iterArg);
1200  loopIt++;
1201  }
1202  if (loopIt == loops.rend())
1203  destinationIterArg = source;
1204  return {dyn_cast<OpResult>(source->get()), destinationIterArg};
1205 }

1207 /// Implementation of fusing producer of a single slice by computing the
1208 /// slice of the producer in-place.
1209 std::optional<scf::SCFFuseProducerOfSliceResult>
1210 mlir::scf::tileAndFuseProducerOfSlice(
1211  RewriterBase &rewriter, tensor::ExtractSliceOp candidateSliceOp,
1212  MutableArrayRef<LoopLikeOpInterface> loops) {
1213  // 1. Get the producer of the source (potentially walking through
1214  // `iter_args` of nested `scf.for`)
1215  auto [fusableProducer, destinationInitArg] =
1216  getUntiledProducerFromSliceSource(&candidateSliceOp.getSourceMutable(),
1217  loops);
1218  if (!fusableProducer)
1219  return std::nullopt;
1220  unsigned resultNumber = fusableProducer.getResultNumber();
1221 
1222  OpBuilder::InsertionGuard g(rewriter);
1223  rewriter.setInsertionPoint(candidateSliceOp);
1224 
1225  // 2. Clone the fused producer
1226  // 2a. Compute the destination operands to use for the cloned operation.
1227  SmallVector<Value> origDestinationTensors, clonedOpDestinationTensors;
1228  Operation *fusableProducerOp = fusableProducer.getOwner();
1229  if (isa<DestinationStyleOpInterface>(fusableProducerOp) &&
1230  failed(tensor::getOrCreateDestinations(
1231  rewriter, fusableProducerOp->getLoc(), fusableProducerOp,
1232  origDestinationTensors)))
1233  return std::nullopt;
1234 
1235  clonedOpDestinationTensors = origDestinationTensors;
1236  if (destinationInitArg &&
1237  isa<DestinationStyleOpInterface>(fusableProducerOp)) {
1238  // 2b. If the producer is also destination style, then to maintain the
1239  // destination passing style, update the destination of the producer to be
1240  // the source of the slice.
1241  clonedOpDestinationTensors[resultNumber] = candidateSliceOp.getSource();
1242  }
1243  // 2c. Clone the fused producer.
1244  Operation *clonedProducerOp = cloneOpAndUpdateDestinationArgs(
1245  rewriter, fusableProducerOp, clonedOpDestinationTensors);
1246  // 2d. Update the source of the candidateSlice to be the cloned producer.
1247  // Easier to just clone the slice with different source since
1248  // replacements and DCE of cloned ops becomes easier
1249  SmallVector<Value> candidateSliceOpOperands =
1250  llvm::to_vector(candidateSliceOp->getOperands());
1251  candidateSliceOpOperands[0] = clonedProducerOp->getResult(resultNumber);
1252  tensor::ExtractSliceOp clonedCandidateSliceOp =
1253  mlir::clone(rewriter, candidateSliceOp,
1254  candidateSliceOp->getResultTypes(), candidateSliceOpOperands);
1255 
1256  // 3. Generate the tiled implementation of the producer of the source
1257  FailureOr<TilingResult> tileAndFuseResult =
1258  tensor::replaceExtractSliceWithTiledProducer(
1259  rewriter, clonedCandidateSliceOp,
1260  clonedProducerOp->getResult(resultNumber));
1261  if (failed(tileAndFuseResult))
1262  return std::nullopt;
1263  // Note: Do not delete the candidateSliceOp, since its passed in from the
1264  // caller.
1265  rewriter.replaceAllUsesWith(candidateSliceOp,
1266  tileAndFuseResult->tiledValues[0]);
1267  rewriter.eraseOp(clonedCandidateSliceOp);
1268  rewriter.eraseOp(clonedProducerOp);
1269 
1270  // 3. If the slice is for a destination operand, for example,
1271  //
1272  // ```mlir
1273  // %0 = linalg.init
1274  // %1 = linalg.fill .. outs(%0 : )
1275  // %2 = scf.for .. iter_args(%arg0 = %1) {
1276  // %3 = scf.for .. iter_args(%arg1 = %arg0) {
1277  // %4 = tensor.extract_slice %arg1 [..]
1278  // .. = linalg.matmul .. outs(%4 : )
1279  // }
1280  // }
1281  // ```
1282  //
1283  // the IR is currently
1284  //
1285  // ```
1286  // %0 = linalg.init
1287  // %1 = linalg.fill
1288  // %2 = scf.for .. iter_args(%arg0 = %1 /* incorrect value */ ) {
1289  // %3 = scf.for .. iter_args(%arg1 = %arg0) {
1290  // %4 = tensor.extract_slice %arg1[..]
1291  // %5 = linalg.fill .. outs(%4 : )
1292  // .. = linalg.matmul .. outs(%5 : )
1293  // }
1294  // }
1295  // ```
1296  //
1297  // The untiled `linalg.fill` is still used as the `init_value` since it
1298  // was originally a destination operand of the untiled `linalg.matmul`.
1299  // When fusing an operand that is a destination operand, the iter_arg of
1300  // the outer most loop should be changed to use the destination of the
1301  // fused operation. With this the IR will be.
1302  //
1303  // ```
1304  // %0 = linalg.init
1305  // %1 = scf.for .. iter_args(%arg0 = %0 /* corrected value */ ) {
1306  // %2 = scf.for .. iter_args(%arg1 = %arg0) {
1307  // %3 = tensor.extract_slice %arg1[..]
1308  // %4 = linalg.fill .. outs(%3 : )
1309  // .. = linalg.matmul .. outs(%4 : )
1310  // }
1311  // }
1312  // ```
1313  if (destinationInitArg &&
1314  isa<DestinationStyleOpInterface>(fusableProducerOp) && !loops.empty()) {
1315  loops.front()
1316  ->getOpOperands()[destinationInitArg.value()->getOperandNumber()]
1317  .set(origDestinationTensors[resultNumber]);
1318  }
1319  return scf::SCFFuseProducerOfSliceResult{
1320  fusableProducer, tileAndFuseResult->tiledValues[0],
1321  tileAndFuseResult->tiledOps, tileAndFuseResult->generatedSlices};
1322 }

1325 FailureOr<SmallVector<Operation *>> mlir::scf::yieldReplacementForFusedProducer(
1326  RewriterBase &rewriter, tensor::ExtractSliceOp sliceOp,

1334  *tiledOwner = fusedProducerInfo.tiledOps[0];
1335 
1336  Location loc = originalOwner->getLoc();
1337  // a. collect all init Value to be appended
1338  SmallVector<unsigned> initNumberList =
1339  yieldResultNumber.empty() ? llvm::to_vector(llvm::seq<unsigned>(
1340  0, originalOwner->getNumResults()))
1341  : llvm::to_vector(yieldResultNumber);
1342  SmallVector<Value> initValueList;
1343  for (const auto &resultNumber : initNumberList) {
1344  FailureOr<Value> initValue = tensor::getOrCreateDestination(
1345  rewriter, loc, originalOwner->getResult(resultNumber));
1346  if (succeeded(initValue)) {
1347  initValueList.push_back(initValue.value());
1348  } else {
1349  return failure();
1350  }
1351  }
1352 
1353  SmallVector<Operation *> generatedSlices;
1354  YieldTiledValuesFn newYieldValuesFn =
1355  [&](RewriterBase &innerRewriter, Location loc, ValueRange /*ivs*/,
1356  ValueRange newRegionIterArgs, SmallVector<Value> &tiledResult,
1357  SmallVector<SmallVector<OpFoldResult>> &tiledOffset,
1358  SmallVector<SmallVector<OpFoldResult>> &tiledSizes) -> LogicalResult {
1359  OpBuilder::InsertionGuard g(innerRewriter);
1360 
1361  // get sliceOp tile information
1362  SmallVector<OpFoldResult> sliceOffset = sliceOp.getMixedOffsets(),
1363  sliceSizes = sliceOp.getMixedSizes();
1364 
1365  // expect all strides of sliceOp being 1
1366  if (!llvm::all_of(sliceOp.getMixedStrides(), isOneInteger))
1367  return failure();

1370  fusedProducerInfo.origProducer.getResultNumber();
1371 
1372  auto tilableOp = cast<TilingInterface>(originalOwner);
1373  // b. get iterDomain Offset and Sizes based on sliceOp tile
1374  SmallVector<OpFoldResult> iterDomainOffset, iterDomainSizes;
1375  // skip tensor.pack/unpack/pad, which expects single opResult
1376  if (tilableOp->getNumResults() > 1 &&
1377  failed(tilableOp.getIterationDomainTileFromResultTile(
1378  rewriter, sliceResultNumber, sliceOffset, sliceSizes,
1379  iterDomainOffset, iterDomainSizes))) {
1380  // In theory, it is unnecessary to raise an error here. Actually
1381  // although it fails to reconstruct the result tensor, it should not
1382  // broke current fusion anyway. The reason why we must return failure
1383  // currently is that the callback function `newYieldValuesFn` will be
1384  // called after new init operand(s) has already been appended. It will
1385  // take more refactoring to make sure the init operands are added
1386  // consistently in the future. For more details, please refer to:
1387  // https://github.com/llvm/llvm-project/pull/93144#discussion_r1643760814
1388  return failure();
1389  }
1390

1392  // on iteration Domain Tile
1393  SmallVector<SmallVector<OpFoldResult>> offsetList, sizesList;
1394  for (const auto &resultNumber : initNumberList) {
1395  if (resultNumber == sliceResultNumber) {
1396  offsetList.push_back(sliceOffset);
1397  sizesList.push_back(sliceSizes);
1398  } else {
1399  assert(!iterDomainOffset.empty() && !iterDomainSizes.empty());
1400  // infer result tile according to the iteration domain tile
1401  SmallVector<OpFoldResult> offset, sizes;
1402  if (failed(tilableOp.getResultTilePosition(
1403  rewriter, resultNumber, iterDomainOffset, iterDomainSizes,
1404  offset, sizes))) {
1405  return failure();
1406  }
1407  offsetList.push_back(offset);
1408  sizesList.push_back(sizes);
1409  }
1410  }
1411 
1412  // d. create `extract_slice` for `iter_args` for DPS operation if
1413  // necessary
1414  if (auto tiledDestStyleOp =
1415  dyn_cast<DestinationStyleOpInterface>(tiledOwner)) {
1416  rewriter.setInsertionPoint(tiledDestStyleOp);
1417  for (const auto &&[index, newRegionArg] :
1418  llvm::enumerate(newRegionIterArgs)) {
1419  auto destSlice = tensor::ExtractSliceOp::create(
1420  rewriter, loc, newRegionArg, offsetList[index], sizesList[index],
1421  SmallVector<OpFoldResult>(offsetList[index].size(),
1422  rewriter.getIndexAttr(1)));
1423  generatedSlices.push_back(destSlice);
1424  unsigned resultNumber = initNumberList[index];
1425  rewriter.modifyOpInPlace(tiledDestStyleOp, [&]() {
1426  tiledDestStyleOp.getDpsInitsMutable()[resultNumber].set(destSlice);
1427  });
1428  }
1429  }
1430 
1431  // e. prepare tiled offset and sizes for later `insert_slice` creation by
1432  // caller
1433  Block *block = rewriter.getInsertionPoint()->getBlock();
1434  rewriter.setInsertionPoint(block->getTerminator());
1435  for (const auto &&[index, resultNumber] : llvm::enumerate(initNumberList)) {
1436  tiledResult.push_back(tiledOwner->getResult(resultNumber));
1437  tiledOffset.emplace_back(offsetList[index]);
1438  tiledSizes.emplace_back(sizesList[index]);
1439  }
1440  return success();
1441  };
1442 
1443  if (failed(addInitOperandsToLoopNest(rewriter, loops, initValueList,
1444  newYieldValuesFn))) {
1445  return failure();
1446  }
1447  return generatedSlices;
1448 }

1592 /// Implementation of tile consumer and fuse producer greedily.
1593 FailureOr<scf::SCFTileAndFuseResult>
1594 mlir::scf::tileConsumerAndFuseProducersUsingSCF(
1595  RewriterBase &rewriter, TilingInterface consumer,
1596  const scf::SCFTileAndFuseOptions &options) {
1597  // This transformation is only valid for ops that return values (i.e. not
1598  // valid to use with operations that have memref operands).
1599  if (!consumer->getNumResults()) {
1600  return rewriter.notifyMatchFailure(
1601  consumer, "invalid pattern for op with no results");
1602  }
1603 
1604  // 1. First tile the consumer.
1605  SetVector<Operation *> fusedProducers, tiledAndFusedOps;
1606 
1607  FailureOr<scf::SCFTilingResult> tilingResult =
1608  tileUsingSCF(rewriter, consumer, options.tilingOptions);
1609 
1610  if (failed(tilingResult))
1611  return rewriter.notifyMatchFailure(consumer, "failed to tile consumer");
1612  tiledAndFusedOps.insert_range(tilingResult->tiledOps);
1613 
1614  DenseMap<Value, Value> replacements;
1615  for (auto [origVal, replacement] :
1616  llvm::zip_equal(consumer->getResults(), tilingResult->replacements)) {
1617  replacements[origVal] = replacement;
1618  }
1619 
1620  // If there are no loops generated, fusion is immaterial.
1621  auto &loops = tilingResult->loops;
1622  if (loops.empty()) {
1623  return scf::SCFTileAndFuseResult{fusedProducers, tiledAndFusedOps, loops,
1624  replacements};
1625  }
1626 
1627  // Since the loop gets potentially replaced during fusion, we need to track
1628  // the mutation of replacement values. To do this, we attach a listener to
1629  // update the replacements as they happen.
1630  OpBuilder::Listener *previousListener = rewriter.getListener();
1631  auto resetListener =
1632  llvm::make_scope_exit([&]() { rewriter.setListener(previousListener); });
1633  ReplacementListener replaceListener(replacements, previousListener);
1634  rewriter.setListener(&replaceListener);
1635 
1636  // 2. Typically, the operands of the tiled operation are slices of the
1637  // operands of the untiled operation. These are expressed in IR using
1638  // `tensor.extract_slice` operations with source being the operands of
1639  // the untiled operation. Create a worklist of these
1640  // `tensor.extract_slice` operations. If the producers of the source of
1641  // the `tensor.extract_slice` can be tiled such that the tiled value is
1642  // generated in-place, that effectively tiles + fuses the operations.
1643  struct WorklistItem {
1644  tensor::ExtractSliceOp candidateSlice;
1645  SCFTileAndFuseOptions::ControlFnResult controlFnResult;
1646  };
1647 
1648  SliceTrackingListener sliceTracker =
1649  SliceTrackingListener(options.cleanupPatterns);
1650 
1651  if (failed(
1652  sliceTracker.insertAndApplyPatterns(tilingResult->generatedSlices))) {
1653  return rewriter.notifyMatchFailure(consumer, "cleanup patterns failed");
1654  }
1655  OpBuilder::InsertionGuard g(rewriter);
1656  while (!sliceTracker.worklist.empty()) {
1657  auto candidateSlice = sliceTracker.worklist.front();
1658  sliceTracker.worklist.pop_front();
1659 
1660  auto [fusableProducer, destinationInitArg] =
1661  getUntiledProducerFromSliceSource(&candidateSlice.getSourceMutable(),
1662  loops);
1663  if (!fusableProducer)
1664  continue;
1665 
1666  std::optional<SCFTileAndFuseOptions::ControlFnResult> controlFnResult =
1667  options.fusionControlFn(candidateSlice, fusableProducer,
1668  destinationInitArg.has_value());
1669  if (!controlFnResult)
1670  continue;
1671 
1672  WorklistItem worklistItem = {candidateSlice, controlFnResult.value()};
1673 
1674  // The operands of the fused producer might themselved be slices of
1675  // values produced by operations that implement the `TilingInterface`.
1676  // Add these operations to the worklist.
1677  std::optional<scf::SCFFuseProducerOfSliceResult> fusedResult =
1678  tileAndFuseProducerOfSlice(rewriter, worklistItem.candidateSlice,
1679  loops);
1680  if (!fusedResult)
1681  continue;
1682 
1683  SmallVector<Operation *> worklistCandidates = fusedResult->generatedSlices;
1684 
1685  if (worklistItem.controlFnResult.yieldProducerReplacement) {
1686  // Reconstruct and yield all opResult of fusableProducerOp by default.
1687  // The caller can specific which one to yield by designating optional
1688  // argument named `yieldResultNumber` of
1689  // `yieldReplacementForFusedProducer`.
1690  Operation *fusableProducerOp = fusedResult->origProducer.getOwner();
1691  FailureOr<SmallVector<Operation *>> newSlices =
1692  yieldReplacementForFusedProducer(rewriter,
1693  worklistItem.candidateSlice,
1694  fusedResult.value(), loops);
1695  if (failed(newSlices)) {
1696  return rewriter.notifyMatchFailure(
1697  fusableProducerOp, "failed to replacement value for this "
1698  "operation from within the tiled loop");
1699  }
1700  worklistCandidates.append(newSlices.value());
1701  for (auto [index, result] :
1702  llvm::enumerate(fusableProducerOp->getResults())) {
1703  replacements[result] = loops.front()->getResult(
1704  loops.front()->getNumResults() -
1705  fusableProducerOp->getNumResults() + index);
1706  }
1707  }
1708  if (Operation *tiledAndFusedOp =
1709  fusedResult->tiledAndFusedProducer.getDefiningOp()) {
1710  fusedProducers.insert(fusedResult->origProducer.getDefiningOp());
1711  tiledAndFusedOps.insert(tiledAndFusedOp);
1712  }
1713 
1714  if (failed(sliceTracker.insertAndApplyPatterns(worklistCandidates))) {
1715  return rewriter.notifyMatchFailure(consumer, "cleanup patterns failed");
1716  }
1717  }
1718 
1719  return scf::SCFTileAndFuseResult{fusedProducers, tiledAndFusedOps, loops,
1720  replacements};
1721 }

1753 /// An utility to get the first user of the given loopOp. If any of user stay
1754 /// in different block of loopOp, return failure.
1755 static FailureOr<Operation *> getFirstUserOfLoop(Operation *loopOp) {
1756  if (!isa<LoopLikeOpInterface>(loopOp))
1757  return failure();
1758  Operation *firstUserOfLoop = nullptr;
1759  for (Operation *userOp : loopOp->getUsers()) {
1760  // `ParallelInsertSlice` located inside `InParallelOp` has no same parent
1761  // block with any other types of operation. Thus, just redirecting to its
1762  // parent `InParallelOp`. E.g.
1763  //
1764  // ```
1765  // %1 = scf.for {
1766  // ...
1767  // }
1768  // %2 = consumerOp ins(%1, ...)
1769  // scf.forall.in_parallel {
1770  // tensor.parallel_insert_slice %1
1771  // }
1772  // ```
1773  // where `InParallelOp` but not `ParallelInsertSlice` stays in the same
1774  // same block with `consumerOp`.
1775  if (isa<tensor::ParallelInsertSliceOp>(userOp))
1776  userOp = userOp->getParentOfType<scf::InParallelOp>();
1777 
1778  if (loopOp->getBlock() != userOp->getBlock())
1779  return failure();
1780 
1781  if (!firstUserOfLoop || userOp->isBeforeInBlock(firstUserOfLoop))
1782  firstUserOfLoop = userOp;
1783  }
1784  return firstUserOfLoop;
1785 }

1787 /// This utility currently checks whether the first userOp of loop is NOT
1788 /// before the last defineOp of consumer operand. Because that we need to move
1789 /// the whole loop structure right before the `firstUserOfLoop`. This utility
1790 /// thus helps ensuring that no invalid IR is formed, i.e. no backward slice
1791 /// of consumerOp is dominated by the `firstUserOfLoop`. Saying that:
1792 ///
1793 /// ```
1794 /// %0 = scf.for() {
1795 /// ...
1796 /// }
1797 /// ...
1798 /// %1 = firstUserOfLoop(%0)
1799 /// ...
1800 /// %2 = lastDefOfConsumerOperand
1801 /// ...
1802 /// %3 = consumerOp(%2)
1803 /// ```
1804 ///
1805 /// If the `firstUserOfLoop` is before `lastDefOfConsumerOperand`, then it
1806 /// would be invalid to move the `loopOp` right before the `firstUserOfLoop`,
1807 /// a.k.a. use-def chain violation:
1808 ///
1809 /// ```
1810 /// %0:2 = scf.for() {
1811 /// // use before define error
1812 /// %3 = tiledConsumerOp(%2)
1813 /// }
1814 /// %1 = firstUserOfLoop(%0)
1815 /// ...
1816 /// %2 = lastDefOfConsumerOperand
1817 /// ```
1818 ///
1819 /// @param loopOp: loop operation
1820 /// @param consumerOp: consumer operation
1821 /// @param reorderOperations: the flag controls whether to reorder the
1822 /// backward slice w.r.t. the defineOp of `consumerOp` operands.
1823 /// @return: computed backward slice of consumerOp, but excluding those
1824 /// already dominates `firstUserOfLoop`.
1825 static FailureOr<llvm::SetVector<Operation *>>
1826 checkAssumptionForLoop(Operation *loopOp, Operation *consumerOp,
1827  bool reorderOperations) {
1828  FailureOr<Operation *> firstUserOfLoop = getFirstUserOfLoop(loopOp);
1829  if (failed(firstUserOfLoop))
1830  return failure();
1831 
1832  BackwardSliceOptions options;
1833  DominanceInfo dominanceInfo;
1834  options.inclusive = true;
1835  options.omitBlockArguments = true;
1836  bool includeLoopOp = false;
1837  options.filter = [&](Operation *op) {
1838  if (op == loopOp) {
1839  includeLoopOp = true;
1840  return false;
1841  }
1842  // Cut off the slice to not include any operation that already dominates
1843  // firstUserOfLoop.
1844  return !dominanceInfo.properlyDominates(op, *firstUserOfLoop);
1845  };
1846  llvm::SetVector<Operation *> slice;
1847  for (auto operand : consumerOp->getOperands()) {
1848  LogicalResult result = getBackwardSlice(operand, &slice, options);
1849  assert(result.succeeded() && "expected a backward slice");
1850  (void)result;
1851  }
1852 
1853  if (!slice.empty()) {
1854  // If consumerOp has one producer, which is also the user of loopOp.
1855  // E.g.
1856  // ```
1857  // %0 = %loopOp
1858  // %1 = consumerOp1 ins(%0)
1859  // %2 = consumerOp2 ins(%0, %1)
1860  // ```
1861  // We can not fuse consumerOp2 into loopOp due to UD chain, unless
1862  // consumerOp1 has already been fused into loopOp before.
1863  if (includeLoopOp || !reorderOperations)
1864  return failure();
1865  }
1866 
1867  return slice;
1868 }

1870 /// Fetches the OpOperand of the first valid user (and use) of the value `val`
1871 /// which implements `TilingInterface` and `DestinationStyleOpInterface`.
1872 /// Returns failure otherwise.
1873 static FailureOr<OpOperand *> getConsumerFromLoopUses(RewriterBase &rewriter,
1874  Operation *loopOp,
1875  unsigned resultNumber) {
1876  if (!isa<LoopLikeOpInterface>(loopOp))
1877  return failure();
1878  Value val = loopOp->getResult(resultNumber);
1879  Block *loopBlock = loopOp->getBlock();
1880  for (OpOperand &opOperand : val.getUses()) {
1881  Operation *consumerOp = opOperand.getOwner();
1882  // Step 1. Check if the user is tilable.
1883  if (!isa<TilingInterface>(consumerOp) ||
1884  !isa<DestinationStyleOpInterface>(consumerOp)) {
1885  // TODO: We have to init result of consumer before scf.for, use
1886  // DestinationStyleOpInterface to get result shape from init for now.
1887  // Add support for other op such as op has InferTypeOpInterface.
1888  continue;
1889  }
1890  // Step 2. Check if user stay in the same block.
1891  if (loopBlock != consumerOp->getBlock())
1892  continue;
1893  // Step 3. Check if user has succeeding user. Otherwise, it usually
1894  // represents already tiled.
1895  if (consumerOp->use_empty())
1896  continue;
1897  // Step 4. Check assumption for loop with `reorderOperations` enabled.
1898  FailureOr<llvm::SetVector<Operation *>> slice =
1899  checkAssumptionForLoop(loopOp, consumerOp, true);
1900  if (failed(slice))
1901  continue;
1902  // Step 5. If backward sice is not empty, move them before
1903  // firstUserOfLoop.
1904  if (!slice->empty()) {
1905  mlir::topologicalSort(*slice);
1906  FailureOr<Operation *> firstUserOfLoop = getFirstUserOfLoop(loopOp);
1907  assert(succeeded(firstUserOfLoop) && "First user of loop is not found");
1908  for (auto op : *slice) {
1909  rewriter.moveOpBefore(op, *firstUserOfLoop);
1910  }
1911  }
1912  return &opOperand;
1913  }
1914  return failure();
1915 }

1917 /// Check that the loop is perfectly nested.
1918 /// The loops are expected to be ordered from outer most to inner most.
1919 /// For example:
1920 /// ```
1921 /// %0 = scf.for()
1922 /// %1 = scf.for()
1923 /// %2 = scf.for()
1924 /// %3 = ...
1925 /// yield %3
1926 /// yield %2
1927 /// yield %1
1928 /// ```
1929 /// Here loops should be [%0, %1].
1930 static bool
1931 isPerfectlyNestedForLoops(MutableArrayRef<LoopLikeOpInterface> loops) {
1932  assert(!loops.empty() && "unexpected empty loop nest");
1933  if (loops.size() == 1) {
1934  return isa_and_nonnull<scf::ForOp>(loops.front().getOperation());
1935  }
1936  for (auto [outerLoop, innerLoop] :
1937  llvm::zip_equal(loops.drop_back(), loops.drop_front())) {
1938  auto outerFor = dyn_cast_or_null<scf::ForOp>(outerLoop.getOperation());
1939  auto innerFor = dyn_cast_or_null<scf::ForOp>(innerLoop.getOperation());
1940  if (!outerFor || !innerFor) {
1941  return false;
1942  }
1943  auto outerBBArgs = outerFor.getRegionIterArgs();
1944  auto innerIterArgs = innerFor.getInitArgs();
1945  if (outerBBArgs.size() != innerIterArgs.size()) {
1946  return false;
1947  }
1948 
1949  for (auto [outerBBArg, innerIterArg] :
1950  llvm::zip_equal(outerBBArgs, innerIterArgs)) {
1951  if (!llvm::hasSingleElement(outerBBArg.getUses()) ||
1952  innerIterArg != outerBBArg) {
1953  return false;
1954  }
1955  }
1956 
1957  ValueRange outerYields =
1958  cast<scf::YieldOp>(outerFor.getBody()->getTerminator())->getOperands();
1959  ValueRange innerResults = innerFor.getResults();
1960  if (outerYields.size() != innerResults.size()) {
1961  return false;
1962  }
1963  for (auto [outerYield, innerResult] :
1964  llvm::zip_equal(outerYields, innerResults)) {
1965  if (!llvm::hasSingleElement(innerResult.getUses()) ||
1966  outerYield != innerResult) {
1967  return false;
1968  }
1969  }
1970  }
1971  return true;
1972 }

1974 /// Fetch the untiled consumer of the outermost scf.for's result which is
1975 /// yielded by a tensor.insert_slice from the innermost scf.for. This function
1976 /// makes the following assumptions :
1977 /// 1. tensor.insert_slice has scf.yield as its only user.
1978 /// 2. scf.for's corresponding result has only one use.
1979 /// 3. The `loops` passed in are perfectly nested `scf.for` operations.
1980 static FailureOr<OpOperand *>
1981 getUntiledConsumerFromSlice(RewriterBase &rewriter,
1982  tensor::InsertSliceOp candidateSliceOp,
1983  MutableArrayRef<LoopLikeOpInterface> loops) {
1984  assert(!loops.empty() && "unexpected loops to be empty");
1985  // 1. Expect slice to be part of the body of the inner most loop.
1986  Operation *containingOp = candidateSliceOp->getParentOp();
1987  if (containingOp != loops.back()) {
1988  return rewriter.notifyMatchFailure(
1989  candidateSliceOp,
1990  "expected slice to be within body of inner-most loop");
1991  }
1992 
1993  // 2. Check that the loop is perfectly nested.
1994  if (!isPerfectlyNestedForLoops(loops)) {
1995  return rewriter.notifyMatchFailure(
1996  candidateSliceOp, "expected passed loops to be perfectly nested.");
1997  }
1998 
1999  if (failed(checkAssumptionForFusingConsumer(candidateSliceOp)))
2000  return failure();
2001  Value sliceResult = candidateSliceOp.getResult();
2002 
2003  // 3. Fetch the corresponding output.
2004  OpOperand &yieldOpOperand = (*sliceResult.getUses().begin());
2005  unsigned resultNumber = yieldOpOperand.getOperandNumber();
2006 
2007  scf::ForOp topLevelForOp = cast<scf::ForOp>(loops.front().getOperation());
2008 
2009  return getConsumerFromLoopUses(rewriter, topLevelForOp, resultNumber);
2010 }

2119 /// Implementation of fusing consumer of a single slice by computing the
2120 /// slice of the consumer in-place for scf loop.
2121 FailureOr<scf::SCFFuseConsumerOfSliceResult>
2122 mlir::scf::tileAndFuseConsumerOfSlices(
2123  RewriterBase &rewriter, ArrayRef<Operation *> candidateSlices,
2124  MutableArrayRef<LoopLikeOpInterface> loops) {
2125  if (candidateSlices.empty()) {
2126  return rewriter.notifyMatchFailure(
2127  rewriter.getUnknownLoc(),
2128  "no candidate slices provided for consumer fusion");
2129  }
2130  // Return if `loops` is empty, return an error for now. Caller is expected
2131  // to handle this case.
2132  if (loops.empty()) {
2133  return rewriter.notifyMatchFailure(
2134  candidateSlices.front(),
2135  "cannot call tile and fuse consumer with an empty loop nest");
2136  }
2137 
2138  if (!(llvm::all_of(candidateSlices, llvm::IsaPred<tensor::InsertSliceOp>) ||
2139  llvm::all_of(candidateSlices,
2140  llvm::IsaPred<tensor::ParallelInsertSliceOp>))) {
2141  return rewriter.notifyMatchFailure(
2142  candidateSlices.front(),
2143  "candidates slices need to be all `tensor.extract_slice`s or "
2144  "`tensor.parallel_insert_slice`s");
2145  }
2146 
2147  // 1. Get the consumer of scf.for for the result yielded by
2148  // tensor.insert_slice/parallel_insert_slice.
2149  SmallVector<OpOperand *> consumerOpOperands;
2150  Operation *consumerOp;
2151  {
2152  FailureOr<SmallVector<OpOperand *>> maybeConsumerOpOperand =
2153  getUntiledConsumerOperandsFromSlices(rewriter, candidateSlices, loops);
2154  if (failed(maybeConsumerOpOperand)) {
2155  return rewriter.notifyMatchFailure(candidateSlices.front(),
2156  "could not fetch consumer to fuse");
2157  }
2158  std::swap(consumerOpOperands, maybeConsumerOpOperand.value());
2159  consumerOp = consumerOpOperands.front()->getOwner();
2160  }
2161 
2162  LoopLikeOpInterface outerMostLoop = loops.front();
2163  LoopLikeOpInterface innerMostLoop = loops.back();
2164 
2165  // Check assumption for loop with `reorderOperations` disabled.
2166  if (failed(checkAssumptionForLoop(outerMostLoop, consumerOp, false))) {
2167  return rewriter.notifyMatchFailure(
2168  outerMostLoop, "the first user of loop should not dominate any define "
2169  "of consumer operand(s)");
2170  }
2171 
2172  OpBuilder::InsertionGuard g(rewriter);
2173 
2174  // 2. Check consumer is not using scf loop's output as init.
2175  auto dstOp = dyn_cast<DestinationStyleOpInterface>(consumerOp);
2176  if (!dstOp)
2177  return rewriter.notifyMatchFailure(consumerOp,
2178  "consumer op is not DPS operation");
2179  if (llvm::any_of(consumerOpOperands, [&](OpOperand *opOperand) {
2180  return dstOp.isDpsInit(opOperand);
2181  })) {
2182  return rewriter.notifyMatchFailure(
2183  consumerOp,
2184  "consumer op taking the result of scf.for as init is not supported");
2185  }
2186  SmallVector<Value> newInits = llvm::to_vector(dstOp.getDpsInits());
2187 
2188  // 3. Move the whole loop structure right before firstUserOfLoop, the
2189  // dominance should be already ensured by `checkAssumptionForLoop`.
2190  FailureOr<Operation *> firstUserOfLoop = getFirstUserOfLoop(outerMostLoop);
2191  if (failed(firstUserOfLoop)) {
2192  return rewriter.notifyMatchFailure(
2193  outerMostLoop, "could not find the first user of outer most loop");
2194  }
2195  rewriter.moveOpBefore(outerMostLoop, *firstUserOfLoop);
2196 
2197  // 4. Set insertion point before terminator op of the loop and create a new
2198  // tensor.insert_slice. In the scf.for case this is a clone of the
2199  // candidateSliceOp whereas in the scf.forall case this is created from the
2200  // operands of tensor.parallel_insert_slice.
2201  if (auto sliceOp =
2202  dyn_cast<tensor::ParallelInsertSliceOp>(candidateSlices.front())) {
2203  auto newForallOp = cast<scf::ForallOp>(innerMostLoop.getOperation());
2204  rewriter.setInsertionPoint(newForallOp.getTerminator());
2205  } else {
2206  rewriter.setInsertionPoint(candidateSlices.front());
2207  }
2208  // 5.a. Clone all the candidate slices as equivalent insert slice ops.
2209  SmallVector<tensor::InsertSliceOp> clonedInsertSlices =
2210  cloneAsInsertSlices(rewriter, candidateSlices);
2211 
2212  // 5.b. Clone consumer op.
2213  auto clonedConsumerOp = cast<TilingInterface>(rewriter.clone(*consumerOp));
2214  SmallVector<unsigned> operandNumbers =
2215  llvm::map_to_vector(consumerOpOperands, [](OpOperand *opOperand) {
2216  return opOperand->getOperandNumber();
2217  });
2218  SmallVector<OpOperand *> clonedOpFusedOperandsList =
2219  llvm::map_to_vector(operandNumbers, [&](unsigned operandNum) {
2220  return &clonedConsumerOp->getOpOperand(operandNum);
2221  });
2222 
2223  // 5.c. Replace all uses of the loop result with the result of the cloned
2224  // tensor.insert_slice.
2225  rewriter.modifyOpInPlace(clonedConsumerOp, [&]() {
2226  for (auto [operandToReplace, clonedSliceOp] :
2227  llvm::zip_equal(clonedOpFusedOperandsList, clonedInsertSlices)) {
2228  operandToReplace->set(clonedSliceOp.getResult());
2229  }
2230  });
2231 
2232  // 6. Perform tiling of the cloned consumer and replace the operand at
2233  // `operandNumber` with the source of the cloned tensor.insert_slice op.
2234  FailureOr<TilingResult> tileAndFuseResult =
