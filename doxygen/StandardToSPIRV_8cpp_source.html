53 /// Converts the given `srcAttr` to a new attribute of the given `dstType`.
54 /// Returns null attribute if conversion fails.
55 static IntegerAttr convertIntegerAttr(IntegerAttr srcAttr, IntegerType dstType,
56  Builder builder) {
57  // If the source number uses less active bits than the target bitwidth, then
58  // it should be safe to convert.
59  if (srcAttr.getValue().isIntN(dstType.getWidth()))
60  return builder.getIntegerAttr(dstType, srcAttr.getInt());
61 
62  // XXX: Try again by interpreting the source number as a signed value.
63  // Although integers in the standard dialect are signless, they can represent
64  // a signed number. It's the operation decides how to interpret. This is
65  // dangerous, but it seems there is no good way of handling this if we still
66  // want to change the bitwidth. Emit a message at least.
67  if (srcAttr.getValue().isSignedIntN(dstType.getWidth())) {
68  auto dstAttr = builder.getIntegerAttr(dstType, srcAttr.getInt());
69  LLVM_DEBUG(llvm::dbgs() << "attribute '" << srcAttr << "' converted to '"
70  << dstAttr << "' for type '" << dstType << "'\n");
71  return dstAttr;
72  }
73 
74  LLVM_DEBUG(llvm::dbgs() << "attribute '" << srcAttr
75  << "' illegal: cannot fit into target type '"
76  << dstType << "'\n");
77  return IntegerAttr();
78 }

80 /// Converts the given `srcAttr` to a new attribute of the given `dstType`.
81 /// Returns null attribute if `dstType` is not 32-bit or conversion fails.
82 static FloatAttr convertFloatAttr(FloatAttr srcAttr, FloatType dstType,
83  Builder builder) {
84  // Only support converting to float for now.
85  if (!dstType.isF32())
86  return FloatAttr();
87 
88  // Try to convert the source floating-point number to single precision.
89  APFloat dstVal = srcAttr.getValue();
90  bool losesInfo = false;
91  APFloat::opStatus status =
92  dstVal.convert(APFloat::IEEEsingle(), APFloat::rmTowardZero, &losesInfo);
93  if (status != APFloat::opOK || losesInfo) {
94  LLVM_DEBUG(llvm::dbgs()
95  << srcAttr << " illegal: cannot fit into converted type '"
96  << dstType << "'\n");
97  return FloatAttr();
98  }
99 
100  return builder.getF32FloatAttr(dstVal.convertToFloat());
101 }

103 /// Returns signed remainder for `lhs` and `rhs` and lets the result follow
104 /// the sign of `signOperand`.
105 ///
106 /// Note that this is needed for Vulkan. Per the Vulkan's SPIR-V environment
107 /// spec, "for the OpSRem and OpSMod instructions, if either operand is negative
108 /// the result is undefined." So we cannot directly use spv.SRem/spv.SMod
109 /// if either operand can be negative. Emulate it via spv.UMod.
110 static Value emulateSignedRemainder(Location loc, Value lhs, Value rhs,
111  Value signOperand, OpBuilder &builder) {
112  assert(lhs.getType() == rhs.getType());
113  assert(lhs == signOperand || rhs == signOperand);
114 
115  Type type = lhs.getType();
116 
117  // Calculate the remainder with spv.UMod.
118  Value lhsAbs = builder.create<spirv::GLSLSAbsOp>(loc, type, lhs);
119  Value rhsAbs = builder.create<spirv::GLSLSAbsOp>(loc, type, rhs);
120  Value abs = builder.create<spirv::UModOp>(loc, lhsAbs, rhsAbs);
121 
122  // Fix the sign.
123  Value isPositive;
124  if (lhs == signOperand)
125  isPositive = builder.create<spirv::IEqualOp>(loc, lhs, lhsAbs);
126  else
127  isPositive = builder.create<spirv::IEqualOp>(loc, rhs, rhsAbs);
128  Value absNegate = builder.create<spirv::SNegateOp>(loc, type, abs);
129  return builder.create<spirv::SelectOp>(loc, type, isPositive, abs, absNegate);
130 }

132 /// Returns the offset of the value in `targetBits` representation.
133 ///
134 /// `srcIdx` is an index into a 1-D array with each element having `sourceBits`.
135 /// It's assumed to be non-negative.
136 ///
137 /// When accessing an element in the array treating as having elements of
138 /// `targetBits`, multiple values are loaded in the same time. The method
139 /// returns the offset where the `srcIdx` locates in the value. For example, if
140 /// `sourceBits` equals to 8 and `targetBits` equals to 32, the x-th element is
141 /// located at (x % 4) * 8. Because there are four elements in one i32, and one
142 /// element has 8 bits.
143 static Value getOffsetForBitwidth(Location loc, Value srcIdx, int sourceBits,
144  int targetBits, OpBuilder &builder) {
145  assert(targetBits % sourceBits == 0);
146  IntegerType targetType = builder.getIntegerType(targetBits);
147  IntegerAttr idxAttr =
148  builder.getIntegerAttr(targetType, targetBits / sourceBits);
149  auto idx = builder.create<spirv::ConstantOp>(loc, targetType, idxAttr);
150  IntegerAttr srcBitsAttr = builder.getIntegerAttr(targetType, sourceBits);
151  auto srcBitsValue =
152  builder.create<spirv::ConstantOp>(loc, targetType, srcBitsAttr);
153  auto m = builder.create<spirv::UModOp>(loc, srcIdx, idx);
154  return builder.create<spirv::IMulOp>(loc, targetType, m, srcBitsValue);
155 }

157 /// Returns an adjusted spirv::AccessChainOp. Based on the
158 /// extension/capabilities, certain integer bitwidths `sourceBits` might not be
159 /// supported. During conversion if a memref of an unsupported type is used,
160 /// load/stores to this memref need to be modified to use a supported higher
161 /// bitwidth `targetBits` and extracting the required bits. For an accessing a
162 /// 1D array (spv.array or spv.rt_array), the last index is modified to load the
163 /// bits needed. The extraction of the actual bits needed are handled
164 /// separately. Note that this only works for a 1-D tensor.
165 static Value adjustAccessChainForBitwidth(SPIRVTypeConverter &typeConverter,
166  spirv::AccessChainOp op,
167  int sourceBits, int targetBits,
168  OpBuilder &builder) {
169  assert(targetBits % sourceBits == 0);
170  const auto loc = op.getLoc();
171  IntegerType targetType = builder.getIntegerType(targetBits);
172  IntegerAttr attr =
173  builder.getIntegerAttr(targetType, targetBits / sourceBits);
174  auto idx = builder.create<spirv::ConstantOp>(loc, targetType, attr);
175  auto lastDim = op->getOperand(op.getNumOperands() - 1);
176  auto indices = llvm::to_vector<4>(op.indices());
177  // There are two elements if this is a 1-D tensor.
178  assert(indices.size() == 2);
179  indices.back() = builder.create<spirv::SDivOp>(loc, lastDim, idx);
180  Type t = typeConverter.convertType(op.component_ptr().getType());
181  return builder.create<spirv::AccessChainOp>(loc, t, op.base_ptr(), indices);
182 }

193 /// Returns true if the allocations of type `t` can be lowered to SPIR-V.
194 static bool isAllocationSupported(MemRefType t) {
195  // Currently only support workgroup local memory allocations with static
196  // shape and int or float or vector of int or float element type.
197  if (!(t.hasStaticShape() &&
198  SPIRVTypeConverter::getMemorySpaceForStorageClass(
199  spirv::StorageClass::Workgroup) == t.getMemorySpaceAsInt()))
200  return false;
201  Type elementType = t.getElementType();
202  if (auto vecType = elementType.dyn_cast<VectorType>())
203  elementType = vecType.getElementType();
204  return elementType.isIntOrFloat();
205 }

237 /// Converts an allocation operation to SPIR-V. Currently only supports lowering
238 /// to Workgroup memory when the size is constant. Note that this pattern needs
239 /// to be applied in a pass that runs at least at spv.module scope since it wil
240 /// ladd global variables into the spv.module.
241 class AllocOpPattern final : public OpConversionPattern<memref::AllocOp> {
242 public:
243  using OpConversionPattern<memref::AllocOp>::OpConversionPattern;
244 
245  LogicalResult
246  matchAndRewrite(memref::AllocOp operation, ArrayRef<Value> operands,
247  ConversionPatternRewriter &rewriter) const override {
248  MemRefType allocType = operation.getType();
249  if (!isAllocationSupported(allocType))
250  return operation.emitError("unhandled allocation type");
251 
252  // Get the SPIR-V type for the allocation.
253  Type spirvType = getTypeConverter()->convertType(allocType);
254 
255  // Insert spv.GlobalVariable for this allocation.
256  Operation *parent =
257  SymbolTable::getNearestSymbolTable(operation->getParentOp());
258  if (!parent)
259  return failure();
260  Location loc = operation.getLoc();
261  spirv::GlobalVariableOp varOp;
262  {
263  OpBuilder::InsertionGuard guard(rewriter);
264  Block &entryBlock = *parent->getRegion(0).begin();
265  rewriter.setInsertionPointToStart(&entryBlock);
266  auto varOps = entryBlock.getOps<spirv::GlobalVariableOp>();
267  std::string varName =
268  std::string("__workgroup_mem__") +
269  std::to_string(std::distance(varOps.begin(), varOps.end()));
270  varOp = rewriter.create<spirv::GlobalVariableOp>(loc, spirvType, varName,
271  /*initializer=*/nullptr);
272  }
273 
274  // Get pointer to global variable at the current scope.
275  rewriter.replaceOpWithNewOp<spirv::AddressOfOp>(operation, varOp);
276  return success();
277  }
278 };

549 /// Converts tensor.extract into loading using access chains from SPIR-V local
550 /// variables.
551 class TensorExtractPattern final
552  : public OpConversionPattern<tensor::ExtractOp> {
553 public:
554  TensorExtractPattern(TypeConverter &typeConverter, MLIRContext *context,
555  int64_t threshold, PatternBenefit benefit = 1)
556  : OpConversionPattern(typeConverter, context, benefit),
557  byteCountThreshold(threshold) {}
558 
559  LogicalResult
560  matchAndRewrite(tensor::ExtractOp extractOp, ArrayRef<Value> operands,
561  ConversionPatternRewriter &rewriter) const override {
562  TensorType tensorType = extractOp.tensor().getType().cast<TensorType>();
563 
564  if (!tensorType.hasStaticShape())
565  return rewriter.notifyMatchFailure(extractOp, "non-static tensor");
566 
567  if (tensorType.getNumElements() * tensorType.getElementTypeBitWidth() >
568  byteCountThreshold * 8)
569  return rewriter.notifyMatchFailure(extractOp,
570  "exceeding byte count threshold");
571 
572  Location loc = extractOp.getLoc();
573  tensor::ExtractOp::Adaptor adaptor(operands);
574 
575  int64_t rank = tensorType.getRank();
576  SmallVector<int64_t, 4> strides(rank, 1);
577  for (int i = rank - 2; i >= 0; --i) {
578  strides[i] = strides[i + 1] * tensorType.getDimSize(i + 1);
579  }
580 
581  Type varType = spirv::PointerType::get(adaptor.tensor().getType(),
582  spirv::StorageClass::Function);
583 
584  spirv::VariableOp varOp;
585  if (adaptor.tensor().getDefiningOp<spirv::ConstantOp>()) {
586  varOp = rewriter.create<spirv::VariableOp>(
587  loc, varType, spirv::StorageClass::Function,

729 // TODO: This probably should be split into the vector case and tensor case,
730 // so that the tensor case can be moved to TensorToSPIRV conversion. But,
731 // std.constant is for the standard dialect though.
732 LogicalResult ConstantCompositeOpPattern::matchAndRewrite(
733  ConstantOp constOp, ArrayRef<Value> operands,
734  ConversionPatternRewriter &rewriter) const {
735  auto srcType = constOp.getType().dyn_cast<ShapedType>();
736  if (!srcType)
737  return failure();
738 
739  // std.constant should only have vector or tenor types.
740  assert((srcType.isa<VectorType, RankedTensorType>()));
741 
742  auto dstType = getTypeConverter()->convertType(srcType);
743  if (!dstType)
744  return failure();
745 
746  auto dstElementsAttr = constOp.value().dyn_cast<DenseElementsAttr>();
747  ShapedType dstAttrType = dstElementsAttr.getType();
748  if (!dstElementsAttr)
749  return failure();
750 
751  // If the composite type has more than one dimensions, perform linearization.
752  if (srcType.getRank() > 1) {
753  if (srcType.isa<RankedTensorType>()) {
754  dstAttrType = RankedTensorType::get(srcType.getNumElements(),
755  srcType.getElementType());
756  dstElementsAttr = dstElementsAttr.reshape(dstAttrType);
757  } else {
758  // TODO: add support for large vectors.
759  return failure();
760  }
761  }
762 
763  Type srcElemType = srcType.getElementType();
764  Type dstElemType;
765  // Tensor types are converted to SPIR-V array types; vector types are
766  // converted to SPIR-V vector/array types.
767  if (auto arrayType = dstType.dyn_cast<spirv::ArrayType>())
768  dstElemType = arrayType.getElementType();
769  else
770  dstElemType = dstType.cast<VectorType>().getElementType();
771 
772  // If the source and destination element types are different, perform
773  // attribute conversion.
774  if (srcElemType != dstElemType) {
775  SmallVector<Attribute, 8> elements;
776  if (srcElemType.isa<FloatType>()) {
777  for (Attribute srcAttr : dstElementsAttr.getAttributeValues()) {
778  FloatAttr dstAttr = convertFloatAttr(
779  srcAttr.cast<FloatAttr>(), dstElemType.cast<FloatType>(), rewriter);
780  if (!dstAttr)
781  return failure();
782  elements.push_back(dstAttr);
783  }
784  } else if (srcElemType.isInteger(1)) {
785  return failure();
786  } else {
787  for (Attribute srcAttr : dstElementsAttr.getAttributeValues()) {
788  IntegerAttr dstAttr =
789  convertIntegerAttr(srcAttr.cast<IntegerAttr>(),
790  dstElemType.cast<IntegerType>(), rewriter);
791  if (!dstAttr)
792  return failure();
793  elements.push_back(dstAttr);
794  }
795  }
796 
797  // Unfortunately, we cannot use dialect-specific types for element
798  // attributes; element attributes only works with builtin types. So we need
799  // to prepare another converted builtin types for the destination elements
800  // attribute.
801  if (dstAttrType.isa<RankedTensorType>())
802  dstAttrType = RankedTensorType::get(dstAttrType.getShape(), dstElemType);
803  else
804  dstAttrType = VectorType::get(dstAttrType.getShape(), dstElemType);
805 
806  dstElementsAttr = DenseElementsAttr::get(dstAttrType, elements);
807  }
808 
809  rewriter.replaceOpWithNewOp<spirv::ConstantOp>(constOp, dstType,
810  dstElementsAttr);
811  return success();
812 }

1021 LogicalResult
1022 IntLoadOpPattern::matchAndRewrite(memref::LoadOp loadOp,
1023  ArrayRef<Value> operands,
1024  ConversionPatternRewriter &rewriter) const {
1025  memref::LoadOpAdaptor loadOperands(operands);
1026  auto loc = loadOp.getLoc();
1027  auto memrefType = loadOp.memref().getType().cast<MemRefType>();
1028  if (!memrefType.getElementType().isSignlessInteger())
1029  return failure();
1030 
1031  auto &typeConverter = *getTypeConverter<SPIRVTypeConverter>();
1032  spirv::AccessChainOp accessChainOp =
1033  spirv::getElementPtr(typeConverter, memrefType, loadOperands.memref(),
1034  loadOperands.indices(), loc, rewriter);
1035 
1036  int srcBits = memrefType.getElementType().getIntOrFloatBitWidth();
1037  bool isBool = srcBits == 1;
1038  if (isBool)
1039  srcBits = typeConverter.getOptions().boolNumBits;
1040  Type pointeeType = typeConverter.convertType(memrefType)
1041  .cast<spirv::PointerType>()
1042  .getPointeeType();
1043  Type structElemType = pointeeType.cast<spirv::StructType>().getElementType(0);
1044  Type dstType;
1045  if (auto arrayType = structElemType.dyn_cast<spirv::ArrayType>())
1046  dstType = arrayType.getElementType();
1047  else
1048  dstType = structElemType.cast<spirv::RuntimeArrayType>().getElementType();
1049 
1050  int dstBits = dstType.getIntOrFloatBitWidth();
1051  assert(dstBits % srcBits == 0);
1052 
1053  // If the rewrited load op has the same bit width, use the loading value
1054  // directly.
1055  if (srcBits == dstBits) {
1056  rewriter.replaceOpWithNewOp<spirv::LoadOp>(loadOp,
1057  accessChainOp.getResult());
1058  return success();
1059  }
1060 
1061  // Assume that getElementPtr() works linearizely. If it's a scalar, the method
1062  // still returns a linearized accessing. If the accessing is not linearized,
1063  // there will be offset issues.
1064  assert(accessChainOp.indices().size() == 2);
1065  Value adjustedPtr = adjustAccessChainForBitwidth(typeConverter, accessChainOp,
1066  srcBits, dstBits, rewriter);
1067  Value spvLoadOp = rewriter.create<spirv::LoadOp>(
1068  loc, dstType, adjustedPtr,
1069  loadOp->getAttrOfType<spirv::MemoryAccessAttr>(
1070  spirv::attributeName<spirv::MemoryAccess>()),
1071  loadOp->getAttrOfType<IntegerAttr>("alignment"));
1072 
1073  // Shift the bits to the rightmost.
1074  // ____XXXX________ -> ____________XXXX
1075  Value lastDim = accessChainOp->getOperand(accessChainOp.getNumOperands() - 1);
1076  Value offset = getOffsetForBitwidth(loc, lastDim, srcBits, dstBits, rewriter);
1077  Value result = rewriter.create<spirv::ShiftRightArithmeticOp>(
1078  loc, spvLoadOp.getType(), spvLoadOp, offset);
1079 
1080  // Apply the mask to extract corresponding bits.
1081  Value mask = rewriter.create<spirv::ConstantOp>(
1082  loc, dstType, rewriter.getIntegerAttr(dstType, (1 << srcBits) - 1));
1083  result = rewriter.create<spirv::BitwiseAndOp>(loc, dstType, result, mask);
1084 
1085  // Apply sign extension on the loading value unconditionally. The signedness
1086  // semantic is carried in the operator itself, we relies other pattern to
1087  // handle the casting.
1088  IntegerAttr shiftValueAttr =
1089  rewriter.getIntegerAttr(dstType, dstBits - srcBits);
1090  Value shiftValue =
1091  rewriter.create<spirv::ConstantOp>(loc, dstType, shiftValueAttr);
1092  result = rewriter.create<spirv::ShiftLeftLogicalOp>(loc, dstType, result,
1093  shiftValue);
1094  result = rewriter.create<spirv::ShiftRightArithmeticOp>(loc, dstType, result,
1095  shiftValue);
1096 
1097  if (isBool) {
1098  dstType = typeConverter.convertType(loadOp.getType());
1099  mask = spirv::ConstantOp::getOne(result.getType(), loc, rewriter);
1100  Value isOne = rewriter.create<spirv::IEqualOp>(loc, result, mask);
1101  Value zero = spirv::ConstantOp::getZero(dstType, loc, rewriter);
1102  Value one = spirv::ConstantOp::getOne(dstType, loc, rewriter);
1103  result = rewriter.create<spirv::SelectOp>(loc, dstType, isOne, one, zero);
1104  } else if (result.getType().getIntOrFloatBitWidth() !=
1105  static_cast<unsigned>(dstBits)) {
1106  result = rewriter.create<spirv::SConvertOp>(loc, dstType, result);
1107  }
1108  rewriter.replaceOp(loadOp, result);
1109 
1110  assert(accessChainOp.use_empty());
1111  rewriter.eraseOp(accessChainOp);
1112 
1113  return success();
1114 }

1185  ArrayRef<Value> operands,
1186  ConversionPatternRewriter &rewriter) const {

1188  auto memrefType = storeOp.memref().getType().cast<MemRefType>();
1189  if (!memrefType.getElementType().isSignlessInteger())
1190  return failure();
1191 
1192  auto loc = storeOp.getLoc();
1193  auto &typeConverter = *getTypeConverter<SPIRVTypeConverter>();
1194  spirv::AccessChainOp accessChainOp =
1195  spirv::getElementPtr(typeConverter, memrefType, storeOperands.memref(),
1196  storeOperands.indices(), loc, rewriter);
1197  int srcBits = memrefType.getElementType().getIntOrFloatBitWidth();
1198 
1199  bool isBool = srcBits == 1;
1200  if (isBool)
1201  srcBits = typeConverter.getOptions().boolNumBits;
1202  Type pointeeType = typeConverter.convertType(memrefType)
1203  .cast<spirv::PointerType>()
1204  .getPointeeType();
1205  Type structElemType = pointeeType.cast<spirv::StructType>().getElementType(0);
1206  Type dstType;
1207  if (auto arrayType = structElemType.dyn_cast<spirv::ArrayType>())
1208  dstType = arrayType.getElementType();
1209  else
1210  dstType = structElemType.cast<spirv::RuntimeArrayType>().getElementType();
1211 
1212  int dstBits = dstType.getIntOrFloatBitWidth();
1213  assert(dstBits % srcBits == 0);
1214 
1215  if (srcBits == dstBits) {
1216  rewriter.replaceOpWithNewOp<spirv::StoreOp>(
1217  storeOp, accessChainOp.getResult(), storeOperands.value());
1218  return success();
1219  }
1220 
1221  // Since there are multi threads in the processing, the emulation will be done
1222  // with atomic operations. E.g., if the storing value is i8, rewrite the
1223  // StoreOp to
1224  // 1) load a 32-bit integer
1225  // 2) clear 8 bits in the loading value
1226  // 3) store 32-bit value back
1227  // 4) load a 32-bit integer
1228  // 5) modify 8 bits in the loading value
1229  // 6) store 32-bit value back
1230  // The step 1 to step 3 are done by AtomicAnd as one atomic step, and the step
1231  // 4 to step 6 are done by AtomicOr as another atomic step.
1232  assert(accessChainOp.indices().size() == 2);
1233  Value lastDim = accessChainOp->getOperand(accessChainOp.getNumOperands() - 1);
1234  Value offset = getOffsetForBitwidth(loc, lastDim, srcBits, dstBits, rewriter);
1235 
1236  // Create a mask to clear the destination. E.g., if it is the second i8 in
1237  // i32, 0xFFFF00FF is created.
1238  Value mask = rewriter.create<spirv::ConstantOp>(
1239  loc, dstType, rewriter.getIntegerAttr(dstType, (1 << srcBits) - 1));
1240  Value clearBitsMask =
1241  rewriter.create<spirv::ShiftLeftLogicalOp>(loc, dstType, mask, offset);
1242  clearBitsMask = rewriter.create<spirv::NotOp>(loc, dstType, clearBitsMask);
1243 
1244  Value storeVal = storeOperands.value();
1245  if (isBool) {
1246  Value zero = spirv::ConstantOp::getZero(dstType, loc, rewriter);
1247  Value one = spirv::ConstantOp::getOne(dstType, loc, rewriter);
1248  storeVal =
1249  rewriter.create<spirv::SelectOp>(loc, dstType, storeVal, one, zero);
1250  }
1251  storeVal = shiftValue(loc, storeVal, offset, mask, dstBits, rewriter);
1252  Value adjustedPtr = adjustAccessChainForBitwidth(typeConverter, accessChainOp,
1253  srcBits, dstBits, rewriter);
1254  Optional<spirv::Scope> scope = getAtomicOpScope(memrefType);
1255  if (!scope)
1256  return failure();
1257  Value result = rewriter.create<spirv::AtomicAndOp>(
1258  loc, dstType, adjustedPtr, *scope, spirv::MemorySemantics::AcquireRelease,
1259  clearBitsMask);
1260  result = rewriter.create<spirv::AtomicOrOp>(
1261  loc, dstType, adjustedPtr, *scope, spirv::MemorySemantics::AcquireRelease,
1262  storeVal);
1263 
1264  // The AtomicOrOp has no side effect. Since it is already inserted, we can
1265  // just remove the original StoreOp. Note that rewriter.replaceOp()
1266  // doesn't work because it only accepts that the numbers of result are the
1267  // same.
1268  rewriter.eraseOp(storeOp);
1269 
1270  assert(accessChainOp.use_empty());
1271  rewriter.eraseOp(accessChainOp);
1272 
1273  return success();
1274 }
