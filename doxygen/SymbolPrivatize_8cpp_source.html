40 LogicalResult SymbolPrivatize::initialize(MLIRContext *context) {

46 void SymbolPrivatize::runOnOperation() {
47  for (Region &region : getOperation()->getRegions()) {
48  for (Block &block : region) {
49  for (Operation &op : block) {
50  auto symbol = dyn_cast<SymbolOpInterface>(op);
51  if (!symbol)
52  continue;
53  if (!excludedSymbols.contains(symbol.getNameAttr()))
54  symbol.setVisibility(SymbolTable::Visibility::Private);
55  }
56  }
57  }
58 }
