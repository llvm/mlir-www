53 /// Attempts to apply the pattern specified as template argument to the given
54 /// operation. The pattern is expected to have a `returningMatchAndRewrite`
55 /// function that returns the "main" result or failure. Returns failure if the
56 /// pattern failed to apply. Extra arguments are forwarded to the pattern
57 /// constructor.
58 template <typename PatternTy, typename... Args>
59 static FailureOr<LinalgOp> tryApply(Operation *operation, Args &&...args) {
60  // Check if the given operation has the type expected by the pattern.
61  using OpTy = typename llvm::function_traits<
62  decltype(&PatternTy::returningMatchAndRewrite)>::template arg_t<0>;
63  auto op = dyn_cast<OpTy>(operation);
64  if (!op)
65  return failure();
66 
67  // Apply the pattern directly to the op.
68  PatternTy pattern(operation->getContext(), std::forward<Args>(args)...);
69  // We want to discourage direct use of PatternRewriter in APIs but In this
70  // very specific case, an IRRewriter is not enough.
71  struct TrivialPatternRewriter : public PatternRewriter {
72  public:
73  explicit TrivialPatternRewriter(MLIRContext *context)
74  : PatternRewriter(context) {}
75  };
76  TrivialPatternRewriter rewriter(operation->getContext());
77  rewriter.setInsertionPoint(operation);
78  auto result = pattern.returningMatchAndRewrite(op, rewriter);
79  if (failed(result))
80  return failure();
81  return cast<LinalgOp>(result->getOperation());
82 }

84 /// Assuming that `ofr` is an index attr or a param of index type
85 /// or a transform dialect handle mapped to exactly one op
86 /// with one index result, return that value.
87 static DiagnosedSilenceableFailure unpackSingleIndexResultPayloadOperations(
88  transform::TransformState &state, TransformOpInterface transformOp,
89  SmallVector<OpFoldResult> &result, ArrayRef<OpFoldResult> ofrs) {
90  for (OpFoldResult ofr : ofrs) {
91  if (auto attr = dyn_cast<Attribute>(ofr)) {
92  if (!isa<IntegerAttr>(attr))
93  return transformOp.emitDefiniteFailure() << "expected IntegerAttr";
94  result.push_back(ofr);
95  continue;
96  }
97 
98  Value transformValue = cast<Value>(ofr);
99  if (isa<TransformParamTypeInterface>(transformValue.getType())) {
100  ArrayRef<Attribute> params = state.getParams(transformValue);
101  if (params.size() != 1)
102  return transformOp.emitDefiniteFailure()
103  << "requires exactly one parameter associated";
104  result.push_back(params[0]);
105  continue;
106  }
107 
108  auto payloadOps = state.getPayloadOps(transformValue);
109  if (!llvm::hasSingleElement(payloadOps)) {
110  DiagnosedSilenceableFailure diag =
111  transformOp.emitSilenceableError()
112  << "handle must be mapped to exactly one payload op";
113  diag.attachNote(transformValue.getLoc())
114  << "mapped to " << llvm::range_size(payloadOps) << " payload ops";
115  return diag;
116  }
117 
118  Operation *op = *payloadOps.begin();
119  if (op->getNumResults() != 1 || !op->getResult(0).getType().isIndex()) {
120  DiagnosedSilenceableFailure diag =
121  transformOp.emitSilenceableError()
122  << "payload op must have exactly 1 index result";
123  diag.attachNote(op->getLoc())
124  << "has " << op->getNumResults() << " results";
125  return diag;
126  }
127  result.push_back(op->getResult(0));
128  }
129 
130  return DiagnosedSilenceableFailure::success();
131 }

133 // Given a list of params that are index attrs or a list of OpFoldResults
134 // that are either index attrs or op handles, return a list of OpFoldResults
135 // of index attrs or a list of OpFoldResults where all op handles are
136 // replaced with the first (and only) OpResult of that payload op.
137 // (There must be exactly one parameter associated with the AnyParamType or
138 // one mapped payload op which must have exactly one index result.)
139 static DiagnosedSilenceableFailure unpackSingleIndexResultPayloadOperations(
140  transform::TransformState &state, TransformOpInterface transformOp,
141  SmallVector<OpFoldResult> &result, Value packedHandle) {
142  if (isa<TransformParamTypeInterface>(packedHandle.getType())) {
143  ArrayRef<Attribute> params = state.getParams(packedHandle);
144  for (auto param : params) {

163  result.push_back(op->getResult(0));

169 /// When possible, converts each `OpFoldResult` in `mixedResult` to
170 /// an integer if the value can be statically inferred. If a result
171 /// is a `Value` then it must be either a `ParamType` or a handle
172 /// to an a constant like op.
173 static DiagnosedSilenceableFailure reifyMixedParamAndHandleResults(
174  TransformState &state, TransformOpInterface &transformOp,
175  ArrayRef<OpFoldResult> mixedResults, SmallVectorImpl<int64_t> &reified) {
176  for (OpFoldResult paramOrHandle : mixedResults) {
177  if (auto attr = dyn_cast<Attribute>(paramOrHandle)) {
178  reified.push_back(cast<IntegerAttr>(attr).getInt());
179  continue;
180  } else if (isa<ParamType>(cast<Value>(paramOrHandle).getType())) {
181  ArrayRef<Attribute> params = state.getParams(cast<Value>(paramOrHandle));
182  if (params.size() != 1)
183  return transformOp.emitSilenceableError() << "expected a single param";
184  reified.push_back(
185  cast<IntegerAttr>(params.front()).getValue().getSExtValue());
186  continue;
187  }
188 
189  Value handle = cast<Value>(paramOrHandle);
190  if (!isa<TransformHandleTypeInterface>(handle.getType()))
191  return transformOp.emitSilenceableError() << "unexpected value handle";
192  auto payload = state.getPayloadOps(handle);
193  if (!llvm::hasSingleElement(payload))
194  return transformOp.emitSilenceableError()
195  << "requires param or handle that is mapped to 1 payload op";
196 
197  Operation *paramOrHandlePayloadOp = *payload.begin();
198  if (paramOrHandlePayloadOp->getNumResults() != 1 ||
199  !paramOrHandlePayloadOp->getResult(0).getType().isIndex()) {
200  return transformOp.emitSilenceableError()
201  << "requires param or handle to be result of op with 1 index "
202  "result";
203  }
204 
205  IntegerAttr attr;
206  if (!matchPattern(paramOrHandlePayloadOp->getResult(0), m_Constant(&attr)))
207  return transformOp.emitSilenceableError()
208  << "requires param or handle to be the result of a constant like "
209  "op";
210 
211  reified.push_back(attr.getInt());
212  }
213  return DiagnosedSilenceableFailure::success();
214 }

334 DiagnosedSilenceableFailure transform::BufferizeToAllocationOp::apply(
335  transform::TransformRewriter &rewriter,
336  transform::TransformResults &results, transform::TransformState &state) {
337  // Attach listener to keep track of newly created ops.
338  OpBuilder::Listener *previousListener = rewriter.getListener();
339  auto resetListener =
340  llvm::make_scope_exit([&]() { rewriter.setListener(previousListener); });
341  NewOpsListener newOpsListener(previousListener);
342  rewriter.setListener(&newOpsListener);
343 
344  linalg::BufferizeToAllocationOptions options;
345  if (getMemcpyOp() == "bufferization.materialize_in_destination") {
346  options.memcpyOp = linalg::BufferizeToAllocationOptions::MemcpyOp::
347  MaterializeInDestination;
348  } else if (getMemcpyOp() == "memref.copy") {
349  options.memcpyOp =
350  linalg::BufferizeToAllocationOptions::MemcpyOp::MemrefCopy;
351  } else if (getMemcpyOp() == "linalg.copy") {
352  options.memcpyOp =
353  linalg::BufferizeToAllocationOptions::MemcpyOp::LinalgCopy;
354  } else {
355  llvm_unreachable("invalid memcpy op");
356  }
357  if (getAllocOp() == "memref.alloc") {
358  options.allocOp =
359  linalg::BufferizeToAllocationOptions::AllocOp::MemrefAlloc;
360  } else if (getAllocOp() == "memref.alloca") {
361  options.allocOp =
362  linalg::BufferizeToAllocationOptions::AllocOp::MemrefAlloca;
363  } else {
364  llvm_unreachable("invalid alloc op");
365  }
366  options.bufferizeDestinationOnly = getBufferizeDestinationOnly();
367  options.emitDealloc = getEmitDealloc();
368 
369  // Bufferize ops.
370  Attribute memorySpace =
371  getMemorySpace().has_value() ? getMemorySpace().value() : Attribute();
372  SmallVector<Value> allocatedBuffers;
373  for (Operation *op : state.getPayloadOps(getTarget())) {
374  Value buffer =
375  linalg::bufferizeToAllocation(rewriter, options, op, memorySpace);
376  if (!buffer) {
377  DiagnosedSilenceableFailure diag = emitSilenceableError()
378  << "failed to bufferize operation";
379  diag.attachNote(op->getLoc()) << "target payload op";
380  return diag;
381  }
382  allocatedBuffers.push_back(buffer);
383  }
384 
385  // Set results.
386  results.setValues(cast<OpResult>(getAllocatedBuffer()), allocatedBuffers);
387  results.set(cast<OpResult>(getNewOps()), newOpsListener.getNewOps());
388  return DiagnosedSilenceableFailure::success();
389 }

456 // Push the decomposed operations (the ones that replaces the values produced by

461  transform::TransformState &state) {
462  auto decomposableOp = dyn_cast<AggregatedOpInterface>(target);
463  if (!decomposableOp) {
464  failed(rewriter.notifyMatchFailure(target,
465  "payload is not a decomposable op"));
466  return emitDefaultSilenceableFailure(target);

469  FailureOr<SmallVector<Value>> maybeNewResults =
470  decomposableOp.decomposeOperation(rewriter);
471  if (failed(maybeNewResults))
472  return emitDefaultSilenceableFailure(target);

475  for (Value val : *maybeNewResults) {
476  Operation *definition = val.getDefiningOp();
477  if (definition)
478  results.push_back(definition);
479  }
480  return DiagnosedSilenceableFailure::success();
481 }

517 /// Apply a tiling transformation to all payload ops and store both the
518 /// tiled operation as well as the created tile loops.
519 template <typename Range>
520 static LogicalResult applyTilingToAll(
521  RewriterBase &rewriter, Operation *transformOp, Range &&payloadOps,
522  unsigned numLoops, transform::TransformResults &transformResults,
523  function_ref<FailureOr<scf::SCFTileAndFuseResult>(TilingInterface)>
524  applyFn) {
525  SmallVector<Operation *> tiledLinalgOps;
526  SmallVector<SmallVector<Operation *>> loopOps(numLoops);
527 
528  for (Operation *target : payloadOps) {
529  auto tilingInterfaceOp = dyn_cast<TilingInterface>(target);
530  if (!tilingInterfaceOp)
531  return transformOp->emitError("only TilingInterface ops are supported");
532 
533  rewriter.setInsertionPoint(target);
534  FailureOr<scf::SCFTileAndFuseResult> tiledResults =
535  applyFn(tilingInterfaceOp);
536  if (failed(tiledResults))
537  return failure();
538 
539  // Perform the replacement of tiled and fused values.
540  SmallVector<Operation *> opsToReplace{target};
541  llvm::append_range(opsToReplace, tiledResults->fusedProducers);
542  for (Operation *toReplace : opsToReplace) {
543  for (OpResult res : toReplace->getResults())
544  if (auto replacement = tiledResults->replacements.lookup(res))
545  rewriter.replaceAllUsesWith(res, replacement);
546  if (toReplace->use_empty()) {
547  rewriter.eraseOp(toReplace);
548  }
549  }
550 
551  // Report back the relevant handles to the transform op.
552  tiledLinalgOps.push_back(tiledResults->tiledAndFusedOps.front());
553  assert(tiledResults->loops.size() == numLoops &&
554  "Mismatched number of loops, tile and fuse transform should have "
555  "failed");
556  for (unsigned int i = 0; i < numLoops; ++i)
557  loopOps[i].push_back(tiledResults->loops[i]);
558  }
559 
560  transformResults.set(transformOp->getOpResult(0), tiledLinalgOps);
561  for (unsigned int i = 0; i < numLoops; ++i)
562  transformResults.set(transformOp->getOpResult(i + 1), loopOps[i]);
563 
564  return success();
565 }

582  tileAndFuseOptions.tilingOptions = tilingOptions;
583 
584  if (getApplyCleanup()) {
585  MLIRContext *context = rewriter.getContext();
586  RewritePatternSet patterns(context);
587  tensor::ExtractSliceOp::getCanonicalizationPatterns(patterns, context);
588  tensor::populateMergeConsecutiveInsertExtractSlicePatterns(patterns);
589  tensor::populateBubbleUpExtractSliceOpPatterns(patterns);
590  tileAndFuseOptions.cleanupPatterns = std::move(patterns);
591  }

593  LogicalResult result = applyTilingToAll(
594  rewriter, getOperation(), state.getPayloadOps(getTarget()),
595  tileSizes.size() - llvm::count(tileSizes, 0), transformResults,
596  [&](TilingInterface tilingInterfaceOp)
597  -> FailureOr<scf::SCFTileAndFuseResult> {
598  return tileConsumerAndFuseProducersUsingSCF(rewriter, tilingInterfaceOp,
599  tileAndFuseOptions);
600  });
601  return failed(result) ? DiagnosedSilenceableFailure::definiteFailure()
602  : DiagnosedSilenceableFailure::success();
603 }

637 /// Add new operands to the forall op for users of the producerOp
638 /// that are dominated by the containing scf.forall op.
639 static Operation *replaceForAllWithNewSignature(
640  RewriterBase &rewriter, Diagnostic &diag, Operation *producerOp,
641  Operation *containingOp, TilingResult &tileAndFuseResult,
642  int64_t resultNumber, SmallVector<OpFoldResult> &offsets,
643  SmallVector<OpFoldResult> &sizes) {
644 
645  // Count number of users not including the containing op
646  SetVector<Operation *> dominatedUsers;
647  DominanceInfo domInfo(containingOp);
648  for (Operation *user : producerOp->getResult(resultNumber).getUsers()) {
649  if (!containingOp->isAncestor(user) &&
650  (domInfo.dominates(containingOp, user))) {
651  dominatedUsers.insert(user);
652  }
653  }
654  if (dominatedUsers.empty())
655  return nullptr;
656 
657  // Create new scf.forall op
658  auto forallOp = cast<scf::ForallOp>(containingOp);
659  OpBuilder::InsertionGuard g(rewriter);
660  rewriter.setInsertionPoint(forallOp);
661 
662  // Get new output
663  Location loc = forallOp.getLoc();
664  auto genericOp = dyn_cast<linalg::GenericOp>(producerOp);
665  if (!genericOp)
666  return nullptr;
667  SmallVector<Value> outputs = genericOp.getOutputs();
668  SmallVector<Value> newOuts(forallOp.getOutputs());
669  newOuts.push_back(outputs[resultNumber]);
670 
671  // Create new scf.forall op
672  auto newforallOp = scf::ForallOp::create(
673  rewriter, loc, forallOp.getMixedLowerBound(),
674  forallOp.getMixedUpperBound(), forallOp.getMixedStep(), newOuts,
675  forallOp.getMapping());
676  rewriter.eraseBlock(newforallOp.getBody());
677  newforallOp.getRegion().takeBody(forallOp.getRegion());
678 
679  // Add additional block argument for new value being returned
680  // and replaces all uses of the new output with corresponding bbArg
681  // inside the scf.forall to enable fusion into this new scf.forall.
682  newforallOp.getBody()->addArgument(newOuts.back().getType(),
683  newOuts.back().getLoc());
684  auto bbArgs = newforallOp.getBody()->getArguments();
685  rewriter.replaceUsesWithIf(newOuts.back(), bbArgs.back(),
686  [&](OpOperand &use) {
687  Operation *op = use.getOwner();
688  return newforallOp->isProperAncestor(op);
689  });
690 
691  // Fix terminator
692  scf::InParallelOp terminatorOp = newforallOp.getTerminator();
693  SmallVector<Operation *> yieldingOps = llvm::to_vector<4>(llvm::map_range(
694  terminatorOp.getYieldingOps(), [](Operation &op) { return &op; }));
695  Operation *firstYieldOp = yieldingOps.front();
696  rewriter.setInsertionPoint(firstYieldOp);
697  Value src = tileAndFuseResult.tiledValues[0];
698  Value dst = newforallOp.getRegionIterArgs().back();
699  SmallVector<OpFoldResult> strides(offsets.size(), rewriter.getIndexAttr(1));
700  tensor::ParallelInsertSliceOp::create(rewriter, firstYieldOp->getLoc(), src,
701  dst, offsets, sizes, strides);
702 
703  for (auto result : llvm::enumerate(forallOp.getResults())) {
704  rewriter.replaceAllUsesWith(result.value(),
705  newforallOp->getResult(result.index()));
706  }
707  rewriter.replaceUsesWithIf(producerOp->getResult(resultNumber),
708  newforallOp->getResults().back(),
709  [&](OpOperand &use) {
710  Operation *user = use.getOwner();
711  return dominatedUsers.contains(user);
712  });
713  return newforallOp;
714 }

716 /// Given two operands coming from a loop iter arg, 'src' and 'dst', return true
717 /// if the operand 'src' is equal to 'dst' or equal to a iter arg present in a
718 /// outer loop. To determine the second condition, this function iterates
719 /// using a worklist over the enclosing loops, trying to find 'src' in any of
720 /// the parent loop's iter args.
721 static bool sameOrEquivalentIterArg(Value src, Value dst) {
722  // Stack like vector containing possible iterArgs candidates. The first one
723  // is dst, and we will transverse the IR from there.
724  SmallVector<Value> destWorklist;
725  destWorklist.push_back(dst);
726 
727  while (!destWorklist.empty()) {
728  Value currentDst = destWorklist.pop_back_val();
729 
730  // We have found the same operand in some iter arg in the loop structure,
731  // so src and dst are equivalent.
732  if (src == currentDst)
733  return true;
734 
735  // The operands are not equivalent, look for enclosing loops over
736  // currentDst.
737  auto bbArg = dyn_cast<BlockArgument>(currentDst);
738  if (!bbArg)
739  continue;
740 
741  Block *parentBlock = bbArg.getOwner();
742  assert(parentBlock && "unlinked block argument");
743 
744  Operation *parentOp = parentBlock->getParentOp();
745  assert(parentOp && "expected block argument with parent operation");
746 
747  // Check if parent is loop-like. If it's not, do not add it to the worklist.
748  auto parentLoop = dyn_cast<LoopLikeOpInterface>(parentOp);
749  if (!parentLoop)
750  continue;
751 
752  for (auto innerIterArg : parentLoop.getRegionIterArgs()) {
753  // No need to check for null as innerIterArg is tied to parentLoop.
754  OpOperand *operand = parentLoop.getTiedLoopInit(innerIterArg);
755  Value loopBlockArgument =
756  parentLoop->getOperand(operand->getOperandNumber());
757  destWorklist.push_back(loopBlockArgument);
758  }
759  }
760 
761  return false;
762 }

764 /// Find the first "extract" user of `producerOp` and tile it right before its
765 /// use. The tiled op is fused under the `containingOp`.
766 /// Return this fused op on success or nullptr if anything fails.
767 /// If tiled op has uses that are dominated by `containingOp`, return
768 /// a new `containingOp` with results of the fused op appended to
769 /// results of the `containingOp` or nullptr if there are no dominated uses.
770 static std::tuple<SmallVector<Operation *>, Operation *>
771 tileAndFuseFirstExtractUse(RewriterBase &rewriter, Diagnostic &diag,
772  Operation *producerOp, Operation *containingOp) {
773  LDBG() << "Try to fuse a direct extract use";
774  auto tileableProducer = dyn_cast<TilingInterface>(producerOp);
775  if (!tileableProducer) {
776  diag.attachNote(producerOp->getLoc())
777  << "producer is not a TileableInterface: " << *producerOp;
778  return {};
779  }
780 
781  // Search the producer slices accessed within the containing operation.
782  // TODO: Generalize to more extract/insert/parallel_insert triples, maybe
783  // evolve into an interface.
784  auto it = llvm::find_if(tileableProducer->getUsers(), [&](Operation *user) {
785  auto sliceOp = dyn_cast<tensor::ExtractSliceOp>(user);
786  return sliceOp && containingOp->isProperAncestor(sliceOp);
787  });
788 
789  // Find a fusion opportunity.
790  if (it == tileableProducer->getUsers().end()) {
791  diag.attachNote(tileableProducer->getLoc())
792  << "could not find fusion opportunity for: " << *tileableProducer;
793  return {};
794  }
795  auto sliceOpToTile = cast<tensor::ExtractSliceOp>(*it);
796 
797  // Try to fuse the producer in-place.
798  OpBuilder::InsertionGuard guard(rewriter);
799  rewriter.setInsertionPoint(sliceOpToTile);
800 
801  // Clone the producer inside the consumer and try to update the producer init
802  // operands using the loop bbArgs if applicable. More precisely, if the bbArg
803  // of the container loop points to a value that it is used by the consumer op,
804  // then, instead of using such value on the consumer, use the value coming
805  // from the bbArg instead. This allows to reuse the output tensor (instead of
806  // creating a new one) of the container when both producer and container write
807  // to the same output.
808  if (LoopLikeOpInterface containerLoop =
809  dyn_cast<LoopLikeOpInterface>(sliceOpToTile->getParentOp())) {
810  Operation *clone = rewriter.clone(*producerOp);
811  rewriter.modifyOpInPlace(clone, [&]() {
812  // Iterate over the outputs of the producer and over the loop bbArgs and
813  // check if any bbArg points to the same value as the producer output. In
814  // such case, make the producer output point to the bbArg directly.
815  for (OpOperand &initOperandPtr :
816  cast<DestinationStyleOpInterface>(clone).getDpsInitsMutable()) {
817  Value producerOperand =
818  clone->getOperand(initOperandPtr.getOperandNumber());
819  for (BlockArgument containerIterArg :
820  containerLoop.getRegionIterArgs()) {
821  OpOperand *bbArg = containerLoop.getTiedLoopInit(containerIterArg);
822  Value consumerOperand =
823  containerLoop->getOperand(bbArg->getOperandNumber());
824  // The producer has the same init as the loop bbArg, use it.
825  if (sameOrEquivalentIterArg(producerOperand, consumerOperand)) {
826  initOperandPtr.set(containerIterArg);
827  }
828  }
829  }
830  });
831 
832  tileableProducer = dyn_cast<TilingInterface>(clone);
833  }
834 
835  // Tile the producer.
836  int64_t resultNumber =
837  cast<OpResult>(sliceOpToTile.getSource()).getResultNumber();
838  LDBG() << "resultNumber: " << resultNumber;
839 
840  SmallVector<OpFoldResult> offsets = sliceOpToTile.getMixedOffsets();
841  SmallVector<OpFoldResult> sizes = sliceOpToTile.getMixedSizes();
842 
843  FailureOr<TilingResult> tileAndFuseResult =
844  tileableProducer.generateResultTileValue(rewriter, resultNumber, offsets,
845  sizes);
846 
847  if (failed(tileAndFuseResult)) {
848  diag.attachNote(tileableProducer->getLoc())
849  << "failed to tile producer op: " << *tileableProducer;
850  return {};
851  }
852 
853 #ifndef NDEBUG
854  for (auto *tiledOp : tileAndFuseResult->tiledOps) {
855  LDBG() << "tiledProducer: " << *tiledOp;
856  }
857 #endif
858 
859  // Replace the extract op.
860  auto maybeRankReduced = tensor::ExtractSliceOp::rankReduceIfNeeded(
861  rewriter, sliceOpToTile->getLoc(), tileAndFuseResult->tiledValues[0],
862  cast<RankedTensorType>(sliceOpToTile->getResult(0).getType()).getShape());
863  if (failed(maybeRankReduced)) {
864  diag.attachNote(producerOp->getLoc())
865  << "shape types don't match (missing canonicalization?):\nTiledOp: "
866  << tileAndFuseResult->tiledValues[0]
867  << "\nSliceOp: " << sliceOpToTile.getOperation() << '\n';
868  return {};
869  }
870  rewriter.replaceOp(sliceOpToTile, *maybeRankReduced);
871 
872  // Add new outputs to containing op, if required
873  Operation *newContainingOp = replaceForAllWithNewSignature(
874  rewriter, diag, producerOp, containingOp, *tileAndFuseResult,
875  resultNumber, offsets, sizes);
876 
877  // Cleanup clone.
878  if (dyn_cast<LoopLikeOpInterface>(containingOp))
879  rewriter.eraseOp(tileableProducer);
880 
881  return std::make_tuple(tileAndFuseResult->tiledOps, newContainingOp);
882 }

884 /// First, find the first "scf::ForallOp" user of `producerOp` and ensure
885 /// it is exactly the `containingOp`, otherwise bail.
886 /// Then, find the first "extract" user of the tied block argument and tile it
887 /// right before its "extract" use. The tiled op is fused under the
888 /// `containingOp`.
889 /// Return this fused op on success or nullptr if anything fails.
890 static SmallVector<Operation *>
891 tileAndFuseFirstExtractUseThroughContainingOpBlockArgument(
892  RewriterBase &rewriter, Diagnostic &diag, Operation *producerOp,
893  Operation *containingOp) {
894  LDBG() << "Try to fuse an extract use through block argument";
895 
896  auto tileableProducer = dyn_cast<TilingInterface>(producerOp);
897  if (!tileableProducer) {
898  diag.attachNote(producerOp->getLoc())
899  << "producer is not a TileableInterface: " << *producerOp;
900  return {};
901  }
902 
903  // Search the first use by a "scf::ForallOp" user.
904  scf::ForallOp forallOp;
905  auto itProducerUses =
906  llvm::find_if(tileableProducer->getUses(), [&](OpOperand &use) {
907  forallOp = dyn_cast<scf::ForallOp>(use.getOwner());
908  return forallOp;
909  });
910  // If it's not from the containing op, return.
911  if (!forallOp || forallOp != containingOp) {
912  diag.attachNote(tileableProducer->getLoc())
913  << "could not find a use by the containing op: " << *tileableProducer;
914  return {};
915  }
916 
917  // Search the producer slices accessed within the containing
918  // operation.
919  // TODO: Generalize to more extract/insert/parallel_insert triples.
920  // Maybe evolve into an interface.
921  OpOperand *pUse = &(*itProducerUses);
922  BlockArgument bbArg = forallOp.getTiedBlockArgument(pUse);
923 
924  // Search the producer slices accessed within the containing operation.
925  // TODO: Generalize to more extract/insert/parallel_insert triples, maybe
926  // evolve into an interface.
927  auto itBBArgUsers = llvm::find_if(bbArg.getUsers(), [&](Operation *user) {
928  auto sliceOp = dyn_cast<tensor::ExtractSliceOp>(user);
929  return sliceOp && containingOp->isProperAncestor(sliceOp);
930  });
931 
932  // Find a fusion opportunity.
933  if (itBBArgUsers == bbArg.getUsers().end()) {
934  diag.attachNote(containingOp->getLoc())
935  << "could not find fusion opportunity for bbArg: " << bbArg;
936  return {};
937  }
938  auto sliceOpToTile = cast<tensor::ExtractSliceOp>(*itBBArgUsers);
939 
940  // Try to fuse the producer in-place.
941  OpBuilder::InsertionGuard guard(rewriter);
942  rewriter.setInsertionPoint(sliceOpToTile);
943 
944  // Replace the use in the tileableProducer before tiling: clone, replace and
945  // then tile.
946  int64_t resultNumber = cast<OpResult>(pUse->get()).getResultNumber();
947  LDBG() << "resultNumber: " << resultNumber;
948 
949  // Gather destination tensors.
950  SmallVector<Value> destinationTensors;
951  if (failed(tensor::getOrCreateDestinations(
952  rewriter, tileableProducer->getLoc(), tileableProducer,
953  destinationTensors))) {
954  diag.attachNote(tileableProducer->getLoc())
955  << "failed to get destination tensors for: " << *tileableProducer;
956  return {};
957  }
958 
959  IRMapping bvm;
960  bvm.map(destinationTensors[resultNumber], bbArg);
961  auto tileableProducerClone =
962  cast<TilingInterface>(rewriter.clone(*tileableProducer, bvm));
963  auto scopeGuard =
964  llvm::make_scope_exit([&]() { rewriter.eraseOp(tileableProducerClone); });
965 
966  // Tile the producer.
967  FailureOr<TilingResult> tileAndFuseResult =
968  tileableProducerClone.generateResultTileValue(
969  rewriter, resultNumber, sliceOpToTile.getMixedOffsets(),
970  sliceOpToTile.getMixedSizes());
971  if (failed(tileAndFuseResult)) {
972  diag.attachNote(tileableProducer->getLoc())
973  << "failed to tile producer op: " << *tileableProducer;
974  return {};
975  }
976 
977  // Replace the extract op.
978  auto maybeRankReduced = tensor::ExtractSliceOp::rankReduceIfNeeded(
979  rewriter, sliceOpToTile->getLoc(), tileAndFuseResult->tiledValues[0],
980  cast<RankedTensorType>(sliceOpToTile->getResult(0).getType()).getShape());
981  assert(succeeded(maybeRankReduced) && "unexpected shape");
982  rewriter.replaceOp(sliceOpToTile, *maybeRankReduced);
983 
984  // Replace the use in containingOp.
985  rewriter.modifyOpInPlace(containingOp, [&]() {
986  containingOp->setOperand(pUse->getOperandNumber(),
987  destinationTensors.front());
988  });
989 
990  return tileAndFuseResult->tiledOps;
991 }

993 static Operation *cloneAndFuseFirstUse(RewriterBase &rewriter, Diagnostic &diag,
994  Operation *producerOp,
995  Operation *containingOp) {
996  LDBG() << "Try to fuse an use by cloning";
997 
998  // Gather all uses inside the containing op.
999  SmallVector<OpOperand *> uses;
1000  for (OpResult result : producerOp->getOpResults()) {
1001  for (OpOperand &use : result.getUses()) {
1002  if (containingOp->isProperAncestor(use.getOwner())) {
1003  uses.push_back(&use);
1004  continue;
1005  }
1006  // Cannot clone and fuse if the use is by the containing op itself: fail
1007  // immediately.
1008  if (containingOp == use.getOwner()) {
1009  diag.attachNote(producerOp->getLoc())
1010  << "producer op use by containing op cannot be fused by cloning";
1011  return nullptr;
1012  }
1013  }
1014  }
1015 
1016  // Check for a non-empty list of fusion opportunities.
1017  if (uses.empty()) {
1018  diag.attachNote(producerOp->getLoc()) << "no fusion opportunity by cloning";
1019  return nullptr;
1020  }
1021 
1022  // Clone and fuse inside the containing op.
1023  Operation *fusedOp = nullptr;
1024  OpOperand *use = uses.front();
1025  // Parallel insert slice is not a valid clone destination.
1026  // TODO: Generalize to other type of ops.
1027  assert(!isa<tensor::ParallelInsertSliceOp>(use->getOwner()) &&
1028  "Parallel insert slice is not a valid clone destination");
1029  unsigned resultNumber = cast<OpResult>(use->get()).getResultNumber();
1030  LDBG() << "resultNumber: " << resultNumber;
1031 
1032  OpBuilder::InsertionGuard guard(rewriter);
1033  rewriter.setInsertionPoint(use->getOwner());
1034  fusedOp = rewriter.clone(*producerOp);
1035  rewriter.modifyOpInPlace(
1036  use->getOwner(), [&] { use->set(fusedOp->getOpResult(resultNumber)); });
1037 
1038  return fusedOp;
1039 }

1046 DiagnosedSilenceableFailure
1047 transform::FuseIntoContainingOp::apply(transform::TransformRewriter &rewriter,
1048  transform::TransformResults &results,
1049  transform::TransformState &state) {
1050  SmallVector<Operation *> fusedOps;
1051  auto producerOps = state.getPayloadOps(getProducerOp());
1052  auto containingOps = state.getPayloadOps(getContainingOp());
1053  if (!llvm::hasSingleElement(containingOps)) {
1054  return emitDefiniteFailure()
1055  << "requires exactly one containing_op handle (got "
1056  << llvm::range_size(containingOps) << ")";
1057  }
1058  Operation *containingOp = *containingOps.begin();
1059 
1060  // If nothing to fuse, propagate success.
1061  if (std::empty(producerOps)) {
1062  results.set(cast<OpResult>(getFusedOp()), SmallVector<mlir::Operation *>{});
1063  results.set(cast<OpResult>(getNewContainingOp()), {containingOp});
1064  return DiagnosedSilenceableFailure::success();
1065  }
1066 
1067  // Helper function to find the next producer that should be fused. Take any
1068  // producer that has a use inside the containing op.
1069  SetVector<Operation *> remainingProducers(llvm::from_range, producerOps);
1070  auto getNextProducer = [&]() -> FailureOr<Operation *> {
1071  for (const auto &it : enumerate(remainingProducers)) {
1072  Operation *producerOp = it.value();
1073  // The containing op may be a user of producerOp: use isAncestor.
1074  int64_t numUsesInContainingOp =
1075  llvm::count_if(producerOp->getUsers(), [&](Operation *op) {
1076  return containingOp->isAncestor(op);
1077  });
1078  // TODO: When resolving the TODO below (no duplicate ops), take an op
1079  // that has no use among the remaining producers. This is a topological
1080  // sorting.
1081  if (numUsesInContainingOp > 0) {
1082  if (numUsesInContainingOp == 1)
1083  remainingProducers.erase(remainingProducers.begin() + it.index());
1084  return producerOp;
1085  }
1086  }
1087  return failure();
1088  };
1089 
1090  while (!remainingProducers.empty()) {
1091  auto nextProducer = getNextProducer();
1092  if (failed(nextProducer)) {
1093  auto diag = mlir::emitSilenceableFailure(getLoc())
1094  << "could not find next producer to fuse into container";
1095  diag.attachNote(containingOp->getLoc()) << "containing op";
1096  return diag;
1097  }
1098 
1099  Operation *producerOp = *nextProducer;
1100 
1101  // Default diagnostic, to be complemented with more failure information.
1102  Diagnostic diag(producerOp->getLoc(), DiagnosticSeverity::Remark);
1103  diag << "could not fuse " << *producerOp << " into " << *containingOp;
1104 
1105  // TODO: If there are multiple uses of the producer in the containing op,
1106  // we currently tile/clone the op multiple times (once per use). In some
1107  // cases, we can tile/clone once and reuse the value for each use.
1108  // Futhermore, producers should then be traversed according to a
1109  // topological sorting.
1110  auto [tiledOps, newContainingOp] =
1111  tileAndFuseFirstExtractUse(rewriter, diag, producerOp, containingOp);
1112  if (!tiledOps.empty()) {
1113  LDBG() << "\nFused a direct extract use\n" << *containingOp;
1114  fusedOps.append(tiledOps);
1115  if (newContainingOp) {
1116  // Update handles associated with the containing op so we don't need to
1117  // invalidate them. This is a hack to support better composability
1118  // between tiling and fusion while a proper mechanism is being
1119  // investigated.
1120  //
1121  // DO NOT replicate this elsewhere unless you understand what you are
1122  // doing.
1123  LogicalResult replacementStatus =
1124  rewriter.notifyPayloadOperationReplaced(containingOp,
1125  newContainingOp);
1126  (void)replacementStatus;
1127  assert(succeeded(replacementStatus) &&
1128  "unable to update transform state mapping");
1129  rewriter.eraseOp(containingOp);
1130  containingOp = newContainingOp;
1131  }
1132  continue;
1133  }
1134 
1135  SmallVector<Operation *> tiledContainingOpOperand =
1136  tileAndFuseFirstExtractUseThroughContainingOpBlockArgument(
1137  rewriter, diag, producerOp, containingOp);
1138  if (!tiledContainingOpOperand.empty()) {
1139  LDBG() << "\nFused an extract use through block argument\n"
1140  << *containingOp;
1141  fusedOps.append(tiledContainingOpOperand);
1142  continue;
1143  }
1144 
1145  Operation *cloned =
1146  cloneAndFuseFirstUse(rewriter, diag, producerOp, containingOp);
1147  if (cloned) {
1148  LDBG() << "\nFused an use by cloning\n" << *containingOp;
1149  fusedOps.push_back(cloned);

1393 DiagnosedSilenceableFailure
1394 transform::MatchOp::apply(transform::TransformRewriter &rewriter,
1395  transform::TransformResults &results,
1396  transform::TransformState &state) {
1397  llvm::StringSet<> strs;
1398  if (getOps().has_value())
1399  strs.insert_range(getOps()->getAsValueRange<StringAttr>());
1400 
1401  auto payloadOps = state.getPayloadOps(getTarget());
1402  if (!llvm::hasSingleElement(payloadOps)) {
1403  return emitDefiniteFailure("requires exactly one target handle");
1404  }
1405 
1406  SmallVector<Operation *> res;
1407  bool incorrectNumOperandTypes = false;
1408  auto matchFun = [&](Operation *op) {
1409  if (getOps().has_value() && !strs.contains(op->getName().getStringRef()))
1410  return;
1411 
1412  // Interfaces cannot be matched by name, just by ID.
1413  // So we specifically encode the interfaces we care about for this op.
1414  if (getInterface().has_value()) {
1415  auto iface = getInterface().value();
1416  if (iface == transform::MatchInterfaceEnum::LinalgOp &&
1417  !isa<LinalgOp>(op))
1418  return;
1419  if (iface == transform::MatchInterfaceEnum::TilingInterface &&
1420  !isa<TilingInterface>(op))
1421  return;
1422  if (iface == transform::MatchInterfaceEnum::LoopLikeInterface &&
1423  !isa<LoopLikeOpInterface>(op))
1424  return;
1425  }
1426 
1427  // Check if all specified attributes match.
1428  if (getOpAttrs().has_value()) {
1429  DictionaryAttr opAttrs = getOpAttrs().value();
1430  for (NamedAttribute attr : opAttrs) {
1431  if (attr.getName() == getInterfaceAttrName() ||
1432  attr.getName() == getOpsAttrName())
1433  continue;
1434  if (!op->hasAttr(attr.getName()))
1435  return;
1436  if (op->getAttr(attr.getName()) != attr.getValue())
1437  return;
1438  }
1439  }
1440 
1441  if (getFilterResultType().has_value()) {
1442  Type t = getFilterResultType().value();
1443  if (op->getNumResults() != 1 || op->getResultTypes().front() != t)
1444  return;
1445  }
1446 
1447  if (getFilterOperandTypes().has_value()) {
1448  mlir::ArrayAttr types = getFilterOperandTypes().value();
1449  auto operandTypes = op->getOperandTypes();
1450 
1451  if (types.size() == 1) {
1452  // All the operands must must be equal to the specified type
1453  auto typeattr =
1454  dyn_cast<mlir::TypeAttr>(getFilterOperandTypes().value()[0]);
1455  Type t = cast<::mlir::Type>(typeattr.getValue());
1456  if (!llvm::all_of(op->getOperandTypes(),
1457  [&](Type operandType) { return operandType == t; }))
1458  return;
1459  } else {
1460  // The operand types must match all the types in the list (in the same
1461  // order in with they are specified)
1462  if (types.size() != operandTypes.size()) {
1463  incorrectNumOperandTypes = true;
1464  return;
1465  }
1466 
1467  for (auto [attr, operandType] :
1468  llvm::zip_equal(getFilterOperandTypes().value(), operandTypes)) {
1469  auto typeattr = cast<mlir::TypeAttr>(attr);
1470  Type type = cast<::mlir::Type>(typeattr.getValue());
1471 
1472  if (type != operandType)
1473  return;
1474  }
1475  }
1476  }
1477 
1478  // All constraints are satisfied.
1479  res.push_back(op);
1480  return;
1481  };
1482 
1483  (*payloadOps.begin())->walk(matchFun);
1484  if (incorrectNumOperandTypes)
1485  return emitDefiniteFailure("If filter_operand_types contains more than a "
1486  "type, then it must contain as much types as "
1487  "the number of operands in the target ops");
1488  results.set(cast<OpResult>(getResult()), res);
1489  return DiagnosedSilenceableFailure::success();
1490 }

1768 /// Return true if `permutation` is a valid permutation of the
1769 /// `outer_dims_perm` (case OuterOrInnerPerm::Outer) or `inner_dims_pos`
1770 /// (OuterOrInnerPerm::Inner) of the `tensor.pack` or `tensor.unpack` `op.
1771 /// This is the case when the `permutation` rank matches the rank expected by
1772 /// `op` and `permutation` is itself a permutation vector.
1773 /// Return true if either `op` or `permutation` are empty to allow a simpler
1774 /// polymorphic implementation.
1775 template <typename RelayoutOpTy>
1776 bool isValidPackingPermutation(
1777  RelayoutOpTy op, ArrayRef<int64_t> permutation,
1778  OuterOrInnerPerm outerOrInnerPerm = OuterOrInnerPerm::Outer) {
1779  static_assert(
1780  llvm::is_one_of<RelayoutOpTy, linalg::PackOp, linalg::UnPackOp>::value,
1781  "applies to only pack or unpack operations");
1782  if (!op || permutation.empty())
1783  return true;
1784  size_t innerRank = op.getInnerDimsPos().size();
1785  if (outerOrInnerPerm == OuterOrInnerPerm::Inner)
1786  return permutation.size() == innerRank && isPermutationVector(permutation);
1787  // op.getOuterDimsPerm() may be empty, in which case it is identity.
1788  // Don't rely on it.
1789  if (std::is_same<RelayoutOpTy, linalg::PackOp>::value) {
1790  return permutation.size() == op.getSourceRank() &&
1791  isPermutationVector(permutation);
1792  }
1793  return permutation.size() == op.getDestRank() &&
1794  isPermutationVector(permutation);
1795 }

1797 DiagnosedSilenceableFailure
1798 transform::PackTransposeOp::apply(transform::TransformRewriter &rewriter,
1799  transform::TransformResults &transformResults,
1800  transform::TransformState &state) {
1801  auto packOrUnpackOps = state.getPayloadOps(getTargetPackOrUnPackOp());
1802  auto linalgOps = state.getPayloadOps(getTargetLinalgOp());
1803  // Step 1. If nothing to pack, propagate success.
1804  if (std::empty(packOrUnpackOps)) {
1805  transformResults.set(cast<OpResult>(getPackedOp()), {});
1806  transformResults.set(cast<OpResult>(getPackOp()), {});
1807  transformResults.set(cast<OpResult>(getUnPackOp()), {});
1808  return DiagnosedSilenceableFailure::success();
1809  }
1810 
1811  // Step 2. Bunch of runtime sanity check and error messages.
1812  // Step 2.1. Fail on multi-op handles.
1813  if (!llvm::hasSingleElement(packOrUnpackOps) ||
1814  !llvm::hasSingleElement(linalgOps)) {
1815  return emitSilenceableError()
1816  << "requires target to map to exactly 1 "
1817  "packing op and 1 packed op ("
1818  << "got " << llvm::range_size(packOrUnpackOps) << " and "
1819  << llvm::range_size(linalgOps) << ")";
1820  }
1821 
1822  // Step 2.2. Fail on wrong type.
1823  auto packOp = dyn_cast<linalg::PackOp>(*packOrUnpackOps.begin());
1824  auto unPackOp = dyn_cast<linalg::UnPackOp>(*packOrUnpackOps.begin());
1825  if ((!packOp && !unPackOp)) {
1826  return emitSilenceableError() << "requires target to map to a "
1827  "linalg.pack or linalg.unpack";
1828  }
1829  LinalgOp linalgOpTarget = dyn_cast<LinalgOp>(*linalgOps.begin());
1830  if (!linalgOpTarget)
1831  return emitSilenceableError() << "requires a LinalgOp target";
1832 
1833  // Step 2.3. Fail if we can't get the producer / consumer Linalg op.
1834  LinalgOp linalgOp;
1835  if (packOp && packOp.getResult().hasOneUse())
1836  linalgOp = dyn_cast<LinalgOp>(*(packOp.getResult().getUsers().begin()));
1837  else if (unPackOp)
1838  linalgOp = unPackOp.getSource().getDefiningOp<LinalgOp>();
1839  if (linalgOp != linalgOpTarget) {
1840  auto errorMsg =
1841  packOp ? StringLiteral{"not a single use by the LinalgOp target"}
1842  : StringLiteral{"not produced by the LinalgOp target"};
1843  return emitSilenceableError() << errorMsg;
1844  }
1845 
1846  // Step 2.4. If we have an UnPackOp, we need to fetch the symmetrical
1847  // PackOp.
1848  if (unPackOp) {
1849  assert(!packOp && "packOp must be null on entry when unPackOp is not null");
1850  OpOperand *packUse = linalgOp.getDpsInitOperand(
1851  cast<OpResult>(unPackOp.getSource()).getResultNumber());
1852  packOp = packUse->get().getDefiningOp<linalg::PackOp>();
1853  if (!packOp || !packOp.getResult().hasOneUse())
1854  return emitSilenceableError() << "could not find matching pack op";
1855  }
1856 
1857  // Step 2.5. Fail if any permutation does not validate.
1858  for (auto permType : {OuterOrInnerPerm::Outer, OuterOrInnerPerm::Inner}) {
1859  ArrayRef<int64_t> perm =
1860  (permType == OuterOrInnerPerm::Outer) ? getOuterPerm() : getInnerPerm();
1861  auto errorMsg = (permType == OuterOrInnerPerm::Outer)
1862  ? StringLiteral{"invalid outer_perm"}
1863  : StringLiteral{"invalid inner_perm"};
1864  if (!isValidPackingPermutation(packOp, perm, permType) ||
1865  !isValidPackingPermutation(unPackOp, perm, permType)) {
1866  Operation *packOrUnpackOp =
1867  unPackOp ? unPackOp.getOperation() : packOp.getOperation();
1868  return emitSilenceableError() << errorMsg << ": " << *packOrUnpackOp;
1869  }
1870  }
1871 
1872  // From here on, packOp and linalgOp are always present, unPackOp may or may
1873  // not be present.
1874  assert(packOp && linalgOp && "unexpected null op");
1875 
1876  // Step 3. Actually transpose the ops.
1877  FailureOr<PackTransposeResult> res = packTranspose(
1878  rewriter, packOp, linalgOp, unPackOp, getOuterPerm(), getInnerPerm());
1879  // Preconditions have been checked, it is an error to fail here.
1880  assert(succeeded(res) && "unexpected packTranspose failure");
1881 
1882  // Step 4. Return results.
1883  transformResults.set(cast<OpResult>(getPackOp()), {res->transposedPackOp});
1884  transformResults.set(cast<OpResult>(getPackedOp()),
1885  {res->transposedLinalgOp});
1886  if (unPackOp) {
1887  transformResults.set(cast<OpResult>(getUnPackOp()),
1888  {res->transposedUnPackOp});
1889  } else {
1890  transformResults.set(cast<OpResult>(getUnPackOp()), {});
1891  }
1892 
1893  return DiagnosedSilenceableFailure::success();
1894 }

2626  llvm::to_vector(state.getPayloadOps(getTarget()));

2629

2633  "multiway split is enabled (got "
2634  << llvm::range_size(payload) << ")";
2635  }
2636 
2637  SmallVector<OpFoldResult> chunkSizes;
2638 
2639  if (!isMultiwaySplit)
2640  chunkSizes.reserve(payload.size());
2641 
2642  if (getDynamicChunkSizes()) {
2643  auto diag = DiagnosedSilenceableFailure::success();
2644  if (isa<TransformHandleTypeInterface>(getDynamicChunkSizes().getType())) {
2645  chunkSizes = llvm::to_vector(llvm::map_range(
2646  state.getPayloadOps(getDynamicChunkSizes()), [&](Operation *op) {
2647  if (op->getNumResults() != 1 ||
2648  !op->getResult(0).getType().isIndex()) {
2649  diag = emitSilenceableError()

2652  diag.attachNote(op->getLoc()) << "dynamic split point";

2654  return OpFoldResult(op->getResult(0));
2655  }));
2656  } else {

2658  llvm::map_range(state.getParams(getDynamicChunkSizes()),
2659  [](Attribute attr) { return OpFoldResult(attr); }));
2660  }
2661  if (diag.isSilenceableFailure())
2662  return diag;

2665  // split points.
2666  if (!isMultiwaySplit && chunkSizes.size() != payload.size()) {
2667  return emitDefiniteFailure()
2668  << "expected the dynamic split point handle to point to as "
2669  "many operations ("
2670  << chunkSizes.size() << ") as the target handle ("
2671  << payload.size() << ")";
2672  }
2673  } else {
2674  chunkSizes.resize(payload.size(),
2675  rewriter.getIndexAttr(getStaticChunkSizes()));
2676  }
2677 
2678  auto checkStructuredOpAndDimensions =
2679  [&](LinalgOp linalgOp, Location loc) -> DiagnosedSilenceableFailure {
2680  if (!linalgOp) {
2681  auto diag = emitSilenceableError() << "only applies to structured ops";
2682  diag.attachNote(loc) << "target op";
2683  return diag;
2684  }
2685

2688  << " does not exist in target op";
2689  diag.attachNote(loc) << "target op";
2690  return diag;
2691  }

2696  [&](bool hasFailed, Location loc) -> DiagnosedSilenceableFailure {
2697  if (hasFailed) {
2698  auto diag = emitDefiniteFailure() << "internal failure in splitting";
2699  diag.attachNote(loc) << "target op";
2700  return diag;
2701  }
2702  return DiagnosedSilenceableFailure::success();
2703  };
2704 
2705  SmallVector<Operation *> opList;
2706  if (isMultiwaySplit) {
2707

2709  TilingInterface head, tail;
2710  Operation *target = payload.front();
2711 
2712  LinalgOp linalgOp = dyn_cast<LinalgOp>(target);
2713 
2714  // Check that the target is a valid LinalgOp with correct dimensions.
2715  DiagnosedSilenceableFailure diag =
2716  checkStructuredOpAndDimensions(linalgOp, target->getLoc());
2717  if (diag.isSilenceableFailure())
2718  return diag;
2719 
2720  for (auto &&[idx, chunkSize] : llvm::enumerate(chunkSizes)) {
2721 
2722  if (idx > 0)
2723  target = tail.getOperation();
2724 
2725  if (!target)
2726  break;
2727 
2728  linalgOp = cast<LinalgOp>(target);
2729  Location loc = target->getLoc();
2730 
2731  rewriter.setInsertionPoint(linalgOp);
2732  std::tie(head, tail) = linalg::splitOp(
2733  rewriter, cast<TilingInterface>(linalgOp.getOperation()),
2734  getDimension(), chunkSize);
2735 
2736  // Propagate errors.
2737  DiagnosedSilenceableFailure diag =
2738  checkFailureInSplitting(!head && !tail, loc);
2739  if (diag.isDefiniteFailure())
2740  return diag;
2741 
2742  opList.push_back(head.getOperation());
2743  }
2744 
2745  // Append any leftover parts to the end of the result list.
2746  if (tail)
2747  opList.push_back(tail.getOperation());
2748 
2749  } else {
2750  // Split each target operation.
2751  SmallVector<Operation *> first, second;
2752  Operation *noSecondPart = nullptr;
2753  for (const auto &pair : llvm::zip(payload, chunkSizes)) {
2754  Operation *target = std::get<0>(pair);
2755  Location loc = target->getLoc();
2756  LinalgOp linalgOp = dyn_cast<LinalgOp>(target);
2757  DiagnosedSilenceableFailure diag =
2758  checkStructuredOpAndDimensions(linalgOp, target->getLoc());
2759 
2760  if (diag.isSilenceableFailure())
2761  return diag;
2762 
2763  rewriter.setInsertionPoint(linalgOp);
2764  std::tie(first.emplace_back(), second.emplace_back()) = linalg::splitOp(
2765  rewriter, cast<TilingInterface>(linalgOp.getOperation()),
2766  getDimension(), std::get<1>(pair));
2767 
2768  // Propagate errors.
2769  DiagnosedSilenceableFailure diagSplit =
2770  checkFailureInSplitting(!first.back() && !second.back(), loc);
2771  if (diagSplit.isDefiniteFailure())
2772  return diag;
2773 
2774  // Do not add null second parts.
2775  if (!second.back()) {
2776  noSecondPart = target;
2777  second.pop_back();
2778  }
2779  }
2780 
2781  if (second.size() != first.size() && !second.empty()) {
2782  auto diag = emitSilenceableError()
2783  << "splitting does not produce the second part for a subset "
2784  "of targets";
2785  diag.attachNote()
2786  << "expected splitting to produce the second part of all "
2787  "or none of the targets";
2788  diag.attachNote(noSecondPart->getLoc())
2789  << "first target with no second part";
2790  return diag;
2791  }
2792 
2793  opList.append(first);
2794  if (second.size())
2795  opList.append(second);
2796  }
2797  results.set(cast<OpResult>(getSplitList()), opList);
2798  return DiagnosedSilenceableFailure::success();
2799 }
