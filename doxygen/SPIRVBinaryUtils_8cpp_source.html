39  // See "2.3. Physical Layout of a SPIR-V Module and Instruction" in the SPIR-V
40  // spec for the definition of the binary module header.
41  //
42  // The first five words of a SPIR-V module must be:
43  // +-------------------------------------------------------------------------+
44  // | Magic number |
45  // +-------------------------------------------------------------------------+
46  // | Version number (bytes: 0 | major number | minor number | 0) |
47  // +-------------------------------------------------------------------------+
48  // | Generator magic number |
49  // +-------------------------------------------------------------------------+
50  // | Bound (all result <id>s in the module guaranteed to be less than it) |
51  // +-------------------------------------------------------------------------+
52  // | 0 (reserved for instruction schema) |
53  // +-------------------------------------------------------------------------+
54  header.push_back(spirv::kMagicNumber);
55  header.push_back((majorVersion << 16) | (minorVersion << 8));
56  header.push_back((kGeneratorNumber << 16) | LLVM_VERSION_MAJOR);
57  header.push_back(idBound); // <id> bound
58  header.push_back(0); // Schema (reserved word)
59 }
