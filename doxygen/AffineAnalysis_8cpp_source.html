36 /// Returns true if `value` (transitively) depends on iteration arguments of the
37 /// given `forOp`.
38 static bool dependsOnIterArgs(Value value, AffineForOp forOp) {
39  // Compute the backward slice of the value.
40  SetVector<Operation *> slice;
41  getBackwardSlice(value, &slice,
42  [&](Operation *op) { return !forOp->isAncestor(op); });
43 
44  // Check that none of the operands of the operations in the backward slice are
45  // loop iteration arguments, and neither is the value itself.
46  auto argRange = forOp.getRegionIterArgs();
47  llvm::SmallPtrSet<Value, 8> iterArgs(argRange.begin(), argRange.end());
48  if (iterArgs.contains(value))
49  return true;
50 
51  for (Operation *op : slice)
52  for (Value operand : op->getOperands())
53  if (iterArgs.contains(operand))
54  return true;
55 
56  return false;
57 }

59 /// Get the value that is being reduced by `pos`-th reduction in the loop if
60 /// such a reduction can be performed by affine parallel loops. This assumes
61 /// floating-point operations are commutative. On success, `kind` will be the
62 /// reduction kind suitable for use in affine parallel loop builder. If the
63 /// reduction is not supported, returns null.
64 static Value getSupportedReduction(AffineForOp forOp, unsigned pos,
65  AtomicRMWKind &kind) {
66  auto yieldOp = cast<AffineYieldOp>(forOp.getBody()->back());
67  Value yielded = yieldOp.operands()[pos];
68  Operation *definition = yielded.getDefiningOp();
69  if (!definition)
70  return nullptr;
71  if (!forOp.getRegionIterArgs()[pos].hasOneUse())
72  return nullptr;
73  if (!yielded.hasOneUse())
74  return nullptr;
75 
76  Optional<AtomicRMWKind> maybeKind =
77  TypeSwitch<Operation *, Optional<AtomicRMWKind>>(definition)
78  .Case<AddFOp>([](Operation *) { return AtomicRMWKind::addf; })
79  .Case<MulFOp>([](Operation *) { return AtomicRMWKind::mulf; })
80  .Case<AddIOp>([](Operation *) { return AtomicRMWKind::addi; })
81  .Case<MulIOp>([](Operation *) { return AtomicRMWKind::muli; })
82  .Default([](Operation *) -> Optional<AtomicRMWKind> {
83  // TODO: AtomicRMW supports other kinds of reductions this is
84  // currently not detecting, add those when the need arises.
85  return llvm::None;
86  });
87  if (!maybeKind)
88  return nullptr;
89 
90  kind = *maybeKind;
91  if (definition->getOperand(0) == forOp.getRegionIterArgs()[pos] &&
92  !dependsOnIterArgs(definition->getOperand(1), forOp))
93  return definition->getOperand(1);
94  if (definition->getOperand(1) == forOp.getRegionIterArgs()[pos] &&
95  !dependsOnIterArgs(definition->getOperand(0), forOp))
96  return definition->getOperand(0);
97 
98  return nullptr;
99 }

101 /// Returns true if `forOp' is a parallel loop. If `parallelReductions` is
102 /// provided, populates it with descriptors of the parallelizable reductions and
103 /// treats them as not preventing parallelization.
104 bool mlir::isLoopParallel(AffineForOp forOp,
105  SmallVectorImpl<LoopReduction> *parallelReductions) {
106  unsigned numIterArgs = forOp.getNumIterOperands();
107 
108  // Loop is not parallel if it has SSA loop-carried dependences and reduction
109  // detection is not requested.
110  if (numIterArgs > 0 && !parallelReductions)
111  return false;
112 
113  // Find supported reductions of requested.
114  if (parallelReductions) {
115  parallelReductions->reserve(forOp.getNumIterOperands());
116  for (unsigned i = 0; i < numIterArgs; ++i) {
117  AtomicRMWKind kind;
118  if (Value value = getSupportedReduction(forOp, i, kind))
119  parallelReductions->emplace_back(LoopReduction{kind, i, value});
120  }
121 
122  // Return later to allow for identifying all parallel reductions even if the
123  // loop is not parallel.
124  if (parallelReductions->size() != numIterArgs)
125  return false;
126  }
127 
128  // Check memory dependences.
129  return isLoopMemoryParallel(forOp);
130 }

132 /// Returns true if `forOp' doesn't have memory dependences preventing
133 /// parallelization. This function doesn't check iter_args and should be used
134 /// only as a building block for full parallel-checking functions.
135 bool mlir::isLoopMemoryParallel(AffineForOp forOp) {
136  // Collect all load and store ops in loop nest rooted at 'forOp'.
137  SmallVector<Operation *, 8> loadAndStoreOps;
138  auto walkResult = forOp.walk([&](Operation *op) -> WalkResult {
139  if (isa<AffineReadOpInterface, AffineWriteOpInterface>(op))
140  loadAndStoreOps.push_back(op);
141  else if (!isa<AffineForOp, AffineYieldOp, AffineIfOp>(op) &&
142  !MemoryEffectOpInterface::hasNoEffect(op))
143  return WalkResult::interrupt();
144 
145  return WalkResult::advance();
146  });
147 
148  // Stop early if the loop has unknown ops with side effects.
149  if (walkResult.wasInterrupted())
150  return false;
151 
152  // Dep check depth would be number of enclosing loops + 1.
153  unsigned depth = getNestingDepth(forOp) + 1;
154 
155  // Check dependences between all pairs of ops in 'loadAndStoreOps'.
156  for (auto *srcOp : loadAndStoreOps) {
157  MemRefAccess srcAccess(srcOp);
158  for (auto *dstOp : loadAndStoreOps) {
159  MemRefAccess dstAccess(dstOp);
160  FlatAffineConstraints dependenceConstraints;
161  DependenceResult result = checkMemrefAccessDependence(
162  srcAccess, dstAccess, depth, &dependenceConstraints,
163  /*dependenceComponents=*/nullptr);
164  if (result.value != DependenceResult::NoDependence)
165  return false;
166  }
167  }
168  return true;
169 }

171 /// Returns the sequence of AffineApplyOp Operations operation in
172 /// 'affineApplyOps', which are reachable via a search starting from 'operands',
173 /// and ending at operands which are not defined by AffineApplyOps.
174 // TODO: Add a method to AffineApplyOp which forward substitutes the
175 // AffineApplyOp into any user AffineApplyOps.
176 void mlir::getReachableAffineApplyOps(
177  ArrayRef<Value> operands, SmallVectorImpl<Operation *> &affineApplyOps) {
178  struct State {
179  // The ssa value for this node in the DFS traversal.
180  Value value;
181  // The operand index of 'value' to explore next during DFS traversal.
182  unsigned operandIndex;
183  };
184  SmallVector<State, 4> worklist;
185  for (auto operand : operands) {
186  worklist.push_back({operand, 0});
187  }
188 
189  while (!worklist.empty()) {
190  State &state = worklist.back();
191  auto *opInst = state.value.getDefiningOp();
192  // Note: getDefiningOp will return nullptr if the operand is not an
193  // Operation (i.e. block argument), which is a terminator for the search.
194  if (!isa_and_nonnull<AffineApplyOp>(opInst)) {
195  worklist.pop_back();
196  continue;
197  }
198 
199  if (state.operandIndex == 0) {
200  // Pre-Visit: Add 'opInst' to reachable sequence.
201  affineApplyOps.push_back(opInst);
202  }
203  if (state.operandIndex < opInst->getNumOperands()) {
204  // Visit: Add next 'affineApplyOp' operand to worklist.
205  // Get next operand to visit at 'operandIndex'.
206  auto nextOperand = opInst->getOperand(state.operandIndex);
207  // Increment 'operandIndex' in 'state'.
208  ++state.operandIndex;
209  // Add 'nextOperand' to worklist.
210  worklist.push_back({nextOperand, 0});
211  } else {
212  // Post-visit: done visiting operands AffineApplyOp, pop off stack.
213  worklist.pop_back();
214  }
215  }
216 }

218 // Builds a system of constraints with dimensional identifiers corresponding to
219 // the loop IVs of the forOps appearing in that order. Any symbols founds in
220 // the bound operands are added as symbols in the system. Returns failure for
221 // the yet unimplemented cases.
222 // TODO: Handle non-unit steps through local variables or stride information in
223 // FlatAffineConstraints. (For eg., by using iv - lb % step = 0 and/or by
224 // introducing a method in FlatAffineConstraints setExprStride(ArrayRef<int64_t>
225 // expr, int64_t stride)
226 LogicalResult mlir::getIndexSet(MutableArrayRef<Operation *> ops,
227  FlatAffineConstraints *domain) {
228  SmallVector<Value, 4> indices;
229  SmallVector<AffineForOp, 8> forOps;
230 
231  for (Operation *op : ops) {
232  assert((isa<AffineForOp, AffineIfOp>(op)) &&
233  "ops should have either AffineForOp or AffineIfOp");
234  if (AffineForOp forOp = dyn_cast<AffineForOp>(op))
235  forOps.push_back(forOp);
236  }
237  extractForInductionVars(forOps, &indices);
238  // Reset while associated Values in 'indices' to the domain.
239  domain->reset(forOps.size(), /*numSymbols=*/0, /*numLocals=*/0, indices);
240  for (Operation *op : ops) {
241  // Add constraints from forOp's bounds.
242  if (AffineForOp forOp = dyn_cast<AffineForOp>(op)) {
243  if (failed(domain->addAffineForOpDomain(forOp)))
244  return failure();
245  } else if (AffineIfOp ifOp = dyn_cast<AffineIfOp>(op)) {
246  domain->addAffineIfOpDomain(ifOp);
247  }
248  }
249  return success();
250 }

265 // ValuePositionMap manages the mapping from Values which represent dimension
266 // and symbol identifiers from 'src' and 'dst' access functions to positions
267 // in new space where some Values are kept separate (using addSrc/DstValue)
268 // and some Values are merged (addSymbolValue).
269 // Position lookups return the absolute position in the new space which
270 // has the following format:
271 //
272 // [src-dim-identifiers] [dst-dim-identifiers] [symbol-identifiers]
273 //
274 // Note: access function non-IV dimension identifiers (that have 'dimension'
275 // positions in the access function position space) are assigned as symbols
276 // in the output position space. Convenience access functions which lookup
277 // an Value in multiple maps are provided (i.e. getSrcDimOrSymPos) to handle
278 // the common case of resolving positions for all access function operands.
279 //
280 // TODO: Generalize this: could take a template parameter for the number of maps
281 // (3 in the current case), and lookups could take indices of maps to check. So
282 // getSrcDimOrSymPos would be "getPos(value, {0, 2})".
283 class ValuePositionMap {
284 public:
285  void addSrcValue(Value value) {
286  if (addValueAt(value, &srcDimPosMap, numSrcDims))
287  ++numSrcDims;
288  }
289  void addDstValue(Value value) {
290  if (addValueAt(value, &dstDimPosMap, numDstDims))
291  ++numDstDims;
292  }
293  void addSymbolValue(Value value) {
294  if (addValueAt(value, &symbolPosMap, numSymbols))
295  ++numSymbols;
296  }
297  unsigned getSrcDimOrSymPos(Value value) const {
298  return getDimOrSymPos(value, srcDimPosMap, 0);
299  }
300  unsigned getDstDimOrSymPos(Value value) const {
301  return getDimOrSymPos(value, dstDimPosMap, numSrcDims);
302  }
303  unsigned getSymPos(Value value) const {
304  auto it = symbolPosMap.find(value);
305  assert(it != symbolPosMap.end());
306  return numSrcDims + numDstDims + it->second;
307  }
308 
309  unsigned getNumSrcDims() const { return numSrcDims; }
310  unsigned getNumDstDims() const { return numDstDims; }
311  unsigned getNumDims() const { return numSrcDims + numDstDims; }
312  unsigned getNumSymbols() const { return numSymbols; }
313 
314 private:
315  bool addValueAt(Value value, DenseMap<Value, unsigned> *posMap,
316  unsigned position) {
317  auto it = posMap->find(value);
318  if (it == posMap->end()) {
319  (*posMap)[value] = position;
320  return true;
321  }
322  return false;
323  }
324  unsigned getDimOrSymPos(Value value,
325  const DenseMap<Value, unsigned> &dimPosMap,
326  unsigned dimPosOffset) const {
327  auto it = dimPosMap.find(value);
328  if (it != dimPosMap.end()) {
329  return dimPosOffset + it->second;
330  }
331  it = symbolPosMap.find(value);
332  assert(it != symbolPosMap.end());
333  return numSrcDims + numDstDims + it->second;
334  }
335 
336  unsigned numSrcDims = 0;
337  unsigned numDstDims = 0;
338  unsigned numSymbols = 0;
339  DenseMap<Value, unsigned> srcDimPosMap;
340  DenseMap<Value, unsigned> dstDimPosMap;
341  DenseMap<Value, unsigned> symbolPosMap;
342 };

345 // Builds a map from Value to identifier position in a new merged identifier
346 // list, which is the result of merging dim/symbol lists from src/dst
347 // iteration domains, the format of which is as follows:
348 //
349 // [src-dim-identifiers, dst-dim-identifiers, symbol-identifiers, const_term]
350 //
351 // This method populates 'valuePosMap' with mappings from operand Values in
352 // 'srcAccessMap'/'dstAccessMap' (as well as those in 'srcDomain'/'dstDomain')
353 // to the position of these values in the merged list.
354 static void buildDimAndSymbolPositionMaps(
355  const FlatAffineConstraints &srcDomain,
356  const FlatAffineConstraints &dstDomain, const AffineValueMap &srcAccessMap,
357  const AffineValueMap &dstAccessMap, ValuePositionMap *valuePosMap,
358  FlatAffineConstraints *dependenceConstraints) {
359 
360  // IsDimState is a tri-state boolean. It is used to distinguish three
361  // different cases of the values passed to updateValuePosMap.
362  // - When it is TRUE, we are certain that all values are dim values.
363  // - When it is FALSE, we are certain that all values are symbol values.
364  // - When it is UNKNOWN, we need to further check whether the value is from a
365  // loop IV to determine its type (dim or symbol).
366 
367  // We need this enumeration because sometimes we cannot determine whether a
368  // Value is a symbol or a dim by the information from the Value itself. If a
369  // Value appears in an affine map of a loop, we can determine whether it is a
370  // dim or not by the function `isForInductionVar`. But when a Value is in the
371  // affine set of an if-statement, there is no way to identify its category
372  // (dim/symbol) by itself. Fortunately, the Values to be inserted into
373  // `valuePosMap` come from `srcDomain` and `dstDomain`, and they hold such
374  // information of Value category: `srcDomain` and `dstDomain` organize Values
375  // by their category, such that the position of each Value stored in
376  // `srcDomain` and `dstDomain` marks which category that a Value belongs to.
377  // Therefore, we can separate Values into dim and symbol groups before passing
378  // them to the function `updateValuePosMap`. Specifically, when passing the
379  // dim group, we set IsDimState to TRUE; otherwise, we set it to FALSE.
380  // However, Values from the operands of `srcAccessMap` and `dstAccessMap` are
381  // not explicitly categorized into dim or symbol, and we have to rely on
382  // `isForInductionVar` to make the decision. IsDimState is set to UNKNOWN in
383  // this case.
384  enum IsDimState { TRUE, FALSE, UNKNOWN };
385 
386  // This function places each given Value (in `values`) under a respective
387  // category in `valuePosMap`. Specifically, the placement rules are:
388  // 1) If `isDim` is FALSE, then every value in `values` are inserted into
389  // `valuePosMap` as symbols.
390  // 2) If `isDim` is UNKNOWN and the value of the current iteration is NOT an
391  // induction variable of a for-loop, we treat it as symbol as well.
392  // 3) For other cases, we decide whether to add a value to the `src` or the
393  // `dst` section of the dim category simply by the boolean value `isSrc`.
394  auto updateValuePosMap = [&](ArrayRef<Value> values, bool isSrc,
395  IsDimState isDim) {
396  for (unsigned i = 0, e = values.size(); i < e; ++i) {
397  auto value = values[i];
398  if (isDim == FALSE || (isDim == UNKNOWN && !isForInductionVar(value))) {
399  assert(isValidSymbol(value) &&
400  "access operand has to be either a loop IV or a symbol");
401  valuePosMap->addSymbolValue(value);
402  } else {
403  if (isSrc)
404  valuePosMap->addSrcValue(value);
405  else
406  valuePosMap->addDstValue(value);
407  }
408  }
409  };
410 
411  // Collect values from the src and dst domains. For each domain, we separate
412  // the collected values into dim and symbol parts.
413  SmallVector<Value, 4> srcDimValues, dstDimValues, srcSymbolValues,
414  dstSymbolValues;
415  srcDomain.getIdValues(0, srcDomain.getNumDimIds(), &srcDimValues);
416  dstDomain.getIdValues(0, dstDomain.getNumDimIds(), &dstDimValues);
417  srcDomain.getIdValues(srcDomain.getNumDimIds(),
418  srcDomain.getNumDimAndSymbolIds(), &srcSymbolValues);
419  dstDomain.getIdValues(dstDomain.getNumDimIds(),
420  dstDomain.getNumDimAndSymbolIds(), &dstSymbolValues);
421 
422  // Update value position map with dim values from src iteration domain.
423  updateValuePosMap(srcDimValues, /*isSrc=*/true, /*isDim=*/TRUE);
424  // Update value position map with dim values from dst iteration domain.
425  updateValuePosMap(dstDimValues, /*isSrc=*/false, /*isDim=*/TRUE);
426  // Update value position map with symbols from src iteration domain.
427  updateValuePosMap(srcSymbolValues, /*isSrc=*/true, /*isDim=*/FALSE);
428  // Update value position map with symbols from dst iteration domain.
429  updateValuePosMap(dstSymbolValues, /*isSrc=*/false, /*isDim=*/FALSE);
430  // Update value position map with identifiers from src access function.
431  updateValuePosMap(srcAccessMap.getOperands(), /*isSrc=*/true,
432  /*isDim=*/UNKNOWN);
433  // Update value position map with identifiers from dst access function.
434  updateValuePosMap(dstAccessMap.getOperands(), /*isSrc=*/false,
435  /*isDim=*/UNKNOWN);
436 }

438 // Sets up dependence constraints columns appropriately, in the format:
439 // [src-dim-ids, dst-dim-ids, symbol-ids, local-ids, const_term]
440 static void initDependenceConstraints(
441  const FlatAffineConstraints &srcDomain,
442  const FlatAffineConstraints &dstDomain, const AffineValueMap &srcAccessMap,
443  const AffineValueMap &dstAccessMap, const ValuePositionMap &valuePosMap,
444  FlatAffineConstraints *dependenceConstraints) {
445  // Calculate number of equalities/inequalities and columns required to
446  // initialize FlatAffineConstraints for 'dependenceDomain'.
447  unsigned numIneq =
448  srcDomain.getNumInequalities() + dstDomain.getNumInequalities();
449  AffineMap srcMap = srcAccessMap.getAffineMap();
450  assert(srcMap.getNumResults() == dstAccessMap.getAffineMap().getNumResults());
451  unsigned numEq = srcMap.getNumResults();
452  unsigned numDims = srcDomain.getNumDimIds() + dstDomain.getNumDimIds();
453  unsigned numSymbols = valuePosMap.getNumSymbols();
454  unsigned numLocals = srcDomain.getNumLocalIds() + dstDomain.getNumLocalIds();
455  unsigned numIds = numDims + numSymbols + numLocals;
456  unsigned numCols = numIds + 1;
457 
458  // Set flat affine constraints sizes and reserving space for constraints.
459  dependenceConstraints->reset(numIneq, numEq, numCols, numDims, numSymbols,
460  numLocals);
461 
462  // Set values corresponding to dependence constraint identifiers.
463  SmallVector<Value, 4> srcLoopIVs, dstLoopIVs;
464  srcDomain.getIdValues(0, srcDomain.getNumDimIds(), &srcLoopIVs);
465  dstDomain.getIdValues(0, dstDomain.getNumDimIds(), &dstLoopIVs);
466 
467  dependenceConstraints->setIdValues(0, srcLoopIVs.size(), srcLoopIVs);
468  dependenceConstraints->setIdValues(
469  srcLoopIVs.size(), srcLoopIVs.size() + dstLoopIVs.size(), dstLoopIVs);
470 
471  // Set values for the symbolic identifier dimensions. `isSymbolDetermined`
472  // indicates whether we are certain that the `values` passed in are all
473  // symbols. If `isSymbolDetermined` is true, then we treat every Value in
474  // `values` as a symbol; otherwise, we let the function `isForInductionVar` to
475  // distinguish whether a Value in `values` is a symbol or not.
476  auto setSymbolIds = [&](ArrayRef<Value> values,
477  bool isSymbolDetermined = true) {
478  for (auto value : values) {
479  if (isSymbolDetermined || !isForInductionVar(value)) {
480  assert(isValidSymbol(value) && "expected symbol");
481  dependenceConstraints->setIdValue(valuePosMap.getSymPos(value), value);
482  }
483  }
484  };
485 
486  // We are uncertain about whether all operands in `srcAccessMap` and
487  // `dstAccessMap` are symbols, so we set `isSymbolDetermined` to false.
488  setSymbolIds(srcAccessMap.getOperands(), /*isSymbolDetermined=*/false);
489  setSymbolIds(dstAccessMap.getOperands(), /*isSymbolDetermined=*/false);
490 
491  SmallVector<Value, 8> srcSymbolValues, dstSymbolValues;
492  srcDomain.getIdValues(srcDomain.getNumDimIds(),
493  srcDomain.getNumDimAndSymbolIds(), &srcSymbolValues);
494  dstDomain.getIdValues(dstDomain.getNumDimIds(),
495  dstDomain.getNumDimAndSymbolIds(), &dstSymbolValues);
496  // Since we only take symbol Values out of `srcDomain` and `dstDomain`,
497  // `isSymbolDetermined` is kept to its default value: true.
498  setSymbolIds(srcSymbolValues);
499  setSymbolIds(dstSymbolValues);
500 
501  for (unsigned i = 0, e = dependenceConstraints->getNumDimAndSymbolIds();
502  i < e; i++)
503  assert(dependenceConstraints->getIds()[i].hasValue());
504 }

506 // Adds iteration domain constraints from 'srcDomain' and 'dstDomain' into
507 // 'dependenceDomain'.
508 // Uses 'valuePosMap' to determine the position in 'dependenceDomain' to which a
509 // srcDomain/dstDomain Value maps.
510 static void addDomainConstraints(const FlatAffineConstraints &srcDomain,
511  const FlatAffineConstraints &dstDomain,
512  const ValuePositionMap &valuePosMap,
513  FlatAffineConstraints *dependenceDomain) {
514  unsigned depNumDimsAndSymbolIds = dependenceDomain->getNumDimAndSymbolIds();
515 
516  SmallVector<int64_t, 4> cst(dependenceDomain->getNumCols());
517 
518  auto addDomain = [&](bool isSrc, bool isEq, unsigned localOffset) {
519  const FlatAffineConstraints &domain = isSrc ? srcDomain : dstDomain;
520  unsigned numCsts =
521  isEq ? domain.getNumEqualities() : domain.getNumInequalities();
522  unsigned numDimAndSymbolIds = domain.getNumDimAndSymbolIds();
523  auto at = [&](unsigned i, unsigned j) -> int64_t {
524  return isEq ? domain.atEq(i, j) : domain.atIneq(i, j);
525  };
526  auto map = [&](unsigned i) -> int64_t {
527  return isSrc ? valuePosMap.getSrcDimOrSymPos(domain.getIdValue(i))
528  : valuePosMap.getDstDimOrSymPos(domain.getIdValue(i));
529  };
530 
531  for (unsigned i = 0; i < numCsts; ++i) {
532  // Zero fill.
533  std::fill(cst.begin(), cst.end(), 0);
534  // Set coefficients for identifiers corresponding to domain.
535  for (unsigned j = 0; j < numDimAndSymbolIds; ++j)
536  cst[map(j)] = at(i, j);
537  // Local terms.
538  for (unsigned j = 0, e = domain.getNumLocalIds(); j < e; j++)
539  cst[depNumDimsAndSymbolIds + localOffset + j] =
540  at(i, numDimAndSymbolIds + j);
541  // Set constant term.
542  cst[cst.size() - 1] = at(i, domain.getNumCols() - 1);
543  // Add constraint.
544  if (isEq)
545  dependenceDomain->addEquality(cst);
546  else
547  dependenceDomain->addInequality(cst);
548  }
549  };

552  addDomain(/*isSrc=*/true, /*isEq=*/true, /*localOffset=*/0);

563 // Adds equality constraints that equate src and dst access functions
564 // represented by 'srcAccessMap' and 'dstAccessMap' for each result.
565 // Requires that 'srcAccessMap' and 'dstAccessMap' have the same results count.
566 // For example, given the following two accesses functions to a 2D memref:
567 //
568 // Source access function:
569 // (a0 * d0 + a1 * s0 + a2, b0 * d0 + b1 * s0 + b2)
570 //
571 // Destination access function:
572 // (c0 * d0 + c1 * s0 + c2, f0 * d0 + f1 * s0 + f2)
573 //
574 // This method constructs the following equality constraints in
575 // 'dependenceDomain', by equating the access functions for each result
576 // (i.e. each memref dim). Notice that 'd0' for the destination access function
577 // is mapped into 'd0' in the equality constraint:
578 //
579 // d0 d1 s0 c
580 // -- -- -- --
581 // a0 -c0 (a1 - c1) (a2 - c2) = 0
582 // b0 -f0 (b1 - f1) (b2 - f2) = 0
583 //
584 // Returns failure if any AffineExpr cannot be flattened (due to it being
585 // semi-affine). Returns success otherwise.
586 static LogicalResult
587 addMemRefAccessConstraints(const AffineValueMap &srcAccessMap,
588  const AffineValueMap &dstAccessMap,
589  const ValuePositionMap &valuePosMap,
590  FlatAffineConstraints *dependenceDomain) {
591  AffineMap srcMap = srcAccessMap.getAffineMap();
592  AffineMap dstMap = dstAccessMap.getAffineMap();
593  assert(srcMap.getNumResults() == dstMap.getNumResults());
594  unsigned numResults = srcMap.getNumResults();
595 
596  unsigned srcNumIds = srcMap.getNumDims() + srcMap.getNumSymbols();
597  ArrayRef<Value> srcOperands = srcAccessMap.getOperands();
598 
599  unsigned dstNumIds = dstMap.getNumDims() + dstMap.getNumSymbols();
600  ArrayRef<Value> dstOperands = dstAccessMap.getOperands();
601 
602  std::vector<SmallVector<int64_t, 8>> srcFlatExprs;
603  std::vector<SmallVector<int64_t, 8>> destFlatExprs;
604  FlatAffineConstraints srcLocalVarCst, destLocalVarCst;
605  // Get flattened expressions for the source destination maps.
606  if (failed(getFlattenedAffineExprs(srcMap, &srcFlatExprs, &srcLocalVarCst)) ||
607  failed(getFlattenedAffineExprs(dstMap, &destFlatExprs, &destLocalVarCst)))
608  return failure();
609 
610  unsigned domNumLocalIds = dependenceDomain->getNumLocalIds();
611  unsigned srcNumLocalIds = srcLocalVarCst.getNumLocalIds();
612  unsigned dstNumLocalIds = destLocalVarCst.getNumLocalIds();
613  unsigned numLocalIdsToAdd = srcNumLocalIds + dstNumLocalIds;
614  for (unsigned i = 0; i < numLocalIdsToAdd; i++) {
615  dependenceDomain->addLocalId(dependenceDomain->getNumLocalIds());
616  }
617 
618  unsigned numDims = dependenceDomain->getNumDimIds();
619  unsigned numSymbols = dependenceDomain->getNumSymbolIds();
620  unsigned numSrcLocalIds = srcLocalVarCst.getNumLocalIds();
621  unsigned newLocalIdOffset = numDims + numSymbols + domNumLocalIds;
622 
623  // Equality to add.
624  SmallVector<int64_t, 8> eq(dependenceDomain->getNumCols());
625  for (unsigned i = 0; i < numResults; ++i) {
626  // Zero fill.
627  std::fill(eq.begin(), eq.end(), 0);
628 
629  // Flattened AffineExpr for src result 'i'.
630  const auto &srcFlatExpr = srcFlatExprs[i];
631  // Set identifier coefficients from src access function.
632  for (unsigned j = 0, e = srcOperands.size(); j < e; ++j)
633  eq[valuePosMap.getSrcDimOrSymPos(srcOperands[j])] = srcFlatExpr[j];
634  // Local terms.
635  for (unsigned j = 0, e = srcNumLocalIds; j < e; j++)
636  eq[newLocalIdOffset + j] = srcFlatExpr[srcNumIds + j];
637  // Set constant term.
638  eq[eq.size() - 1] = srcFlatExpr[srcFlatExpr.size() - 1];
639 
640  // Flattened AffineExpr for dest result 'i'.
641  const auto &destFlatExpr = destFlatExprs[i];
642  // Set identifier coefficients from dst access function.
643  for (unsigned j = 0, e = dstOperands.size(); j < e; ++j)
644  eq[valuePosMap.getDstDimOrSymPos(dstOperands[j])] -= destFlatExpr[j];
645  // Local terms.
646  for (unsigned j = 0, e = dstNumLocalIds; j < e; j++)
647  eq[newLocalIdOffset + numSrcLocalIds + j] = -destFlatExpr[dstNumIds + j];
648  // Set constant term.
649  eq[eq.size() - 1] -= destFlatExpr[destFlatExpr.size() - 1];
650 
651  // Add equality constraint.
652  dependenceDomain->addEquality(eq);
653  }
654 
655  // Add equality constraints for any operands that are defined by constant ops.
656  auto addEqForConstOperands = [&](ArrayRef<Value> operands) {
657  for (unsigned i = 0, e = operands.size(); i < e; ++i) {
658  if (isForInductionVar(operands[i]))
659  continue;
660  auto symbol = operands[i];
661  assert(isValidSymbol(symbol));
662  // Check if the symbol is a constant.
663  if (auto cOp = symbol.getDefiningOp<ConstantIndexOp>())
664  dependenceDomain->setIdToConstant(valuePosMap.getSymPos(symbol),
665  cOp.getValue());
666  }
667  };
668 
669  // Add equality constraints for any src symbols defined by constant ops.
670  addEqForConstOperands(srcOperands);
671  // Add equality constraints for any dst symbols defined by constant ops.
672  addEqForConstOperands(dstOperands);
673 
674  // By construction (see flattener), local var constraints will not have any
675  // equalities.
676  assert(srcLocalVarCst.getNumEqualities() == 0 &&
677  destLocalVarCst.getNumEqualities() == 0);
678  // Add inequalities from srcLocalVarCst and destLocalVarCst into the
679  // dependence domain.
680  SmallVector<int64_t, 8> ineq(dependenceDomain->getNumCols());
681  for (unsigned r = 0, e = srcLocalVarCst.getNumInequalities(); r < e; r++) {
682  std::fill(ineq.begin(), ineq.end(), 0);
683 
684  // Set identifier coefficients from src local var constraints.
685  for (unsigned j = 0, e = srcOperands.size(); j < e; ++j)
686  ineq[valuePosMap.getSrcDimOrSymPos(srcOperands[j])] =
687  srcLocalVarCst.atIneq(r, j);
688  // Local terms.
689  for (unsigned j = 0, e = srcNumLocalIds; j < e; j++)
690  ineq[newLocalIdOffset + j] = srcLocalVarCst.atIneq(r, srcNumIds + j);
691  // Set constant term.
692  ineq[ineq.size() - 1] =
693  srcLocalVarCst.atIneq(r, srcLocalVarCst.getNumCols() - 1);
694  dependenceDomain->addInequality(ineq);
695  }
696 
697  for (unsigned r = 0, e = destLocalVarCst.getNumInequalities(); r < e; r++) {
698  std::fill(ineq.begin(), ineq.end(), 0);
699  // Set identifier coefficients from dest local var constraints.
700  for (unsigned j = 0, e = dstOperands.size(); j < e; ++j)
701  ineq[valuePosMap.getDstDimOrSymPos(dstOperands[j])] =
702  destLocalVarCst.atIneq(r, j);
703  // Local terms.
704  for (unsigned j = 0, e = dstNumLocalIds; j < e; j++)
705  ineq[newLocalIdOffset + numSrcLocalIds + j] =
706  destLocalVarCst.atIneq(r, dstNumIds + j);
707  // Set constant term.
708  ineq[ineq.size() - 1] =
709  destLocalVarCst.atIneq(r, destLocalVarCst.getNumCols() - 1);
710 
711  dependenceDomain->addInequality(ineq);
712  }
713  return success();
714 }

716 // Returns the number of outer loop common to 'src/dstDomain'.
717 // Loops common to 'src/dst' domains are added to 'commonLoops' if non-null.
718 static unsigned
719 getNumCommonLoops(const FlatAffineConstraints &srcDomain,
720  const FlatAffineConstraints &dstDomain,
721  SmallVectorImpl<AffineForOp> *commonLoops = nullptr) {
722  // Find the number of common loops shared by src and dst accesses.
723  unsigned minNumLoops =
724  std::min(srcDomain.getNumDimIds(), dstDomain.getNumDimIds());
725  unsigned numCommonLoops = 0;
726  for (unsigned i = 0; i < minNumLoops; ++i) {
727  if (!isForInductionVar(srcDomain.getIdValue(i)) ||
728  !isForInductionVar(dstDomain.getIdValue(i)) ||
729  srcDomain.getIdValue(i) != dstDomain.getIdValue(i))
730  break;
731  if (commonLoops != nullptr)
732  commonLoops->push_back(getForInductionVarOwner(srcDomain.getIdValue(i)));
733  ++numCommonLoops;
734  }
735  if (commonLoops != nullptr)
736  assert(commonLoops->size() == numCommonLoops);
737  return numCommonLoops;
738 }

740 /// Returns Block common to 'srcAccess.opInst' and 'dstAccess.opInst'.
741 static Block *getCommonBlock(const MemRefAccess &srcAccess,
742  const MemRefAccess &dstAccess,
743  const FlatAffineConstraints &srcDomain,
744  unsigned numCommonLoops) {
745  // Get the chain of ancestor blocks to the given `MemRefAccess` instance. The
746  // search terminates when either an op with the `AffineScope` trait or
747  // `endBlock` is reached.
748  auto getChainOfAncestorBlocks = [&](const MemRefAccess &access,
749  SmallVector<Block *, 4> &ancestorBlocks,
750  Block *endBlock = nullptr) {
751  Block *currBlock = access.opInst->getBlock();
752  // Loop terminates when the currBlock is nullptr or equals to the endBlock,
753  // or its parent operation holds an affine scope.
754  while (currBlock && currBlock != endBlock &&
755  !currBlock->getParentOp()->hasTrait<OpTrait::AffineScope>()) {
756  ancestorBlocks.push_back(currBlock);
757  currBlock = currBlock->getParentOp()->getBlock();
758  }
759  };
760 
761  if (numCommonLoops == 0) {
762  Block *block = srcAccess.opInst->getBlock();
763  while (!llvm::isa<FuncOp>(block->getParentOp())) {
764  block = block->getParentOp()->getBlock();
765  }
766  return block;
767  }
768  Value commonForIV = srcDomain.getIdValue(numCommonLoops - 1);
769  AffineForOp forOp = getForInductionVarOwner(commonForIV);
770  assert(forOp && "commonForValue was not an induction variable");
771 
772  // Find the closest common block including those in AffineIf.
773  SmallVector<Block *, 4> srcAncestorBlocks, dstAncestorBlocks;
774  getChainOfAncestorBlocks(srcAccess, srcAncestorBlocks, forOp.getBody());
775  getChainOfAncestorBlocks(dstAccess, dstAncestorBlocks, forOp.getBody());
776 
777  Block *commonBlock = forOp.getBody();
778  for (int i = srcAncestorBlocks.size() - 1, j = dstAncestorBlocks.size() - 1;
779  i >= 0 && j >= 0 && srcAncestorBlocks[i] == dstAncestorBlocks[j];
780  i--, j--)
781  commonBlock = srcAncestorBlocks[i];
782 
783  return commonBlock;
784 }

786 // Returns true if the ancestor operation of 'srcAccess' appears before the
787 // ancestor operation of 'dstAccess' in the common ancestral block. Returns
788 // false otherwise.
789 // Note that because 'srcAccess' or 'dstAccess' may be nested in conditionals,
790 // the function is named 'srcAppearsBeforeDstInCommonBlock'. Note that
791 // 'numCommonLoops' is the number of contiguous surrounding outer loops.
792 static bool srcAppearsBeforeDstInAncestralBlock(
793  const MemRefAccess &srcAccess, const MemRefAccess &dstAccess,
794  const FlatAffineConstraints &srcDomain, unsigned numCommonLoops) {
795  // Get Block common to 'srcAccess.opInst' and 'dstAccess.opInst'.
796  auto *commonBlock =
797  getCommonBlock(srcAccess, dstAccess, srcDomain, numCommonLoops);
798  // Check the dominance relationship between the respective ancestors of the
799  // src and dst in the Block of the innermost among the common loops.
800  auto *srcInst = commonBlock->findAncestorOpInBlock(*srcAccess.opInst);
801  assert(srcInst != nullptr);
802  auto *dstInst = commonBlock->findAncestorOpInBlock(*dstAccess.opInst);
803  assert(dstInst != nullptr);
804 
805  // Determine whether dstInst comes after srcInst.
806  return srcInst->isBeforeInBlock(dstInst);
807 }

809 // Adds ordering constraints to 'dependenceDomain' based on number of loops
810 // common to 'src/dstDomain' and requested 'loopDepth'.
811 // Note that 'loopDepth' cannot exceed the number of common loops plus one.
812 // EX: Given a loop nest of depth 2 with IVs 'i' and 'j':
813 // *) If 'loopDepth == 1' then one constraint is added: i' >= i + 1
814 // *) If 'loopDepth == 2' then two constraints are added: i == i' and j' > j + 1
815 // *) If 'loopDepth == 3' then two constraints are added: i == i' and j == j'
816 static void addOrderingConstraints(const FlatAffineConstraints &srcDomain,
817  const FlatAffineConstraints &dstDomain,
818  unsigned loopDepth,
819  FlatAffineConstraints *dependenceDomain) {
820  unsigned numCols = dependenceDomain->getNumCols();
821  SmallVector<int64_t, 4> eq(numCols);
822  unsigned numSrcDims = srcDomain.getNumDimIds();
823  unsigned numCommonLoops = getNumCommonLoops(srcDomain, dstDomain);
824  unsigned numCommonLoopConstraints = std::min(numCommonLoops, loopDepth);
825  for (unsigned i = 0; i < numCommonLoopConstraints; ++i) {
826  std::fill(eq.begin(), eq.end(), 0);
827  eq[i] = -1;
828  eq[i + numSrcDims] = 1;
829  if (i == loopDepth - 1) {
830  eq[numCols - 1] = -1;
831  dependenceDomain->addInequality(eq);
832  } else {
833  dependenceDomain->addEquality(eq);
834  }
835  }
836 }

838 // Computes distance and direction vectors in 'dependences', by adding
839 // variables to 'dependenceDomain' which represent the difference of the IVs,
840 // eliminating all other variables, and reading off distance vectors from
841 // equality constraints (if possible), and direction vectors from inequalities.
842 static void computeDirectionVector(
843  const FlatAffineConstraints &srcDomain,
844  const FlatAffineConstraints &dstDomain, unsigned loopDepth,
845  FlatAffineConstraints *dependenceDomain,
846  SmallVector<DependenceComponent, 2> *dependenceComponents) {
847  // Find the number of common loops shared by src and dst accesses.
848  SmallVector<AffineForOp, 4> commonLoops;
849  unsigned numCommonLoops =
850  getNumCommonLoops(srcDomain, dstDomain, &commonLoops);
851  if (numCommonLoops == 0)
852  return;
853  // Compute direction vectors for requested loop depth.
854  unsigned numIdsToEliminate = dependenceDomain->getNumIds();
855  // Add new variables to 'dependenceDomain' to represent the direction
856  // constraints for each shared loop.
857  for (unsigned j = 0; j < numCommonLoops; ++j) {
858  dependenceDomain->addDimId(j);
859  }
860 
861  // Add equality constraints for each common loop, setting newly introduced
862  // variable at column 'j' to the 'dst' IV minus the 'src IV.
863  SmallVector<int64_t, 4> eq;
864  eq.resize(dependenceDomain->getNumCols());
865  unsigned numSrcDims = srcDomain.getNumDimIds();
866  // Constraint variables format:
867  // [num-common-loops][num-src-dim-ids][num-dst-dim-ids][num-symbols][constant]
868  for (unsigned j = 0; j < numCommonLoops; ++j) {
869  std::fill(eq.begin(), eq.end(), 0);
870  eq[j] = 1;
871  eq[j + numCommonLoops] = 1;
872  eq[j + numCommonLoops + numSrcDims] = -1;
873  dependenceDomain->addEquality(eq);
874  }
875 
876  // Eliminate all variables other than the direction variables just added.
877  dependenceDomain->projectOut(numCommonLoops, numIdsToEliminate);
878 
879  // Scan each common loop variable column and set direction vectors based
880  // on eliminated constraint system.
881  dependenceComponents->resize(numCommonLoops);
882  for (unsigned j = 0; j < numCommonLoops; ++j) {
883  (*dependenceComponents)[j].op = commonLoops[j].getOperation();
884  auto lbConst = dependenceDomain->getConstantLowerBound(j);
885  (*dependenceComponents)[j].lb =
886  lbConst.getValueOr(std::numeric_limits<int64_t>::min());
887  auto ubConst = dependenceDomain->getConstantUpperBound(j);
888  (*dependenceComponents)[j].ub =
889  ubConst.getValueOr(std::numeric_limits<int64_t>::max());
890  }
891 }

910 // Builds a flat affine constraint system to check if there exists a dependence
911 // between memref accesses 'srcAccess' and 'dstAccess'.
912 // Returns 'NoDependence' if the accesses can be definitively shown not to
913 // access the same element.
914 // Returns 'HasDependence' if the accesses do access the same element.
915 // Returns 'Failure' if an error or unsupported case was encountered.
916 // If a dependence exists, returns in 'dependenceComponents' a direction
917 // vector for the dependence, with a component for each loop IV in loops
918 // common to both accesses (see Dependence in AffineAnalysis.h for details).
919 //
920 // The memref access dependence check is comprised of the following steps:
921 // *) Compute access functions for each access. Access functions are computed
922 // using AffineValueMaps initialized with the indices from an access, then
923 // composed with AffineApplyOps reachable from operands of that access,
924 // until operands of the AffineValueMap are loop IVs or symbols.
925 // *) Build iteration domain constraints for each access. Iteration domain
926 // constraints are pairs of inequality constraints representing the
927 // upper/lower loop bounds for each AffineForOp in the loop nest associated
928 // with each access.
929 // *) Build dimension and symbol position maps for each access, which map
930 // Values from access functions and iteration domains to their position
931 // in the merged constraint system built by this method.
932 //
933 // This method builds a constraint system with the following column format:
934 //
935 // [src-dim-identifiers, dst-dim-identifiers, symbols, constant]
936 //
937 // For example, given the following MLIR code with "source" and "destination"
938 // accesses to the same memref label, and symbols %M, %N, %K:
939 //
940 // affine.for %i0 = 0 to 100 {
941 // affine.for %i1 = 0 to 50 {
942 // %a0 = affine.apply
943 // (d0, d1) -> (d0 * 2 - d1 * 4 + s1, d1 * 3 - s0) (%i0, %i1)[%M, %N]
944 // // Source memref access.
945 // store %v0, %m[%a0#0, %a0#1] : memref<4x4xf32>
946 // }
947 // }
948 //
949 // affine.for %i2 = 0 to 100 {
950 // affine.for %i3 = 0 to 50 {
951 // %a1 = affine.apply
952 // (d0, d1) -> (d0 * 7 + d1 * 9 - s1, d1 * 11 + s0) (%i2, %i3)[%K, %M]
953 // // Destination memref access.
954 // %v1 = load %m[%a1#0, %a1#1] : memref<4x4xf32>
955 // }
956 // }
957 //
958 // The access functions would be the following:
959 //
960 // src: (%i0 * 2 - %i1 * 4 + %N, %i1 * 3 - %M)
961 // dst: (%i2 * 7 + %i3 * 9 - %M, %i3 * 11 - %K)
962 //
963 // The iteration domains for the src/dst accesses would be the following:
964 //
965 // src: 0 <= %i0 <= 100, 0 <= %i1 <= 50
966 // dst: 0 <= %i2 <= 100, 0 <= %i3 <= 50
967 //
968 // The symbols by both accesses would be assigned to a canonical position order
969 // which will be used in the dependence constraint system:
970 //
971 // symbol name: %M %N %K
972 // symbol pos: 0 1 2
973 //
974 // Equality constraints are built by equating each result of src/destination
975 // access functions. For this example, the following two equality constraints
976 // will be added to the dependence constraint system:
977 //
978 // [src_dim0, src_dim1, dst_dim0, dst_dim1, sym0, sym1, sym2, const]
979 // 2 -4 -7 -9 1 1 0 0 = 0
980 // 0 3 0 -11 -1 0 1 0 = 0
981 //
982 // Inequality constraints from the iteration domain will be meged into
983 // the dependence constraint system
984 //
985 // [src_dim0, src_dim1, dst_dim0, dst_dim1, sym0, sym1, sym2, const]
986 // 1 0 0 0 0 0 0 0 >= 0
987 // -1 0 0 0 0 0 0 100 >= 0
988 // 0 1 0 0 0 0 0 0 >= 0
989 // 0 -1 0 0 0 0 0 50 >= 0
990 // 0 0 1 0 0 0 0 0 >= 0
991 // 0 0 -1 0 0 0 0 100 >= 0
992 // 0 0 0 1 0 0 0 0 >= 0
993 // 0 0 0 -1 0 0 0 50 >= 0
994 //
995 //
996 // TODO: Support AffineExprs mod/floordiv/ceildiv.
997 DependenceResult mlir::checkMemrefAccessDependence(
998  const MemRefAccess &srcAccess, const MemRefAccess &dstAccess,
999  unsigned loopDepth, FlatAffineConstraints *dependenceConstraints,
1000  SmallVector<DependenceComponent, 2> *dependenceComponents, bool allowRAR) {
1001  LLVM_DEBUG(llvm::dbgs() << "Checking for dependence at depth: "
1002  << Twine(loopDepth) << " between:\n";);
1003  LLVM_DEBUG(srcAccess.opInst->dump(););
1004  LLVM_DEBUG(dstAccess.opInst->dump(););
1005 
1006  // Return 'NoDependence' if these accesses do not access the same memref.
1007  if (srcAccess.memref != dstAccess.memref)
1008  return DependenceResult::NoDependence;
1009 
1010  // Return 'NoDependence' if one of these accesses is not an
1011  // AffineWriteOpInterface.
1012  if (!allowRAR && !isa<AffineWriteOpInterface>(srcAccess.opInst) &&
1013  !isa<AffineWriteOpInterface>(dstAccess.opInst))
1014  return DependenceResult::NoDependence;
1015 
1016  // Get composed access function for 'srcAccess'.
1017  AffineValueMap srcAccessMap;
1018  srcAccess.getAccessMap(&srcAccessMap);
1019 
1020  // Get composed access function for 'dstAccess'.
1021  AffineValueMap dstAccessMap;
1022  dstAccess.getAccessMap(&dstAccessMap);
1023 
1024  // Get iteration domain for the 'srcAccess' operation.
1025  FlatAffineConstraints srcDomain;
1026  if (failed(getOpIndexSet(srcAccess.opInst, &srcDomain)))
1027  return DependenceResult::Failure;
1028 
1029  // Get iteration domain for 'dstAccess' operation.
1030  FlatAffineConstraints dstDomain;
1031  if (failed(getOpIndexSet(dstAccess.opInst, &dstDomain)))
1032  return DependenceResult::Failure;
1033 
1034  // Return 'NoDependence' if loopDepth > numCommonLoops and if the ancestor
1035  // operation of 'srcAccess' does not properly dominate the ancestor
1036  // operation of 'dstAccess' in the same common operation block.
1037  // Note: this check is skipped if 'allowRAR' is true, because because RAR
1038  // deps can exist irrespective of lexicographic ordering b/w src and dst.
1039  unsigned numCommonLoops = getNumCommonLoops(srcDomain, dstDomain);
1040  assert(loopDepth <= numCommonLoops + 1);
1041  if (!allowRAR && loopDepth > numCommonLoops &&
1042  !srcAppearsBeforeDstInAncestralBlock(srcAccess, dstAccess, srcDomain,
1043  numCommonLoops)) {
1044  return DependenceResult::NoDependence;
1045  }
1046  // Build dim and symbol position maps for each access from access operand
1047  // Value to position in merged constraint system.
1048  ValuePositionMap valuePosMap;
1049  buildDimAndSymbolPositionMaps(srcDomain, dstDomain, srcAccessMap,
1050  dstAccessMap, &valuePosMap,
1051  dependenceConstraints);
1052  initDependenceConstraints(srcDomain, dstDomain, srcAccessMap, dstAccessMap,
1053  valuePosMap, dependenceConstraints);
1054 
1055  assert(valuePosMap.getNumDims() ==
1056  srcDomain.getNumDimIds() + dstDomain.getNumDimIds());
1057 
1058  // Create memref access constraint by equating src/dst access functions.
