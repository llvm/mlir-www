34 /// Return the ancestor op in the region or nullptr if the region is not
35 /// an ancestor of the op.
36 static Operation *findAncestorOpInRegion(Region *region, Operation *op) {
37  for (; op != nullptr && op->getParentRegion() != region;
38  op = op->getParentOp())
39  ;
40  return op;

66 /// Return true if there is a path from start operation to dest operation,
67 /// otherwise return false. The operations have to be in the same region.
68 bool TransferOptimization::isReachable(Operation *start, Operation *dest) {
69  assert(start->getParentRegion() == dest->getParentRegion() &&
70  "This function only works for ops i the same region");
71  // Simple case where the start op dominate the destination.
72  if (dominators.dominates(start, dest))
73  return true;
74  return start->getBlock()->isReachable(dest->getBlock());
75 }

77 /// For transfer_write to overwrite fully another transfer_write must:
78 /// 1. Access the same memref with the same indices and vector type.
79 /// 2. Post-dominate the other transfer_write operation.
80 /// If several candidates are available, one must be post-dominated by all the
81 /// others since they are all post-dominating the same transfer_write. We only
82 /// consider the transfer_write post-dominated by all the other candidates as
83 /// this will be the first transfer_write executed after the potentially dead
84 /// transfer_write.
85 /// If we found such an overwriting transfer_write we know that the original
86 /// transfer_write is dead if all reads that can be reached from the potentially
87 /// dead transfer_write are dominated by the overwriting transfer_write.
88 void TransferOptimization::deadStoreOp(vector::TransferWriteOp write) {
89  LDBG() << "Candidate for dead store: " << *write.getOperation();
90  llvm::SmallVector<Operation *, 8> blockingAccesses;
91  Operation *firstOverwriteCandidate = nullptr;
92  Value source = memref::skipViewLikeOps(cast<MemrefValue>(write.getBase()));
93  llvm::SmallVector<Operation *, 32> users(source.getUsers().begin(),
94  source.getUsers().end());
95  llvm::SmallDenseSet<Operation *, 32> processed;
96  while (!users.empty()) {
97  Operation *user = users.pop_back_val();
98  // If the user has already been processed skip.
99  if (!processed.insert(user).second)
100  continue;
101  if (isa<ViewLikeOpInterface>(user)) {
102  users.append(user->getUsers().begin(), user->getUsers().end());
103  continue;
104  }
105  if (isMemoryEffectFree(user))
106  continue;
107  if (user == write.getOperation())
108  continue;
109  if (auto nextWrite = dyn_cast<vector::TransferWriteOp>(user)) {
110  // Check candidate that can override the store.
111  if (memref::isSameViewOrTrivialAlias(
112  cast<MemrefValue>(nextWrite.getBase()),
113  cast<MemrefValue>(write.getBase())) &&
114  checkSameValueWAW(nextWrite, write) &&
115  postDominators.postDominates(nextWrite, write)) {
116  if (firstOverwriteCandidate == nullptr ||
117  postDominators.postDominates(firstOverwriteCandidate, nextWrite))
118  firstOverwriteCandidate = nextWrite;
119  else
120  assert(
121  postDominators.postDominates(nextWrite, firstOverwriteCandidate));
122  continue;
123  }
124  }
125  if (auto transferOp = dyn_cast<VectorTransferOpInterface>(user)) {
126  // Don't need to consider disjoint accesses.
127  if (vector::isDisjointTransferSet(
128  cast<VectorTransferOpInterface>(write.getOperation()),
129  cast<VectorTransferOpInterface>(transferOp.getOperation()),
130  /*testDynamicValueUsingBounds=*/true))
131  continue;
132  }
133  blockingAccesses.push_back(user);
134  }
135  if (firstOverwriteCandidate == nullptr)
136  return;
137  Region *topRegion = firstOverwriteCandidate->getParentRegion();
138  Operation *writeAncestor = findAncestorOpInRegion(topRegion, write);
139  assert(writeAncestor &&
140  "write op should be recursively part of the top region");
141 
142  for (Operation *access : blockingAccesses) {
143  Operation *accessAncestor = findAncestorOpInRegion(topRegion, access);
144  // TODO: if the access and write have the same ancestor we could recurse in
145  // the region to know if the access is reachable with more precision.
146  if (accessAncestor == nullptr ||
147  !isReachable(writeAncestor, accessAncestor))
148  continue;
149  if (!dominators.dominates(firstOverwriteCandidate, accessAncestor)) {
150  LDBG() << "Store may not be dead due to op: " << *accessAncestor;
151  return;
152  }
153  }
154  LDBG() << "Found dead store: " << *write.getOperation()
155  << " overwritten by: " << *firstOverwriteCandidate;
156  opToErase.push_back(write.getOperation());
157 }

159 /// A transfer_write candidate to storeToLoad forwarding must:
160 /// 1. Access the same memref with the same indices and vector type as the
161 /// transfer_read.
162 /// 2. Dominate the transfer_read operation.
163 /// If several candidates are available, one must be dominated by all the others
164 /// since they are all dominating the same transfer_read. We only consider the
165 /// transfer_write dominated by all the other candidates as this will be the
166 /// last transfer_write executed before the transfer_read.
167 /// If we found such a candidate we can do the forwarding if all the other
168 /// potentially aliasing ops that may reach the transfer_read are post-dominated
169 /// by the transfer_write.
170 void TransferOptimization::storeToLoadForwarding(vector::TransferReadOp read) {
171  if (read.hasOutOfBoundsDim())
172  return;
173  LDBG() << "Candidate for Forwarding: " << *read.getOperation();
174  SmallVector<Operation *, 8> blockingWrites;
175  vector::TransferWriteOp lastwrite = nullptr;
176  Value source = memref::skipViewLikeOps(cast<MemrefValue>(read.getBase()));
177  llvm::SmallVector<Operation *, 32> users(source.getUsers().begin(),
178  source.getUsers().end());
179  llvm::SmallDenseSet<Operation *, 32> processed;
180  while (!users.empty()) {
181  Operation *user = users.pop_back_val();
182  // If the user has already been processed skip.
183  if (!processed.insert(user).second)
184  continue;
185  if (isa<ViewLikeOpInterface>(user)) {
186  users.append(user->getUsers().begin(), user->getUsers().end());
187  continue;
188  }
189  if (isMemoryEffectFree(user) || isa<vector::TransferReadOp>(user))
190  continue;
191  if (auto write = dyn_cast<vector::TransferWriteOp>(user)) {
192  // If there is a write, but we can prove that it is disjoint we can ignore
193  // the write.
194  if (vector::isDisjointTransferSet(
195  cast<VectorTransferOpInterface>(write.getOperation()),
196  cast<VectorTransferOpInterface>(read.getOperation()),
197  /*testDynamicValueUsingBounds=*/true))
198  continue;
199  if (memref::isSameViewOrTrivialAlias(
200  cast<MemrefValue>(read.getBase()),
201  cast<MemrefValue>(write.getBase())) &&
202  dominators.dominates(write, read) && checkSameValueRAW(write, read)) {
203  if (lastwrite == nullptr || dominators.dominates(lastwrite, write))
204  lastwrite = write;
205  else
206  assert(dominators.dominates(write, lastwrite));
207  continue;
208  }
209  }
210  blockingWrites.push_back(user);
211  }
212 
213  if (lastwrite == nullptr)
214  return;
215 
216  Region *topRegion = lastwrite->getParentRegion();
217  Operation *readAncestor = findAncestorOpInRegion(topRegion, read);
218  assert(readAncestor &&
219  "read op should be recursively part of the top region");
220 
221  for (Operation *write : blockingWrites) {
222  Operation *writeAncestor = findAncestorOpInRegion(topRegion, write);
223  // TODO: if the store and read have the same ancestor we could recurse in
224  // the region to know if the read is reachable with more precision.
225  if (writeAncestor == nullptr || !isReachable(writeAncestor, readAncestor))
226  continue;
227  if (!postDominators.postDominates(lastwrite, write)) {
228  LDBG() << "Fail to do write to read forwarding due to op: " << *write;
229  return;
230  }
231  }
232 
233  LDBG() << "Forward value from " << *lastwrite.getOperation()
234  << " to: " << *read.getOperation();
235  read.replaceAllUsesWith(lastwrite.getVector());
236  opToErase.push_back(read.getOperation());
237 }

239 /// Converts OpFoldResults to int64_t shape without unit dims.
240 static SmallVector<int64_t> getReducedShape(ArrayRef<OpFoldResult> mixedSizes) {
241  SmallVector<int64_t> reducedShape;
242  for (const auto size : mixedSizes) {
243  if (llvm::dyn_cast_if_present<Value>(size)) {
244  reducedShape.push_back(ShapedType::kDynamic);
245  continue;
246  }
247 
248  auto value = cast<IntegerAttr>(cast<Attribute>(size)).getValue();
249  if (value == 1)
250  continue;
251  reducedShape.push_back(value.getSExtValue());
252  }
253  return reducedShape;
254 }

267 /// Creates a rank-reducing memref.subview op that drops unit dims from its
268 /// input. Or just returns the input if it was already without unit dims.
269 static Value rankReducingSubviewDroppingUnitDims(PatternRewriter &rewriter,
270  mlir::Location loc,
271  Value input) {
272  MemRefType inputType = cast<MemRefType>(input.getType());
273  SmallVector<OpFoldResult> offsets(inputType.getRank(),
274  rewriter.getIndexAttr(0));
275  SmallVector<OpFoldResult> sizes = memref::getMixedSizes(rewriter, loc, input);
276  SmallVector<OpFoldResult> strides(inputType.getRank(),
277  rewriter.getIndexAttr(1));
278  MemRefType resultType = dropUnitDims(inputType, offsets, sizes, strides);
279 
280  if (resultType.canonicalizeStridedLayout() ==
281  inputType.canonicalizeStridedLayout())
282  return input;
283  return memref::SubViewOp::create(rewriter, loc, resultType, input, offsets,
284  sizes, strides);
285 }

292 /// Trims non-scalable one dimensions from `oldType` and returns the result
293 /// type.
294 static VectorType trimNonScalableUnitDims(VectorType oldType) {
295  SmallVector<int64_t> newShape;
296  SmallVector<bool> newScalableDims;
297  for (auto [dimIdx, dimSize] : llvm::enumerate(oldType.getShape())) {
298  if (dimSize == 1 && !oldType.getScalableDims()[dimIdx])
299  continue;
300  newShape.push_back(dimSize);
301  newScalableDims.push_back(oldType.getScalableDims()[dimIdx]);
302  }
303  return VectorType::get(newShape, oldType.getElementType(), newScalableDims);
304 }

306 // Rewrites vector.create_mask 'op' to drop non-scalable one dimensions.
307 static FailureOr<Value>
308 createMaskDropNonScalableUnitDims(PatternRewriter &rewriter, Location loc,
309  vector::CreateMaskOp op) {
310  auto type = op.getType();
311  VectorType reducedType = trimNonScalableUnitDims(type);
312  if (reducedType.getRank() == type.getRank())
313  return failure();
314 
315  SmallVector<Value> reducedOperands;
316  for (auto [dim, dimIsScalable, operand] : llvm::zip_equal(
317  type.getShape(), type.getScalableDims(), op.getOperands())) {
318  if (dim == 1 && !dimIsScalable) {
319  // If the mask for the unit dim is not a constant of 1, do nothing.
320  auto constant = operand.getDefiningOp<arith::ConstantIndexOp>();
321  if (!constant || (constant.value() != 1))
322  return failure();
323  continue;
324  }
325  reducedOperands.push_back(operand);
326  }
327  return vector::CreateMaskOp::create(rewriter, loc, reducedType,
328  reducedOperands)
329  .getResult();
330 }

334 /// Rewrites `vector.transfer_read` ops where the source has unit dims, by
335 /// inserting a memref.subview dropping those unit dims. The vector shapes are
336 /// also reduced accordingly.
337 class TransferReadDropUnitDimsPattern
338  : public vector::MaskableOpRewritePattern<vector::TransferReadOp> {
339  using MaskableOpRewritePattern::MaskableOpRewritePattern;
340 
341  FailureOr<Value>
342  matchAndRewriteMaskableOp(vector::TransferReadOp transferReadOp,
343  vector::MaskingOpInterface maskingOp,
344  PatternRewriter &rewriter) const override {
345  auto loc = transferReadOp.getLoc();
346  Value vector = transferReadOp.getVector();
347  VectorType vectorType = cast<VectorType>(vector.getType());
348  Value source = transferReadOp.getBase();
349  MemRefType sourceType = dyn_cast<MemRefType>(source.getType());
350  // TODO: support tensor types.
351  if (!sourceType)
352  return failure();
353  // TODO: generalize this pattern, relax the requirements here.
354  if (transferReadOp.hasOutOfBoundsDim())
355  return failure();
356  if (!transferReadOp.getPermutationMap().isMinorIdentity())
357  return failure();
358  // Check if the source shape can be further reduced.
359  int reducedRank = getReducedRank(sourceType.getShape());
360  if (reducedRank == sourceType.getRank())
361  return failure();
362  // TODO: Extend vector.mask to support 0-d vectors. In the meantime, bail
363  // out.
364  if (reducedRank == 0 && maskingOp)
365  return failure();
366  // Check if the reduced vector shape matches the reduced source shape.
367  // Otherwise, this case is not supported yet.
368  VectorType reducedVectorType = trimNonScalableUnitDims(vectorType);
369  if (reducedRank != reducedVectorType.getRank())
370  return failure();
371  if (llvm::any_of(transferReadOp.getIndices(), [](Value v) {
372  return getConstantIntValue(v) != static_cast<int64_t>(0);
373  }))
374  return failure();
375 
376  Value maskOp = transferReadOp.getMask();
377  if (maskOp) {
378  auto createMaskOp = maskOp.getDefiningOp<vector::CreateMaskOp>();
379  if (!createMaskOp)
380  return rewriter.notifyMatchFailure(
381  transferReadOp, "unsupported mask op, only 'vector.create_mask' is "
382  "currently supported");
383  FailureOr<Value> rankReducedCreateMask =
384  createMaskDropNonScalableUnitDims(rewriter, loc, createMaskOp);
385  if (failed(rankReducedCreateMask))
386  return failure();
387  maskOp = *rankReducedCreateMask;
388  }
389 
390  Value reducedShapeSource =
391  rankReducingSubviewDroppingUnitDims(rewriter, loc, source);
392  Value c0 = arith::ConstantIndexOp::create(rewriter, loc, 0);
393  SmallVector<Value> zeros(reducedRank, c0);
394  auto identityMap = rewriter.getMultiDimIdentityMap(reducedRank);
395  SmallVector<bool> inBounds(reducedVectorType.getRank(), true);
396  Operation *newTransferReadOp = vector::TransferReadOp::create(
397  rewriter, loc, reducedVectorType, reducedShapeSource, zeros,
398  identityMap, transferReadOp.getPadding(), maskOp,
399  rewriter.getBoolArrayAttr(inBounds));
400 
401  if (maskingOp) {
402  auto shapeCastMask = rewriter.createOrFold<vector::ShapeCastOp>(
403  loc, reducedVectorType.cloneWith(std::nullopt, rewriter.getI1Type()),
404  maskingOp.getMask());
405  newTransferReadOp = mlir::vector::maskOperation(
406  rewriter, newTransferReadOp, shapeCastMask);
407  }
408 
409  auto shapeCast = rewriter.createOrFold<vector::ShapeCastOp>(
410  loc, vectorType, newTransferReadOp->getResults()[0]);
411 
412  return shapeCast;
413  }
414 };

503 /// Creates a memref.collapse_shape collapsing all inner dimensions of the
504 /// input starting at `firstDimToCollapse`.
505 static Value collapseInnerDims(PatternRewriter &rewriter, mlir::Location loc,
506  Value input, int64_t firstDimToCollapse) {
507  ShapedType inputType = cast<ShapedType>(input.getType());
508  if (inputType.getRank() == 1)
509  return input;
510  SmallVector<ReassociationIndices> reassociation;
511  for (int64_t i = 0; i < firstDimToCollapse; ++i)
512  reassociation.push_back(ReassociationIndices{i});
513  ReassociationIndices collapsedIndices;
514  for (int64_t i = firstDimToCollapse; i < inputType.getRank(); ++i)
515  collapsedIndices.push_back(i);
516  reassociation.push_back(collapsedIndices);
517  return memref::CollapseShapeOp::create(rewriter, loc, input, reassociation);
518 }

520 /// Returns the new indices that collapses the inner dimensions starting from
521 /// the `firstDimToCollapse` dimension.
522 static SmallVector<Value> getCollapsedIndices(RewriterBase &rewriter,
523  Location loc,
524  ArrayRef<int64_t> shape,
525  ValueRange indices,
526  int64_t firstDimToCollapse) {
527  assert(firstDimToCollapse < static_cast<int64_t>(indices.size()));
528 
529  // If all the collapsed indices are zero then no extra logic is needed.
530  // Otherwise, a new offset/index has to be computed.
531  SmallVector<Value> indicesAfterCollapsing(
532  indices.begin(), indices.begin() + firstDimToCollapse);
533  SmallVector<Value> indicesToCollapse(indices.begin() + firstDimToCollapse,
534  indices.end());
535  if (llvm::all_of(indicesToCollapse, isZeroInteger)) {
536  indicesAfterCollapsing.push_back(indicesToCollapse[0]);
537  return indicesAfterCollapsing;
538  }
539 
540  // Compute the remaining trailing index/offset required for reading from
541  // the collapsed memref:
542  //
543  // offset = 0
544  // for (i = firstDimToCollapse; i < outputRank; ++i)
545  // offset += sourceType.getDimSize(i) * transferReadOp.indices[i]
546  //
547  // For this example:
548  // %2 = vector.transfer_read/write %arg4[%c0, %arg0, %c0] (...) :
549  // memref<1x43x2xi32>, vector<1x2xi32>
550  // which would be collapsed to:
551  // %1 = vector.transfer_read/write %collapse_shape[%c0, %offset] (...) :
552  // memref<1x86xi32>, vector<2xi32>
553  // one would get the following offset:
554  // %offset = %arg0 * 43
555  OpFoldResult collapsedOffset =
556  arith::ConstantIndexOp::create(rewriter, loc, 0).getResult();
557 
558  auto collapsedStrides = computeSuffixProduct(
559  ArrayRef<int64_t>(shape.begin() + firstDimToCollapse, shape.end()));
560 
561  // Compute the collapsed offset.
562  auto &&[collapsedExpr, collapsedVals] =
563  computeLinearIndex(collapsedOffset, collapsedStrides, indicesToCollapse);
564  collapsedOffset = affine::makeComposedFoldedAffineApply(
565  rewriter, loc, collapsedExpr, collapsedVals);
566 
567  if (auto value = dyn_cast<Value>(collapsedOffset)) {
568  indicesAfterCollapsing.push_back(value);
569  } else {
570  indicesAfterCollapsing.push_back(arith::ConstantIndexOp::create(
571  rewriter, loc, *getConstantIntValue(collapsedOffset)));

578 /// Rewrites contiguous row-major vector.transfer_read ops by inserting
579 /// memref.collapse_shape on the source so that the resulting
580 /// vector.transfer_read has a 1D source. Requires the source shape to be
581 /// already reduced i.e. without unit dims.
582 ///
583 /// If `targetVectorBitwidth` is provided, the flattening will only happen if
584 /// the trailing dimension of the vector read is smaller than the provided
585 /// bitwidth.
586 class FlattenContiguousRowMajorTransferReadPattern
587  : public OpRewritePattern<vector::TransferReadOp> {
588 public:
589  FlattenContiguousRowMajorTransferReadPattern(MLIRContext *context,
590  unsigned vectorBitwidth,
591  PatternBenefit benefit)
592  : OpRewritePattern<vector::TransferReadOp>(context, benefit),
593  targetVectorBitwidth(vectorBitwidth) {}
594 
595  LogicalResult matchAndRewrite(vector::TransferReadOp transferReadOp,
596  PatternRewriter &rewriter) const override {
597  auto loc = transferReadOp.getLoc();
598  Value vector = transferReadOp.getVector();
599  VectorType vectorType = cast<VectorType>(vector.getType());
600  auto source = transferReadOp.getBase();
601  MemRefType sourceType = dyn_cast<MemRefType>(source.getType());
602 
603  // 0. Check pre-conditions
604  // Contiguity check is valid on tensors only.
605  if (!sourceType)
606  return failure();
607  // If this is already 0D/1D, there's nothing to do.
608  if (vectorType.getRank() <= 1)
609  return failure();
610  if (!vectorType.getElementType().isSignlessIntOrFloat())
611  return failure();
612  unsigned trailingVectorDimBitwidth =
613  vectorType.getShape().back() * vectorType.getElementTypeBitWidth();
614  if (trailingVectorDimBitwidth >= targetVectorBitwidth)
615  return failure();
616  if (!vector::isContiguousSlice(sourceType, vectorType))
617  return failure();
618  // TODO: generalize this pattern, relax the requirements here.
619  if (transferReadOp.hasOutOfBoundsDim())
620  return failure();
621  if (!transferReadOp.getPermutationMap().isMinorIdentity())
622  return failure();
623  if (transferReadOp.getMask())
624  return failure();
625 
626  // Determine the first memref dimension to collapse - just enough so we can
627  // read a flattened vector.
628  int64_t firstDimToCollapse =
629  sourceType.getRank() -
630  vectorType.getShape().drop_while([](auto v) { return v == 1; }).size();
631 
632  // 1. Collapse the source memref
633  Value collapsedSource =
634  collapseInnerDims(rewriter, loc, source, firstDimToCollapse);
635  MemRefType collapsedSourceType =
636  cast<MemRefType>(collapsedSource.getType());
637  int64_t collapsedRank = collapsedSourceType.getRank();
638  assert(collapsedRank == firstDimToCollapse + 1);
639 
640  // 2. Generate input args for a new vector.transfer_read that will read
641  // from the collapsed memref.
642  // 2.1. New dim exprs + affine map
643  SmallVector<AffineExpr, 1> dimExprs{
644  getAffineDimExpr(firstDimToCollapse, rewriter.getContext())};
645  auto collapsedMap =
646  AffineMap::get(collapsedRank, 0, dimExprs, rewriter.getContext());
647 
648  // 2.2 New indices
649  SmallVector<Value> collapsedIndices =
650  getCollapsedIndices(rewriter, loc, sourceType.getShape(),
651  transferReadOp.getIndices(), firstDimToCollapse);
652 
653  // 3. Create new vector.transfer_read that reads from the collapsed memref
654  VectorType flatVectorType = VectorType::get({vectorType.getNumElements()},
655  vectorType.getElementType());
656  vector::TransferReadOp flatRead = vector::TransferReadOp::create(
657  rewriter, loc, flatVectorType, collapsedSource, collapsedIndices,
658  transferReadOp.getPadding(), collapsedMap);
659  flatRead.setInBoundsAttr(rewriter.getBoolArrayAttr({true}));
660 
661  // 4. Replace the old transfer_read with the new one reading from the
662  // collapsed shape
663  rewriter.replaceOpWithNewOp<vector::ShapeCastOp>(
664  transferReadOp, cast<VectorType>(vector.getType()), flatRead);
665  return success();
666  }
667 
668 private:
669  // Minimum bitwidth that the trailing vector dimension should have after
670  // flattening.
671  unsigned targetVectorBitwidth;
672 };

772 /// Rewrite `vector.extract(vector.transfer_read)` to `memref.load`.
773 ///
774 /// All the users of the transfer op must be `vector.extract` ops. If
775 /// `allowMultipleUses` is set to true, rewrite transfer ops with any number of
776 /// users. Otherwise, rewrite only if the extract op is the single user of the
777 /// transfer op. Rewriting a single vector load with multiple scalar loads may
778 /// negatively affect performance.
779 class RewriteScalarExtractOfTransferRead
780  : public OpRewritePattern<vector::ExtractOp> {
781 public:
782  RewriteScalarExtractOfTransferRead(MLIRContext *context,
783  PatternBenefit benefit,
784  bool allowMultipleUses)
785  : OpRewritePattern(context, benefit),
786  allowMultipleUses(allowMultipleUses) {}
787 
788  LogicalResult matchAndRewrite(vector::ExtractOp extractOp,
789  PatternRewriter &rewriter) const override {
790  // Match phase.
791  auto xferOp = extractOp.getVector().getDefiningOp<vector::TransferReadOp>();
792  if (!xferOp)
793  return failure();
794  // Check that we are extracting a scalar and not a sub-vector.
795  if (isa<VectorType>(extractOp.getResult().getType()))
796  return failure();
797  // If multiple uses are not allowed, check if xfer has a single use.
798  if (!allowMultipleUses && !xferOp.getResult().hasOneUse())
799  return failure();
800  // If multiple uses are allowed, check if all the xfer uses are extract ops.
801  if (allowMultipleUses &&
802  !llvm::all_of(xferOp->getUses(), [](OpOperand &use) {
803  return isa<vector::ExtractOp>(use.getOwner());
804  }))
805  return failure();
806  // Mask not supported.
807  if (xferOp.getMask())
808  return failure();
809  // Map not supported.
810  if (!xferOp.getPermutationMap().isMinorIdentity())
811  return failure();
812  // Cannot rewrite if the indices may be out of bounds.
813  if (xferOp.hasOutOfBoundsDim())
814  return failure();
815 
816  // Rewrite phase: construct scalar load.
817  SmallVector<Value> newIndices(xferOp.getIndices().begin(),
818  xferOp.getIndices().end());
819  for (auto [i, pos] : llvm::enumerate(extractOp.getMixedPosition())) {
820  int64_t idx = newIndices.size() - extractOp.getNumIndices() + i;
821 
822  // Compute affine expression `newIndices[idx] + pos` where `pos` can be
823  // either a constant or a value.
824  OpFoldResult composedIdx;
825  if (auto attr = dyn_cast<Attribute>(pos)) {
826  int64_t offset = cast<IntegerAttr>(attr).getInt();
827  composedIdx = affine::makeComposedFoldedAffineApply(
828  rewriter, extractOp.getLoc(),
829  rewriter.getAffineSymbolExpr(0) + offset, {newIndices[idx]});
830  } else {
831  Value dynamicOffset = cast<Value>(pos);
832  AffineExpr sym0, sym1;
833  bindSymbols(rewriter.getContext(), sym0, sym1);
834  composedIdx = affine::makeComposedFoldedAffineApply(
835  rewriter, extractOp.getLoc(), sym0 + sym1,
836  {newIndices[idx], dynamicOffset});
837  }
838 
839  // Update the corresponding index with the folded result.
840  if (auto value = dyn_cast<Value>(composedIdx)) {
841  newIndices[idx] = value;
842  } else {
843  newIndices[idx] = arith::ConstantIndexOp::create(
844  rewriter, extractOp.getLoc(), *getConstantIntValue(composedIdx));
845  }
846  }
847  if (isa<MemRefType>(xferOp.getBase().getType())) {
848  rewriter.replaceOpWithNewOp<memref::LoadOp>(extractOp, xferOp.getBase(),
849  newIndices);
850  } else {
851  rewriter.replaceOpWithNewOp<tensor::ExtractOp>(
852  extractOp, xferOp.getBase(), newIndices);
853  }
854 
855  return success();
856  }
857 
858 private:
859  bool allowMultipleUses;
860 };
