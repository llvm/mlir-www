91 // Helper to drop dimension from vector type.
92 static Type adjustType(VectorType tp, int64_t index) {
93  int64_t rank = tp.getRank();
94  Type eltType = tp.getElementType();
95  if (rank == 1) {
96  assert(index == 0 && "index for scalar result out of bounds");
97  return eltType;
98  }
99  SmallVector<int64_t, 4> adjustedShape;
100  for (int64_t i = 0; i < rank; ++i) {
101  // Omit dimension at the given index.
102  if (i == index)
103  continue;
104  // Otherwise, add dimension back.
105  adjustedShape.push_back(tp.getDimSize(i));
106  }
107  return VectorType::get(adjustedShape, eltType);
108 }

141  VectorType type, int64_t index, int64_t pos,

145  return val;

176 /// Return the target shape for unrolling for the given `op`. Return llvm::None
177 /// if the op shouldn't be or cannot be unrolled.
178 static Optional<SmallVector<int64_t, 4>>
179 getTargetShape(const vector::UnrollVectorOptions &options, Operation *op) {
180  if (options.filterConstraint && failed(options.filterConstraint(op)))
181  return llvm::None;
182  assert(options.nativeShape &&
183  "vector unrolling expects the native shape or native"
184  "shape call back function to be set");
185  auto unrollableVectorOp = dyn_cast<VectorUnrollOpInterface>(op);
186  if (!unrollableVectorOp)
187  return llvm::None;
188  auto maybeUnrollShape = unrollableVectorOp.getShapeForUnroll();
189  if (!maybeUnrollShape)
190  return llvm::None;
191  Optional<SmallVector<int64_t, 4>> targetShape = options.nativeShape(op);
192  if (!targetShape)
193  return llvm::None;
194  auto maybeShapeRatio = shapeRatio(*maybeUnrollShape, *targetShape);
195  if (!maybeShapeRatio ||
196  llvm::all_of(*maybeShapeRatio, [](int64_t v) { return v == 1; }))
197  return llvm::None;
198  return targetShape;
199 }

214 /// Compute the indices of the slice `index` for a tranfer op.
215 static SmallVector<Value>
216 sliceTransferIndices(int64_t index, ArrayRef<int64_t> originalShape,
217  ArrayRef<int64_t> targetShape, ArrayRef<Value> indices,
218  AffineMap permutationMap, Location loc,
219  OpBuilder &builder) {
220  MLIRContext *ctx = builder.getContext();
221  auto isBroadcast = [](AffineExpr expr) {
222  if (auto constExpr = expr.dyn_cast<AffineConstantExpr>())
223  return constExpr.getValue() == 0;
224  return false;
225  };
226  SmallVector<int64_t, 4> elementOffsets =
227  getVectorOffset(originalShape, targetShape, index);
228  // Compute 'sliceIndices' by adding 'sliceOffsets[i]' to 'indices[i]'.
229  SmallVector<Value> slicedIndices(indices.begin(), indices.end());
230  for (auto dim : llvm::enumerate(permutationMap.getResults())) {
231  if (isBroadcast(dim.value()))
232  continue;
233  unsigned pos = dim.value().cast<AffineDimExpr>().getPosition();
234  auto expr = getAffineDimExpr(0, builder.getContext()) +
235  getAffineConstantExpr(elementOffsets[dim.index()], ctx);
236  auto map = AffineMap::get(/*dimCount=*/1, /*symbolCount=*/0, expr);
237  slicedIndices[pos] = builder.create<AffineApplyOp>(loc, map, indices[pos]);
238  }
239  return slicedIndices;
240 }

346 SmallVector<T> permute(AffineMap map, llvm::ArrayRef<T> source) {
347  SmallVector<T> result;
348  result.reserve(map.getNumResults());
349  for (unsigned i : llvm::seq(unsigned(0), map.getNumResults())) {
350  unsigned dim = map.getDimPosition(i);
351  result.push_back(source[dim]);
352  }
353  return result;

358  struct OffsetMapInfo {
359  static SmallVector<int64_t> getEmptyKey() { return {int64_t(-1)}; }

368  static bool isEqual(const SmallVector<int64_t> &lhs,
369  const SmallVector<int64_t> &rhs) {
370  return lhs == rhs;
371  }
372  };
373  UnrollContractionPattern(MLIRContext *context,
374  const vector::UnrollVectorOptions &options)
375  : OpRewritePattern<vector::ContractionOp>(context, /*benefit=*/1),
376  options(options) {}
377 
378  LogicalResult matchAndRewrite(vector::ContractionOp contractOp,
379  PatternRewriter &rewriter) const override {
380  auto targetShape = getTargetShape(options, contractOp);
381  if (!targetShape)
382  return failure();
383  auto dstVecType = contractOp.getResultType().cast<VectorType>();
384  SmallVector<int64_t, 4> originalSize = *contractOp.getShapeForUnroll();
385  SmallVector<int64_t, 4> ratio = *shapeRatio(originalSize, *targetShape);
386 
387  // Compute shape ratio of 'shape' and 'sizes'.
388  int64_t sliceCount = computeMaxLinearIndex(ratio);
389  Location loc = contractOp.getLoc();
390  unsigned accIndex = vector::ContractionOp::getAccOperandIndex();
391  AffineMap dstAffineMap = contractOp.getIndexingMaps()[accIndex];
392  llvm::MapVector<
393  SmallVector<int64_t>, Value,
394  llvm::DenseMap<SmallVector<int64_t>, unsigned, OffsetMapInfo>>
395  accCache;
396  for (int64_t i = 0; i < sliceCount; i++) {
397  SmallVector<int64_t, 4> offsets =
398  getVectorOffset(originalSize, *targetShape, i);
399  SmallVector<Value, 4> slicesOperands(contractOp.getNumOperands());
400 
401  // Helper to coompute the new shape of each operand and extract the slice.
402  auto extractOperand = [&](unsigned index, Value operand,
403  AffineMap permutationMap,
404  ArrayRef<int64_t> operandOffets) {
405  SmallVector<int64_t> operandShape =
406  permute(permutationMap, ArrayRef<int64_t>(*targetShape));
407  SmallVector<int64_t, 4> operandStrides(operandOffets.size(), 1);
408  slicesOperands[index] = rewriter.create<vector::ExtractStridedSliceOp>(
409  loc, operand, operandOffets, operandShape, operandStrides);
410  };
411 
412  // Extract the new lhs operand.
413  AffineMap lhsPermutationMap = contractOp.getIndexingMaps()[0];
414  SmallVector<int64_t> lhsOffets =
415  permute(lhsPermutationMap, ArrayRef<int64_t>(offsets));
416  extractOperand(0, contractOp.lhs(), lhsPermutationMap, lhsOffets);
417  // If there is a mask associated to lhs, extract it as well.
418  if (slicesOperands.size() > 3)
419  extractOperand(3, contractOp.masks()[0], lhsPermutationMap, lhsOffets);
420 
421  // Extract the new rhs operand.
422  AffineMap rhsPermutationMap = contractOp.getIndexingMaps()[1];
423  SmallVector<int64_t> rhsOffets =
424  permute(rhsPermutationMap, ArrayRef<int64_t>(offsets));
425  extractOperand(1, contractOp.rhs(), rhsPermutationMap, rhsOffets);
426  // If there is a mask associated to rhs, extract it as well.
427  if (slicesOperands.size() > 4)
428  extractOperand(4, contractOp.masks()[1], rhsPermutationMap, rhsOffets);
429 
430  AffineMap accPermutationMap = contractOp.getIndexingMaps()[2];
431  SmallVector<int64_t> accOffets =
432  permute(accPermutationMap, ArrayRef<int64_t>(offsets));
433  // If a version of the accumulator has already been computed, use it
434  // otherwise extract the first version from the original operand.
435  auto accIt = accCache.find(accOffets);
436  if (accIt != accCache.end())
437  slicesOperands[2] = accIt->second;
438  else
439  extractOperand(2, contractOp.acc(), accPermutationMap, accOffets);
440 
441  SmallVector<int64_t> dstShape =
442  permute(dstAffineMap, ArrayRef<int64_t>(*targetShape));
443  auto targetType = VectorType::get(dstShape, dstVecType.getElementType());
444  Operation *newOp = cloneOpWithOperandsAndTypes(
445  rewriter, loc, contractOp, slicesOperands, targetType);
446 
447  SmallVector<int64_t> dstOffets =
448  permute(dstAffineMap, ArrayRef<int64_t>(offsets));
449  // Save the accumulated value untill all the loops are unrolled since
450  // reduction loop keep updating the accumulator.
451  accCache[dstOffets] = newOp->getResult(0);
452  }
453  // Assemble back the accumulator into a single vector.
454  Value result = rewriter.create<ConstantOp>(
455  loc, dstVecType, rewriter.getZeroAttr(dstVecType));
456  for (const auto &it : accCache) {
457  SmallVector<int64_t> dstStrides(it.first.size(), 1);
458  result = rewriter.create<vector::InsertStridedSliceOp>(
459  loc, it.second, result, it.first, dstStrides);
460  }
461  rewriter.replaceOp(contractOp, result);
462  return success();
463  }

520 /// ShapeCastOpFolder folds cancelling ShapeCastOps away.
521 //
522 // Example:
523 //
524 // The following MLIR with cancelling ShapeCastOps:
525 //
526 // %0 = source : vector<5x4x2xf32>
527 // %1 = shape_cast %0 : vector<5x4x2xf32> to vector<20x2xf32>
528 // %2 = shape_cast %1 : vector<20x2xf32> to vector<5x4x2xf32>
529 // %3 = user %2 : vector<5x4x2xf32>
530 //
531 // Should canonicalize to the following:
532 //
533 // %0 = source : vector<5x4x2xf32>
534 // %1 = user %0 : vector<5x4x2xf32>
535 //
536 struct ShapeCastOpFolder : public OpRewritePattern<vector::ShapeCastOp> {
537  using OpRewritePattern<vector::ShapeCastOp>::OpRewritePattern;
538 
539  LogicalResult matchAndRewrite(vector::ShapeCastOp shapeCastOp,
540  PatternRewriter &rewriter) const override {
541  // Check if 'shapeCastOp' has vector source/result type.
542  auto sourceVectorType =
543  shapeCastOp.source().getType().dyn_cast_or_null<VectorType>();
544  auto resultVectorType =
545  shapeCastOp.result().getType().dyn_cast_or_null<VectorType>();
546  if (!sourceVectorType || !resultVectorType)
547  return failure();
548 
549  // Check if shape cast op source operand is also a shape cast op.
550  auto sourceShapeCastOp = dyn_cast_or_null<vector::ShapeCastOp>(
551  shapeCastOp.source().getDefiningOp());
552  if (!sourceShapeCastOp)
553  return failure();

560  operandResultVectorType != sourceVectorType)

568 /// Progressive lowering of BroadcastOp.
569 class BroadcastOpLowering : public OpRewritePattern<vector::BroadcastOp> {
570 public:
571  using OpRewritePattern<vector::BroadcastOp>::OpRewritePattern;
572 
573  LogicalResult matchAndRewrite(vector::BroadcastOp op,
574  PatternRewriter &rewriter) const override {
575  auto loc = op.getLoc();
576  VectorType dstType = op.getVectorType();
577  VectorType srcType = op.getSourceType().dyn_cast<VectorType>();
578  Type eltType = dstType.getElementType();
579 
580  // Determine rank of source and destination.
581  int64_t srcRank = srcType ? srcType.getRank() : 0;
582  int64_t dstRank = dstType.getRank();
583 
584  // Duplicate this rank.
585  // For example:
586  // %x = broadcast %y : k-D to n-D, k < n
587  // becomes:
588  // %b = broadcast %y : k-D to (n-1)-D
589  // %x = [%b,%b,%b,%b] : n-D
590  // becomes:
591  // %b = [%y,%y] : (n-1)-D
592  // %x = [%b,%b,%b,%b] : n-D
593  if (srcRank < dstRank) {
594  // Scalar to any vector can use splat.
595  if (srcRank == 0) {
596  rewriter.replaceOpWithNewOp<SplatOp>(op, dstType, op.source());
597  return success();
598  }
599  // Duplication.
600  VectorType resType =
601  VectorType::get(dstType.getShape().drop_front(), eltType);
602  Value bcst =
603  rewriter.create<vector::BroadcastOp>(loc, resType, op.source());
604  Value result = rewriter.create<ConstantOp>(loc, dstType,
605  rewriter.getZeroAttr(dstType));
606  for (int64_t d = 0, dim = dstType.getDimSize(0); d < dim; ++d)
607  result = rewriter.create<vector::InsertOp>(loc, bcst, result, d);
608  rewriter.replaceOp(op, result);
609  return success();
610  }
611 
612  // Find non-matching dimension, if any.
613  assert(srcRank == dstRank);
614  int64_t m = -1;
615  for (int64_t r = 0; r < dstRank; r++)
616  if (srcType.getDimSize(r) != dstType.getDimSize(r)) {
617  m = r;
618  break;
619  }
620 
621  // All trailing dimensions are the same. Simply pass through.
622  if (m == -1) {
623  rewriter.replaceOp(op, op.source());
624  return success();
625  }
626 
627  // Stretching scalar inside vector (e.g. vector<1xf32>) can use splat.
628  if (srcRank == 1) {
629  assert(m == 0);
630  Value ext = rewriter.create<vector::ExtractOp>(loc, op.source(), 0);
631  rewriter.replaceOpWithNewOp<SplatOp>(op, dstType, ext);
632  return success();
633  }
634 
635  // Any non-matching dimension forces a stretch along this rank.
636  // For example:
637  // %x = broadcast %y : vector<4x1x2xf32> to vector<4x2x2xf32>
638  // becomes:
639  // %a = broadcast %y[0] : vector<1x2xf32> to vector<2x2xf32>
640  // %b = broadcast %y[1] : vector<1x2xf32> to vector<2x2xf32>
641  // %c = broadcast %y[2] : vector<1x2xf32> to vector<2x2xf32>
642  // %d = broadcast %y[3] : vector<1x2xf32> to vector<2x2xf32>
643  // %x = [%a,%b,%c,%d]
644  // becomes:
645  // %u = broadcast %y[0][0] : vector<2xf32> to vector <2x2xf32>
646  // %v = broadcast %y[1][0] : vector<2xf32> to vector <2x2xf32>
647  // %a = [%u, %v]
648  // ..
649  // %x = [%a,%b,%c,%d]
650  VectorType resType =
651  VectorType::get(dstType.getShape().drop_front(), eltType);
652  Value result = rewriter.create<ConstantOp>(loc, dstType,
653  rewriter.getZeroAttr(dstType));
654  if (m == 0) {
655  // Stetch at start.
656  Value ext = rewriter.create<vector::ExtractOp>(loc, op.source(), 0);
657  Value bcst = rewriter.create<vector::BroadcastOp>(loc, resType, ext);
658  for (int64_t d = 0, dim = dstType.getDimSize(0); d < dim; ++d)
659  result = rewriter.create<vector::InsertOp>(loc, bcst, result, d);
660  } else {
661  // Stetch not at start.
662  for (int64_t d = 0, dim = dstType.getDimSize(0); d < dim; ++d) {
663  Value ext = rewriter.create<vector::ExtractOp>(loc, op.source(), d);
664  Value bcst = rewriter.create<vector::BroadcastOp>(loc, resType, ext);
665  result = rewriter.create<vector::InsertOp>(loc, bcst, result, d);
666  }
667  }
668  rewriter.replaceOp(op, result);
669  return success();
670  }
671 };

673 /// Progressive lowering of TransposeOp.
674 /// One:
675 /// %x = vector.transpose %y, [1, 0]
676 /// is replaced by:
677 /// %z = constant dense<0.000000e+00>
678 /// %0 = vector.extract %y[0, 0]
679 /// %1 = vector.insert %0, %z [0, 0]
680 /// ..
681 /// %x = vector.insert .., .. [.., ..]
682 class TransposeOpLowering : public OpRewritePattern<vector::TransposeOp> {
683 public:
684  using OpRewritePattern<vector::TransposeOp>::OpRewritePattern;
685 
686  TransposeOpLowering(vector::VectorTransformsOptions vectorTransformsOptions,
687  MLIRContext *context)
688  : OpRewritePattern<vector::TransposeOp>(context),
689  vectorTransformsOptions(vectorTransformsOptions) {}
690 
691  LogicalResult matchAndRewrite(vector::TransposeOp op,
692  PatternRewriter &rewriter) const override {
693  auto loc = op.getLoc();
694 
695  VectorType resType = op.getResultType();
696 
697  // Set up convenience transposition table.
698  SmallVector<int64_t, 4> transp;
699  for (auto attr : op.transp())
700  transp.push_back(attr.cast<IntegerAttr>().getInt());
701 
702  // Handle a true 2-D matrix transpose differently when requested.
703  if (vectorTransformsOptions.vectorTransposeLowering ==
704  vector::VectorTransposeLowering::Flat &&
705  resType.getRank() == 2 && transp[0] == 1 && transp[1] == 0) {
706  Type flattenedType =
707  VectorType::get(resType.getNumElements(), resType.getElementType());
708  auto matrix =
709  rewriter.create<vector::ShapeCastOp>(loc, flattenedType, op.vector());
710  auto rows = rewriter.getI32IntegerAttr(resType.getShape()[0]);
711  auto columns = rewriter.getI32IntegerAttr(resType.getShape()[1]);
712  Value trans = rewriter.create<vector::FlatTransposeOp>(
713  loc, flattenedType, matrix, rows, columns);
714  rewriter.replaceOpWithNewOp<vector::ShapeCastOp>(op, resType, trans);
715  return success();
716  }
717 
718  // Generate fully unrolled extract/insert ops.
719  Value result = rewriter.create<ConstantOp>(loc, resType,
720  rewriter.getZeroAttr(resType));
721  SmallVector<int64_t, 4> lhs(transp.size(), 0);
722  SmallVector<int64_t, 4> rhs(transp.size(), 0);
723  rewriter.replaceOp(op, expandIndices(loc, resType, 0, transp, lhs, rhs,
724  op.vector(), result, rewriter));
725  return success();
726  }
727 
728 private:
729  // Builds the indices arrays for the lhs and rhs. Generates the extract/insert
730  // operation when al ranks are exhausted.
731  Value expandIndices(Location loc, VectorType resType, int64_t pos,
732  SmallVector<int64_t, 4> &transp,
733  SmallVector<int64_t, 4> &lhs,
734  SmallVector<int64_t, 4> &rhs, Value input, Value result,
735  PatternRewriter &rewriter) const {
736  if (pos >= resType.getRank()) {
737  auto ridx = rewriter.getI64ArrayAttr(rhs);
738  auto lidx = rewriter.getI64ArrayAttr(lhs);
739  Type eltType = resType.getElementType();
740  Value e = rewriter.create<vector::ExtractOp>(loc, eltType, input, ridx);
741  return rewriter.create<vector::InsertOp>(loc, resType, e, result, lidx);
742  }
743  for (int64_t d = 0, e = resType.getDimSize(pos); d < e; ++d) {
744  lhs[pos] = d;
745  rhs[transp[pos]] = d;
746  result = expandIndices(loc, resType, pos + 1, transp, lhs, rhs, input,
747  result, rewriter);
748  }
749  return result;
750  }

752  /// Options to control the vector patterns.
753  vector::VectorTransformsOptions vectorTransformsOptions;

759 /// is replaced by:
760 /// %z = zero-result
761 /// %0 = vector.extract %lhs[0]
762 /// %1 = vector.broadcast %0

764 /// %3 = vector.fma %1, %rhs, %2
765 /// %4 = vector.insert %3, %z[0]

767 /// %x = vector.insert %.., %..[N-1]

770 public:
771  using OpRewritePattern<vector::OuterProductOp>::OpRewritePattern;
772 
773  LogicalResult matchAndRewrite(vector::OuterProductOp op,

775  auto loc = op.getLoc();
776 
777  VectorType lhsType = op.getOperandVectorTypeLHS();
778  VectorType rhsType = op.getOperandTypeRHS().dyn_cast<VectorType>();
779  VectorType resType = op.getVectorType();
780  Type eltType = resType.getElementType();
781  bool isInt = eltType.isa<IntegerType, IndexType>();
782  Value acc = (op.acc().empty()) ? nullptr : op.acc()[0];
783  vector::CombiningKind kind = op.kind();
784 
785  if (!rhsType) {
786  // Special case: AXPY operation.
787  Value b = rewriter.create<vector::BroadcastOp>(loc, lhsType, op.rhs());
788  Optional<Value> mult =
789  isInt ? genMultI(loc, op.lhs(), b, acc, kind, rewriter)
790  : genMultF(loc, op.lhs(), b, acc, kind, rewriter);
791  if (!mult.hasValue())
792  return failure();
793  rewriter.replaceOp(op, mult.getValue());
794  return success();
795  }
796 
797  Value result = rewriter.create<ConstantOp>(loc, resType,
798  rewriter.getZeroAttr(resType));
799  for (int64_t d = 0, e = resType.getDimSize(0); d < e; ++d) {
800  auto pos = rewriter.getI64ArrayAttr(d);

802  Value a = rewriter.create<vector::BroadcastOp>(loc, rhsType, x);
803  Value r = nullptr;
804  if (acc)
805  r = rewriter.create<vector::ExtractOp>(loc, rhsType, acc, pos);
806  Optional<Value> m = isInt ? genMultI(loc, a, op.rhs(), r, kind, rewriter)
807  : genMultF(loc, a, op.rhs(), r, kind, rewriter);
808  if (!m.hasValue())
809  return failure();

811  result, pos);
812  }
813  rewriter.replaceOp(op, result);
814  return success();

817 private:
818  static Optional<Value> genMultI(Location loc, Value x, Value y, Value acc,
819  vector::CombiningKind kind,
820  PatternRewriter &rewriter) {
821  using vector::CombiningKind;
822 
823  MulIOp mul = rewriter.create<MulIOp>(loc, x, y);
824  if (!acc)
825  return Optional<Value>(mul);
826 
827  Value combinedResult;
828  switch (kind) {
829  case CombiningKind::ADD:
830  combinedResult = rewriter.create<AddIOp>(loc, mul, acc);
831  break;
832  case CombiningKind::MUL:
833  combinedResult = rewriter.create<MulIOp>(loc, mul, acc);
834  break;
835  case CombiningKind::MIN:
836  combinedResult = rewriter.create<SelectOp>(
837  loc, rewriter.create<CmpIOp>(loc, CmpIPredicate::slt, mul, acc), mul,
838  acc);
839  break;
840  case CombiningKind::MAX:
841  combinedResult = rewriter.create<SelectOp>(
842  loc, rewriter.create<CmpIOp>(loc, CmpIPredicate::sge, mul, acc), mul,
843  acc);
844  break;
845  case CombiningKind::AND:
846  combinedResult = rewriter.create<AndOp>(loc, mul, acc);
847  break;
848  case CombiningKind::OR:
849  combinedResult = rewriter.create<OrOp>(loc, mul, acc);
850  break;
851  case CombiningKind::XOR:
852  combinedResult = rewriter.create<XOrOp>(loc, mul, acc);
853  break;
854  }
855  return Optional<Value>(combinedResult);
856  }

858  static Optional<Value> genMultF(Location loc, Value x, Value y, Value acc,
859  vector::CombiningKind kind,
860  PatternRewriter &rewriter) {
861  using vector::CombiningKind;
862 
863  // Special case for fused multiply-add.
864  if (acc && kind == CombiningKind::ADD) {
865  return Optional<Value>(rewriter.create<vector::FMAOp>(loc, x, y, acc));
866  }
867 
868  MulFOp mul = rewriter.create<MulFOp>(loc, x, y);
869 
870  if (!acc)
871  return Optional<Value>(mul);
872 
873  Value combinedResult;
874  switch (kind) {
875  case CombiningKind::MUL:
876  combinedResult = rewriter.create<MulFOp>(loc, mul, acc);
877  break;
878  case CombiningKind::MIN:
879  combinedResult = rewriter.create<SelectOp>(
880  loc, rewriter.create<CmpFOp>(loc, CmpFPredicate::OLE, mul, acc), mul,
881  acc);
882  break;
883  case CombiningKind::MAX:
884  combinedResult = rewriter.create<SelectOp>(
885  loc, rewriter.create<CmpFOp>(loc, CmpFPredicate::OGT, mul, acc), mul,
886  acc);
887  break;
888  case CombiningKind::ADD: // Already handled this special case above.
889  case CombiningKind::AND: // Only valid for integer types.
890  case CombiningKind::OR: // Only valid for integer types.
891  case CombiningKind::XOR: // Only valid for integer types.
892  return Optional<Value>();
893  }
894  return Optional<Value>(combinedResult);
895  }

1065 // We typically should not lower general shape cast operations into data
1066 // movement instructions, since the assumption is that these casts are
1067 // optimized away during progressive lowering. For completeness, however,
1068 // we fall back to a reference implementation that moves all elements
1069 // into the right place if we get here.
1070 class ShapeCastOpRewritePattern : public OpRewritePattern<vector::ShapeCastOp> {
1071 public:
1072  using OpRewritePattern<vector::ShapeCastOp>::OpRewritePattern;
1073 
1074  LogicalResult matchAndRewrite(vector::ShapeCastOp op,
1075  PatternRewriter &rewriter) const override {
1076  Location loc = op.getLoc();
1077  auto sourceVectorType = op.getSourceVectorType();
1078  auto resultVectorType = op.getResultVectorType();
1079  // Intended 2D/1D lowerings with better implementations.
1080  int64_t srcRank = sourceVectorType.getRank();
1081  int64_t resRank = resultVectorType.getRank();
1082  if ((srcRank == 2 && resRank == 1) || (srcRank == 1 && resRank == 2))
1083  return failure();
1084  // Compute number of elements involved in the reshape.
1085  int64_t numElts = 1;
1086  for (int64_t r = 0; r < srcRank; r++)
1087  numElts *= sourceVectorType.getDimSize(r);
1088  // Replace with data movement operations:
1089  // x[0,0,0] = y[0,0]
1090  // x[0,0,1] = y[0,1]
1091  // x[0,1,0] = y[0,2]
1092  // etc., incrementing the two index vectors "row-major"
1093  // within the source and result shape.
1094  SmallVector<int64_t, 4> srcIdx(srcRank);
1095  SmallVector<int64_t, 4> resIdx(resRank);
1096  Value result = rewriter.create<ConstantOp>(
1097  loc, resultVectorType, rewriter.getZeroAttr(resultVectorType));
1098  for (int64_t i = 0; i < numElts; i++) {
1099  if (i != 0) {
1100  incIdx(srcIdx, sourceVectorType, srcRank - 1);
1101  incIdx(resIdx, resultVectorType, resRank - 1);
1102  }
1103  Value e = rewriter.create<vector::ExtractOp>(loc, op.source(), srcIdx);

1143 /// semantics to:
1144 /// ```
1145 /// %mta = maybe_transpose
1146 /// %mtb = maybe_transpose
1147 /// %flattened_a = vector.shape_cast %mta
1148 /// %flattened_b = vector.shape_cast %mtb
1149 /// %flattened_d = vector.matmul %flattened_a, %flattened_b
1150 /// %mtd = vector.shape_cast %flattened_d
1151 /// %d = maybe_untranspose %mtd
1152 /// %e = add %c, %d
1153 /// ```
1154 /// `vector.matmul` later lowers to `llvm.matrix.multiply`.
1155 //
1156 /// This only kicks in when VectorTransformsOptions is set to `Matmul`.
1157 /// vector.transpose operations are inserted if the vector.contract op is not a
1158 /// row-major matrix multiply.
1159 LogicalResult
1160 ContractionOpToMatmulOpLowering::matchAndRewrite(vector::ContractionOp op,
1161  PatternRewriter &rew) const {
1162  // TODO: implement masks
1163  if (llvm::size(op.masks()) != 0)
1164  return failure();
1165  if (vectorTransformsOptions.vectorContractLowering !=
1166  vector::VectorContractLowering::Matmul)
1167  return failure();
1168  if (failed(filter(op)))
1169  return failure();
1170 
1171  auto iteratorTypes = op.iterator_types().getValue();
1172  if (!isParallelIterator(iteratorTypes[0]) ||
1173  !isParallelIterator(iteratorTypes[1]) ||
1174  !isReductionIterator(iteratorTypes[2]))
1175  return failure();
1176 
1177  Type elementType = op.getLhsType().getElementType();
1178  if (!elementType.isIntOrFloat())
1179  return failure();
1180 
1181  // Perform lhs + rhs transpositions to conform to matmul row-major semantics.
1182  // Bail out if the contraction cannot be put in this form.
1183  MLIRContext *ctx = op.getContext();
1184  Location loc = op.getLoc();
1185  AffineExpr m, n, k;
1186  bindDims(rew.getContext(), m, n, k);
1187  // LHS must be A(m, k) or A(k, m).
1188  Value lhs = op.lhs();
1189  auto lhsMap = op.indexing_maps()[0].cast<AffineMapAttr>().getValue();
1190  if (lhsMap == AffineMap::get(3, 0, {k, m}, ctx))
1191  lhs = rew.create<vector::TransposeOp>(loc, lhs, ArrayRef<int64_t>{1, 0});
1192  else if (lhsMap != AffineMap::get(3, 0, {m, k}, ctx))
1193  return failure();
1194 
1195  // RHS must be B(k, n) or B(n, k).
1196  Value rhs = op.rhs();
1197  auto rhsMap = op.indexing_maps()[1].cast<AffineMapAttr>().getValue();
1198  if (rhsMap == AffineMap::get(3, 0, {n, k}, ctx))
1199  rhs = rew.create<vector::TransposeOp>(loc, rhs, ArrayRef<int64_t>{1, 0});
1200  else if (rhsMap != AffineMap::get(3, 0, {k, n}, ctx))
1201  return failure();
1202 
1203  // At this point lhs and rhs are in row-major.
1204  VectorType lhsType = lhs.getType().cast<VectorType>();
1205  VectorType rhsType = rhs.getType().cast<VectorType>();
1206  int64_t lhsRows = lhsType.getDimSize(0);
1207  int64_t lhsColumns = lhsType.getDimSize(1);
1208  int64_t rhsColumns = rhsType.getDimSize(1);
1209 
1210  Type flattenedLHSType =
1211  VectorType::get(lhsType.getNumElements(), lhsType.getElementType());
1212  lhs = rew.create<vector::ShapeCastOp>(loc, flattenedLHSType, lhs);
1213 
1214  Type flattenedRHSType =
1215  VectorType::get(rhsType.getNumElements(), rhsType.getElementType());
1216  rhs = rew.create<vector::ShapeCastOp>(loc, flattenedRHSType, rhs);
1217 
1218  Value mul = rew.create<vector::MatmulOp>(loc, lhs, rhs, lhsRows, lhsColumns,
1219  rhsColumns);
1220  mul = rew.create<vector::ShapeCastOp>(
1221  loc,
1222  VectorType::get({lhsRows, rhsColumns},
1223  getElementTypeOrSelf(op.acc().getType())),
1224  mul);
1225 
1226  // ACC must be C(m, n) or C(n, m).
1227  auto accMap = op.indexing_maps()[2].cast<AffineMapAttr>().getValue();
1228  if (accMap == AffineMap::get(3, 0, {n, m}, ctx))
1229  mul = rew.create<vector::TransposeOp>(loc, mul, ArrayRef<int64_t>{1, 0});
1230  else if (accMap != AffineMap::get(3, 0, {m, n}, ctx))
1231  llvm_unreachable("invalid contraction semantics");
1232 
1233  Value res = elementType.isa<IntegerType>()
1234  ? static_cast<Value>(rew.create<AddIOp>(loc, op.acc(), mul))
1235  : static_cast<Value>(rew.create<AddFOp>(loc, op.acc(), mul));
1236 
1237  rew.replaceOp(op, res);
1238  return success();
1239 }

1307 /// Progressively lower a `vector.contract %a, %b, %c` with row-major matmul
1308 /// semantics to a reduction_size-unrolled sequence:
1309 /// ```
1310 /// %at = vector.transpose %a, [1, 0]
1311 /// %bRow0 = vector.extract %b[0]
1312 /// %atRow0 = vector.extract %at[0]
1313 /// %c0 = vector.outerproduct %atRow0, %bRow0, %c
1314 /// ...
1315 /// %bRowK = vector.extract %b[K]
1316 /// %atRowK = vector.extract %at[K]
1317 /// %cK = vector.outerproduct %atRowK, %bRowK, %cK-1
1318 /// ```
1319 ///
1320 /// This only kicks in when VectorTransformsOptions is set to OuterProduct but
1321 /// otherwise supports any layout permutation of the matrix-multiply.
1322 LogicalResult ContractionOpToOuterProductOpLowering::matchAndRewrite(
1323  vector::ContractionOp op, PatternRewriter &rewriter) const {
1324  // TODO: implement masks
1325  if (llvm::size(op.masks()) != 0)
1326  return failure();
1327 
1328  if (vectorTransformsOptions.vectorContractLowering !=
1329  vector::VectorContractLowering::OuterProduct)
1330  return failure();
1331 
1332  if (failed(filter(op)))
1333  return failure();
1334 
1335  VectorType lhsType = op.getLhsType();
1336  Value lhs = op.lhs(), rhs = op.rhs(), res = op.acc();
1337 
1338  // Set up the parallel/reduction structure in right form.
1339  AffineExpr m, n, k;
1340  bindDims(rewriter.getContext(), m, n, k);
1341 
1342  //
1343  // Two outer parallel, one inner reduction (matmat flavor).
1344  //
1345  UnrolledOuterProductEmitter e(rewriter, op);
1346  if (e.iters({Par(), Par(), Red()})) {
1347  // Classical row-major matmul: Just permute the lhs.
1348  if (e.layout({{m, k}, {k, n}, {m, n}}))
1349  return e.outer_prod(e.t(lhs), rhs, res, lhsType.getDimSize(1));
1350  // TODO: may be better to fail and use some vector<k> -> scalar reduction.
1351  if (e.layout({{m, k}, {n, k}, {m, n}})) {
1352  Value tlhs = e.t(lhs);
1353  return e.outer_prod(tlhs, e.t(rhs), res, lhsType.getDimSize(1));
1354  }
1355  // No need to permute anything.
1356  if (e.layout({{k, m}, {k, n}, {m, n}}))
1357  return e.outer_prod(lhs, rhs, res, lhsType.getDimSize(0));
1358  // Just permute the rhs.
1359  if (e.layout({{k, m}, {n, k}, {m, n}}))
1360  return e.outer_prod(lhs, e.t(rhs), res, lhsType.getDimSize(0));
1361  // Transposed output: swap RHS and LHS.
1362  // Classical row-major matmul: permute the lhs.
1363  if (e.layout({{m, k}, {k, n}, {n, m}}))
1364  return e.outer_prod(rhs, e.t(lhs), res, lhsType.getDimSize(1));
1365  // TODO: may be better to fail and use some vector<k> -> scalar reduction.
1366  if (e.layout({{m, k}, {n, k}, {n, m}})) {
1367  Value trhs = e.t(rhs);
1368  return e.outer_prod(trhs, e.t(lhs), res, lhsType.getDimSize(1));
1369  }
1370  if (e.layout({{k, m}, {k, n}, {n, m}}))
1371  return e.outer_prod(rhs, lhs, res, lhsType.getDimSize(0));
1372  if (e.layout({{k, m}, {n, k}, {n, m}}))
1373  return e.outer_prod(e.t(rhs), lhs, res, lhsType.getDimSize(0));
1374  return failure();
1375  }
1376 
1377  //
1378  // One outer parallel, one inner reduction (matvec flavor)
1379  //
1380  if (e.iters({Par(), Red()})) {
1381  // Case mat-vec: transpose.
1382  if (e.layout({{m, n}, {n}, {m}}))
1383  return e.outer_prod(e.t(lhs), rhs, res, lhsType.getDimSize(1));
1384  // Case mat-trans-vec: ready to go.
1385  if (e.layout({{n, m}, {n}, {m}}))
1386  return e.outer_prod(lhs, rhs, res, lhsType.getDimSize(0));
1387  // Case vec-mat: swap and transpose.
1388  if (e.layout({{n}, {m, n}, {m}}))
1389  return e.outer_prod(e.t(rhs), lhs, res, lhsType.getDimSize(0));
1390  // Case vec-mat-trans: swap and ready to go.
1391  if (e.layout({{n}, {n, m}, {m}}))
1392  return e.outer_prod(rhs, lhs, res, lhsType.getDimSize(0));
1393  return failure();
1394  }
1395 
1396  return failure();
1397 }

1540  return failure();

1544  getElementTypeOrSelf(op.getAccType()) ||
1545  op.getRhsType().getElementType() != getElementTypeOrSelf(op.getAccType()))
1546  return failure();

1554  if (succeeded(pat2.matchAndRewrite(op, rewriter)))

1557  if (succeeded(pat3.matchAndRewrite(op, rewriter)))

1576  rhsContractingDimSet.insert(dimPair.second);

1623  "parallel index should be free in LHS or batch in LHS/RHS");
1624  dimSize = lhsType.getDimSize(lhsIndex);
1625  } else {
1626  assert(rhsIndex >= 0 && "missing parallel index");
1627  iterIndex = iMap[1].getDimPosition(rhsIndex);
1628  dimSize = rhsType.getDimSize(rhsIndex);
1629  }
1630  assert(iterIndex >= 0 && "parallel index not listed in operand mapping");
1631  Optional<int64_t> lookup = getResultIndex(iMap[2], iterIndex);
1632  assert(lookup.hasValue() && "parallel index not listed in reduction");
1633  int64_t resIndex = lookup.getValue();
1634  // Construct new iterator types and affine map array attribute.
1635  std::array<AffineMap, 3> lowIndexingMaps = {
1636  adjustMap(iMap[0], iterIndex, rewriter),
1637  adjustMap(iMap[1], iterIndex, rewriter),
1638  adjustMap(iMap[2], iterIndex, rewriter)};
1639  auto lowAffine = rewriter.getAffineMapArrayAttr(lowIndexingMaps);
1640  auto lowIter =
1641  rewriter.getArrayAttr(adjustIter(op.iterator_types(), iterIndex));
1642  // Unroll into a series of lower dimensional vector.contract ops.
1643  Location loc = op.getLoc();
1644  Value result =
1645  rewriter.create<ConstantOp>(loc, resType, rewriter.getZeroAttr(resType));
1646  for (int64_t d = 0; d < dimSize; ++d) {
1647  auto lhs = reshapeLoad(loc, op.lhs(), lhsType, lhsIndex, d, rewriter);
1648  auto rhs = reshapeLoad(loc, op.rhs(), rhsType, rhsIndex, d, rewriter);
1649  auto acc = reshapeLoad(loc, op.acc(), resType, resIndex, d, rewriter);
1650  Value lowContract = rewriter.create<vector::ContractionOp>(
1651  loc, lhs, rhs, acc, lowAffine, lowIter);
1652  result =
1653  reshapeStore(loc, lowContract, result, resType, resIndex, d, rewriter);
1654  }
1655  return result;

1658 // Lower one reduction dimension.
1659 Value ContractionOpLowering::lowerReduction(vector::ContractionOp op,
1660  PatternRewriter &rewriter) const {
1661  auto loc = op.getLoc();
1662  VectorType lhsType = op.getLhsType();
1663  VectorType rhsType = op.getRhsType();
1664  Type resType = op.getResultType();
1665  assert(!resType.isa<VectorType>());
1666  bool isInt = resType.isa<IntegerType>();
1667  // Use iterator index 0.
1668  int64_t iterIndex = 0;
1669  SmallVector<AffineMap, 4> iMap = op.getIndexingMaps();
1670  Optional<int64_t> lookupLhs = getResultIndex(iMap[0], iterIndex);
1671  Optional<int64_t> lookupRhs = getResultIndex(iMap[1], iterIndex);
1672  assert(lookupLhs.hasValue() && "missing LHS parallel index");
1673  assert(lookupRhs.hasValue() && "missing RHS parallel index");
1674  int64_t lhsIndex = lookupLhs.getValue();
1675  int64_t rhsIndex = lookupRhs.getValue();
1676  int64_t dimSize = lhsType.getDimSize(lhsIndex);
1677  assert(dimSize == rhsType.getDimSize(rhsIndex) && "corrupt shape");
1678  // Base case.
1679  if (lhsType.getRank() == 1) {
1680  assert(rhsType.getRank() == 1 && "corrupt contraction");
1681  Value m = createMul(loc, op.lhs(), op.rhs(), isInt, rewriter);
1682  StringAttr kind = rewriter.getStringAttr("add");
1683  Value res = rewriter.create<vector::ReductionOp>(loc, resType, kind, m,
1684  ValueRange{});
1685  if (auto acc = op.acc())
1686  res = createAdd(op.getLoc(), res, acc, isInt, rewriter);
1687  return res;
1688  }
1689  // Construct new iterator types and affine map array attribute.
1690  std::array<AffineMap, 3> lowIndexingMaps = {
1691  adjustMap(iMap[0], iterIndex, rewriter),
1692  adjustMap(iMap[1], iterIndex, rewriter),
1693  adjustMap(iMap[2], iterIndex, rewriter)};
1694  auto lowAffine = rewriter.getAffineMapArrayAttr(lowIndexingMaps);
1695  auto lowIter =
1696  rewriter.getArrayAttr(adjustIter(op.iterator_types(), iterIndex));
1697  // Unroll into a series of lower dimensional vector.contract ops.
1698  // By feeding the initial accumulator into the first contraction,
1699  // and the result of each contraction into the next, eventually
1700  // the sum of all reductions is computed.
1701  Value result = op.acc();
1702  for (int64_t d = 0; d < dimSize; ++d) {
1703  auto lhs = reshapeLoad(loc, op.lhs(), lhsType, lhsIndex, d, rewriter);
1704  auto rhs = reshapeLoad(loc, op.rhs(), rhsType, rhsIndex, d, rewriter);
1705  result = rewriter.create<vector::ContractionOp>(loc, lhs, rhs, result,
1706  lowAffine, lowIter);
1707  }
1708  return result;
1709 }

1733 // Operates under a scoped context to build the condition to ensure that a
1734 // particular VectorTransferOpInterface is in-bounds.
1735 static Value createInBoundsCond(OpBuilder &b,
1736  VectorTransferOpInterface xferOp) {
1737  assert(xferOp.permutation_map().isMinorIdentity() &&
1738  "Expected minor identity map");
1739  Value inBoundsCond;
1740  xferOp.zipResultAndIndexing([&](int64_t resultIdx, int64_t indicesIdx) {
1741  // Zip over the resulting vector shape and memref indices.
1742  // If the dimension is known to be in-bounds, it does not participate in
1743  // the construction of `inBoundsCond`.
1744  if (xferOp.isDimInBounds(resultIdx))
1745  return;
1746  // Fold or create the check that `index + vector_size` <= `memref_size`.
1747  Location loc = xferOp.getLoc();
1748  ImplicitLocOpBuilder lb(loc, b);
1749  int64_t vectorSize = xferOp.getVectorType().getDimSize(resultIdx);
1750  auto d0 = getAffineDimExpr(0, xferOp.getContext());
1751  auto vs = getAffineConstantExpr(vectorSize, xferOp.getContext());

1753  makeComposedAffineApply(b, loc, d0 + vs, xferOp.indices()[indicesIdx]);
1754  Value cond = createFoldedSLE(
1755  b, sum, vector::createOrFoldDimOp(b, loc, xferOp.source(), indicesIdx));
1756  if (!cond)
1757  return;

1759  if (inBoundsCond)
1760  inBoundsCond = lb.create<AndOp>(inBoundsCond, cond);
1761  else
1762  inBoundsCond = cond;
1763  });

1783 /// Given two MemRefTypes `aT` and `bT`, return a MemRefType to which both can
1784 /// be cast. If the MemRefTypes don't have the same rank or are not strided,
1785 /// return null; otherwise:
1786 /// 1. if `aT` and `bT` are cast-compatible, return `aT`.
1787 /// 2. else return a new MemRefType obtained by iterating over the shape and
1788 /// strides and:
1789 /// a. keeping the ones that are static and equal across `aT` and `bT`.
1790 /// b. using a dynamic shape and/or stride for the dimensions that don't
1791 /// agree.
1792 static MemRefType getCastCompatibleMemRefType(MemRefType aT, MemRefType bT) {
1793  if (memref::CastOp::areCastCompatible(aT, bT))
1794  return aT;
1795  if (aT.getRank() != bT.getRank())
1796  return MemRefType();
1797  int64_t aOffset, bOffset;
1798  SmallVector<int64_t, 4> aStrides, bStrides;
1799  if (failed(getStridesAndOffset(aT, aStrides, aOffset)) ||
1800  failed(getStridesAndOffset(bT, bStrides, bOffset)) ||
1801  aStrides.size() != bStrides.size())
1802  return MemRefType();
1803 
1804  ArrayRef<int64_t> aShape = aT.getShape(), bShape = bT.getShape();
1805  int64_t resOffset;
1806  SmallVector<int64_t, 4> resShape(aT.getRank(), 0),
1807  resStrides(bT.getRank(), 0);
1808  for (int64_t idx = 0, e = aT.getRank(); idx < e; ++idx) {
1809  resShape[idx] =
1810  (aShape[idx] == bShape[idx]) ? aShape[idx] : MemRefType::kDynamicSize;
1811  resStrides[idx] = (aStrides[idx] == bStrides[idx])
1812  ? aStrides[idx]
1813  : MemRefType::kDynamicStrideOrOffset;
1814  }
1815  resOffset =
1816  (aOffset == bOffset) ? aOffset : MemRefType::kDynamicStrideOrOffset;
1817  return MemRefType::get(
1818  resShape, aT.getElementType(),
1819  makeStridedLinearLayoutMap(resStrides, resOffset, aT.getContext()));
1820 }

1822 /// Operates under a scoped context to build the intersection between the
1823 /// view `xferOp.source()` @ `xferOp.indices()` and the view `alloc`.
1824 // TODO: view intersection/union/differences should be a proper std op.
1825 static Value createSubViewIntersection(OpBuilder &b,
1826  VectorTransferOpInterface xferOp,
1827  Value alloc) {
1828  ImplicitLocOpBuilder lb(xferOp.getLoc(), b);
1829  int64_t memrefRank = xferOp.getShapedType().getRank();
1830  // TODO: relax this precondition, will require rank-reducing subviews.
1831  assert(memrefRank == alloc.getType().cast<MemRefType>().getRank() &&
1832  "Expected memref rank to match the alloc rank");
1833  ValueRange leadingIndices =
1834  xferOp.indices().take_front(xferOp.getLeadingShapedRank());
1835  SmallVector<OpFoldResult, 4> sizes;
1836  sizes.append(leadingIndices.begin(), leadingIndices.end());
1837  auto isaWrite = isa<vector::TransferWriteOp>(xferOp);
1838  xferOp.zipResultAndIndexing([&](int64_t resultIdx, int64_t indicesIdx) {
1839  using MapList = ArrayRef<ArrayRef<AffineExpr>>;
1840  Value dimMemRef = vector::createOrFoldDimOp(b, xferOp.getLoc(),
1841  xferOp.source(), indicesIdx);
1842  Value dimAlloc = lb.create<memref::DimOp>(alloc, resultIdx);
1843  Value index = xferOp.indices()[indicesIdx];
1844  AffineExpr i, j, k;
1845  bindDims(xferOp.getContext(), i, j, k);
1846  SmallVector<AffineMap, 4> maps =
1847  AffineMap::inferFromExprList(MapList{{i - j, k}});
1848  // affine_min(%dimMemRef - %index, %dimAlloc)
1849  Value affineMin = lb.create<AffineMinOp>(
1850  index.getType(), maps[0], ValueRange{dimMemRef, index, dimAlloc});
1851  sizes.push_back(affineMin);
1852  });
1853 
1854  SmallVector<OpFoldResult, 4> indices = llvm::to_vector<4>(llvm::map_range(
1855  xferOp.indices(), [](Value idx) -> OpFoldResult { return idx; }));
1856  return lb.create<memref::SubViewOp>(
1857  isaWrite ? alloc : xferOp.source(), indices, sizes,
1858  SmallVector<OpFoldResult>(memrefRank, OpBuilder(xferOp).getIndexAttr(1)));
1859 }

2068 /// masking) fastpath and a slowpath.
2069 ///
2070 /// For vector.transfer_read:
2071 /// If `ifOp` is not null and the result is `success, the `ifOp` points to the

2078 /// ```
2079 /// %1 = vector.transfer_read %0[...], %pad : memref<A...>, vector<...>
2080 /// ```
2081 /// is transformed into:
2082 /// ```
2083 /// %1:3 = scf.if (%inBounds) {
2084 /// // fastpath, direct cast
2085 /// memref.cast %A: memref<A...> to compatibleMemRefType
2086 /// scf.yield %view : compatibleMemRefType, index, index
2087 /// } else {
2088 /// // slowpath, not in-bounds vector.transfer or linalg.copy.
2089 /// memref.cast %alloc: memref<B...> to compatibleMemRefType
2090 /// scf.yield %4 : compatibleMemRefType, index, index
2091 // }
2092 /// %0 = vector.transfer_read %1#0[%1#1, %1#2] {in_bounds = [true ... true]}
2093 /// ```
2094 /// where `alloc` is a top of the function alloca'ed buffer of one vector.
2095 ///
2096 /// For vector.transfer_write:
2097 /// There are 2 conditional blocks. First a block to decide which memref and
2098 /// indices to use for an unmasked, inbounds write. Then a conditional block to
2099 /// further copy a partial buffer into the final result in the slow path case.
2100 ///
2101 /// Example (a 2-D vector.transfer_write):
2102 /// ```
2103 /// vector.transfer_write %arg, %0[...], %pad : memref<A...>, vector<...>
2104 /// ```
2105 /// is transformed into:
2106 /// ```

2141  // ensure they aren't used in the wrong scopes further down.

3351 // Helper that returns a vector comparison that constructs a mask:
3352 // mask = [0,1,..,n-1] + [o,o,..,o] < [b,b,..,b]
