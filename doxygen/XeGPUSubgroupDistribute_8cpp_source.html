104 /// Helper class for tracking the analysis state of an mlir value. For layout
105 /// propagation, the analysis state is simply the lane_layout and lane_data of
106 /// each value. Purpose of this analysis to propagate some unique layout for
107 /// each value in the program starting from a set of anchor operations (like
108 /// DPAS, StoreNd, etc.).
109 ///
110 /// Given this, LayoutInfo satisifies the following properties:
111 /// 1) A LayoutInfo value can be in one of two states - `assigned` or `not
112 /// assigned`.
113 /// 2) Two LayoutInfo values are equal if they are both assigned or
114 /// both not assigned. The concrete value of assigned state does not matter.
115 /// 3) The meet operator works as follows:
116 /// - If current state is assigned, return the current state. (already
117 /// a unique layout is assigned. don't change it)
118 /// - Otherwise, return the other state.
119 
120 struct LayoutInfo {
121 private:
122  LaneLayout laneLayout;
123  LaneData laneData;
124 
125 public:
126  LayoutInfo() = default;
127  LayoutInfo(const LaneLayout &layout, const LaneData &data)
128  : laneLayout(layout), laneData(data) {}
129 
130  // Two lattice values are equal if they have `some` layout. The actual
131  // content of the layout does not matter.
132  bool operator==(const LayoutInfo &other) const {
133  return this->isAssigned() == other.isAssigned();
134  }
135 
136  static LayoutInfo meet(const LayoutInfo &lhs, const LayoutInfo &rhs);
137 
138  static LayoutInfo join(const LayoutInfo &lhs, const LayoutInfo &rhs);
139 
140  void print(raw_ostream &os) const;
141 
142  bool isAssigned() const {
143  return laneLayout.size() > 0 && laneData.size() > 0;
144  }
145 
146  LayoutInfo getTransposedLayout(ArrayRef<int64_t> permutation) const;
147 
148  const LaneLayout &getLayout() const { return laneLayout; }
149  const LaneData &getData() const { return laneData; }
150  ArrayRef<int64_t> getLayoutAsArrayRef() const { return laneLayout.layout; }
151  ArrayRef<int64_t> getDataAsArrayRef() const { return laneData.layout; }
152 };

199 /// Helper Functions to get default layouts. A `default layout` is a layout that
200 /// is assigned to a value when the layout is not fixed by some anchor operation
201 /// (like DPAS).
202 
203 /// Helper Function to get the default layout for uniform values like constants.
204 /// For 1D vector, lane_layout is [subgroupSize] and lane_data is [1].
205 /// For 2D vector, lane_layout is [1, subgroupSize] and lane_data is [1, 1].
206 static LayoutInfo getDefaultLayoutInfo(unsigned rank) {
207  assert((rank == 1 || rank == 2) && "Expected 1D or 2D vector.");
208  if (rank == 1)
209  return LayoutInfo(LaneLayout({subgroupSize}), LaneData({1}));
210  return LayoutInfo(LaneLayout({1, subgroupSize}), LaneData({1, 1}));
211 }
212 
213 /// Helper to get the default layout for a vector type.
214 static LayoutInfo getDefaultLayoutInfo(VectorType vectorTy) {
215  // Expecting a 1D or 2D vector.
216  assert((vectorTy.getRank() == 1 || vectorTy.getRank() == 2) &&
217  "Expected 1D or 2D vector.");
218  // Expecting int or float element type.
219  assert(vectorTy.getElementType().isIntOrFloat() &&
220  "Expected int or float element type.");
221  // If the rank is 1, then return default layout for 1D vector.
222  if (vectorTy.getRank() == 1)
223  return getDefaultLayoutInfo(1);
224  // Packing factor is determined by the element type bitwidth.
225  int packingFactor = 1;
226  unsigned bitwidth = vectorTy.getElementType().getIntOrFloatBitWidth();
227  if (bitwidth < packedSizeInBitsForDefault)
228  packingFactor = packedSizeInBitsForDefault / bitwidth;
229  return LayoutInfo(LaneLayout({1, subgroupSize}),
230  LaneData({1, packingFactor}));
231 }

233 /// Helper Function to get the expected layouts for DPAS operands. `lane_data`
234 /// is set according to the following criteria:
235 /// * For A operand, the data must be packed in minimum
236 /// `packedSizeInBitsForDefault`
237 /// * For B operand, the data must be packed in minimum
238 /// `packedSizeInBitsForDpasB`
239 static LayoutInfo getLayoutInfoForDPASOperand(VectorType vectorTy,
240  unsigned operandNum) {
241  Type elementTy = vectorTy.getElementType();
242  assert(elementTy.isIntOrFloat() &&
243  "Expected int or float type in DPAS operands");
244  LaneLayout layout({1, subgroupSize});
245  // For B operand, data must be packed in minimum `packedDpasBSizeInBits` and
246  // must have the VNNI format.
247  if (operandNum == 1 &&
248  elementTy.getIntOrFloatBitWidth() < packedSizeInBitsForDpasB) {
249  LaneData data(
250  {packedSizeInBitsForDpasB / elementTy.getIntOrFloatBitWidth(), 1});
251  return LayoutInfo(layout, data);
252  }
253  // Otherwise, return the default layout for the vector type.
254  return getDefaultLayoutInfo(vectorTy);
255 }

337 LogicalResult LayoutInfoPropagation::visitOperation(
338  Operation *op, ArrayRef<LayoutInfoLattice *> operands,
339  ArrayRef<const LayoutInfoLattice *> results) {
340  TypeSwitch<Operation *>(op)
341  .Case<xegpu::DpasOp>(
342  [&](auto dpasOp) { visitDpasOp(dpasOp, operands, results); })
343  .Case<xegpu::StoreNdOp>(
344  [&](auto storeNdOp) { visitStoreNdOp(storeNdOp, operands, results); })
345  .Case<xegpu::StoreScatterOp>([&](auto storeScatterOp) {
346  visitStoreScatterOp(storeScatterOp, operands, results);
347  })
348  .Case<xegpu::LoadNdOp>(
349  [&](auto loadNdOp) { visitLoadNdOp(loadNdOp, operands, results); })
350  .Case<xegpu::LoadGatherOp>([&](auto loadGatherOp) {
351  visitLoadGatherOp(loadGatherOp, operands, results);
352  })
353  .Case<xegpu::CreateDescOp>([&](auto createDescOp) {
354  visitCreateDescOp(createDescOp, operands, results);
355  })
356  .Case<xegpu::UpdateNdOffsetOp>([&](auto updateNdOffsetOp) {
357  visitUpdateNdOffsetOp(updateNdOffsetOp, operands, results);
358  })
359  .Case<xegpu::PrefetchNdOp>([&](auto prefetchNdOp) {
360  visitPrefetchNdOp(prefetchNdOp, operands, results);
361  })
362  // No need to propagate the layout to operands in CreateNdDescOp because
363  // they are scalars (offsets, sizes, etc.).
364  .Case<xegpu::CreateNdDescOp>([&](auto createNdDescOp) {})
365  .Case<vector::TransposeOp>([&](auto transposeOp) {
366  visitTransposeOp(transposeOp, operands, results);
367  })
368  .Case<vector::BitCastOp>([&](auto bitcastOp) {
369  visitVectorBitcastOp(bitcastOp, operands, results);
370  })
371  .Case<vector::MultiDimReductionOp>([&](auto reductionOp) {
372  visitVectorMultiReductionOp(reductionOp, operands, results);
373  })
374  // All other ops.
375  .Default([&](Operation *op) {
376  for (const LayoutInfoLattice *r : results) {
377  for (LayoutInfoLattice *operand : operands) {
378  // Propagate the layout of the result to the operand.
379  if (r->getValue().isAssigned())
380  meet(operand, *r);
381  }
382  }
383  });
384  // Add a dependency from each result to program point after the operation.
385  for (const LayoutInfoLattice *r : results) {
386  addDependency(const_cast<LayoutInfoLattice *>(r), getProgramPointAfter(op));
387  }
388  return success();
389 }

500 /// For vector::BitCastOp, the lane_data of the source layout is changed based
501 /// on the bit width of the source and result types.
502 void LayoutInfoPropagation::visitVectorBitcastOp(
503  vector::BitCastOp bitcast, ArrayRef<LayoutInfoLattice *> operands,
504  ArrayRef<const LayoutInfoLattice *> results) {
505  // Need the layout of bitcast result to propagate to the operands.
506  LayoutInfo resultLayout = results[0]->getValue();
507  if (!resultLayout.isAssigned())
508  return;
509  int inElemTyBitWidth =
510  bitcast.getSourceVectorType().getElementType().getIntOrFloatBitWidth();
511  int outElemTyBitWidth =
512  bitcast.getResultVectorType().getElementType().getIntOrFloatBitWidth();
513 
514  // LaneLayout does not change.
515  const LaneLayout &newLaneLayout = resultLayout.getLayout();
516  const LaneData &currData = resultLayout.getData();
517  LaneData newLaneData;
518  // It's a widening bitcast
519  if (inElemTyBitWidth < outElemTyBitWidth) {
520  int ratio = outElemTyBitWidth / inElemTyBitWidth;
521  newLaneData = resultLayout.getData()[0] == 1
522  ? LaneData({1, currData[1] * ratio})
523  : LaneData({currData[0] * ratio, 1});
524  } else {
525  // It's a narrowing bitcast
526  int ratio = inElemTyBitWidth / outElemTyBitWidth;
527  newLaneData = resultLayout.getData()[0] == 1
528  ? LaneData({1, currData[1] / ratio})
529  : LaneData({currData[0] / ratio, 1});
530  }
531 
532  propagateIfChanged(operands[0],
533  operands[0]->meet(LayoutInfo(newLaneLayout, newLaneData)));
534 }

573  LayoutInfo layout = getDefaultLayoutInfo(1);

577 /// Set the layout for the value, tensor descriptor, and mask operands in the
578 /// StoreScatterOp.
579 void LayoutInfoPropagation::visitStoreScatterOp(
580  xegpu::StoreScatterOp storeScatter, ArrayRef<LayoutInfoLattice *> operands,
581  ArrayRef<const LayoutInfoLattice *> results) {
582  // Currently, for 2D StoreScatterOp we expect that the height dimension of
583  // the tensor descriptor is equal to the subgroup size. This is ensured by
584  // the op verifier.
585  ArrayRef<int64_t> tdescShape = storeScatter.getTensorDescType().getShape();
586  if (tdescShape.size() > 1)
587  assert(
588  tdescShape[0] == subgroupSize &&
589  "Expected the first dimension of 2D tensor descriptor to be equal to "
590  "subgroup size.");
591 
592  LayoutInfo valueLayout = getDefaultLayoutInfo(storeScatter.getValueType());
593  LayoutInfo storeScatterLayout = valueLayout;
594  if (storeScatter.getTranspose()) {
595  // StoreScatteOp allows transpose effect. However, at the stage of this
596  // analyis this effect is not expected and should be abstracted away. Emit
597  // a warning.
598  storeScatter.emitWarning("Transpose effect is not expected for "
599  "StoreScatterOp at LayoutInfoPropagation stage.");
600  storeScatterLayout = valueLayout.getTransposedLayout({1, 0});
601  }
602  // Propagate the value layout.
603  propagateIfChanged(operands[0], operands[0]->meet(valueLayout));
604  // Propagate the tensor descriptor layout.
605  propagateIfChanged(operands[1], operands[1]->meet(storeScatterLayout));
606  // Use default 1D layout for mask operand.
607  LayoutInfo maskLayout = getDefaultLayoutInfo(1);
608  propagateIfChanged(operands[2], operands[2]->meet(maskLayout));
609 }

753 /// Assign xegpu::LayoutAttr to the op and its users. The layout is assigned
754 /// based on the layout propagation analysis result.
755 LogicalResult LayoutAttrAssignment::assign(Operation *op) {
756  // For function ops, propagate the function argument layout to the users.
757  if (auto func = dyn_cast<FunctionOpInterface>(op)) {
758  for (BlockArgument arg : func.getArguments()) {
759  xegpu::LayoutAttr layoutInfo = getLayoutAttrForValue(arg);
760  if (layoutInfo) {
761  assignToUsers(arg, layoutInfo);
762  }
763  }
764  return success();
765  }
766  // If no results, move on.
767  if (op->getNumResults() == 0)
768  return success();
769  // If all the results are scalars, move on.
770  if (llvm::all_of(op->getResultTypes(),
771  [](Type t) { return t.isIntOrIndexOrFloat(); }))
772  return success();
773  // If the op has more than one result and at least one result is a tensor
774  // descriptor, exit. This case is not supported yet.
775  // TODO: Support this case.
776  if (op->getNumResults() > 1 && llvm::any_of(op->getResultTypes(), [](Type t) {
777  return isa<xegpu::TensorDescType>(t);
778  })) {
779  LLVM_DEBUG(
780  DBGS() << op->getName()
781  << " op has more than one result and at least one is a tensor "
782  "descriptor. This case is not handled.\n");
783  return failure();
784  }
785  // If the result is a tensor descriptor, attach the layout to the tensor
786  // descriptor itself.
787  if (auto tensorDescTy =
788  dyn_cast<xegpu::TensorDescType>(op->getResultTypes()[0])) {
789  xegpu::LayoutAttr layoutInfo = getLayoutAttrForValue(op->getResult(0));
790  if (!layoutInfo) {
791  LLVM_DEBUG(DBGS() << "No layout for result of " << *op << "\n");
792  return failure();
793  }
794 
795  // Clone the op, attach the layout to the result tensor descriptor, and
796  // remove the original op.
797  OpBuilder builder(op);
798  Operation *newOp = builder.clone(*op);
799  auto newTensorDescTy = xegpu::TensorDescType::get(
800  tensorDescTy.getContext(), tensorDescTy.getShape(),
801  tensorDescTy.getElementType(), tensorDescTy.getEncoding(), layoutInfo);
802  newOp->getResult(0).setType(newTensorDescTy);
803  op->replaceAllUsesWith(newOp->getResults());
804  op->erase();
805  return success();
806  }
807  // Otherwise simply attach the layout to the op itself.
808  for (auto [i, r] : llvm::enumerate(op->getResults())) {
809  xegpu::LayoutAttr layoutInfo = getLayoutAttrForValue(r);
810  if (layoutInfo) {
811  std::string attrName = resultLayoutNamePrefix + std::to_string(i);
812  op->setAttr(attrName, layoutInfo);
813  // Attach the layout attribute to the users of the result.
814  assignToUsers(r, layoutInfo);
815  }
816  }
817  return success();
818 }

848 /// Helper function to get distributed vector type for a source vector type
849 /// according to the lane_layout. We simply divide each dimension of tensor
850 /// descriptor shape by corresponding lane_layout dimension. If
851 /// array_length > 1, that is appended to the front of the ditributed shape.
852 /// NOTE: This is the vector type that will be returned by the
853 /// gpu.warp_execute_on_lane0 op.
854 ///
855 /// Examples:
856 /// | original vector shape | lane_layout | distributed vector shape |
857 /// |-----------------------|-------------|--------------------------|
858 /// | 32x16 | [1, 16] | 32x1 |
859 /// | 32x16 | [2, 8] | 16x2 |
860 /// | 2x32x16 | [1, 16] | 2x32x1 |
861 static FailureOr<VectorType>
862 getDistVecTypeBasedOnLaneLayout(xegpu::LayoutAttr layout,
863  VectorType originalType) {
864  if (!layout)
865  return failure();
866 
867  auto laneLayout = layout.getLaneLayout().asArrayRef();
868  assert(originalType.getShape().size() >= laneLayout.size() &&
869  "Rank of the original vector type should be greater or equal to the "
870  "size of the lane layout to distribute the vector type.");
871  SmallVector<int64_t> distributedShape(originalType.getShape());
872  // Only distribute the last `laneLayout.size()` dimensions. The remaining
873  // dimensions are not distributed.
874  unsigned distributionStart = originalType.getRank() - laneLayout.size();
875  for (auto [i, dim] : llvm::enumerate(originalType.getShape())) {
876  if (i < distributionStart) {
877  continue;
878  }
879  // Check if the dimension can be distributed evenly.
880  if (dim % laneLayout[i - distributionStart] != 0)
881  return failure();
882  distributedShape[i] = dim / laneLayout[i - distributionStart];
883  }
884  return VectorType::get(distributedShape, originalType.getElementType());
885 }

887 /// Helper function to resolve types if the distributed type out of
888 /// gpu.warp_execute_on_lane0 is different from the expected xegpu SIMT type.
889 /// Example 1:
890 /// distributed type: vector<8x1xf32>
891 /// expected type: vector<8xf32>
892 /// resolved using,
893 /// %0 = vector.shape_cast %1 : vector<8x1xf32> to vector<8xf32>
894 /// Example 2:
895 /// distributed type: xegpu.tensor_desc<8x16xf32, #xegpu.layout<...>>
896 /// expected type: xegpu.tensor_desc<8x16xf32>
897 /// resolved using,
898 /// %0 = unrealized_conversion_cast %1 :
899 /// xegpu.tensor_desc<8x16xf32, #xegpu.layout<..>> ->
900 /// xegpu.tensor_desc<8x16xf32>
901 template <typename T>
902 static Value resolveDistributedTy(Value orig, T expected,
903  PatternRewriter &rewriter) {
904  // If orig and expected types are the same, return orig.
905  if (orig.getType() == expected)
906  return orig;
907  // If orig is a vector type, create a shape cast op to reconcile the types.
908  if (isa<VectorType>(orig.getType())) {
909  auto castOp =
910  rewriter.create<vector::ShapeCastOp>(orig.getLoc(), expected, orig);
911  return castOp.getResult();
912  }
913  // If orig is a tensor descriptor type, create an unrealized conversion cast
914  // op to reconcile the types.
915  if (isa<xegpu::TensorDescType>(orig.getType())) {
916  auto castOp = rewriter.create<UnrealizedConversionCastOp>(orig.getLoc(),
917  expected, orig);
918  return castOp.getResult(0);
919  }
920  llvm_unreachable("Unsupported type for reconciliation");
921  return orig;
922 }

951 /// Given a GPUFuncOp, this pattern creates a new GPUFuncOp and moves the body
952 /// of the original GPUFuncOp to the new GPUFuncOp such that entire body is
953 /// contained within a WarpExecuteOnLane0Op.
954 /// Example:
955 ///
956 /// ```
957 /// gpu.func @foo(%arg0: memref<*xf16>) -> vector<8x16xf32> {
958 /// ...
959 /// ...
960 /// gpu.return %result: vector<8x16xf32>
961 /// }
962 /// ```
963 /// To
964 /// ```
965 /// gpu.func @foo(%arg0: memref<*xf16>) -> vector<8x16xf32> {
966 /// %laneid = gpu.lane_id : index
967 /// %0 = gpu.warp_execute_on_lane_0(%laneid) -> vector<8x16xf32> {
968 /// ...
969 /// ...
970 /// gpu.yield %result: vector<8x16xf32>
971 /// }
972 /// return %0
973 /// }
974 struct MoveFuncBodyToWarpExecuteOnLane0
975  : public OpRewritePattern<gpu::GPUFuncOp> {
976  using OpRewritePattern<gpu::GPUFuncOp>::OpRewritePattern;
977  LogicalResult matchAndRewrite(gpu::GPUFuncOp gpuFuncOp,
978  PatternRewriter &rewriter) const override {
979  // If the function only contains a single void return, skip.
980  if (llvm::all_of(gpuFuncOp.getBody().getOps(), [](Operation &op) {
981  return isa<gpu::ReturnOp>(op) && !op.getNumOperands();
982  }))
983  return failure();
984  // If the function already moved inside a warp_execute_on_lane0, skip.
985  if (llvm::any_of(gpuFuncOp.getBody().getOps(), [](Operation &op) {
986  return isa<gpu::WarpExecuteOnLane0Op>(op);
987  }))
988  return failure();
989  // Create a new function with the same signature.
990  auto newGpuFunc = rewriter.create<gpu::GPUFuncOp>(
991  gpuFuncOp.getLoc(), gpuFuncOp.getName(), gpuFuncOp.getFunctionType());
992  // Create a WarpExecuteOnLane0Op with same arguments and results as the
993  // original gpuFuncOp.
994  rewriter.setInsertionPointToEnd(&newGpuFunc.getFunctionBody().front());
995  auto laneId = rewriter.create<gpu::LaneIdOp>(
996  newGpuFunc.getLoc(), rewriter.getIndexType(),
997  /** upperBound = **/ mlir::IntegerAttr());
998  ArrayRef<Type> gpuFuncResultType = gpuFuncOp.getFunctionType().getResults();
999  auto warpOp = rewriter.create<gpu::WarpExecuteOnLane0Op>(
1000  laneId.getLoc(), gpuFuncResultType, laneId, subgroupSize,
1001  newGpuFunc.getArguments(), newGpuFunc.getArgumentTypes());
1002  Block &warpBodyBlock = warpOp.getBodyRegion().front();
1003  // Replace the ReturnOp of the original gpu function with a YieldOp.
1004  auto origRetunOp =
1005  cast<gpu::ReturnOp>(gpuFuncOp.getBlocks().back().getTerminator());
1006  rewriter.setInsertionPointAfter(origRetunOp);
1007  rewriter.create<gpu::YieldOp>(origRetunOp.getLoc(),
1008  origRetunOp.getOperands());
1009  rewriter.eraseOp(origRetunOp);
1010  // Move the original function body to the WarpExecuteOnLane0Op body.
1011  rewriter.inlineRegionBefore(gpuFuncOp.getBody(), warpOp.getBodyRegion(),
1012  warpOp.getBodyRegion().begin());
1013  rewriter.eraseBlock(&warpBodyBlock);
1014  // Insert a new ReturnOp after the WarpExecuteOnLane0Op.
1015  rewriter.setInsertionPointAfter(warpOp);
1016  rewriter.create<gpu::ReturnOp>(newGpuFunc.getLoc(), warpOp.getResults());
1017  rewriter.replaceOp(gpuFuncOp, newGpuFunc);
1018  return success();
1019  }

1022 /// Distribute a create_nd_tdesc feeding into vector.yield op of the enclosing
1023 /// `gpu.warp_execute_on_lane_0` region. After the sinking, the warp op will
1024 /// still contain the original op that will not be used by the yield op (and
1025 /// should be cleaned up later). The yield op will bypass the create_nd_tdesc's
1026 /// arguments. Tensor descriptor shape is not distributed because it is a
1027 /// uniform value across all work items within the subgroup. However, the
1028 /// layout information is dropped in the new tensor descriptor type.
1029 ///
1030 /// Example:
1031 ///
1032 /// ```
1033 /// #layout0 = #xegpu.layout<wi_layout = [1, 8], wi_data = [1, 1]>
1034 /// %r = gpu.warp_execute_on_lane_0(%laneid) ->
1035 /// (!xegpu.tensor_desc<4x8xf32, #layout0>) {

1037 /// %td = xegpu.create_nd_tdesc %arg0[0, 0]
1038 /// : memref<4x8xf32> -> !xegpu.tensor_desc<4x8xf32, #layout0>
1039 /// vector.yield %td
1040 /// }
1041 /// ```

1043 /// ```
1044 /// %r:2 = gpu.warp_execute_on_lane_0(%laneid) -> (...) {
1045 /// ...

1047 /// : memref<4x8xf32> -> !xegpu.tensor_desc<4x8xf32, #layout0>
1048 /// vector.yield %arg0, %dead
1049 /// }
1050 /// %td = xegpu.create_nd_tdesc %r#0[0, 0]: memref<4x8xf32>
1051 /// -> !xegpu.tensor_desc<4x8xf32>
1052 ///
1053 /// ```

1055  using gpu::WarpDistributionPattern::WarpDistributionPattern;
1056  LogicalResult matchAndRewrite(gpu::WarpExecuteOnLane0Op subgroupOp,
1057  PatternRewriter &rewriter) const override {
1058  OpOperand *operand =
1059  getWarpResult(subgroupOp, llvm::IsaPred<xegpu::CreateNdDescOp>);
1060  if (!operand)
1061  return rewriter.notifyMatchFailure(
1062  subgroupOp, "warp result is not a xegpu::CreateNdDesc op");
1063  auto descOp = operand->get().getDefiningOp<xegpu::CreateNdDescOp>();
1064  unsigned operandIdx = operand->getOperandNumber();
1065 
1066  xegpu::LayoutAttr layout = descOp.getType().getLayoutAttr();
1067  if (!layout)
1068  return rewriter.notifyMatchFailure(
1069  descOp, "the tensor descriptor lacks layout attribute");
1070 
1071  SmallVector<size_t> newRetIndices;
1072  SmallVector<Value> newYieldValues;
1073  SmallVector<Type> newYieldTypes;
1074 
1075  for (Value operand : descOp->getOperands()) {
1076  newYieldValues.push_back(operand);
1077  newYieldTypes.push_back(operand.getType());
1078  }
1079  rewriter.setInsertionPoint(subgroupOp);
1080  gpu::WarpExecuteOnLane0Op newWarpOp = moveRegionToNewWarpOpAndAppendReturns(
1081  rewriter, subgroupOp, /* new yieled values = */ newYieldValues,
1082  /* new yielded types = */ newYieldTypes, newRetIndices);
1083

1085  for (size_t i : newRetIndices) {
1086  newDescOperands.push_back(newWarpOp.getResult(i));
1087  }
1088  rewriter.setInsertionPointAfter(newWarpOp);
1089  xegpu::TensorDescType distributedTensorDescTy =
1090  descOp.getType().dropLayouts(); // Distributed tensor descriptor type
1091  // does not contain layout info.
1092  auto newDescOp = rewriter.create<xegpu::CreateNdDescOp>(
1093  newWarpOp.getLoc(), distributedTensorDescTy, newDescOperands,
1094  descOp->getAttrs());
1095 
1096  Value distributedVal = newWarpOp.getResult(operandIdx);
1097  rewriter.replaceAllUsesWith(distributedVal, newDescOp);
1098  return success();
1099  }
