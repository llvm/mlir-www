304 Type GPUDialect::parseType(DialectAsmParser &parser) const {
305  // Parse the main keyword for the type.
306  StringRef keyword;
307  if (parser.parseKeyword(&keyword))
308  return Type();
309  MLIRContext *context = getContext();
310 
311  // Handle 'async token' types.
312  if (keyword == "async.token")
313  return AsyncTokenType::get(context);
314 
315  if (keyword == "mma_matrix") {
316  SMLoc beginLoc = parser.getNameLoc();
317 
318  // Parse '<'.
319  if (parser.parseLess())
320  return nullptr;
321 
322  // Parse the size and elementType.
323  SmallVector<int64_t> shape;
324  Type elementType;
325  if (parser.parseDimensionList(shape, /*allowDynamic=*/false) ||
326  parser.parseType(elementType))
327  return nullptr;
328 
329  // Parse ','
330  if (parser.parseComma())
331  return nullptr;
332 
333  // Parse operand.
334  std::string operand;
335  if (failed(parser.parseOptionalString(&operand)))
336  return nullptr;
337 
338  // Parse '>'.
339  if (parser.parseGreater())
340  return nullptr;
341 
342  return MMAMatrixType::getChecked(mlir::detail::getDefaultDiagnosticEmitFn(
343  parser.getEncodedSourceLoc(beginLoc)),
344  shape, elementType, operand);
345  }
346 
347  if (keyword == getSparseHandleKeyword(SparseHandleKind::DnTensor))
348  return SparseDnTensorHandleType::get(context);
349  if (keyword == getSparseHandleKeyword(SparseHandleKind::SpMat))
350  return SparseSpMatHandleType::get(context);
351  if (keyword == getSparseHandleKeyword(SparseHandleKind::SpGEMMOp))
352  return SparseSpGEMMOpHandleType::get(context);
353 
354  parser.emitError(parser.getNameLoc(), "unknown gpu type: " + keyword);
355  return Type();
356 }

393 LogicalResult GPUDialect::verifyOperationAttribute(Operation *op,
394  NamedAttribute attr) {
395  if (attr.getName() == getKnownBlockSizeAttrHelper().getName())
396  return verifyKnownLaunchSizeAttr(op, attr);
397  if (attr.getName() == getKnownGridSizeAttrHelper().getName())
398  return verifyKnownLaunchSizeAttr(op, attr);
399  if (!llvm::isa<UnitAttr>(attr.getValue()) ||
400  attr.getName() != getContainerModuleAttrName())
401  return success();
402 
403  auto module = dyn_cast<ModuleOp>(op);
404  if (!module)
405  return op->emitError("expected '")
406  << getContainerModuleAttrName() << "' attribute to be attached to '"
407  << ModuleOp::getOperationName() << '\'';
408 
409  auto walkResult = module.walk([&module](LaunchFuncOp launchOp) -> WalkResult {
410  // Ignore launches that are nested more or less deep than functions in the
411  // module we are currently checking.
412  if (!launchOp->getParentOp() ||
413  launchOp->getParentOp()->getParentOp() != module)
414  return success();
415 
416  // Ignore launch ops with missing attributes here. The errors will be
417  // reported by the verifiers of those ops.
418  if (!launchOp->getAttrOfType<SymbolRefAttr>(
419  LaunchFuncOp::getKernelAttrName(launchOp->getName())))
420  return success();
421 
422  // Check that `launch_func` refers to a well-formed GPU kernel container.
423  StringAttr kernelContainerName = launchOp.getKernelModuleName();
424  Operation *kernelContainer = module.lookupSymbol(kernelContainerName);
425  if (!kernelContainer)
426  return launchOp.emitOpError()
427  << "kernel container '" << kernelContainerName.getValue()
428  << "' is undefined";
429 
430  // If the container is a GPU binary op return success.
431  if (isa<BinaryOp>(kernelContainer))
432  return success();
433 
434  auto kernelModule = dyn_cast<GPUModuleOp>(kernelContainer);
435  if (!kernelModule)
436  return launchOp.emitOpError()
437  << "kernel module '" << kernelContainerName.getValue()
438  << "' is undefined";
439 
440  // Check that `launch_func` refers to a well-formed kernel function.
441  Operation *kernelFunc = module.lookupSymbol(launchOp.getKernelAttr());
442  if (!kernelFunc)
443  return launchOp.emitOpError("kernel function '")
444  << launchOp.getKernel() << "' is undefined";
445  auto kernelConvertedFunction = dyn_cast<FunctionOpInterface>(kernelFunc);
446  if (!kernelConvertedFunction) {
447  InFlightDiagnostic diag = launchOp.emitOpError()
448  << "referenced kernel '" << launchOp.getKernel()
449  << "' is not a function";
450  diag.attachNote(kernelFunc->getLoc()) << "see the kernel definition here";
451  return diag;
452  }
453 
454  if (!kernelFunc->getAttrOfType<mlir::UnitAttr>(
455  GPUDialect::getKernelFuncAttrName()))
456  return launchOp.emitOpError("kernel function is missing the '")
457  << GPUDialect::getKernelFuncAttrName() << "' attribute";
458 
459  // TODO: If the kernel isn't a GPU function (which happens during separate
460  // compilation), do not check type correspondence as it would require the
461  // verifier to be aware of the type conversion.
462  auto kernelGPUFunction = dyn_cast<gpu::GPUFuncOp>(kernelFunc);
463  if (!kernelGPUFunction)
464  return success();
465 
466  unsigned actualNumArguments = launchOp.getNumKernelOperands();
467  unsigned expectedNumArguments = kernelGPUFunction.getNumArguments();
468  if (expectedNumArguments != actualNumArguments)
469  return launchOp.emitOpError("got ")
470  << actualNumArguments << " kernel operands but expected "
471  << expectedNumArguments;
472 
473  auto functionType = kernelGPUFunction.getFunctionType();
474  for (unsigned i = 0; i < expectedNumArguments; ++i) {
475  if (launchOp.getKernelOperand(i).getType() != functionType.getInput(i)) {
476  return launchOp.emitOpError("type of function argument ")
477  << i << " does not match";
478  }
479  }
480 
481  return success();
482  });
483 
484  return walkResult.wasInterrupted() ? failure() : success();
485 }

598 LogicalResult gpu::AllReduceOp::verifyRegions() {
599  if (getBody().empty() != getOp().has_value())
600  return emitError("expected either an op attribute or a non-empty body");
601  if (!getBody().empty()) {
602  if (getBody().getNumArguments() != 2)
603  return emitError("expected two region arguments");
604  for (auto argument : getBody().getArguments()) {
605  if (argument.getType() != getType())
606  return emitError("incorrect region argument type");
607  }
608  unsigned yieldCount = 0;
609  for (Block &block : getBody()) {
610  if (auto yield = dyn_cast<gpu::YieldOp>(block.getTerminator())) {
611  if (yield.getNumOperands() != 1)
612  return emitError("expected one gpu.yield operand");
613  if (yield.getOperand(0).getType() != getType())

753  Value gridSizeX, Value gridSizeY, Value gridSizeZ,
754  Value getBlockSizeX, Value getBlockSizeY,
755  Value getBlockSizeZ, Value dynamicSharedMemorySize,
756  Type asyncTokenType, ValueRange asyncDependencies,
757  TypeRange workgroupAttributions,
758  TypeRange privateAttributions, Value clusterSizeX,
759  Value clusterSizeY, Value clusterSizeZ) {

761 
762  // Add a WorkGroup attribution attribute. This attribute is required to
763  // identify private attributions in the list of block argguments.
764  result.addAttribute(getNumWorkgroupAttributionsAttrName(),
765  builder.getI64IntegerAttr(workgroupAttributions.size()));
766 
767  // Add Op operands.
768  result.addOperands(asyncDependencies);
769  if (asyncTokenType)
770  result.types.push_back(builder.getType<AsyncTokenType>());

773  result.addOperands({gridSizeX, gridSizeY, gridSizeZ, getBlockSizeX,

785  // attributions, where the first kNumConfigRegionAttributes arguments have
786  // `index` type and the rest have the same types as the data operands.
787  Region *kernelRegion = result.addRegion();

790  for (unsigned i = 0; i < kNumConfigRegionAttributes; ++i)

792  // Add WorkGroup & Private attributions to the region arguments.

872 LogicalResult LaunchOp::verifyRegions() {
873  // Kernel launch takes kNumConfigOperands leading operands for grid/block
874  // sizes and transforms them into kNumConfigRegionAttributes region arguments
875  // for block/thread identifiers and grid/block sizes.
876  if (!getBody().empty()) {
877  if (getBody().getNumArguments() <
878  kNumConfigRegionAttributes + getNumWorkgroupAttributions())
879  return emitOpError("unexpected number of region arguments");
880  }

882  // Verify Attributions Address Spaces.
883  if (failed(verifyAttributions(getOperation(), getWorkgroupAttributions(),

885  failed(verifyAttributions(getOperation(), getPrivateAttributions(),
886  GPUDialect::getPrivateAddressSpace())))

902  .append("in '", LaunchOp::getOperationName(), "' body region");
903  }
904  }

906  if (getNumResults() == 0 && getAsyncToken())
907  return emitOpError("needs to be named when async keyword is specified");
908 
909  return success();

958 // Parse the size assignment blocks for blocks and threads. These have the form
959 // (%region_arg, %region_arg, %region_arg) in
960 // (%region_arg = %operand, %region_arg = %operand, %region_arg = %operand)
961 // where %region_arg are percent-identifiers for the region arguments to be
962 // introduced further (SSA defs), and %operand are percent-identifiers for the
963 // SSA value uses.
964 static ParseResult
965 parseSizeAssignment(OpAsmParser &parser,
966  MutableArrayRef<OpAsmParser::UnresolvedOperand> sizes,
967  MutableArrayRef<OpAsmParser::UnresolvedOperand> regionSizes,
968  MutableArrayRef<OpAsmParser::UnresolvedOperand> indices) {
969  assert(indices.size() == 3 && "space for three indices expected");
970  SmallVector<OpAsmParser::UnresolvedOperand, 3> args;
971  if (parser.parseOperandList(args, OpAsmParser::Delimiter::Paren,
972  /*allowResultNumber=*/false) ||
973  parser.parseKeyword("in") || parser.parseLParen())
974  return failure();
975  std::move(args.begin(), args.end(), indices.begin());
976 
977  for (int i = 0; i < 3; ++i) {
978  if (i != 0 && parser.parseComma())
979  return failure();
980  if (parser.parseOperand(regionSizes[i], /*allowResultNumber=*/false) ||
981  parser.parseEqual() || parser.parseOperand(sizes[i]))
982  return failure();
983  }
984 
985  return parser.parseRParen();
986 }

988 /// Parses a Launch operation.
989 /// operation ::= `gpu.launch` (`async` `[` ssa-id-list `]`)?
990 /// `clusters` `(` ssa-id-list `)` `in` ssa-reassignment (Optional)
991 /// `blocks` `(` ssa-id-list `)` `in` ssa-reassignment
992 /// `threads` `(` ssa-id-list `)` `in` ssa-reassignment
993 /// memory-attribution
994 /// region attr-dict?
995 /// ssa-reassignment ::= `(` ssa-id `=` ssa-use (`,` ssa-id `=` ssa-use)* `)`
996 ParseResult LaunchOp::parse(OpAsmParser &parser, OperationState &result) {
997  // Sizes of the grid and block.
998  SmallVector<OpAsmParser::UnresolvedOperand, LaunchOp::kNumConfigOperands>
999  sizes(LaunchOp::kNumConfigOperands);
1000 
1001  // Region arguments to be created.
1002  SmallVector<OpAsmParser::UnresolvedOperand, 16> regionArgs(
1003  LaunchOp::kNumConfigRegionAttributes);
1004 
1005  // Parse optional async dependencies.
1006  SmallVector<OpAsmParser::UnresolvedOperand, 4> asyncDependencies;
1007  Type asyncTokenType;
1008  if (failed(
1009  parseAsyncDependencies(parser, asyncTokenType, asyncDependencies)) ||
1010  parser.resolveOperands(asyncDependencies, asyncTokenType,
1011  result.operands))
1012  return failure();
1013  if (parser.getNumResults() > 0)
1014  result.types.push_back(asyncTokenType);
1015 
1016  bool hasCluster = false;
1017  if (succeeded(
1018  parser.parseOptionalKeyword(LaunchOp::getClustersKeyword().data()))) {
1019  hasCluster = true;
1020  sizes.resize(9);
1021  regionArgs.resize(18);
1022  }
1023  MutableArrayRef<OpAsmParser::UnresolvedOperand> sizesRef(sizes);
1024  MutableArrayRef<OpAsmParser::UnresolvedOperand> regionArgsRef(regionArgs);
1025 
1026  // Last three segment assigns the cluster size. In the region argument
1027  // list, this is last 6 arguments.
1028  if (hasCluster) {
1029  if (parseSizeAssignment(parser, sizesRef.drop_front(6),
1030  regionArgsRef.slice(15, 3),
1031  regionArgsRef.slice(12, 3)))
1032  return failure();
1033  }
1034  // Parse the size assignment segments: the first segment assigns grid sizes
1035  // and defines values for block identifiers; the second segment assigns block
1036  // sizes and defines values for thread identifiers. In the region argument
1037  // list, identifiers precede sizes, and block-related values precede
1038  // thread-related values.
1039  if (parser.parseKeyword(LaunchOp::getBlocksKeyword().data()) ||
1040  parseSizeAssignment(parser, sizesRef.take_front(3),
1041  regionArgsRef.slice(6, 3),
1042  regionArgsRef.slice(0, 3)) ||
1043  parser.parseKeyword(LaunchOp::getThreadsKeyword().data()) ||
1044  parseSizeAssignment(parser, sizesRef.drop_front(3),
1045  regionArgsRef.slice(9, 3),
1046  regionArgsRef.slice(3, 3)) ||
1047  parser.resolveOperands(sizes, parser.getBuilder().getIndexType(),
1048  result.operands))
1049  return failure();
1050 
1051  OpAsmParser::UnresolvedOperand dynamicSharedMemorySize;
1052  bool hasDynamicSharedMemorySize = false;
1053  if (!parser.parseOptionalKeyword(
1054  LaunchOp::getDynamicSharedMemorySizeKeyword())) {
1055  hasDynamicSharedMemorySize = true;
1056  if (parser.parseOperand(dynamicSharedMemorySize) ||
1057  parser.resolveOperand(dynamicSharedMemorySize,
1058  parser.getBuilder().getI32Type(),
1059  result.operands))
1060  return failure();
1061  }
1062 
1063  // Create the region arguments, it has kNumConfigRegionAttributes arguments
1064  // that correspond to block/thread identifiers and grid/block sizes, all
1065  // having `index` type, a variadic number of WorkGroup Attributions and
1066  // a variadic number of Private Attributions. The number of WorkGroup
1067  // Attributions is stored in the attr with name:
1068  // LaunchOp::getNumWorkgroupAttributionsAttrName().
1069  Type index = parser.getBuilder().getIndexType();
1070  SmallVector<Type, LaunchOp::kNumConfigRegionAttributes> dataTypes(
1071  LaunchOp::kNumConfigRegionAttributes + 6, index);
1072 
1073  SmallVector<OpAsmParser::Argument> regionArguments;
1074  for (auto ssaValueAndType : llvm::zip(regionArgs, dataTypes)) {
1075  OpAsmParser::Argument arg;
1076  arg.ssaName = std::get<0>(ssaValueAndType);
1077  arg.type = std::get<1>(ssaValueAndType);
1078  regionArguments.push_back(arg);
1079  }
1080 
1081  Builder &builder = parser.getBuilder();
1082  // Parse workgroup memory attributions.
1083  if (failed(parseAttributions(parser, LaunchOp::getWorkgroupKeyword(),
1084  regionArguments)))
1085  return failure();
1086 
1087  // Store the number of operands we just parsed as the number of workgroup
1088  // memory attributions.
1089  unsigned numWorkgroupAttrs = regionArguments.size() -
1090  LaunchOp::kNumConfigRegionAttributes -
1091  (hasCluster ? 6 : 0);
1092  result.addAttribute(LaunchOp::getNumWorkgroupAttributionsAttrName(),
1093  builder.getI64IntegerAttr(numWorkgroupAttrs));
1094 
1095  // Parse private memory attributions.
1096  if (failed(parseAttributions(parser, LaunchOp::getPrivateKeyword(),
1097  regionArguments)))
1098  return failure();
1099 
1100  // Introduce the body region and parse it. The region has
1101  // kNumConfigRegionAttributes arguments that correspond to
1102  // block/thread identifiers and grid/block sizes, all having `index` type.
1103  Region *body = result.addRegion();
1104  if (parser.parseRegion(*body, regionArguments) ||
1105  parser.parseOptionalAttrDict(result.attributes))
1106  return failure();
1107 
1108  SmallVector<int32_t, 11> segmentSizes(11, 1);
1109  segmentSizes.front() = asyncDependencies.size();
1110 
1111  if (!hasCluster) {
1112  segmentSizes[7] = 0;
1113  segmentSizes[8] = 0;
1114  segmentSizes[9] = 0;
1115  }
1116  segmentSizes.back() = hasDynamicSharedMemorySize ? 1 : 0;
1117  result.addAttribute(LaunchOp::getOperandSegmentSizeAttr(),
1118  parser.getBuilder().getDenseI32ArrayAttr(segmentSizes));
1119  return success();
1120 }

1122 /// Simplify the gpu.launch when the range of a thread or block ID is
1123 /// trivially known to be one.
1124 struct FoldLaunchArguments : public OpRewritePattern<LaunchOp> {
1125  using OpRewritePattern<LaunchOp>::OpRewritePattern;
1126  LogicalResult matchAndRewrite(LaunchOp op,
1127  PatternRewriter &rewriter) const override {
1128  // If the range implies a single value for `id`, replace `id`'s uses by
1129  // zero.
1130  Value zero;
1131  bool simplified = false;
1132  auto constPropIdUses = [&](Value id, Value size) {
1133  // Check if size is trivially one.
1134  if (!matchPattern(size, m_One()))
1135  return;
1136  if (id.getUses().empty())
1137  return;
1138  if (!simplified) {
1139  // Create a zero value the first time.
1140  OpBuilder::InsertionGuard guard(rewriter);
1141  rewriter.setInsertionPointToStart(&op.getBody().front());
1142  zero =
1143  arith::ConstantIndexOp::create(rewriter, op.getLoc(), /*value=*/0);
1144  }
1145  rewriter.replaceAllUsesWith(id, zero);
1146  simplified = true;
1147  };
1148  constPropIdUses(op.getBlockIds().x, op.getGridSizeX());
1149  constPropIdUses(op.getBlockIds().y, op.getGridSizeY());
1150  constPropIdUses(op.getBlockIds().z, op.getGridSizeZ());
1151  constPropIdUses(op.getThreadIds().x, op.getBlockSizeX());
1152  constPropIdUses(op.getThreadIds().y, op.getBlockSizeY());
1153  constPropIdUses(op.getThreadIds().z, op.getBlockSizeZ());
1154 
1155  return success(simplified);
1156  }
1157 };

1481 /// Parses a GPU function memory attribution.
1482 ///
1483 /// memory-attribution ::= (`workgroup` `(` ssa-id-and-type-list `)`)?
1484 /// (`private` `(` ssa-id-and-type-list `)`)?
1485 ///
1486 /// Note that this function parses only one of the two similar parts, with the
1487 /// keyword provided as argument.
1488 static ParseResult
1489 parseAttributions(OpAsmParser &parser, StringRef keyword,
1490  SmallVectorImpl<OpAsmParser::Argument> &args,
1491  Attribute &attributionAttrs) {
1492  // If we could not parse the keyword, just assume empty list and succeed.
1493  if (failed(parser.parseOptionalKeyword(keyword)))
1494  return success();
1495 
1496  size_t existingArgs = args.size();
1497  ParseResult result =
1498  parser.parseArgumentList(args, OpAsmParser::Delimiter::Paren,
1499  /*allowType=*/true, /*allowAttrs=*/true);
1500  if (failed(result))
1501  return result;
1502 
1503  bool hadAttrs = llvm::any_of(ArrayRef(args).drop_front(existingArgs),
1504  [](const OpAsmParser::Argument &arg) -> bool {
1505  return arg.attrs && !arg.attrs.empty();
1506  });
1507  if (!hadAttrs) {
1508  attributionAttrs = nullptr;
1509  return result;
1510  }
1511 
1512  Builder &builder = parser.getBuilder();
1513  SmallVector<Attribute> attributionAttrsVec;
1514  for (const auto &argument : ArrayRef(args).drop_front(existingArgs)) {
1515  if (!argument.attrs)
1516  attributionAttrsVec.push_back(builder.getDictionaryAttr({}));
1517  else
1518  attributionAttrsVec.push_back(argument.attrs);
1519  }
1520  attributionAttrs = builder.getArrayAttr(attributionAttrsVec);
1521  return result;
1522 }

1524 /// Parses a GPU function.
1525 ///
1526 /// <operation> ::= `gpu.func` symbol-ref-id `(` argument-list `)`
1527 /// (`->` function-result-list)? memory-attribution `kernel`?
1528 /// function-attributes? region
1529 ParseResult GPUFuncOp::parse(OpAsmParser &parser, OperationState &result) {
1530  SmallVector<OpAsmParser::Argument> entryArgs;
1531  SmallVector<DictionaryAttr> resultAttrs;
1532  SmallVector<Type> resultTypes;
1533  bool isVariadic;
1534 
1535  // Parse the function name.
1536  StringAttr nameAttr;
1537  if (parser.parseSymbolName(nameAttr, ::mlir::SymbolTable::getSymbolAttrName(),
1538  result.attributes))
1539  return failure();
1540 
1541  auto signatureLocation = parser.getCurrentLocation();
1542  if (failed(function_interface_impl::parseFunctionSignatureWithArguments(
1543  parser, /*allowVariadic=*/false, entryArgs, isVariadic, resultTypes,
1544  resultAttrs)))
1545  return failure();
1546 
1547  if (!entryArgs.empty() && entryArgs[0].ssaName.name.empty())
1548  return parser.emitError(signatureLocation)
1549  << "gpu.func requires named arguments";
1550 
1551  // Construct the function type. More types will be added to the region, but
1552  // not to the function type.
1553  Builder &builder = parser.getBuilder();
1554 
1555  SmallVector<Type> argTypes;
1556  for (auto &arg : entryArgs)
1557  argTypes.push_back(arg.type);
1558  auto type = builder.getFunctionType(argTypes, resultTypes);
1559  result.addAttribute(getFunctionTypeAttrName(result.name),
1560  TypeAttr::get(type));
1561 
1562  call_interface_impl::addArgAndResultAttrs(
1563  builder, result, entryArgs, resultAttrs, getArgAttrsAttrName(result.name),
1564  getResAttrsAttrName(result.name));
1565 
1566  Attribute workgroupAttributionAttrs;
1567  // Parse workgroup memory attributions.
1568  if (failed(parseAttributions(parser, GPUFuncOp::getWorkgroupKeyword(),
1569  entryArgs, workgroupAttributionAttrs)))
1570  return failure();
1571 
1572  // Store the number of operands we just parsed as the number of workgroup
1573  // memory attributions.
1574  unsigned numWorkgroupAttrs = entryArgs.size() - type.getNumInputs();
1575  result.addAttribute(GPUFuncOp::getNumWorkgroupAttributionsAttrName(),
1576  builder.getI64IntegerAttr(numWorkgroupAttrs));
1577  if (workgroupAttributionAttrs)
1578  result.addAttribute(GPUFuncOp::getWorkgroupAttribAttrsAttrName(result.name),
1579  workgroupAttributionAttrs);
1580 
1581  Attribute privateAttributionAttrs;
1582  // Parse private memory attributions.
1583  if (failed(parseAttributions(parser, GPUFuncOp::getPrivateKeyword(),
1584  entryArgs, privateAttributionAttrs)))
1585  return failure();
1586  if (privateAttributionAttrs)
1587  result.addAttribute(GPUFuncOp::getPrivateAttribAttrsAttrName(result.name),
1588  privateAttributionAttrs);
1589 
1590  // Parse the kernel attribute if present.
1591  if (succeeded(parser.parseOptionalKeyword(GPUFuncOp::getKernelKeyword())))
1592  result.addAttribute(GPUDialect::getKernelFuncAttrName(),
1593  builder.getUnitAttr());
1594 
1595  // Parse attributes.
1596  if (failed(parser.parseOptionalAttrDictWithKeyword(result.attributes)))
1597  return failure();
1598 
1599  // Parse the region. If no argument names were provided, take all names
1600  // (including those of attributions) from the entry block.
1601  auto *body = result.addRegion();
1602  return parser.parseRegion(*body, entryArgs);
1603 }

1745  return;
1746  elems.emplace_back(name, value);
1747  }
1748  if (mustSort) {
