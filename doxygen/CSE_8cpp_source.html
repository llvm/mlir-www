35 struct SimpleOperationInfo : public llvm::DenseMapInfo<Operation *> {
36  static unsigned getHashValue(const Operation *opC) {
37  return OperationEquivalence::computeHash(
38  const_cast<Operation *>(opC),
39  /*hashOperands=*/OperationEquivalence::directHashValue,
40  /*hashResults=*/OperationEquivalence::ignoreHashValue,
41  OperationEquivalence::IgnoreLocations);
42  }
43  static bool isEqual(const Operation *lhsC, const Operation *rhsC) {
44  auto *lhs = const_cast<Operation *>(lhsC);
45  auto *rhs = const_cast<Operation *>(rhsC);
46  if (lhs == rhs)
47  return true;
48  if (lhs == getTombstoneKey() || lhs == getEmptyKey() ||
49  rhs == getTombstoneKey() || rhs == getEmptyKey())
50  return false;
51  return OperationEquivalence::isEquivalentTo(
52  const_cast<Operation *>(lhsC), const_cast<Operation *>(rhsC),
53  OperationEquivalence::IgnoreLocations);
54  }
55 };

130 void CSEDriver::replaceUsesAndDelete(ScopedMapTy &knownValues, Operation *op,
131  Operation *existing,
132  bool hasSSADominance) {
133  // If we find one then replace all uses of the current operation with the
134  // existing one and mark it for deletion. We can only replace an operand in
135  // an operation if it has not been visited yet.
136  if (hasSSADominance) {
137  // If the region has SSA dominance, then we are guaranteed to have not
138  // visited any use of the current operation.
139  if (auto *rewriteListener =
140  dyn_cast_if_present<RewriterBase::Listener>(rewriter.getListener()))
141  rewriteListener->notifyOperationReplaced(op, existing);
142  // Replace all uses, but do not remove the operation yet. This does not
143  // notify the listener because the original op is not erased.
144  rewriter.replaceAllUsesWith(op->getResults(), existing->getResults());
145  opsToErase.push_back(op);
146  } else {
147  // When the region does not have SSA dominance, we need to check if we
148  // have visited a use before replacing any use.
149  auto wasVisited = [&](OpOperand &operand) {
150  return !knownValues.count(operand.getOwner());
151  };
152  if (auto *rewriteListener =
153  dyn_cast_if_present<RewriterBase::Listener>(rewriter.getListener()))
154  for (Value v : op->getResults())
155  if (all_of(v.getUses(), wasVisited))
156  rewriteListener->notifyOperationReplaced(op, existing);
157 
158  // Replace all uses, but do not remove the operation yet. This does not
159  // notify the listener because the original op is not erased.
160  rewriter.replaceUsesWithIf(op->getResults(), existing->getResults(),
161  wasVisited);
162 
163  // There may be some remaining uses of the operation.
164  if (op->use_empty())
165  opsToErase.push_back(op);
166  }
167 
168  // If the existing operation has an unknown location and the current
169  // operation doesn't, then set the existing op's location to that of the
170  // current op.
171  if (isa<UnknownLoc>(existing->getLoc()) && !isa<UnknownLoc>(op->getLoc()))
172  existing->setLoc(op->getLoc());
173 
174  ++numCSE;
175 }

177 bool CSEDriver::hasOtherSideEffectingOpInBetween(Operation *fromOp,
178  Operation *toOp) {
179  assert(fromOp->getBlock() == toOp->getBlock());
180  assert(
181  isa<MemoryEffectOpInterface>(fromOp) &&
182  cast<MemoryEffectOpInterface>(fromOp).hasEffect<MemoryEffects::Read>() &&
183  isa<MemoryEffectOpInterface>(toOp) &&
184  cast<MemoryEffectOpInterface>(toOp).hasEffect<MemoryEffects::Read>());
185  Operation *nextOp = fromOp->getNextNode();
186  auto result =
187  memEffectsCache.try_emplace(fromOp, std::make_pair(fromOp, nullptr));
188  if (result.second) {
189  auto memEffectsCachePair = result.first->second;
190  if (memEffectsCachePair.second == nullptr) {
191  // No MemoryEffects::Write has been detected until the cached operation.
192  // Continue looking from the cached operation to toOp.
193  nextOp = memEffectsCachePair.first;
194  } else {
195  // MemoryEffects::Write has been detected before so there is no need to
196  // check further.
197  return true;
198  }
199  }
200  while (nextOp && nextOp != toOp) {
201  std::optional<SmallVector<MemoryEffects::EffectInstance>> effects =
202  getEffectsRecursively(nextOp);
203  if (!effects) {
204  // TODO: Do we need to handle other effects generically?
205  // If the operation does not implement the MemoryEffectOpInterface we
206  // conservatively assume it writes.
207  result.first->second =
208  std::make_pair(nextOp, MemoryEffects::Write::get());
209  return true;
210  }
211 
212  for (const MemoryEffects::EffectInstance &effect : *effects) {
213  if (isa<MemoryEffects::Write>(effect.getEffect())) {
214  result.first->second = {nextOp, MemoryEffects::Write::get()};
215  return true;
216  }
217  }
218  nextOp = nextOp->getNextNode();
219  }
220  result.first->second = std::make_pair(toOp, nullptr);
221  return false;
222 }

224 /// Attempt to eliminate a redundant operation.
225 LogicalResult CSEDriver::simplifyOperation(ScopedMapTy &knownValues,
226  Operation *op,
227  bool hasSSADominance) {
228  // Don't simplify terminator operations.
229  if (op->hasTrait<OpTrait::IsTerminator>())
230  return failure();
231 
232  // If the operation is already trivially dead just add it to the erase list.
233  if (isOpTriviallyDead(op)) {
234  opsToErase.push_back(op);
235  ++numDCE;
236  return success();
237  }
238 
239  // Don't simplify operations with regions that have multiple blocks.
240  // TODO: We need additional tests to verify that we handle such IR correctly.
241  if (!llvm::all_of(op->getRegions(),
242  [](Region &r) { return r.empty() || r.hasOneBlock(); }))
243  return failure();
244 
245  // Some simple use case of operation with memory side-effect are dealt with
246  // here. Operations with no side-effect are done after.
247  if (!isMemoryEffectFree(op)) {
248  auto memEffects = dyn_cast<MemoryEffectOpInterface>(op);
249  // TODO: Only basic use case for operations with MemoryEffects::Read can be
250  // eleminated now. More work needs to be done for more complicated patterns
251  // and other side-effects.
252  if (!memEffects || !memEffects.onlyHasEffect<MemoryEffects::Read>())
253  return failure();
254 
255  // Look for an existing definition for the operation.
256  if (auto *existing = knownValues.lookup(op)) {
257  if (existing->getBlock() == op->getBlock() &&
258  !hasOtherSideEffectingOpInBetween(existing, op)) {
259  // The operation that can be deleted has been reach with no
260  // side-effecting operations in between the existing operation and
261  // this one so we can remove the duplicate.
262  replaceUsesAndDelete(knownValues, op, existing, hasSSADominance);
263  return success();
264  }
265  }
266  knownValues.insert(op, op);
267  return failure();
268  }
269 
270  // Look for an existing definition for the operation.
271  if (auto *existing = knownValues.lookup(op)) {
272  replaceUsesAndDelete(knownValues, op, existing, hasSSADominance);
273  ++numCSE;
274  return success();
275  }
276 
277  // Otherwise, we add this operation to the known values map.
278  knownValues.insert(op, op);
279  return failure();
280 }

282 void CSEDriver::simplifyBlock(ScopedMapTy &knownValues, Block *bb,
283  bool hasSSADominance) {
284  for (auto &op : *bb) {
285  // Most operations don't have regions, so fast path that case.
286  if (op.getNumRegions() != 0) {
287  // If this operation is isolated above, we can't process nested regions
288  // with the given 'knownValues' map. This would cause the insertion of
289  // implicit captures in explicit capture only regions.
290  if (op.mightHaveTrait<OpTrait::IsIsolatedFromAbove>()) {
291  ScopedMapTy nestedKnownValues;
292  for (auto &region : op.getRegions())
293  simplifyRegion(nestedKnownValues, region);
294  } else {
295  // Otherwise, process nested regions normally.
296  for (auto &region : op.getRegions())
297  simplifyRegion(knownValues, region);
298  }
299  }
300 
301  // If the operation is simplified, we don't process any held regions.
302  if (succeeded(simplifyOperation(knownValues, &op, hasSSADominance)))
303  continue;
304  }
305  // Clear the MemoryEffects cache since its usage is by block only.
306  memEffectsCache.clear();
307 }

309 void CSEDriver::simplifyRegion(ScopedMapTy &knownValues, Region &region) {
310  // If the region is empty there is nothing to do.
311  if (region.empty())
312  return;
313 
314  bool hasSSADominance = domInfo->hasSSADominance(&region);
315 
316  // If the region only contains one block, then simplify it directly.
317  if (region.hasOneBlock()) {
318  ScopedMapTy::ScopeTy scope(knownValues);
319  simplifyBlock(knownValues, &region.front(), hasSSADominance);
320  return;
321  }
322 
323  // If the region does not have dominanceInfo, then skip it.
324  // TODO: Regions without SSA dominance should define a different
325  // traversal order which is appropriate and can be used here.
326  if (!hasSSADominance)
327  return;
328 
329  // Note, deque is being used here because there was significant performance
330  // gains over vector when the container becomes very large due to the
331  // specific access patterns. If/when these performance issues are no
332  // longer a problem we can change this to vector. For more information see
333  // the llvm mailing list discussion on this:
334  // http://lists.llvm.org/pipermail/llvm-commits/Week-of-Mon-20120116/135228.html
335  std::deque<std::unique_ptr<CFGStackNode>> stack;
336 
337  // Process the nodes of the dom tree for this region.
338  stack.emplace_back(std::make_unique<CFGStackNode>(
339  knownValues, domInfo->getRootNode(&region)));
340 
341  while (!stack.empty()) {
342  auto &currentNode = stack.back();
343 
344  // Check to see if we need to process this node.
345  if (!currentNode->processed) {
346  currentNode->processed = true;
347  simplifyBlock(knownValues, currentNode->node->getBlock(),
348  hasSSADominance);
349  }
350 
351  // Otherwise, check to see if we need to process a child node.
352  if (currentNode->childIterator != currentNode->node->end()) {
353  auto *childNode = *(currentNode->childIterator++);
354  stack.emplace_back(
355  std::make_unique<CFGStackNode>(knownValues, childNode));
356  } else {
357  // Finally, if the node and all of its children have been processed
358  // then we delete the node.
359  stack.pop_back();
360  }
361  }
362 }

364 void CSEDriver::simplify(Operation *op, bool *changed) {
365  /// Simplify all regions.
366  ScopedMapTy knownValues;
367  for (auto &region : op->getRegions())
368  simplifyRegion(knownValues, region);
369 
370  /// Erase any operations that were marked as dead during simplification.
371  for (auto *op : opsToErase)
372  rewriter.eraseOp(op);
373  if (changed)
374  *changed = !opsToErase.empty();
375 
376  // Note: CSE does currently not remove ops with regions, so DominanceInfo
377  // does not have to be invalidated.
378 }

394 void CSE::runOnOperation() {
395  // Simplify the IR.
396  IRRewriter rewriter(&getContext());
397  CSEDriver driver(rewriter, &getAnalysis<DominanceInfo>());
398  bool changed = false;
399  driver.simplify(getOperation(), &changed);
400 
401  // Set statistics.
402  numCSE = driver.getNumCSE();
403  numDCE = driver.getNumDCE();
404 
405  // If there was no change to the IR, we mark all analyses as preserved.
406  if (!changed)
407  return markAllAnalysesPreserved();
408 
409  // We currently don't remove region operations, so mark dominance as
410  // preserved.
411  markAnalysesPreserved<DominanceInfo, PostDominanceInfo>();
412 }
