117 /// Parse an enum from the keyword, or default to the provided default value.
118 /// The return type is the enum type by default, unless overridden with the
119 /// second template argument.
120 template <typename EnumTy, typename RetTy = EnumTy>
121 static RetTy parseOptionalLLVMKeyword(OpAsmParser &parser,
122  EnumTy defaultValue) {
123  SmallVector<StringRef, 10> names;
124  for (unsigned i = 0, e = EnumTraits<EnumTy>::getMaxEnumVal(); i <= e; ++i)
125  names.push_back(EnumTraits<EnumTy>::stringify(static_cast<EnumTy>(i)));

128  if (index == -1)
129  return static_cast<RetTy>(defaultValue);
130  return static_cast<RetTy>(index);

181 static ParseResult parseOneOpBundle(
182  OpAsmParser &p,
183  SmallVector<SmallVector<OpAsmParser::UnresolvedOperand>> &opBundleOperands,
184  SmallVector<SmallVector<Type>> &opBundleOperandTypes,
185  SmallVector<Attribute> &opBundleTags) {
186  SMLoc currentParserLoc = p.getCurrentLocation();
187  SmallVector<OpAsmParser::UnresolvedOperand> operands;
188  SmallVector<Type> types;
189  std::string tag;
190 
191  if (p.parseString(&tag))
192  return p.emitError(currentParserLoc, "expect operand bundle tag");
193 
194  if (p.parseLParen())
195  return failure();
196 
197  if (p.parseOptionalRParen()) {
198  if (p.parseOperandList(operands) || p.parseColon() ||
199  p.parseTypeList(types) || p.parseRParen())
200  return failure();
201  }
202 
203  opBundleOperands.push_back(std::move(operands));
204  opBundleOperandTypes.push_back(std::move(types));
205  opBundleTags.push_back(StringAttr::get(p.getContext(), tag));
206 
207  return success();
208 }

255 // <operation> ::= `llvm.icmp` string-literal ssa-use `,` ssa-use
256 // attribute-dict? `:` type
257 // <operation> ::= `llvm.fcmp` string-literal ssa-use `,` ssa-use
258 // attribute-dict? `:` type
259 template <typename CmpPredicateType>
260 static ParseResult parseCmpOp(OpAsmParser &parser, OperationState &result) {
261  StringAttr predicateAttr;
262  OpAsmParser::UnresolvedOperand lhs, rhs;
263  Type type;
264  SMLoc predicateLoc, trailingTypeLoc;
265  if (parser.getCurrentLocation(&predicateLoc) ||
266  parser.parseAttribute(predicateAttr, "predicate", result.attributes) ||
267  parser.parseOperand(lhs) || parser.parseComma() ||
268  parser.parseOperand(rhs) ||
269  parser.parseOptionalAttrDict(result.attributes) || parser.parseColon() ||
270  parser.getCurrentLocation(&trailingTypeLoc) || parser.parseType(type) ||
271  parser.resolveOperand(lhs, type, result.operands) ||
272  parser.resolveOperand(rhs, type, result.operands))
273  return failure();
274 
275  // Replace the string attribute `predicate` with an integer attribute.
276  int64_t predicateValue = 0;
277  if (std::is_same<CmpPredicateType, ICmpPredicate>()) {
278  std::optional<ICmpPredicate> predicate =
279  symbolizeICmpPredicate(predicateAttr.getValue());
280  if (!predicate)
281  return parser.emitError(predicateLoc)
282  << "'" << predicateAttr.getValue()
283  << "' is an incorrect value of the 'predicate' attribute";
284  predicateValue = static_cast<int64_t>(*predicate);
285  } else {
286  std::optional<FCmpPredicate> predicate =
287  symbolizeFCmpPredicate(predicateAttr.getValue());
288  if (!predicate)
289  return parser.emitError(predicateLoc)
290  << "'" << predicateAttr.getValue()
291  << "' is an incorrect value of the 'predicate' attribute";
292  predicateValue = static_cast<int64_t>(*predicate);
293  }
294 
295  result.attributes.set("predicate",
296  parser.getBuilder().getI64IntegerAttr(predicateValue));
297 
298  // The result type is either i1 or a vector type <? x i1> if the inputs are
299  // vectors.
300  if (!isCompatibleType(type))
301  return parser.emitError(trailingTypeLoc,
302  "expected LLVM dialect-compatible type");
303  result.addTypes(getI1SameShape(type));
304  return success();
305 }

324 OpFoldResult ICmpOp::fold(FoldAdaptor adaptor) {
325  if (getPredicate() != ICmpPredicate::eq &&
326  getPredicate() != ICmpPredicate::ne)
327  return {};
328 
329  // cmpi(eq/ne, x, x) -> true/false
330  if (getLhs() == getRhs())
331  return getBoolAttribute(getType(), getContext(),
332  getPredicate() == ICmpPredicate::eq);
333 
334  // cmpi(eq/ne, alloca, null) -> false/true
335  if (getLhs().getDefiningOp<AllocaOp>() && getRhs().getDefiningOp<ZeroOp>())
336  return getBoolAttribute(getType(), getContext(),
337  getPredicate() == ICmpPredicate::ne);
338 
339  // cmpi(eq/ne, null, alloca) -> cmpi(eq/ne, alloca, null)
340  if (getLhs().getDefiningOp<ZeroOp>() && getRhs().getDefiningOp<AllocaOp>()) {
341  Value lhs = getLhs();
342  Value rhs = getRhs();
343  getLhsMutable().assign(rhs);
344  getRhsMutable().assign(lhs);
345  return getResult();
346  }
347 
348  return {};
349 }

374 // attribute-dict? `:` type `,` type
375 ParseResult AllocaOp::parse(OpAsmParser &parser, OperationState &result) {
376  OpAsmParser::UnresolvedOperand arraySize;
377  Type type, elemType;
378  SMLoc trailingTypeLoc;
379 
380  if (succeeded(parser.parseOptionalKeyword("inalloca")))
381  result.addAttribute(getInallocaAttrName(result.name),
382  UnitAttr::get(parser.getContext()));
383 
384  if (parser.parseOperand(arraySize) || parser.parseKeyword("x") ||
385  parser.parseType(elemType) ||
386  parser.parseOptionalAttrDict(result.attributes) || parser.parseColon() ||
387  parser.getCurrentLocation(&trailingTypeLoc) || parser.parseType(type))
388  return failure();
389 
390  std::optional<NamedAttribute> alignmentAttr =
391  result.attributes.getNamed("alignment");
392  if (alignmentAttr.has_value()) {
393  auto alignmentInt = llvm::dyn_cast<IntegerAttr>(alignmentAttr->getValue());
394  if (!alignmentInt)
395  return parser.emitError(parser.getNameLoc(),
396  "expected integer alignment");
397  if (alignmentInt.getValue().isZero())
398  result.attributes.erase("alignment");
399  }
400 
401  // Extract the result type from the trailing function type.
402  auto funcType = llvm::dyn_cast<FunctionType>(type);
403  if (!funcType || funcType.getNumInputs() != 1 ||
404  funcType.getNumResults() != 1)
405  return parser.emitError(
406  trailingTypeLoc,
407  "expected trailing function type with one argument and one result");
408 
409  if (parser.resolveOperand(arraySize, funcType.getInput(0), result.operands))
410  return failure();
411 
412  Type resultType = funcType.getResult(0);
413  if (auto ptrResultType = llvm::dyn_cast<LLVMPointerType>(resultType))
414  result.addAttribute(kElemTypeAttrName, TypeAttr::get(elemType));
415 
416  result.addTypes({funcType.getResult(0)});
417  return success();
418 }

513 /// <cases> ::= `[` (case (`,` case )* )? `]`
514 /// <case> ::= integer `:` bb-id (`(` ssa-use-and-type-list `)`)?
515 static ParseResult parseSwitchOpCases(
516  OpAsmParser &parser, Type flagType, DenseIntElementsAttr &caseValues,
517  SmallVectorImpl<Block *> &caseDestinations,
518  SmallVectorImpl<SmallVector<OpAsmParser::UnresolvedOperand>> &caseOperands,
519  SmallVectorImpl<SmallVector<Type>> &caseOperandTypes) {
520  if (failed(parser.parseLSquare()))
521  return failure();
522  if (succeeded(parser.parseOptionalRSquare()))
523  return success();
524  SmallVector<APInt> values;
525  unsigned bitWidth = flagType.getIntOrFloatBitWidth();
526  auto parseCase = [&]() {
527  int64_t value = 0;
528  if (failed(parser.parseInteger(value)))
529  return failure();
530  values.push_back(APInt(bitWidth, value, /*isSigned=*/true));
531 
532  Block *destination;
533  SmallVector<OpAsmParser::UnresolvedOperand> operands;
534  SmallVector<Type> operandTypes;
535  if (parser.parseColon() || parser.parseSuccessor(destination))
536  return failure();
537  if (!parser.parseOptionalLParen()) {
538  if (parser.parseOperandList(operands, OpAsmParser::Delimiter::None,
539  /*allowResultNumber=*/false) ||
540  parser.parseColonTypeList(operandTypes) || parser.parseRParen())
541  return failure();
542  }
543  caseDestinations.push_back(destination);
544  caseOperands.emplace_back(operands);
545  caseOperandTypes.emplace_back(operandTypes);
546  return success();
547  };
548  if (failed(parser.parseCommaSeparatedList(parseCase)))
549  return failure();
550 
551  ShapedType caseValueType =
552  VectorType::get(static_cast<int64_t>(values.size()), flagType);
553  caseValues = DenseIntElementsAttr::get(caseValueType, values);
554  return parser.parseRSquare();
555 }

623 /// Destructures the 'indices' parameter into 'rawConstantIndices' and
624 /// 'dynamicIndices', encoding the former in the process. In the process,
625 /// dynamic indices which are used to index into a structure type are converted
626 /// to constant indices when possible. To do this, the GEPs element type should
627 /// be passed as first parameter.
628 static void destructureIndices(Type currType, ArrayRef<GEPArg> indices,
629  SmallVectorImpl<int32_t> &rawConstantIndices,
630  SmallVectorImpl<Value> &dynamicIndices) {
631  for (const GEPArg &iter : indices) {
632  // If the thing we are currently indexing into is a struct we must turn
633  // any integer constants into constant indices. If this is not possible
634  // we don't do anything here. The verifier will catch it and emit a proper
635  // error. All other canonicalization is done in the fold method.
636  bool requiresConst = !rawConstantIndices.empty() &&
637  isa_and_nonnull<LLVMStructType>(currType);
638  if (Value val = llvm::dyn_cast_if_present<Value>(iter)) {
639  APInt intC;
640  if (requiresConst && matchPattern(val, m_ConstantInt(&intC)) &&
641  intC.isSignedIntN(kGEPConstantBitWidth)) {
642  rawConstantIndices.push_back(intC.getSExtValue());
643  } else {
644  rawConstantIndices.push_back(GEPOp::kDynamicIndex);
645  dynamicIndices.push_back(val);
646  }
647  } else {
648  rawConstantIndices.push_back(cast<GEPConstantIndex>(iter));
649  }
650 
651  // Skip for very first iteration of this loop. First index does not index
652  // within the aggregates, but is just a pointer offset.
653  if (rawConstantIndices.size() == 1 || !currType)
654  continue;
655 
656  currType = TypeSwitch<Type, Type>(currType)
657  .Case<VectorType, LLVMArrayType>([](auto containerType) {
658  return containerType.getElementType();
659  })
660  .Case([&](LLVMStructType structType) -> Type {
661  int64_t memberIndex = rawConstantIndices.back();
662  if (memberIndex >= 0 && static_cast<size_t>(memberIndex) <
663  structType.getBody().size())
664  return structType.getBody()[memberIndex];
665  return nullptr;
666  })
667  .Default(Type(nullptr));
668  }
669 }

739 /// For the given `indices`, check if they comply with `baseGEPType`,
740 /// especially check against LLVMStructTypes nested within.
741 static LogicalResult
742 verifyStructIndices(Type baseGEPType, unsigned indexPos,
743  GEPIndicesAdaptor<ValueRange> indices,
744  function_ref<InFlightDiagnostic()> emitOpError) {
745  if (indexPos >= indices.size())
746  // Stop searching
747  return success();
748 
749  return TypeSwitch<Type, LogicalResult>(baseGEPType)
750  .Case<LLVMStructType>([&](LLVMStructType structType) -> LogicalResult {
751  auto attr = dyn_cast<IntegerAttr>(indices[indexPos]);
752  if (!attr)
753  return emitOpError() << "expected index " << indexPos
754  << " indexing a struct to be constant";
755 
756  int32_t gepIndex = attr.getInt();
757  ArrayRef<Type> elementTypes = structType.getBody();
758  if (gepIndex < 0 ||
759  static_cast<size_t>(gepIndex) >= elementTypes.size())
760  return emitOpError() << "index " << indexPos
761  << " indexing a struct is out of bounds";
762 
763  // Instead of recursively going into every children types, we only
764  // dive into the one indexed by gepIndex.
765  return verifyStructIndices(elementTypes[gepIndex], indexPos + 1,
766  indices, emitOpError);
767  })
768  .Case<VectorType, LLVMArrayType>(
769  [&](auto containerType) -> LogicalResult {
770  return verifyStructIndices(containerType.getElementType(),
771  indexPos + 1, indices, emitOpError);
772  })
773  .Default([&](auto otherType) -> LogicalResult {
774  return emitOpError()
775  << "type " << otherType << " cannot be indexed (index #"
776  << indexPos << ")";
777  });
778 }

821 /// Returns true if the given type is supported by atomic operations. All
822 /// integer, float, and pointer types with a power-of-two bitsize and a minimal
823 /// size of 8 bits are supported.
824 static bool isTypeCompatibleWithAtomicOp(Type type,
825  const DataLayout &dataLayout) {
826  if (!isa<IntegerType, LLVMPointerType>(type))
827  if (!isCompatibleFloatingPointType(type))
828  return false;
829 
830  llvm::TypeSize bitWidth = dataLayout.getTypeSizeInBits(type);
831  if (bitWidth.isScalable())
832  return false;
833  // Needs to be at least 8 bits and a power of two.
834  return bitWidth >= 8 && (bitWidth & (bitWidth - 1)) == 0;
835 }

1201

1207  << funcType.getNumParams() << ")";

1211  << (getCalleeOperands().size() - isIndirect)

1216  if (getOperand(i + isIndirect).getType() != funcType.getParamType(i))
1217  return emitOpError() << "operand type mismatch for operand " << i << ": "

1220

1222  !llvm::isa<LLVM::LLVMVoidType>(funcType.getReturnType()))

1224

1226  llvm::isa<LLVM::LLVMVoidType>(funcType.getReturnType()))

1228  << "calling function with void result must not produce values";
1229 
1230  if (getNumResults() > 1)
1231  return emitOpError()

1235  return emitOpError() << "result type mismatch: " << getResult().getType()
1236  << " != " << funcType.getReturnType();
1237

1292 /// Parses the type of a call operation and resolves the operands if the parsing
1293 /// succeeds. Returns failure otherwise.
1294 static ParseResult parseCallTypeAndResolveOperands(
1295  OpAsmParser &parser, OperationState &result, bool isDirect,
1296  ArrayRef<OpAsmParser::UnresolvedOperand> operands,
1297  SmallVectorImpl<DictionaryAttr> &argAttrs,
1298  SmallVectorImpl<DictionaryAttr> &resultAttrs) {
1299  SMLoc trailingTypesLoc = parser.getCurrentLocation();
1300  SmallVector<Type> types;
1301  if (parser.parseColon())
1302  return failure();
1303  if (!isDirect) {
1304  types.emplace_back();
1305  if (parser.parseType(types.back()))
1306  return failure();
1307  if (parser.parseOptionalComma())
1308  return parser.emitError(
1309  trailingTypesLoc, "expected indirect call to have 2 trailing types");
1310  }
1311  SmallVector<Type> argTypes;
1312  SmallVector<Type> resTypes;
1313  if (call_interface_impl::parseFunctionSignature(parser, argTypes, argAttrs,
1314  resTypes, resultAttrs)) {
1315  if (isDirect)
1316  return parser.emitError(trailingTypesLoc,
1317  "expected direct call to have 1 trailing types");
1318  return parser.emitError(trailingTypesLoc,
1319  "expected trailing function type");
1320  }
1321 
1322  if (resTypes.size() > 1)
1323  return parser.emitError(trailingTypesLoc,
1324  "expected function with 0 or 1 result");
1325  if (resTypes.size() == 1 && llvm::isa<LLVM::LLVMVoidType>(resTypes[0]))
1326  return parser.emitError(trailingTypesLoc,
1327  "expected a non-void result type");
1328 
1329  // The head element of the types list matches the callee type for
1330  // indirect calls, while the types list is emtpy for direct calls.
1331  // Append the function input types to resolve the call operation
1332  // operands.
1333  llvm::append_range(types, argTypes);
1334  if (parser.resolveOperands(operands, types, parser.getNameLoc(),
1335  result.operands))
1336  return failure();
1337  if (resTypes.size() != 0)
1338  result.addTypes(resTypes);
1339 
1340  return success();
1341 }

1445  result.addAttribute(CallOp::getOpBundleTagsAttrName(result.name).getValue(),
1446  opBundleTags);
1447 
1448  if (parser.parseOptionalAttrDict(result.attributes))
1449  return failure();

1452  SmallVector<DictionaryAttr> argAttrs;
1453  SmallVector<DictionaryAttr> resultAttrs;
1454  if (parseCallTypeAndResolveOperands(parser, result, isDirect, operands,
1455  argAttrs, resultAttrs))

1458  parser.getBuilder(), result, argAttrs, resultAttrs,
1459  getArgAttrsAttrName(result.name), getResAttrsAttrName(result.name));
1460  if (resolveOpBundleOperands(parser, opBundlesLoc, result, opBundleOperands,
1461  opBundleOperandTypes,
1462  getOpBundleSizesAttrName(result.name)))
1463  return failure();

1467  numOpBundleOperands += operands.size();

1470  CallOp::getOperandSegmentSizeAttr(),
1471  parser.getBuilder().getDenseI32ArrayAttr(
1472  {static_cast<int32_t>(operands.size()), numOpBundleOperands}));
1473  return success();
1474 }

1817 /// Extract the type at `position` in the LLVM IR aggregate type
1818 /// `containerType`. Each element of `position` is an index into a nested
1819 /// aggregate type. Return the resulting type or emit an error.
1820 static Type getInsertExtractValueElementType(
1821  function_ref<InFlightDiagnostic(StringRef)> emitError, Type containerType,
1822  ArrayRef<int64_t> position) {
1823  Type llvmType = containerType;
1824  if (!isCompatibleType(containerType)) {
1825  emitError("expected LLVM IR Dialect type, got ") << containerType;
1826  return {};
1827  }
1828 
1829  // Infer the element type from the structure type: iteratively step inside the
1830  // type by taking the element type, indexed by the position attribute for
1831  // structures. Check the position index before accessing, it is supposed to
1832  // be in bounds.
1833  for (int64_t idx : position) {
1834  if (auto arrayType = llvm::dyn_cast<LLVMArrayType>(llvmType)) {
1835  if (idx < 0 || static_cast<unsigned>(idx) >= arrayType.getNumElements()) {
1836  emitError("position out of bounds: ") << idx;
1837  return {};
1838  }
1839  llvmType = arrayType.getElementType();
1840  } else if (auto structType = llvm::dyn_cast<LLVMStructType>(llvmType)) {
1841  if (idx < 0 ||
1842  static_cast<unsigned>(idx) >= structType.getBody().size()) {
1843  emitError("position out of bounds: ") << idx;
1844  return {};
1845  }
1846  llvmType = structType.getBody()[idx];
1847  } else {
1848  emitError("expected LLVM IR structure/array type, got: ") << llvmType;
1849  return {};
1850  }
1851  }
1852  return llvmType;
1853 }

1868 OpFoldResult LLVM::ExtractValueOp::fold(FoldAdaptor adaptor) {
1869  if (auto extractValueOp = getContainer().getDefiningOp<ExtractValueOp>()) {
1870  SmallVector<int64_t, 4> newPos(extractValueOp.getPosition());
1871  newPos.append(getPosition().begin(), getPosition().end());
1872  setPosition(newPos);
1873  getContainerMutable().set(extractValueOp.getContainer());
1874  return getResult();
1875  }
1876 
1877  {
1878  DenseElementsAttr constval;
1879  matchPattern(getContainer(), m_Constant(&constval));
1880  if (constval && constval.getElementType() == getType()) {
1881  if (isa<SplatElementsAttr>(constval))
1882  return constval.getSplatValue<Attribute>();
1883  if (getPosition().size() == 1)
1884  return constval.getValues<Attribute>()[getPosition()[0]];
1885  }
1886  }
1887 
1888  auto insertValueOp = getContainer().getDefiningOp<InsertValueOp>();
1889  OpFoldResult result = {};
1890  ArrayRef<int64_t> extractPos = getPosition();
1891  bool switchedToInsertedValue = false;
1892  while (insertValueOp) {
1893  ArrayRef<int64_t> insertPos = insertValueOp.getPosition();
1894  auto extractPosSize = extractPos.size();
1895  auto insertPosSize = insertPos.size();
1896 
1897  // Case 1: Exact match of positions.
1898  if (extractPos == insertPos)
1899  return insertValueOp.getValue();
1900 
1901  // Case 2: Insert position is a prefix of extract position. Continue
1902  // traversal with the inserted value. Example:
1903  // ```
1904  // %0 = llvm.insertvalue %arg1, %undef[0] : !llvm.struct<(i32, i32, i32)>
1905  // %1 = llvm.insertvalue %arg2, %0[1] : !llvm.struct<(i32, i32, i32)>
1906  // %2 = llvm.insertvalue %arg3, %1[2] : !llvm.struct<(i32, i32, i32)>
1907  // %3 = llvm.insertvalue %2, %foo[0]
1908  // : !llvm.struct<(struct<(i32, i32, i32)>, i64)>
1909  // %4 = llvm.extractvalue %3[0, 0]
1910  // : !llvm.struct<(struct<(i32, i32, i32)>, i64)>
1911  // ```
1912  // In the above example, %4 is folded to %arg1.
1913  if (extractPosSize > insertPosSize &&
1914  extractPos.take_front(insertPosSize) == insertPos) {
1915  insertValueOp = insertValueOp.getValue().getDefiningOp<InsertValueOp>();
1916  extractPos = extractPos.drop_front(insertPosSize);
1917  switchedToInsertedValue = true;
1918  continue;
1919  }
1920 
1921  // Case 3: Try to continue the traversal with the container value.
1922  unsigned min = std::min(extractPosSize, insertPosSize);
1923 
1924  // If one is fully prefix of the other, stop propagating back as it will
1925  // miss dependencies. For instance, %3 should not fold to %f0 in the
1926  // following example:
1927  // ```
1928  // %1 = llvm.insertvalue %f0, %0[0, 0] :
1929  // !llvm.array<4 x !llvm.array<4 x f32>>
1930  // %2 = llvm.insertvalue %arr, %1[0] :
1931  // !llvm.array<4 x !llvm.array<4 x f32>>
1932  // %3 = llvm.extractvalue %2[0, 0] : !llvm.array<4 x !llvm.array<4 x f32>>
1933  // ```
1934  if (extractPos.take_front(min) == insertPos.take_front(min))
1935  return result;
1936  // If neither a prefix, nor the exact position, we can extract out of the
1937  // value being inserted into. Moreover, we can try again if that operand
1938  // is itself an insertvalue expression.
1939  if (!switchedToInsertedValue) {
1940  // Do not swap out the container operand if we decided earlier to
1941  // continue the traversal with the inserted value (Case 2).
1942  getContainerMutable().assign(insertValueOp.getContainer());
1943  result = getResult();
1944  }
1945  insertValueOp = insertValueOp.getContainer().getDefiningOp<InsertValueOp>();
1946  }
1947  return result;
1948 }
