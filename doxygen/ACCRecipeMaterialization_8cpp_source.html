7//===----------------------------------------------------------------------===//
8//
9// Overview:
10// ---------
11// OpenACC compute constructs (acc.parallel, acc.serial, acc.kernels) and
12// acc.loop can carry data clauses (acc.private, acc.firstprivate,
13// acc.reduction) that refer to recipes (acc.private.recipe,
14// acc.firstprivate.recipe, acc.reduction.recipe). Recipes define how to
15// initialize, copy, combine, or destroy a particular variable. This pass clones
16// those regions into the construct and ensures the materialized SSA values are
17// used instead.
18//
19// Transforms:
20// -----------
21// 1. Firstprivate: Inserts acc.firstprivate_map so the initial value is
22// available on the device, then clones the recipe init and copy regions
23// into the construct and replaces uses with the materialized alloca.
24// Optional destroy region is cloned before the region terminator.
25//
26// 2. Private: Clones the recipe init region into the construct (at the
27// region entry or at the loop op for acc.loop private). Replaces uses
28// of the recipe result with the materialized alloca. Optional destroy
29// region is cloned before the region terminator.
30//
31// 3. Reduction: Creates acc.reduction_init (init region inlined) and
32// acc.reduction_combine_region (combiner region inlined). Uses within
33// the region are updated to the reduction init result.
34//
35// Requirements:
36// -------------
37// 1. OpenACCSupport: The pass uses the `acc::OpenACCSupport` analysis
38// including emitNYI for unsupported cases.
39//
40//===----------------------------------------------------------------------===//

76static void setLocation(Region &region, Location loc) {
77 // Since recipes are generated per type and not per variable, the location
78 // of the recipe operations which get inlined will not necessarily be the
79 // same as the location of the op that is being materialized. Force an update
80 // of the location of the recipe operations to the location of the op that is
81 // being materialized.
82 region.walk([&](Operation *op) { op->setLoc(loc); });
83}

85static void saveVarName(StringRef name, Value dst) {
86 if (name.empty())
87 return;
88 if (Operation *dstOp = dst.getDefiningOp()) {
89 if (dstOp->getAttrOfType<acc::VarNameAttr>(acc::getVarNameAttrName()))
90 return;
91 if (isa<ACC_DATA_ENTRY_OPS>(dstOp))
92 return;
93 dstOp->setAttr(acc::getVarNameAttrName(),
94 acc::VarNameAttr::get(dstOp->getContext(), name));
95 return;
96 }
97 auto blockArg = dyn_cast<BlockArgument>(dst);
98 if (!blockArg)
99 return;
100 Block *block = blockArg.getOwner();
101 Region *region = block ? block->getParent() : nullptr;
102 if (!region || !block->isEntryBlock())
103 return;
104 Operation *parent = region->getParentOp();
105 if (!parent)
106 return;
107 auto funcOp = dyn_cast<FunctionOpInterface>(parent);
108 if (!funcOp)
109 return;
110 unsigned argIdx = blockArg.getArgNumber();
111 if (argIdx >= funcOp.getNumArguments())
112 return;
113 if (funcOp.getArgAttr(argIdx, acc::getVarNameAttrName()))
114 return;
115 funcOp.setArgAttr(argIdx, acc::getVarNameAttrName(),
116 acc::VarNameAttr::get(parent->getContext(), name));
117}

119static void saveVarName(Value src, Value dst) {
120 saveVarName(acc::getVariableName(src), dst);
121}

123static void resolveVarNamePlaceholders(Block *block, Block::iterator ip,
124 StringRef name) {
125 StringRef placeholder = acc::getVarNamePlaceholder();
126 for (auto it = block->begin(); it != std::next(ip); ++it) {
127 auto attr = it->getAttrOfType<acc::VarNameAttr>(acc::getVarNameAttrName());
128 if (attr && attr.getName() == placeholder) {
129 if (name.empty())
130 it->removeAttr(acc::getVarNameAttrName());
131 else
132 it->setAttr(acc::getVarNameAttrName(),
133 acc::VarNameAttr::get(it->getContext(), name));
134 }
135 }

138// Clone the destroy region of the recipe before the terminator of the provided
139// block. Values must be provided for the destroy region block arguments
140// according to the recipe specifications.
141template <typename RecipeOpTy>
142static void cloneDestroy(Location loc, RecipeOpTy recipe, mlir::Block *block,
143 Block::iterator ip,
144 const llvm::SmallVector<mlir::Value> &arguments) {
145 IRMapping mapping{};
146 Region &destroyRegion = recipe.getDestroyRegion();
147 assert(destroyRegion.getBlocks().front().getNumArguments() ==
148 arguments.size() &&
149 "unexpected acc recipe destroy block arguments");
150
151 setLocation(destroyRegion, loc);
152
153 mapping.map(destroyRegion.getBlocks().front().getArguments(), arguments);
154 acc::cloneACCRegionInto(&destroyRegion, block, ip, mapping,
155 /*resultsToReplace=*/{});
156}

158class ACCRecipeMaterialization
159 : public acc::impl::ACCRecipeMaterializationBase<ACCRecipeMaterialization> {
160public:
161 using acc::impl::ACCRecipeMaterializationBase<
162 ACCRecipeMaterialization>::ACCRecipeMaterializationBase;
163 void runOnOperation() override;
164
165private:
166 // When handling firstprivate, the initial value needs to be available on
167 // the GPU. One way to get that value there is to map the variable through
168 // global memory.
169 // Thus, when we materialize a firstprivate, we materialize it into
170 // a mapping action first. This function ends up with doing the following:
171 // %dev = acc.firstprivate var(%var)
172 // =>
173 // %copy = acc.firstprivate_map var(%var)
174 // %dev = acc.firstprivate var(%copy)
175 // When the recipe materialization happens, the `acc.firstprivate` ends up
176 // being removed. But because of the way we chain it to the
177 // `acc.firstprivate_map`, then its result becomes live-in to the
178 // compute region and used as the variable the initial value is loaded from.
179 void handleFirstprivateMapping(acc::FirstprivateOp firstprivateOp) const;
180 template <typename OpTy>
181 void removeRecipe(OpTy op, ModuleOp moduleOp) const;
182 template <typename OpTy, typename RecipeOpTy, typename AccOpTy>
183 LogicalResult materialize(OpTy op, RecipeOpTy recipe, AccOpTy accOp,
184 acc::OpenACCSupport &accSupport) const;
185 template <typename OpTy>
186 LogicalResult materializeForACCOp(OpTy accOp,
187 acc::OpenACCSupport &accSupport) const;
188};

190void ACCRecipeMaterialization::handleFirstprivateMapping(
191 acc::FirstprivateOp firstprivateOp) const {
192 OpBuilder builder(firstprivateOp);
193 auto mapFirstprivateOp = acc::FirstprivateMapInitialOp::create(
194 builder, firstprivateOp.getLoc(), firstprivateOp.getVar(),
195 firstprivateOp.getStructured(), firstprivateOp.getImplicit(),
196 firstprivateOp.getBounds());
197 mapFirstprivateOp.setName(firstprivateOp.getName());
198 firstprivateOp.getVarMutable().assign(mapFirstprivateOp.getAccVar());
199}

201template <typename OpTy>
202void ACCRecipeMaterialization::removeRecipe(OpTy op, ModuleOp moduleOp) const {
203 auto recipeName = op.getNameAttr();
204 if (SymbolTable::symbolKnownUseEmpty(recipeName, moduleOp)) {
205 LLVM_DEBUG(llvm::dbgs() << "erasing recipe: " << recipeName << "\n");
206 op.erase();
207 } else {
208 LLVM_DEBUG({
209 std::optional<SymbolTable::UseRange> symbolUses =
210 op.getSymbolUses(moduleOp);
211 if (symbolUses.has_value()) {
212 for (SymbolTable::SymbolUse symbolUse : *symbolUses) {
213 llvm::dbgs() << "symbol use: ";
214 symbolUse.getUser()->dump();
215 }
216 }
217 });
218 llvm_unreachable("expected no use of recipe symbol");
219 }

222template <typename OpTy, typename RecipeOpTy, typename AccOpTy>
223LogicalResult
224ACCRecipeMaterialization::materialize(OpTy op, RecipeOpTy recipe, AccOpTy accOp,
225 acc::OpenACCSupport &accSupport) const {
226 Region &region = accOp.getRegion();
227 Value origPtr = op.getVar();
228 Value accPtr = op.getAccVar();
229 assert(accPtr && "invalid op: null acc var");
230
231 OpBuilder b(op);
232 SmallVector<Value> triples;
233
234 // Clone init block into the region at the insertion point specified.
235 Region &initRegion = recipe.getInitRegion();
236 unsigned initNumArguments =
237 initRegion.getBlocks().front().getArguments().size();
238 if (initNumArguments > 1) {
239 // Code from C/C++ will most likely only provide extent arguments to the
240 // recipe arguments.
241 if ((initNumArguments - 1) % 3 != 0) {
242 (void)accSupport.emitNYI(recipe.getLoc(),
243 "privatization of array section with extents");
244 return failure();
245 }
246 // The remaining arguments must be the bounds triples
247 // (lower-bound, upper-bound, step), ...
248 unsigned argIdx = 1;
249 // Cast the given value to the type of the combiner region's argument
250 // at position argIdx, and increment argIdx.
251 auto castValueToArgType = [&](Location loc, Value v) {
252 return convertScalarToDtype(
253 b, loc, v,
254 initRegion.getBlocks().front().getArgument(argIdx++).getType(),
255 /*isUnsignedCast=*/false);
256 };
257 for (Value bound : acc::getBounds(op)) {
258 auto dataBound = bound.getDefiningOp<acc::DataBoundsOp>();
259 assert(dataBound &&
260 "acc.reduction's bound must be defined by acc.bounds");
261 // NOTE: we should probably generate get_lowerbound, get_upperbound
262 // and get_stride here, so that we can stop looking for the acc.bounds
263 // operation above, and just use the `bound` value.
264 Value lb =
265 castValueToArgType(dataBound.getLoc(), dataBound.getLowerbound());
266 Value ub =
267 castValueToArgType(dataBound.getLoc(), dataBound.getUpperbound());
268 Value step =
269 castValueToArgType(dataBound.getLoc(), dataBound.getStride());
270 triples.append({lb, ub, step});
271 }
272 assert(triples.size() + 1 == initNumArguments &&
273 "mismatch between number bounds and number of recipe init block "
274 "arguments");
275 }
276
277 IRMapping mapping;
278 SmallVector<Value> initArgs{origPtr};
279 initArgs.append(triples);
280 mapping.map(initRegion.getBlocks().front().getArguments(), initArgs);
281
282 Location loc = op.getLoc();
283 setLocation(initRegion, loc);
284
285 if constexpr (std::is_same_v<OpTy, acc::PrivateOp>) {
286 // Clone the init region for a private.
287 Block *block = &region.front();
288 auto [results, ip] = acc::cloneACCRegionInto(
289 &initRegion, block, block->begin(), mapping, {accPtr});
290 assert(results.size() == 1 && "expected single result from init region");
291 saveVarName(op.getAccVar(), results[0]);
292 resolveVarNamePlaceholders(block, ip, acc::getVariableName(op.getAccVar()));
293 // Clone the destroy region for a private, if it exists.
294 if (!recipe.getDestroyRegion().empty()) {
295 results.insert(results.begin(), origPtr);
296 results.append(triples);
297 cloneDestroy(loc, recipe, block, std::prev(block->end()), results);
298 }
299 } else if constexpr (std::is_same_v<OpTy, acc::FirstprivateOp>) {
300 // Clone the init region for a firstprivate.
301 Block *block = &region.front();
302 auto [results, ip] = acc::cloneACCRegionInto(
303 &initRegion, block, block->begin(), mapping, {accPtr});
304 assert(results.size() == 1 && "expected single result from init region");
305 saveVarName(op.getAccVar(), results[0]);
306 resolveVarNamePlaceholders(block, ip, acc::getVariableName(op.getAccVar()));
307 // We want the copy to store the origPtr to private
308 results.insert(results.begin(), origPtr);
309 results.append(triples);
310
311 // Clone the copy region for a firstprivate
312 mapping.clear();
313 mapping.map(recipe.getCopyRegion().front().getArguments(), results);
314 // Clone the copy region for a firstprivate.
315 Region &copyRegion = recipe.getCopyRegion();
316 setLocation(copyRegion, loc);
317 acc::cloneACCRegionInto(&copyRegion, block, std::next(ip), mapping, {});
318 if (!recipe.getDestroyRegion().empty()) {
319 // origPtr was already pushed.
320 cloneDestroy(loc, recipe, block, std::prev(block->end()), results);
321 }
322 } else if constexpr (std::is_same_v<OpTy, acc::ReductionOp>) {
323 auto cloneRegionIntoAccRegion = [&](Region *src, Region *dest,
324 bool hasResult) {
325 src->cloneInto(dest, mapping);
326 Block *block = &dest->front();
327 Operation *terminator = block->getTerminator();
328 b.setInsertionPoint(terminator);
329 if (hasResult)
330 acc::YieldOp::create(b, op.getLoc(), terminator->getOperands());
331 else
332 acc::YieldOp::create(b, op.getLoc(), ValueRange{});
333 terminator->erase();
334 };
335
336 // Clone the init region into acc.reduction_init.
337 if constexpr (std::is_same_v<AccOpTy, acc::ParallelOp>)
338 b.setInsertionPointToStart(&region.front());
339 else if constexpr (std::is_same_v<AccOpTy, acc::LoopOp>)
340 b.setInsertionPoint(op);
341 else
342 llvm_unreachable("unexpected acc op with reduction recipe");
343
344 auto reductionOp = acc::ReductionInitOp::create(
345 b, op.getLoc(), origPtr, recipe.getReductionOperatorAttr());
346 saveVarName(op.getAccVar(), reductionOp.getResult());
347 cloneRegionIntoAccRegion(&initRegion, &reductionOp.getRegion(),
348 /*hasResult=*/true);
349 Block *initBlock = &reductionOp.getRegion().front();
350 resolveVarNamePlaceholders(initBlock, std::prev(initBlock->end()),
351 acc::getVariableName(op.getAccVar()));
352
353 // Update the uses within the loop to use the reduction op result.
354 replaceAllUsesInRegionWith(accPtr, reductionOp.getResult(), region);
355
356 // Clone the combiner region into acc.reduction_combine_region.
357 Region &combinerRegion = recipe.getCombinerRegion();
358 setLocation(combinerRegion, loc);
359
360 Block *entryBlock = &combinerRegion.front();
361
362 if constexpr (std::is_same_v<AccOpTy, acc::ParallelOp>)
363 b.setInsertionPoint(region.back().getTerminator());
364 else if constexpr (std::is_same_v<AccOpTy, acc::LoopOp>)
365 b.setInsertionPointAfter(accOp);
366 else
367 llvm_unreachable("unexpected acc op with reduction recipe");
368
369 // Map the first two block arguments to the original and private
370 // reduction variables. If the recipe's combiner region has the bounds
371 // arguments, we have to map them to the corresponding operands of
372 // acc.reduction operation.
373 mapping.clear();
374 SmallVector<Value, 2> argsRemapping{origPtr, reductionOp.getResult()};
375 argsRemapping.append(triples);
376 mapping.map(entryBlock->getArguments(), argsRemapping);
377
378 auto combineRegionOp = acc::ReductionCombineRegionOp::create(
379 b, op.getLoc(), origPtr, reductionOp.getResult());
380 cloneRegionIntoAccRegion(&combinerRegion, &combineRegionOp.getRegion(),
381 /*hasResult=*/false);
382
383 auto setSeqParDimsForRecipeLoops = [](Region *r) {
384 r->walk([](LoopLikeOpInterface loopLike) {
385 loopLike->setAttr(
386 acc::GPUParallelDimsAttr::name,
387 acc::GPUParallelDimsAttr::seq(loopLike->getContext()));
388 });
389 };
390 setSeqParDimsForRecipeLoops(&reductionOp.getRegion());
391 setSeqParDimsForRecipeLoops(&combineRegionOp.getRegion());
392
393 if (!recipe.getDestroyRegion().empty()) {
394 SmallVector<Value> results{origPtr, reductionOp.getResult()};
395 results.append(triples);
396 Block::iterator ip = std::next(Block::iterator(combineRegionOp));
397 cloneDestroy(loc, recipe, combineRegionOp->getBlock(), ip, results);
398 }
399 } else {
400 llvm_unreachable("unexpected op type");
401 }
402
403 op.erase();
404 return success();
405}

407template <typename OpTy>
408LogicalResult ACCRecipeMaterialization::materializeForACCOp(
409 OpTy accOp, acc::OpenACCSupport &accSupport) const {
410 assert(isa<ACC_COMPUTE_CONSTRUCT_AND_LOOP_OPS>(accOp));
411
412 if (!accOp.getFirstprivateOperands().empty()) {
413 // Clear the firstprivate operands list so there will be no uses after
414 // the recipe is materialized.
415 SmallVector<Value> operands(accOp.getFirstprivateOperands());
416 accOp.getFirstprivateOperandsMutable().clear();
417 for (Value operand : operands) {
418 auto firstprivateOp = cast<acc::FirstprivateOp>(operand.getDefiningOp());
419 auto symbolRef = cast<SymbolRefAttr>(firstprivateOp.getRecipeAttr());
420 auto decl = SymbolTable::lookupNearestSymbolFrom(accOp, symbolRef);
421 auto recipeOp = cast<acc::FirstprivateRecipeOp>(decl);
422 LLVM_DEBUG(llvm::dbgs() << "materializing: " << firstprivateOp << "\n"
423 << symbolRef << "\n");
424 handleFirstprivateMapping(firstprivateOp);
425 if (failed(materialize(firstprivateOp, recipeOp, accOp, accSupport)))
426 return failure();
427 }
428 }
429
430 if (!accOp.getPrivateOperands().empty()) {
431 // Clear the private operands list so there will be no uses after
432 // the recipe is materialized.
433 SmallVector<Value> operands(accOp.getPrivateOperands());
434 accOp.getPrivateOperandsMutable().clear();
435 for (Value operand : operands) {
436 auto privateOp = cast<acc::PrivateOp>(operand.getDefiningOp());
437 auto symbolRef = cast<SymbolRefAttr>(privateOp.getRecipeAttr());
438 auto decl = SymbolTable::lookupNearestSymbolFrom(accOp, symbolRef);
439 auto recipeOp = cast<acc::PrivateRecipeOp>(decl);
440 LLVM_DEBUG(llvm::dbgs() << "materializing: " << privateOp << "\n"
441 << symbolRef << "\n");
442 if (failed(materialize(privateOp, recipeOp, accOp, accSupport)))
443 return failure();
444 }
445 }
446
447 if (!accOp.getReductionOperands().empty()) {
448 // Clear the reduction operands list so there will be no uses after
449 // the recipe is materialized.
450 SmallVector<Value> operands(accOp.getReductionOperands());
451 accOp.getReductionOperandsMutable().clear();
452 for (Value operand : operands) {
453 auto reductionOp = cast<acc::ReductionOp>(operand.getDefiningOp());
454 auto symbolRef = cast<SymbolRefAttr>(reductionOp.getRecipeAttr());
455 auto decl = SymbolTable::lookupNearestSymbolFrom(accOp, symbolRef);
456 auto recipeOp = cast<acc::ReductionRecipeOp>(decl);
457 LLVM_DEBUG(llvm::dbgs() << "materializing: " << reductionOp << "\n"
458 << symbolRef << "\n");
459 if (failed(materialize(reductionOp, recipeOp, accOp, accSupport)))
460 return failure();
461 }
462 }
463 return success();
464}

466void ACCRecipeMaterialization::runOnOperation() {
467 ModuleOp moduleOp = getOperation();
468 acc::OpenACCSupport &accSupport = getAnalysis<acc::OpenACCSupport>();
469
470 // Materialize all recipes for all compute constructs and loop constructs.
471 bool anyFailed = false;
472 moduleOp.walk([&](Operation *op) {
473 if (anyFailed)
474 return;
475 TypeSwitch<Operation *>(op).Case<ACC_COMPUTE_CONSTRUCT_AND_LOOP_OPS>(
476 [&](auto constructOp) {
477 if (failed(materializeForACCOp(constructOp, accSupport)))
478 anyFailed = true;
479 });
480 });
481 if (anyFailed) {
482 signalPassFailure();
483 return;
484 }
485
486 // Remove all recipes.
487 moduleOp.walk([&](Operation *op) {
488 if (auto recipe = dyn_cast<acc::ReductionRecipeOp>(op))
489 removeRecipe(recipe, moduleOp);
490 else if (auto recipe = dyn_cast<acc::PrivateRecipeOp>(op))
491 removeRecipe(recipe, moduleOp);
492 else if (auto recipe = dyn_cast<acc::FirstprivateRecipeOp>(op))
493 removeRecipe(recipe, moduleOp);
494 });
495}
