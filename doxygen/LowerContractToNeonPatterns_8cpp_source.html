34 /// Get the operand of a `vector.contract`. This function is intended to
35 /// abstract away from the particular way a value is extended before feeding it
36 /// into the `vector.contract` - via zero-extend or an explicit or implicit
37 /// sign-extend (for implicit sign-extension see `vector.contract`
38 /// documentation).
39 ///
40 /// The template parameter `Op` indicates the extension operation (explicit or
41 /// implicit) for which we are checking.
42 ///
43 // Return success only for extensions from `iN` (N <= 8) to `i32`.
44 template <typename Op>
45 std::optional<Value> getExtOperand(Value v) {
46 
47  static_assert(llvm::is_one_of<Op, arith::ExtSIOp, arith::ExtUIOp>::value,
48  "Must be instantiated with either sign- or zero- extension op");
49 
50  // If the operand is not defined by an explicit extend operation of the
51  // accepted operation type allow for an implicit sign-extension.
52  auto extOp = v.getDefiningOp<Op>();
53  if (!extOp) {
54  if constexpr (std::is_same<Op, arith::ExtSIOp>::value) {
55  auto eltTy = cast<VectorType>(v.getType()).getElementType();
56  if (!eltTy.isSignlessInteger() || eltTy.getIntOrFloatBitWidth() > 8)
57  return {};
58  return v;
59  }
60  return {};
61  }
62 
63  // If the operand is defined by an explicit extend operation of the accepted
64  // operation type, check it's extended from `iN` (N <= 8) to `i32`.
65  auto inOp = extOp.getIn();
66  auto inTy = dyn_cast<VectorType>(inOp.getType());
67  if (!inTy)
68  return {};
69  auto inEltTy = inTy.getElementType();
70  if (!inEltTy.isSignlessInteger() || inEltTy.getIntOrFloatBitWidth() > 8)
71  return {};
72 
73  auto outTy = dyn_cast<VectorType>(extOp.getType());
74  if (!(outTy && outTy.getElementType().isSignlessInteger(32)))
75  return {};
76 
77  return inOp;
78 }

80 /// Helper function to extend a vector with elements iN, N < 8 to
81 /// a vector of i8. Do sign extension if the parameter `signExt` is true,
82 /// zero extension otherwise.
83 Value extendSmallIntVector(Location loc, VectorType srcTy, Value val,
84  bool signExt, PatternRewriter &rewriter) {
85  Type targetTy = srcTy.clone(rewriter.getI8Type());
86  return signExt ? rewriter.createOrFold<arith::ExtSIOp>(loc, targetTy, val)
87  : rewriter.createOrFold<arith::ExtUIOp>(loc, targetTy, val);
88 }

90 class VectorContractRewriter {
91 protected:
92  // Designate the operation (resp. instruction) used to do sub-tile matrix
93  // multiplications.
94  enum class MMLA {
95  Nop,
96  SignedInt, // smmla
97  UnsignedInt, // ummla
98  MixedInt, // usmmla
99  Bfloat // bfmmla
100  };
101 
102  // Lower-level operation to be emitted.
103  MMLA mmlaOp = MMLA::Nop;
104 
105  // Indicate if the operands for the ArmNeon dialect operation need to be
106  // swapped. Currently this is needed in order to emulate an "summla"
107  // operation.
108  bool swapOperands = false;
109 
110  // The operand tiles. These are not necessarily the operands of
111  // `vector.contract`, for example they could be operands to `arith.extsi`
112  // that is in turn fed into `vector.contract`.
113  Value lhs;
114  Value rhs;
115  Value acc;
116 
117  // The dimensions logically corresponding to matrix multiplication of
118  // MxK * KxN -> MxN. The operands and the result do not necessarily have these
119  // shapes, for example RHS could be NxK with a transposing indexing map.
120  int64_t dimM = 0;
121  int64_t dimN = 0;
122  int64_t dimK = 0;
123 
124  // Unroll iteration bounds. See documentaiton for `StaticTileOffsetRange`.
125  SmallVector<int64_t> iterationBounds;
126 
127  // Sub-tile shape. The algorithm handles operand shapes, which are multiples
128  // of this shape.
129  SmallVector<int64_t> subTileShape;
130 
131  // Create the matrix multiply and accumulate operation according to `mmlaOp`.
132  Value createMMLA(PatternRewriter &rewriter, Location loc, Value acc,
133  Value lhs, Value rhs) {
134 
135  if (swapOperands)
136  std::swap(lhs, rhs);
137  switch (mmlaOp) {
138  case MMLA::SignedInt:
139  return rewriter.createOrFold<arm_neon::SmmlaOp>(loc, acc.getType(), acc,
140  lhs, rhs);
141  case MMLA::UnsignedInt:
142  return rewriter.createOrFold<arm_neon::UmmlaOp>(loc, acc.getType(), acc,
143  lhs, rhs);
144  case MMLA::MixedInt:
145  return rewriter.createOrFold<arm_neon::UsmmlaOp>(loc, acc.getType(), acc,
146  lhs, rhs);
147  case MMLA::Bfloat:
148  return arm_neon::BfmmlaOp::create(rewriter, loc, acc.getType(), acc, lhs,
149  rhs);
150  case MMLA::Nop:
151  llvm_unreachable("Uninitialized operation type");
152  }
153  }
154 
155  // Check common preconditions for applying the patterns and initialize
156  // logical dimensions.
157  LogicalResult matchAndInit(vector::ContractionOp op,
158  PatternRewriter &rewriter) {
159  // Check iterator types for matrix multiplication.
160  SmallVector<vector::IteratorType> itTypes = op.getIteratorTypesArray();
161  if (!((itTypes.size() == 3 &&
162  (itTypes[0] == vector::IteratorType::parallel &&
163  itTypes[1] == vector::IteratorType::parallel &&
164  itTypes[2] == vector::IteratorType::reduction)) ||
165  (itTypes.size() == 2 &&
166  (itTypes[0] == vector::IteratorType::parallel &&
167  itTypes[1] == vector::IteratorType::reduction))))
168  return rewriter.notifyMatchFailure(
169  op, "iterator types do not correspond to matrix multiplication");
170 
171  // Avoid 0-D vectors and 1-D rhs:
172  VectorType lhsType = op.getLhsType();
173  VectorType rhsType = op.getRhsType();
174  if (!lhsType.hasRank() || !rhsType.hasRank() || lhsType.getRank() > 2 ||
175  rhsType.getRank() != 2)
176  return rewriter.notifyMatchFailure(op, "Invalid operand rank");
177 
178  // This codegen does not work for scalable vectors. Return failure so this
179  // pattern is not accidentally chosen over patterns that lower to ArmSVE.
180  if (lhsType.isScalable() || rhsType.isScalable())
181  return rewriter.notifyMatchFailure(op,
182  "Not applicable to scalable vectors");
183 
184  // Initialize dimensions and check for a matching K dimension.
185  dimM = lhsType.getDimSize(0);
186  dimN = rhsType.getDimSize(0);
187  dimK = rhsType.getDimSize(1);
188 
189  int64_t lhsDimK;
190  if (lhsType.getRank() == 1) {
191  dimM = 1;
192  lhsDimK = lhsType.getDimSize(0);
193  } else {
194  lhsDimK = lhsType.getDimSize(1);
195  }
196 
197  if (lhsDimK != dimK)
198  return rewriter.notifyMatchFailure(op, "Dimensions mismatch");
199 
200  return success();
201  }

203 public:
204  void lower(vector::ContractionOp op, PatternRewriter &rewriter) {
205  // Create some convenience types.
206  auto inputElementType = cast<ShapedType>(lhs.getType()).getElementType();
207  auto accElementType = cast<ShapedType>(acc.getType()).getElementType();
208  auto inputExpandedType =
209  VectorType::get({2, subTileShape.back()}, inputElementType);
210  auto outputExpandedType = VectorType::get({2, 2}, accElementType);
211 
212  // One-dimensional representation of logical sub-tiles as required by the
213  // ArmNeon ops.
214  auto collapsedInputType =
215  VectorType::get(inputExpandedType.getNumElements(), inputElementType);
216  auto collapsedOutputType =
217  VectorType::get(outputExpandedType.getNumElements(), accElementType);
218 
219  // Get indexing maps for a more concise/convenient access.
220  auto indexingMaps = op.getIndexingMapsArray();
221  AffineMap &lhsPermutationMap = indexingMaps[0];
222  AffineMap &rhsPermutationMap = indexingMaps[1];
223  AffineMap &accPermutationMap = indexingMaps[2];
224 
225  Location loc = op.getLoc();
226 
227  // Initial accumulator for the final result. This is the un-tiled result if
228  // tiling is done.
229  Value result =
230  arith::ConstantOp::create(rewriter, loc, op.getResultType(),
231  rewriter.getZeroAttr(op.getResultType()));
232 
233  SmallVector<int64_t, 3> loopOrder = {0, 1};
234  if (iterationBounds.size() == 3)
235  loopOrder.push_back(2);
236 
237  // Keep track of the previous accumulator when tiling over K.
238  Value kAcc;
239  for (SmallVector<int64_t> offsets :
240  StaticTileOffsetRange(iterationBounds, subTileShape, loopOrder)) {
241  // Helper to compute the new shape of each operand and extract the slice.
242  auto extractOperand = [&](Value operand, AffineMap permutationMap,
243  ArrayRef<int64_t> operandOffsets) {
244  SmallVector<int64_t> operandShape = applyPermutationMap(
245  permutationMap, ArrayRef<int64_t>(subTileShape));
246  SmallVector<int64_t> operandStrides(operandOffsets.size(), 1);
247  return rewriter.createOrFold<vector::ExtractStridedSliceOp>(
248  loc, operand, operandOffsets, operandShape, operandStrides);
249  };
250 
251  // Extract tiled lhs, rhs, and acc
252  SmallVector<int64_t> lhsOffsets =
253  applyPermutationMap(lhsPermutationMap, ArrayRef<int64_t>(offsets));
254  Value tiledLhs = extractOperand(lhs, lhsPermutationMap, lhsOffsets);
255  SmallVector<int64_t> rhsOffsets =
256  applyPermutationMap(rhsPermutationMap, ArrayRef<int64_t>(offsets));
257  Value tiledRhs = extractOperand(rhs, rhsPermutationMap, rhsOffsets);
258  SmallVector<int64_t> accOffsets =
259  applyPermutationMap(accPermutationMap, ArrayRef<int64_t>(offsets));
260  Value tiledAcc = extractOperand(acc, accPermutationMap, accOffsets);
261 
262  // With vecmat, tiled LHS and ACC will contain only one of 2 necessary
263  // rows along dimM. Expand their shapes to match the ArmNeon op.
264  if (dimM == 1) {
265  auto expandRowVector = [&](Value tiledOperand,
266  VectorType expandedTypeType) {
267  auto emptyOperand =
268  arith::ConstantOp::create(rewriter, loc, expandedTypeType,
269  rewriter.getZeroAttr(expandedTypeType));
270  SmallVector<int64_t> offsets(
271  cast<ShapedType>(emptyOperand.getType()).getRank(), 0);
272  SmallVector<int64_t> strides(
273  cast<ShapedType>(tiledOperand.getType()).getRank(), 1);
274  return rewriter.createOrFold<vector::InsertStridedSliceOp>(
275  loc, tiledOperand, emptyOperand, offsets, strides);
276  };
277  tiledLhs = expandRowVector(tiledLhs, inputExpandedType);
278  tiledAcc = expandRowVector(tiledAcc, outputExpandedType);
279  }
280 
281  // Transpose ACC if doing signed by unsigned multiplication, because we're
282  // using the instruction for unsigned by signed multiplication with
283  // reversed operands.
284  if (swapOperands)
285  tiledAcc = vector::TransposeOp::create(rewriter, loc, tiledAcc,
286  ArrayRef<int64_t>({1, 0}));
287 
288  // Collapse tiled operands to 1D vectors required by the ArmNeon ops
289  auto collapsedLhs = rewriter.createOrFold<vector::ShapeCastOp>(
290  tiledLhs.getLoc(), collapsedInputType, tiledLhs);
291  auto collapsedRhs = rewriter.createOrFold<vector::ShapeCastOp>(
292  tiledRhs.getLoc(), collapsedInputType, tiledRhs);
293 
294  bool initialKAcc = offsets.back() == 0;
295  Value collapsedRes;
296  if (!initialKAcc) {
297  collapsedRes = kAcc;
298  } else {
299  collapsedRes = rewriter.createOrFold<vector::ShapeCastOp>(
300  tiledAcc.getLoc(), collapsedOutputType, tiledAcc);
301  }
302 
303  // Insert contract op
304  kAcc =
305  createMMLA(rewriter, loc, collapsedRes, collapsedLhs, collapsedRhs);
306 
307  // Reshape output back to 2D
308  Value tiledRes = rewriter.createOrFold<vector::ShapeCastOp>(
309  kAcc.getLoc(), tiledAcc.getType(), kAcc);
310 
311  // Because of the reversed operands the result is obtained transposed.
312  // Transpose it back,
313  if (swapOperands)
314  tiledRes = vector::TransposeOp::create(rewriter, loc, tiledRes,
315  ArrayRef<int64_t>({1, 0}));
316 
317  // With vecmat, only one row of tiled ACC can be inserted into the final
318  // result
319  if (dimM == 1)
320  tiledRes = rewriter.createOrFold<vector::ExtractOp>(loc, tiledRes, 0);
321 
322  // Insert the tiled result back into the non tiled result of the
323  // contract op.
324  SmallVector<int64_t> strides(
325  cast<ShapedType>(tiledRes.getType()).getRank(), 1);
326  result = rewriter.createOrFold<vector::InsertStridedSliceOp>(
327  loc, tiledRes, result, accOffsets, strides);
328  }
329 
330  rewriter.replaceOp(op, result);
331  }
332 };

334 class VectorContractRewriterI8MM : public VectorContractRewriter {
335 public:
336  LogicalResult matchAndInit(vector::ContractionOp op,
337  PatternRewriter &rewriter) {
338  if (failed(VectorContractRewriter::matchAndInit(op, rewriter)))
339  return failure();
340 
341  // Unrolling patterns can handle any [2, 2, 8] shaped multiple of inputs for
342  // tiling.
343  if ((dimM != 1 && dimM % 2 != 0) || dimN % 2 != 0 || dimK % 8 != 0)
344  return rewriter.notifyMatchFailure(op, "Unsupported operand shapes");
345 
346  // Check inputs are sign-/zero- extensions from iN (N <= 8) to i32. Get the
347  // values before the extension. All four signed/unsigned combinations for
348  // input operands are supported, but they are lowered to different
349  // operations. Determine which is the appropriate operation to lower to.
350  mmlaOp = MMLA::SignedInt;
351  auto maybeLhs = getExtOperand<arith::ExtSIOp>(op.getLhs());
352  if (!maybeLhs) {
353  mmlaOp = MMLA::UnsignedInt;
354  maybeLhs = getExtOperand<arith::ExtUIOp>(op.getLhs());
355  }
356  if (!maybeLhs)
357  return rewriter.notifyMatchFailure(
358  op, "LHS is not a sign- or zero- extended iN, N <= 8");
359 
360  auto maybeRhs = getExtOperand<arith::ExtSIOp>(op.getRhs());
361  if (maybeRhs) {
362  if (mmlaOp == MMLA::UnsignedInt)
363  mmlaOp = MMLA::MixedInt;
364  } else {
365  if (mmlaOp == MMLA::SignedInt) {
366  mmlaOp = MMLA::MixedInt;
367  swapOperands = true;
368  }
369  maybeRhs = getExtOperand<arith::ExtUIOp>(op.getRhs());
370  }
371 
372  if (!maybeRhs)
373  return rewriter.notifyMatchFailure(
374  op, "RHS is not a sign- or zero- extended iN, N <= 8");
375 
376  lhs = *maybeLhs;
377  rhs = *maybeRhs;
378  acc = op.getAcc();
379 
380  // Extend inputs from iN, N < 8 to i8.
381  Location loc = op.getLoc();
382  auto lhsExtInType = cast<VectorType>(lhs.getType());
383  if (lhsExtInType.getElementTypeBitWidth() < 8)
384  lhs = extendSmallIntVector(loc, lhsExtInType, lhs,
385  /* signExt */
386  (mmlaOp == MMLA::SignedInt ||
387  (mmlaOp == MMLA::MixedInt && !swapOperands)),
388  rewriter);
389 
390  auto rhsExtInType = cast<VectorType>(rhs.getType());
391  if (rhsExtInType.getElementTypeBitWidth() < 8)
392  rhs = extendSmallIntVector(loc, rhsExtInType, rhs,
393  /* signExt */
394  (mmlaOp == MMLA::SignedInt ||
395  (mmlaOp == MMLA::MixedInt && swapOperands)),
396  rewriter);
397 
398  // Initialize parameters for unrolling.
399  iterationBounds = *op.getShapeForUnroll();
400  if (iterationBounds.size() == 3)
401  subTileShape = SmallVector<int64_t>({dimM == 1 ? 1 : 2, 2, 8});
402  else
403  subTileShape = SmallVector<int64_t>({2, 8});
404 
405  return success();
406  }

409 class VectorContractRewriterBFMMLA : public VectorContractRewriter {
410 public:
411  LogicalResult matchAndInit(vector::ContractionOp op,
412  PatternRewriter &rewriter) {
413 
414  if (failed(VectorContractRewriter::matchAndInit(op, rewriter)))
