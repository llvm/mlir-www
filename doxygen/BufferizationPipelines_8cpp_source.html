24void mlir::bufferization::buildBufferDeallocationPipeline(
25 OpPassManager &pm, const BufferDeallocationPipelineOptions &options) {
26 memref::ExpandReallocPassOptions expandAllocPassOptions{
27 /*emitDeallocs=*/false};
28 pm.addPass(memref::createExpandReallocPass(expandAllocPassOptions));
29 pm.addPass(createCanonicalizerPass());
30
31 OwnershipBasedBufferDeallocationPassOptions deallocationOptions{
32 options.privateFunctionDynamicOwnership};
33 pm.addPass(createOwnershipBasedBufferDeallocationPass(deallocationOptions));
34 pm.addPass(createCanonicalizerPass());
35 pm.addPass(createBufferDeallocationSimplificationPass());
36 pm.addPass(createLowerDeallocationsPass());
37 pm.addPass(createCSEPass());
38 pm.addPass(createCanonicalizerPass());
39}

50 "(except `memref.realloc`) may not be present already.",
