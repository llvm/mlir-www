42 /// This is a common class used for patterns of the form
43 /// "someop(memrefcast) -> someop". It folds the source of any memref.cast
44 /// into the root operation directly.
45 LogicalResult mlir::memref::foldMemRefCast(Operation *op, Value inner) {
46  bool folded = false;
47  for (OpOperand &operand : op->getOpOperands()) {
48  auto cast = operand.get().getDefiningOp<CastOp>();
49  if (cast && operand.get() != inner &&
50  !llvm::isa<UnrankedMemRefType>(cast.getOperand().getType())) {
51  operand.set(cast.getOperand());
52  folded = true;
53  }
54  }
55  return success(folded);
56 }

90 /// Helper function that sets values[i] to constValues[i] if the latter is a
91 /// static value, as indicated by ShapedType::kDynamic.
92 ///
93 /// If constValues[i] is dynamic, tries to extract a constant value from
94 /// value[i] to allow for additional folding opportunities. Also convertes all
95 /// existing attributes to index attributes. (They may be i64 attributes.)
96 static void constifyIndexValues(SmallVectorImpl<OpFoldResult> &values,
97  ArrayRef<int64_t> constValues) {
98  assert(constValues.size() == values.size() &&
99  "incorrect number of const values");
100  for (auto [i, cstVal] : llvm::enumerate(constValues)) {
101  Builder builder(values[i].getContext());
102  if (ShapedType::isStatic(cstVal)) {
103  // Constant value is known, use it directly.
104  values[i] = builder.getIndexAttr(cstVal);
105  continue;
106  }
107  if (std::optional<int64_t> cst = getConstantIntValue(values[i])) {
108  // Try to extract a constant or convert an existing to index.
109  values[i] = builder.getIndexAttr(*cst);
110  }
111  }
112 }

114 /// Helper function to retrieve a lossless memory-space cast, and the
115 /// corresponding new result memref type.
116 static std::tuple<MemorySpaceCastOpInterface, PtrLikeTypeInterface, Type>
117 getMemorySpaceCastInfo(BaseMemRefType resultTy, Value src) {
118  MemorySpaceCastOpInterface castOp =
119  MemorySpaceCastOpInterface::getIfPromotableCast(src);
120 
121  // Bail if the cast is not lossless.
122  if (!castOp)
123  return {};
124 
125  // Transform the source and target type of `castOp` to have the same metadata
126  // as `resultTy`. Bail if not possible.
127  FailureOr<PtrLikeTypeInterface> srcTy = resultTy.clonePtrWith(
128  castOp.getSourcePtr().getType().getMemorySpace(), std::nullopt);
129  if (failed(srcTy))
130  return {};
131 
132  FailureOr<PtrLikeTypeInterface> tgtTy = resultTy.clonePtrWith(
133  castOp.getTargetPtr().getType().getMemorySpace(), std::nullopt);
134  if (failed(tgtTy))
135  return {};
136 
137  // Check if this is a valid memory-space cast.
138  if (!castOp.isValidMemorySpaceCast(*tgtTy, *srcTy))
139  return {};
140 
141  return std::make_tuple(castOp, *tgtTy, *srcTy);
142 }

144 /// Implementation of `bubbleDownCasts` method for memref operations that
145 /// return a single memref result.
146 template <typename ConcreteOpTy>
147 static FailureOr<std::optional<SmallVector<Value>>>
148 bubbleDownCastsPassthroughOpImpl(ConcreteOpTy op, OpBuilder &builder,
149  OpOperand &src) {
150  auto [castOp, tgtTy, resTy] = getMemorySpaceCastInfo(op.getType(), src.get());
151  // Bail if we cannot cast.
152  if (!castOp)
153  return failure();
154 
155  // Create the new operands.
156  SmallVector<Value> operands;
157  llvm::append_range(operands, op->getOperands());
158  operands[src.getOperandNumber()] = castOp.getSourcePtr();
159 
160  // Create the new op and results.
161  auto newOp = ConcreteOpTy::create(
162  builder, op.getLoc(), TypeRange(resTy), operands, op.getProperties(),
163  llvm::to_vector_of<NamedAttribute>(op->getDiscardableAttrs()));
164 
165  // Insert a memory-space cast to the original memory space of the op.
166  MemorySpaceCastOpInterface result = castOp.cloneMemorySpaceCastOp(
167  builder, tgtTy,
168  cast<TypedValue<PtrLikeTypeInterface>>(newOp.getResult()));
169  return std::optional<SmallVector<Value>>(
170  SmallVector<Value>({result.getTargetPtr()}));
171 }

189  static_assert(llvm::is_one_of<AllocLikeOp, AllocOp, AllocaOp>::value,
190  "applies to only alloc or alloca");
191  auto memRefType = llvm::dyn_cast<MemRefType>(op.getResult().getType());
192  if (!memRefType)
193  return op.emitOpError("result must be a memref");

203  return op.emitOpError("symbol operand count does not equal memref symbol "
204  "count: expected ")
205  << numSymbols << ", got " << op.getSymbolOperands().size();
206 
207  return success();

222 /// Fold constant dimensions into an alloc like operation.
223 template <typename AllocLikeOp>
224 struct SimplifyAllocConst : public OpRewritePattern<AllocLikeOp> {
225  using OpRewritePattern<AllocLikeOp>::OpRewritePattern;
226 
227  LogicalResult matchAndRewrite(AllocLikeOp alloc,
228  PatternRewriter &rewriter) const override {
229  // Check to see if any dimensions operands are constants. If so, we can
230  // substitute and drop them.
231  if (llvm::none_of(alloc.getDynamicSizes(), [](Value operand) {
232  APInt constSizeArg;
233  if (!matchPattern(operand, m_ConstantInt(&constSizeArg)))
234  return false;
235  return constSizeArg.isNonNegative();
236  }))
237  return failure();
238 
239  auto memrefType = alloc.getType();
240 
241  // Ok, we have one or more constant operands. Collect the non-constant ones
242  // and keep track of the resultant memref type to build.
243  SmallVector<int64_t, 4> newShapeConstants;
244  newShapeConstants.reserve(memrefType.getRank());
245  SmallVector<Value, 4> dynamicSizes;
246 
247  unsigned dynamicDimPos = 0;
248  for (unsigned dim = 0, e = memrefType.getRank(); dim < e; ++dim) {
249  int64_t dimSize = memrefType.getDimSize(dim);
250  // If this is already static dimension, keep it.
251  if (ShapedType::isStatic(dimSize)) {
252  newShapeConstants.push_back(dimSize);
253  continue;
254  }
255  auto dynamicSize = alloc.getDynamicSizes()[dynamicDimPos];
256  APInt constSizeArg;
257  if (matchPattern(dynamicSize, m_ConstantInt(&constSizeArg)) &&
258  constSizeArg.isNonNegative()) {
259  // Dynamic shape dimension will be folded.
260  newShapeConstants.push_back(constSizeArg.getZExtValue());
261  } else {
262  // Dynamic shape dimension not folded; copy dynamicSize from old memref.
263  newShapeConstants.push_back(ShapedType::kDynamic);
264  dynamicSizes.push_back(dynamicSize);
265  }
266  dynamicDimPos++;
267  }
268 
269  // Create new memref type (which will have fewer dynamic dimensions).
270  MemRefType newMemRefType =
271  MemRefType::Builder(memrefType).setShape(newShapeConstants);
272  assert(dynamicSizes.size() == newMemRefType.getNumDynamicDims());
273 
274  // Create and insert the alloc op for the new memref.
275  auto newAlloc = AllocLikeOp::create(rewriter, alloc.getLoc(), newMemRefType,
276  dynamicSizes, alloc.getSymbolOperands(),
277  alloc.getAlignmentAttr());
278  // Insert a cast so we have the same type as the old alloc.
279  rewriter.replaceOpWithNewOp<CastOp>(alloc, alloc.getType(), newAlloc);
280  return success();
281  }
282 };

284 /// Fold alloc operations with no users or only store and dealloc uses.
285 template <typename T>
286 struct SimplifyDeadAlloc : public OpRewritePattern<T> {
287  using OpRewritePattern<T>::OpRewritePattern;
288 
289  LogicalResult matchAndRewrite(T alloc,
290  PatternRewriter &rewriter) const override {
291  if (llvm::any_of(alloc->getUsers(), [&](Operation *op) {
292  if (auto storeOp = dyn_cast<StoreOp>(op))
293  return storeOp.getValue() == alloc;
294  return !isa<DeallocOp>(op);
295  }))
296  return failure();
297 
298  for (Operation *user : llvm::make_early_inc_range(alloc->getUsers()))
299  rewriter.eraseOp(user);
300 
301  rewriter.eraseOp(alloc);
302  return success();
303  }
304 };

323  auto sourceType = llvm::cast<MemRefType>(getOperand(0).getType());
324  MemRefType resultType = getType();
325 
326  // The source memref should have identity layout (or none).
327  if (!sourceType.getLayout().isIdentity())
328  return emitError("unsupported layout for source memref type ")
329  << sourceType;
330 
331  // The result memref should have identity layout (or none).
332  if (!resultType.getLayout().isIdentity())
333  return emitError("unsupported layout for result memref type ")
334  << resultType;
335 
336  // The source memref and the result memref should be in the same memory space.
337  if (sourceType.getMemorySpace() != resultType.getMemorySpace())
338  return emitError("different memory spaces specified for source memref "
339  "type ")
340  << sourceType << " and result memref type " << resultType;
341 
342  // The source memref and the result memref should have the same element type.
343  if (sourceType.getElementType() != resultType.getElementType())
344  return emitError("different element types specified for source memref "
345  "type ")
346  << sourceType << " and result memref type " << resultType;
347 
348  // Verify that we have the dynamic dimension operand when it is needed.
349  if (resultType.getNumDynamicDims() && !getDynamicResultSize())
350  return emitError("missing dimension operand for result type ")
351  << resultType;
352  if (!resultType.getNumDynamicDims() && getDynamicResultSize())
353  return emitError("unnecessary dimension operand for result type ")
354  << resultType;
355 
356  return success();
357 }

415 /// Given an operation, return whether this op is guaranteed to
416 /// allocate an AutomaticAllocationScopeResource
417 static bool isGuaranteedAutomaticAllocation(Operation *op) {
418  MemoryEffectOpInterface interface = dyn_cast<MemoryEffectOpInterface>(op);
419  if (!interface)
420  return false;
421  for (auto res : op->getResults()) {
422  if (auto effect =
423  interface.getEffectOnValue<MemoryEffects::Allocate>(res)) {
424  if (isa<SideEffects::AutomaticAllocationScopeResource>(
425  effect->getResource()))
426  return true;
427  }
428  }
429  return false;
430 }

432 /// Given an operation, return whether this op itself could
433 /// allocate an AutomaticAllocationScopeResource. Note that
434 /// this will not check whether an operation contained within
435 /// the op can allocate.
436 static bool isOpItselfPotentialAutomaticAllocation(Operation *op) {
437  // This op itself doesn't create a stack allocation,
438  // the inner allocation should be handled separately.
439  if (op->hasTrait<OpTrait::HasRecursiveMemoryEffects>())
440  return false;
441  MemoryEffectOpInterface interface = dyn_cast<MemoryEffectOpInterface>(op);
442  if (!interface)
443  return true;
444  for (auto res : op->getResults()) {
445  if (auto effect =
446  interface.getEffectOnValue<MemoryEffects::Allocate>(res)) {
447  if (isa<SideEffects::AutomaticAllocationScopeResource>(
448  effect->getResource()))
449  return true;
450  }
451  }
452  return false;
453 }

465 /// Inline an AllocaScopeOp if either the direct parent is an allocation scope
466 /// or it contains no allocation.
467 struct AllocaScopeInliner : public OpRewritePattern<AllocaScopeOp> {
468  using OpRewritePattern<AllocaScopeOp>::OpRewritePattern;
469 
470  LogicalResult matchAndRewrite(AllocaScopeOp op,
471  PatternRewriter &rewriter) const override {
472  bool hasPotentialAlloca =
473  op->walk<WalkOrder::PreOrder>([&](Operation *alloc) {
474  if (alloc == op)
475  return WalkResult::advance();
476  if (isOpItselfPotentialAutomaticAllocation(alloc))
477  return WalkResult::interrupt();
478  if (alloc->hasTrait<OpTrait::AutomaticAllocationScope>())
479  return WalkResult::skip();
480  return WalkResult::advance();
481  }).wasInterrupted();
482 
483  // If this contains no potential allocation, it is always legal to
484  // inline. Otherwise, consider two conditions:
485  if (hasPotentialAlloca) {
486  // If the parent isn't an allocation scope, or we are not the last
487  // non-terminator op in the parent, we will extend the lifetime.
488  if (!op->getParentOp()->hasTrait<OpTrait::AutomaticAllocationScope>())
489  return failure();
490  if (!lastNonTerminatorInRegion(op))
491  return failure();
492  }
493 
494  Block *block = &op.getRegion().front();
495  Operation *terminator = block->getTerminator();
496  ValueRange results = terminator->getOperands();
497  rewriter.inlineBlockBefore(block, op);
498  rewriter.replaceOp(op, results);
499  rewriter.eraseOp(terminator);
500  return success();
501  }
502 };

504 /// Move allocations into an allocation scope, if it is legal to
505 /// move them (e.g. their operands are available at the location
506 /// the op would be moved to).
507 struct AllocaScopeHoister : public OpRewritePattern<AllocaScopeOp> {
508  using OpRewritePattern<AllocaScopeOp>::OpRewritePattern;
509 
510  LogicalResult matchAndRewrite(AllocaScopeOp op,
511  PatternRewriter &rewriter) const override {
512 
513  if (!op->getParentWithTrait<OpTrait::AutomaticAllocationScope>())
514  return failure();
515 
516  Operation *lastParentWithoutScope = op->getParentOp();
517 
518  if (!lastParentWithoutScope ||
519  lastParentWithoutScope->hasTrait<OpTrait::AutomaticAllocationScope>())
520  return failure();
521 
522  // Only apply to if this is this last non-terminator
523  // op in the block (lest lifetime be extended) of a one
524  // block region
525  if (!lastNonTerminatorInRegion(op) ||
526  !lastNonTerminatorInRegion(lastParentWithoutScope))
527  return failure();
528 
529  while (!lastParentWithoutScope->getParentOp()
530  ->hasTrait<OpTrait::AutomaticAllocationScope>()) {
531  lastParentWithoutScope = lastParentWithoutScope->getParentOp();
532  if (!lastParentWithoutScope ||
533  !lastNonTerminatorInRegion(lastParentWithoutScope))
534  return failure();
535  }
536  assert(lastParentWithoutScope->getParentOp()
537  ->hasTrait<OpTrait::AutomaticAllocationScope>());
538 
539  Region *containingRegion = nullptr;
540  for (auto &r : lastParentWithoutScope->getRegions()) {
541  if (r.isAncestor(op->getParentRegion())) {
542  assert(containingRegion == nullptr &&
543  "only one region can contain the op");
544  containingRegion = &r;
545  }
546  }
547  assert(containingRegion && "op must be contained in a region");
548 
549  SmallVector<Operation *> toHoist;
550  op->walk([&](Operation *alloc) {
551  if (!isGuaranteedAutomaticAllocation(alloc))
552  return WalkResult::skip();
553 
554  // If any operand is not defined before the location of
555  // lastParentWithoutScope (i.e. where we would hoist to), skip.
556  if (llvm::any_of(alloc->getOperands(), [&](Value v) {
557  return containingRegion->isAncestor(v.getParentRegion());
558  }))
559  return WalkResult::skip();
560  toHoist.push_back(alloc);
561  return WalkResult::advance();
562  });
563 
564  if (toHoist.empty())
565  return failure();
566  rewriter.setInsertionPoint(lastParentWithoutScope);
567  for (auto *op : toHoist) {
568  auto *cloned = rewriter.clone(*op);
569  rewriter.replaceOp(op, cloned->getResults());
570  }
571  return success();
572  }
573 };

617 /// Determines whether MemRef_CastOp casts to a more dynamic version of the
618 /// source memref. This is useful to fold a memref.cast into a consuming op
619 /// and implement canonicalization patterns for ops in different dialects that
620 /// may consume the results of memref.cast operations. Such foldable memref.cast
621 /// operations are typically inserted as `view` and `subview` ops are
622 /// canonicalized, to preserve the type compatibility of their uses.
623 ///
624 /// Returns true when all conditions are met:
625 /// 1. source and result are ranked memrefs with strided semantics and same
626 /// element type and rank.
627 /// 2. each of the source's size, offset or stride has more static information
628 /// than the corresponding result's size, offset or stride.
629 ///
630 /// Example 1:
631 /// ```mlir
632 /// %1 = memref.cast %0 : memref<8x16xf32> to memref<?x?xf32>
633 /// %2 = consumer %1 ... : memref<?x?xf32> ...
634 /// ```
635 ///
636 /// may fold into:
637 ///
638 /// ```mlir
639 /// %2 = consumer %0 ... : memref<8x16xf32> ...
640 /// ```
641 ///
642 /// Example 2:
643 /// ```
644 /// %1 = memref.cast %0 : memref<?x16xf32, affine_map<(i, j)->(16 * i + j)>>
645 /// to memref<?x?xf32>
646 /// consumer %1 : memref<?x?xf32> ...
647 /// ```
648 ///
649 /// may fold into:
650 ///
651 /// ```
652 /// consumer %0 ... : memref<?x16xf32, affine_map<(i, j)->(16 * i + j)>>
653 /// ```
654 bool CastOp::canFoldIntoConsumerOp(CastOp castOp) {
655  MemRefType sourceType =
656  llvm::dyn_cast<MemRefType>(castOp.getSource().getType());
657  MemRefType resultType = llvm::dyn_cast<MemRefType>(castOp.getType());
658 
659  // Requires ranked MemRefType.
660  if (!sourceType || !resultType)
661  return false;
662 
663  // Requires same elemental type.
664  if (sourceType.getElementType() != resultType.getElementType())
665  return false;
666 
667  // Requires same rank.
668  if (sourceType.getRank() != resultType.getRank())
669  return false;
670 
671  // Only fold casts between strided memref forms.
672  int64_t sourceOffset, resultOffset;
673  SmallVector<int64_t, 4> sourceStrides, resultStrides;
674  if (failed(sourceType.getStridesAndOffset(sourceStrides, sourceOffset)) ||
675  failed(resultType.getStridesAndOffset(resultStrides, resultOffset)))
676  return false;
677 
678  // If cast is towards more static sizes along any dimension, don't fold.
679  for (auto it : llvm::zip(sourceType.getShape(), resultType.getShape())) {
680  auto ss = std::get<0>(it), st = std::get<1>(it);
681  if (ss != st)
682  if (ShapedType::isDynamic(ss) && ShapedType::isStatic(st))
683  return false;
684  }
685 
686  // If cast is towards more static offset along any dimension, don't fold.
687  if (sourceOffset != resultOffset)
688  if (ShapedType::isDynamic(sourceOffset) &&
689  ShapedType::isStatic(resultOffset))
690  return false;
691 
692  // If cast is towards more static strides along any dimension, don't fold.
693  for (auto it : llvm::zip(sourceStrides, resultStrides)) {
694  auto ss = std::get<0>(it), st = std::get<1>(it);
695  if (ss != st)
696  if (ShapedType::isDynamic(ss) && ShapedType::isStatic(st))
697  return false;
698  }
699 
700  return true;
701 }

703 bool CastOp::areCastCompatible(TypeRange inputs, TypeRange outputs) {
704  if (inputs.size() != 1 || outputs.size() != 1)
705  return false;
706  Type a = inputs.front(), b = outputs.front();
707  auto aT = llvm::dyn_cast<MemRefType>(a);
708  auto bT = llvm::dyn_cast<MemRefType>(b);
709 
710  auto uaT = llvm::dyn_cast<UnrankedMemRefType>(a);
711  auto ubT = llvm::dyn_cast<UnrankedMemRefType>(b);
712 
713  if (aT && bT) {
714  if (aT.getElementType() != bT.getElementType())
715  return false;
716  if (aT.getLayout() != bT.getLayout()) {
717  int64_t aOffset, bOffset;
718  SmallVector<int64_t, 4> aStrides, bStrides;
719  if (failed(aT.getStridesAndOffset(aStrides, aOffset)) ||
720  failed(bT.getStridesAndOffset(bStrides, bOffset)) ||
721  aStrides.size() != bStrides.size())
722  return false;
723 
724  // Strides along a dimension/offset are compatible if the value in the
725  // source memref is static and the value in the target memref is the
726  // same. They are also compatible if either one is dynamic (see
727  // description of MemRefCastOp for details).
728  auto checkCompatible = [](int64_t a, int64_t b) {
729  return (ShapedType::isDynamic(a) || ShapedType::isDynamic(b) || a == b);
730  };
731  if (!checkCompatible(aOffset, bOffset))
732  return false;
733  for (const auto &aStride : enumerate(aStrides))
734  if (!checkCompatible(aStride.value(), bStrides[aStride.index()]))
735  return false;
736  }
737  if (aT.getMemorySpace() != bT.getMemorySpace())
738  return false;
739 
740  // They must have the same rank, and any specified dimensions must match.
741  if (aT.getRank() != bT.getRank())
742  return false;
743 
744  for (unsigned i = 0, e = aT.getRank(); i != e; ++i) {
745  int64_t aDim = aT.getDimSize(i), bDim = bT.getDimSize(i);
746  if (ShapedType::isStatic(aDim) && ShapedType::isStatic(bDim) &&
747  aDim != bDim)
748  return false;
749  }
750  return true;
751  } else {
752  if (!aT && !uaT)
753  return false;
754  if (!bT && !ubT)
755  return false;
756  // Unranked to unranked casting is unsupported
757  if (uaT && ubT)
758  return false;
759 
760  auto aEltType = (aT) ? aT.getElementType() : uaT.getElementType();
761  auto bEltType = (bT) ? bT.getElementType() : ubT.getElementType();
762  if (aEltType != bEltType)
763  return false;
764 
765  auto aMemSpace = (aT) ? aT.getMemorySpace() : uaT.getMemorySpace();
766  auto bMemSpace = (bT) ? bT.getMemorySpace() : ubT.getMemorySpace();
767  return aMemSpace == bMemSpace;
768  }
769 
770  return false;
771 }

909 /// Given the `originalType` and a `candidateReducedType` whose shape is assumed
910 /// to be a subset of `originalType` with some `1` entries erased, return the
911 /// set of indices that specifies which of the entries of `originalShape` are
912 /// dropped to obtain `reducedShape`.
913 /// This accounts for cases where there are multiple unit-dims, but only a
914 /// subset of those are dropped. For MemRefTypes these can be disambiguated
915 /// using the strides. If a dimension is dropped the stride must be dropped too.
916 static FailureOr<llvm::SmallBitVector>
917 computeMemRefRankReductionMask(MemRefType originalType, MemRefType reducedType,
918  ArrayRef<OpFoldResult> sizes) {
919  llvm::SmallBitVector unusedDims(originalType.getRank());
920  if (originalType.getRank() == reducedType.getRank())
921  return unusedDims;
922 
923  for (const auto &dim : llvm::enumerate(sizes))
924  if (auto attr = llvm::dyn_cast_if_present<Attribute>(dim.value()))
925  if (llvm::cast<IntegerAttr>(attr).getInt() == 1)
926  unusedDims.set(dim.index());
927 
928  // Early exit for the case where the number of unused dims matches the number
929  // of ranks reduced.
930  if (static_cast<int64_t>(unusedDims.count()) + reducedType.getRank() ==
931  originalType.getRank())
932  return unusedDims;
933 
934  SmallVector<int64_t> originalStrides, candidateStrides;
935  int64_t originalOffset, candidateOffset;
936  if (failed(
937  originalType.getStridesAndOffset(originalStrides, originalOffset)) ||
938  failed(
939  reducedType.getStridesAndOffset(candidateStrides, candidateOffset)))
940  return failure();
941 
942  // For memrefs, a dimension is truly dropped if its corresponding stride is
943  // also dropped. This is particularly important when more than one of the dims
944  // is 1. Track the number of occurences of the strides in the original type
945  // and the candidate type. For each unused dim that stride should not be
946  // present in the candidate type. Note that there could be multiple dimensions
947  // that have the same size. We dont need to exactly figure out which dim
948  // corresponds to which stride, we just need to verify that the number of
949  // reptitions of a stride in the original + number of unused dims with that
950  // stride == number of repititions of a stride in the candidate.
951  std::map<int64_t, unsigned> currUnaccountedStrides =
952  getNumOccurences(originalStrides);
953  std::map<int64_t, unsigned> candidateStridesNumOccurences =
954  getNumOccurences(candidateStrides);
955  for (size_t dim = 0, e = unusedDims.size(); dim != e; ++dim) {
956  if (!unusedDims.test(dim))
957  continue;
958  int64_t originalStride = originalStrides[dim];
959  if (currUnaccountedStrides[originalStride] >
960  candidateStridesNumOccurences[originalStride]) {
961  // This dim can be treated as dropped.
962  currUnaccountedStrides[originalStride]--;
963  continue;
964  }
965  if (currUnaccountedStrides[originalStride] ==
966  candidateStridesNumOccurences[originalStride]) {
967  // The stride for this is not dropped. Keep as is.
968  unusedDims.reset(dim);
969  continue;
970  }
971  if (currUnaccountedStrides[originalStride] <
972  candidateStridesNumOccurences[originalStride]) {
973  // This should never happen. Cant have a stride in the reduced rank type
974  // that wasnt in the original one.
975  return failure();
976  }
977  }
978 
979  if ((int64_t)unusedDims.count() + reducedType.getRank() !=
980  originalType.getRank())
981  return failure();
982  return unusedDims;
983 }

994 OpFoldResult DimOp::fold(FoldAdaptor adaptor) {
995  // All forms of folding require a known index.
996  auto index = llvm::dyn_cast_if_present<IntegerAttr>(adaptor.getIndex());
997  if (!index)
998  return {};
999 
1000  // Folding for unranked types (UnrankedMemRefType) is not supported.
1001  auto memrefType = llvm::dyn_cast<MemRefType>(getSource().getType());
1002  if (!memrefType)
1003  return {};
1004 
1005  // Out of bound indices produce undefined behavior but are still valid IR.
1006  // Don't choke on them.
1007  int64_t indexVal = index.getInt();
1008  if (indexVal < 0 || indexVal >= memrefType.getRank())
1009  return {};
1010 
1011  // Fold if the shape extent along the given index is known.
1012  if (!memrefType.isDynamicDim(index.getInt())) {
1013  Builder builder(getContext());
1014  return builder.getIndexAttr(memrefType.getShape()[index.getInt()]);
1015  }
1016 
1017  // The size at the given index is now known to be a dynamic size.
1018  unsigned unsignedIndex = index.getValue().getZExtValue();
1019 
1020  // Fold dim to the size argument for an `AllocOp`, `ViewOp`, or `SubViewOp`.
1021  Operation *definingOp = getSource().getDefiningOp();
1022 
1023  if (auto alloc = dyn_cast_or_null<AllocOp>(definingOp))
1024  return *(alloc.getDynamicSizes().begin() +
1025  memrefType.getDynamicDimIndex(unsignedIndex));
1026 
1027  if (auto alloca = dyn_cast_or_null<AllocaOp>(definingOp))
1028  return *(alloca.getDynamicSizes().begin() +
1029  memrefType.getDynamicDimIndex(unsignedIndex));
1030 
1031  if (auto view = dyn_cast_or_null<ViewOp>(definingOp))
1032  return *(view.getDynamicSizes().begin() +
1033  memrefType.getDynamicDimIndex(unsignedIndex));
1034 
1035  if (auto subview = dyn_cast_or_null<SubViewOp>(definingOp)) {
1036  llvm::SmallBitVector unusedDims = subview.getDroppedDims();
1037  unsigned resultIndex = 0;
1038  unsigned sourceRank = subview.getSourceType().getRank();
1039  unsigned sourceIndex = 0;
1040  for (auto i : llvm::seq<unsigned>(0, sourceRank)) {
1041  if (unusedDims.test(i))
1042  continue;
1043  if (resultIndex == unsignedIndex) {
1044  sourceIndex = i;
1045  break;
1046  }
1047  resultIndex++;
1048  }
1049  assert(subview.isDynamicSize(sourceIndex) &&
1050  "expected dynamic subview size");
1051  return subview.getDynamicSize(sourceIndex);
1052  }
1053 
1054  if (auto sizeInterface =
1055  dyn_cast_or_null<OffsetSizeAndStrideOpInterface>(definingOp)) {
1056  assert(sizeInterface.isDynamicSize(unsignedIndex) &&
1057  "Expected dynamic subview size");
1058  return sizeInterface.getDynamicSize(unsignedIndex);
1059  }
1060 
1061  // dim(memrefcast) -> dim
1062  if (succeeded(foldMemRefCast(*this)))
1063  return getResult();
1064 
1065  return {};
1066 }

1069 /// Fold dim of a memref reshape operation to a load into the reshape's shape
1070 /// operand.
1071 struct DimOfMemRefReshape : public OpRewritePattern<DimOp> {
1072  using OpRewritePattern<DimOp>::OpRewritePattern;
1073 
1074  LogicalResult matchAndRewrite(DimOp dim,
1075  PatternRewriter &rewriter) const override {
1076  auto reshape = dim.getSource().getDefiningOp<ReshapeOp>();
1077 
1078  if (!reshape)
1079  return rewriter.notifyMatchFailure(
1080  dim, "Dim op is not defined by a reshape op.");
1081 
1082  // dim of a memref reshape can be folded if dim.getIndex() dominates the
1083  // reshape. Instead of using `DominanceInfo` (which is usually costly) we
1084  // cheaply check that either of the following conditions hold:
1085  // 1. dim.getIndex() is defined in the same block as reshape but before
1086  // reshape.
1087  // 2. dim.getIndex() is defined in a parent block of
1088  // reshape.
1089 
1090  // Check condition 1
1091  if (dim.getIndex().getParentBlock() == reshape->getBlock()) {
1092  if (auto *definingOp = dim.getIndex().getDefiningOp()) {
1093  if (reshape->isBeforeInBlock(definingOp)) {
1094  return rewriter.notifyMatchFailure(
1095  dim,
1096  "dim.getIndex is not defined before reshape in the same block.");
1097  }
1098  } // else dim.getIndex is a block argument to reshape->getBlock and
1099  // dominates reshape
1100  } // Check condition 2
1101  else if (dim->getBlock() != reshape->getBlock() &&
1102  !dim.getIndex().getParentRegion()->isProperAncestor(
1103  reshape->getParentRegion())) {
1104  // If dim and reshape are in the same block but dim.getIndex() isn't, we
1105  // already know dim.getIndex() dominates reshape without calling
1106  // `isProperAncestor`
1107  return rewriter.notifyMatchFailure(
1108  dim, "dim.getIndex does not dominate reshape.");
1109  }
1110 
1111  // Place the load directly after the reshape to ensure that the shape memref
1112  // was not mutated.
1113  rewriter.setInsertionPointAfter(reshape);
1114  Location loc = dim.getLoc();
1115  Value load =
1116  LoadOp::create(rewriter, loc, reshape.getShape(), dim.getIndex());
1117  if (load.getType() != dim.getType())
1118  load = arith::IndexCastOp::create(rewriter, loc, dim.getType(), load);
1119  rewriter.replaceOp(dim, load);
1120  return success();
1121  }
1122 };

1163 // Ex:
1164 // %dma_id = dma_start %src[%i, %j], %dst[%k, %l], %size,
1165 // %tag[%index], %stride, %num_elt_per_stride :
1166 // : memref<3076 x f32, 0>,
1167 // memref<1024 x f32, 2>,
1168 // memref<1 x i32>
1169 //
1170 ParseResult DmaStartOp::parse(OpAsmParser &parser, OperationState &result) {
1171  OpAsmParser::UnresolvedOperand srcMemRefInfo;
1172  SmallVector<OpAsmParser::UnresolvedOperand, 4> srcIndexInfos;
1173  OpAsmParser::UnresolvedOperand dstMemRefInfo;
1174  SmallVector<OpAsmParser::UnresolvedOperand, 4> dstIndexInfos;
1175  OpAsmParser::UnresolvedOperand numElementsInfo;
1176  OpAsmParser::UnresolvedOperand tagMemrefInfo;
1177  SmallVector<OpAsmParser::UnresolvedOperand, 4> tagIndexInfos;
1178  SmallVector<OpAsmParser::UnresolvedOperand, 2> strideInfo;
1179 
1180  SmallVector<Type, 3> types;
1181  auto indexType = parser.getBuilder().getIndexType();
1182 
1183  // Parse and resolve the following list of operands:
1184  // *) source memref followed by its indices (in square brackets).
1185  // *) destination memref followed by its indices (in square brackets).
1186  // *) dma size in KiB.
1187  if (parser.parseOperand(srcMemRefInfo) ||
1188  parser.parseOperandList(srcIndexInfos, OpAsmParser::Delimiter::Square) ||
1189  parser.parseComma() || parser.parseOperand(dstMemRefInfo) ||
1190  parser.parseOperandList(dstIndexInfos, OpAsmParser::Delimiter::Square) ||
1191  parser.parseComma() || parser.parseOperand(numElementsInfo) ||
1192  parser.parseComma() || parser.parseOperand(tagMemrefInfo) ||
1193  parser.parseOperandList(tagIndexInfos, OpAsmParser::Delimiter::Square))
1194  return failure();
1195 
1196  // Parse optional stride and elements per stride.
1197  if (parser.parseTrailingOperandList(strideInfo))
1198  return failure();
1199 
1200  bool isStrided = strideInfo.size() == 2;
1201  if (!strideInfo.empty() && !isStrided) {
1202  return parser.emitError(parser.getNameLoc(),
1203  "expected two stride related operands");
1204  }
1205 
1206  if (parser.parseColonTypeList(types))
1207  return failure();
1208  if (types.size() != 3)
1209  return parser.emitError(parser.getNameLoc(), "fewer/more types expected");
1210 
1211  if (parser.resolveOperand(srcMemRefInfo, types[0], result.operands) ||
1212  parser.resolveOperands(srcIndexInfos, indexType, result.operands) ||
1213  parser.resolveOperand(dstMemRefInfo, types[1], result.operands) ||

1230 LogicalResult DmaStartOp::verify() {

1247  !llvm::all_of(getSrcIndices().getTypes(),
1248  [](Type t) { return t.isIndex(); }))

1250

1252  if (!llvm::isa<MemRefType>(getDstMemRef().getType()))
1253  return emitOpError("expected destination to be of memref type");

1259  !llvm::all_of(getDstIndices().getTypes(),

1265  return emitOpError("expected num elements to be of index type");
1266

1268  if (!llvm::isa<MemRefType>(getTagMemRef().getType()))
1269  return emitOpError("expected tag to be of memref type");

1271  if (numOperands < numExpectedOperands)
1272  return emitOpError() << "expected at least " << numExpectedOperands

1276  [](Type t) { return t.isIndex(); }))
1277  return emitOpError("expected tag indices to be of index type");

1373 /// Helper function to perform the replacement of all constant uses of `values`
1374 /// by a materialized constant extracted from `maybeConstants`.
1375 /// `values` and `maybeConstants` are expected to have the same size.
1376 template <typename Container>
1377 static bool replaceConstantUsesOf(OpBuilder &rewriter, Location loc,
1378  Container values,
1379  ArrayRef<OpFoldResult> maybeConstants) {
1380  assert(values.size() == maybeConstants.size() &&
1381  " expected values and maybeConstants of the same size");
1382  bool atLeastOneReplacement = false;
1383  for (auto [maybeConstant, result] : llvm::zip(maybeConstants, values)) {
1384  // Don't materialize a constant if there are no uses: this would indice
1385  // infinite loops in the driver.
1386  if (result.use_empty() || maybeConstant == getAsOpFoldResult(result))
1387  continue;
1388  assert(isa<Attribute>(maybeConstant) &&
1389  "The constified value should be either unchanged (i.e., == result) "
1390  "or a constant");
1391  Value constantVal = arith::ConstantIndexOp::create(
1392  rewriter, loc,
1393  llvm::cast<IntegerAttr>(cast<Attribute>(maybeConstant)).getInt());
1394  for (Operation *op : llvm::make_early_inc_range(result.getUsers())) {
1395  // modifyOpInPlace: lambda cannot capture structured bindings in C++17
1396  // yet.
1397  op->replaceUsesOfWith(result, constantVal);
1398  atLeastOneReplacement = true;
1399  }
1400  }
1401  return atLeastOneReplacement;
1402 }

2026 /// Replace the sequence:
2027 /// ```
2028 /// base, offset, sizes, strides = extract_strided_metadata src
2029 /// dst = reinterpret_cast base to offset, sizes, strides
2030 /// ```
2031 /// With
2032 ///
2033 /// ```
2034 /// dst = memref.cast src
2035 /// ```
2036 ///
2037 /// Note: The cast operation is only inserted when the type of dst and src
2038 /// are not the same. E.g., when going from <4xf32> to <?xf32>.
2039 ///
2040 /// This pattern also matches when the offset, sizes, and strides don't come
2041 /// directly from the `extract_strided_metadata`'s results but it can be
2042 /// statically proven that they would hold the same values.
2043 ///
2044 /// For instance, the following sequence would be replaced:
2045 /// ```
2046 /// base, offset, sizes, strides =
2047 /// extract_strided_metadata memref : memref<3x4xty>
2048 /// dst = reinterpret_cast base to 0, [3, 4], strides
2049 /// ```
2050 /// Because we know (thanks to the type of the input memref) that variable
2051 /// `offset` and `sizes` will respectively hold 0 and [3, 4].
2052 ///
2053 /// Similarly, the following sequence would be replaced:
2054 /// ```
2055 /// c0 = arith.constant 0
2056 /// c4 = arith.constant 4
2057 /// base, offset, sizes, strides =
2058 /// extract_strided_metadata memref : memref<3x4xty>
2059 /// dst = reinterpret_cast base to c0, [3, c4], strides
2060 /// ```
2061 /// Because we know that `offset`and `c0` will hold 0
2062 /// and `c4` will hold 4.
2063 ///
2064 /// If the pattern above does not match, the input of the
2065 /// extract_strided_metadata is always folded into the input of the
2066 /// reinterpret_cast operator. This allows for dead code elimination to get rid
2067 /// of the extract_strided_metadata in some cases.
2068 struct ReinterpretCastOpExtractStridedMetadataFolder
2069  : public OpRewritePattern<ReinterpretCastOp> {
2070 public:
2071  using OpRewritePattern<ReinterpretCastOp>::OpRewritePattern;
2072 
2073  LogicalResult matchAndRewrite(ReinterpretCastOp op,
2074  PatternRewriter &rewriter) const override {
2075  auto extractStridedMetadata =
2076  op.getSource().getDefiningOp<ExtractStridedMetadataOp>();
2077  if (!extractStridedMetadata)
2078  return failure();
2079 
2080  // Check if the reinterpret cast reconstructs a memref with the exact same
2081  // properties as the extract strided metadata.
2082  auto isReinterpretCastNoop = [&]() -> bool {
2083  // First, check that the strides are the same.
2084  if (!llvm::equal(extractStridedMetadata.getConstifiedMixedStrides(),
2085  op.getConstifiedMixedStrides()))
2086  return false;
2087 
2088  // Second, check the sizes.
2089  if (!llvm::equal(extractStridedMetadata.getConstifiedMixedSizes(),
2090  op.getConstifiedMixedSizes()))
2091  return false;
2092 
2093  // Finally, check the offset.
2094  assert(op.getMixedOffsets().size() == 1 &&
2095  "reinterpret_cast with more than one offset should have been "
2096  "rejected by the verifier");
2097  return extractStridedMetadata.getConstifiedMixedOffset() ==
2098  op.getConstifiedMixedOffset();
2099  };
2100 
2101  if (!isReinterpretCastNoop()) {
2102  // If the extract_strided_metadata / reinterpret_cast pair can't be
2103  // completely folded, then we could fold the input of the
2104  // extract_strided_metadata into the input of the reinterpret_cast
2105  // input. For some cases (e.g., static dimensions) the
2106  // the extract_strided_metadata is eliminated by dead code elimination.
2107  //
2108  // reinterpret_cast(extract_strided_metadata(x)) -> reinterpret_cast(x).
2109  //
2110  // We can always fold the input of a extract_strided_metadata operator
2111  // to the input of a reinterpret_cast operator, because they point to
2112  // the same memory. Note that the reinterpret_cast does not use the
2113  // layout of its input memref, only its base memory pointer which is
2114  // the same as the base pointer returned by the extract_strided_metadata
2115  // operator and the base pointer of the extract_strided_metadata memref
2116  // input.
2117  rewriter.modifyOpInPlace(op, [&]() {
2118  op.getSourceMutable().assign(extractStridedMetadata.getSource());
2119  });
2120  return success();
2121  }
2122 
2123  // At this point, we know that the back and forth between extract strided
2124  // metadata and reinterpret cast is a noop. However, the final type of the
2125  // reinterpret cast may not be exactly the same as the original memref.
2126  // E.g., it could be changing a dimension from static to dynamic. Check that
2127  // here and add a cast if necessary.
2128  Type srcTy = extractStridedMetadata.getSource().getType();
2129  if (srcTy == op.getResult().getType())
2130  rewriter.replaceOp(op, extractStridedMetadata.getSource());
2131  else
2132  rewriter.replaceOpWithNewOp<CastOp>(op, op.getType(),
2133  extractStridedMetadata.getSource());
2134 
2135  return success();
2136  }
2137 };

2267 /// Compute the layout map after expanding a given source MemRef type with the
2268 /// specified reassociation indices.
2269 static FailureOr<StridedLayoutAttr>
2270 computeExpandedLayoutMap(MemRefType srcType, ArrayRef<int64_t> resultShape,
2271  ArrayRef<ReassociationIndices> reassociation) {
2272  int64_t srcOffset;
2273  SmallVector<int64_t> srcStrides;
2274  if (failed(srcType.getStridesAndOffset(srcStrides, srcOffset)))
2275  return failure();
2276  assert(srcStrides.size() == reassociation.size() && "invalid reassociation");
2277 
2278  // 1-1 mapping between srcStrides and reassociation packs.
2279  // Each srcStride starts with the given value and gets expanded according to
2280  // the proper entries in resultShape.
2281  // Example:
2282  // srcStrides = [10000, 1 , 100 ],
2283  // reassociations = [ [0], [1], [2, 3, 4]],
2284  // resultSizes = [2, 5, 4, 3, 2] = [ [2], [5], [4, 3, 2]]
2285  // -> For the purpose of stride calculation, the useful sizes are:
2286  // [x, x, x, 3, 2] = [ [x], [x], [x, 3, 2]].
2287  // resultStrides = [10000, 1, 600, 200, 100]
2288  // Note that a stride does not get expanded along the first entry of each
2289  // shape pack.
2290  SmallVector<int64_t> reverseResultStrides;
2291  reverseResultStrides.reserve(resultShape.size());
2292  unsigned shapeIndex = resultShape.size() - 1;
2293  for (auto it : llvm::reverse(llvm::zip(reassociation, srcStrides))) {
2294  ReassociationIndices reassoc = std::get<0>(it);
2295  int64_t currentStrideToExpand = std::get<1>(it);
2296  for (unsigned idx = 0, e = reassoc.size(); idx < e; ++idx) {
2297  reverseResultStrides.push_back(currentStrideToExpand);
2298  currentStrideToExpand =
2299  (SaturatedInteger::wrap(currentStrideToExpand) *
2300  SaturatedInteger::wrap(resultShape[shapeIndex--]))
2301  .asInteger();
2302  }
2303  }
2304  auto resultStrides = llvm::to_vector<8>(llvm::reverse(reverseResultStrides));
2305  resultStrides.resize(resultShape.size(), 1);
2306  return StridedLayoutAttr::get(srcType.getContext(), srcOffset, resultStrides);
2307 }

2462 /// Compute the layout map after collapsing a given source MemRef type with the
2463 /// specified reassociation indices.
2464 ///
2465 /// Note: All collapsed dims in a reassociation group must be contiguous. It is
2466 /// not possible to check this by inspecting a MemRefType in the general case.
2467 /// If non-contiguity cannot be checked statically, the collapse is assumed to
2468 /// be valid (and thus accepted by this function) unless `strict = true`.
2469 static FailureOr<StridedLayoutAttr>
2470 computeCollapsedLayoutMap(MemRefType srcType,
2471  ArrayRef<ReassociationIndices> reassociation,
2472  bool strict = false) {
2473  int64_t srcOffset;
2474  SmallVector<int64_t> srcStrides;
2475  auto srcShape = srcType.getShape();
2476  if (failed(srcType.getStridesAndOffset(srcStrides, srcOffset)))
2477  return failure();
2478 
2479  // The result stride of a reassociation group is the stride of the last entry
2480  // of the reassociation. (TODO: Should be the minimum stride in the
2481  // reassociation because strides are not necessarily sorted. E.g., when using
2482  // memref.transpose.) Dimensions of size 1 should be skipped, because their
2483  // strides are meaningless and could have any arbitrary value.
2484  SmallVector<int64_t> resultStrides;
2485  resultStrides.reserve(reassociation.size());
2486  for (const ReassociationIndices &reassoc : reassociation) {
2487  ArrayRef<int64_t> ref = llvm::ArrayRef(reassoc);
2488  while (srcShape[ref.back()] == 1 && ref.size() > 1)
2489  ref = ref.drop_back();
2490  if (ShapedType::isStatic(srcShape[ref.back()]) || ref.size() == 1) {
2491  resultStrides.push_back(srcStrides[ref.back()]);
2492  } else {
2493  // Dynamically-sized dims may turn out to be dims of size 1 at runtime, so
2494  // the corresponding stride may have to be skipped. (See above comment.)
2495  // Therefore, the result stride cannot be statically determined and must
2496  // be dynamic.
2497  resultStrides.push_back(ShapedType::kDynamic);
2498  }
2499  }
2500 
2501  // Validate that each reassociation group is contiguous.
2502  unsigned resultStrideIndex = resultStrides.size() - 1;
2503  for (const ReassociationIndices &reassoc : llvm::reverse(reassociation)) {
2504  auto trailingReassocs = ArrayRef<int64_t>(reassoc).drop_front();
2505  auto stride = SaturatedInteger::wrap(resultStrides[resultStrideIndex--]);
2506  for (int64_t idx : llvm::reverse(trailingReassocs)) {
2507  stride = stride * SaturatedInteger::wrap(srcShape[idx]);
2508 
2509  // Both source and result stride must have the same static value. In that
2510  // case, we can be sure, that the dimensions are collapsible (because they
2511  // are contiguous).
2512  // If `strict = false` (default during op verification), we accept cases
2513  // where one or both strides are dynamic. This is best effort: We reject
2514  // ops where obviously non-contiguous dims are collapsed, but accept ops
2515  // where we cannot be sure statically. Such ops may fail at runtime. See
2516  // the op documentation for details.
2517  auto srcStride = SaturatedInteger::wrap(srcStrides[idx - 1]);
2518  if (strict && (stride.saturated || srcStride.saturated))
2519  return failure();
2520 
2521  // Dimensions of size 1 should be skipped, because their strides are
2522  // meaningless and could have any arbitrary value.
2523  if (srcShape[idx - 1] == 1)
2524  continue;
2525 
2526  if (!stride.saturated && !srcStride.saturated && stride != srcStride)
2527  return failure();
2528  }
2529  }
2530  return StridedLayoutAttr::get(srcType.getContext(), srcOffset, resultStrides);
2531 }

2766 /// static representation of offsets, sizes and strides. Special sentinels
2767 /// encode the dynamic case.
2768 MemRefType SubViewOp::inferResultType(MemRefType sourceMemRefType,
2769  ArrayRef<int64_t> staticOffsets,
2770  ArrayRef<int64_t> staticSizes,
2771  ArrayRef<int64_t> staticStrides) {
2772  unsigned rank = sourceMemRefType.getRank();
2773  (void)rank;
2774  assert(staticOffsets.size() == rank && "staticOffsets length mismatch");
2775  assert(staticSizes.size() == rank && "staticSizes length mismatch");
2776  assert(staticStrides.size() == rank && "staticStrides length mismatch");
2777 
2778  // Extract source offset and strides.
2779  auto [sourceStrides, sourceOffset] = sourceMemRefType.getStridesAndOffset();

2782  // `sourceOffset + sum_i(staticOffset_i * sourceStrides_i)`.
2783  int64_t targetOffset = sourceOffset;
2784  for (auto it : llvm::zip(staticOffsets, sourceStrides)) {
2785  auto staticOffset = std::get<0>(it), sourceStride = std::get<1>(it);
2786  targetOffset = (SaturatedInteger::wrap(targetOffset) +
2787  SaturatedInteger::wrap(staticOffset) *
2788  SaturatedInteger::wrap(sourceStride))
2789  .asInteger();
2790  }
2791 
2792  // Compute target stride whose value is:
2793  // `sourceStrides_i * staticStrides_i`.
2794  SmallVector<int64_t, 4> targetStrides;
2795  targetStrides.reserve(staticOffsets.size());

2797  auto sourceStride = std::get<0>(it), staticStride = std::get<1>(it);
2798  targetStrides.push_back((SaturatedInteger::wrap(sourceStride) *
2799  SaturatedInteger::wrap(staticStride))
2800  .asInteger());
2801  }
2802 
2803  // The type is now known.
2804  return MemRefType::get(staticSizes, sourceMemRefType.getElementType(),
2805  StridedLayoutAttr::get(sourceMemRefType.getContext(),
2806  targetOffset, targetStrides),
2807  sourceMemRefType.getMemorySpace());
