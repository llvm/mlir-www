64/// Returns a placeholder string for use as an acc.var_name attribute value when

128/// Emit an OpenACC remark with lazy message generation.
129///
130/// The messageFn is only invoked if remarks are enabled, allowing callers
131/// to avoid constructing expensive messages when remarks are disabled.
132///
133/// \param op The operation to emit the remark for.
134/// \param messageFn A callable that returns the remark message.
135/// \param category Optional category for the remark. Defaults to "openacc".
136/// \return An in-flight remark object that can be used to append
137/// additional information to the remark.
138remark::detail::InFlightRemark
139emitRemark(mlir::Operation *op, const std::function<std::string()> &messageFn,
140 llvm::StringRef category = "openacc");
141
142/// Emit an OpenACC remark for the given operation with the given message.
143///
144/// \param op The operation to emit the remark for.
145/// \param message The remark message.
146/// \param category Optional category for the remark. Defaults to "openacc".
147/// \return An in-flight remark object that can be used to append
148/// additional information to the remark.
149inline remark::detail::InFlightRemark
150emitRemark(mlir::Operation *op, const llvm::Twine &message,
151 llvm::StringRef category = "openacc") {
152 return emitRemark(
153 op, std::function<std::string()>([msg = message.str()]() { return msg; }),
154 category);
155}
