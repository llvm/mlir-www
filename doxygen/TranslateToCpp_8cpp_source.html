34 /// Convenience functions to produce interleaved output with functions returning
35 /// a LogicalResult. This is different than those in STLExtras as functions used
36 /// on each element doesn't return a string.
37 template <typename ForwardIterator, typename UnaryFunctor,
38  typename NullaryFunctor>
39 static inline LogicalResult
40 interleaveWithError(ForwardIterator begin, ForwardIterator end,
41  UnaryFunctor eachFn, NullaryFunctor betweenFn) {
42  if (begin == end)
43  return success();
44  if (failed(eachFn(*begin)))
45  return failure();
46  ++begin;
47  for (; begin != end; ++begin) {
48  betweenFn();
49  if (failed(eachFn(*begin)))
50  return failure();
51  }
52  return success();
53 }

69 /// Return the precedence of a operator as an integer, higher values
70 /// imply higher precedence.
71 static FailureOr<int> getOperatorPrecedence(Operation *operation) {
72  return llvm::TypeSwitch<Operation *, FailureOr<int>>(operation)
73  .Case<emitc::AddOp>([&](auto op) { return 12; })
74  .Case<emitc::ApplyOp>([&](auto op) { return 15; })
75  .Case<emitc::BitwiseAndOp>([&](auto op) { return 7; })
76  .Case<emitc::BitwiseLeftShiftOp>([&](auto op) { return 11; })
77  .Case<emitc::BitwiseNotOp>([&](auto op) { return 15; })
78  .Case<emitc::BitwiseOrOp>([&](auto op) { return 5; })
79  .Case<emitc::BitwiseRightShiftOp>([&](auto op) { return 11; })
80  .Case<emitc::BitwiseXorOp>([&](auto op) { return 6; })
81  .Case<emitc::CallOp>([&](auto op) { return 16; })
82  .Case<emitc::CallOpaqueOp>([&](auto op) { return 16; })
83  .Case<emitc::CastOp>([&](auto op) { return 15; })
84  .Case<emitc::CmpOp>([&](auto op) -> FailureOr<int> {
85  switch (op.getPredicate()) {
86  case emitc::CmpPredicate::eq:
87  case emitc::CmpPredicate::ne:
88  return 8;
89  case emitc::CmpPredicate::lt:
90  case emitc::CmpPredicate::le:
91  case emitc::CmpPredicate::gt:
92  case emitc::CmpPredicate::ge:
93  return 9;
94  case emitc::CmpPredicate::three_way:
95  return 10;
96  }
97  return op->emitError("unsupported cmp predicate");
98  })
99  .Case<emitc::ConditionalOp>([&](auto op) { return 2; })
100  .Case<emitc::ConstantOp>([&](auto op) { return 17; })
101  .Case<emitc::DivOp>([&](auto op) { return 13; })
102  .Case<emitc::LoadOp>([&](auto op) { return 16; })
103  .Case<emitc::LogicalAndOp>([&](auto op) { return 4; })
104  .Case<emitc::LogicalNotOp>([&](auto op) { return 15; })
105  .Case<emitc::LogicalOrOp>([&](auto op) { return 3; })
106  .Case<emitc::MulOp>([&](auto op) { return 13; })
107  .Case<emitc::RemOp>([&](auto op) { return 13; })
108  .Case<emitc::SubOp>([&](auto op) { return 12; })
109  .Case<emitc::UnaryMinusOp>([&](auto op) { return 15; })
110  .Case<emitc::UnaryPlusOp>([&](auto op) { return 15; })
111  .Default([](auto op) { return op->emitError("unsupported operation"); });
112 }

348 /// Determine whether expression \p expressionOp should be emitted inline, i.e.
349 /// as part of its user. This function recommends inlining of any expressions
350 /// that can be inlined unless it is used by another expression, under the
351 /// assumption that any expression fusion/re-materialization was taken care of
352 /// by transformations run by the backend.
353 static bool shouldBeInlined(ExpressionOp expressionOp) {
354  // Do not inline if expression is marked as such.
355  if (expressionOp.getDoNotInline())
356  return false;
357 
358  // Do not inline expressions with side effects to prevent side-effect
359  // reordering.
360  if (expressionOp.hasSideEffects())
361  return false;
362 
363  // Do not inline expressions with multiple uses.
364  Value result = expressionOp.getResult();
365  if (!result.hasOneUse())
366  return false;
367 
368  Operation *user = *result.getUsers().begin();
369 
370  // Do not inline expressions used by operations with deferred emission, since
371  // their translation requires the materialization of variables.
372  if (hasDeferredEmission(user))
373  return false;
374 
375  // Do not inline expressions used by other expressions or by ops with the
376  // CExpressionInterface. If this was intended, the user could have been merged
377  // into the expression op.
378  return !isa<emitc::ExpressionOp, emitc::CExpressionInterface>(*user);
379 }

381 static LogicalResult printConstantOp(CppEmitter &emitter, Operation *operation,
382  Attribute value) {
383  OpResult result = operation->getResult(0);
384 
385  // Only emit an assignment as the variable was already declared when printing
386  // the FuncOp.
387  if (emitter.shouldDeclareVariablesAtTop()) {
388  // Skip the assignment if the emitc.constant has no value.
389  if (auto oAttr = dyn_cast<emitc::OpaqueAttr>(value)) {
390  if (oAttr.getValue().empty())
391  return success();
392  }
393 
394  if (failed(emitter.emitVariableAssignment(result)))
395  return failure();
396  return emitter.emitAttribute(operation->getLoc(), value);
397  }
398 
399  // Emit a variable declaration for an emitc.constant op without value.
400  if (auto oAttr = dyn_cast<emitc::OpaqueAttr>(value)) {
401  if (oAttr.getValue().empty())
402  // The semicolon gets printed by the emitOperation function.
403  return emitter.emitVariableDeclaration(result,
404  /*trailingSemicolon=*/false);
405  }
406 
407  // Emit a variable declaration.
408  if (failed(emitter.emitAssignPrefix(*operation)))
409  return failure();
410  return emitter.emitAttribute(operation->getLoc(), value);
411 }

1517 LogicalResult CppEmitter::emitOperand(Value value) {
1518  if (isPartOfCurrentExpression(value)) {
1519  Operation *def = value.getDefiningOp();
1520  assert(def && "Expected operand to be defined by an operation");
1521  FailureOr<int> precedence = getOperatorPrecedence(def);
1522  if (failed(precedence))
1523  return failure();
1524 
1525  // Sub-expressions with equal or lower precedence need to be parenthesized,
1526  // as they might be evaluated in the wrong order depending on the shape of
1527  // the expression tree.
1528  bool encloseInParenthesis = precedence.value() <= getExpressionPrecedence();
1529  if (encloseInParenthesis)
1530  os << "(";
1531  pushExpressionPrecedence(precedence.value());
1532 
1533  if (failed(emitOperation(*def, /*trailingSemicolon=*/false)))
1534  return failure();
1535 
1536  if (encloseInParenthesis)
1537  os << ")";
1538 
1539  popExpressionPrecedence();
1540  return success();
1541  }
1542 
1543  auto expressionOp = value.getDefiningOp<ExpressionOp>();
1544  if (expressionOp && shouldBeInlined(expressionOp))
1545  return emitExpression(expressionOp);
1546 
1547  if (BlockArgument arg = dyn_cast<BlockArgument>(value)) {
1548  // If this operand is a block argument of an expression, emit instead the
1549  // matching expression parameter.
1550  Operation *argOp = arg.getParentBlock()->getParentOp();
1551  if (auto expressionOp = dyn_cast<ExpressionOp>(argOp)) {
1552  // This scenario is only expected when one of the operations within the
1553  // expression being emitted references one of the expression's block
1554  // arguments.
1555  assert(expressionOp == emittedExpression &&
1556  "Expected expression being emitted");
1557  value = expressionOp->getOperand(arg.getArgNumber());
1558  }
1559  }
1560 
1561  os << getOrCreateName(value);
1562  return success();
1563 }
