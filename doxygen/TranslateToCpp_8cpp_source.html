33 /// Convenience functions to produce interleaved output with functions returning
34 /// a LogicalResult. This is different than those in STLExtras as functions used
35 /// on each element doesn't return a string.
36 template <typename ForwardIterator, typename UnaryFunctor,
37  typename NullaryFunctor>
38 inline LogicalResult
39 interleaveWithError(ForwardIterator begin, ForwardIterator end,
40  UnaryFunctor eachFn, NullaryFunctor betweenFn) {
41  if (begin == end)
42  return success();
43  if (failed(eachFn(*begin)))
44  return failure();
45  ++begin;
46  for (; begin != end; ++begin) {
47  betweenFn();
48  if (failed(eachFn(*begin)))
49  return failure();
50  }
51  return success();
52 }

68 /// Return the precedence of a operator as an integer, higher values
69 /// imply higher precedence.
70 static FailureOr<int> getOperatorPrecedence(Operation *operation) {
71  return llvm::TypeSwitch<Operation *, FailureOr<int>>(operation)
72  .Case<emitc::AddOp>([&](auto op) { return 12; })
73  .Case<emitc::ApplyOp>([&](auto op) { return 15; })
74  .Case<emitc::BitwiseAndOp>([&](auto op) { return 7; })
75  .Case<emitc::BitwiseLeftShiftOp>([&](auto op) { return 11; })
76  .Case<emitc::BitwiseNotOp>([&](auto op) { return 15; })
77  .Case<emitc::BitwiseOrOp>([&](auto op) { return 5; })
78  .Case<emitc::BitwiseRightShiftOp>([&](auto op) { return 11; })
79  .Case<emitc::BitwiseXorOp>([&](auto op) { return 6; })
80  .Case<emitc::CallOp>([&](auto op) { return 16; })
81  .Case<emitc::CallOpaqueOp>([&](auto op) { return 16; })
82  .Case<emitc::CastOp>([&](auto op) { return 15; })
83  .Case<emitc::CmpOp>([&](auto op) -> FailureOr<int> {
84  switch (op.getPredicate()) {
85  case emitc::CmpPredicate::eq:
86  case emitc::CmpPredicate::ne:
87  return 8;
88  case emitc::CmpPredicate::lt:
89  case emitc::CmpPredicate::le:
90  case emitc::CmpPredicate::gt:
91  case emitc::CmpPredicate::ge:
92  return 9;
93  case emitc::CmpPredicate::three_way:
94  return 10;
95  }
96  return op->emitError("unsupported cmp predicate");
97  })
98  .Case<emitc::ConditionalOp>([&](auto op) { return 2; })
99  .Case<emitc::DivOp>([&](auto op) { return 13; })
100  .Case<emitc::LoadOp>([&](auto op) { return 16; })
101  .Case<emitc::LogicalAndOp>([&](auto op) { return 4; })
102  .Case<emitc::LogicalNotOp>([&](auto op) { return 15; })
103  .Case<emitc::LogicalOrOp>([&](auto op) { return 3; })
104  .Case<emitc::MulOp>([&](auto op) { return 13; })
105  .Case<emitc::RemOp>([&](auto op) { return 13; })
106  .Case<emitc::SubOp>([&](auto op) { return 12; })
107  .Case<emitc::UnaryMinusOp>([&](auto op) { return 15; })
108  .Case<emitc::UnaryPlusOp>([&](auto op) { return 15; })
109  .Default([](auto op) { return op->emitError("unsupported operation"); });
110 }

339 /// Determine whether expression \p expressionOp should be emitted inline, i.e.
340 /// as part of its user. This function recommends inlining of any expressions
341 /// that can be inlined unless it is used by another expression, under the
342 /// assumption that any expression fusion/re-materialization was taken care of
343 /// by transformations run by the backend.
344 static bool shouldBeInlined(ExpressionOp expressionOp) {
345  // Do not inline if expression is marked as such.
346  if (expressionOp.getDoNotInline())
347  return false;
348 
349  // Do not inline expressions with side effects to prevent side-effect
350  // reordering.
351  if (expressionOp.hasSideEffects())
352  return false;
353 
354  // Do not inline expressions with multiple uses.
355  Value result = expressionOp.getResult();
356  if (!result.hasOneUse())
357  return false;
358 
359  Operation *user = *result.getUsers().begin();
360 
361  // Do not inline expressions used by operations with deferred emission, since
362  // their translation requires the materialization of variables.
363  if (hasDeferredEmission(user))
364  return false;
365 
366  // Do not inline expressions used by ops with the CExpressionInterface. If
367  // this was intended, the user could have been merged into the expression op.
368  return !isa<emitc::CExpressionInterface>(*user);
369 }

371 static LogicalResult printConstantOp(CppEmitter &emitter, Operation *operation,
372  Attribute value) {
373  OpResult result = operation->getResult(0);
374 
375  // Only emit an assignment as the variable was already declared when printing
376  // the FuncOp.
377  if (emitter.shouldDeclareVariablesAtTop()) {
378  // Skip the assignment if the emitc.constant has no value.
379  if (auto oAttr = dyn_cast<emitc::OpaqueAttr>(value)) {
380  if (oAttr.getValue().empty())
381  return success();
382  }
383 
384  if (failed(emitter.emitVariableAssignment(result)))
385  return failure();
386  return emitter.emitAttribute(operation->getLoc(), value);
387  }
388 
389  // Emit a variable declaration for an emitc.constant op without value.
390  if (auto oAttr = dyn_cast<emitc::OpaqueAttr>(value)) {
391  if (oAttr.getValue().empty())
392  // The semicolon gets printed by the emitOperation function.
393  return emitter.emitVariableDeclaration(result,
394  /*trailingSemicolon=*/false);
395  }
396 
397  // Emit a variable declaration.
398  if (failed(emitter.emitAssignPrefix(*operation)))
399  return failure();
400  return emitter.emitAttribute(operation->getLoc(), value);
401 }

1519  if (isPartOfCurrentExpression(value)) {

1527  // as they might be evaluated in the wrong order depending on the shape of
1528  // the expression tree.

1530  if (encloseInParenthesis)

1532  pushExpressionPrecedence(precedence.value());
1533 
1534  if (failed(emitOperation(*def, /*trailingSemicolon=*/false)))
1535  return failure();
1536 
1537  if (encloseInParenthesis)
1538  os << ")";
1539 
1540  popExpressionPrecedence();
1541  return success();
1542  }
1543 
1544  auto expressionOp = value.getDefiningOp<ExpressionOp>();
1545  if (expressionOp && shouldBeInlined(expressionOp))
1546  return emitExpression(expressionOp);
1547 
1548  os << getOrCreateName(value);
1549  return success();
1550 }
