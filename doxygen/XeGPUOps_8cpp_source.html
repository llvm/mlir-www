88 LogicalResult CreateNdDescOp::verify() {
89  auto rank = (int64_t)getMixedOffsets().size();
90  bool invalidRank = (rank != 2);
91  bool invalidElemTy = false;
92 
93  // check source type matches the rank if it is a memref.
94  // It also should have the same ElementType as TensorDesc.
95  auto memrefTy = getSourceType().dyn_cast<MemRefType>();
96  if (memrefTy) {
97  invalidRank |= (memrefTy.getRank() != rank);
98  invalidElemTy |= memrefTy.getElementType() != getElementType();
99  }
100 
101  // check result type matches the rank
102  invalidRank = (getType().getRank() != rank);
103 
104  // mismatches among shape, strides, and offsets are
105  // already handeled by OffsetSizeAndStrideOpInterface.
106  // So they are not check here.
107  if (invalidRank)
108  return emitOpError(
109  "Expecting the rank of shape, strides, offsets, "
110  "source memref type (if source is a memref) and TensorDesc "
111  "should match with each other. They currenlty are 2D.");
112 
113  if (invalidElemTy)
114  return emitOpError("TensorDesc should have the same element "
115  "type with the source if it is a memref.\n");
116 
117  return success();
118 }

123 LogicalResult LoadNdOp::verify() {
124  auto tdescTy = getTensorDescType();
125  auto valueTy = getType();
126 
127  if (tdescTy.getRank() != 2)
128  return emitOpError(
129  "The TensorDesc for LoadNdOp should be a 2D TensorDesc.");
130 
131  if (!valueTy)
132  return emitOpError("Invalid result, it should be a VectorType.\n");
133 
134  auto tdescElemTy = tdescTy.getElementType();
135  auto valueElemTy = valueTy.getElementType();
136 
137  if (tdescElemTy != valueElemTy)
138  return emitOpError(
139  "Value should have the same element type as TensorDesc.");
140 
141  auto array_len = tdescTy.getArrayLength();
142  auto tdescShape = tdescTy.getShape().vec();
143  auto valueShape = valueTy.getShape().vec();
144 
145  if (getTranspose()) {
146  auto trans = getTranspose().value();
147  if (tdescShape.size() >= trans.size())
148  transpose(trans, tdescShape);
149  else
150  emitWarning("Invalid transpose attr. It is ignored.");
151  }
152 
153  if (getVnniAxis()) {
154  auto axis = getVnniAxis().value();
155  auto vnni_factor = valueShape.back();
156  tdescShape[axis] /= vnni_factor;
157  tdescShape.push_back(vnni_factor);
158  }
159 
160  if (array_len > 1) {
161  auto it = tdescShape.begin();
162  tdescShape.insert(it, array_len);
163  }
164 
165  if (tdescShape != valueShape)
166  return emitOpError() << "Result shape doesn't match TensorDesc shape."
167  << "The expected shape is " << makeString(tdescShape)
168  << ". But the given shape is "
169  << makeString(valueShape) << ".\n";
170  return success();
171 }
