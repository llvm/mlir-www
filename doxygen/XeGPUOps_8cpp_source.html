25 static std::string makeString(T array, bool breakline = false) {
26  std::string buf;
27  buf.clear();
28  llvm::raw_string_ostream os(buf);
29  os << "[";
30  for (size_t i = 1; i < array.size(); i++) {
31  os << array[i - 1] << ", ";
32  if (breakline)
33  os << "\n\t\t";
34  }
35  os << array.back() << "]";
36  return buf;

71 static LogicalResult
72 isValidGatherScatterParams(Type maskTy, VectorType valueTy,
73  TensorDescType tdescTy,
74  function_ref<InFlightDiagnostic()> emitError) {
75 
76  if (!tdescTy.isScattered())
77  return emitError() << "Expects a scattered TensorDesc.";
78 
79  if (!valueTy)
80  return emitError() << "Expecting a vector type result.";
81 
82  auto maskShape = getShapeOf(maskTy);
83  auto valueShape = getShapeOf(valueTy);
84  auto tdescShape = getShapeOf(tdescTy);
85  auto chunkSize = tdescTy.getChunkSizeAsInt();
86 
87  if (valueTy.getElementType() != tdescTy.getElementType())
88  return emitError()
89  << "Value should have the same element type as TensorDesc.";
90 
91  llvm::SmallVector<int64_t> expectedMaskShape(tdescShape);
92  if (chunkSize > 1)
93  expectedMaskShape.pop_back();
94  if (expectedMaskShape != maskShape)
95  return emitError()
96  << "Mask should match TensorDesc except the chunk size dim.";
97 
98  // a valid shape for SIMT case
99  if (valueTy.getRank() == 1 && valueTy.getNumElements() == chunkSize) {
100  if (tdescTy.getLayoutAttr())
101  return emitError() << "TensorDesc doesn't need LayoutAttr for SIMT code";
102  return success();
103  }
104 
105  if (tdescShape != valueShape)
106  return emitError() << "Value shape " << makeString(valueShape)
107  << " is neither a valid distribution for SIMT nor "
108  "consistent with the tensor descriptor for SIMD "
109  << tdescTy;
110  return success();
111 }

223 void CreateNdDescOp::build(OpBuilder &builder, OperationState &state,
224  Type tdesc, Value source,
225  llvm::ArrayRef<OpFoldResult> offsets,
226  llvm::ArrayRef<OpFoldResult> shape,
227  llvm::ArrayRef<OpFoldResult> strides) {
228  assert(shape.size() && offsets.size() && strides.size() &&
229  shape.size() == strides.size() && shape.size() == offsets.size());
230 
231  Type srcTy = source.getType();
232  assert((isa<IntegerType, MemRefType>(srcTy)) &&
233  "Source has to be either int or memref.");
234 
235  llvm::SmallVector<Value> dynamicOffsets;
236  llvm::SmallVector<Value> dynamicShape;
237  llvm::SmallVector<Value> dynamicStrides;
238 
239  llvm::SmallVector<int64_t> staticOffsets;
240  llvm::SmallVector<int64_t> staticShape;
241  llvm::SmallVector<int64_t> staticStrides;
242 
243  dispatchIndexOpFoldResults(offsets, dynamicOffsets, staticOffsets);
244  dispatchIndexOpFoldResults(shape, dynamicShape, staticShape);
245  dispatchIndexOpFoldResults(strides, dynamicStrides, staticStrides);
246 
247  auto staticOffsetsAttr = builder.getDenseI64ArrayAttr(staticOffsets);
248  auto staticShapeAttr = builder.getDenseI64ArrayAttr(staticShape);
249  auto staticStridesAttr = builder.getDenseI64ArrayAttr(staticStrides);
250 
251  if (auto memrefTy = dyn_cast<MemRefType>(srcTy)) {
252  auto memrefShape = memrefTy.getShape();
253  auto [memrefStrides, _] = memrefTy.getStridesAndOffset();
254 
255  // if shape and strides are from Memref, we don't need attributes for them
256  // to keep the IR print clean.
257  if (staticShape == memrefShape && staticStrides == memrefStrides) {
258  staticShapeAttr = DenseI64ArrayAttr();
259  staticStridesAttr = DenseI64ArrayAttr();
260  }
261  }
262 
263  build(builder, state, tdesc, source, dynamicOffsets, dynamicShape,
264  dynamicStrides, staticOffsetsAttr, staticShapeAttr, staticStridesAttr);
265 }

267 LogicalResult CreateNdDescOp::verify() {
268  auto rank = (int64_t)getMixedOffsets().size();
269  bool invalidRank = false;
270  bool invalidElemTy = false;
271 
272  // Memory space of created TensorDesc should match with the source.
273  // Both source and TensorDesc are considered for global memory by default,
274  // if the memory scope attr is not specified. If source is an integer,
275  // it is considered as ptr to global memory.
276  auto srcMemorySpace = getSourceMemorySpace();
277  auto tdescMemorySpace = static_cast<unsigned>(getType().getMemorySpace());
278  if (srcMemorySpace != tdescMemorySpace)
279  return emitOpError("Memory space mismatch.")
280  << " Source: " << srcMemorySpace
281  << ", TensorDesc: " << tdescMemorySpace;
282 
283  // check source type matches the rank if it is a memref.
284  // It also should have the same ElementType as TensorDesc.
285  auto memrefTy = dyn_cast<MemRefType>(getSourceType());
286  if (memrefTy) {
287  invalidRank |= (memrefTy.getRank() != rank);
288  invalidElemTy |= memrefTy.getElementType() != getElementType();
289  }
290 
291  if (llvm::isa<IntegerType>(getSourceType())) {
292  // strides and shape must present for integer source.
293  if (getMixedStrides().empty() || getMixedSizes().empty())
294  return emitOpError("Expecting strides and shape to be present for "
295  "integer source.");
296  }
297 
298  // mismatches among shape, strides, and offsets are
299  // already handeled by OffsetSizeAndStrideOpInterface.
300  // So they are not check here.
301  if (invalidRank)
302  return emitOpError(
303  "Expecting the rank of shape, strides, offsets, and source (if source "
304  "is a memref) should match with each other.");
305 
306  // check result TensorDesc rank
307  if (getType().getRank() > rank)
308  return emitOpError(
309  "Expecting the TensorDesc rank is not greater than the "
310  "ranks of shape, strides, offsets or the memref source.");
311 
312  if (invalidElemTy)
313  return emitOpError("TensorDesc should have the same element "
314  "type with the source if it is a memref.\n");
315 
316  if (getType().isScattered())
317  return emitOpError("Expects a non-scattered TensorDesc.\n");
318 
319  return success();
320 }

322 ParseResult parseOptionalDynamicIndexList(
323  OpAsmParser &parser,
324  SmallVectorImpl<OpAsmParser::UnresolvedOperand> &values,
325  DenseI64ArrayAttr &integers, SmallVectorImpl<Type> *valueTypes = nullptr,
326  AsmParser::Delimiter delimiter = AsmParser::Delimiter::Square) {
327 
328  SmallVector<int64_t, 4> integerVals;
329  auto parseIntegerOrValue = [&]() {
330  OpAsmParser::UnresolvedOperand operand;
331  auto res = parser.parseOptionalOperand(operand);
332 
333  if (res.has_value() && succeeded(res.value())) {
334  values.push_back(operand);
335  integerVals.push_back(ShapedType::kDynamic);
336  if (valueTypes && parser.parseColonType(valueTypes->emplace_back()))
337  return failure();
338  } else {
339  int64_t integer;
340  if (failed(parser.parseInteger(integer)))
341  return failure();
342  integerVals.push_back(integer);
343  }
344  return success();
345  };
346 
347  // If the optional values are given there must be left bracket
348  if (parser.parseOptionalLSquare().succeeded()) {
349  if (parser.parseCommaSeparatedList(parseIntegerOrValue) ||
350  parser.parseRSquare())
351  return parser.emitError(parser.getNameLoc())
352  << "expected a list of SSA values or integers";
353  integers = parser.getBuilder().getDenseI64ArrayAttr(integerVals);
354  return success();
355  }
356 
357  return success();
358 }

426 LogicalResult LoadNdOp::verify() {
427  auto tdescTy = getTensorDescType();
428  auto valueTy = getType();
429 
430  if (tdescTy.isScattered())
431  return emitOpError("Expects a non-scattered TensorDesc.\n");
432 
433  if (tdescTy.getRank() > 2)
434  return emitOpError("Expects a 1D or 2D TensorDesc.\n");
435 
436  if (!valueTy)
437  return emitOpError("Invalid result, it should be a VectorType.\n");
438 
439  if (!isReadHintOrNone(getL1HintAttr()))
440  return emitOpError("invalid l1_hint: ") << getL1HintAttr();
441 
442  if (!isReadHintOrNone(getL2HintAttr()))
443  return emitOpError("invalid l2_hint: ") << getL2HintAttr();
444 
445  if (!isReadHintOrNone(getL3HintAttr()))
446  return emitOpError("invalid l3_hint: ") << getL3HintAttr();
447 
448  int tdescElems = tdescTy.getNumElements() * tdescTy.getArrayLength();
449  int valueElems = valueTy.getNumElements();
450 
451  // If the result vector is 1D and has less elements than the tensor
452  // descriptor, it is supposed to be a SIMT op. The layout attribute in
453  // tensor_desc is not needed.
454  if (valueElems < tdescElems && valueTy.getRank() == 1) {
455  // SIMT mode doesn't need LayoutAttr.
456  if (tdescTy.getLayoutAttr())
457  return emitOpError()
458  << "TensorDesc doesn't need LayoutAttr for SIMT code";
459 
460  // For SIMT code, the load is evenly distributed across all lanes in a
461  // subgroup. Since subgroup size is arch dependent, we only check even
462  // distribution here.
463  if (tdescElems % valueElems)
464  return emitOpError()
465  << "Result shape " << makeString(getShapeOf(valueTy))
466  << " is not a valid distribution for tensor descriptor "
467  << tdescTy;
468 
469  return success();
470  }
471 
472  // Check SIMD mode.
473  auto tdescShape = getShapeOf(tdescTy);
474  auto valueShape = getShapeOf(valueTy);
475 
476  if (getTranspose()) {
477  auto trans = getTranspose().value();
478  // Make sure the transpose value is valid, and apply it
479  if (llvm::all_of(trans, [&](size_t s) { return s < tdescShape.size(); }))
480  tdescShape = applyPermutation(tdescShape, trans);
481  else
482  mlir::emitWarning(getLoc()) << "Invalid transpose attr. It is ignored.";
483  }
484 
485  if (getPacked()) {
486  if (tdescTy.getRank() == 2) {
487  const int axis = 0;
488  auto vnni_factor = valueShape.back();
489  tdescShape[axis] /= vnni_factor;
490  tdescShape.push_back(vnni_factor);
491  } else {
492  mlir::emitWarning(getLoc())
493  << "Invalid Packed Attr. It is ignored (available for 2D "
494  "TensorDesc only).";
495  }
496  }
497 
498  auto array_len = tdescTy.getArrayLength();
499  if (array_len > 1)
500  tdescShape.insert(tdescShape.begin(), array_len);
501 
502  if (tdescShape != valueShape)
503  return emitOpError() << "Result shape " << makeString(valueShape)
504  << " is not consistent with tensor descriptor "
505  << tdescTy;
506 
507  int64_t tDescRank = tdescTy.getRank();
508  int64_t offsetSize = static_cast<int64_t>(getOffsets().size());
509  int64_t constOffsetSize =
510  getConstOffsetsAttr() ? getConstOffsetsAttr().size() : 0;
511  if (((offsetSize != 0) && (offsetSize != tDescRank)) ||
512  ((constOffsetSize != 0) && (constOffsetSize != tDescRank)))
513  return emitOpError(
514  "Mismatched ranks between offsets and tensor descriptor");
515 
516  return success();
517 }

588  int64_t constOffsetSize =

590  if (((offsetSize != 0) && (offsetSize != tDescRank)) ||
591  ((constOffsetSize != 0) && (constOffsetSize != tDescRank)))

837 // XeGPU_DpasOp
838 //===----------------------------------------------------------------------===//
839 LogicalResult DpasOp::verify() {
840  int64_t lhsRank = getLhsType().getRank();
841  int64_t rhsRank = getRhsType().getRank();
842  int64_t resRank = getResultType().getRank();
843  auto lhsShape = getLhsType().getShape();
844  auto rhsShape = getRhsType().getShape();
845  auto resShape = getResultType().getShape();
846 
847  if (getAcc() && getAcc().getType() != getResultType())
848  return emitOpError("Expecting the acc type to be the same as result.");
849 
850  // SIMT code: the size of the B operand has to be a multiple of 32 bits.
851  // It skips the semantic check since lack of architecture information.
852  // Users need to ensure the correctness.
853  if (lhsRank == 1 && rhsRank == 1 && resRank == 1) {
854  auto numElems = getRhsType().getNumElements();
855  auto elemTy = getRhsType().getElementType();
856  auto factor = 32 / elemTy.getIntOrFloatBitWidth();
857  if (numElems % factor != 0)
858  return emitOpError("Expecting B operand to be a multiple of 32 bits.");
859  return success();
860  }
861 
862  // SIMD code
863  if (lhsRank != 2 || (rhsRank != 2 && rhsRank != 3) || resRank != 2)
864  return emitOpError(
865  "expecting lhs and result to be a 2D vector, and rhs to be either "
866  "2D or 3D (packed) vector.");
867  auto bK = rhsRank == 3 ? rhsShape[0] * rhsShape[2] : rhsShape[0];
868  if (bK != lhsShape[1])
869  return emitOpError("K-dimension mismatch.");
870  if (lhsShape[0] != resShape[0])
871  return emitOpError("M-dimension mismatch.");
872  if (rhsShape[1] != resShape[1])
873  return emitOpError("N-dimension mismatch.");
874 
875  return success();
876 }
