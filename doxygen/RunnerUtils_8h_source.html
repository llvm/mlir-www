39 template <typename T, typename StreamType>
40 void printMemRefMetaData(StreamType &os, const DynamicMemRefType<T> &V) {
41  os << "base@ = " << reinterpret_cast<void *>(V.data) << " rank = " << V.rank
42  << " offset = " << V.offset;
43  auto print = [&](const int64_t *ptr) {
44  if (V.rank == 0)
45  return;
46  os << ptr[0];
47  for (int64_t i = 1; i < V.rank; ++i)
48  os << ", " << ptr[i];
49  };
50  os << " sizes = [";
51  print(V.sizes);
52  os << "] strides = [";
53  print(V.strides);
54  os << "]";
55 }

98 template <typename T, int M, int... Dims>
99 void VectorDataPrinter<T, M, Dims...>::print(std::ostream &os,
100  const Vector<T, M, Dims...> &val) {
101  static_assert(M > 0, "0 dimensioned tensor");
102  static_assert(sizeof(val) == M * StaticSizeMult<Dims...>::value * sizeof(T),
103  "Incorrect vector size!");
104  // First
105  os << "(" << val[0];
106  if (M > 1)
107  os << ", ";
108  if (sizeof...(Dims) > 1)
109  os << "\n";
110  // Kernel
111  for (unsigned i = 1; i + 1 < M; ++i) {
112  printSpace(os, 2 * sizeof...(Dims));
113  os << val[i] << ", ";
114  if (sizeof...(Dims) > 1)
115  os << "\n";
116  }
117  // Last
118  if (M > 1) {
119  printSpace(os, sizeof...(Dims));
120  os << val[M - 1];
121  }
122  os << ")";
123 }

144 template <typename T>
145 void MemRefDataPrinter<T>::printFirst(std::ostream &os, T *base, int64_t dim,
146  int64_t rank, int64_t offset,
147  const int64_t *sizes,
148  const int64_t *strides) {
149  os << "[";
150  print(os, base, dim - 1, rank, offset, sizes + 1, strides + 1);
151  // If single element, close square bracket and return early.
152  if (sizes[0] <= 1) {
153  os << "]";
154  return;
155  }
156  os << ", ";
157  if (dim > 1)
158  os << "\n";
159 }

161 template <typename T>
162 void MemRefDataPrinter<T>::print(std::ostream &os, T *base, int64_t dim,
163  int64_t rank, int64_t offset,
164  const int64_t *sizes, const int64_t *strides) {
165  if (dim == 0) {
166  os << base[offset];
167  return;
168  }
169  printFirst(os, base, dim, rank, offset, sizes, strides);
170  for (unsigned i = 1; i + 1 < sizes[0]; ++i) {
171  printSpace(os, rank - dim + 1);
172  print(os, base, dim - 1, rank, offset + i * strides[0], sizes + 1,
173  strides + 1);
174  os << ", ";
175  if (dim > 1)
176  os << "\n";
177  }
178  if (sizes[0] <= 1)
179  return;
180  printLast(os, base, dim, rank, offset, sizes, strides);
181 }

183 template <typename T>
184 void MemRefDataPrinter<T>::printLast(std::ostream &os, T *base, int64_t dim,
185  int64_t rank, int64_t offset,
186  const int64_t *sizes,
187  const int64_t *strides) {
188  printSpace(os, rank - dim + 1);
189  print(os, base, dim - 1, rank, offset + (sizes[0] - 1) * (*strides),
190  sizes + 1, strides + 1);
191  os << "]";

229 /// Verify the result of two computations are equivalent up to a small
230 /// numerical error and return the number of errors.
231 template <typename T>
232 struct MemRefDataVerifier {
233  /// Maximum number of errors printed by the verifier.
234  static constexpr int printLimit = 10;
235 
236  /// Verify the relative difference of the values is smaller than epsilon.
237  static bool verifyRelErrorSmallerThan(T actual, T expected, T epsilon);
238 
239  /// Verify the values are equivalent (integers) or are close (floating-point).
240  static bool verifyElem(T actual, T expected);
241 
242  /// Verify the data element-by-element and return the number of errors.
243  static int64_t verify(std::ostream &os, T *actualBasePtr, T *expectedBasePtr,
244  int64_t dim, int64_t offset, const int64_t *sizes,
245  const int64_t *strides, int64_t &printCounter);
246 };
247 
248 template <typename T>
249 bool MemRefDataVerifier<T>::verifyRelErrorSmallerThan(T actual, T expected,
250  T epsilon) {
251  // Return an error if one of the values is infinite or NaN.
252  if (!std::isfinite(actual) || !std::isfinite(expected))
253  return false;
254  // Return true if the relative error is smaller than epsilon.
255  T delta = std::abs(actual - expected);
256  return (delta <= epsilon * std::abs(expected));
257 }
258 
259 template <typename T>
260 bool MemRefDataVerifier<T>::verifyElem(T actual, T expected) {
261  return actual == expected;
262 }
263 
264 template <>
265 inline bool MemRefDataVerifier<double>::verifyElem(double actual,
266  double expected) {
267  return verifyRelErrorSmallerThan(actual, expected, 1e-12);
268 }
269 
270 template <>
271 inline bool MemRefDataVerifier<float>::verifyElem(float actual,
272  float expected) {
273  return verifyRelErrorSmallerThan(actual, expected, 1e-6f);
274 }

276 template <typename T>
277 int64_t MemRefDataVerifier<T>::verify(std::ostream &os, T *actualBasePtr,
278  T *expectedBasePtr, int64_t dim,
279  int64_t offset, const int64_t *sizes,
280  const int64_t *strides,
281  int64_t &printCounter) {
282  int64_t errors = 0;
283  // Verify the elements at the current offset.
284  if (dim == 0) {
285  if (!verifyElem(actualBasePtr[offset], expectedBasePtr[offset])) {
286  if (printCounter < printLimit) {
287  os << actualBasePtr[offset] << " != " << expectedBasePtr[offset]
288  << " offset = " << offset << "\n";
289  printCounter++;
290  }
291  errors++;
292  }
293  } else {
294  // Iterate the current dimension and verify recursively.
295  for (int64_t i = 0; i < sizes[0]; ++i) {
296  errors +=
297  verify(os, actualBasePtr, expectedBasePtr, dim - 1,
298  offset + i * strides[0], sizes + 1, strides + 1, printCounter);
299  }
300  }
301  return errors;
302 }

304 /// Verify the equivalence of two dynamic memrefs and return the number of
305 /// errors or -1 if the shape of the memrefs do not match.
306 template <typename T>
307 int64_t verifyMemRef(const DynamicMemRefType<T> &actual,
308  const DynamicMemRefType<T> &expected) {
309  // Check if the memref shapes match.
310  for (int64_t i = 0; i < actual.rank; ++i) {
311  if (expected.rank != actual.rank || actual.offset != expected.offset ||
312  actual.sizes[i] != expected.sizes[i] ||
313  actual.strides[i] != expected.strides[i]) {
314  printMemRefMetaData(std::cerr, actual);
315  printMemRefMetaData(std::cerr, expected);
316  return -1;
317  }
318  }
319  // Return the number of errors.
320  int64_t printCounter = 0;
321  return MemRefDataVerifier<T>::verify(
322  std::cerr, actual.basePtr, expected.basePtr, actual.rank, actual.offset,
323  actual.sizes, actual.strides, printCounter);
324 }
