51 LogicalResult
52 AbstractSparseForwardDataFlowAnalysis::initialize(Operation *top) {
53  // Mark the entry block arguments as having reached their pessimistic
54  // fixpoints.
55  for (Region &region : top->getRegions()) {
56  if (region.empty())
57  continue;
58  for (Value argument : region.front().getArguments())
59  setToEntryState(getLatticeElement(argument));
60  }
61 
62  return initializeRecursively(top);
63 }

65 LogicalResult
66 AbstractSparseForwardDataFlowAnalysis::initializeRecursively(Operation *op) {
67  // Initialize the analysis by visiting every owner of an SSA value (all
68  // operations and blocks).
69  if (failed(visitOperation(op)))
70  return failure();
71 
72  for (Region &region : op->getRegions()) {
73  for (Block &block : region) {
74  getOrCreate<Executable>(getProgramPointBefore(&block))
75  ->blockContentSubscribe(this);
76  visitBlock(&block);
77  for (Operation &op : block)
78  if (failed(initializeRecursively(&op)))
79  return failure();
80  }
81  }
82 
83  return success();
84 }

94 LogicalResult
95 AbstractSparseForwardDataFlowAnalysis::visitOperation(Operation *op) {
96  // Exit early on operations with no results.
97  if (op->getNumResults() == 0)
98  return success();
99 
100  // If the containing block is not executable, bail out.
101  if (op->getBlock() != nullptr &&
102  !getOrCreate<Executable>(getProgramPointBefore(op->getBlock()))->isLive())
103  return success();
104 
105  // Get the result lattices.
106  SmallVector<AbstractSparseLattice *> resultLattices;
107  resultLattices.reserve(op->getNumResults());
108  for (Value result : op->getResults()) {
109  AbstractSparseLattice *resultLattice = getLatticeElement(result);
110  resultLattices.push_back(resultLattice);
111  }
112 
113  // The results of a region branch operation are determined by control-flow.
114  if (auto branch = dyn_cast<RegionBranchOpInterface>(op)) {
115  visitRegionSuccessors(getProgramPointAfter(branch), branch,
116  /*successor=*/RegionBranchPoint::parent(),
117  resultLattices);
118  return success();
119  }
120 
121  // Grab the lattice elements of the operands.
122  SmallVector<const AbstractSparseLattice *> operandLattices;
123  operandLattices.reserve(op->getNumOperands());
124  for (Value operand : op->getOperands()) {
125  AbstractSparseLattice *operandLattice = getLatticeElement(operand);
126  operandLattice->useDefSubscribe(this);
127  operandLattices.push_back(operandLattice);
128  }
129 
130  if (auto call = dyn_cast<CallOpInterface>(op))
131  return visitCallOperation(call, operandLattices, resultLattices);
132 
133  // Invoke the operation transfer function.
134  return visitOperationImpl(op, operandLattices, resultLattices);
135 }

137 void AbstractSparseForwardDataFlowAnalysis::visitBlock(Block *block) {
138  // Exit early on blocks with no arguments.
139  if (block->getNumArguments() == 0)
140  return;
141 
142  // If the block is not executable, bail out.
143  if (!getOrCreate<Executable>(getProgramPointBefore(block))->isLive())
144  return;
145 
146  // Get the argument lattices.
147  SmallVector<AbstractSparseLattice *> argLattices;
148  argLattices.reserve(block->getNumArguments());
149  for (BlockArgument argument : block->getArguments()) {
150  AbstractSparseLattice *argLattice = getLatticeElement(argument);
151  argLattices.push_back(argLattice);
152  }
153 
154  // The argument lattices of entry blocks are set by region control-flow or the
155  // callgraph.
156  if (block->isEntryBlock()) {
157  // Check if this block is the entry block of a callable region.
158  auto callable = dyn_cast<CallableOpInterface>(block->getParentOp());
159  if (callable && callable.getCallableRegion() == block->getParent())
160  return visitCallableOperation(callable, argLattices);
161 
162  // Check if the lattices can be determined from region control flow.
163  if (auto branch = dyn_cast<RegionBranchOpInterface>(block->getParentOp())) {
164  return visitRegionSuccessors(getProgramPointBefore(block), branch,
165  block->getParent(), argLattices);
166  }
167 
168  // Otherwise, we can't reason about the data-flow.
169  return visitNonControlFlowArgumentsImpl(block->getParentOp(),
170  RegionSuccessor(block->getParent()),
171  argLattices, /*firstIndex=*/0);
172  }
173 
174  // Iterate over the predecessors of the non-entry block.
175  for (Block::pred_iterator it = block->pred_begin(), e = block->pred_end();
176  it != e; ++it) {
177  Block *predecessor = *it;
178 
179  // If the edge from the predecessor block to the current block is not live,
180  // bail out.
181  auto *edgeExecutable =
182  getOrCreate<Executable>(getLatticeAnchor<CFGEdge>(predecessor, block));
183  edgeExecutable->blockContentSubscribe(this);
184  if (!edgeExecutable->isLive())
185  continue;
186 
187  // Check if we can reason about the data-flow from the predecessor.
188  if (auto branch =
189  dyn_cast<BranchOpInterface>(predecessor->getTerminator())) {
190  SuccessorOperands operands =
191  branch.getSuccessorOperands(it.getSuccessorIndex());
192  for (auto [idx, lattice] : llvm::enumerate(argLattices)) {
193  if (Value operand = operands[idx]) {
194  join(lattice,
195  *getLatticeElementFor(getProgramPointBefore(block), operand));
196  } else {
197  // Conservatively consider internally produced arguments as entry
198  // points.
199  setAllToEntryStates(lattice);
200  }
201  }
202  } else {
203  return setAllToEntryStates(argLattices);
204  }
205  }
206 }

208 LogicalResult AbstractSparseForwardDataFlowAnalysis::visitCallOperation(
209  CallOpInterface call,
210  ArrayRef<const AbstractSparseLattice *> operandLattices,
211  ArrayRef<AbstractSparseLattice *> resultLattices) {
212  // If the call operation is to an external function, attempt to infer the
213  // results from the call arguments.
214  auto callable =
215  dyn_cast_if_present<CallableOpInterface>(call.resolveCallable());
216  if (!getSolverConfig().isInterprocedural() ||
217  (callable && !callable.getCallableRegion())) {
218  visitExternalCallImpl(call, operandLattices, resultLattices);
219  return success();
220  }
221 
222  // Otherwise, the results of a call operation are determined by the
223  // callgraph.
224  const auto *predecessors = getOrCreateFor<PredecessorState>(
225  getProgramPointAfter(call), getProgramPointAfter(call));
226  // If not all return sites are known, then conservatively assume we can't
227  // reason about the data-flow.
228  if (!predecessors->allPredecessorsKnown()) {
229  setAllToEntryStates(resultLattices);
230  return success();
231  }
232  for (Operation *predecessor : predecessors->getKnownPredecessors())
233  for (auto &&[operand, resLattice] :
234  llvm::zip(predecessor->getOperands(), resultLattices))
235  join(resLattice,
236  *getLatticeElementFor(getProgramPointAfter(call), operand));
237  return success();
238 }

240 void AbstractSparseForwardDataFlowAnalysis::visitCallableOperation(
241  CallableOpInterface callable,
242  ArrayRef<AbstractSparseLattice *> argLattices) {
243  Block *entryBlock = &callable.getCallableRegion()->front();
244  const auto *callsites = getOrCreateFor<PredecessorState>(
245  getProgramPointBefore(entryBlock), getProgramPointAfter(callable));
246  // If not all callsites are known, conservatively mark all lattices as
247  // having reached their pessimistic fixpoints.
248  if (!callsites->allPredecessorsKnown() ||
249  !getSolverConfig().isInterprocedural()) {
250  return setAllToEntryStates(argLattices);
251  }
252  for (Operation *callsite : callsites->getKnownPredecessors()) {
253  auto call = cast<CallOpInterface>(callsite);
254  for (auto it : llvm::zip(call.getArgOperands(), argLattices))
255  join(std::get<1>(it),
256  *getLatticeElementFor(getProgramPointBefore(entryBlock),
257  std::get<0>(it)));
258  }
259 }

261 void AbstractSparseForwardDataFlowAnalysis::visitRegionSuccessors(
262  ProgramPoint *point, RegionBranchOpInterface branch,
263  RegionBranchPoint successor, ArrayRef<AbstractSparseLattice *> lattices) {
264  const auto *predecessors = getOrCreateFor<PredecessorState>(point, point);
265  assert(predecessors->allPredecessorsKnown() &&
266  "unexpected unresolved region successors");
267 
268  for (Operation *op : predecessors->getKnownPredecessors()) {
269  // Get the incoming successor operands.
270  std::optional<OperandRange> operands;
271 
272  // Check if the predecessor is the parent op.
273  if (op == branch) {
274  operands = branch.getEntrySuccessorOperands(successor);
275  // Otherwise, try to deduce the operands from a region return-like op.
276  } else if (auto regionTerminator =
277  dyn_cast<RegionBranchTerminatorOpInterface>(op)) {
278  operands = regionTerminator.getSuccessorOperands(successor);
279  }
280 
281  if (!operands) {
282  // We can't reason about the data-flow.
283  return setAllToEntryStates(lattices);
284  }
285 
286  ValueRange inputs = predecessors->getSuccessorInputs(op);
287  assert(inputs.size() == operands->size() &&
288  "expected the same number of successor inputs as operands");
289 
290  unsigned firstIndex = 0;
291  if (inputs.size() != lattices.size()) {
292  if (!point->isBlockStart()) {
293  if (!inputs.empty())
294  firstIndex = cast<OpResult>(inputs.front()).getResultNumber();
295  visitNonControlFlowArgumentsImpl(
296  branch,
297  RegionSuccessor(
298  branch->getResults().slice(firstIndex, inputs.size())),
299  lattices, firstIndex);
300  } else {
301  if (!inputs.empty())
302  firstIndex = cast<BlockArgument>(inputs.front()).getArgNumber();
303  Region *region = point->getBlock()->getParent();
304  visitNonControlFlowArgumentsImpl(
305  branch,
306  RegionSuccessor(region, region->getArguments().slice(
307  firstIndex, inputs.size())),
308  lattices, firstIndex);
309  }
310  }
311 
312  for (auto it : llvm::zip(*operands, lattices.drop_front(firstIndex)))
313  join(std::get<1>(it), *getLatticeElementFor(point, std::get<0>(it)));
314  }
315 }

351 LogicalResult
352 AbstractSparseBackwardDataFlowAnalysis::initializeRecursively(Operation *op) {
353  if (failed(visitOperation(op)))
354  return failure();
355 
356  for (Region &region : op->getRegions()) {
357  for (Block &block : region) {
358  getOrCreate<Executable>(getProgramPointBefore(&block))
359  ->blockContentSubscribe(this);
360  // Initialize ops in reverse order, so we can do as much initial
361  // propagation as possible without having to go through the
362  // solver queue.
363  for (auto it = block.rbegin(); it != block.rend(); it++)
364  if (failed(initializeRecursively(&*it)))
365  return failure();
366  }
367  }
368  return success();
369 }

377  if (point->isBlockStart())

410 LogicalResult
411 AbstractSparseBackwardDataFlowAnalysis::visitOperation(Operation *op) {
412  // If we're in a dead block, bail out.
413  if (op->getBlock() != nullptr &&
414  !getOrCreate<Executable>(getProgramPointBefore(op->getBlock()))->isLive())
415  return success();
416 
417  SmallVector<AbstractSparseLattice *> operandLattices =
418  getLatticeElements(op->getOperands());
419  SmallVector<const AbstractSparseLattice *> resultLattices =
420  getLatticeElementsFor(getProgramPointAfter(op), op->getResults());
421 
422  // Block arguments of region branch operations flow back into the operands
423  // of the parent op
424  if (auto branch = dyn_cast<RegionBranchOpInterface>(op)) {
425  visitRegionSuccessors(branch, operandLattices);
426  return success();
427  }
428 
429  if (auto branch = dyn_cast<BranchOpInterface>(op)) {
430  // Block arguments of successor blocks flow back into our operands.
431 
432  // We remember all operands not forwarded to any block in a BitVector.
433  // We can't just cut out a range here, since the non-forwarded ops might
434  // be non-contiguous (if there's more than one successor).
435  BitVector unaccounted(op->getNumOperands(), true);
436 
437  for (auto [index, block] : llvm::enumerate(op->getSuccessors())) {
438  SuccessorOperands successorOperands = branch.getSuccessorOperands(index);
439  OperandRange forwarded = successorOperands.getForwardedOperands();
440  if (!forwarded.empty()) {
441  MutableArrayRef<OpOperand> operands = op->getOpOperands().slice(
442  forwarded.getBeginOperandIndex(), forwarded.size());
443  for (OpOperand &operand : operands) {
444  unaccounted.reset(operand.getOperandNumber());
445  if (std::optional<BlockArgument> blockArg =
446  detail::getBranchSuccessorArgument(
447  successorOperands, operand.getOperandNumber(), block)) {
448  meet(getLatticeElement(operand.get()),
449  *getLatticeElementFor(getProgramPointAfter(op), *blockArg));
450  }
451  }
452  }
453  }
454  // Operands not forwarded to successor blocks are typically parameters
455  // of the branch operation itself (for example the boolean for if/else).
456  for (int index : unaccounted.set_bits()) {
457  OpOperand &operand = op->getOpOperand(index);
458  visitBranchOperand(operand);
459  }
460  return success();
461  }
462 
463  // For function calls, connect the arguments of the entry blocks to the
464  // operands of the call op that are forwarded to these arguments.
465  if (auto call = dyn_cast<CallOpInterface>(op)) {
466  Operation *callableOp = call.resolveCallableInTable(&symbolTable);
467  if (auto callable = dyn_cast_or_null<CallableOpInterface>(callableOp)) {
468  // Not all operands of a call op forward to arguments. Such operands are
469  // stored in `unaccounted`.
470  BitVector unaccounted(op->getNumOperands(), true);
471 
472  // If the call invokes an external function (or a function treated as
473  // external due to config), defer to the corresponding extension hook.
474  // By default, it just does `visitCallOperand` for all operands.
475  OperandRange argOperands = call.getArgOperands();
476  MutableArrayRef<OpOperand> argOpOperands =
477  operandsToOpOperands(argOperands);
478  Region *region = callable.getCallableRegion();
479  if (!region || region->empty() ||
480  !getSolverConfig().isInterprocedural()) {
481  visitExternalCallImpl(call, operandLattices, resultLattices);
482  return success();
483  }
484 
485  // Otherwise, propagate information from the entry point of the function
486  // back to operands whenever possible.
487  Block &block = region->front();
488  for (auto [blockArg, argOpOperand] :
489  llvm::zip(block.getArguments(), argOpOperands)) {
490  meet(getLatticeElement(argOpOperand.get()),
491  *getLatticeElementFor(getProgramPointAfter(op), blockArg));
492  unaccounted.reset(argOpOperand.getOperandNumber());
493  }
494 
495  // Handle the operands of the call op that aren't forwarded to any
496  // arguments.
497  for (int index : unaccounted.set_bits()) {
498  OpOperand &opOperand = op->getOpOperand(index);
499  visitCallOperand(opOperand);
500  }
501  return success();
502  }
503  }
504 
505  // When the region of an op implementing `RegionBranchOpInterface` has a
506  // terminator implementing `RegionBranchTerminatorOpInterface` or a
507  // return-like terminator, the region's successors' arguments flow back into
508  // the "successor operands" of this terminator.
509  //
510  // A successor operand with respect to an op implementing
511  // `RegionBranchOpInterface` is an operand that is forwarded to a region
512  // successor's input. There are two types of successor operands: the operands
513  // of this op itself and the operands of the terminators of the regions of
514  // this op.
515  if (auto terminator = dyn_cast<RegionBranchTerminatorOpInterface>(op)) {
516  if (auto branch = dyn_cast<RegionBranchOpInterface>(op->getParentOp())) {
517  visitRegionSuccessorsFromTerminator(terminator, branch);
518  return success();
519  }
520  }
521 
522  if (op->hasTrait<OpTrait::ReturnLike>()) {
523  // Going backwards, the operands of the return are derived from the
524  // results of all CallOps calling this CallableOp.
525  if (auto callable = dyn_cast<CallableOpInterface>(op->getParentOp()))
526  return visitCallableOperation(op, callable, operandLattices);
527  }
528 
529  return visitOperationImpl(op, operandLattices, resultLattices);
530 }

532 LogicalResult AbstractSparseBackwardDataFlowAnalysis::visitCallableOperation(
533  Operation *op, CallableOpInterface callable,
534  ArrayRef<AbstractSparseLattice *> operandLattices) {
535  const PredecessorState *callsites = getOrCreateFor<PredecessorState>(
536  getProgramPointAfter(op), getProgramPointAfter(callable));
537  if (callsites->allPredecessorsKnown()) {
538  for (Operation *call : callsites->getKnownPredecessors()) {
539  SmallVector<const AbstractSparseLattice *> callResultLattices =
540  getLatticeElementsFor(getProgramPointAfter(op), call->getResults());
541  for (auto [op, result] : llvm::zip(operandLattices, callResultLattices))
542  meet(op, *result);
543  }
544  } else {
545  // If we don't know all the callers, we can't know where the
546  // returned values go. Note that, in particular, this will trigger
547  // for the return ops of any public functions.
548  setAllToExitStates(operandLattices);
549  }
550  return success();
551 }

553 void AbstractSparseBackwardDataFlowAnalysis::visitRegionSuccessors(
554  RegionBranchOpInterface branch,
555  ArrayRef<AbstractSparseLattice *> operandLattices) {
556  Operation *op = branch.getOperation();
557  SmallVector<RegionSuccessor> successors;
558  SmallVector<Attribute> operands(op->getNumOperands(), nullptr);
559  branch.getEntrySuccessorRegions(operands, successors);
560 
561  // All operands not forwarded to any successor. This set can be non-contiguous
562  // in the presence of multiple successors.
563  BitVector unaccounted(op->getNumOperands(), true);
564 
565  for (RegionSuccessor &successor : successors) {
566  OperandRange operands = branch.getEntrySuccessorOperands(successor);
567  MutableArrayRef<OpOperand> opoperands = operandsToOpOperands(operands);
568  ValueRange inputs = successor.getSuccessorInputs();
569  for (auto [operand, input] : llvm::zip(opoperands, inputs)) {
570  meet(getLatticeElement(operand.get()),
571  *getLatticeElementFor(getProgramPointAfter(op), input));
572  unaccounted.reset(operand.getOperandNumber());
573  }
574  }
575  // All operands not forwarded to regions are typically parameters of the
576  // branch operation itself (for example the boolean for if/else).
577  for (int index : unaccounted.set_bits()) {
578  visitBranchOperand(op->getOpOperand(index));
579  }
580 }

582 void AbstractSparseBackwardDataFlowAnalysis::
583  visitRegionSuccessorsFromTerminator(
584  RegionBranchTerminatorOpInterface terminator,
585  RegionBranchOpInterface branch) {
586  assert(isa<RegionBranchTerminatorOpInterface>(terminator) &&
587  "expected a `RegionBranchTerminatorOpInterface` op");
588  assert(terminator->getParentOp() == branch.getOperation() &&
589  "expected `branch` to be the parent op of `terminator`");
590 
591  SmallVector<Attribute> operandAttributes(terminator->getNumOperands(),
592  nullptr);
593  SmallVector<RegionSuccessor> successors;
594  terminator.getSuccessorRegions(operandAttributes, successors);
595  // All operands not forwarded to any successor. This set can be
596  // non-contiguous in the presence of multiple successors.
597  BitVector unaccounted(terminator->getNumOperands(), true);
598 
599  for (const RegionSuccessor &successor : successors) {
600  ValueRange inputs = successor.getSuccessorInputs();
601  OperandRange operands = terminator.getSuccessorOperands(successor);
602  MutableArrayRef<OpOperand> opOperands = operandsToOpOperands(operands);
603  for (auto [opOperand, input] : llvm::zip(opOperands, inputs)) {
604  meet(getLatticeElement(opOperand.get()),
605  *getLatticeElementFor(getProgramPointAfter(terminator), input));
606  unaccounted.reset(const_cast<OpOperand &>(opOperand).getOperandNumber());
607  }
608  }
609  // Visit operands of the branch op not forwarded to the next region.
610  // (Like e.g. the boolean of `scf.conditional`)
611  for (int index : unaccounted.set_bits()) {
612  visitBranchOperand(terminator->getOpOperand(index));
613  }
614 }
