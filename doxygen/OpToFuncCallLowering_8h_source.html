19 /// Rewriting that replace SourceOp with a CallOp to `f32Func` or `f64Func` or
20 /// `f32ApproxFunc` or `f16Func` depending on the element type and the
21 /// fastMathFlag of that Op. The function declaration is added in case it was
22 /// not added before.
23 ///
24 /// If the input values are of bf16 type (or f16 type if f16Func is empty), the
25 /// value is first casted to f32, the function called and then the result casted
26 /// back.
27 ///
28 /// Example with NVVM:
29 /// %exp_f32 = math.exp %arg_f32 : f32
30 ///
31 /// will be transformed into
32 /// llvm.call @__nv_expf(%arg_f32) : (f32) -> f32
33 ///
34 /// If the fastMathFlag attribute of SourceOp is `afn` or `fast`, this Op lowers
35 /// to the approximate calculation function.
36 ///
37 /// Also example with NVVM:
38 /// %exp_f32 = math.exp %arg_f32 fastmath<afn> : f32
39 ///
40 /// will be transformed into
41 /// llvm.call @__nv_fast_expf(%arg_f32) : (f32) -> f32
42 template <typename SourceOp>
43 struct OpToFuncCallLowering : public ConvertOpToLLVMPattern<SourceOp> {
44 public:
45  explicit OpToFuncCallLowering(const LLVMTypeConverter &lowering,
46  StringRef f32Func, StringRef f64Func,
47  StringRef f32ApproxFunc, StringRef f16Func)
48  : ConvertOpToLLVMPattern<SourceOp>(lowering), f32Func(f32Func),
49  f64Func(f64Func), f32ApproxFunc(f32ApproxFunc), f16Func(f16Func) {}
50 
51  LogicalResult
52  matchAndRewrite(SourceOp op, typename SourceOp::Adaptor adaptor,
53  ConversionPatternRewriter &rewriter) const override {
54  using LLVM::LLVMFuncOp;
55 
56  static_assert(
57  std::is_base_of<OpTrait::OneResult<SourceOp>, SourceOp>::value,
58  "expected single result op");
59 
60  static_assert(std::is_base_of<OpTrait::SameOperandsAndResultType<SourceOp>,
61  SourceOp>::value,
62  "expected op with same operand and result types");
63 
64  SmallVector<Value, 1> castedOperands;
65  for (Value operand : adaptor.getOperands())
66  castedOperands.push_back(maybeCast(operand, rewriter));
67 
68  Type resultType = castedOperands.front().getType();
69  Type funcType = getFunctionType(resultType, castedOperands);
70  StringRef funcName =
71  getFunctionName(cast<LLVM::LLVMFunctionType>(funcType).getReturnType(),
72  op.getFastmath());
73  if (funcName.empty())
74  return failure();
75 
76  LLVMFuncOp funcOp = appendOrGetFuncOp(funcName, funcType, op);
77  auto callOp =
78  rewriter.create<LLVM::CallOp>(op->getLoc(), funcOp, castedOperands);
79 
80  if (resultType == adaptor.getOperands().front().getType()) {
81  rewriter.replaceOp(op, {callOp.getResult()});
82  return success();
83  }
84 
85  Value truncated = rewriter.create<LLVM::FPTruncOp>(
86  op->getLoc(), adaptor.getOperands().front().getType(),
87  callOp.getResult());
88  rewriter.replaceOp(op, {truncated});
89  return success();
90  }
91 
92 private:
93  Value maybeCast(Value operand, PatternRewriter &rewriter) const {
94  Type type = operand.getType();
95  if (!isa<Float16Type, BFloat16Type>(type))
96  return operand;
97 
98  // if there's a f16 function, no need to cast f16 values
99  if (!f16Func.empty() && isa<Float16Type>(type))
100  return operand;
101 
102  return rewriter.create<LLVM::FPExtOp>(
103  operand.getLoc(), Float32Type::get(rewriter.getContext()), operand);
104  }
105 
106  Type getFunctionType(Type resultType, ValueRange operands) const {
107  SmallVector<Type> operandTypes(operands.getTypes());
108  return LLVM::LLVMFunctionType::get(resultType, operandTypes);
109  }
110 
111  StringRef getFunctionName(Type type, arith::FastMathFlags flag) const {
112  if (isa<Float16Type>(type))
113  return f16Func;
114  if (isa<Float32Type>(type)) {
115  if (((uint32_t)arith::FastMathFlags::afn & (uint32_t)flag) &&
116  !f32ApproxFunc.empty())
117  return f32ApproxFunc;
118  else
119  return f32Func;
120  }
121  if (isa<Float64Type>(type))
122  return f64Func;
123  return "";
124  }
125 
126  LLVM::LLVMFuncOp appendOrGetFuncOp(StringRef funcName, Type funcType,
127  Operation *op) const {
128  using LLVM::LLVMFuncOp;
129 
130  auto funcAttr = StringAttr::get(op->getContext(), funcName);
131  Operation *funcOp = SymbolTable::lookupNearestSymbolFrom(op, funcAttr);
132  if (funcOp)
133  return cast<LLVMFuncOp>(*funcOp);
134 
135  mlir::OpBuilder b(op->getParentOfType<FunctionOpInterface>());
136  return b.create<LLVMFuncOp>(op->getLoc(), funcName, funcType);
137  }
138 
139  const std::string f32Func;
140  const std::string f64Func;
141  const std::string f32ApproxFunc;
142  const std::string f16Func;
143 };
