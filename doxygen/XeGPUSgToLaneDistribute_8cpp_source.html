48/// Casts the given vector value `v` to the expected vector type `expectedTy`.
49static Value castValueTo(ConversionPatternRewriter &rewriter,
50 TypedValue<VectorType> v, VectorType expectedTy) {
51 // If the type matches, simply return the value itself.
52 if (v.getType() == expectedTy)
53 return v;
54 // If only shape differs, use shape cast.
55 if (isa<VectorType>(v.getType()) &&
56 v.getType().getNumElements() == expectedTy.getNumElements())
57 return vector::ShapeCastOp::create(rewriter, v.getLoc(), expectedTy, v);
58
59 // Else create an unrealized cast.
60 auto newOp = UnrealizedConversionCastOp::create(rewriter, v.getLoc(),
61 expectedTy, ValueRange{v});
62 return newOp.getResult(0);
63}

65/// A vector::MultiDimReductionOp at subgroup level in expected form if, it has
66/// exactly 1 reduction dimension, it had valid result layout attribute, and
67/// result type can be distributed to lanes using the layout.
68static bool isValidSubgroupMultiReductionOp(vector::MultiDimReductionOp op) {
69 auto resLayout = xegpu::getTemporaryLayout(op->getOpResult(0));
70 // If no layout, not valid.
71 if (!resLayout || !resLayout.isForSubgroup())
72 return false;
73 // Scalar result (e.g., vector<32xf32> to f32) is valid.
74 if (op.getType().isIntOrFloat())
75 return op.getReductionDims().size() == 1;
76 VectorType resTy = dyn_cast<VectorType>(op.getType());
77 if (!resTy)
78 return false;
79 // Compute the distributed result vector type based on the layout.
80 FailureOr<VectorType> resDistTypeOrFailure =
81 getDistVecTypeBasedOnLaneLayout(resLayout, resTy);
82 if (failed(resDistTypeOrFailure))
83 return false;
84 return op.getReductionDims().size() == 1;
85}

87/// A vector::MultiDimReductionOp is doing lane-local reduction if each lane
88/// is doing its own local reduction. In this case the result layout ensures
89/// that result vector is distributed to lanes, i.e. the result vector type is
90/// different from the distributed result vector type.
91static bool isReductionLaneLocal(vector::MultiDimReductionOp op) {
92 // Must be valid MultiDimReductionOp.
93 assert(isValidSubgroupMultiReductionOp(op) && "Expecting a valid subgroup "
94 "MultiDimReductionOp");
95 auto resLayout = xegpu::getTemporaryLayout(op->getOpResult(0));
96 VectorType resTy = dyn_cast<VectorType>(op.getType());
97 auto resDistTypeOrFailure = getDistVecTypeBasedOnLaneLayout(resLayout, resTy);
98 return resTy != resDistTypeOrFailure.value();
99}

101/// Given a vector type and its distributed vector type, return the list of
102/// dimensions that are distributed.
103static SmallVector<int64_t> getDistributedDims(VectorType originalType,
104 VectorType distributedType) {
105 assert(originalType.getRank() == distributedType.getRank() &&
106 "original and distributed vector types must have the same rank");
107 SmallVector<int64_t> distributedDims;
108 for (int64_t i = 0; i < originalType.getRank(); ++i) {
109 if (distributedType.getDimSize(i) != originalType.getDimSize(i))
110 distributedDims.push_back(i);
111 }
112 return distributedDims;
113}

137/// Distributes a subgroup-level LoadNd op to lane-level LoadNd op. Output
138/// of lane-level LoadNd op is 1D. ShapeCast is added to restore the
139/// original rank.
140struct SgToLaneLoadNd : public OpConversionPattern<xegpu::LoadNdOp> {
141 using OpConversionPattern<xegpu::LoadNdOp>::OpConversionPattern;
142
143 LogicalResult
144 matchAndRewrite(xegpu::LoadNdOp op, OpAdaptor adaptor,
145 ConversionPatternRewriter &rewriter) const override {
146 xegpu::DistributeLayoutAttr layout = op.getAnchorLayout();
147 // If no layout, nothing to do.
148 if (!layout)
149 return failure();
150 // Check if the layout attached to the tensor descriptor is same as the
151 // anchor layout. Otherwise, this is a conflict.
152 if (op.getTensorDescType().getLayout() != layout)
153 return rewriter.notifyMatchFailure(
154 op, "conflicting layout attributes on tensor descriptor and anchor");
155 const auto *uArch =
156 xegpu::uArch::getUArch(xegpu::getChipStr(op).value_or(""));
157 if (!uArch)
158 return rewriter.notifyMatchFailure(
159 op, "xegpu::LoadNdOp require target attribute attached to "
160 "determine transpose "
161 "requirement");
162 auto supportedLaneResultTyOrFailure =
163 xegpu::getDistributedVectorType(op.getTensorDescType());
164 auto expectedLaneResultTyOrFailure =
165 xegpu::getDistVecTypeBasedOnLaneLayout(layout, op.getType());
166 if (failed(supportedLaneResultTyOrFailure))
167 return rewriter.notifyMatchFailure(
168 op, "unable to compute the lane vector type for LoadNdOp");
169 if (failed(expectedLaneResultTyOrFailure))
170 return rewriter.notifyMatchFailure(
171 op, "unable to compute expected lane vector type from lane layout");
172 auto newOp = xegpu::LoadNdOp::create(
173 rewriter, op.getLoc(), supportedLaneResultTyOrFailure.value(),
174 adaptor.getTensorDesc(), op.getMixedOffsets(), op.getPackedAttr(),
175 op.getTransposeAttr(), op.getL1HintAttr(), op.getL2HintAttr(),
176 op.getL3HintAttr(), /**layout**/ nullptr);
177 // Set the packed attribute if the layout requires it.
178 newOp.setPacked(xegpu::requirePacked(cast<xegpu::LayoutAttr>(layout)));
179 // Set the transpose attribute if the layout requires it.
180 if (xegpu::requireTranspose(cast<xegpu::LayoutAttr>(layout), uArch))
181 newOp.setTranspose(DenseI64ArrayAttr::get(rewriter.getContext(), {1, 0}));
182 rewriter.replaceOp(op, castValueTo(rewriter, newOp.getResult(),
183 expectedLaneResultTyOrFailure.value()));
184 return success();
185 }

229/// Distributes a subgroup-level Dpas op to lane-level Dpas op. All inpputs
230/// and output of lane-level Dpas op are 1D. Necessary casts are added to
231/// convert the inputs and output to/from 1D.
232struct SgToLaneDpas : public OpConversionPattern<xegpu::DpasOp> {
233 using OpConversionPattern<xegpu::DpasOp>::OpConversionPattern;
234
235 LogicalResult
236 matchAndRewrite(xegpu::DpasOp op, OpAdaptor adaptor,
237 ConversionPatternRewriter &rewriter) const override {
238 // Check if the op has A, B and CD layouts attached.
239 auto layoutA = cast<xegpu::LayoutAttr>(op.getLayoutAAttr());
240 auto layoutB = cast<xegpu::LayoutAttr>(op.getLayoutBAttr());
241 auto layoutCd = cast<xegpu::LayoutAttr>(op.getLayoutCdAttr());
242 if (!layoutA || !layoutB || !layoutCd)
243 return failure();
244 auto laneResultTyOrFailure =
245 xegpu::getDistributedVectorType(op.getType(), layoutCd);
246 auto laneATypeOrFailure =
247 xegpu::getDistributedVectorType(op.getLhs().getType(), layoutA);
248 auto laneBTypeOrFailure =
249 xegpu::getDistributedVectorType(op.getRhs().getType(), layoutB);
250 auto expectedLaneResultTyOrFailure =
251 xegpu::getDistVecTypeBasedOnLaneLayout(layoutCd, op.getType());
252 if (failed(laneResultTyOrFailure) || failed(laneATypeOrFailure) ||
253 failed(laneBTypeOrFailure))
254 return rewriter.notifyMatchFailure(
255 op, "failed to calculate supported lane vector types for DpasOp "
256 "from layouts");
257 if (failed(expectedLaneResultTyOrFailure))
258 return rewriter.notifyMatchFailure(
259 op, "unable to compute expected lane vector type for DpasOp from "
260 "lane layout");
261
262 // Validate bit widths match uArch packed format requirements
263 const auto *uArch =
264 xegpu::uArch::getUArch(xegpu::getChipStr(op).value_or(""));
265 if (uArch) {
266 const auto *uArchInstruction =
267 dyn_cast<xegpu::uArch::SubgroupMatrixMultiplyAcc>(
268 uArch->getInstruction(
269 xegpu::uArch::InstructionKind::SubgroupMatrixMultiplyAcc));
270 if (uArchInstruction) {
271 auto laneAType = laneATypeOrFailure.value();
272 auto laneBType = laneBTypeOrFailure.value();
273 // Calculate total packed bit width = element bit width * vector size
274 unsigned aPackedBitWidth =
275 laneAType.getElementTypeBitWidth() * laneAType.getNumElements();
276 unsigned bPackedBitWidth =
277 laneBType.getElementTypeBitWidth() * laneBType.getNumElements();
278 unsigned expectedABitSize = uArchInstruction->getPackedFormatBitSizeA();
279 unsigned expectedBBitSize = uArchInstruction->getPackedFormatBitSizeB();
280
281 if (aPackedBitWidth % expectedABitSize != 0)
282 return rewriter.notifyMatchFailure(
283 op,
284 "A operand packed bit width must be a multiple of uArch packed "
285 "format requirement");
286 if (bPackedBitWidth % expectedBBitSize != 0)
287 return rewriter.notifyMatchFailure(
288 op,
289 "B operand packed bit width must be a multiple of uArch packed "
290 "format requirement");
291 }
292 }
293
294 auto newOp = xegpu::DpasOp::create(
295 rewriter, op->getLoc(), laneResultTyOrFailure.value(),
296 castValueTo(rewriter, cast<TypedValue<VectorType>>(adaptor.getLhs()),
297 laneATypeOrFailure.value()),
298 castValueTo(rewriter, cast<TypedValue<VectorType>>(adaptor.getRhs()),
299 laneBTypeOrFailure.value()),
300 castValueTo(rewriter, cast<TypedValue<VectorType>>(adaptor.getAcc()),
301 laneResultTyOrFailure.value()),
302 /** layoutA**/ nullptr,
303 /** layoutB**/ nullptr, /** layoutCd**/ nullptr);
304 // Explicitly set the new types to enable correct type materializations.
305 rewriter.replaceOp(op, castValueTo(rewriter, newOp.getResult(),
306 expectedLaneResultTyOrFailure.value()));
307 return success();
308 }
309};

311/// Distributes elementwise ops to lane-level elementwise ops. This
312/// currently handles elementwise ops with single result only.
313struct SgToLaneElementWise : public ConversionPattern {
314 SgToLaneElementWise(TypeConverter &typeConverter, MLIRContext *ctx)
315 : ConversionPattern(MatchAnyOpTypeTag(), /*benefit=*/1, ctx) {}
316
317 LogicalResult
318 matchAndRewrite(Operation *op, ArrayRef<Value> operands,
319 ConversionPatternRewriter &rewriter) const override {
320 // Only match ops with elementwise trait and single result.
321 if (!OpTrait::hasElementwiseMappableTraits(op) || op->getNumResults() != 1)
322 return failure();
323
324 auto resultType = dyn_cast<VectorType>(op->getResult(0).getType());
325 if (!resultType)
326 return rewriter.notifyMatchFailure(
327 op, "operation result is not a vector type");
328
329 xegpu::DistributeLayoutAttr layout =
330 xegpu::getTemporaryLayout(llvm::cast<OpResult>(op->getResult(0)));
331 if (!layout || !layout.isForSubgroup())
332 return rewriter.notifyMatchFailure(
333 op, "operation result does not have subgroup distribute layout");
334
335 auto laneShapeOrFailure =
336 xegpu::getDistVecTypeBasedOnLaneLayout(layout, resultType);
337
338 if (failed(laneShapeOrFailure))
339 return rewriter.notifyMatchFailure(
340 op, "unable to compute lane vector type from the layout");
341
342 VectorType newResultType = laneShapeOrFailure.value();
343 OperationState state(op->getLoc(), op->getName());
344 state.addOperands(operands);
345 state.addTypes(newResultType);
346 // Copy all attributes except for DistributeLayoutAttr.
347 for (auto attr : op->getAttrs()) {
348 if (!isa<xegpu::DistributeLayoutAttr>(attr.getValue()))
349 state.addAttribute(attr.getName(), attr.getValue());
350 }
351 Operation *newOp = rewriter.create(state);
352
353 rewriter.replaceOp(op, newOp->getResult(0));
354 return success();
355 }

421/// Distributes a subgroup-level LoadGather (xegpu.load) op to lane-level.
422///
423/// Example 1 (1D, no chunk size):
424/// layout = #xegpu.layout<lane_layout = [16], lane_data = [1]>
425/// %mask = producer_op : vector<16xi1>
426/// %offset = producer_op : vector<16xindex>
427/// %0 = xegpu.load %src[%offset], %mask : memref<256xf16>,
428/// vector<16xindex>, vector<16xi1> -> vector<16xf16>
429/// Distributed to:
430/// %mask = producer_op : vector<1xi1>
431/// %offset = producer_op : vector<1xindex>
432/// %0 = xegpu.load %src[%offset], %mask : memref<256xf16>,
433/// vector<1xindex>, vector<1xi1> -> vector<1xf16>
434///
435/// Example 2 (2D with chunk size, same mask & offset):
436/// layout = #xegpu.layout<lane_layout = [16, 1], lane_data = [1, 1]>
437/// %0 = xegpu.load %src[%offset], %mask <{chunk_size=8}> :
438/// memref<256xf16>, vector<16xindex>, vector<16xi1> -> vector<16x8xf16>
439/// Distributed to:
440/// %0 = xegpu.load %src[%offset], %mask <{chunk_size=8}> :
441/// memref<256xf16>, vector<1xindex>, vector<1xi1> -> vector<8xf16>
442///
443/// Example 3 (3D with leading unit dims):
444/// layout = #xegpu.layout<lane_layout = [1, 1, 16], lane_data = [1, 1, 1]>
445/// %mask = producer_op : vector<1x1x16xi1>
446/// %offset = producer_op : vector<1x1x16xindex>
447/// %0 = xegpu.load %src[%offset], %mask : memref<256xf16>,
448/// vector<1x1x16xindex>, vector<1x1x16xi1> -> vector<1x1x16xf16>
449/// Distributed to:
450/// %mask = producer_op : vector<1x1x1xi1>
451/// %offset = producer_op : vector<1x1x1xindex>
452/// %0 = xegpu.load %src[%offset], %mask : memref<256xf16>,
453/// vector<1xindex>, vector<1xi1> -> vector<1xf16>
454struct SgToLaneLoadGather : public OpConversionPattern<xegpu::LoadGatherOp> {
455 using OpConversionPattern<xegpu::LoadGatherOp>::OpConversionPattern;
456
457 LogicalResult
458 matchAndRewrite(xegpu::LoadGatherOp op, OpAdaptor adaptor,
459 ConversionPatternRewriter &rewriter) const override {
460 xegpu::DistributeLayoutAttr layout = op.getAnchorLayout();
461 if (!layout)
462 return failure();
463
464 VectorType origResultTy = op.getValueType();
465 if (!origResultTy)
466 return failure();
467
468 // Check that leading dimensions are unit.
469 int chunkSize = op.getChunkSize().value_or(1);
470 int effectiveVecRank = (chunkSize == 1) ? 1 : 2;
471 ArrayRef<int64_t> shape = origResultTy.getShape();
472 if (llvm::any_of(
473 shape.take_front(origResultTy.getRank() - effectiveVecRank),
474 [](int64_t d) { return d != 1; }))
475 return rewriter.notifyMatchFailure(
476 op, "Only unit dimensions allowed for the leading "
477 "dimensions of the load vector!");
478
479 auto distResultTyOrFailure =
480 xegpu::getDistVecTypeBasedOnLaneLayout(layout, origResultTy);
481 if (failed(distResultTyOrFailure))
482 return rewriter.notifyMatchFailure(
483 op, "unable to compute expected lane vector type from lane layout");
484
485 VectorType distResultTy = distResultTyOrFailure.value();
486 VectorType distResultTy1D = VectorType::get({distResultTy.getNumElements()},
487 distResultTy.getElementType());
488
489 // Flatten offsets and mask to 1D to match the 1D result type.
490 Value distOffsets = adaptor.getOffsets();
491 auto distOffsetsTy = cast<VectorType>(distOffsets.getType());
492 VectorType offsetsTy1D = VectorType::get({distOffsetsTy.getNumElements()},
493 distOffsetsTy.getElementType());
494 distOffsets = castValueTo(
495 rewriter, cast<TypedValue<VectorType>>(distOffsets), offsetsTy1D);
496
497 Value distMask = adaptor.getMask();
498 auto distMaskTy = cast<VectorType>(distMask.getType());
499 VectorType maskTy1D = VectorType::get({distMaskTy.getNumElements()},
500 distMaskTy.getElementType());
501 distMask =
502 castValueTo(rewriter, cast<TypedValue<VectorType>>(distMask), maskTy1D);
503
504 Value distSource = adaptor.getSource();
505 auto newOp = xegpu::LoadGatherOp::create(
506 rewriter, op.getLoc(), distResultTy1D, distSource, distOffsets,
507 distMask, op.getChunkSizeAttr(), op.getL1HintAttr(), op.getL2HintAttr(),
508 op.getL3HintAttr(), /*layout=*/nullptr, /*contiguity=*/nullptr);
509
510 Value result = newOp->getResult(0);
511 if (distResultTy1D != distResultTy)
512 result = castValueTo(rewriter, cast<TypedValue<VectorType>>(result),
513 distResultTy);
514 rewriter.replaceOp(op, result);
515 return success();
516 }

519/// This pattern distributes a subgroup-level vector.reduction op to
520/// lane-level. This require shuffling the data across the lanes (using
521/// gpu::ShuffleOp) and reducing in stages until all lanes have the final
522/// result.
523struct SgToLaneVectorReduction
524 : public OpConversionPattern<vector::ReductionOp> {
525 using OpConversionPattern<vector::ReductionOp>::OpConversionPattern;
526
527 LogicalResult
528 matchAndRewrite(vector::ReductionOp op, OpAdaptor adaptor,
529 ConversionPatternRewriter &rewriter) const override {
530 auto layout = xegpu::getDistributeLayoutAttr(op.getVector());
531
532 // If no layout, nothing to do.
533 if (!layout || !layout.isForSubgroup())
534 return failure();
535
536 VectorType srcVecType = op.getSourceVectorType();
537 // Only rank 1 vectors supported.
538 if (srcVecType.getRank() != 1)
539 return rewriter.notifyMatchFailure(
540 op, "Only rank 1 reductions can be distributed.");
541 // Lane layout must have the same rank as the vector.
542 if (layout.getRank() != srcVecType.getRank())
543 return rewriter.notifyMatchFailure(
544 op, "Layout rank does not match vector rank.");
545
546 // Get the subgroup size from the layout.
547 int64_t sgSize = layout.getEffectiveLaneLayoutAsInt()[0];
548 const auto *uArch =
549 xegpu::uArch::getUArch(xegpu::getChipStr(op).value_or(""));
550 if (!uArch)
551 return rewriter.notifyMatchFailure(
552 op, "xegpu::ReductionOp require target attribute attached to "
553 "determine subgroup size");
554
555 // Only subgroup-sized vectors supported.
556 if (sgSize != uArch->getSubgroupSize() ||
557 srcVecType.getShape()[0] % sgSize != 0)
558 return rewriter.notifyMatchFailure(op,
559 "Invalid layout or reduction vector "
560 "dimension must match subgroup size.");
561
562 if (!op.getType().isIntOrFloat())
563 return rewriter.notifyMatchFailure(
564 op, "Reduction distribution currently only supports floats and "
565 "integer types.");

567 // Get the distributed vector (per lane portion).
568 Value laneValVec = adaptor.getVector();
569
570 // Distribute and reduce across lanes in the subgroup.
571 Value fullReduce = xegpu::subgroupReduction(
572 op.getLoc(), rewriter, laneValVec, op.getKind(), sgSize);
573
574 // If there's an accumulator, combine it with the reduced value.
575 if (adaptor.getAcc())
576 fullReduce = vector::makeArithReduction(
577 rewriter, op.getLoc(), op.getKind(), fullReduce, adaptor.getAcc());
578
579 rewriter.replaceOp(op, fullReduce);
580 return success();

584/// This pattern distributes a subgroup-level vector.multi_reduction op to
585/// lane-level only if the reduction is lane-local. This means that
586/// reduction dimension is not distributed to lanes and each lane does its own
587/// local reduction.
588struct SgToLaneMultiDimReduction
589 : public OpConversionPattern<vector::MultiDimReductionOp> {
590 using OpConversionPattern<vector::MultiDimReductionOp>::OpConversionPattern;
591
592 LogicalResult
593 matchAndRewrite(vector::MultiDimReductionOp op, OpAdaptor adaptor,
594 ConversionPatternRewriter &rewriter) const override {
595 Value result;
596 ArrayRef<int64_t> reductionDims = op.getReductionDims();
597 assert(reductionDims.size() == 1 &&
598 "Expecting single reduction dimension for subgroup multi "
599 "reduction op");
600 // For rank > 2, ensure leading dimensions are unit.
601 VectorType sourceType = op.getSourceVectorType();
602 int64_t rank = sourceType.getRank();
603 if (rank > 2) {
604 ArrayRef<int64_t> shape = sourceType.getShape();
605 if (llvm::any_of(shape.take_front(rank - 2),
606 [](int64_t d) { return d != 1; }))
607 return rewriter.notifyMatchFailure(
608 op, "only unit leading dimensions are supported for "
609 "multi_reduction with rank > 2");
610 }
611 // Handle scalar result: full reduction of a distributed vector to a
612 // scalar. First do a local vector reduction, then cross-lane shuffles.
613 if (op.getType().isIntOrFloat()) {
614 auto reductionDim = reductionDims[0];
615 VectorType origSourceType = op.getSourceVectorType();
616 int64_t reductionDimSize = origSourceType.getShape()[reductionDim];
617 // Local reduction to scalar, then cross-lane butterfly shuffles.
618 result =
619 xegpu::subgroupReduction(op.getLoc(), rewriter, adaptor.getSource(),
620 op.getKind(), reductionDimSize);
621 // Combine with accumulator if present.
622 if (adaptor.getAcc())
623 result = vector::makeArithReduction(rewriter, op.getLoc(), op.getKind(),
624 result, adaptor.getAcc());
625 } else if (isReductionLaneLocal(op)) {
626 // For lane-local reduction, lower to a sequence of vector.reduction ops
627 // over 1D slices extracted from the distributed source vector. This is
628 // required so we dont have 2D source vectors at xegpu-linearize.
629 auto reductionDim = reductionDims[0];
630 result = xegpu::lowerToVectorReductions(
631 cast<TypedValue<VectorType>>(adaptor.getSource()),
632 cast<TypedValue<VectorType>>(adaptor.getAcc()), op.getKind(),
633 reductionDim, op.getLoc(), rewriter);
634 } else {
635 auto reductionDim = reductionDims[0];
636 VectorType sourceType = op.getSourceVectorType();
637 int64_t reductionDimSize = sourceType.getShape()[reductionDim];
638 result = xegpu::lowerCrossLaneReductionToShuffles(
639 cast<TypedValue<VectorType>>(adaptor.getSource()),
640 cast<TypedValue<VectorType>>(adaptor.getAcc()), op.getKind(),
641 reductionDim, reductionDimSize, op.getLoc(), rewriter);
642 }
643 rewriter.replaceOp(op, result);
644 return success();
645 }

787/// Distributes a subgroup-level vector.create_mask or vector.constant_mask op
788/// to lane-level. Uses `computeDistributedCoords()` to obtain the
789/// coordinates each lane owns, then compares each coordinate against the
790/// original mask bounds using `arith.cmpi slt`. The per-element boolean
791/// results are assembled into the distributed mask vector.
792///
793/// For multi-dimensional masks, the element is in-bounds when ALL dimensions
794/// satisfy `coord[i] < bound[i]`.
795///
796/// Example (1D):
797/// layout = #xegpu.layout<lane_layout = [16], lane_data = [1]>
798/// %mask = vector.create_mask %m0 : vector<16xi1>
799/// For lane k, computeDistributedCoords gives coord = [k], so:
800/// %in_bounds = arith.cmpi slt, %coord, %m0 → i1
801/// %mask = vector.broadcast %in_bounds : i1 to vector<1xi1>
802///
803/// Example (2D):
804/// layout = #xegpu.layout<lane_layout = [8, 2], lane_data = [1, 1]>
805/// %mask = vector.create_mask %m0, %m1 : vector<8x4xi1>
806/// Each WI owns a 1x2 slice. computeDistributedCoords returns 2 coords:
807/// [[r0, c0], [r0, c1]]
808/// For each coord: in_bounds = (r < m0) && (c < m1)
809/// %mask = vector.from_elements %bit0, %bit1 : vector<1x2xi1>
810template <typename OpType,
811 typename = std::enable_if_t<llvm::is_one_of<
812 OpType, vector::CreateMaskOp, vector::ConstantMaskOp>::value>>
813struct SgToLaneCreateMask : public OpConversionPattern<OpType> {
814 using OpConversionPattern<OpType>::OpConversionPattern;
815
816 LogicalResult
817 matchAndRewrite(OpType op, typename OpType::Adaptor adaptor,
818 ConversionPatternRewriter &rewriter) const override {
819 xegpu::DistributeLayoutAttr layout =
820 xegpu::getTemporaryLayout(op->getOpResult(0));
821 if (!layout || !layout.isForSubgroup())
822 return rewriter.notifyMatchFailure(
823 op, "operation result does not have subgroup distribute layout");
824
825 VectorType origType = op.getType();
826 FailureOr<VectorType> distTypeOrFailure =
827 getDistVecTypeBasedOnLaneLayout(layout, origType);
828 if (failed(distTypeOrFailure))
829 return rewriter.notifyMatchFailure(
830 op, "unable to compute lane vector type from the layout");
831
832 VectorType distType = distTypeOrFailure.value();
833 Location loc = op.getLoc();
834
835 // Materialize the original mask bounds as Values.
836 SmallVector<Value> origBounds;
837 if constexpr (std::is_same_v<OpType, vector::CreateMaskOp>) {
838 origBounds.append(op.getOperands().begin(), op.getOperands().end());
839 } else {
840 auto dimSizes = op.getMaskDimSizesAttr().asArrayRef();
841 for (auto dimSize : dimSizes)
842 origBounds.push_back(
843 arith::ConstantIndexOp::create(rewriter, loc, dimSize).getResult());
844 }
845
846 ArrayRef<int64_t> origShape = origType.getShape();
847
848 // Use computeDistributedCoords to get the coordinates each WI owns.
849 Value laneId = gpu::LaneIdOp::create(rewriter, loc, rewriter.getIndexType(),
850 /*upperBound=*/mlir::IntegerAttr());
851 auto maybeCoordsVec =
852 layout.computeDistributedCoords(rewriter, loc, laneId, origShape);
853 if (failed(maybeCoordsVec))
854 return rewriter.notifyMatchFailure(
855 op, "failed to compute distributed coordinates from layout");
856
857 SmallVector<SmallVector<Value>> coordsVec = maybeCoordsVec.value();
858 int64_t numElements = distType.getNumElements();
859 assert(static_cast<int64_t>(coordsVec.size()) == numElements &&
860 "number of coordinate sets must match number of distributed "
861 "elements");
862
863 // For each element, compare all coordinates against bounds.
864 Value trueVal =
865 arith::ConstantIntOp::create(rewriter, loc, /*value=*/1, /*width=*/1);
866 SmallVector<Value> maskBits;
867 for (auto &coords : coordsVec) {
868 Value inBounds = trueVal;
869 for (size_t i = 0; i < coords.size(); ++i) {
870 Value cmp = arith::CmpIOp::create(
871 rewriter, loc, arith::CmpIPredicate::slt, coords[i], origBounds[i]);
872 inBounds = arith::AndIOp::create(rewriter, loc, inBounds, cmp);
873 }
874 maskBits.push_back(inBounds);
875 }
876
877 // Build the distributed mask vector.
878 Value result;
879 if (numElements == 1) {
880 result =
881 vector::BroadcastOp::create(rewriter, loc, distType, maskBits[0]);
882 } else {
883 result =
884 vector::FromElementsOp::create(rewriter, loc, distType, maskBits);
885 }
886 rewriter.replaceOp(op, result);
887 return success();
888 }
889};

1039 op.getL1HintAttr(), op.getL2HintAttr(),
1040 op.getL3HintAttr(), /*layout=*/nullptr,

1058 xegpu::getTemporaryLayout(op->getResult(0));
1059 if (!resultLayout || !resultLayout.isForSubgroup())
1060 return rewriter.notifyMatchFailure(
1061 op, "the result vector of the step op lacks subgroup layout");
1062
1063 auto loc = op.getLoc();
1064 auto stepResultVecTy = op.getResult().getType();
1065 auto laneShapeOrFailure =
1066 xegpu::getDistVecTypeBasedOnLaneLayout(resultLayout, stepResultVecTy);
1067 if (failed(laneShapeOrFailure))

1071
1072 Value laneId = gpu::LaneIdOp::create(rewriter, loc, rewriter.getIndexType(),
1073 /*upperBound=*/mlir::IntegerAttr());

1075 rewriter, loc, laneId, stepResultVecTy.getShape());
1076 if (failed(laneDataBlockCoords))
1077 return rewriter.notifyMatchFailure(
1078 op, "failed to compute lane data block coordinates");
1079
1080 auto laneDataBlockCoordsVec = laneDataBlockCoords.value();
1081 auto laneDataBlockLength = resultLayout.getEffectiveLaneDataAsInt()[0];
1082 assert(static_cast<int64_t>(laneDataBlockCoordsVec.size()) ==

1097 stepVals.push_back(arith::AddIOp::create(
1098 rewriter, loc, laneDataBlockStartCoord, offset));
1099 }
1100 }
1101 assert(static_cast<int64_t>(stepVals.size()) == newVecTy.getNumElements() &&
1102 "Expecting the number of step values to match the number of "
1103 "elements in the vector");
1104 auto stepOpVal =
1105 vector::FromElementsOp::create(rewriter, loc, newVecTy, stepVals);
1106 rewriter.replaceOp(op, stepOpVal);

1272/// This pattern distributes a subgroup-level `vector.broadcast` op to
1273/// lane-level. The pattern supports three cases:
1274///
1275/// 1) Broadcast a low-rank vector to high-rank vector: The low-rank input
1276/// vector must have a slice layout of the result. If the distributed source
1277/// and target vector types are identical, this lowers to a no-op; otherwise,
1278/// it remains a broadcast but operates on distributed vectors.
1279///
1280/// 2) Broadcast a same-rank vector with identical layouts for source and
1281/// target: The source vector must have unit dimensions, and lane_data must
1282/// be unit size for those unit dims. This always lowers to a no-op.
1283///
1284/// 3) Broadcast a scalar with no layout: This always lowers to a broadcast
1285/// from scalar to distributed result type.
1286///
1287/// Example 1 (low-rank to high-rank broadcast):
1288/// ```
1289/// %0 = "some_op"() {layout_result_0 =
1290/// #xegpu.slice<#xegpu.layout<lane_layout = [1, 16], lane_data = [1, 1]>,
1291/// dims = [0]>} : () -> vector<16xf16>
1292/// %1 = vector.broadcast %0 {layout_result_0 =
1293/// #xegpu.layout<lane_layout = [1, 16], lane_data = [1, 1]>}
1294/// : vector<16xf16> to vector<16x16xf16>
1295/// ```
1296/// is distributed to:
1297/// ```
1298/// %0 = "some_op"() : () -> vector<1xf16>
1299/// %1 = vector.broadcast %0 : vector<1xf16> to vector<16x1xf16>
1300/// ```
1301///
1302/// Example 2 (same-rank broadcast, no-op):
1303/// ```
1304/// %0 = "some_op"() {layout_result_0 =
1305/// #xegpu.layout<lane_layout = [1, 16], lane_data = [1, 1]>}
1306/// : () -> vector<16x1xf16>
1307/// %1 = vector.broadcast %0 {layout_result_0 =
1308/// #xegpu.layout<lane_layout = [1, 16], lane_data = [1, 1]>}
1309/// : vector<16x1xf16> to vector<16x16xf16>
1310/// ```
1311/// is distributed to (no-op, source already matches distributed result type):
1312/// ```
1313/// %0 = "some_op"() : () -> vector<16x1xf16>
1314/// // broadcast is eliminated, %0 is used directly
1315/// ```
1316///
1317/// Example 3 (scalar to vector broadcast):
1318/// ```
1319/// %0 = "some_op"() : () -> f16
1320/// %1 = vector.broadcast %0 {layout_result_0 =
1321/// #xegpu.layout<lane_layout = [1, 16], lane_data = [1, 1]>}
1322/// : f16 to vector<16x16xf16>
1323/// ```
1324/// is distributed to:
1325/// ```
1326/// %0 = "some_op"() : f16
1327/// %1 = vector.broadcast %0 : f16 to vector<16x1xf16>
1328/// ```
1329struct SgToLaneBroadcast : public OpConversionPattern<vector::BroadcastOp> {
1330 using OpConversionPattern<vector::BroadcastOp>::OpConversionPattern;
1331
1332 LogicalResult
1333 matchAndRewrite(vector::BroadcastOp op, OpAdaptor adaptor,
1334 ConversionPatternRewriter &rewriter) const override {
1335 xegpu::DistributeLayoutAttr resultLayout =
1336 xegpu::getTemporaryLayout(cast<OpResult>(op.getResult()));
1337 if (!resultLayout || !resultLayout.isForSubgroup())
1338 return rewriter.notifyMatchFailure(
1339 op, "result does not have subgroup distribute layout");
1340
1341 VectorType destType = op.getResultVectorType();
1342 VectorType sourceType = dyn_cast<VectorType>(op.getSourceType());
1343
1344 xegpu::DistributeLayoutAttr sourceLayout =
1345 xegpu::getTemporaryLayout(op->getOpOperand(0));
1346
1347 if (sourceType) {
1348 int64_t rankDiff = destType.getRank() - sourceType.getRank();
1349 if (rankDiff > 0) {
1350 // Case 1: Low-rank to high-rank broadcast.
1351 if (!sourceLayout || !sourceLayout.isSliceOf(resultLayout))
1352 op.emitWarning(
1353 "broadcast source layout must be a slice of result layout");
1354 } else if (rankDiff == 0) {
1355 // Case 2: Same-rank broadcast.
1356 auto broadcastUnitDimsSet = op.computeBroadcastedUnitDims();
1357 SmallVector<int64_t> broadcastUnitDims(broadcastUnitDimsSet.begin(),
1358 broadcastUnitDimsSet.end());
1359 assert(sourceLayout.isEqualTo(
1360 sourceLayout.setUnitDimData(broadcastUnitDims)) &&
1361 "The sg_data for unit dimensions should be set as 1");
1362 sourceLayout = sourceLayout.setUnitDimLayout(broadcastUnitDims);
1363 }
1364 } else {
1365 // Case 3: Scalar to vector broadcast.
1366 if (sourceLayout)
1367 return rewriter.notifyMatchFailure(
1368 op, "broadcast from scalar must not have a layout attribute");
1369 }
1370
1371 auto destDistType =
1372 xegpu::getDistVecTypeBasedOnLaneLayout(resultLayout, destType);
1373 if (failed(destDistType))
1374 return rewriter.notifyMatchFailure(
1375 op, "failed to distribute the result vector type");
1376
1377 Value source = adaptor.getSource();
1378 // If the adapted source already matches the dest dist type, it's a no-op.
1379 if (source.getType() == destDistType.value()) {
1380 rewriter.replaceOp(op, source);
1381 return success();
1382 }
1383
1384 auto newOp = vector::BroadcastOp::create(rewriter, op.getLoc(),
1385 destDistType.value(), source);
1386 rewriter.replaceOp(op, newOp);
1387 return success();
1388 }

1522/// Redistributes `src` for a `convert_layout` that changes only the
1523/// `lane_layout` along the outer (distributed) dimension, shrinking it from
1524/// `currentLaneNum` to `targetLaneNum` lanes (a partial-subgroup
1525/// distribution). Because the data is no longer replicated across all lanes,
1526/// each surviving lane must gather the values that previously lived in the
1527/// lanes that are dropped. The values are gathered with `gpu.shuffle` and
1528/// concatenated with the lane-local data using `vector.shuffle`, which doubles
1529/// the distributed outer dimension when the lane count is halved.
1530///
1531/// Only halving the lane count (a factor of two) is currently supported.
1532/// Returns the redistributed value on success, or failure if `src` cannot be
1533/// shuffled (e.g. it is not a rank-2 vector or its bit width is not a multiple
1534/// of 32).
1535static FailureOr<Value>
1536shuffleDataAsLaneLayoutChange(ConversionPatternRewriter &rewriter, Location loc,
1537 Value src, int64_t currentLaneNum,
1538 int64_t targetLaneNum) {
1539 VectorType srcTy = dyn_cast<VectorType>(src.getType());
1540 if (!srcTy || srcTy.getRank() != 2)
1541 return failure();
1542 // Only halving the lane count (factor of two) is supported for now.
1543 if (targetLaneNum <= 0 || currentLaneNum != targetLaneNum * 2)
1544 return failure();
1545 // gpu.shuffle operates on i32, so the data must be a multiple of 32 bits.
1546 int64_t vectorBitWidth =
1547 srcTy.getNumElements() * srcTy.getElementTypeBitWidth();
1548 if (vectorBitWidth % 32 != 0)
1549 return failure();
1550
1551 // A vector cannot be shuffled across lanes directly:
1552 // -- cast the source to a 1D vector of i32
1553 // -- create a temp 1D vector of i32 initialized to zero
1554 // -- for each i32 element:
1555 // ---- extract it from the source bundle
1556 // ---- gpu.shuffle to gather the value from the partner lane
1557 // ---- insert it into the temp bundle
1558 // -- cast the temp back to the source vector type
1559 // -- vector.shuffle the source and temp to concatenate along the outer dim
1560 Type shuffleElemTy = rewriter.getI32Type();
1561 int64_t numShuffles = vectorBitWidth / 32;
1562 VectorType shuffleBundleTy = VectorType::get({numShuffles}, shuffleElemTy);
1563 // Initialize temp to zero.
1564 Value temp = arith::ConstantOp::create(
1565 rewriter, loc,
1566 DenseElementsAttr::get(shuffleBundleTy,
1567 IntegerAttr::get(shuffleElemTy, 0)));
1568 VectorType flatSrcTy =
1569 VectorType::get({srcTy.getNumElements()}, srcTy.getElementType());
1570 Value flatSrc = vector::ShapeCastOp::create(rewriter, loc, flatSrcTy, src);
1571 Value shuffleBundle =
1572 vector::BitCastOp::create(rewriter, loc, shuffleBundleTy, flatSrc);
1573 for (int64_t i = 0; i < numShuffles; i++) {
1574 Value shuffleElem =
1575 vector::ExtractOp::create(rewriter, loc, shuffleBundle, i);
1576 shuffleElem = gpu::ShuffleOp::create(rewriter, loc, shuffleElem, 0,
1577 targetLaneNum, gpu::ShuffleMode::UP)
1578 .getResult(0);
1579 temp = vector::InsertOp::create(rewriter, loc, shuffleElem, temp, i);
1580 }
1581 temp = vector::BitCastOp::create(rewriter, loc, flatSrcTy, temp);
1582 temp = vector::ShapeCastOp::create(rewriter, loc, srcTy, temp);
1583
1584 // Concatenate the lane-local and gathered data along the outer dimension.
1585 SmallVector<int64_t> indices(srcTy.getShape()[0] * 2);
1586 std::iota(indices.begin(), indices.end(), 0);
1587 Value res = vector::ShuffleOp::create(rewriter, loc, src, temp, indices);
1588 return res;
1589}

1591/// Folds a subgroup-level ConvertLayout op with compatible lane layouts.
1592struct SgToLaneConvertLayout
1593 : public OpConversionPattern<xegpu::ConvertLayoutOp> {
1594 using OpConversionPattern<xegpu::ConvertLayoutOp>::OpConversionPattern;
1595
1596 LogicalResult
1597 matchAndRewrite(xegpu::ConvertLayoutOp op, OpAdaptor adaptor,
1598 ConversionPatternRewriter &rewriter) const override {
1599 auto inputLayout = op.getInputLayoutAttr();
1600 auto targetLayout = op.getTargetLayoutAttr();
1601 Type valType = op.getResult().getType();
1602
1603 if (valType.isIntOrFloat()) {
1604 rewriter.replaceOp(op, op.getSource());
1605 return success();
1606 }
1607
1608 auto resShape = cast<VectorType>(valType).getShape();
1609 SmallVector<int64_t> resShapeVec(resShape.begin(), resShape.end());
1610
1611 // Equivalent layouts: the convert_layout is a no-op and folds to its
1612 // source.
1613 if (inputLayout.isCompatibleWith(targetLayout, resShapeVec,
1614 xegpu::LayoutKind::Lane)) {
1615 rewriter.replaceOp(op, adaptor.getSource());
1616 return success();
1617 }
1618
1619 // Handle the special case where the conversion redistributes a value
1620 // across a fraction of the subgroup: the lane_layout shrinks along the
1621 // outer (distributed) dimension while lane_data stays the same. Only a
1622 // pure outer-dimension lane_layout change is supported, so the inner
1623 // lane_layout must be unit (making the outer dim the only distributed one)
1624 // and the outer lane_layout must be genuinely distributed (> 1), which
1625 // also rules out the degenerate [1, 1] layout.
1626 if (inputLayout.getEffectiveOrderAsInt() ==
1627 targetLayout.getEffectiveOrderAsInt() &&
1628 inputLayout.getRank() == 2 && targetLayout.getRank() == 2) {
1629 auto laneLayout = inputLayout.getEffectiveLaneLayoutAsInt();
1630 auto targetLaneLayout = targetLayout.getEffectiveLaneLayoutAsInt();
1631 auto laneData = inputLayout.getEffectiveLaneDataAsInt();
1632 auto targetLaneData = targetLayout.getEffectiveLaneDataAsInt();
1633 if (laneLayout.size() == 2 && targetLaneLayout.size() == 2 &&
1634 laneData == targetLaneData && laneLayout[1] == 1 &&
1635 targetLaneLayout[1] == 1 && laneLayout[0] > 1 &&
1636 laneLayout[0] != targetLaneLayout[0]) {
1637 FailureOr<Value> res = shuffleDataAsLaneLayoutChange(
1638 rewriter, op.getLoc(), adaptor.getSource(), laneLayout[0],
1639 targetLaneLayout[0]);
1640 if (succeeded(res)) {
1641 rewriter.replaceOp(op, *res);
1642 return success();
1643 }
1644 }
1645 }
1646
1647 return rewriter.notifyMatchFailure(
1648 op, "lowering incompatible convert_layout not yet supported");
1649 }
