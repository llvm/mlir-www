105 /// Returns the iterator of a certain type.
106 inline unsigned getNumIterators(StringRef name, ArrayAttr iteratorTypes) {
107  auto names = getAllIteratorTypeNames();
108  (void)names;
109  assert(llvm::is_contained(names, name));
110  return llvm::count_if(iteratorTypes, [name](Attribute a) {
111  return a.cast<StringAttr>().getValue() == name;
112  });
113 }
114 
115 inline unsigned getNumIterators(ArrayAttr iteratorTypes) {
116  unsigned res = 0;
117  for (auto n : getAllIteratorTypeNames())
118  res += getNumIterators(n, iteratorTypes);
119  return res;
120 }
