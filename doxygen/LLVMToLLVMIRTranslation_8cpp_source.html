33 static llvm::FastMathFlags getFastmathFlags(FastmathFlagsInterface &op) {
34  using llvmFMF = llvm::FastMathFlags;
35  using FuncT = void (llvmFMF::*)(bool);
36  const std::pair<FastmathFlags, FuncT> handlers[] = {
37  // clang-format off
38  {FastmathFlags::nnan, &llvmFMF::setNoNaNs},
39  {FastmathFlags::ninf, &llvmFMF::setNoInfs},
40  {FastmathFlags::nsz, &llvmFMF::setNoSignedZeros},
41  {FastmathFlags::arcp, &llvmFMF::setAllowReciprocal},
42  {FastmathFlags::contract, &llvmFMF::setAllowContract},
43  {FastmathFlags::afn, &llvmFMF::setApproxFunc},
44  {FastmathFlags::reassoc, &llvmFMF::setAllowReassoc},
45  // clang-format on
46  };
47  llvm::FastMathFlags ret;
48  ::mlir::LLVM::FastmathFlags fmfMlir = op.getFastmathAttr().getValue();
49  for (auto it : handlers)
50  if (bitEnumContainsAll(fmfMlir, it.first))
51  (ret.*(it.second))(true);
52  return ret;
53 }

70 /// Get the declaration of an overloaded llvm intrinsic. First we get the
71 /// overloaded argument types and/or result type from the CallIntrinsicOp, and
72 /// then use those to get the correct declaration of the overloaded intrinsic.
73 static FailureOr<llvm::Function *>
74 getOverloadedDeclaration(CallIntrinsicOp op, llvm::Intrinsic::ID id,
75  llvm::Module *module,
76  LLVM::ModuleTranslation &moduleTranslation) {
77  SmallVector<llvm::Type *, 8> allArgTys;
78  for (Type type : op->getOperandTypes())
79  allArgTys.push_back(moduleTranslation.convertType(type));
80 
81  llvm::Type *resTy;
82  if (op.getNumResults() == 0)
83  resTy = llvm::Type::getVoidTy(module->getContext());
84  else
85  resTy = moduleTranslation.convertType(op.getResult(0).getType());
86 
87  // ATM we do not support variadic intrinsics.
88  llvm::FunctionType *ft = llvm::FunctionType::get(resTy, allArgTys, false);
89 
90  SmallVector<llvm::Intrinsic::IITDescriptor, 8> table;
91  getIntrinsicInfoTableEntries(id, table);
92  ArrayRef<llvm::Intrinsic::IITDescriptor> tableRef = table;
93 
94  SmallVector<llvm::Type *, 8> overloadedArgTys;
95  if (llvm::Intrinsic::matchIntrinsicSignature(ft, tableRef,
96  overloadedArgTys) !=
97  llvm::Intrinsic::MatchIntrinsicTypesResult::MatchIntrinsicTypes_Match) {
98  return mlir::emitError(op.getLoc(), "call intrinsic signature ")
99  << diagStr(ft) << " to overloaded intrinsic " << op.getIntrinAttr()
100  << " does not match any of the overloads";
101  }
102 
103  ArrayRef<llvm::Type *> overloadedArgTysRef = overloadedArgTys;
104  return llvm::Intrinsic::getOrInsertDeclaration(module, id,
105  overloadedArgTysRef);
106 }

120  LLVM::ModuleTranslation &moduleTranslation) {
121  SmallVector<llvm::OperandBundleDef> bundles;

140 /// Builder for LLVM_CallIntrinsicOp
141 static LogicalResult
142 convertCallLLVMIntrinsicOp(CallIntrinsicOp op, llvm::IRBuilderBase &builder,
143  LLVM::ModuleTranslation &moduleTranslation) {
144  llvm::Module *module = builder.GetInsertBlock()->getModule();
145  llvm::Intrinsic::ID id =
146  llvm::Intrinsic::lookupIntrinsicID(op.getIntrinAttr());
147  if (!id)
148  return mlir::emitError(op.getLoc(), "could not find LLVM intrinsic: ")
149  << op.getIntrinAttr();
150 
151  llvm::Function *fn = nullptr;
152  if (llvm::Intrinsic::isOverloaded(id)) {
153  auto fnOrFailure =
154  getOverloadedDeclaration(op, id, module, moduleTranslation);
155  if (failed(fnOrFailure))
156  return failure();
157  fn = *fnOrFailure;
158  } else {
159  fn = llvm::Intrinsic::getOrInsertDeclaration(module, id, {});
160  }
161 
162  // Check the result type of the call.
163  const llvm::Type *intrinType =
164  op.getNumResults() == 0
165  ? llvm::Type::getVoidTy(module->getContext())
166  : moduleTranslation.convertType(op.getResultTypes().front());
167  if (intrinType != fn->getReturnType()) {
168  return mlir::emitError(op.getLoc(), "intrinsic call returns ")
169  << diagStr(intrinType) << " but " << op.getIntrinAttr()
170  << " actually returns " << diagStr(fn->getReturnType());
171  }
172 
173  // Check the argument types of the call. If the function is variadic, check
174  // the subrange of required arguments.
175  if (!fn->getFunctionType()->isVarArg() &&
176  op.getArgs().size() != fn->arg_size()) {
177  return mlir::emitError(op.getLoc(), "intrinsic call has ")
178  << op.getArgs().size() << " operands but " << op.getIntrinAttr()
179  << " expects " << fn->arg_size();
180  }
181  if (fn->getFunctionType()->isVarArg() &&
182  op.getArgs().size() < fn->arg_size()) {
183  return mlir::emitError(op.getLoc(), "intrinsic call has ")
184  << op.getArgs().size() << " operands but variadic "
185  << op.getIntrinAttr() << " expects at least " << fn->arg_size();
186  }
187  // Check the arguments up to the number the function requires.
188  for (unsigned i = 0, e = fn->arg_size(); i != e; ++i) {
189  const llvm::Type *expected = fn->getArg(i)->getType();
190  const llvm::Type *actual =
191  moduleTranslation.convertType(op.getOperandTypes()[i]);
192  if (actual != expected) {
193  return mlir::emitError(op.getLoc(), "intrinsic call operand #")
194  << i << " has type " << diagStr(actual) << " but "
195  << op.getIntrinAttr() << " expects " << diagStr(expected);
196  }
197  }
198 
199  FastmathFlagsInterface itf = op;
200  builder.setFastMathFlags(getFastmathFlags(itf));
201 
202  auto *inst = builder.CreateCall(
203  fn, moduleTranslation.lookupValues(op.getArgs()),
204  convertOperandBundles(op.getOpBundleOperands(), op.getOpBundleTags(),
205  moduleTranslation));
206 
207  if (failed(moduleTranslation.convertArgAndResultAttrs(op, inst)))
208  return failure();
209 
210  if (op.getNumResults() == 1)
211  moduleTranslation.mapValue(op->getResults().front()) = inst;
212  return success();
213 }

218  llvm::Module *llvmModule = moduleTranslation.getLLVMModule();

233 static llvm::Metadata *
234 convertModuleFlagValue(StringRef key, ArrayAttr arrayAttr,
235  llvm::IRBuilderBase &builder,
236  LLVM::ModuleTranslation &moduleTranslation) {
237  llvm::LLVMContext &context = builder.getContext();
238  llvm::MDBuilder mdb(context);
239  SmallVector<llvm::Metadata *> nodes;
240 
241  if (key == LLVMDialect::getModuleFlagKeyCGProfileName()) {
242  for (auto entry : arrayAttr.getAsRange<ModuleFlagCGProfileEntryAttr>()) {
243  llvm::Metadata *fromMetadata =
244  entry.getFrom()
245  ? llvm::ValueAsMetadata::get(moduleTranslation.lookupFunction(
246  entry.getFrom().getValue()))
247  : nullptr;
248  llvm::Metadata *toMetadata =
249  entry.getTo()
250  ? llvm::ValueAsMetadata::get(
251  moduleTranslation.lookupFunction(entry.getTo().getValue()))
252  : nullptr;
253 
254  llvm::Metadata *vals[] = {
255  fromMetadata, toMetadata,
256  mdb.createConstant(llvm::ConstantInt::get(
257  llvm::Type::getInt64Ty(context), entry.getCount()))};
258  nodes.push_back(llvm::MDNode::get(context, vals));
259  }
260  return llvm::MDTuple::getDistinct(context, nodes);
261  }
262  return nullptr;
263 }

265 static llvm::Metadata *convertModuleFlagProfileSummaryAttr(
266  StringRef key, ModuleFlagProfileSummaryAttr summaryAttr,
267  llvm::IRBuilderBase &builder, LLVM::ModuleTranslation &moduleTranslation) {
268  llvm::LLVMContext &context = builder.getContext();
269  llvm::MDBuilder mdb(context);
270 
271  auto getIntTuple = [&](StringRef key, uint64_t val) -> llvm::MDTuple * {
272  SmallVector<llvm::Metadata *> tupleNodes{
273  mdb.createString(key), mdb.createConstant(llvm::ConstantInt::get(
274  llvm::Type::getInt64Ty(context), val))};
275  return llvm::MDTuple::get(context, tupleNodes);
276  };
277 
278  SmallVector<llvm::Metadata *> fmtNode{
279  mdb.createString("ProfileFormat"),
280  mdb.createString(
281  stringifyProfileSummaryFormatKind(summaryAttr.getFormat()))};
282 
283  SmallVector<llvm::Metadata *> vals = {
284  llvm::MDTuple::get(context, fmtNode),
285  getIntTuple("TotalCount", summaryAttr.getTotalCount()),
286  getIntTuple("MaxCount", summaryAttr.getMaxCount()),
287  getIntTuple("MaxInternalCount", summaryAttr.getMaxInternalCount()),
288  getIntTuple("MaxFunctionCount", summaryAttr.getMaxFunctionCount()),
289  getIntTuple("NumCounts", summaryAttr.getNumCounts()),
290  getIntTuple("NumFunctions", summaryAttr.getNumFunctions()),
291  };
292 
293  if (summaryAttr.getIsPartialProfile())
294  vals.push_back(
295  getIntTuple("IsPartialProfile", *summaryAttr.getIsPartialProfile()));
296 
297  if (summaryAttr.getPartialProfileRatio()) {
298  SmallVector<llvm::Metadata *> tupleNodes{
299  mdb.createString("PartialProfileRatio"),
300  mdb.createConstant(llvm::ConstantFP::get(
301  llvm::Type::getDoubleTy(context),
302  summaryAttr.getPartialProfileRatio().getValue()))};
303  vals.push_back(llvm::MDTuple::get(context, tupleNodes));
304  }
305 
306  SmallVector<llvm::Metadata *> detailedEntries;
307  llvm::Type *llvmInt64Type = llvm::Type::getInt64Ty(context);
308  for (ModuleFlagProfileSummaryDetailedAttr detailedEntry :
309  summaryAttr.getDetailedSummary()) {
310  SmallVector<llvm::Metadata *> tupleNodes{
311  mdb.createConstant(
312  llvm::ConstantInt::get(llvmInt64Type, detailedEntry.getCutOff())),
313  mdb.createConstant(
314  llvm::ConstantInt::get(llvmInt64Type, detailedEntry.getMinCount())),
315  mdb.createConstant(llvm::ConstantInt::get(
316  llvmInt64Type, detailedEntry.getNumCounts()))};
317  detailedEntries.push_back(llvm::MDTuple::get(context, tupleNodes));
318  }
319  SmallVector<llvm::Metadata *> detailedSummary{
320  mdb.createString("DetailedSummary"),
321  llvm::MDTuple::get(context, detailedEntries)};
322  vals.push_back(llvm::MDTuple::get(context, detailedSummary));
323 
324  return llvm::MDNode::get(context, vals);
325 }

327 static void convertModuleFlagsOp(ArrayAttr flags, llvm::IRBuilderBase &builder,
328  LLVM::ModuleTranslation &moduleTranslation) {
329  llvm::Module *llvmModule = moduleTranslation.getLLVMModule();
330  for (auto flagAttr : flags.getAsRange<ModuleFlagAttr>()) {
331  llvm::Metadata *valueMetadata =
332  llvm::TypeSwitch<Attribute, llvm::Metadata *>(flagAttr.getValue())
333  .Case<StringAttr>([&](auto strAttr) {
334  return llvm::MDString::get(builder.getContext(),
335  strAttr.getValue());
336  })
337  .Case<IntegerAttr>([&](auto intAttr) {
338  return llvm::ConstantAsMetadata::get(llvm::ConstantInt::get(
339  llvm::Type::getInt32Ty(builder.getContext()),
340  intAttr.getInt()));
341  })
342  .Case<ArrayAttr>([&](auto arrayAttr) {
343  return convertModuleFlagValue(flagAttr.getKey().getValue(),
344  arrayAttr, builder,
345  moduleTranslation);
346  })
347  .Case([&](ModuleFlagProfileSummaryAttr summaryAttr) {
348  return convertModuleFlagProfileSummaryAttr(
349  flagAttr.getKey().getValue(), summaryAttr, builder,
350  moduleTranslation);
351  })
352  .Default([](auto) { return nullptr; });
353 
354  assert(valueMetadata && "expected valid metadata");
355  llvmModule->addModuleFlag(
356  convertModFlagBehaviorToLLVM(flagAttr.getBehavior()),
357  flagAttr.getKey().getValue(), valueMetadata);
358  }
359 }

361 static LogicalResult
362 convertOperationImpl(Operation &opInst, llvm::IRBuilderBase &builder,
363  LLVM::ModuleTranslation &moduleTranslation) {
364 
365  llvm::IRBuilder<>::FastMathFlagGuard fmfGuard(builder);
366  if (auto fmf = dyn_cast<FastmathFlagsInterface>(opInst))
367  builder.setFastMathFlags(getFastmathFlags(fmf));
368 
369 #include "mlir/Dialect/LLVMIR/LLVMConversions.inc"
370 #include "mlir/Dialect/LLVMIR/LLVMIntrinsicConversions.inc"
371 
372  // Emit function calls. If the "callee" attribute is present, this is a
373  // direct function call and we also need to look up the remapped function
374  // itself. Otherwise, this is an indirect call and the callee is the first
375  // operand, look it up as a normal value.
376  if (auto callOp = dyn_cast<LLVM::CallOp>(opInst)) {
377  auto operands = moduleTranslation.lookupValues(callOp.getCalleeOperands());
378  SmallVector<llvm::OperandBundleDef> opBundles =
379  convertOperandBundles(callOp.getOpBundleOperands(),
380  callOp.getOpBundleTags(), moduleTranslation);
381  ArrayRef<llvm::Value *> operandsRef(operands);
382  llvm::CallInst *call;
383  if (auto attr = callOp.getCalleeAttr()) {
384  if (llvm::Function *function =
385  moduleTranslation.lookupFunction(attr.getValue())) {
386  call = builder.CreateCall(function, operandsRef, opBundles);
387  } else {
388  Operation *moduleOp = parentLLVMModule(&opInst);
389  Operation *ifuncOp =
390  moduleTranslation.symbolTable().lookupSymbolIn(moduleOp, attr);
391  llvm::GlobalValue *ifunc = moduleTranslation.lookupIFunc(ifuncOp);
392  llvm::FunctionType *calleeType = llvm::cast<llvm::FunctionType>(
393  moduleTranslation.convertType(callOp.getCalleeFunctionType()));
394  call = builder.CreateCall(calleeType, ifunc, operandsRef, opBundles);
395  }
396  } else {
397  llvm::FunctionType *calleeType = llvm::cast<llvm::FunctionType>(
398  moduleTranslation.convertType(callOp.getCalleeFunctionType()));
399  call = builder.CreateCall(calleeType, operandsRef.front(),
400  operandsRef.drop_front(), opBundles);
401  }
402  call->setCallingConv(convertCConvToLLVM(callOp.getCConv()));
403  call->setTailCallKind(convertTailCallKindToLLVM(callOp.getTailCallKind()));
404  if (callOp.getConvergentAttr())
405  call->addFnAttr(llvm::Attribute::Convergent);
406  if (callOp.getNoUnwindAttr())
407  call->addFnAttr(llvm::Attribute::NoUnwind);
408  if (callOp.getWillReturnAttr())
409  call->addFnAttr(llvm::Attribute::WillReturn);
410  if (callOp.getNoInlineAttr())
411  call->addFnAttr(llvm::Attribute::NoInline);
412  if (callOp.getAlwaysInlineAttr())
413  call->addFnAttr(llvm::Attribute::AlwaysInline);
414  if (callOp.getInlineHintAttr())
415  call->addFnAttr(llvm::Attribute::InlineHint);
416 
417  if (failed(moduleTranslation.convertArgAndResultAttrs(callOp, call)))
418  return failure();
419 
420  if (MemoryEffectsAttr memAttr = callOp.getMemoryEffectsAttr()) {
421  llvm::MemoryEffects memEffects =
422  llvm::MemoryEffects(llvm::MemoryEffects::Location::ArgMem,
423  convertModRefInfoToLLVM(memAttr.getArgMem())) |
424  llvm::MemoryEffects(
425  llvm::MemoryEffects::Location::InaccessibleMem,
426  convertModRefInfoToLLVM(memAttr.getInaccessibleMem())) |
427  llvm::MemoryEffects(llvm::MemoryEffects::Location::Other,
428  convertModRefInfoToLLVM(memAttr.getOther()));
429  call->setMemoryEffects(memEffects);
430  }
431 
432  moduleTranslation.setAccessGroupsMetadata(callOp, call);
433  moduleTranslation.setAliasScopeMetadata(callOp, call);
434  moduleTranslation.setTBAAMetadata(callOp, call);
435  // If the called function has a result, remap the corresponding value. Note
436  // that LLVM IR dialect CallOp has either 0 or 1 result.
437  if (opInst.getNumResults() != 0)
438  moduleTranslation.mapValue(opInst.getResult(0), call);
439  // Check that LLVM call returns void for 0-result functions.
440  else if (!call->getType()->isVoidTy())
441  return failure();
442  moduleTranslation.mapCall(callOp, call);
443  return success();
444  }
445 
446  if (auto inlineAsmOp = dyn_cast<LLVM::InlineAsmOp>(opInst)) {
447  // TODO: refactor function type creation which usually occurs in std-LLVM
448  // conversion.
449  SmallVector<Type, 8> operandTypes;
450  llvm::append_range(operandTypes, inlineAsmOp.getOperands().getTypes());
451 
452  Type resultType;
453  if (inlineAsmOp.getNumResults() == 0) {
454  resultType = LLVM::LLVMVoidType::get(&moduleTranslation.getContext());
455  } else {
456  assert(inlineAsmOp.getNumResults() == 1);
457  resultType = inlineAsmOp.getResultTypes()[0];
458  }
459  auto ft = LLVM::LLVMFunctionType::get(resultType, operandTypes);
460  llvm::InlineAsm *inlineAsmInst =
461  inlineAsmOp.getAsmDialect()
462  ? llvm::InlineAsm::get(
463  static_cast<llvm::FunctionType *>(
464  moduleTranslation.convertType(ft)),
465  inlineAsmOp.getAsmString(), inlineAsmOp.getConstraints(),
466  inlineAsmOp.getHasSideEffects(),
467  inlineAsmOp.getIsAlignStack(),
468  convertAsmDialectToLLVM(*inlineAsmOp.getAsmDialect()))
469  : llvm::InlineAsm::get(static_cast<llvm::FunctionType *>(
470  moduleTranslation.convertType(ft)),
471  inlineAsmOp.getAsmString(),
472  inlineAsmOp.getConstraints(),
473  inlineAsmOp.getHasSideEffects(),
474  inlineAsmOp.getIsAlignStack());
475  llvm::CallInst *inst = builder.CreateCall(
476  inlineAsmInst,
477  moduleTranslation.lookupValues(inlineAsmOp.getOperands()));
478  inst->setTailCallKind(convertTailCallKindToLLVM(
479  inlineAsmOp.getTailCallKindAttr().getTailCallKind()));
480  if (auto maybeOperandAttrs = inlineAsmOp.getOperandAttrs()) {
481  llvm::AttributeList attrList;
482  for (const auto &it : llvm::enumerate(*maybeOperandAttrs)) {
483  Attribute attr = it.value();
484  if (!attr)
485  continue;
486  DictionaryAttr dAttr = cast<DictionaryAttr>(attr);
487  if (dAttr.empty())
488  continue;
489  TypeAttr tAttr =
490  cast<TypeAttr>(dAttr.get(InlineAsmOp::getElementTypeAttrName()));
491  llvm::AttrBuilder b(moduleTranslation.getLLVMContext());
492  llvm::Type *ty = moduleTranslation.convertType(tAttr.getValue());
493  b.addTypeAttr(llvm::Attribute::ElementType, ty);
494  // shift to account for the returned value (this is always 1 aggregate
495  // value in LLVM).
496  int shift = (opInst.getNumResults() > 0) ? 1 : 0;
497  attrList = attrList.addAttributesAtIndex(
498  moduleTranslation.getLLVMContext(), it.index() + shift, b);
499  }
500  inst->setAttributes(attrList);
501  }
502 
503  if (opInst.getNumResults() != 0)
504  moduleTranslation.mapValue(opInst.getResult(0), inst);
505  return success();
506  }
507 
508  if (auto invOp = dyn_cast<LLVM::InvokeOp>(opInst)) {
509  auto operands = moduleTranslation.lookupValues(invOp.getCalleeOperands());
510  SmallVector<llvm::OperandBundleDef> opBundles =
511  convertOperandBundles(invOp.getOpBundleOperands(),
512  invOp.getOpBundleTags(), moduleTranslation);
513  ArrayRef<llvm::Value *> operandsRef(operands);
514  llvm::InvokeInst *result;
515  if (auto attr = opInst.getAttrOfType<FlatSymbolRefAttr>("callee")) {
516  result = builder.CreateInvoke(
517  moduleTranslation.lookupFunction(attr.getValue()),
518  moduleTranslation.lookupBlock(invOp.getSuccessor(0)),
519  moduleTranslation.lookupBlock(invOp.getSuccessor(1)), operandsRef,
520  opBundles);
521  } else {
522  llvm::FunctionType *calleeType = llvm::cast<llvm::FunctionType>(
523  moduleTranslation.convertType(invOp.getCalleeFunctionType()));
524  result = builder.CreateInvoke(
525  calleeType, operandsRef.front(),
526  moduleTranslation.lookupBlock(invOp.getSuccessor(0)),
527  moduleTranslation.lookupBlock(invOp.getSuccessor(1)),
528  operandsRef.drop_front(), opBundles);
529  }
530  result->setCallingConv(convertCConvToLLVM(invOp.getCConv()));
531  if (failed(moduleTranslation.convertArgAndResultAttrs(invOp, result)))
532  return failure();
533  moduleTranslation.mapBranch(invOp, result);
534  // InvokeOp can only have 0 or 1 result
535  if (invOp->getNumResults() != 0) {
536  moduleTranslation.mapValue(opInst.getResult(0), result);
537  return success();
538  }
539  return success(result->getType()->isVoidTy());
540  }
541 
542  if (auto lpOp = dyn_cast<LLVM::LandingpadOp>(opInst)) {
543  llvm::Type *ty = moduleTranslation.convertType(lpOp.getType());
544  llvm::LandingPadInst *lpi =
545  builder.CreateLandingPad(ty, lpOp.getNumOperands());
546  lpi->setCleanup(lpOp.getCleanup());
547 
548  // Add clauses
549  for (llvm::Value *operand :
550  moduleTranslation.lookupValues(lpOp.getOperands())) {
551  // All operands should be constant - checked by verifier
552  if (auto *constOperand = dyn_cast<llvm::Constant>(operand))
553  lpi->addClause(constOperand);
554  }
555  moduleTranslation.mapValue(lpOp.getResult(), lpi);
556  return success();
557  }
558 
559  // Emit branches. We need to look up the remapped blocks and ignore the
560  // block arguments that were transformed into PHI nodes.
561  if (auto brOp = dyn_cast<LLVM::BrOp>(opInst)) {
562  llvm::BranchInst *branch =
563  builder.CreateBr(moduleTranslation.lookupBlock(brOp.getSuccessor()));
564  moduleTranslation.mapBranch(&opInst, branch);
565  moduleTranslation.setLoopMetadata(&opInst, branch);
566  return success();
567  }
568  if (auto condbrOp = dyn_cast<LLVM::CondBrOp>(opInst)) {
569  llvm::BranchInst *branch = builder.CreateCondBr(
570  moduleTranslation.lookupValue(condbrOp.getOperand(0)),
571  moduleTranslation.lookupBlock(condbrOp.getSuccessor(0)),
572  moduleTranslation.lookupBlock(condbrOp.getSuccessor(1)));
573  moduleTranslation.mapBranch(&opInst, branch);
574  moduleTranslation.setLoopMetadata(&opInst, branch);
575  return success();
576  }
577  if (auto switchOp = dyn_cast<LLVM::SwitchOp>(opInst)) {
578  llvm::SwitchInst *switchInst = builder.CreateSwitch(
579  moduleTranslation.lookupValue(switchOp.getValue()),
580  moduleTranslation.lookupBlock(switchOp.getDefaultDestination()),
581  switchOp.getCaseDestinations().size());
582 
583  // Handle switch with zero cases.
584  if (!switchOp.getCaseValues())
585  return success();
586 
587  auto *ty = llvm::cast<llvm::IntegerType>(
588  moduleTranslation.convertType(switchOp.getValue().getType()));
589  for (auto i :
590  llvm::zip(llvm::cast<DenseIntElementsAttr>(*switchOp.getCaseValues()),
591  switchOp.getCaseDestinations()))
592  switchInst->addCase(
593  llvm::ConstantInt::get(ty, std::get<0>(i).getLimitedValue()),
594  moduleTranslation.lookupBlock(std::get<1>(i)));
595 
596  moduleTranslation.mapBranch(&opInst, switchInst);
597  return success();
598  }
599  if (auto indBrOp = dyn_cast<LLVM::IndirectBrOp>(opInst)) {
600  llvm::IndirectBrInst *indBr = builder.CreateIndirectBr(
601  moduleTranslation.lookupValue(indBrOp.getAddr()),
602  indBrOp->getNumSuccessors());
603  for (auto *succ : indBrOp.getSuccessors())
604  indBr->addDestination(moduleTranslation.lookupBlock(succ));
605  moduleTranslation.mapBranch(&opInst, indBr);
606  return success();
607  }
608 
609  // Emit addressof. We need to look up the global value referenced by the
610  // operation and store it in the MLIR-to-LLVM value mapping. This does not
611  // emit any LLVM instruction.
612  if (auto addressOfOp = dyn_cast<LLVM::AddressOfOp>(opInst)) {
613  LLVM::GlobalOp global =
614  addressOfOp.getGlobal(moduleTranslation.symbolTable());
615  LLVM::LLVMFuncOp function =
616  addressOfOp.getFunction(moduleTranslation.symbolTable());
617  LLVM::AliasOp alias = addressOfOp.getAlias(moduleTranslation.symbolTable());
618  LLVM::IFuncOp ifunc = addressOfOp.getIFunc(moduleTranslation.symbolTable());
619 
620  // The verifier should not have allowed this.
621  assert((global || function || alias || ifunc) &&
622  "referencing an undefined global, function, alias, or ifunc");
623 
624  llvm::Value *llvmValue = nullptr;
625  if (global)
626  llvmValue = moduleTranslation.lookupGlobal(global);
627  else if (alias)
628  llvmValue = moduleTranslation.lookupAlias(alias);
629  else if (function)
630  llvmValue = moduleTranslation.lookupFunction(function.getName());
631  else
632  llvmValue = moduleTranslation.lookupIFunc(ifunc);
633 
634  moduleTranslation.mapValue(addressOfOp.getResult(), llvmValue);
635  return success();
636  }
637 
638  // Emit dso_local_equivalent. We need to look up the global value referenced
639  // by the operation and store it in the MLIR-to-LLVM value mapping.
640  if (auto dsoLocalEquivalentOp =
641  dyn_cast<LLVM::DSOLocalEquivalentOp>(opInst)) {
642  LLVM::LLVMFuncOp function =
643  dsoLocalEquivalentOp.getFunction(moduleTranslation.symbolTable());
644  LLVM::AliasOp alias =
645  dsoLocalEquivalentOp.getAlias(moduleTranslation.symbolTable());
646 
647  // The verifier should not have allowed this.
648  assert((function || alias) &&
649  "referencing an undefined function, or alias");
650 
651  llvm::Value *llvmValue = nullptr;
652  if (alias)
653  llvmValue = moduleTranslation.lookupAlias(alias);
654  else
655  llvmValue = moduleTranslation.lookupFunction(function.getName());
656 
657  moduleTranslation.mapValue(
658  dsoLocalEquivalentOp.getResult(),
659  llvm::DSOLocalEquivalent::get(cast<llvm::GlobalValue>(llvmValue)));
660  return success();
661  }
662 
663  // Emit blockaddress. We first need to find the LLVM block referenced by this
664  // operation and then create a LLVM block address for it.
665  if (auto blockAddressOp = dyn_cast<LLVM::BlockAddressOp>(opInst)) {
666  BlockAddressAttr blockAddressAttr = blockAddressOp.getBlockAddr();
667  llvm::BasicBlock *llvmBlock =
668  moduleTranslation.lookupBlockAddress(blockAddressAttr);
669 
670  llvm::Value *llvmValue = nullptr;
671  StringRef fnName = blockAddressAttr.getFunction().getValue();
672  if (llvmBlock) {
673  llvm::Function *llvmFn = moduleTranslation.lookupFunction(fnName);
674  llvmValue = llvm::BlockAddress::get(llvmFn, llvmBlock);
675  } else {
676  // The matching LLVM block is not yet emitted, a placeholder is created
677  // in its place. When the LLVM block is emitted later in translation,
678  // the llvmValue is replaced with the actual llvm::BlockAddress.
679  // A GlobalVariable is chosen as placeholder because in general LLVM
680  // constants are uniqued and are not proper for RAUW, since that could
681  // harm unrelated uses of the constant.
682  llvmValue = new llvm::GlobalVariable(
683  *moduleTranslation.getLLVMModule(),
684  llvm::PointerType::getUnqual(moduleTranslation.getLLVMContext()),
685  /*isConstant=*/true, llvm::GlobalValue::LinkageTypes::ExternalLinkage,
686  /*Initializer=*/nullptr,
687  Twine("__mlir_block_address_")
688  .concat(Twine(fnName))
689  .concat(Twine((uint64_t)blockAddressOp.getOperation())));
690  moduleTranslation.mapUnresolvedBlockAddress(blockAddressOp, llvmValue);
691  }
692 
693  moduleTranslation.mapValue(blockAddressOp.getResult(), llvmValue);
694  return success();
695  }
696 
697  // Emit block label. If this label is seen before BlockAddressOp is
698  // translated, go ahead and already map it.
699  if (auto blockTagOp = dyn_cast<LLVM::BlockTagOp>(opInst)) {
700  auto funcOp = blockTagOp->getParentOfType<LLVMFuncOp>();
701  BlockAddressAttr blockAddressAttr = BlockAddressAttr::get(
702  &moduleTranslation.getContext(),
703  FlatSymbolRefAttr::get(&moduleTranslation.getContext(),
704  funcOp.getName()),
705  blockTagOp.getTag());
706  moduleTranslation.mapBlockAddress(blockAddressAttr,
707  builder.GetInsertBlock());
708  return success();
709  }
710 
711  return failure();
712 }
