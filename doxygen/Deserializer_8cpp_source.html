63 LogicalResult spirv::Deserializer::deserialize() {
64  LLVM_DEBUG({
65  logger.resetIndent();
66  logger.startLine()
67  << "//+++---------- start deserialization ----------+++//\n";
68  });
69 
70  if (failed(processHeader()))
71  return failure();
72 
73  spirv::Opcode opcode = spirv::Opcode::OpNop;
74  ArrayRef<uint32_t> operands;
75  auto binarySize = binary.size();
76  while (curOffset < binarySize) {
77  // Slice the next instruction out and populate `opcode` and `operands`.
78  // Internally this also updates `curOffset`.
79  if (failed(sliceInstruction(opcode, operands)))
80  return failure();
81 
82  if (failed(processInstruction(opcode, operands)))
83  return failure();
84  }
85 
86  assert(curOffset == binarySize &&
87  "deserializer should never index beyond the binary end");
88 
89  for (auto &deferred : deferredInstructions) {
90  if (failed(processInstruction(deferred.first, deferred.second, false))) {
91  return failure();
92  }
93  }
94 
95  attachVCETriple();
96 
97  LLVM_DEBUG(logger.startLine()
98  << "//+++-------- completed deserialization --------+++//\n");
99  return success();
100 }

117 LogicalResult spirv::Deserializer::processHeader() {
118  if (binary.size() < spirv::kHeaderWordCount)
119  return emitError(unknownLoc,
120  "SPIR-V binary module must have a 5-word header");
121 
122  if (binary[0] != spirv::kMagicNumber)
123  return emitError(unknownLoc, "incorrect magic number");
124 
125  // Version number bytes: 0 | major number | minor number | 0
126  uint32_t majorVersion = (binary[1] << 8) >> 24;
127  uint32_t minorVersion = (binary[1] << 16) >> 24;
128  if (majorVersion == 1) {
129  switch (minorVersion) {
130 #define MIN_VERSION_CASE(v) \
131  case v: \
132  version = spirv::Version::V_1_##v; \
133  break
134 
135  MIN_VERSION_CASE(0);
136  MIN_VERSION_CASE(1);
137  MIN_VERSION_CASE(2);
138  MIN_VERSION_CASE(3);
139  MIN_VERSION_CASE(4);
140  MIN_VERSION_CASE(5);
141 #undef MIN_VERSION_CASE
142  default:
143  return emitError(unknownLoc, "unsupported SPIR-V minor version: ")
144  << minorVersion;
145  }
146  } else {
147  return emitError(unknownLoc, "unsupported SPIR-V major version: ")
148  << majorVersion;
149  }
150 
151  // TODO: generator number, bound, schema
152  curOffset = spirv::kHeaderWordCount;
153  return success();
154 }

231 LogicalResult deserializeCacheControlDecoration(
232  Location loc, OpBuilder &opBuilder,
233  DenseMap<uint32_t, NamedAttrList> &decorations, ArrayRef<uint32_t> words,
234  StringAttr symbol, StringRef decorationName, StringRef cacheControlKind) {
235  if (words.size() != 4) {
236  return emitError(loc, "OpDecoration with ")
237  << decorationName << "needs a cache control integer literal and a "
238  << cacheControlKind << " cache control literal";
239  }
240  unsigned cacheLevel = words[2];
241  auto cacheControlAttr = static_cast<EnumTy>(words[3]);
242  auto value = opBuilder.getAttr<AttrTy>(cacheLevel, cacheControlAttr);
243  SmallVector<Attribute> attrs;
244  if (auto attrList =
245  llvm::dyn_cast_or_null<ArrayAttr>(decorations[words[0]].get(symbol)))
246  llvm::append_range(attrs, attrList);
247  attrs.push_back(value);
248  decorations[words[0]].set(symbol, opBuilder.getArrayAttr(attrs));
249  return success();
250 }

252 LogicalResult spirv::Deserializer::processDecoration(ArrayRef<uint32_t> words) {
253  // TODO: This function should also be auto-generated. For now, since only a
254  // few decorations are processed/handled in a meaningful manner, going with a
255  // manual implementation.
256  if (words.size() < 2) {
257  return emitError(
258  unknownLoc, "OpDecorate must have at least result <id> and Decoration");
259  }
260  auto decorationName =
261  stringifyDecoration(static_cast<spirv::Decoration>(words[1]));
262  if (decorationName.empty()) {
263  return emitError(unknownLoc, "invalid Decoration code : ") << words[1];
264  }
265  auto symbol = getSymbolDecoration(decorationName);
266  switch (static_cast<spirv::Decoration>(words[1])) {
267  case spirv::Decoration::FPFastMathMode:
268  if (words.size() != 3) {
269  return emitError(unknownLoc, "OpDecorate with ")
270  << decorationName << " needs a single integer literal";
271  }
272  decorations[words[0]].set(
273  symbol, FPFastMathModeAttr::get(opBuilder.getContext(),
274  static_cast<FPFastMathMode>(words[2])));
275  break;
276  case spirv::Decoration::FPRoundingMode:
277  if (words.size() != 3) {
278  return emitError(unknownLoc, "OpDecorate with ")
279  << decorationName << " needs a single integer literal";
280  }
281  decorations[words[0]].set(
282  symbol, FPRoundingModeAttr::get(opBuilder.getContext(),
283  static_cast<FPRoundingMode>(words[2])));
284  break;
285  case spirv::Decoration::DescriptorSet:
286  case spirv::Decoration::Binding:
287  if (words.size() != 3) {
288  return emitError(unknownLoc, "OpDecorate with ")
289  << decorationName << " needs a single integer literal";
290  }
291  decorations[words[0]].set(
292  symbol, opBuilder.getI32IntegerAttr(static_cast<int32_t>(words[2])));
293  break;
294  case spirv::Decoration::BuiltIn:
295  if (words.size() != 3) {
296  return emitError(unknownLoc, "OpDecorate with ")
297  << decorationName << " needs a single integer literal";
298  }
299  decorations[words[0]].set(
300  symbol, opBuilder.getStringAttr(
301  stringifyBuiltIn(static_cast<spirv::BuiltIn>(words[2]))));
302  break;
303  case spirv::Decoration::ArrayStride:
304  if (words.size() != 3) {
305  return emitError(unknownLoc, "OpDecorate with ")
306  << decorationName << " needs a single integer literal";
307  }
308  typeDecorations[words[0]] = words[2];
309  break;
310  case spirv::Decoration::LinkageAttributes: {
311  if (words.size() < 4) {
312  return emitError(unknownLoc, "OpDecorate with ")
313  << decorationName
314  << " needs at least 1 string and 1 integer literal";
315  }
316  // LinkageAttributes has two parameters ["linkageName", linkageType]
317  // e.g., OpDecorate %imported_func LinkageAttributes "outside.func" Import
318  // "linkageName" is a stringliteral encoded as uint32_t,
319  // hence the size of name is variable length which results in words.size()
320  // being variable length, words.size() = 3 + strlen(name)/4 + 1 or
321  // 3 + ceildiv(strlen(name), 4).
322  unsigned wordIndex = 2;
323  auto linkageName = spirv::decodeStringLiteral(words, wordIndex).str();
324  auto linkageTypeAttr = opBuilder.getAttr<::mlir::spirv::LinkageTypeAttr>(
325  static_cast<::mlir::spirv::LinkageType>(words[wordIndex++]));
326  auto linkageAttr = opBuilder.getAttr<::mlir::spirv::LinkageAttributesAttr>(
327  StringAttr::get(context, linkageName), linkageTypeAttr);
328  decorations[words[0]].set(symbol, llvm::dyn_cast<Attribute>(linkageAttr));
329  break;
330  }
331  case spirv::Decoration::Aliased:
332  case spirv::Decoration::AliasedPointer:
333  case spirv::Decoration::Block:
334  case spirv::Decoration::BufferBlock:
335  case spirv::Decoration::Flat:
336  case spirv::Decoration::NonReadable:
337  case spirv::Decoration::NonWritable:
338  case spirv::Decoration::NoPerspective:
339  case spirv::Decoration::NoSignedWrap:
340  case spirv::Decoration::NoUnsignedWrap:
341  case spirv::Decoration::RelaxedPrecision:
342  case spirv::Decoration::Restrict:
343  case spirv::Decoration::RestrictPointer:
344  case spirv::Decoration::NoContraction:
345  case spirv::Decoration::Constant:
346  if (words.size() != 2) {
347  return emitError(unknownLoc, "OpDecoration with ")
348  << decorationName << "needs a single target <id>";
349  }
350  decorations[words[0]].set(symbol, opBuilder.getUnitAttr());
351  break;
352  case spirv::Decoration::Location:
353  case spirv::Decoration::SpecId:
354  if (words.size() != 3) {
355  return emitError(unknownLoc, "OpDecoration with ")
356  << decorationName << "needs a single integer literal";
357  }
358  decorations[words[0]].set(
359  symbol, opBuilder.getI32IntegerAttr(static_cast<int32_t>(words[2])));
360  break;
361  case spirv::Decoration::CacheControlLoadINTEL: {
362  LogicalResult res = deserializeCacheControlDecoration<
363  CacheControlLoadINTELAttr, LoadCacheControlAttr, LoadCacheControl>(
364  unknownLoc, opBuilder, decorations, words, symbol, decorationName,
365  "load");
366  if (failed(res))
367  return res;
368  break;
369  }
370  case spirv::Decoration::CacheControlStoreINTEL: {
371  LogicalResult res = deserializeCacheControlDecoration<
372  CacheControlStoreINTELAttr, StoreCacheControlAttr, StoreCacheControl>(
373  unknownLoc, opBuilder, decorations, words, symbol, decorationName,
374  "store");
375  if (failed(res))
376  return res;
377  break;
378  }
379  default:
380  return emitError(unknownLoc, "unhandled Decoration : '") << decorationName;
381  }
382  return success();
383 }

421 LogicalResult spirv::Deserializer::setFunctionArgAttrs(
422  uint32_t argID, SmallVectorImpl<Attribute> &argAttrs, size_t argIndex) {
423  if (!decorations.contains(argID)) {
424  argAttrs[argIndex] = DictionaryAttr::get(context, {});
425  return success();
426  }
427 
428  spirv::DecorationAttr foundDecorationAttr;
429  for (NamedAttribute decAttr : decorations[argID]) {
430  for (auto decoration :
431  {spirv::Decoration::Aliased, spirv::Decoration::Restrict,
432  spirv::Decoration::AliasedPointer,
433  spirv::Decoration::RestrictPointer}) {
434 
435  if (decAttr.getName() !=
436  getSymbolDecoration(stringifyDecoration(decoration)))
437  continue;
438 
439  if (foundDecorationAttr)
440  return emitError(unknownLoc,
441  "more than one Aliased/Restrict decorations for "
442  "function argument with result <id> ")
443  << argID;
444 
445  foundDecorationAttr = spirv::DecorationAttr::get(context, decoration);
446  break;
447  }
448 
449  if (decAttr.getName() == getSymbolDecoration(stringifyDecoration(
450  spirv::Decoration::RelaxedPrecision))) {
451  // TODO: Current implementation supports only one decoration per function
452  // parameter so RelaxedPrecision cannot be applied at the same time as,
453  // for example, Aliased/Restrict/etc. This should be relaxed to allow any
454  // combination of decoration allowed by the spec to be supported.
455  if (foundDecorationAttr)
456  return emitError(unknownLoc, "already found a decoration for function "
457  "argument with result <id> ")
458  << argID;
459 
460  foundDecorationAttr = spirv::DecorationAttr::get(
461  context, spirv::Decoration::RelaxedPrecision);
462  }
463  }
464 
465  if (!foundDecorationAttr)
466  return emitError(unknownLoc, "unimplemented decoration support for "
467  "function argument with result <id> ")
468  << argID;
469 
470  NamedAttribute attr(StringAttr::get(context, spirv::DecorationAttr::name),
471  foundDecorationAttr);
472  argAttrs[argIndex] = DictionaryAttr::get(context, attr);
473  return success();
474 }

476 LogicalResult
477 spirv::Deserializer::processFunction(ArrayRef<uint32_t> operands) {
478  if (curFunction) {
479  return emitError(unknownLoc, "found function inside function");
480  }
481 
482  // Get the result type
483  if (operands.size() != 4) {
484  return emitError(unknownLoc, "OpFunction must have 4 parameters");
485  }
486  Type resultType = getType(operands[0]);
487  if (!resultType) {
488  return emitError(unknownLoc, "undefined result type from <id> ")
489  << operands[0];
490  }
491 
492  uint32_t fnID = operands[1];
493  if (funcMap.count(fnID)) {
494  return emitError(unknownLoc, "duplicate function definition/declaration");
495  }
496 
497  auto fnControl = spirv::symbolizeFunctionControl(operands[2]);
498  if (!fnControl) {
499  return emitError(unknownLoc, "unknown Function Control: ") << operands[2];
500  }
501 
502  Type fnType = getType(operands[3]);
503  if (!fnType || !isa<FunctionType>(fnType)) {
504  return emitError(unknownLoc, "unknown function type from <id> ")
505  << operands[3];
506  }
507  auto functionType = cast<FunctionType>(fnType);
508 
509  if ((isVoidType(resultType) && functionType.getNumResults() != 0) ||
510  (functionType.getNumResults() == 1 &&
511  functionType.getResult(0) != resultType)) {
512  return emitError(unknownLoc, "mismatch in function type ")
513  << functionType << " and return type " << resultType << " specified";
514  }
515 
516  std::string fnName = getFunctionSymbol(fnID);
517  auto funcOp = spirv::FuncOp::create(opBuilder, unknownLoc, fnName,
518  functionType, fnControl.value());
519  // Processing other function attributes.
520  if (decorations.count(fnID)) {
521  for (auto attr : decorations[fnID].getAttrs()) {
522  funcOp->setAttr(attr.getName(), attr.getValue());
523  }
524  }
525  curFunction = funcMap[fnID] = funcOp;
526  auto *entryBlock = funcOp.addEntryBlock();
527  LLVM_DEBUG({
528  logger.startLine()
529  << "//===-------------------------------------------===//\n";
530  logger.startLine() << "[fn] name: " << fnName << "\n";
531  logger.startLine() << "[fn] type: " << fnType << "\n";
532  logger.startLine() << "[fn] ID: " << fnID << "\n";
533  logger.startLine() << "[fn] entry block: " << entryBlock << "\n";
534  logger.indent();
535  });
536 
537  SmallVector<Attribute> argAttrs;
538  argAttrs.resize(functionType.getNumInputs());
539 
540  // Parse the op argument instructions
541  if (functionType.getNumInputs()) {
542  for (size_t i = 0, e = functionType.getNumInputs(); i != e; ++i) {
543  auto argType = functionType.getInput(i);
544  spirv::Opcode opcode = spirv::Opcode::OpNop;
545  ArrayRef<uint32_t> operands;
546  if (failed(sliceInstruction(opcode, operands,
547  spirv::Opcode::OpFunctionParameter))) {
548  return failure();
549  }
550  if (opcode != spirv::Opcode::OpFunctionParameter) {
551  return emitError(
552  unknownLoc,
553  "missing OpFunctionParameter instruction for argument ")
554  << i;
555  }
556  if (operands.size() != 2) {
557  return emitError(
558  unknownLoc,
559  "expected result type and result <id> for OpFunctionParameter");
560  }
561  auto argDefinedType = getType(operands[0]);
562  if (!argDefinedType || argDefinedType != argType) {
563  return emitError(unknownLoc,
564  "mismatch in argument type between function type "
565  "definition ")
566  << functionType << " and argument type definition "
567  << argDefinedType << " at argument " << i;
568  }
569  if (getValue(operands[1])) {
570  return emitError(unknownLoc, "duplicate definition of result <id> ")
571  << operands[1];
572  }
573  if (failed(setFunctionArgAttrs(operands[1], argAttrs, i))) {
574  return failure();
575  }
576 
577  auto argValue = funcOp.getArgument(i);
578  valueMap[operands[1]] = argValue;
579  }
580  }
581 
582  if (llvm::any_of(argAttrs, [](Attribute attr) {
583  auto argAttr = cast<DictionaryAttr>(attr);
584  return !argAttr.empty();
585  }))
586  funcOp.setArgAttrsAttr(ArrayAttr::get(context, argAttrs));
587 
588  // entryBlock is needed to access the arguments, Once that is done, we can
589  // erase the block for functions with 'Import' LinkageAttributes, since these
590  // are essentially function declarations, so they have no body.
591  auto linkageAttr = funcOp.getLinkageAttributes();
592  auto hasImportLinkage =
593  linkageAttr && (linkageAttr.value().getLinkageType().getValue() ==
594  spirv::LinkageType::Import);
595  if (hasImportLinkage)
596  funcOp.eraseBody();
597 
598  // RAII guard to reset the insertion point to the module's region after
599  // deserializing the body of this function.
600  OpBuilder::InsertionGuard moduleInsertionGuard(opBuilder);
601 
602  spirv::Opcode opcode = spirv::Opcode::OpNop;
603  ArrayRef<uint32_t> instOperands;
604 
605  // Special handling for the entry block. We need to make sure it starts with
606  // an OpLabel instruction. The entry block takes the same parameters as the
607  // function. All other blocks do not take any parameter. We have already
608  // created the entry block, here we need to register it to the correct label
609  // <id>.
610  if (failed(sliceInstruction(opcode, instOperands,
611  spirv::Opcode::OpFunctionEnd))) {
612  return failure();
613  }
614  if (opcode == spirv::Opcode::OpFunctionEnd) {
615  return processFunctionEnd(instOperands);
616  }
617  if (opcode != spirv::Opcode::OpLabel) {
618  return emitError(unknownLoc, "a basic block must start with OpLabel");
619  }
620  if (instOperands.size() != 1) {
621  return emitError(unknownLoc, "OpLabel should only have result <id>");
622  }
623  blockMap[instOperands[0]] = entryBlock;
624  if (failed(processLabel(instOperands))) {
625  return failure();
626  }
627 
628  // Then process all the other instructions in the function until we hit
629  // OpFunctionEnd.
630  while (succeeded(sliceInstruction(opcode, instOperands,
631  spirv::Opcode::OpFunctionEnd)) &&
632  opcode != spirv::Opcode::OpFunctionEnd) {
633  if (failed(processInstruction(opcode, instOperands))) {

724 spirv::Deserializer::processGlobalVariable(ArrayRef<uint32_t> operands) {
725  unsigned wordIndex = 0;
726  if (operands.size() < 3) {
727  return emitError(
728  unknownLoc,
729  "OpVariable needs at least 3 operands, type, <id> and storage class");
730  }
731 
732  // Result Type.
733  auto type = getType(operands[wordIndex]);
734  if (!type) {
735  return emitError(unknownLoc, "unknown result type <id> : ")
736  << operands[wordIndex];
737  }
738  auto ptrType = dyn_cast<spirv::PointerType>(type);
739  if (!ptrType) {
740  return emitError(unknownLoc,
741  "expected a result type <id> to be a spirv.ptr, found : ")
742  << type;
743  }
744  wordIndex++;
745 
746  // Result <id>.
747  auto variableID = operands[wordIndex];
748  auto variableName = nameMap.lookup(variableID).str();
749  if (variableName.empty()) {
750  variableName = "spirv_var_" + std::to_string(variableID);
751  }
752  wordIndex++;
753 
754  // Storage class.
755  auto storageClass = static_cast<spirv::StorageClass>(operands[wordIndex]);
756  if (ptrType.getStorageClass() != storageClass) {
757  return emitError(unknownLoc, "mismatch in storage class of pointer type ")
758  << type << " and that specified in OpVariable instruction : "
759  << stringifyStorageClass(storageClass);
760  }
761  wordIndex++;
762 
763  // Initializer.
764  FlatSymbolRefAttr initializer = nullptr;
765 
766  if (wordIndex < operands.size()) {
767  Operation *op = nullptr;
768 
769  if (auto initOp = getGlobalVariable(operands[wordIndex]))
770  op = initOp;
771  else if (auto initOp = getSpecConstant(operands[wordIndex]))
772  op = initOp;
773  else if (auto initOp = getSpecConstantComposite(operands[wordIndex]))
774  op = initOp;
775  else
776  return emitError(unknownLoc, "unknown <id> ")
777  << operands[wordIndex] << "used as initializer";
778 
779  initializer = SymbolRefAttr::get(op);
780  wordIndex++;
781  }
782  if (wordIndex != operands.size()) {
783  return emitError(unknownLoc,
784  "found more operands than expected when deserializing "
785  "OpVariable instruction, only ")
786  << wordIndex << " of " << operands.size() << " processed";
787  }
788  auto loc = createFileLineColLoc(opBuilder);
789  auto varOp = spirv::GlobalVariableOp::create(
790  opBuilder, loc, TypeAttr::get(type),
791  opBuilder.getStringAttr(variableName), initializer);
792 
793  // Decorations.
794  if (decorations.count(variableID)) {
795  for (auto attr : decorations[variableID].getAttrs())
796  varOp->setAttr(attr.getName(), attr.getValue());
797  }
798  globalVariableMap[variableID] = varOp;
799  return success();
800 }

832 LogicalResult spirv::Deserializer::processType(spirv::Opcode opcode,
833  ArrayRef<uint32_t> operands) {
834  if (operands.empty()) {
835  return emitError(unknownLoc, "type instruction with opcode ")
836  << spirv::stringifyOpcode(opcode) << " needs at least one <id>";
837  }
838 
839  /// TODO: Types might be forward declared in some instructions and need to be
840  /// handled appropriately.
841  if (typeMap.count(operands[0])) {
842  return emitError(unknownLoc, "duplicate definition for result <id> ")
843  << operands[0];
844  }
845 
846  switch (opcode) {
847  case spirv::Opcode::OpTypeVoid:
848  if (operands.size() != 1)
849  return emitError(unknownLoc, "OpTypeVoid must have no parameters");
850  typeMap[operands[0]] = opBuilder.getNoneType();
851  break;
852  case spirv::Opcode::OpTypeBool:
853  if (operands.size() != 1)
854  return emitError(unknownLoc, "OpTypeBool must have no parameters");
855  typeMap[operands[0]] = opBuilder.getI1Type();
856  break;
857  case spirv::Opcode::OpTypeInt: {
858  if (operands.size() != 3)
859  return emitError(
860  unknownLoc, "OpTypeInt must have bitwidth and signedness parameters");
861 
862  // SPIR-V OpTypeInt "Signedness specifies whether there are signed semantics
863  // to preserve or validate.
864  // 0 indicates unsigned, or no signedness semantics
865  // 1 indicates signed semantics."
866  //
867  // So we cannot differentiate signless and unsigned integers; always use
868  // signless semantics for such cases.
869  auto sign = operands[2] == 1 ? IntegerType::SignednessSemantics::Signed
870  : IntegerType::SignednessSemantics::Signless;
871  typeMap[operands[0]] = IntegerType::get(context, operands[1], sign);
872  } break;
873  case spirv::Opcode::OpTypeFloat: {
874  if (operands.size() != 2 && operands.size() != 3)
875  return emitError(unknownLoc,
876  "OpTypeFloat expects either 2 operands (type, bitwidth) "
877  "or 3 operands (type, bitwidth, encoding), but got ")
878  << operands.size();
879  uint32_t bitWidth = operands[1];
880 
881  Type floatTy;
882  switch (bitWidth) {
883  case 16:
884  floatTy = opBuilder.getF16Type();
885  break;
886  case 32:
887  floatTy = opBuilder.getF32Type();
888  break;
889  case 64:
890  floatTy = opBuilder.getF64Type();
891  break;
892  default:
893  return emitError(unknownLoc, "unsupported OpTypeFloat bitwidth: ")
894  << bitWidth;
895  }
896 
897  if (operands.size() == 3) {
898  if (spirv::FPEncoding(operands[2]) != spirv::FPEncoding::BFloat16KHR)
899  return emitError(unknownLoc, "unsupported OpTypeFloat FP encoding: ")
900  << operands[2];
901  if (bitWidth != 16)
902  return emitError(unknownLoc,
903  "invalid OpTypeFloat bitwidth for bfloat16 encoding: ")
904  << bitWidth << " (expected 16)";
905  floatTy = opBuilder.getBF16Type();
906  }
907 
908  typeMap[operands[0]] = floatTy;
909  } break;
910  case spirv::Opcode::OpTypeVector: {
911  if (operands.size() != 3) {
912  return emitError(
913  unknownLoc,
914  "OpTypeVector must have element type and count parameters");
915  }
916  Type elementTy = getType(operands[1]);
917  if (!elementTy) {
918  return emitError(unknownLoc, "OpTypeVector references undefined <id> ")
919  << operands[1];
920  }
921  typeMap[operands[0]] = VectorType::get({operands[2]}, elementTy);
922  } break;
923  case spirv::Opcode::OpTypePointer: {
924  return processOpTypePointer(operands);
925  } break;
926  case spirv::Opcode::OpTypeArray:
927  return processArrayType(operands);
928  case spirv::Opcode::OpTypeCooperativeMatrixKHR:
929  return processCooperativeMatrixTypeKHR(operands);
930  case spirv::Opcode::OpTypeFunction:
931  return processFunctionType(operands);
932  case spirv::Opcode::OpTypeImage:
933  return processImageType(operands);
934  case spirv::Opcode::OpTypeSampledImage:
935  return processSampledImageType(operands);
936  case spirv::Opcode::OpTypeRuntimeArray:
937  return processRuntimeArrayType(operands);
938  case spirv::Opcode::OpTypeStruct:
939  return processStructType(operands);
940  case spirv::Opcode::OpTypeMatrix:
941  return processMatrixType(operands);
942  case spirv::Opcode::OpTypeTensorARM:
943  return processTensorARMType(operands);
944  default:
945  return emitError(unknownLoc, "unhandled type instruction");
946  }
947  return success();
948 }

950 LogicalResult
951 spirv::Deserializer::processOpTypePointer(ArrayRef<uint32_t> operands) {
952  if (operands.size() != 3)
953  return emitError(unknownLoc, "OpTypePointer must have two parameters");
954 
955  auto pointeeType = getType(operands[2]);
956  if (!pointeeType)
957  return emitError(unknownLoc, "unknown OpTypePointer pointee type <id> ")
958  << operands[2];
959 
960  uint32_t typePointerID = operands[0];
961  auto storageClass = static_cast<spirv::StorageClass>(operands[1]);
962  typeMap[typePointerID] = spirv::PointerType::get(pointeeType, storageClass);
963 
964  for (auto *deferredStructIt = std::begin(deferredStructTypesInfos);
965  deferredStructIt != std::end(deferredStructTypesInfos);) {
966  for (auto *unresolvedMemberIt =
967  std::begin(deferredStructIt->unresolvedMemberTypes);
968  unresolvedMemberIt !=
969  std::end(deferredStructIt->unresolvedMemberTypes);) {
970  if (unresolvedMemberIt->first == typePointerID) {
971  // The newly constructed pointer type can resolve one of the
972  // deferred struct type members; update the memberTypes list and
973  // clean the unresolvedMemberTypes list accordingly.
974  deferredStructIt->memberTypes[unresolvedMemberIt->second] =
975  typeMap[typePointerID];
976  unresolvedMemberIt =
977  deferredStructIt->unresolvedMemberTypes.erase(unresolvedMemberIt);
978  } else {
979  ++unresolvedMemberIt;
980  }
981  }
982 
983  if (deferredStructIt->unresolvedMemberTypes.empty()) {
984  // All deferred struct type members are now resolved, set the struct body.
985  auto structType = deferredStructIt->deferredStructType;
986 
987  assert(structType && "expected a spirv::StructType");
988  assert(structType.isIdentified() && "expected an indentified struct");
989 
990  if (failed(structType.trySetBody(
991  deferredStructIt->memberTypes, deferredStructIt->offsetInfo,
992  deferredStructIt->memberDecorationsInfo,
993  deferredStructIt->structDecorationsInfo)))
994  return failure();
995 
996  deferredStructIt = deferredStructTypesInfos.erase(deferredStructIt);
997  } else {
998  ++deferredStructIt;
999  }
1000  }
1001 
1002  return success();
1003 }

1140 LogicalResult
1141 spirv::Deserializer::processStructType(ArrayRef<uint32_t> operands) {
1142  // TODO: Find a way to handle identified structs when debug info is stripped.
1143 
1144  if (operands.empty()) {
1145  return emitError(unknownLoc, "OpTypeStruct must have at least result <id>");
1146  }
1147 
1148  if (operands.size() == 1) {
1149  // Handle empty struct.
1150  typeMap[operands[0]] =
1151  spirv::StructType::getEmpty(context, nameMap.lookup(operands[0]).str());
1152  return success();
1153  }
1154 
1155  // First element is operand ID, second element is member index in the struct.
1156  SmallVector<std::pair<uint32_t, unsigned>, 0> unresolvedMemberTypes;
1157  SmallVector<Type, 4> memberTypes;
1158 
1159  for (auto op : llvm::drop_begin(operands, 1)) {
1160  Type memberType = getType(op);
1161  bool typeForwardPtr = (typeForwardPointerIDs.count(op) != 0);
1162 
1163  if (!memberType && !typeForwardPtr)
1164  return emitError(unknownLoc, "OpTypeStruct references undefined <id> ")
1165  << op;
1166 
1167  if (!memberType)
1168  unresolvedMemberTypes.emplace_back(op, memberTypes.size());
1169 
1170  memberTypes.push_back(memberType);
1171  }
1172 
1173  SmallVector<spirv::StructType::OffsetInfo, 0> offsetInfo;
1174  SmallVector<spirv::StructType::MemberDecorationInfo, 0> memberDecorationsInfo;
1175  if (memberDecorationMap.count(operands[0])) {
1176  auto &allMemberDecorations = memberDecorationMap[operands[0]];
1177  for (auto memberIndex : llvm::seq<uint32_t>(0, memberTypes.size())) {
1178  if (allMemberDecorations.count(memberIndex)) {
1179  for (auto &memberDecoration : allMemberDecorations[memberIndex]) {
1180  // Check for offset.
1181  if (memberDecoration.first == spirv::Decoration::Offset) {
1182  // If offset info is empty, resize to the number of members;
1183  if (offsetInfo.empty()) {
1184  offsetInfo.resize(memberTypes.size());
1185  }
1186  offsetInfo[memberIndex] = memberDecoration.second[0];
1187  } else {
1188  auto intType = mlir::IntegerType::get(context, 32);
1189  if (!memberDecoration.second.empty()) {
1190  memberDecorationsInfo.emplace_back(
1191  memberIndex, memberDecoration.first,
1192  IntegerAttr::get(intType, memberDecoration.second[0]));
1193  } else {
1194  memberDecorationsInfo.emplace_back(
1195  memberIndex, memberDecoration.first, UnitAttr::get(context));
1196  }
1197  }
1198  }
1199  }
1200  }
1201  }
1202 
1203  SmallVector<spirv::StructType::StructDecorationInfo, 0> structDecorationsInfo;
1204  if (decorations.count(operands[0])) {
1205  NamedAttrList &allDecorations = decorations[operands[0]];
1206  for (NamedAttribute &decorationAttr : allDecorations) {
1207  std::optional<spirv::Decoration> decoration = spirv::symbolizeDecoration(
1208  llvm::convertToCamelFromSnakeCase(decorationAttr.getName(), true));
1209  assert(decoration.has_value());
1210  structDecorationsInfo.emplace_back(decoration.value(),
1211  decorationAttr.getValue());
1212  }
1213  }
1214 
1215  uint32_t structID = operands[0];
1216  std::string structIdentifier = nameMap.lookup(structID).str();
1217 
1218  if (structIdentifier.empty()) {
1219  assert(unresolvedMemberTypes.empty() &&
1220  "didn't expect unresolved member types");
1221  typeMap[structID] = spirv::StructType::get(
1222  memberTypes, offsetInfo, memberDecorationsInfo, structDecorationsInfo);
1223  } else {
1224  auto structTy = spirv::StructType::getIdentified(context, structIdentifier);
1225  typeMap[structID] = structTy;
1226 
1227  if (!unresolvedMemberTypes.empty())
1228  deferredStructTypesInfos.push_back(
1229  {structTy, unresolvedMemberTypes, memberTypes, offsetInfo,
1230  memberDecorationsInfo, structDecorationsInfo});
1231  else if (failed(structTy.trySetBody(memberTypes, offsetInfo,
1232  memberDecorationsInfo,
1233  structDecorationsInfo)))
1234  return failure();
1235  }
1236 
1237  // TODO: Update StructType to have member name as attribute as
1238  // well.
1239  return success();
1240 }

1391 LogicalResult spirv::Deserializer::processConstant(ArrayRef<uint32_t> operands,
1392  bool isSpec) {
1393  StringRef opname = isSpec ? "OpSpecConstant" : "OpConstant";
1394

1397  << opname << " must have type <id> and result <id>";
1398  }
1399  if (operands.size() < 3) {
1400  return emitError(unknownLoc)
1401  << opname << " must have at least 1 more parameter";
1402  }
1403 
1404  Type resultType = getType(operands[0]);
1405  if (!resultType) {
1406  return emitError(unknownLoc, "undefined result type from <id> ")
1407  << operands[0];
1408  }
1409 
1410  auto checkOperandSizeForBitwidth = [&](unsigned bitwidth) -> LogicalResult {
1411  if (bitwidth == 64) {
1412  if (operands.size() == 4) {
1413  return success();
1414  }
1415  return emitError(unknownLoc)
1416  << opname << " should have 2 parameters for 64-bit values";
1417  }
1418  if (bitwidth <= 32) {
1419  if (operands.size() == 3) {
1420  return success();
1421  }
1422 
1423  return emitError(unknownLoc)
1424  << opname
1425  << " should have 1 parameter for values with no more than 32 bits";
1426  }
1427  return emitError(unknownLoc, "unsupported OpConstant bitwidth: ")
1428  << bitwidth;
1429  };
1430 
1431  auto resultID = operands[1];
1432 
1433  if (auto intType = dyn_cast<IntegerType>(resultType)) {
1434  auto bitwidth = intType.getWidth();
1435  if (failed(checkOperandSizeForBitwidth(bitwidth))) {
1436  return failure();
1437  }
1438 
1439  APInt value;
1440  if (bitwidth == 64) {
1441  // 64-bit integers are represented with two SPIR-V words. According to
1442  // SPIR-V spec: "When the types bit width is larger than one word, the
1443  // literals low-order words appear first."
1444  struct DoubleWord {
1445  uint32_t word1;
1446  uint32_t word2;
1447  } words = {operands[2], operands[3]};
1448  value = APInt(64, llvm::bit_cast<uint64_t>(words), /*isSigned=*/true);
1449  } else if (bitwidth <= 32) {
1450  value = APInt(bitwidth, operands[2], /*isSigned=*/true,
1451  /*implicitTrunc=*/true);
1452  }
1453 
1454  auto attr = opBuilder.getIntegerAttr(intType, value);
1455 
1456  if (isSpec) {
1457  createSpecConstant(unknownLoc, resultID, attr);
1458  } else {
1459  // For normal constants, we just record the attribute (and its type) for
1460  // later materialization at use sites.
1461  constantMap.try_emplace(resultID, attr, intType);
1462  }
1463 
1464  return success();
1465  }
1466 
1467  if (auto floatType = dyn_cast<FloatType>(resultType)) {
1468  auto bitwidth = floatType.getWidth();
1469  if (failed(checkOperandSizeForBitwidth(bitwidth))) {
1470  return failure();
1471  }
1472 
1473  APFloat value(0.f);
1474  if (floatType.isF64()) {
1475  // Double values are represented with two SPIR-V words. According to
1476  // SPIR-V spec: "When the types bit width is larger than one word, the
1477  // literals low-order words appear first."
1478  struct DoubleWord {
1479  uint32_t word1;
1480  uint32_t word2;
1481  } words = {operands[2], operands[3]};
1482  value = APFloat(llvm::bit_cast<double>(words));
1483  } else if (floatType.isF32()) {
1484  value = APFloat(llvm::bit_cast<float>(operands[2]));
1485  } else if (floatType.isF16()) {
1486  APInt data(16, operands[2]);
1487  value = APFloat(APFloat::IEEEhalf(), data);
1488  } else if (floatType.isBF16()) {
1489  APInt data(16, operands[2]);
1490  value = APFloat(APFloat::BFloat(), data);
1491  }
1492 
1493  auto attr = opBuilder.getFloatAttr(floatType, value);
1494  if (isSpec) {
1495  createSpecConstant(unknownLoc, resultID, attr);
1496  } else {
1497  // For normal constants, we just record the attribute (and its type) for
1498  // later materialization at use sites.
1499  constantMap.try_emplace(resultID, attr, floatType);
1500  }
1501 
1502  return success();
1503  }
1504 
1505  return emitError(unknownLoc, "OpConstant can only generate values of "
1506  "scalar integer or floating-point type");
1507 }

1721 Value spirv::Deserializer::materializeSpecConstantOperation(
1722  uint32_t resultID, spirv::Opcode enclosedOpcode, uint32_t resultTypeID,
1723  ArrayRef<uint32_t> enclosedOpOperands) {
1724 
1725  Type resultType = getType(resultTypeID);
1726 
1727  // Instructions wrapped by OpSpecConstantOp need an ID for their
1728  // Deserializer::processOp<op_name>(...) to emit the corresponding SPIR-V
1729  // dialect wrapped op. For that purpose, a new value map is created and "fake"
1730  // ID in that map is assigned to the result of the enclosed instruction. Note
1731  // that there is no need to update this fake ID since we only need to
1732  // reference the created Value for the enclosed op from the spv::YieldOp
1733  // created later in this method (both of which are the only values in their
1734  // region: the SpecConstantOperation's region). If we encounter another
1735  // SpecConstantOperation in the module, we simply re-use the fake ID since the
1736  // previous Value assigned to it isn't visible in the current scope anyway.
1737  DenseMap<uint32_t, Value> newValueMap;
1738  llvm::SaveAndRestore valueMapGuard(valueMap, newValueMap);
1739  constexpr uint32_t fakeID = static_cast<uint32_t>(-3);
1740 
1741  SmallVector<uint32_t, 4> enclosedOpResultTypeAndOperands;
1742  enclosedOpResultTypeAndOperands.push_back(resultTypeID);
1743  enclosedOpResultTypeAndOperands.push_back(fakeID);
1744  enclosedOpResultTypeAndOperands.append(enclosedOpOperands.begin(),
1745  enclosedOpOperands.end());
1746 
1747  // Process enclosed instruction before creating the enclosing
1748  // specConstantOperation (and its region). This way, references to constants,
1749  // global variables, and spec constants will be materialized outside the new
1750  // op's region. For more info, see Deserializer::getValue's implementation.
1751  if (failed(
1752  processInstruction(enclosedOpcode, enclosedOpResultTypeAndOperands)))
1753  return Value();
1754 
1755  // Since the enclosed op is emitted in the current block, split it in a
1756  // separate new block.
1757  Block *enclosedBlock = curBlock->splitBlock(&curBlock->back());
1758 
1759  auto loc = createFileLineColLoc(opBuilder);
1760  auto specConstOperationOp =
1761  spirv::SpecConstantOperationOp::create(opBuilder, loc, resultType);
1762 
1763  Region &body = specConstOperationOp.getBody();
1764  // Move the new block into SpecConstantOperation's body.
1765  body.getBlocks().splice(body.end(), curBlock->getParent()->getBlocks(),
1766  Region::iterator(enclosedBlock));
1767  Block &block = body.back();
1768 
1769  // RAII guard to reset the insertion point to the module's region after
1770  // deserializing the body of the specConstantOperation.
1771  OpBuilder::InsertionGuard moduleInsertionGuard(opBuilder);
1772  opBuilder.setInsertionPointToEnd(&block);
1773 
1774  spirv::YieldOp::create(opBuilder, loc, block.front().getResult(0));
1775  return specConstOperationOp.getResult();
1776 }

2092 LogicalResult ControlFlowStructurizer::structurize() {
2093  Operation *op = nullptr;
2094  bool isLoop = continueBlock != nullptr;
2095  if (isLoop) {
2096  if (auto loopOp = createLoopOp(control))
2097  op = loopOp.getOperation();
2098  } else {
2099  if (auto selectionOp = createSelectionOp(control))
2100  op = selectionOp.getOperation();
2101  }
2102  if (!op)
2103  return failure();
2104  Region &body = op->getRegion(0);
2105 
2106  IRMapping mapper;
2107  // All references to the old merge block should be directed to the
2108  // selection/loop merge block in the SelectionOp/LoopOp's region.
2109  mapper.map(mergeBlock, &body.back());
2110 
2111  collectBlocksInConstruct();
2112 
2113  // We've identified all blocks belonging to the selection/loop's region. Now
2114  // need to "move" them into the selection/loop. Instead of really moving the
2115  // blocks, in the following we copy them and remap all values and branches.
2116  // This is because:
2117  // * Inserting a block into a region requires the block not in any region
2118  // before. But selections/loops can nest so we can create selection/loop ops
2119  // in a nested manner, which means some blocks may already be in a
2120  // selection/loop region when to be moved again.
2121  // * It's much trickier to fix up the branches into and out of the loop's
2122  // region: we need to treat not-moved blocks and moved blocks differently:
2123  // Not-moved blocks jumping to the loop header block need to jump to the
2124  // merge point containing the new loop op but not the loop continue block's
2125  // back edge. Moved blocks jumping out of the loop need to jump to the
2126  // merge block inside the loop region but not other not-moved blocks.
2127  // We cannot use replaceAllUsesWith clearly and it's harder to follow the
2128  // logic.
2129 
2130  // Create a corresponding block in the SelectionOp/LoopOp's region for each
2131  // block in this loop construct.
2132  OpBuilder builder(body);
2133  for (auto *block : constructBlocks) {
2134  // Create a block and insert it before the selection/loop merge block in the
2135  // SelectionOp/LoopOp's region.
2136  auto *newBlock = builder.createBlock(&body.back());
2137  mapper.map(block, newBlock);
2138  LLVM_DEBUG(logger.startLine() << "[cf] cloned block " << newBlock
2139  << " from block " << block << "\n");
2140  if (!isFnEntryBlock(block)) {
2141  for (BlockArgument blockArg : block->getArguments()) {
2142  auto newArg =
2143  newBlock->addArgument(blockArg.getType(), blockArg.getLoc());
2144  mapper.map(blockArg, newArg);
2145  LLVM_DEBUG(logger.startLine() << "[cf] remapped block argument "
2146  << blockArg << " to " << newArg << "\n");
2147  }
2148  } else {
2149  LLVM_DEBUG(logger.startLine()
2150  << "[cf] block " << block << " is a function entry block\n");
2151  }
2152 
2153  for (auto &op : *block)
2154  newBlock->push_back(op.clone(mapper));
2155  }
2156 
2157  // Go through all ops and remap the operands.
2158  auto remapOperands = [&](Operation *op) {
2159  for (auto &operand : op->getOpOperands())
2160  if (Value mappedOp = mapper.lookupOrNull(operand.get()))
2161  operand.set(mappedOp);
2162  for (auto &succOp : op->getBlockOperands())
2163  if (Block *mappedOp = mapper.lookupOrNull(succOp.get()))
2164  succOp.set(mappedOp);
2165  };
2166  for (auto &block : body)
2167  block.walk(remapOperands);
2168 
2169  // We have created the SelectionOp/LoopOp and "moved" all blocks belonging to
2170  // the selection/loop construct into its region. Next we need to fix the
2171  // connections between this new SelectionOp/LoopOp with existing blocks.
2172 
2173  // All existing incoming branches should go to the merge block, where the
2174  // SelectionOp/LoopOp resides right now.
2175  headerBlock->replaceAllUsesWith(mergeBlock);
2176 
2177  LLVM_DEBUG({
2178  logger.startLine() << "[cf] after cloning and fixing references:\n";
2179  headerBlock->getParentOp()->print(logger.getOStream());
2180  logger.startLine() << "\n";
2181  });
2182 
2183  if (isLoop) {
2184  if (!mergeBlock->args_empty()) {
2185  return mergeBlock->getParentOp()->emitError(
2186  "OpPhi in loop merge block unsupported");
2187  }
2188 
2189  // The loop header block may have block arguments. Since now we place the
2190  // loop op inside the old merge block, we need to make sure the old merge
2191  // block has the same block argument list.
2192  for (BlockArgument blockArg : headerBlock->getArguments())
2193  mergeBlock->addArgument(blockArg.getType(), blockArg.getLoc());
2194 
2195  // If the loop header block has block arguments, make sure the spirv.Branch
2196  // op matches.
2197  SmallVector<Value, 4> blockArgs;
2198  if (!headerBlock->args_empty())
2199  blockArgs = {mergeBlock->args_begin(), mergeBlock->args_end()};
2200 
2201  // The loop entry block should have a unconditional branch jumping to the
2202  // loop header block.
2203  builder.setInsertionPointToEnd(&body.front());
2204  spirv::BranchOp::create(builder, location, mapper.lookupOrNull(headerBlock),
2205  ArrayRef<Value>(blockArgs));
2206  }
2207 
2208  // Values defined inside the selection region that need to be yielded outside
2209  // the region.
2210  SmallVector<Value> valuesToYield;
2211  // Outside uses of values that were sunk into the selection region. Those uses
2212  // will be replaced with values returned by the SelectionOp.
2213  SmallVector<Value> outsideUses;
2214 
2215  // Move block arguments of the original block (`mergeBlock`) into the merge
2216  // block inside the selection (`body.back()`). Values produced by block
2217  // arguments will be yielded by the selection region. We do not update uses or
2218  // erase original block arguments yet. It will be done later in the code.
2219  //
2220  // Code below is not executed for loops as it would interfere with the logic
2221  // above. Currently block arguments in the merge block are not supported, but
2222  // instead, the code above copies those arguments from the header block into
2223  // the merge block. As such, running the code would yield those copied
2224  // arguments that is most likely not a desired behaviour. This may need to be
2225  // revisited in the future.
2226  if (!isLoop)
2227  for (BlockArgument blockArg : mergeBlock->getArguments()) {
2228  // Create new block arguments in the last block ("merge block") of the
2229  // selection region. We create one argument for each argument in
2230  // `mergeBlock`. This new value will need to be yielded, and the original
2231  // value replaced, so add them to appropriate vectors.
2232  body.back().addArgument(blockArg.getType(), blockArg.getLoc());
2233  valuesToYield.push_back(body.back().getArguments().back());
2234  outsideUses.push_back(blockArg);
2235  }
2236 
2237  // All the blocks cloned into the SelectionOp/LoopOp's region can now be
2238  // cleaned up.
2239  LLVM_DEBUG(logger.startLine() << "[cf] cleaning up blocks after clone\n");
2240  // First we need to drop all operands' references inside all blocks. This is
2241  // needed because we can have blocks referencing SSA values from one another.
2242  for (auto *block : constructBlocks)
2243  block->dropAllReferences();
2244 
2245  // All internal uses should be removed from original blocks by now, so
2246  // whatever is left is an outside use and will need to be yielded from
2247  // the newly created selection / loop region.
2248  for (Block *block : constructBlocks) {
2249  for (Operation &op : *block) {
2250  if (!op.use_empty())
2251  for (Value result : op.getResults()) {
2252  valuesToYield.push_back(mapper.lookupOrNull(result));
2253  outsideUses.push_back(result);
2254  }
2255  }
2256  for (BlockArgument &arg : block->getArguments()) {
2257  if (!arg.use_empty()) {
2258  valuesToYield.push_back(mapper.lookupOrNull(arg));
2259  outsideUses.push_back(arg);
2260  }
2261  }
2262  }
2263 
2264  assert(valuesToYield.size() == outsideUses.size());
2265 
2266  // If we need to yield any values from the selection / loop region we will
2267  // take care of it here.
2268  if (!valuesToYield.empty()) {
2269  LLVM_DEBUG(logger.startLine()
2270  << "[cf] yielding values from the selection / loop region\n");
2271 
2272  // Update `mlir.merge` with values to be yield.
2273  auto mergeOps = body.back().getOps<spirv::MergeOp>();
2274  Operation *merge = llvm::getSingleElement(mergeOps);
2275  assert(merge);
2276  merge->setOperands(valuesToYield);
2277 
2278  // MLIR does not allow changing the number of results of an operation, so
2279  // we create a new SelectionOp / LoopOp with required list of results and
2280  // move the region from the initial SelectionOp / LoopOp. The initial
2281  // operation is then removed. Since we move the region to the new op all
2282  // links between blocks and remapping we have previously done should be
2283  // preserved.
2284  builder.setInsertionPoint(&mergeBlock->front());
2285 
2286  Operation *newOp = nullptr;
2287 
2288  if (isLoop)
2289  newOp = spirv::LoopOp::create(builder, location,
2290  TypeRange(ValueRange(outsideUses)),
2291  static_cast<spirv::LoopControl>(control));
2292  else
2293  newOp = spirv::SelectionOp::create(
2294  builder, location, TypeRange(ValueRange(outsideUses)),
2295  static_cast<spirv::SelectionControl>(control));
2296 
2297  newOp->getRegion(0).takeBody(body);
2298 
2299  // Remove initial op and swap the pointer to the newly created one.
2300  op->erase();
2301  op = newOp;
2302 
2303  // Update all outside uses to use results of the SelectionOp / LoopOp and
2304  // remove block arguments from the original merge block.
2305  for (unsigned i = 0, e = outsideUses.size(); i != e; ++i)
2306  outsideUses[i].replaceAllUsesWith(op->getResult(i));
2307 
2308  // We do not support block arguments in loop merge block. Also running this
2309  // function with loop would break some of the loop specific code above
2310  // dealing with block arguments.
2311  if (!isLoop)
2312  mergeBlock->eraseArguments(0, mergeBlock->getNumArguments());
2313  }
2314 
2315  // Check that whether some op in the to-be-erased blocks still has uses. Those
2316  // uses come from blocks that won't be sinked into the SelectionOp/LoopOp's
2317  // region. We cannot handle such cases given that once a value is sinked into
2318  // the SelectionOp/LoopOp's region, there is no escape for it.
2319  for (auto *block : constructBlocks) {
2320  for (Operation &op : *block)
2321  if (!op.use_empty())
2322  return op.emitOpError("failed control flow structurization: value has "
2323  "uses outside of the "
2324  "enclosing selection/loop construct");
2325  for (BlockArgument &arg : block->getArguments())
2326  if (!arg.use_empty())
2327  return emitError(arg.getLoc(), "failed control flow structurization: "
2328  "block argument has uses outside of the "
2329  "enclosing selection/loop construct");
2330  }
2331 
2332  // Then erase all old blocks.
2333  for (auto *block : constructBlocks) {
2334  // We've cloned all blocks belonging to this construct into the structured
2335  // control flow op's region. Among these blocks, some may compose another
2336  // selection/loop. If so, they will be recorded within blockMergeInfo.
2337  // We need to update the pointers there to the newly remapped ones so we can
2338  // continue structurizing them later.
2339  //
2340  // We need to walk each block as constructBlocks do not include blocks
2341  // internal to ops already structured within those blocks. It is not
2342  // fully clear to me why the mergeInfo of blocks (yet to be structured)
2343  // inside already structured selections/loops get invalidated and needs
2344  // updating, however the following example code can cause a crash (depending
2345  // on the structuring order), when the most inner selection is being
2346  // structured after the outer selection and loop have been already
2347  // structured:
2348  //
2349  // spirv.mlir.for {
2350  // // ...
2351  // spirv.mlir.selection {
2352  // // ..
2353  // // A selection region that hasn't been yet structured!
2354  // // ..
2355  // }
2356  // // ...
2357  // }
2358  //
2359  // If the loop gets structured after the outer selection, but before the
2360  // inner selection. Moving the already structured selection inside the loop
2361  // will invalidate the mergeInfo of the region that is not yet structured.
2362  // Just going over constructBlocks will not check and updated header blocks
2363  // inside the already structured selection region. Walking block fixes that.
2364  //
2365  // TODO: If structuring was done in a fixed order starting with inner
2366  // most constructs this most likely not be an issue and the whole code
2367  // section could be removed. However, with the current non-deterministic
2368  // order this is not possible.
2369  //
2370  // TODO: The asserts in the following assumes input SPIR-V blob forms
2371  // correctly nested selection/loop constructs. We should relax this and
2372  // support error cases better.
2373  auto updateMergeInfo = [&](Block *block) -> WalkResult {
2374  auto it = blockMergeInfo.find(block);
2375  if (it != blockMergeInfo.end()) {
2376  // Use the original location for nested selection/loop ops.
2377  Location loc = it->second.loc;
2378 
2379  Block *newHeader = mapper.lookupOrNull(block);
2380  if (!newHeader)
2381  return emitError(loc, "failed control flow structurization: nested "
2382  "loop header block should be remapped!");
2383 
2384  Block *newContinue = it->second.continueBlock;
2385  if (newContinue) {
2386  newContinue = mapper.lookupOrNull(newContinue);
2387  if (!newContinue)
2388  return emitError(loc, "failed control flow structurization: nested "
2389  "loop continue block should be remapped!");
2390  }
2391 
2392  Block *newMerge = it->second.mergeBlock;
2393  if (Block *mappedTo = mapper.lookupOrNull(newMerge))
2394  newMerge = mappedTo;
2395 
2396  // The iterator should be erased before adding a new entry into
2397  // blockMergeInfo to avoid iterator invalidation.
2398  blockMergeInfo.erase(it);
2399  blockMergeInfo.try_emplace(newHeader, loc, it->second.control, newMerge,
2400  newContinue);
2401  }
2402 
2403  return WalkResult::advance();
2404  };
2405 
2406  if (block->walk(updateMergeInfo).wasInterrupted())
2407  return failure();
2408 
2409  // The structured selection/loop's entry block does not have arguments.
2410  // If the function's header block is also part of the structured control
2411  // flow, we cannot just simply erase it because it may contain arguments
2412  // matching the function signature and used by the cloned blocks.
2413  if (isFnEntryBlock(block)) {
2414  LLVM_DEBUG(logger.startLine() << "[cf] changing entry block " << block
2415  << " to only contain a spirv.Branch op\n");
2416  // Still keep the function entry block for the potential block arguments,
2417  // but replace all ops inside with a branch to the merge block.
2418  block->clear();
2419  builder.setInsertionPointToEnd(block);
2420  spirv::BranchOp::create(builder, location, mergeBlock);
2421  } else {
2422  LLVM_DEBUG(logger.startLine() << "[cf] erasing block " << block << "\n");
2423  block->erase();
2424  }
2425  }
2426 
2427  LLVM_DEBUG(logger.startLine()
2428  << "[cf] after structurizing construct with header block "
2429  << headerBlock << ":\n"
2430  << *op << "\n");
2431

2433 }

2515 LogicalResult spirv::Deserializer::splitConditionalBlocks() {
2516  // Create a copy, so we can modify keys in the original.
2517  BlockMergeInfoMap blockMergeInfoCopy = blockMergeInfo;
2518  for (auto it = blockMergeInfoCopy.begin(), e = blockMergeInfoCopy.end();
2519  it != e; ++it) {
2520  auto &[block, mergeInfo] = *it;
2521 
2522  // Skip processing loop regions. For loop regions continueBlock is non-null.
2523  if (mergeInfo.continueBlock)
2524  continue;
2525 
2526  if (!block->mightHaveTerminator())
2527  continue;
2528 
2529  Operation *terminator = block->getTerminator();
2530  assert(terminator);
2531 
2532  if (!isa<spirv::BranchConditionalOp>(terminator))
2533  continue;
2534 
2535  // Check if the current header block is a merge block of another construct.
2536  bool splitHeaderMergeBlock = false;
2537  for (const auto &[_, mergeInfo] : blockMergeInfo) {
2538  if (mergeInfo.mergeBlock == block)
2539  splitHeaderMergeBlock = true;
2540  }
2541 
2542  // Do not split a block that only contains a conditional branch, unless it
2543  // is also a merge block of another construct - in that case we want to
2544  // split the block. We do not want two constructs to share header / merge
2545  // block.
2546  if (!llvm::hasSingleElement(*block) || splitHeaderMergeBlock) {
2547  Block *newBlock = block->splitBlock(terminator);
2548  OpBuilder builder(block, block->end());
2549  spirv::BranchOp::create(builder, block->getParent()->getLoc(), newBlock);
2550 
2551  // After splitting we need to update the map to use the new block as a
2552  // header.
2553  blockMergeInfo.erase(block);
2554  blockMergeInfo.try_emplace(newBlock, mergeInfo);
2555  }
2556  }
2557
