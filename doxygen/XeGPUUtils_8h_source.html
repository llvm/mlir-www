32 /// If tensor descriptor has a layout attribute it is used in SIMT mode.
33 /// In this mode, the distributed vector shape is determined as follows:
34 /// Definitions:
35 /// lane_data_size = lane_data[0] × lane_data[1]
36 /// subgroup_size = lane_layout[0] × lane_layout[1]
37 /// distribution_unit_size = subgroup_size × lane_data_size
38 ///
39 /// Case 1: Regular loads/stores.
40 /// The following conditions must be met:
41 /// * tensor_desc[0] == lane_layout[0]
42 /// Distributed vector is a 1D vector with shape:
43 /// [chunk_size]
44 ///
45 /// Case 2: Block loads/stores
46 /// Additional definitions:
47 /// tensor_size = tensor_desc[0] * .. * tensor_desc[r-1] * array_length
48 /// n_distribution_units = tensor_size / distribution_unit_size
49 /// fragment_size = n_distribution_units * lane_data_size
50 /// Given above definitions, the following conditions must be met:
51 /// * tensor_desc[0] % (lane_layout[0] × lane_data[0]) == 0
52 /// * tensor_desc[1] % (lane_layout[1] × lane_data[1]) == 0
53 /// Distributed vector is a 1D vector with shape:
54 /// [fragment_size]
55 FailureOr<VectorType> getDistributedVectorType(xegpu::TensorDescType tdescTy);
