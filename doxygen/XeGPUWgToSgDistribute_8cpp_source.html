37 // Check if there is sg id range attached to the scf.if op.
38 static bool isSgIdRangeSpecified(Operation *op, int64_t &startOfRange,
39  int64_t &endOfRange) {
40  Operation *parent = op->getParentOp();
41  // Find the outermost scf::IfOp with xegpu.sg_id_range.
42  while (parent) {
43  if (auto ifOp = dyn_cast<scf::IfOp>(parent)) {
44  if (auto attr = llvm::dyn_cast_or_null<xegpu::RangeAttr>(
45  ifOp->getAttr("sg_id_range"))) {
46  startOfRange = attr.getStart().getInt();
47  endOfRange = attr.getEnd().getInt();
48  break;
49  }
50  }
51  parent = parent->getParentOp();
52  }
53  // Return false if startOfRange is 0
54  return (startOfRange > 0 && endOfRange > startOfRange);
55 }

57 static std::pair<SmallVector<int64_t>, int>
58 getSgShapeAndCount(ArrayRef<int64_t> shape, xegpu::LayoutAttr layout) {
59  int count = 1;
60  SmallVector<int64_t> sgShape(shape);
61 
62  if (layout && layout.isWgLayout()) {
63  DenseI32ArrayAttr sgLayoutAttr = layout.getSgLayout();
64  auto sgLayout = llvm::to_vector_of<int64_t>(sgLayoutAttr.asArrayRef());
65  if (DenseI32ArrayAttr sgDataAttr = layout.getSgData())
66  sgShape = llvm::to_vector_of<int64_t>(sgDataAttr.asArrayRef());
67  else
68  sgShape = computeShapeRatio(shape, sgLayout).value_or(sgShape);
69  SmallVector<int64_t> distUnit = computeElementwiseMul(sgLayout, sgShape);
70  // Clamp distUnit to the original shape to handle cases where data is
71  // shared among subgroups, which may cause distUnit to exceed the original
72  // shape.
73  for (size_t i = 0; i < distUnit.size(); ++i)
74  distUnit[i] = std::min(shape[i], distUnit[i]);
75  count = computeProduct(shape) / computeProduct(distUnit);
76  }
77  return std::make_pair(sgShape, count);
78 }

80 /// This pattern transforms the CreateNdDescOp to create a subgroup descriptor
81 /// from a workgroup descriptor. It replaces the offsets and sizes with
82 /// appropriate values for the subgroup.
83 /// It uses round-robin assignment to distribute the work to the subgroups.
84 /// Following create_nd_desc operation:,
85 /// %tdesc = xegpu.create_nd_tdesc %src[0, 0] : memref<24x24xf32>
86 /// -> !xegpu.tensor_desc<24x24xf32, #xegpu.layout<sg_layout = [4, 4],
87 /// sg_data = [2, 2], lane_layout = [2, 2], lane_data = [1, 1]>>
88 /// is converted to 9 subgroup level operations based on the sg_layout &
89 /// sg_data:
90 /// %tdesc = xegpu.create_nd_tdesc %src[off1, off2] : memref<24x24xf32> ->
91 /// !xegpu.tensor_desc<2x2xf32, #xegpu.layout<lane_layout = [2, 2],
92 /// lane_data = [1, 1]>>
93 ///
94 /// The sg_layout and sg_data attributes are dropped after the pass as they are
95 /// no longer needed.
96 ///
97 /// 24x24 matrix distribution example:
98 /// sg_layout = [4, 4], sg_data = [2, 2]
99 /// Each 8x8 matrix within the 24x24 matrix is called a distribution unit.
100 /// dist_unit_shape = [8, 8] --> sg_layout[i] * sg_data[i]
101 ///
102 /// +------------------------+
103 /// | 8x8 | 8x8 | 8x8 | <- 3 tiles across
104 /// |-----+-----+-----|
105 /// | 8x8 | 8x8 | 8x8 | <- 3 tiles down
106 /// |-----+-----+-----|
107 /// | 8x8 | 8x8 | 8x8 |
108 /// +------------------------+
109 ///
110 /// Each 8x8 tile is further subdivided among subgroups:
111 /// +------------------------+
112 /// | 2x2 2x2 2x2 2x2 | <- 4 subgroups across (each handles 2 columns)
113 /// | 2x2 2x2 2x2 2x2 | <- 4 subgroups down (each handles 2 rows)
114 /// | 2x2 2x2 2x2 2x2 |
115 /// | 2x2 2x2 2x2 2x2 |
116 /// +------------------------+
117 ///
118 /// Since the 24x24 matrix is divided into 8x8 distribution units, there will be
119 /// 9 distribution units (3x3) in total. Hence the 9 subgroup level operations.
120 
121 /// The pass currently has entire distribution logic in the WgToSgCreateNdOp
122 /// pattern and all the other ops just follow.
123 /// TODO: Decouple the distribution logic from WgToSgCreateNdOp for all the
124 /// ops in the pass.
125 struct WgToSgCreateNdOp : public OpConversionPattern<xegpu::CreateNdDescOp> {
126  using OpConversionPattern<xegpu::CreateNdDescOp>::OpConversionPattern;
127 
128  // Calculate offset for each subgroup
129  static SmallVector<OpFoldResult>
130  calculateGlobalOffsets(ConversionPatternRewriter &rewriter, Location loc,
131  const SmallVector<OpFoldResult> &originalOffsets,
132  const SmallVector<Value> &localOffset,
133  const SmallVector<int64_t> &distUnitBaseAddr,
134  const SmallVector<int64_t> &distUnitShape) {
135  assert(localOffset.size() == distUnitBaseAddr.size() &&
136  "localOffset and distUnitBaseAddr must have the same rank");
137 
138  SmallVector<OpFoldResult> globalOffsets(originalOffsets.begin(),
139  originalOffsets.end());
140  size_t rank = localOffset.size();
141  for (size_t i = 0; i < rank; ++i) {
142  size_t dimIdx = originalOffsets.size() - rank + i;
143  Value constOffset =
144  arith::ConstantIndexOp::create(rewriter, loc, distUnitBaseAddr[i]);
145  Value offset =
146  rewriter.createOrFold<index::AddOp>(loc, localOffset[i], constOffset);
147  Value modValue =
148  arith::ConstantIndexOp::create(rewriter, loc, distUnitShape[i]);
149  Value offsetMod =
150  rewriter.createOrFold<index::RemUOp>(loc, offset, modValue);
151  Value origOffset = getValueOrCreateConstantIndexOp(
152  rewriter, loc, originalOffsets[dimIdx]);
153  Value globalOffset =
154  rewriter.createOrFold<index::AddOp>(loc, origOffset, offsetMod);
155  globalOffsets[dimIdx] = globalOffset;
156  }
157 
158  return globalOffsets;
159  }

161  LogicalResult
162  matchAndRewrite(xegpu::CreateNdDescOp op, OneToNOpAdaptor adaptor,
163  ConversionPatternRewriter &rewriter) const override {
164  Location loc = op.getLoc();
165  MLIRContext *ctx = op.getContext();
166  xegpu::TensorDescType tdescTy = op.getType();
167  auto layout = dyn_cast<xegpu::LayoutAttr>(tdescTy.getLayout());
168  if (!layout)
169  return failure();
170  Type elemTy = tdescTy.getElementType();
171  ArrayRef<int64_t> wgShape = tdescTy.getShape();
172  // sgLayout must be present for workgroup-level distribution.
173  SmallVector<int64_t> sgLayout;
174  if (auto sgLayoutAttr = layout.getSgLayout())
175  sgLayout = llvm::to_vector_of<int64_t>(sgLayoutAttr.asArrayRef());
176  else
177  return rewriter.notifyMatchFailure(
178  op, "sgLayout attribute is required in layout");
179 
180  SmallVector<int64_t> sgShape = getSgShapeAndCount(wgShape, layout).first;
181 
182  // TODO : Handle order attribute
183  // Get the subgroup ID
184  auto linearSgId =
185  gpu::SubgroupIdOp::create(rewriter, loc, /*upper_bound=*/nullptr);
186 
187  // Create constants for layout dimensions
188  SmallVector<Value> sgLayoutDim(sgLayout.size());
189  SmallVector<Value> sgDataDim(sgShape.size());
190 
191  for (size_t i = 0; i < sgLayout.size(); i++) {
192  sgLayoutDim[i] =
193  arith::ConstantIndexOp::create(rewriter, loc, sgLayout[i]);
194  sgDataDim[i] = arith::ConstantIndexOp::create(rewriter, loc, sgShape[i]);
195  }
196 
197  int64_t startOfRange = -1, endOfRange = -1;
198  bool sgIdRangeSpecified =
199  isSgIdRangeSpecified(op, startOfRange, endOfRange);
200 
201  Value adjustedSgId = linearSgId;
202  if (sgIdRangeSpecified) {
203  int64_t sgCount = endOfRange - startOfRange;
204  if (computeProduct(sgLayout) != sgCount)
205  return rewriter.notifyMatchFailure(
206  op, "sg_layout size must match the sg_id_range");
207  // Subtract startOfRange from the original subgroup id to get the adjusted
208  // sg id
209  Value startOfRangeVal =
210  arith::ConstantIndexOp::create(rewriter, loc, startOfRange);
211  adjustedSgId =
212  rewriter.createOrFold<index::SubOp>(loc, linearSgId, startOfRangeVal);
213  }
214 
215  auto deLinearizeSgId =
216  affine::delinearizeIndex(rewriter, loc, adjustedSgId, sgLayoutDim);
217  if (failed(deLinearizeSgId))
218  return failure();
219  SmallVector<Value> sgIds = *deLinearizeSgId;
220 
221  // Calculate distribution unit shape and local offsets for subgroup
222  SmallVector<int64_t> distUnitShape(sgLayout.size());
223  SmallVector<Value> localOffset(sgLayout.size());
224  for (size_t i = 0; i < sgLayout.size(); i++) {
225  distUnitShape[i] = std::min(sgLayout[i] * sgShape[i], wgShape[i]);
226  localOffset[i] =
227  rewriter.createOrFold<index::MulOp>(loc, sgIds[i], sgDataDim[i]);
228  }
229 
230  SmallVector<OpFoldResult> originalOffsets = op.getMixedOffsets();
231 
232  xegpu::TensorDescType newTdescTy =
233  xegpu::TensorDescType::get(ctx, sgShape, elemTy, tdescTy.getEncoding(),
234  layout.dropSgLayoutAndData());
235  SmallVector<Value> newCreateNdOps;
236  for (SmallVector<int64_t> distUnitBaseAddr :
237  StaticTileOffsetRange(wgShape, distUnitShape)) {
238  SmallVector<OpFoldResult> globalOffsets =
239  calculateGlobalOffsets(rewriter, loc, originalOffsets, localOffset,
240  distUnitBaseAddr, distUnitShape);
241 
242  auto newCreateNdOp = xegpu::CreateNdDescOp::create(
243  rewriter, loc, newTdescTy, op.getSource(), globalOffsets,
244  op.getMixedSizes(), op.getMixedStrides());
245  newCreateNdOps.push_back(newCreateNdOp);
246  }
247 
248  rewriter.replaceOpWithMultiple(op, {newCreateNdOps});
249  return success();
250  }
251 };

323 /// This pattern transforms the DpasOp to work at subgroup level.
324 struct WgToSgDpasOp : public OpConversionPattern<xegpu::DpasOp> {
325  using OpConversionPattern<xegpu::DpasOp>::OpConversionPattern;
326  LogicalResult
327  matchAndRewrite(xegpu::DpasOp op, OneToNOpAdaptor adaptor,
328  ConversionPatternRewriter &rewriter) const override {
329  Location loc = op.getLoc();
330  VectorType resultTy = op.getResult().getType();
331  if (resultTy.getRank() != 2)
332  return failure();
333 
334  auto originalLayout = xegpu::getLayoutAttr(op.getResult());
335  if (!originalLayout)
336  return failure();
337 
338  size_t i = 0;
339  SmallVector<Value> newDpasOps;
340  for (auto aVec : adaptor.getLhs()) {
341  for (auto bVec : adaptor.getRhs()) {
342 
343  llvm::SmallVector<Value> operands({aVec, bVec});
344  Value tmpC;
345  if (op.getAcc()) {
346  tmpC = adaptor.getAcc()[i++];
347  operands.push_back(tmpC);
348  }
349 
350  ArrayRef<int64_t> aVecShape =
351  llvm::cast<VectorType>(aVec.getType()).getShape();
352  ArrayRef<int64_t> bVecShape =
353  llvm::cast<VectorType>(bVec.getType()).getShape();
354  VectorType resTy = VectorType::get({aVecShape[0], bVecShape[1]},
355  resultTy.getElementType());
356  tmpC = xegpu::DpasOp::create(rewriter, loc, resTy, operands);
357  xegpu::setLayoutAttr(cast<OpResult>(tmpC),
358  originalLayout.dropSgLayoutAndData());
359 
360  newDpasOps.push_back(tmpC);
361  }
362  }
363  rewriter.replaceOpWithMultiple(op, {newDpasOps});
364  return success();
365  }

388 struct WgToSgVectorBroadcastOp
389  : public OpConversionPattern<vector::BroadcastOp> {

396  ArrayRef<int64_t> wgShape = resultType.getShape();
397 
398  xegpu::LayoutAttr layout = xegpu::getLayoutAttr(op.getResult());
399  if (!layout || !layout.getSgLayout())

404  auto srcType =
405  dyn_cast<VectorType>(adaptor.getOperands().front()[0].getType());
406  if (!srcType || srcType.getRank() != resultType.getRank())
407  return failure();
408 
409  SmallVector<int64_t> sgShape = getSgShapeAndCount(wgShape, layout).first;
410  VectorType newResultType =
411  VectorType::get(sgShape, resultType.getElementType());
412 
413  // Check if the output layout is distributable
414  SmallVector<int64_t> sgLayout;
415  if (auto sgLayoutAttr = layout.getSgLayout())
416  sgLayout = llvm::to_vector_of<int64_t>(sgLayoutAttr.asArrayRef());
417  else
418  return failure();
419 
420  if (!xegpu::XeGPUDialect::isEvenlyDistributable(wgShape, layout))
421  return failure();
422 
423  // Check if the srcShape has unit dim in dimensions being broadcasted,
424  // and the other dimensions are the same as the destination type
425  // TODO: Generalize it
426  auto srcShape = srcType.getShape();
427  for (size_t i = 0; i < srcShape.size(); ++i) {
428  if (srcShape[i] != 1 && srcShape[i] != sgShape[i])
429  return failure();
430  }
431 
432  SmallVector<Value> newBroadcastOps;
433  for (auto operand : adaptor.getOperands().front()) {
434  auto newBroadcast = vector::BroadcastOp::create(rewriter, op.getLoc(),
435  newResultType, operand);
436  xegpu::setLayoutAttr(newBroadcast->getResult(0),
437  layout.dropSgLayoutAndData());
438  newBroadcastOps.push_back(newBroadcast.getResult());
439  }
440 
441  rewriter.replaceOpWithMultiple(op, {newBroadcastOps});
442  return success();
443  }

446 // This pattern transforms elementwise ops to work at subgroup level.
447 struct WgToSgElementwiseOp : public ConversionPattern {
448  WgToSgElementwiseOp(MLIRContext *ctx)
449  : ConversionPattern(MatchAnyOpTypeTag(), /*benefit=*/1, ctx) {}
450 
451  LogicalResult
452  matchAndRewrite(Operation *op, ArrayRef<ValueRange> operands,
453  ConversionPatternRewriter &rewriter) const override {
454  // Only match ops with elementwise trait and single result.
455  if (!OpTrait::hasElementwiseMappableTraits(op) || op->getNumResults() != 1)
456  return failure();
457 
458  auto resultType = dyn_cast<VectorType>(op->getResult(0).getType());
459  assert(resultType && "Expected result to be a VectorType");
460 
461  ArrayRef<int64_t> wgShape = resultType.getShape();
462 
463  xegpu::LayoutAttr layout = xegpu::getLayoutAttr(op->getResult(0));
464  if (!layout || !layout.getSgLayout())
465  return failure();
466 
467  SmallVector<int64_t> sgShape = getSgShapeAndCount(wgShape, layout).first;
468 
469  size_t numVariants = operands.empty() ? 0 : operands.front().size();
470 
471  if (llvm::any_of(operands, [&](const ValueRange &operandVec) {
472  return operandVec.size() != numVariants;
473  }))
474  return failure();
475 
476  SmallVector<Value> newResults;
477  VectorType newResultType =
478  VectorType::get(sgShape, resultType.getElementType());
479 
480  for (size_t i = 0; i < numVariants; ++i) {
481  SmallVector<Value> opOperands;
482  for (auto &operandVec : operands)
483  opOperands.push_back(operandVec[i]);
484 
485  OperationState state(op->getLoc(), op->getName());
486  state.addOperands(opOperands);
487  state.addTypes(newResultType);
488  // Copy all attributes, but update "layout_result_0" to drop
489  // sgLayout/sgData
490  for (auto attr : op->getAttrs()) {
491  if (auto layout = dyn_cast<xegpu::LayoutAttr>(attr.getValue())) {
492  if (auto newLayout = layout.dropSgLayoutAndData())
493  state.addAttribute(attr.getName(), newLayout);
494  } else {
495  state.addAttribute(attr.getName(), attr.getValue());
496  }
497  }
498  Operation *newOp = rewriter.create(state);
499  newResults.push_back(newOp->getResult(0));
500  }
501 
502  rewriter.replaceOpWithMultiple(op, {newResults});
503  return success();
504  }

507 // clang-format off
508 // Pattern for lowering ConvertLayoutOp based on sg_layout and sg_data.
509 // If input_layout and target_layout have identical sg_layout and sg_data,
510 // the op is rewritten to a subgroup-level ConvertLayoutOp with these fields
511 // dropped. For example:
512 // #a = #xegpu.layout<sg_layout = [2, 2], sg_data = [16, 16], inst_data = [16, 16]>
513 // #b = #xegpu.layout<sg_layout = [2, 2], sg_data = [16, 16], inst_data = [8, 16]>
514 // xegpu.convert_layout %1 <{input_layout = #a, target_layout = #b}> : vector<32x64xf32>
515 // becomes:
516 // #a = #xegpu.layout<inst_data = [16, 16]>
517 // #b = #xegpu.layout<inst_data = [8, 16]>
518 // xegpu.convert_layout %1 <{input_layout = #a, target_layout = #b}> : vector<16x16xf32>
519 // (vector<16x16xf32> is determined by sg_data = [16, 16])
520 //
521 // If sg_layout or sg_data differ, SLM is used to redistribute data across subgroups.
522 // For example:
523 // #a = #xegpu.layout<sg_layout = [1, 4], sg_data = [32, 16], inst_data = [16, 16]>
524 // #b = #xegpu.layout<sg_layout = [2, 2], sg_data = [16, 32], inst_data = [8, 16]>
525 // xegpu.convert_layout %1 <{input_layout = #a, target_layout = #b}> : vector<32x64xf32>
526 // is lowered to:
527 // #a = #xegpu.layout<inst_data = [16, 16]>
528 // #b = #xegpu.layout<inst_data = [8, 16]>
529 // store_matrix %1, %slm <{layout_input_0 = #a}> : vector<32x16>, matrix_desc<32x64xf32>
530 // %d = load_matrix %slm <{layout_result_0 = #a}> : matrix_desc<32x64xf32> -> vector<16x32xf32>

533 struct WgToSgConvertLayoutOp
534  : public OpConversionPattern<xegpu::ConvertLayoutOp> {
535  using OpConversionPattern<xegpu::ConvertLayoutOp>::OpConversionPattern;
536  LogicalResult
537  matchAndRewrite(xegpu::ConvertLayoutOp op, OneToNOpAdaptor adaptor,
538  ConversionPatternRewriter &rewriter) const override {
539  xegpu::LayoutAttr input = op.getInputLayout();
540  xegpu::LayoutAttr target = op.getTargetLayout();
541 
542  if (!input || !target || !input.isWgLayout() || !target.isWgLayout())
543  return rewriter.notifyMatchFailure(
544  op, "Input and target layouts must have subgroup layout");
545 
546  DenseI32ArrayAttr inputSgLayout = input.getSgLayout();
547  DenseI32ArrayAttr inputSgData = input.getSgData();
548  DenseI32ArrayAttr inputOrder = input.getOrder();
549  DenseI32ArrayAttr targetSgLayout = target.getSgLayout();
550  DenseI32ArrayAttr targetSgData = target.getSgData();
551  DenseI32ArrayAttr targetOrder = target.getOrder();
552 
553  // TODO: currently we only support for optimal case, where input and
554  // output has the same sg_layout and sg_data, so SLM is not involved.
555  if (inputSgLayout != targetSgLayout || inputSgData != targetSgData ||
556  inputOrder != targetOrder)
557  return failure();
558 
559  input = input.dropSgLayoutAndData();
560  target = target.dropSgLayoutAndData();
561 
562  SmallVector<Value> newOps(adaptor.getSource());
563  if (input && target) {
564  // keep the ConvertLayoutOp for rest fields, e.g., inst_data.
565  for (auto [i, src] : llvm::enumerate(adaptor.getSource())) {
566  auto newOp = xegpu::ConvertLayoutOp::create(
567  rewriter, op.getLoc(), src.getType(), src, input, target);
568  newOps[i] = newOp;
569  }
570  }
571  rewriter.replaceOpWithMultiple(op, {newOps});
572  return success();
573  }

576 // Handles UnrealizedConversionCastOp generated during
577 // SCFStructuralTypeConversions (step 1). This op may appear as either a
578 // target or source materialization for Vector values, e.g.:
579 // 1. unrealized_cast %1 : vector<256xf32> to vector<16xf32>, ...
580 // 2. unrealized_cast %1 : vector<16xf32>, ... to vector<256xf32>
581 // it could be either 1:N or N:1 cast. In both cases, the pattern
582 // simply forwards the inputs to the outputs using 1:1 or 1:N interface.
583 // for example, the following scf::forOp
584 // ```
585 // %for = scf.for ... iter_args(%arg1 = %0)->(vector<128x128xf16>) {
586 // %n = use(%arg1): vector<128x128xf16>
587 // scf.yield %n : vector<128x128xf16>
588 // }
589 // ```
590 // Could be converted to:
591 // ```
592 // %1 = unrealized_conversion_cast %0
593 // : vector<128x128xf16> to vector<16x16xf16>, vector<16x16xf16>
594 // %for:2 = scf.for ... iter_args(%arg1 = %1#1, %arg2 = %1#2)
595 // -> (vector<16x16xf16>, vector<16x16xf16) {
596 // %m = unrealized_conversion_cast %arg1, %arg2
597 // : vector<16x16xf16>, vector<16x16xf16> to vector<128x128xf16>
598 // %n = use(%m): vector<128x128xf16>
599 // %b = unrealized_conversion_cast %n
600 // : vector<128x128xf16> to vector<16x16xf16>, vector<16x16xf16>
601 // scf.yield %b#1, %b#2 : vector<16x16xf16>, vector<16x16xf16>
602 // }
603 // %cast = unrealized_conversion_cast %for:2
604 // : vector<16x16xf16>, vector<16x16xf16> to vector<128x128xf16>
605 // ```
606 // TODO: remove it when context-aware type converter is ready.
607 struct UnrealizedConversionCastOpPattern
608  : public OpConversionPattern<mlir::UnrealizedConversionCastOp> {
609  using OpConversionPattern<
610  mlir::UnrealizedConversionCastOp>::OpConversionPattern;
611 
612  mlir::LogicalResult
613  matchAndRewrite(mlir::UnrealizedConversionCastOp op, OneToNOpAdaptor adaptor,
614  ConversionPatternRewriter &rewriter) const override {
615  SmallVector<Value> inputs = xegpu::flattenValues(adaptor.getInputs());
616 
617  auto inputTy = dyn_cast<VectorType>(inputs[0].getType());
618  auto outputTy = dyn_cast<VectorType>(op->getOpResult(0).getType());
619 
620  if (!inputTy || !outputTy || !llvm::all_equal(op->getResultTypes()) ||
621  !llvm::all_equal(ValueRange(inputs).getTypes()))
622  return failure();
623 
624  // Handles the case "cast %1 : vector<256xf32> to vector<16xf32>, ...".
625  // It is generated by source materialization (e.g., inits to scf forOp).
626  // The input values provided by the adaptor should already be distributed,
627  // and their types should correspond exactly to the result types of the
628  // operation.
629  if (op.getNumOperands() == 1 &&
630  llvm::equal(ValueRange(inputs).getTypes(), op->getResultTypes())) {
631  rewriter.replaceOp(op, inputs);
632  return success();
633  }
634 
635  // Handles the case "cast %1 : vector<16xf32>, ... to vector<256xf32>".
636  // It is generated by target materialization (e.g., arguments/results
637  // of scf forOp). All input values must have the same vector type, and
638  // their shape must be evenly divisible by the output vector's shape
639  // (determined by the nature of the workgroup to subgroup distribution).
640  // TODO: it is not safe to do such forward, since such N:1 cast could be
641  // from others.
642  if (op.getNumResults() == 1 &&
643  computeShapeRatio(outputTy.getShape(), inputTy.getShape())) {
644  rewriter.replaceOpWithMultiple(op, {inputs});
645  return success();
646  }
647 
648  return mlir::failure();
649  }
650 };

673 void XeGPUWgToSgDistributePass::runOnOperation() {
674  // Track existing UnrealizedConversionCastOps
675  SmallVector<Operation *> existingCastOps;
676  getOperation()->walk([&](UnrealizedConversionCastOp castOp) {
677  existingCastOps.push_back(castOp.getOperation());
678  });
679 
680  {
681  // Step 1: Apply SCFStructuralTypeConversions to SCF operations with
682  // VectorType operands. This first converts such operands to
683  // RankedTensorType, propagates the layout attribute into the encoding
684  // attribute, and finally converts the RankedTensorType to VectorType based
685  // on the encoding.
686 
687  TypeConverter converter;
688  converter.addConversion([&](Type type) -> Type { return type; });
689  converter.addConversion(
690  [&](RankedTensorType type,
691  SmallVectorImpl<Type> &result) -> std::optional<LogicalResult> {
692  Type elemTy = type.getElementType();
693  ArrayRef<int64_t> shape = type.getShape();
694 
695  int count;
696  SmallVector<int64_t> subShape;
697  std::tie(subShape, count) = getSgShapeAndCount(
698  shape,
699  dyn_cast_if_present<xegpu::LayoutAttr>(type.getEncoding()));
700 
701  auto newTy = VectorType::get(subShape, elemTy);
702  result.append(count, newTy);
703  return success();
704  });
705 
706  xegpu::doSCFStructuralTypeConversionWithTensorType(getOperation(),
707  converter);
708  }
709 
710  // Step 2: Perform workgroup to subgroup distribution for TensorDesc values,
711  // as well as XeGPU, Arith, and Vector operations.
712  MLIRContext *ctx = &getContext();
713  RewritePatternSet patterns(ctx);
714  ConversionTarget target(*ctx);
715  TypeConverter converter;
716  converter.addConversion([&](Type type) -> Type { return type; });
717  converter.addConversion(
718  [&](xegpu::TensorDescType type,
719  SmallVectorImpl<Type> &result) -> std::optional<LogicalResult> {
720  Type elemTy = type.getElementType();
721  ArrayRef<int64_t> shape = type.getShape();
722 
723  int count;
724  SmallVector<int64_t> subShape;
725  xegpu::LayoutAttr layout = type.getLayoutAttr();
726  std::tie(subShape, count) = getSgShapeAndCount(shape, layout);
727 
728  if (layout)
729  layout = layout.dropSgLayoutAndData();
730 
731  auto newTy = xegpu::TensorDescType::get(
732  type.getContext(), subShape, elemTy, type.getEncoding(), layout);
733  result.append(count, newTy);
734  return success();
735  });
736 
737  auto getTensorDescType = [](Operation *op) -> xegpu::TensorDescType {
738  if (auto createOp = dyn_cast<xegpu::CreateNdDescOp>(op))
739  return createOp.getType();
740  if (auto loadOp = dyn_cast<xegpu::LoadNdOp>(op))
741  return loadOp.getTensorDescType();
742  if (auto storeOp = dyn_cast<xegpu::StoreNdOp>(op))
743  return storeOp.getTensorDescType();
744  if (auto updateOp = dyn_cast<xegpu::UpdateNdOffsetOp>(op))
745  return updateOp.getType();
746  if (auto prefetchOp = dyn_cast<xegpu::PrefetchNdOp>(op))
747  return prefetchOp.getTensorDescType();
748  return xegpu::TensorDescType();
749  };
750 
751  auto isLegal = [&](xegpu::LayoutAttr layout) -> bool {
752  return !layout || !layout.isWgLayout();
753  };
754 
755  target.addDynamicallyLegalOp<xegpu::CreateNdDescOp, xegpu::LoadNdOp,
756  xegpu::StoreNdOp, xegpu::UpdateNdOffsetOp,
757  xegpu::PrefetchNdOp>([=](Operation *op) -> bool {
758  auto tdescTy = getTensorDescType(op);
759  auto layout = dyn_cast_if_present<xegpu::LayoutAttr>(tdescTy.getLayout());
760  return isLegal(layout);
761  });
762 
763  target.addDynamicallyLegalOp<xegpu::DpasOp>([=](xegpu::DpasOp op) -> bool {
764  auto layout = xegpu::getLayoutAttr(op.getResult());
765  return isLegal(layout);
766  });
767 
768  target.addDynamicallyLegalOp<vector::BroadcastOp>(
769  [=](vector::BroadcastOp op) -> bool {
770  return isLegal(xegpu::getLayoutAttr(op.getResult()));
771  });
772 
773  target.addDynamicallyLegalOp<xegpu::ConvertLayoutOp>(
774  [=](xegpu::ConvertLayoutOp op) -> bool {
775  return isLegal(op.getInputLayout()) && isLegal(op.getTargetLayout());
776  });
777 
778  target.addDynamicallyLegalDialect<math::MathDialect, arith::ArithDialect>(
779  [=](Operation *op) -> std::optional<bool> {
780  // Only handle elementwise mappable ops
781  if (!OpTrait::hasElementwiseMappableTraits(op))
782  return true;
783 
784  VectorType resultType =
785  dyn_cast<VectorType>(op->getResult(0).getType());
786  if (!resultType)
787  return true;
788 
789  // Check if all operands are vectors of the same shape
790  // TODO: Support other types.
791  for (Value operand : op->getOperands()) {
792  VectorType operandType = dyn_cast<VectorType>(operand.getType());
793  if (!operandType || operandType.getShape() != resultType.getShape()) {
794  return true;
795  }
796  }
797 
798  xegpu::LayoutAttr layout = xegpu::getLayoutAttr(op->getResult(0));
799  return isLegal(layout);
800  });
801 
802  target.addDynamicallyLegalOp<UnrealizedConversionCastOp>(
803  [=](UnrealizedConversionCastOp op) {
804  return llvm::is_contained(existingCastOps, op.getOperation());
805  });
806 
807  target.markUnknownOpDynamicallyLegal([](Operation *) { return true; });
808 
809  scf::populateSCFStructuralTypeConversionsAndLegality(converter, patterns,
810  target);
811  xegpu::populateXeGPUWgToSgDistributePatterns(patterns);
812  if (failed(
813  applyPartialConversion(getOperation(), target, std::move(patterns))))
814  return signalPassFailure();
815 
816  // Remove sg_layout and sg_data attributes from the Layout
817  // attribute for each VectorType result of the operation.
818  // For Structured Control Flow ops, the layout is simply removed,
819  // since in 1:N case, the layout for new results are missing.
820  // Layout propagation pass will activated.
821  getOperation()->walk([](Operation *op) {
822  for (OpResult result : op->getOpResults()) {
823  std::string name = xegpu::getLayoutName(result);
824  if (auto layout = op->getAttrOfType<xegpu::LayoutAttr>(name)) {
825  op->removeAttr(name);
826  if (!isa<scf::IfOp, scf::ForOp, scf::WhileOp, scf::ConditionOp>(op)) {
827  if (auto newLayout = layout.dropSgLayoutAndData())
828  op->setAttr(name, newLayout);
829  }
830  }
831  }
832  });
833 }
