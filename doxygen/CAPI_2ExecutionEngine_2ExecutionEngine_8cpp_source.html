22extern "C" MlirExecutionEngine
23mlirExecutionEngineCreate(MlirModule op, int optLevel, int numPaths,
24 const MlirStringRef *sharedLibPaths,
25 bool enableObjectDump, bool enablePIC) {
26 static bool initOnce = [] {
27 llvm::InitializeNativeTarget();
28 llvm::InitializeNativeTargetAsmParser(); // needed for inline_asm
29 llvm::InitializeNativeTargetAsmPrinter();
30 return true;
31 }();
32 (void)initOnce;
33
34 auto &ctx = *unwrap(op)->getContext();
35 mlir::registerBuiltinDialectTranslation(ctx);
36 mlir::registerLLVMDialectTranslation(ctx);
37 mlir::registerOpenMPDialectTranslation(ctx);
38
39 auto tmBuilderOrError = llvm::orc::JITTargetMachineBuilder::detectHost();
40 if (!tmBuilderOrError) {
41 llvm::errs() << "Failed to create a JITTargetMachineBuilder for the host "
42 "because: \n";
43 consumeError(tmBuilderOrError.takeError());
44 return MlirExecutionEngine{nullptr};
45 }
46 if (enablePIC)
47 tmBuilderOrError->setRelocationModel(llvm::Reloc::PIC_);
48 auto tmOrError = tmBuilderOrError->createTargetMachine();
49 if (!tmOrError) {
50 llvm::errs() << "Failed to create a TargetMachine for the host because: \n";
51 consumeError(tmOrError.takeError());
52 return MlirExecutionEngine{nullptr};
53 }
54
55 SmallVector<StringRef> libPaths;
56 for (unsigned i = 0; i < static_cast<unsigned>(numPaths); ++i)
57 libPaths.push_back(sharedLibPaths[i].data);
58
59 // Create a transformer to run all LLVM optimization passes at the
60 // specified optimization level.
61 auto transformer = mlir::makeOptimizingTransformer(
62 optLevel, /*sizeLevel=*/0, /*targetMachine=*/tmOrError->get());
63 ExecutionEngineOptions jitOptions;
64 jitOptions.transformer = transformer;
65 jitOptions.jitCodeGenOptLevel = static_cast<llvm::CodeGenOptLevel>(optLevel);
66 jitOptions.sharedLibPaths = libPaths;
67 jitOptions.enableObjectDump = enableObjectDump;
68 auto jitOrError = ExecutionEngine::create(unwrap(op), jitOptions,
69 std::move(tmOrError.get()));
70 if (!jitOrError) {
71 llvm::errs() << "Failed to create an ExecutionEngine because: \n";
72 consumeError(jitOrError.takeError());
73 return MlirExecutionEngine{nullptr};
74 }
75 return wrap(jitOrError->release());
76}
