21 static Value materializeTensorLoad(OpBuilder &builder, TensorType type,
22  ValueRange inputs, Location loc) {
23  assert(inputs.size() == 1);
24  assert(inputs[0].getType().isa<BaseMemRefType>());

28 /// Registers conversions into BufferizeTypeConverter
29 BufferizeTypeConverter::BufferizeTypeConverter() {
30  // Keep all types unchanged.
31  addConversion([](Type type) { return type; });
32  // Convert RankedTensorType to MemRefType.
33  addConversion([](RankedTensorType type) -> Type {
34  return MemRefType::get(type.getShape(), type.getElementType());
35  });
36  // Convert UnrankedTensorType to UnrankedMemRefType.
37  addConversion([](UnrankedTensorType type) -> Type {
38  return UnrankedMemRefType::get(type.getElementType(), 0);
39  });
40  addArgumentMaterialization(materializeTensorLoad);
41  addSourceMaterialization(materializeTensorLoad);
42  addTargetMaterialization([](OpBuilder &builder, BaseMemRefType type,
43  ValueRange inputs, Location loc) -> Value {
44  assert(inputs.size() == 1);
45  assert(inputs[0].getType().isa<TensorType>());
46  return builder.create<memref::BufferCastOp>(loc, type, inputs[0]);
47  });
48 }

93 struct FinalizingBufferizePass
94  : public FinalizingBufferizeBase<FinalizingBufferizePass> {
95  using FinalizingBufferizeBase<
96  FinalizingBufferizePass>::FinalizingBufferizeBase;
97 
98  void runOnFunction() override {
99  auto func = getFunction();
100  auto *context = &getContext();
101 
102  BufferizeTypeConverter typeConverter;
103  RewritePatternSet patterns(context);
104  ConversionTarget target(*context);
105 
106  populateEliminateBufferizeMaterializationsPatterns(typeConverter, patterns);
107 
108  // If all result types are legal, and all block arguments are legal (ensured
109  // by func conversion above), then all types in the program are legal.
110  //
111  // We also check that the operand types are legal to avoid creating invalid
112  // IR. For example, this prevents
113  // populateEliminateBufferizeMaterializationsPatterns from updating the
114  // types of the operands to a return op without updating the enclosing
115  // function.
116  target.markUnknownOpDynamicallyLegal(
117  [&](Operation *op) { return typeConverter.isLegal(op); });
118 
119  if (failed(applyFullConversion(func, target, std::move(patterns))))
120  signalPassFailure();
121  }
122 };
