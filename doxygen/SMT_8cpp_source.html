69 const MlirType *domainTypes,
70 MlirType rangeType) {
71 SmallVector<Type> domainTypesVec;
72 domainTypesVec.reserve(numberOfDomainTypes);

75 domainTypesVec.push_back(unwrap(domainTypes[i]));
