9// This pass strips OpenACC constructs that are invalid or unnecessary inside
10// device code (specialized acc routines or compute construct regions).
11//
12// Overview:
13// ---------
14// In a specialized acc routine or compute construct, many OpenACC operations
15// do not make sense because they are host-side constructs. This pass removes
16// or transforms these operations appropriately:
17//
18// - Data operations that manage device memory from host perspective
19// - Compute constructs that launch kernels (we're already on device)
20// - Runtime operations like init/shutdown/set/wait
21//
22// Transformations:
23// ----------------
24// The pass applies the following transformations:
25//
26// 1. Data Entry Ops (replaced with var operand):
27// acc.attach, acc.copyin, acc.create, acc.declare_device_resident,
28// acc.declare_link, acc.deviceptr, acc.get_deviceptr, acc.nocreate,
29// acc.present, acc.update_device, acc.use_device
30//
31// 2. Data Exit Ops (erased):
32// acc.copyout, acc.delete, acc.detach, acc.update_host
33//
34// 3. Structured Data/Compute Constructs (region inlined):
35// acc.data, acc.host_data, acc.kernel_environment, acc.parallel,
36// acc.serial, acc.kernels
37//
38// 4. Unstructured Data Ops (erased):
39// acc.enter_data, acc.exit_data, acc.update, acc.declare_enter,
40// acc.declare_exit
41//
42// 5. Runtime Ops (erased):
43// acc.init, acc.shutdown, acc.set, acc.wait
44//
45// Scope of Application:
46// ---------------------
47// - For functions with `acc.specialized_routine` attribute: patterns are
48// applied to the entire function body.
49// - For non-specialized functions: patterns are applied only to ACC
50// operations INSIDE compute constructs (parallel, serial, kernels),
51// not to the compute constructs themselves or their data operands.
52//
53// Note: acc.cache, acc.private, acc.reduction, acc.firstprivate are NOT
54// transformed by this pass as they are valid in device code.
55//
56//===----------------------------------------------------------------------===//

78class ACCSpecializeForDevice
79 : public acc::impl::ACCSpecializeForDeviceBase<ACCSpecializeForDevice> {
80public:
81 using ACCSpecializeForDeviceBase<
82 ACCSpecializeForDevice>::ACCSpecializeForDeviceBase;
83
84 void runOnOperation() override {
85 func::FuncOp func = getOperation();
86
87 RewritePatternSet patterns(&getContext());
88 acc::populateACCSpecializeForDevicePatterns(patterns);
89 GreedyRewriteConfig config;
90 config.setUseTopDownTraversal(true);
91
92 if (acc::isSpecializedAccRoutine(func)) {
93 // For specialized acc routines, apply patterns to the entire function
94 (void)applyPatternsGreedily(func, std::move(patterns), config);
95 } else {
96 // For non-specialized functions, apply patterns only to ACC operations
97 // inside compute constructs (not to the compute constructs themselves).
98 // Use ExistingOps strictness so the greedy driver does not expand the
99 // worklist to parent ops, which would accidentally unwrap the compute
100 // construct (e.g. after inlining acc routines with their own data
101 // regions).
102 config.setStrictness(GreedyRewriteStrictness::ExistingOps);
103 SmallVector<Operation *> opsToTransform;
104 func.walk([&](Operation *op) {
105 if (isa<ACC_COMPUTE_CONSTRUCT_OPS>(op)) {
106 // Walk inside the compute construct and collect ACC ops
107 op->walk([&](Operation *innerOp) {
108 // Skip the compute construct itself
109 if (innerOp == op)
110 return;
111 if (isa<acc::OpenACCDialect>(innerOp->getDialect()))
112 opsToTransform.push_back(innerOp);
113 });
114 }
115 });
116 if (!opsToTransform.empty())
117 (void)applyOpPatternsGreedily(opsToTransform, std::move(patterns),
118 config);
119 }
120 }
121};
122
123} // namespace

129void mlir::acc::populateACCSpecializeForDevicePatterns(
130 RewritePatternSet &patterns) {
131 MLIRContext *context = patterns.getContext();
132
133 // Declare patterns - erase declare_enter and its associated declare_exit
134 patterns.insert<ACCDeclareEnterOpConversion>(context);
135
136 // Data entry ops - replaced with their var operand
137 // Note: acc.cache, acc.private, acc.reduction, acc.firstprivate are NOT
138 // included here - they are valid in device code
139 patterns.insert<ACCOpReplaceWithVarConversion<acc::AttachOp>,
140 ACCOpReplaceWithVarConversion<acc::CopyinOp>,
141 ACCOpReplaceWithVarConversion<acc::CreateOp>,
142 ACCOpReplaceWithVarConversion<acc::DeclareDeviceResidentOp>,
143 ACCOpReplaceWithVarConversion<acc::DeclareLinkOp>,
144 ACCOpReplaceWithVarConversion<acc::DevicePtrOp>,
145 ACCOpReplaceWithVarConversion<acc::GetDevicePtrOp>,
146 ACCOpReplaceWithVarConversion<acc::NoCreateOp>,
147 ACCOpReplaceWithVarConversion<acc::PresentOp>,
148 ACCOpReplaceWithVarConversion<acc::UpdateDeviceOp>,
149 ACCOpReplaceWithVarConversion<acc::UseDeviceOp>>(context);
150
151 // Data exit ops - simply erased (no results)
152 patterns.insert<ACCOpEraseConversion<acc::CopyoutOp>,
153 ACCOpEraseConversion<acc::DeleteOp>,
154 ACCOpEraseConversion<acc::DetachOp>,
155 ACCOpEraseConversion<acc::UpdateHostOp>>(context);
156
157 // Structured data constructs - unwrap their regions
158 patterns.insert<ACCRegionUnwrapConversion<acc::DataOp>,
159 ACCRegionUnwrapConversion<acc::HostDataOp>,
160 ACCRegionUnwrapConversion<acc::KernelEnvironmentOp>>(context);
161
162 // Compute constructs - unwrap their regions
163 patterns.insert<ACCRegionUnwrapConversion<acc::ParallelOp>,
164 ACCRegionUnwrapConversion<acc::SerialOp>,
165 ACCRegionUnwrapConversion<acc::KernelsOp>>(context);
166
167 // Unstructured data operations - erase them
168 patterns.insert<ACCOpEraseConversion<acc::EnterDataOp>,
169 ACCOpEraseConversion<acc::ExitDataOp>,
170 ACCOpEraseConversion<acc::UpdateOp>>(context);
171
172 // Runtime operations - erase them
173 patterns.insert<
174 ACCOpEraseConversion<acc::InitOp>, ACCOpEraseConversion<acc::ShutdownOp>,
175 ACCOpEraseConversion<acc::SetOp>, ACCOpEraseConversion<acc::WaitOp>>(
176 context);
177}
