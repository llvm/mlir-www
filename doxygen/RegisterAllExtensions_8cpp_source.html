67/// This function may be called to register all MLIR dialect extensions with the
68/// provided registry.
69/// If you're building a compiler, you generally shouldn't use this: you would
70/// individually register the specific extensions that are useful for the
71/// pipelines and transformations you are using.
72void mlir::registerAllExtensions(DialectRegistry &registry) {
73 // Register all conversions to LLVM extensions.
74 registerConvertArithToEmitCInterface(registry);
75 arith::registerConvertArithToLLVMInterface(registry);
76 registerConvertComplexToLLVMInterface(registry);
77 cf::registerConvertControlFlowToLLVMInterface(registry);
78 func::registerAllExtensions(registry);
79 tensor::registerAllExtensions(registry);
80 registerConvertFuncToEmitCInterface(registry);
81 registerConvertFuncToLLVMInterface(registry);
82 index::registerConvertIndexToLLVMInterface(registry);
83 registerConvertMathToLLVMInterface(registry);
84 mpi::registerConvertMPIToLLVMInterface(registry);
85 registerConvertMemRefToEmitCInterface(registry);
86 registerConvertMemRefToLLVMInterface(registry);
87 registerConvertNVVMToLLVMInterface(registry);
88 ptr::registerConvertPtrToLLVMInterface(registry);
89 registerConvertOpenMPToLLVMInterface(registry);
90 registerConvertSCFToEmitCInterface(registry);
91 ub::registerConvertUBToLLVMInterface(registry);
92 registerConvertAMXToLLVMInterface(registry);
93 gpu::registerConvertGpuToLLVMInterface(registry);
94 NVVM::registerConvertGpuToNVVMInterface(registry);
95 vector::registerConvertVectorToLLVMInterface(registry);
96 registerConvertXeVMToLLVMInterface(registry);
97
98 // Register all transform dialect extensions.
99 affine::registerTransformDialectExtension(registry);
100 bufferization::registerTransformDialectExtension(registry);
101 dlti::registerTransformDialectExtension(registry);
102 func::registerTransformDialectExtension(registry);
103 gpu::registerTransformDialectExtension(registry);
104 linalg::registerTransformDialectExtension(registry);
105 memref::registerTransformDialectExtension(registry);
106 nvgpu::registerTransformDialectExtension(registry);
107 scf::registerTransformDialectExtension(registry);
108 sparse_tensor::registerTransformDialectExtension(registry);
109 tensor::registerTransformDialectExtension(registry);
110 transform::registerDebugExtension(registry);
111 transform::registerIRDLExtension(registry);
112 transform::registerLoopExtension(registry);
113 transform::registerPDLExtension(registry);
114 transform::registerSMTExtension(registry);
115 transform::registerTuneExtension(registry);
116 vector::registerTransformDialectExtension(registry);

118 xegpu::registerTransformDialectExtension(registry);
119 arm_neon::registerTransformDialectExtension(registry);
120 arm_sve::registerTransformDialectExtension(registry);
121
122 // Translation extensions need to be registered by calling
123 // `registerAllToLLVMIRTranslations` (see All.h).
124}
