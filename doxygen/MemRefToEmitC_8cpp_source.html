28 static bool isMemRefTypeLegalForEmitC(MemRefType memRefType) {
29  return memRefType.hasStaticShape() && memRefType.getLayout().isIdentity() &&
30  memRefType.getRank() != 0 &&
31  !llvm::is_contained(memRefType.getShape(), 0);
32 }

61 struct ConvertAlloca final : public OpConversionPattern<memref::AllocaOp> {
62  using OpConversionPattern::OpConversionPattern;
63 
64  LogicalResult
65  matchAndRewrite(memref::AllocaOp op, OpAdaptor operands,
66  ConversionPatternRewriter &rewriter) const override {
67 
68  if (!op.getType().hasStaticShape()) {
69  return rewriter.notifyMatchFailure(
70  op.getLoc(), "cannot transform alloca with dynamic shape");
71  }
72 
73  if (op.getAlignment().value_or(1) > 1) {
74  // TODO: Allow alignment if it is not more than the natural alignment
75  // of the C array.
76  return rewriter.notifyMatchFailure(
77  op.getLoc(), "cannot transform alloca with alignment requirement");
78  }
79 
80  auto resultTy = getTypeConverter()->convertType(op.getType());
81  if (!resultTy) {
82  return rewriter.notifyMatchFailure(op.getLoc(), "cannot convert type");
83  }
84  auto noInit = emitc::OpaqueAttr::get(getContext(), "");
85  rewriter.replaceOpWithNewOp<emitc::VariableOp>(op, resultTy, noInit);
86  return success();
87  }
88 };

90 Type convertMemRefType(MemRefType opTy, const TypeConverter *typeConverter) {
91  Type resultTy;
92  if (opTy.getRank() == 0) {
93  resultTy = typeConverter->convertType(mlir::getElementTypeOrSelf(opTy));
94  } else {
95  resultTy = typeConverter->convertType(opTy);
96  }
97  return resultTy;
98 }

100 struct ConvertAlloc final : public OpConversionPattern<memref::AllocOp> {
101  using OpConversionPattern::OpConversionPattern;
102  LogicalResult
103  matchAndRewrite(memref::AllocOp allocOp, OpAdaptor operands,
104  ConversionPatternRewriter &rewriter) const override {
105  Location loc = allocOp.getLoc();
106  MemRefType memrefType = allocOp.getType();
107  if (!isMemRefTypeLegalForEmitC(memrefType)) {
108  return rewriter.notifyMatchFailure(
109  loc, "incompatible memref type for EmitC conversion");
110  }
111 
112  Type sizeTType = emitc::SizeTType::get(rewriter.getContext());
113  Type elementType = memrefType.getElementType();
114  IndexType indexType = rewriter.getIndexType();
115  emitc::CallOpaqueOp sizeofElementOp = rewriter.create<emitc::CallOpaqueOp>(
116  loc, sizeTType, rewriter.getStringAttr("sizeof"), ValueRange{},
117  ArrayAttr::get(rewriter.getContext(), {TypeAttr::get(elementType)}));
118 
119  int64_t numElements = 1;
120  for (int64_t dimSize : memrefType.getShape()) {
121  numElements *= dimSize;
122  }
123  Value numElementsValue = rewriter.create<emitc::ConstantOp>(
124  loc, indexType, rewriter.getIndexAttr(numElements));
125 
126  Value totalSizeBytes = rewriter.create<emitc::MulOp>(
127  loc, sizeTType, sizeofElementOp.getResult(0), numElementsValue);
128 
129  emitc::CallOpaqueOp allocCall;
130  StringAttr allocFunctionName;
131  Value alignmentValue;
132  SmallVector<Value, 2> argsVec;
133  if (allocOp.getAlignment()) {
134  allocFunctionName = rewriter.getStringAttr(alignedAllocFunctionName);
135  alignmentValue = rewriter.create<emitc::ConstantOp>(
136  loc, sizeTType,
137  rewriter.getIntegerAttr(indexType,
138  allocOp.getAlignment().value_or(0)));
139  argsVec.push_back(alignmentValue);
140  } else {
141  allocFunctionName = rewriter.getStringAttr(mallocFunctionName);
142  }
143 
144  argsVec.push_back(totalSizeBytes);
145  ValueRange args(argsVec);
146 
147  allocCall = rewriter.create<emitc::CallOpaqueOp>(
148  loc,
149  emitc::PointerType::get(
150  emitc::OpaqueType::get(rewriter.getContext(), "void")),
151  allocFunctionName, args);
152 
153  emitc::PointerType targetPointerType = emitc::PointerType::get(elementType);
154  emitc::CastOp castOp = rewriter.create<emitc::CastOp>(
155  loc, targetPointerType, allocCall.getResult(0));
156 
157  rewriter.replaceOp(allocOp, castOp);
158  return success();
159  }
160 };

162 struct ConvertGlobal final : public OpConversionPattern<memref::GlobalOp> {
163  using OpConversionPattern::OpConversionPattern;
164 
165  LogicalResult
166  matchAndRewrite(memref::GlobalOp op, OpAdaptor operands,
167  ConversionPatternRewriter &rewriter) const override {
168  MemRefType opTy = op.getType();
169  if (!op.getType().hasStaticShape()) {
170  return rewriter.notifyMatchFailure(
171  op.getLoc(), "cannot transform global with dynamic shape");
172  }
173 
174  if (op.getAlignment().value_or(1) > 1) {
175  // TODO: Extend GlobalOp to specify alignment via the `alignas` specifier.
176  return rewriter.notifyMatchFailure(
177  op.getLoc(), "global variable with alignment requirement is "
178  "currently not supported");
179  }
180 
181  Type resultTy = convertMemRefType(opTy, getTypeConverter());
182 
183  if (!resultTy) {
184  return rewriter.notifyMatchFailure(op.getLoc(),
185  "cannot convert result type");
186  }
187 
188  SymbolTable::Visibility visibility = SymbolTable::getSymbolVisibility(op);
189  if (visibility != SymbolTable::Visibility::Public &&
190  visibility != SymbolTable::Visibility::Private) {
191  return rewriter.notifyMatchFailure(
192  op.getLoc(),
193  "only public and private visibility is currently supported");
194  }
195  // We are explicit in specifing the linkage because the default linkage
196  // for constants is different in C and C++.
197  bool staticSpecifier = visibility == SymbolTable::Visibility::Private;
198  bool externSpecifier = !staticSpecifier;
199 
200  Attribute initialValue = operands.getInitialValueAttr();
201  if (opTy.getRank() == 0) {
202  auto elementsAttr = llvm::cast<ElementsAttr>(*op.getInitialValue());
203  initialValue = elementsAttr.getSplatValue<Attribute>();
204  }
205  if (isa_and_present<UnitAttr>(initialValue))
206  initialValue = {};
207 
208  rewriter.replaceOpWithNewOp<emitc::GlobalOp>(
209  op, operands.getSymName(), resultTy, initialValue, externSpecifier,
210  staticSpecifier, operands.getConstant());
211  return success();
212  }
213 };

236  rewriter.replaceOpWithNewOp<emitc::ApplyOp>(

238  return success();
239  }
240  rewriter.replaceOpWithNewOp<emitc::GetGlobalOp>(op, resultTy,
241  operands.getNameAttr());
242  return success();
243  }

293 void mlir::populateMemRefToEmitCTypeConversion(TypeConverter &typeConverter) {
294  typeConverter.addConversion(
295  [&](MemRefType memRefType) -> std::optional<Type> {
296  if (!isMemRefTypeLegalForEmitC(memRefType)) {
297  return {};
298  }
299  Type convertedElementType =
300  typeConverter.convertType(memRefType.getElementType());
301  if (!convertedElementType)
302  return {};
303  return emitc::ArrayType::get(memRefType.getShape(),
304  convertedElementType);
305  });
306 
307  auto materializeAsUnrealizedCast = [](OpBuilder &builder, Type resultType,
308  ValueRange inputs,
309  Location loc) -> Value {
310  if (inputs.size() != 1)
311  return Value();
312 
313  return UnrealizedConversionCastOp::create(builder, loc, resultType, inputs)
314  .getResult(0);
315  };
316 
317  typeConverter.addSourceMaterialization(materializeAsUnrealizedCast);
318  typeConverter.addTargetMaterialization(materializeAsUnrealizedCast);
319 }
