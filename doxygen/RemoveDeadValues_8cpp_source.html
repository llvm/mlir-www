9 // The goal of this pass is optimization (reducing runtime) by removing
10 // unnecessary instructions. Unlike other passes that rely on local information
11 // gathered from patterns to accomplish optimization, this pass uses a full
12 // analysis of the IR, specifically, liveness analysis, and is thus more
13 // powerful.
14 //
15 // Currently, this pass performs the following optimizations:
16 // (A) Removes function arguments that are not live,
17 // (B) Removes function return values that are not live across all callers of
18 // the function,
19 // (C) Removes unneccesary operands, results, region arguments, and region
20 // terminator operands of region branch ops, and,
21 // (D) Removes simple and region branch ops that have all non-live results and
22 // don't affect memory in any way,
23 //
24 // iff
25 //
26 // the IR doesn't have any non-function symbol ops, non-call symbol user ops and
27 // branch ops.
28 //
29 // Here, a "simple op" refers to an op that isn't a symbol op, symbol-user op,
30 // region branch op, branch op, region branch terminator op, or return-like.
31 //
32 //===----------------------------------------------------------------------===//

116 /// Return true iff at least one value in `values` is live, given the liveness
117 /// information in `la`.
118 static bool hasLive(ValueRange values, const DenseSet<Value> &nonLiveSet,
119  RunLivenessAnalysis &la) {
120  for (Value value : values) {
121  if (nonLiveSet.contains(value)) {
122  LDBG() << "Value " << value << " is already marked non-live (dead)";
123  continue;
124  }
125 
126  const Liveness *liveness = la.getLiveness(value);
127  if (!liveness) {
128  LDBG() << "Value " << value
129  << " has no liveness info, conservatively considered live";
130  return true;
131  }
132  if (liveness->isLive) {
133  LDBG() << "Value " << value << " is live according to liveness analysis";
134  return true;
135  } else {
136  LDBG() << "Value " << value << " is dead according to liveness analysis";
137  }
138  }
139  return false;
140 }

142 /// Return a BitVector of size `values.size()` where its i-th bit is 1 iff the
143 /// i-th value in `values` is live, given the liveness information in `la`.
144 static BitVector markLives(ValueRange values, const DenseSet<Value> &nonLiveSet,
145  RunLivenessAnalysis &la) {
146  BitVector lives(values.size(), true);
147 
148  for (auto [index, value] : llvm::enumerate(values)) {
149  if (nonLiveSet.contains(value)) {
150  lives.reset(index);
151  LDBG() << "Value " << value
152  << " is already marked non-live (dead) at index " << index;
153  continue;
154  }
155 
156  const Liveness *liveness = la.getLiveness(value);
157  // It is important to note that when `liveness` is null, we can't tell if
158  // `value` is live or not. So, the safe option is to consider it live. Also,
159  // the execution of this pass might create new SSA values when erasing some
160  // of the results of an op and we know that these new values are live
161  // (because they weren't erased) and also their liveness is null because
162  // liveness analysis ran before their creation.
163  if (!liveness) {
164  LDBG() << "Value " << value << " at index " << index
165  << " has no liveness info, conservatively considered live";
166  continue;
167  }
168  if (!liveness->isLive) {
169  lives.reset(index);
170  LDBG() << "Value " << value << " at index " << index
171  << " is dead according to liveness analysis";
172  } else {
173  LDBG() << "Value " << value << " at index " << index
174  << " is live according to liveness analysis";
175  }
176  }
177 
178  return lives;
179 }

181 /// Collects values marked as "non-live" in the provided range and inserts them
182 /// into the nonLiveSet. A value is considered "non-live" if the corresponding
183 /// index in the `nonLive` bit vector is set.
184 static void collectNonLiveValues(DenseSet<Value> &nonLiveSet, ValueRange range,
185  const BitVector &nonLive) {
186  for (auto [index, result] : llvm::enumerate(range)) {
187  if (!nonLive[index])
188  continue;
189  nonLiveSet.insert(result);
190  LDBG() << "Marking value " << result << " as non-live (dead) at index "
191  << index;
192  }
193 }

195 /// Drop the uses of the i-th result of `op` and then erase it iff toErase[i]
196 /// is 1.
197 static void dropUsesAndEraseResults(Operation *op, BitVector toErase) {
198  assert(op->getNumResults() == toErase.size() &&
199  "expected the number of results in `op` and the size of `toErase` to "
200  "be the same");
201 
202  std::vector<Type> newResultTypes;
203  for (OpResult result : op->getResults())
204  if (!toErase[result.getResultNumber()])
205  newResultTypes.push_back(result.getType());
206  OpBuilder builder(op);
207  builder.setInsertionPointAfter(op);
208  OperationState state(op->getLoc(), op->getName().getStringRef(),
209  op->getOperands(), newResultTypes, op->getAttrs());
210  for (unsigned i = 0, e = op->getNumRegions(); i < e; ++i)
211  state.addRegion();
212  Operation *newOp = builder.create(state);
213  for (const auto &[index, region] : llvm::enumerate(op->getRegions())) {
214  Region &newRegion = newOp->getRegion(index);
215  // Move all blocks of `region` into `newRegion`.
216  Block *temp = new Block();
217  newRegion.push_back(temp);
218  while (!region.empty())
219  region.front().moveBefore(temp);
220  temp->erase();
221  }
222 
223  unsigned indexOfNextNewCallOpResultToReplace = 0;
224  for (auto [index, result] : llvm::enumerate(op->getResults())) {
225  assert(result && "expected result to be non-null");
226  if (toErase[index]) {
227  result.dropAllUses();
228  } else {
229  result.replaceAllUsesWith(
230  newOp->getResult(indexOfNextNewCallOpResultToReplace++));
231  }
232  }
233  op->erase();
234 }

236 /// Convert a list of `Operand`s to a list of `OpOperand`s.
237 static SmallVector<OpOperand *> operandsToOpOperands(OperandRange operands) {
238  OpOperand *values = operands.getBase();
239  SmallVector<OpOperand *> opOperands;
240  for (unsigned i = 0, e = operands.size(); i < e; i++)
241  opOperands.push_back(&values[i]);
242  return opOperands;
243 }

245 /// Process a simple operation `op` using the liveness analysis `la`.
246 /// If the operation has no memory effects and none of its results are live:
247 /// 1. Add the operation to a list for future removal, and
248 /// 2. Mark all its results as non-live values
249 ///
250 /// The operation `op` is assumed to be simple. A simple operation is one that
251 /// is NOT:
252 /// - Function-like
253 /// - Call-like
254 /// - A region branch operation
255 /// - A branch operation
256 /// - A region branch terminator
257 /// - Return-like
258 static void processSimpleOp(Operation *op, RunLivenessAnalysis &la,
259  DenseSet<Value> &nonLiveSet,
260  RDVFinalCleanupList &cl) {
261  LDBG() << "Processing simple op: " << *op;
262  if (!isMemoryEffectFree(op) || hasLive(op->getResults(), nonLiveSet, la)) {
263  LDBG()
264  << "Simple op is not memory effect free or has live results, skipping: "
265  << *op;
266  return;
267  }
268 
269  LDBG()
270  << "Simple op has all dead results and is memory effect free, scheduling "
271  "for removal: "
272  << *op;
273  cl.operations.push_back(op);
274  collectNonLiveValues(nonLiveSet, op->getResults(),
275  BitVector(op->getNumResults(), true));
276 }

278 /// Process a function-like operation `funcOp` using the liveness analysis `la`
279 /// and the IR in `module`. If it is not public or external:
280 /// (1) Adding its non-live arguments to a list for future removal.
281 /// (2) Marking their corresponding operands in its callers for removal.
282 /// (3) Identifying and enqueueing unnecessary terminator operands
283 /// (return values that are non-live across all callers) for removal.
284 /// (4) Enqueueing the non-live arguments and return values for removal.
285 /// (5) Collecting the uses of these return values in its callers for future
286 /// removal.
287 /// (6) Marking all its results as non-live values.
288 static void processFuncOp(FunctionOpInterface funcOp, Operation *module,
289  RunLivenessAnalysis &la, DenseSet<Value> &nonLiveSet,
290  RDVFinalCleanupList &cl) {
291  LDBG() << "Processing function op: " << funcOp.getOperation()->getName();
292  if (funcOp.isPublic() || funcOp.isExternal()) {
293  LDBG() << "Function is public or external, skipping: "
294  << funcOp.getOperation()->getName();
295  return;
296  }
297 
298  // Get the list of unnecessary (non-live) arguments in `nonLiveArgs`.
299  SmallVector<Value> arguments(funcOp.getArguments());
300  BitVector nonLiveArgs = markLives(arguments, nonLiveSet, la);
301  nonLiveArgs = nonLiveArgs.flip();
302 
303  // Do (1).
304  for (auto [index, arg] : llvm::enumerate(arguments))
305  if (arg && nonLiveArgs[index]) {
306  cl.values.push_back(arg);
307  nonLiveSet.insert(arg);
308  }
309 
310  // Do (2).
311  SymbolTable::UseRange uses = *funcOp.getSymbolUses(module);
312  for (SymbolTable::SymbolUse use : uses) {
313  Operation *callOp = use.getUser();
314  assert(isa<CallOpInterface>(callOp) && "expected a call-like user");
315  // The number of operands in the call op may not match the number of
316  // arguments in the func op.
317  BitVector nonLiveCallOperands(callOp->getNumOperands(), false);
318  SmallVector<OpOperand *> callOpOperands =
319  operandsToOpOperands(cast<CallOpInterface>(callOp).getArgOperands());
320  for (int index : nonLiveArgs.set_bits())
321  nonLiveCallOperands.set(callOpOperands[index]->getOperandNumber());
322  cl.operands.push_back({callOp, nonLiveCallOperands});
323  }
324 
325  // Do (3).
326  // Get the list of unnecessary terminator operands (return values that are
327  // non-live across all callers) in `nonLiveRets`. There is a very important
328  // subtlety here. Unnecessary terminator operands are NOT the operands of the
329  // terminator that are non-live. Instead, these are the return values of the
330  // callers such that a given return value is non-live across all callers. Such
331  // corresponding operands in the terminator could be live. An example to
332  // demonstrate this:
333  // func.func private @f(%arg0: memref<i32>) -> (i32, i32) {
334  // %c0_i32 = arith.constant 0 : i32
335  // %0 = arith.addi %c0_i32, %c0_i32 : i32
336  // memref.store %0, %arg0[] : memref<i32>
337  // return %c0_i32, %0 : i32, i32
338  // }
339  // func.func @main(%arg0: i32, %arg1: memref<i32>) -> (i32) {
340  // %1:2 = call @f(%arg1) : (memref<i32>) -> i32
341  // return %1#0 : i32
342  // }
343  // Here, we can see that %1#1 is never used. It is non-live. Thus, @f doesn't
344  // need to return %0. But, %0 is live. And, still, we want to stop it from
345  // being returned, in order to optimize our IR. So, this demonstrates how we
346  // can make our optimization strong by even removing a live return value (%0),
347  // since it forwards only to non-live value(s) (%1#1).
348  Operation *lastReturnOp = funcOp.back().getTerminator();
349  size_t numReturns = lastReturnOp->getNumOperands();
350  BitVector nonLiveRets(numReturns, true);
351  for (SymbolTable::SymbolUse use : uses) {
352  Operation *callOp = use.getUser();
353  assert(isa<CallOpInterface>(callOp) && "expected a call-like user");
354  BitVector liveCallRets = markLives(callOp->getResults(), nonLiveSet, la);
355  nonLiveRets &= liveCallRets.flip();
356  }
357 
358  // Note that in the absence of control flow ops forcing the control to go from
359  // the entry (first) block to the other blocks, the control never reaches any
360  // block other than the entry block, because every block has a terminator.
361  for (Block &block : funcOp.getBlocks()) {
362  Operation *returnOp = block.getTerminator();
363  if (returnOp && returnOp->getNumOperands() == numReturns)
364  cl.operands.push_back({returnOp, nonLiveRets});
365  }
366 
367  // Do (4).
368  cl.functions.push_back({funcOp, nonLiveArgs, nonLiveRets});
369 
370  // Do (5) and (6).
371  if (numReturns == 0)
372  return;
373  for (SymbolTable::SymbolUse use : uses) {
374  Operation *callOp = use.getUser();
375  assert(isa<CallOpInterface>(callOp) && "expected a call-like user");
376  cl.results.push_back({callOp, nonLiveRets});
377  collectNonLiveValues(nonLiveSet, callOp->getResults(), nonLiveRets);
378  }
379 }

381 /// Process a region branch operation `regionBranchOp` using the liveness
382 /// information in `la`. The processing involves two scenarios:
383 ///
384 /// Scenario 1: If the operation has no memory effects and none of its results
385 /// are live:
386 /// (1') Enqueue all its uses for deletion.
387 /// (2') Enqueue the branch itself for deletion.
388 ///
389 /// Scenario 2: Otherwise:
390 /// (1) Collect its unnecessary operands (operands forwarded to unnecessary
391 /// results or arguments).
392 /// (2) Process each of its regions.
393 /// (3) Collect the uses of its unnecessary results (results forwarded from
394 /// unnecessary operands
395 /// or terminator operands).
396 /// (4) Add these results to the deletion list.
397 ///
398 /// Processing a region includes:
399 /// (a) Collecting the uses of its unnecessary arguments (arguments forwarded
400 /// from unnecessary operands
401 /// or terminator operands).
402 /// (b) Collecting these unnecessary arguments.
403 /// (c) Collecting its unnecessary terminator operands (terminator operands
404 /// forwarded to unnecessary results
405 /// or arguments).
406 ///
407 /// Value Flow Note: In this operation, values flow as follows:
408 /// - From operands and terminator operands (successor operands)
409 /// - To arguments and results (successor inputs).
410 static void processRegionBranchOp(RegionBranchOpInterface regionBranchOp,
411  RunLivenessAnalysis &la,
412  DenseSet<Value> &nonLiveSet,
413  RDVFinalCleanupList &cl) {
414  LDBG() << "Processing region branch op: "
415  << OpWithFlags(regionBranchOp, OpPrintingFlags().skipRegions());
416  // Mark live results of `regionBranchOp` in `liveResults`.
417  auto markLiveResults = [&](BitVector &liveResults) {
418  liveResults = markLives(regionBranchOp->getResults(), nonLiveSet, la);
419  };
420 
421  // Mark live arguments in the regions of `regionBranchOp` in `liveArgs`.
422  auto markLiveArgs = [&](DenseMap<Region *, BitVector> &liveArgs) {
423  for (Region &region : regionBranchOp->getRegions()) {
424  if (region.empty())
425  continue;
426  SmallVector<Value> arguments(region.front().getArguments());
427  BitVector regionLiveArgs = markLives(arguments, nonLiveSet, la);
428  liveArgs[&region] = regionLiveArgs;
429  }
430  };
431 
432  // Return the successors of `region` if the latter is not null. Else return
433  // the successors of `regionBranchOp`.
434  auto getSuccessors = [&](Region *region = nullptr) {
435  auto point = region ? region : RegionBranchPoint::parent();
436  SmallVector<RegionSuccessor> successors;
437  regionBranchOp.getSuccessorRegions(point, successors);
438  return successors;
439  };
440 
441  // Return the operands of `terminator` that are forwarded to `successor` if
442  // the former is not null. Else return the operands of `regionBranchOp`
443  // forwarded to `successor`.
444  auto getForwardedOpOperands = [&](const RegionSuccessor &successor,
445  Operation *terminator = nullptr) {
446  OperandRange operands =
447  terminator ? cast<RegionBranchTerminatorOpInterface>(terminator)
448  .getSuccessorOperands(successor)
449  : regionBranchOp.getEntrySuccessorOperands(successor);
450  SmallVector<OpOperand *> opOperands = operandsToOpOperands(operands);
451  return opOperands;
452  };
453 
454  // Mark the non-forwarded operands of `regionBranchOp` in
455  // `nonForwardedOperands`.
456  auto markNonForwardedOperands = [&](BitVector &nonForwardedOperands) {
457  nonForwardedOperands.resize(regionBranchOp->getNumOperands(), true);
458  for (const RegionSuccessor &successor : getSuccessors()) {
459  for (OpOperand *opOperand : getForwardedOpOperands(successor))
460  nonForwardedOperands.reset(opOperand->getOperandNumber());
461  }
462  };
463 
464  // Mark the non-forwarded terminator operands of the various regions of
465  // `regionBranchOp` in `nonForwardedRets`.
466  auto markNonForwardedReturnValues =
467  [&](DenseMap<Operation *, BitVector> &nonForwardedRets) {
468  for (Region &region : regionBranchOp->getRegions()) {
469  if (region.empty())
470  continue;
471  Operation *terminator = region.front().getTerminator();
472  nonForwardedRets[terminator] =
473  BitVector(terminator->getNumOperands(), true);
474  for (const RegionSuccessor &successor : getSuccessors(&region)) {
475  for (OpOperand *opOperand :
476  getForwardedOpOperands(successor, terminator))
477  nonForwardedRets[terminator].reset(opOperand->getOperandNumber());
478  }
479  }
480  };
481 
482  // Update `valuesToKeep` (which is expected to correspond to operands or
483  // terminator operands) based on `resultsToKeep` and `argsToKeep`, given
484  // `region`. When `valuesToKeep` correspond to operands, `region` is null.
485  // Else, `region` is the parent region of the terminator.
486  auto updateOperandsOrTerminatorOperandsToKeep =
487  [&](BitVector &valuesToKeep, BitVector &resultsToKeep,
488  DenseMap<Region *, BitVector> &argsToKeep, Region *region = nullptr) {
489  Operation *terminator =
490  region ? region->front().getTerminator() : nullptr;
491 
492  for (const RegionSuccessor &successor : getSuccessors(region)) {
493  Region *successorRegion = successor.getSuccessor();
494  for (auto [opOperand, input] :
495  llvm::zip(getForwardedOpOperands(successor, terminator),
496  successor.getSuccessorInputs())) {
497  size_t operandNum = opOperand->getOperandNumber();
498  bool updateBasedOn =
499  successorRegion
500  ? argsToKeep[successorRegion]
501  [cast<BlockArgument>(input).getArgNumber()]
502  : resultsToKeep[cast<OpResult>(input).getResultNumber()];
503  valuesToKeep[operandNum] = valuesToKeep[operandNum] | updateBasedOn;
504  }
505  }
506  };
507 
508  // Recompute `resultsToKeep` and `argsToKeep` based on `operandsToKeep` and
509  // `terminatorOperandsToKeep`. Store true in `resultsOrArgsToKeepChanged` if a
510  // value is modified, else, false.
511  auto recomputeResultsAndArgsToKeep =
512  [&](BitVector &resultsToKeep, DenseMap<Region *, BitVector> &argsToKeep,
513  BitVector &operandsToKeep,
514  DenseMap<Operation *, BitVector> &terminatorOperandsToKeep,
515  bool &resultsOrArgsToKeepChanged) {
516  resultsOrArgsToKeepChanged = false;
517 
518  // Recompute `resultsToKeep` and `argsToKeep` based on `operandsToKeep`.
519  for (const RegionSuccessor &successor : getSuccessors()) {
520  Region *successorRegion = successor.getSuccessor();
521  for (auto [opOperand, input] :
522  llvm::zip(getForwardedOpOperands(successor),
523  successor.getSuccessorInputs())) {
524  bool recomputeBasedOn =
525  operandsToKeep[opOperand->getOperandNumber()];
526  bool toRecompute =
527  successorRegion
528  ? argsToKeep[successorRegion]
529  [cast<BlockArgument>(input).getArgNumber()]
530  : resultsToKeep[cast<OpResult>(input).getResultNumber()];
531  if (!toRecompute && recomputeBasedOn)
532  resultsOrArgsToKeepChanged = true;
533  if (successorRegion) {
534  argsToKeep[successorRegion][cast<BlockArgument>(input)
535  .getArgNumber()] =
536  argsToKeep[successorRegion]
537  [cast<BlockArgument>(input).getArgNumber()] |
538  recomputeBasedOn;
539  } else {
540  resultsToKeep[cast<OpResult>(input).getResultNumber()] =
541  resultsToKeep[cast<OpResult>(input).getResultNumber()] |
542  recomputeBasedOn;
543  }
544  }
545  }
546 
547  // Recompute `resultsToKeep` and `argsToKeep` based on
548  // `terminatorOperandsToKeep`.
549  for (Region &region : regionBranchOp->getRegions()) {
550  if (region.empty())
551  continue;
552  Operation *terminator = region.front().getTerminator();
553  for (const RegionSuccessor &successor : getSuccessors(&region)) {
554  Region *successorRegion = successor.getSuccessor();
555  for (auto [opOperand, input] :
556  llvm::zip(getForwardedOpOperands(successor, terminator),
557  successor.getSuccessorInputs())) {
558  bool recomputeBasedOn =
559  terminatorOperandsToKeep[region.back().getTerminator()]
560  [opOperand->getOperandNumber()];
561  bool toRecompute =
562  successorRegion
563  ? argsToKeep[successorRegion]
564  [cast<BlockArgument>(input).getArgNumber()]
565  : resultsToKeep[cast<OpResult>(input).getResultNumber()];
566  if (!toRecompute && recomputeBasedOn)
567  resultsOrArgsToKeepChanged = true;
568  if (successorRegion) {
569  argsToKeep[successorRegion][cast<BlockArgument>(input)
570  .getArgNumber()] =
571  argsToKeep[successorRegion]
572  [cast<BlockArgument>(input).getArgNumber()] |
573  recomputeBasedOn;
574  } else {
575  resultsToKeep[cast<OpResult>(input).getResultNumber()] =
576  resultsToKeep[cast<OpResult>(input).getResultNumber()] |
577  recomputeBasedOn;
578  }
579  }
580  }
581  }
582  };

584  // Mark the values that we want to keep in `resultsToKeep`, `argsToKeep`,
585  // `operandsToKeep`, and `terminatorOperandsToKeep`.
586  auto markValuesToKeep =
587  [&](BitVector &resultsToKeep, DenseMap<Region *, BitVector> &argsToKeep,
588  BitVector &operandsToKeep,
589  DenseMap<Operation *, BitVector> &terminatorOperandsToKeep) {
590  bool resultsOrArgsToKeepChanged = true;
591  // We keep updating and recomputing the values until we reach a point
592  // where they stop changing.
593  while (resultsOrArgsToKeepChanged) {
594  // Update the operands that need to be kept.
595  updateOperandsOrTerminatorOperandsToKeep(operandsToKeep,
596  resultsToKeep, argsToKeep);
597 
598  // Update the terminator operands that need to be kept.
599  for (Region &region : regionBranchOp->getRegions()) {
600  if (region.empty())
601  continue;
602  updateOperandsOrTerminatorOperandsToKeep(
603  terminatorOperandsToKeep[region.back().getTerminator()],
604  resultsToKeep, argsToKeep, &region);
605  }
606 
607  // Recompute the results and arguments that need to be kept.
608  recomputeResultsAndArgsToKeep(
609  resultsToKeep, argsToKeep, operandsToKeep,
610  terminatorOperandsToKeep, resultsOrArgsToKeepChanged);
611  }
612  };
613 
614  // Scenario 1. This is the only case where the entire `regionBranchOp`
615  // is removed. It will not happen in any other scenario. Note that in this
616  // case, a non-forwarded operand of `regionBranchOp` could be live/non-live.
617  // It could never be live because of this op but its liveness could have been
618  // attributed to something else.
619  // Do (1') and (2').
620  if (isMemoryEffectFree(regionBranchOp.getOperation()) &&
621  !hasLive(regionBranchOp->getResults(), nonLiveSet, la)) {
622  cl.operations.push_back(regionBranchOp.getOperation());
623  return;
624  }
625 
626  // Scenario 2.
627  // At this point, we know that every non-forwarded operand of `regionBranchOp`
628  // is live.
629 
630  // Stores the results of `regionBranchOp` that we want to keep.
631  BitVector resultsToKeep;
632  // Stores the mapping from regions of `regionBranchOp` to their arguments that
633  // we want to keep.
634  DenseMap<Region *, BitVector> argsToKeep;
635  // Stores the operands of `regionBranchOp` that we want to keep.
636  BitVector operandsToKeep;
637  // Stores the mapping from region terminators in `regionBranchOp` to their
638  // operands that we want to keep.
639  DenseMap<Operation *, BitVector> terminatorOperandsToKeep;
640 
641  // Initializing the above variables...
642 
643  // The live results of `regionBranchOp` definitely need to be kept.
644  markLiveResults(resultsToKeep);
645  // Similarly, the live arguments of the regions in `regionBranchOp` definitely
646  // need to be kept.
647  markLiveArgs(argsToKeep);
648  // The non-forwarded operands of `regionBranchOp` definitely need to be kept.
649  // A live forwarded operand can be removed but no non-forwarded operand can be
650  // removed since it "controls" the flow of data in this control flow op.
651  markNonForwardedOperands(operandsToKeep);
652  // Similarly, the non-forwarded terminator operands of the regions in
653  // `regionBranchOp` definitely need to be kept.
654  markNonForwardedReturnValues(terminatorOperandsToKeep);
655 
656  // Mark the values (results, arguments, operands, and terminator operands)
657  // that we want to keep.
658  markValuesToKeep(resultsToKeep, argsToKeep, operandsToKeep,
659  terminatorOperandsToKeep);
660 
661  // Do (1).
662  cl.operands.push_back({regionBranchOp, operandsToKeep.flip()});
663 
664  // Do (2.a) and (2.b).
665  for (Region &region : regionBranchOp->getRegions()) {
666  if (region.empty())
667  continue;
668  BitVector argsToRemove = argsToKeep[&region].flip();
669  cl.blocks.push_back({&region.front(), argsToRemove});
670  collectNonLiveValues(nonLiveSet, region.front().getArguments(),
671  argsToRemove);
672  }
673 
674  // Do (2.c).
675  for (Region &region : regionBranchOp->getRegions()) {
676  if (region.empty())
677  continue;
678  Operation *terminator = region.front().getTerminator();
679  cl.operands.push_back(
680  {terminator, terminatorOperandsToKeep[terminator].flip()});
681  }
682 
683  // Do (3) and (4).
684  BitVector resultsToRemove = resultsToKeep.flip();
685  collectNonLiveValues(nonLiveSet, regionBranchOp.getOperation()->getResults(),
686  resultsToRemove);
687  cl.results.push_back({regionBranchOp.getOperation(), resultsToRemove});
688 }

690 /// Steps to process a `BranchOpInterface` operation:
691 /// Iterate through each successor block of `branchOp`.
692 /// (1) For each successor block, gather all operands from all successors.
693 /// (2) Fetch their associated liveness analysis data and collect for future
694 /// removal.
695 /// (3) Identify and collect the dead operands from the successor block
696 /// as well as their corresponding arguments.
697 
698 static void processBranchOp(BranchOpInterface branchOp, RunLivenessAnalysis &la,
699  DenseSet<Value> &nonLiveSet,
700  RDVFinalCleanupList &cl) {
701  LDBG() << "Processing branch op: " << *branchOp;
702  unsigned numSuccessors = branchOp->getNumSuccessors();
703 
704  for (unsigned succIdx = 0; succIdx < numSuccessors; ++succIdx) {
705  Block *successorBlock = branchOp->getSuccessor(succIdx);
706 
707  // Do (1)
708  SuccessorOperands successorOperands =
709  branchOp.getSuccessorOperands(succIdx);
710  SmallVector<Value> operandValues;
711  for (unsigned operandIdx = 0; operandIdx < successorOperands.size();
712  ++operandIdx) {
713  operandValues.push_back(successorOperands[operandIdx]);
714  }
715 
716  // Do (2)
717  BitVector successorNonLive =
718  markLives(operandValues, nonLiveSet, la).flip();
719  collectNonLiveValues(nonLiveSet, successorBlock->getArguments(),
720  successorNonLive);
721 
722  // Do (3)
723  cl.blocks.push_back({successorBlock, successorNonLive});
724  cl.successorOperands.push_back({branchOp, succIdx, successorNonLive});
725  }
726 }

728 /// Removes dead values collected in RDVFinalCleanupList.
729 /// To be run once when all dead values have been collected.
730 static void cleanUpDeadVals(RDVFinalCleanupList &list) {
731  // 1. Operations
732  for (auto &op : list.operations) {
733  op->dropAllUses();
734  op->erase();
735  }
736 
737  // 2. Values
738  for (auto &v : list.values) {
739  v.dropAllUses();
740  }
741 
742  // 3. Functions
743  for (auto &f : list.functions) {
744  // Some functions may not allow erasing arguments or results. These calls
745  // return failure in such cases without modifying the function, so it's okay
746  // to proceed.
747  (void)f.funcOp.eraseArguments(f.nonLiveArgs);
748  (void)f.funcOp.eraseResults(f.nonLiveRets);
749  }
750 
751  // 4. Operands
752  for (OperationToCleanup &o : list.operands) {
753  if (o.op->getNumOperands() > 0)
754  o.op->eraseOperands(o.nonLive);
755  }
756 
757  // 5. Results
758  for (auto &r : list.results) {
759  dropUsesAndEraseResults(r.op, r.nonLive);
760  }
761 
762  // 6. Blocks
763  for (auto &b : list.blocks) {
764  // blocks that are accessed via multiple codepaths processed once
765  if (b.b->getNumArguments() != b.nonLiveArgs.size())
766  continue;
767  // it iterates backwards because erase invalidates all successor indexes
768  for (int i = b.nonLiveArgs.size() - 1; i >= 0; --i) {
769  if (!b.nonLiveArgs[i])
770  continue;
771  b.b->getArgument(i).dropAllUses();
772  b.b->eraseArgument(i);
773  }
774  }
775 
776  // 7. Successor Operands
777  for (auto &op : list.successorOperands) {
778  SuccessorOperands successorOperands =
779  op.branch.getSuccessorOperands(op.successorIndex);
780  // blocks that are accessed via multiple codepaths processed once
781  if (successorOperands.size() != op.nonLiveOperands.size())
782  continue;
783  // it iterates backwards because erase invalidates all successor indexes
784  for (int i = successorOperands.size() - 1; i >= 0; --i) {
785  if (!op.nonLiveOperands[i])
786  continue;
787  successorOperands.erase(i);
788  }
789  }
790 }

797 void RemoveDeadValues::runOnOperation() {
798  auto &la = getAnalysis<RunLivenessAnalysis>();
799  Operation *module = getOperation();
800 
801  // Tracks values eligible for erasure - complements liveness analysis to
802  // identify "droppable" values.
803  DenseSet<Value> deadVals;
804 
805  // Maintains a list of Ops, values, branches, etc., slated for cleanup at the
806  // end of this pass.
807  RDVFinalCleanupList finalCleanupList;
808 
809  module->walk([&](Operation *op) {
810  if (auto funcOp = dyn_cast<FunctionOpInterface>(op)) {
811  processFuncOp(funcOp, module, la, deadVals, finalCleanupList);
812  } else if (auto regionBranchOp = dyn_cast<RegionBranchOpInterface>(op)) {
813  processRegionBranchOp(regionBranchOp, la, deadVals, finalCleanupList);
814  } else if (auto branchOp = dyn_cast<BranchOpInterface>(op)) {
815  processBranchOp(branchOp, la, deadVals, finalCleanupList);
816  } else if (op->hasTrait<::mlir::OpTrait::IsTerminator>()) {
817  // Nothing to do here because this is a terminator op and it should be
818  // honored with respect to its parent
819  } else if (isa<CallOpInterface>(op)) {
820  // Nothing to do because this op is associated with a function op and gets
821  // cleaned when the latter is cleaned.
822  } else {
823  processSimpleOp(op, la, deadVals, finalCleanupList);
824  }
825  });
826 
827  cleanUpDeadVals(finalCleanupList);
828 }
