43 /// Given a predecessor `block` for a block with arguments, returns the block
44 /// that should be used as the parent block for SPIR-V OpPhi instructions
45 /// corresponding to the block arguments.
46 static Block *getPhiIncomingBlock(Block *block) {
47  // If the predecessor block in question is the entry block for a
48  // spirv.mlir.loop, we jump to this spirv.mlir.loop from its enclosing block.
49  if (block->isEntryBlock()) {
50  if (auto loopOp = dyn_cast<spirv::LoopOp>(block->getParentOp())) {
51  // Then the incoming parent block for OpPhi should be the merge block of
52  // the structured control flow op before this loop.
53  Operation *op = loopOp.getOperation();
54  while ((op = op->getPrevNode()) != nullptr)
55  if (Block *incomingBlock = getStructuredControlFlowOpMergeBlock(op))
56  return incomingBlock;
57  // Or the enclosing block itself if no structured control flow ops
58  // exists before this loop.
59  return loopOp->getBlock();
60  }
61  }
62 
63  // Otherwise, we jump from the given predecessor block. Try to see if there is
64  // a structured control flow op inside it.
65  for (Operation &op : llvm::reverse(block->getOperations())) {
66  if (Block *incomingBlock = getStructuredControlFlowOpMergeBlock(&op))
67  return incomingBlock;
68  }
69  return block;
70 }

72 static bool isZeroValue(Attribute attr) {
73  if (auto floatAttr = dyn_cast<FloatAttr>(attr)) {
74  return floatAttr.getValue().isZero();
75  }
76  if (auto boolAttr = dyn_cast<BoolAttr>(attr)) {
77  return !boolAttr.getValue();
78  }
79  if (auto intAttr = dyn_cast<IntegerAttr>(attr)) {
80  return intAttr.getValue().isZero();
81  }
82  if (auto splatElemAttr = dyn_cast<SplatElementsAttr>(attr)) {
83  return isZeroValue(splatElemAttr.getSplatValue<Attribute>());
84  }
85  if (auto denseElemAttr = dyn_cast<DenseElementsAttr>(attr)) {
86  return all_of(denseElemAttr.getValues<Attribute>(), isZeroValue);
87  }
88  return false;
89 }

107 LogicalResult Serializer::serialize() {
108  LLVM_DEBUG(llvm::dbgs() << "+++ starting serialization +++\n");
109 
110  if (failed(module.verifyInvariants()))
111  return failure();
112 
113  // TODO: handle the other sections
114  processCapability();
115  processExtension();
116  processMemoryModel();
117  processDebugInfo();
118 
119  // Iterate over the module body to serialize it. Assumptions are that there is
120  // only one basic block in the moduleOp
121  for (auto &op : *module.getBody()) {
122  if (failed(processOperation(&op))) {
123  return failure();
124  }
125  }
126 
127  LLVM_DEBUG(llvm::dbgs() << "+++ completed serialization +++\n");
128  return success();
129 }

132  auto moduleSize = spirv::kHeaderWordCount + capabilities.size() +
133  extensions.size() + extendedSets.size() +
134  memoryModel.size() + entryPoints.size() +
135  executionModes.size() + decorations.size() +
136  typesGlobalValues.size() + functions.size();
137 
138  binary.clear();
139  binary.reserve(moduleSize);
140 
141  spirv::appendModuleHeader(binary, module.getVceTriple()->getVersion(),
142  nextID);
143  binary.append(capabilities.begin(), capabilities.end());
144  binary.append(extensions.begin(), extensions.end());
145  binary.append(extendedSets.begin(), extendedSets.end());
146  binary.append(memoryModel.begin(), memoryModel.end());
147  binary.append(entryPoints.begin(), entryPoints.end());
148  binary.append(executionModes.begin(), executionModes.end());
149  binary.append(debug.begin(), debug.end());
150  binary.append(names.begin(), names.end());
151  binary.append(decorations.begin(), decorations.end());
152  binary.append(typesGlobalValues.begin(), typesGlobalValues.end());
153  binary.append(functions.begin(), functions.end());

247 template <typename AttrTy, typename EmitF>
248 LogicalResult processDecorationList(Location loc, Decoration decoration,
249  Attribute attrList, StringRef attrName,
250  EmitF emitter) {
251  auto arrayAttr = dyn_cast<ArrayAttr>(attrList);
252  if (!arrayAttr) {
253  return emitError(loc, "expecting array attribute of ")
254  << attrName << " for " << stringifyDecoration(decoration);
255  }
256  if (arrayAttr.empty()) {
257  return emitError(loc, "expecting non-empty array attribute of ")
258  << attrName << " for " << stringifyDecoration(decoration);
259  }
260  for (Attribute attr : arrayAttr.getValue()) {
261  auto cacheControlAttr = dyn_cast<AttrTy>(attr);
262  if (!cacheControlAttr) {
263  return emitError(loc, "expecting array attribute of ")
264  << attrName << " for " << stringifyDecoration(decoration);
265  }
266  // This named attribute encodes several decorations. Emit one per
267  // element in the array.
268  if (failed(emitter(cacheControlAttr)))
269  return failure();
270  }
271  return success();
272 }

274 LogicalResult Serializer::processDecorationAttr(Location loc, uint32_t resultID,
275  Decoration decoration,
276  Attribute attr) {
277  SmallVector<uint32_t, 1> args;
278  switch (decoration) {
279  case spirv::Decoration::LinkageAttributes: {
280  // Get the value of the Linkage Attributes
281  // e.g., LinkageAttributes=["linkageName", linkageType].
282  auto linkageAttr = llvm::dyn_cast<spirv::LinkageAttributesAttr>(attr);
283  auto linkageName = linkageAttr.getLinkageName();
284  auto linkageType = linkageAttr.getLinkageType().getValue();
285  // Encode the Linkage Name (string literal to uint32_t).
286  spirv::encodeStringLiteralInto(args, linkageName);
287  // Encode LinkageType & Add the Linkagetype to the args.
288  args.push_back(static_cast<uint32_t>(linkageType));
289  break;
290  }
291  case spirv::Decoration::FPFastMathMode:
292  if (auto intAttr = dyn_cast<FPFastMathModeAttr>(attr)) {
293  args.push_back(static_cast<uint32_t>(intAttr.getValue()));
294  break;
295  }
296  return emitError(loc, "expected FPFastMathModeAttr attribute for ")
297  << stringifyDecoration(decoration);
298  case spirv::Decoration::FPRoundingMode:
299  if (auto intAttr = dyn_cast<FPRoundingModeAttr>(attr)) {
300  args.push_back(static_cast<uint32_t>(intAttr.getValue()));
301  break;
302  }
303  return emitError(loc, "expected FPRoundingModeAttr attribute for ")
304  << stringifyDecoration(decoration);
305  case spirv::Decoration::Binding:
306  case spirv::Decoration::DescriptorSet:
307  case spirv::Decoration::Location:
308  if (auto intAttr = dyn_cast<IntegerAttr>(attr)) {
309  args.push_back(intAttr.getValue().getZExtValue());
310  break;
311  }
312  return emitError(loc, "expected integer attribute for ")
313  << stringifyDecoration(decoration);
314  case spirv::Decoration::BuiltIn:
315  if (auto strAttr = dyn_cast<StringAttr>(attr)) {
316  auto enumVal = spirv::symbolizeBuiltIn(strAttr.getValue());
317  if (enumVal) {
318  args.push_back(static_cast<uint32_t>(*enumVal));
319  break;
320  }
321  return emitError(loc, "invalid ")
322  << stringifyDecoration(decoration) << " decoration attribute "
323  << strAttr.getValue();
324  }
325  return emitError(loc, "expected string attribute for ")
326  << stringifyDecoration(decoration);
327  case spirv::Decoration::Aliased:
328  case spirv::Decoration::AliasedPointer:
329  case spirv::Decoration::Flat:
330  case spirv::Decoration::NonReadable:
331  case spirv::Decoration::NonWritable:
332  case spirv::Decoration::NoPerspective:
333  case spirv::Decoration::NoSignedWrap:
334  case spirv::Decoration::NoUnsignedWrap:
335  case spirv::Decoration::RelaxedPrecision:
336  case spirv::Decoration::Restrict:
337  case spirv::Decoration::RestrictPointer:
338  case spirv::Decoration::NoContraction:
339  case spirv::Decoration::Constant:
340  case spirv::Decoration::Block:
341  // For unit attributes and decoration attributes, the args list
342  // has no values so we do nothing.
343  if (isa<UnitAttr, DecorationAttr>(attr))
344  break;
345  return emitError(loc,
346  "expected unit attribute or decoration attribute for ")
347  << stringifyDecoration(decoration);
348  case spirv::Decoration::CacheControlLoadINTEL:
349  return processDecorationList<CacheControlLoadINTELAttr>(
350  loc, decoration, attr, "CacheControlLoadINTEL",
351  [&](CacheControlLoadINTELAttr attr) {
352  unsigned cacheLevel = attr.getCacheLevel();
353  LoadCacheControl loadCacheControl = attr.getLoadCacheControl();
354  return emitDecoration(
355  resultID, decoration,
356  {cacheLevel, static_cast<uint32_t>(loadCacheControl)});
357  });
358  case spirv::Decoration::CacheControlStoreINTEL:
359  return processDecorationList<CacheControlStoreINTELAttr>(
360  loc, decoration, attr, "CacheControlStoreINTEL",
361  [&](CacheControlStoreINTELAttr attr) {
362  unsigned cacheLevel = attr.getCacheLevel();
363  StoreCacheControl storeCacheControl = attr.getStoreCacheControl();
364  return emitDecoration(
365  resultID, decoration,
366  {cacheLevel, static_cast<uint32_t>(storeCacheControl)});
367  });
368  default:
369  return emitError(loc, "unhandled decoration ")
370  << stringifyDecoration(decoration);
371  }
372  return emitDecoration(resultID, decoration, args);
373 }

466 LogicalResult
467 Serializer::processTypeImpl(Location loc, Type type, uint32_t &typeID,
468  SetVector<StringRef> &serializationCtx) {
469 
470  // Map unsigned integer types to singless integer types.
471  // This is needed otherwise the generated spirv assembly will contain
472  // twice a type declaration (like OpTypeInt 32 0) which is no permitted and
473  // such module fails validation. Indeed at MLIR level the two types are
474  // different and lookup in the cache below misses.
475  // Note: This conversion needs to happen here before the type is looked up in
476  // the cache.
477  if (type.isUnsignedInteger()) {
478  type = IntegerType::get(loc->getContext(), type.getIntOrFloatBitWidth(),
479  IntegerType::SignednessSemantics::Signless);
480  }
481 
482  typeID = getTypeID(type);
483  if (typeID)
484  return success();
485 
486  typeID = getNextID();
487  SmallVector<uint32_t, 4> operands;
488 
489  operands.push_back(typeID);
490  auto typeEnum = spirv::Opcode::OpTypeVoid;
491  bool deferSerialization = false;
492 
493  if ((isa<FunctionType>(type) &&
494  succeeded(prepareFunctionType(loc, cast<FunctionType>(type), typeEnum,
495  operands))) ||
496  succeeded(prepareBasicType(loc, type, typeID, typeEnum, operands,
497  deferSerialization, serializationCtx))) {
498  if (deferSerialization)
499  return success();
500 
501  typeIDMap[type] = typeID;
502 
503  encodeInstructionInto(typesGlobalValues, typeEnum, operands);
504 
505  if (recursiveStructInfos.count(type) != 0) {
506  // This recursive struct type is emitted already, now the OpTypePointer
507  // instructions referring to recursive references are emitted as well.
508  for (auto &ptrInfo : recursiveStructInfos[type]) {
509  // TODO: This might not work if more than 1 recursive reference is
510  // present in the struct.
511  SmallVector<uint32_t, 4> ptrOperands;
512  ptrOperands.push_back(ptrInfo.pointerTypeID);
513  ptrOperands.push_back(static_cast<uint32_t>(ptrInfo.storageClass));
514  ptrOperands.push_back(typeIDMap[type]);
515 
516  encodeInstructionInto(typesGlobalValues, spirv::Opcode::OpTypePointer,
517  ptrOperands);
518  }
519 
520  recursiveStructInfos[type].clear();
521  }
522 
523  return success();
524  }
525 
526  return failure();
527 }

529 LogicalResult Serializer::prepareBasicType(
530  Location loc, Type type, uint32_t resultID, spirv::Opcode &typeEnum,
531  SmallVectorImpl<uint32_t> &operands, bool &deferSerialization,
532  SetVector<StringRef> &serializationCtx) {
533  deferSerialization = false;
534 
535  if (isVoidType(type)) {
536  typeEnum = spirv::Opcode::OpTypeVoid;
537  return success();
538  }
539 
540  if (auto intType = dyn_cast<IntegerType>(type)) {
541  if (intType.getWidth() == 1) {
542  typeEnum = spirv::Opcode::OpTypeBool;
543  return success();
544  }
545 
546  typeEnum = spirv::Opcode::OpTypeInt;
547  operands.push_back(intType.getWidth());
548  // SPIR-V OpTypeInt "Signedness specifies whether there are signed semantics
549  // to preserve or validate.
550  // 0 indicates unsigned, or no signedness semantics
551  // 1 indicates signed semantics."
552  operands.push_back(intType.isSigned() ? 1 : 0);
553  return success();
554  }
555 
556  if (auto floatType = dyn_cast<FloatType>(type)) {
557  typeEnum = spirv::Opcode::OpTypeFloat;
558  operands.push_back(floatType.getWidth());
559  if (floatType.isBF16()) {
560  operands.push_back(static_cast<uint32_t>(spirv::FPEncoding::BFloat16KHR));
561  }
562  return success();
563  }
564 
565  if (auto vectorType = dyn_cast<VectorType>(type)) {
566  uint32_t elementTypeID = 0;
567  if (failed(processTypeImpl(loc, vectorType.getElementType(), elementTypeID,
568  serializationCtx))) {
569  return failure();
570  }
571  typeEnum = spirv::Opcode::OpTypeVector;
572  operands.push_back(elementTypeID);
573  operands.push_back(vectorType.getNumElements());
574  return success();
575  }
576 
577  if (auto imageType = dyn_cast<spirv::ImageType>(type)) {
578  typeEnum = spirv::Opcode::OpTypeImage;
579  uint32_t sampledTypeID = 0;
580  if (failed(processType(loc, imageType.getElementType(), sampledTypeID)))
581  return failure();
582 
583  llvm::append_values(operands, sampledTypeID,
584  static_cast<uint32_t>(imageType.getDim()),
585  static_cast<uint32_t>(imageType.getDepthInfo()),
586  static_cast<uint32_t>(imageType.getArrayedInfo()),
587  static_cast<uint32_t>(imageType.getSamplingInfo()),
588  static_cast<uint32_t>(imageType.getSamplerUseInfo()),
589  static_cast<uint32_t>(imageType.getImageFormat()));
590  return success();
591  }
592 
593  if (auto arrayType = dyn_cast<spirv::ArrayType>(type)) {
594  typeEnum = spirv::Opcode::OpTypeArray;
595  uint32_t elementTypeID = 0;
596  if (failed(processTypeImpl(loc, arrayType.getElementType(), elementTypeID,
597  serializationCtx))) {
598  return failure();
599  }
600  operands.push_back(elementTypeID);
601  if (auto elementCountID = prepareConstantInt(
602  loc, mlirBuilder.getI32IntegerAttr(arrayType.getNumElements()))) {
603  operands.push_back(elementCountID);
604  }
605  return processTypeDecoration(loc, arrayType, resultID);
606  }
607 
608  if (auto ptrType = dyn_cast<spirv::PointerType>(type)) {
609  uint32_t pointeeTypeID = 0;
610  spirv::StructType pointeeStruct =
611  dyn_cast<spirv::StructType>(ptrType.getPointeeType());
612 
613  if (pointeeStruct && pointeeStruct.isIdentified() &&
614  serializationCtx.count(pointeeStruct.getIdentifier()) != 0) {
615  // A recursive reference to an enclosing struct is found.
616  //
617  // 1. Prepare an OpTypeForwardPointer with resultID and the ptr storage
618  // class as operands.
619  SmallVector<uint32_t, 2> forwardPtrOperands;
620  forwardPtrOperands.push_back(resultID);
621  forwardPtrOperands.push_back(
622  static_cast<uint32_t>(ptrType.getStorageClass()));
623 
624  encodeInstructionInto(typesGlobalValues,
625  spirv::Opcode::OpTypeForwardPointer,
626  forwardPtrOperands);
627 
628  // 2. Find the pointee (enclosing) struct.
629  auto structType = spirv::StructType::getIdentified(
630  module.getContext(), pointeeStruct.getIdentifier());
631 
632  if (!structType)
633  return failure();
634 
635  // 3. Mark the OpTypePointer that is supposed to be emitted by this call
636  // as deferred.
637  deferSerialization = true;
638 
639  // 4. Record the info needed to emit the deferred OpTypePointer
640  // instruction when the enclosing struct is completely serialized.
641  recursiveStructInfos[structType].push_back(
642  {resultID, ptrType.getStorageClass()});
643  } else {
644  if (failed(processTypeImpl(loc, ptrType.getPointeeType(), pointeeTypeID,
645  serializationCtx)))
646  return failure();
647  }
648 
649  typeEnum = spirv::Opcode::OpTypePointer;
650  operands.push_back(static_cast<uint32_t>(ptrType.getStorageClass()));
651  operands.push_back(pointeeTypeID);
652 
653  // TODO: Now struct decorations are supported this code may not be
654  // necessary. However, it is left to support backwards compatibility.
655  // Ideally, Block decorations should be inserted when converting to SPIR-V.
656  if (isInterfaceStructPtrType(ptrType)) {
657  auto structType = cast<spirv::StructType>(ptrType.getPointeeType());
658  if (!structType.hasDecoration(spirv::Decoration::Block))
659  if (failed(emitDecoration(getTypeID(pointeeStruct),
660  spirv::Decoration::Block)))
661  return emitError(loc, "cannot decorate ")
662  << pointeeStruct << " with Block decoration";
663  }
664 
665  return success();
666  }
667 
668  if (auto runtimeArrayType = dyn_cast<spirv::RuntimeArrayType>(type)) {
669  uint32_t elementTypeID = 0;
670  if (failed(processTypeImpl(loc, runtimeArrayType.getElementType(),
671  elementTypeID, serializationCtx))) {
672  return failure();
673  }
674  typeEnum = spirv::Opcode::OpTypeRuntimeArray;
675  operands.push_back(elementTypeID);
676  return processTypeDecoration(loc, runtimeArrayType, resultID);
677  }
678 
679  if (auto sampledImageType = dyn_cast<spirv::SampledImageType>(type)) {
680  typeEnum = spirv::Opcode::OpTypeSampledImage;
681  uint32_t imageTypeID = 0;
682  if (failed(
683  processType(loc, sampledImageType.getImageType(), imageTypeID))) {
684  return failure();
685  }
686  operands.push_back(imageTypeID);
687  return success();
688  }
689 
690  if (auto structType = dyn_cast<spirv::StructType>(type)) {
691  if (structType.isIdentified()) {
692  if (failed(processName(resultID, structType.getIdentifier())))
693  return failure();
694  serializationCtx.insert(structType.getIdentifier());
695  }
696

698  for (auto elementIndex :
699  llvm::seq<uint32_t>(0, structType.getNumElements())) {

702  elementTypeID, serializationCtx))) {
703  return failure();

705  operands.push_back(elementTypeID);

708  // Decorate each struct member with an offset
709  spirv::StructType::MemberDecorationInfo offsetDecoration{
710  elementIndex, spirv::Decoration::Offset,
711  IntegerAttr::get(intType,
712  structType.getMemberOffset(elementIndex))};

714  return emitError(loc, "cannot decorate ")
715  << elementIndex << "-th member of " << structType
716  << " with its offset";
717  }
718  }
719  }

721  structType.getMemberDecorations(memberDecorations);
722 
723  for (auto &memberDecoration : memberDecorations) {
724  if (failed(processMemberDecoration(resultID, memberDecoration))) {
725  return emitError(loc, "cannot decorate ")
726  << static_cast<uint32_t>(memberDecoration.memberIndex)
727  << "-th member of " << structType << " with "
728  << stringifyDecoration(memberDecoration.decoration);
729  }
730  }
731

733  structType.getStructDecorations(structDecorations);
734 
735  for (spirv::StructType::StructDecorationInfo &structDecoration :
736  structDecorations) {
737  if (failed(processDecorationAttr(loc, resultID,
738  structDecoration.decoration,
739  structDecoration.decorationValue))) {
740  return emitError(loc, "cannot decorate struct ")
741  << structType << " with "
742  << stringifyDecoration(structDecoration.decoration);
743  }
744  }

747 
748  if (structType.isIdentified())
749  serializationCtx.remove(structType.getIdentifier());
750 
751  return success();
752  }
753 
754  if (auto cooperativeMatrixType =
755  dyn_cast<spirv::CooperativeMatrixType>(type)) {
756  uint32_t elementTypeID = 0;
757  if (failed(processTypeImpl(loc, cooperativeMatrixType.getElementType(),
758  elementTypeID, serializationCtx))) {
759  return failure();
760  }
761  typeEnum = spirv::Opcode::OpTypeCooperativeMatrixKHR;
762  auto getConstantOp = [&](uint32_t id) {
763  auto attr = IntegerAttr::get(IntegerType::get(type.getContext(), 32), id);
764  return prepareConstantInt(loc, attr);
765  };
766  llvm::append_values(
767  operands, elementTypeID,
768  getConstantOp(static_cast<uint32_t>(cooperativeMatrixType.getScope())),
769  getConstantOp(cooperativeMatrixType.getRows()),
770  getConstantOp(cooperativeMatrixType.getColumns()),
771  getConstantOp(static_cast<uint32_t>(cooperativeMatrixType.getUse())));
772  return success();
773  }
774 
775  if (auto matrixType = dyn_cast<spirv::MatrixType>(type)) {
776  uint32_t elementTypeID = 0;
777  if (failed(processTypeImpl(loc, matrixType.getColumnType(), elementTypeID,
778  serializationCtx))) {
779  return failure();
780  }
781  typeEnum = spirv::Opcode::OpTypeMatrix;
782  llvm::append_values(operands, elementTypeID, matrixType.getNumColumns());
783  return success();
784  }
785 
786  if (auto tensorArmType = llvm::dyn_cast<TensorArmType>(type)) {
787  uint32_t elementTypeID = 0;
788  uint32_t rank = 0;
789  uint32_t shapeID = 0;
790  uint32_t rankID = 0;
791  if (failed(processTypeImpl(loc, tensorArmType.getElementType(),
792  elementTypeID, serializationCtx))) {
793  return failure();
794  }
795  if (tensorArmType.hasRank()) {
796  ArrayRef<int64_t> dims = tensorArmType.getShape();
797  rank = dims.size();
798  rankID = prepareConstantInt(loc, mlirBuilder.getI32IntegerAttr(rank));
799  if (rankID == 0) {
800  return failure();
801  }
802 
803  bool shaped = llvm::all_of(dims, [](const auto &dim) { return dim > 0; });
804  if (rank > 0 && shaped) {
805  auto I32Type = IntegerType::get(type.getContext(), 32);
806  auto shapeType = ArrayType::get(I32Type, rank);
807  if (rank == 1) {
808  SmallVector<uint64_t, 1> index(rank);
809  shapeID = prepareDenseElementsConstant(
810  loc, shapeType,
811  mlirBuilder.getI32TensorAttr(SmallVector<int32_t>(dims)), 0,
812  index);
813  } else {
814  shapeID = prepareArrayConstant(
815  loc, shapeType,
816  mlirBuilder.getI32ArrayAttr(SmallVector<int32_t>(dims)));
817  }
818  if (shapeID == 0) {
819  return failure();
820  }
821  }
822  }
823  typeEnum = spirv::Opcode::OpTypeTensorARM;
824  operands.push_back(elementTypeID);
825  if (rankID == 0)
826  return success();
827  operands.push_back(rankID);
828  if (shapeID == 0)
829  return success();
830  operands.push_back(shapeID);
831  return success();
832  }
833

835  return emitError(loc, "unhandled type in serialization: ") << type;
836 }

929 uint32_t
930 Serializer::prepareDenseElementsConstant(Location loc, Type constType,
931  DenseElementsAttr valueAttr, int dim,
932  MutableArrayRef<uint64_t> index) {
933  auto shapedType = dyn_cast<ShapedType>(valueAttr.getType());
934  assert(dim <= shapedType.getRank());
935  if (shapedType.getRank() == dim) {
936  if (auto attr = dyn_cast<DenseIntElementsAttr>(valueAttr)) {
937  return attr.getType().getElementType().isInteger(1)
938  ? prepareConstantBool(loc, attr.getValues<BoolAttr>()[index])
939  : prepareConstantInt(loc,
940  attr.getValues<IntegerAttr>()[index]);
941  }
942  if (auto attr = dyn_cast<DenseFPElementsAttr>(valueAttr)) {
943  return prepareConstantFp(loc, attr.getValues<FloatAttr>()[index]);
944  }
945  return 0;
946  }
947 
948  uint32_t typeID = 0;
949  if (failed(processType(loc, constType, typeID))) {
950  return 0;
951  }
952 
953  int64_t numberOfConstituents = shapedType.getDimSize(dim);
954  uint32_t resultID = getNextID();
955  SmallVector<uint32_t, 4> operands = {typeID, resultID};
956  auto elementType = cast<spirv::CompositeType>(constType).getElementType(0);
957 
958  // "If the Result Type is a cooperative matrix type, then there must be only
959  // one Constituent, with scalar type matching the cooperative matrix Component
960  // Type, and all components of the matrix are initialized to that value."
961  // (https://github.khronos.org/SPIRV-Registry/extensions/KHR/SPV_KHR_cooperative_matrix.html)
962  if (isa<spirv::CooperativeMatrixType>(constType)) {
963  if (!valueAttr.isSplat()) {
964  emitError(
965  loc,
966  "cannot serialize a non-splat value for a cooperative matrix type");
967  return 0;
968  }
969  // numberOfConstituents is 1, so we only need one more elements in the
970  // SmallVector, so the total is 3 (1 + 2).
971  operands.reserve(3);
972  // We set dim directly to `shapedType.getRank()` so the recursive call
973  // directly returns the scalar type.
974  if (auto elementID = prepareDenseElementsConstant(
975  loc, elementType, valueAttr, /*dim=*/shapedType.getRank(), index)) {
976  operands.push_back(elementID);
977  } else {
978  return 0;
979  }
980  } else if (isa<spirv::TensorArmType>(constType)) {
981  if (isZeroValue(valueAttr)) {
982  encodeInstructionInto(typesGlobalValues, spirv::Opcode::OpConstantNull,
983  {typeID, resultID});
984  return resultID;
985  }
986  numberOfConstituents = shapedType.getNumElements();
987  operands.reserve(numberOfConstituents + 2);
988  for (int i = 0; i < numberOfConstituents; ++i) {
989  uint32_t elementID = 0;
990  if (auto attr = dyn_cast<DenseIntElementsAttr>(valueAttr)) {
991  elementID =
992  elementType.isInteger(1)
993  ? prepareConstantBool(loc, attr.getValues<BoolAttr>()[i])
994  : prepareConstantInt(loc, attr.getValues<IntegerAttr>()[i]);
995  }
996  if (auto attr = dyn_cast<DenseFPElementsAttr>(valueAttr)) {
997  elementID = prepareConstantFp(loc, attr.getValues<FloatAttr>()[i]);
998  }
999  if (!elementID) {
1000  return 0;
1001  }
1002  operands.push_back(elementID);
1003  }
1004  } else {
1005  operands.reserve(numberOfConstituents + 2);
1006  for (int i = 0; i < numberOfConstituents; ++i) {
1007  index[dim] = i;
1008  if (auto elementID = prepareDenseElementsConstant(
1009  loc, elementType, valueAttr, dim + 1, index)) {
1010  operands.push_back(elementID);
1011  } else {
1012  return 0;
1013  }
1014  }
1015  }
1016  spirv::Opcode opcode = spirv::Opcode::OpConstantComposite;
1017  encodeInstructionInto(typesGlobalValues, opcode, operands);
1018 
1019  return resultID;
1020 }

1066 uint32_t Serializer::prepareConstantInt(Location loc, IntegerAttr intAttr,
1067  bool isSpec) {
1068  if (!isSpec) {
1069  // We can de-duplicate normal constants, but not specialization constants.
1070  if (auto id = getConstantID(intAttr)) {
1071  return id;
1072  }
1073  }
1074 
1075  // Process the type for this integer literal
1076  uint32_t typeID = 0;
1077  if (failed(processType(loc, intAttr.getType(), typeID))) {
1078  return 0;
1079  }
1080 
1081  auto resultID = getNextID();
1082  APInt value = intAttr.getValue();
1083  unsigned bitwidth = value.getBitWidth();
1084  bool isSigned = intAttr.getType().isSignedInteger();
1085  auto opcode =
1086  isSpec ? spirv::Opcode::OpSpecConstant : spirv::Opcode::OpConstant;
1087 
1088  switch (bitwidth) {
1089  // According to SPIR-V spec, "When the type's bit width is less than
1090  // 32-bits, the literal's value appears in the low-order bits of the word,
1091  // and the high-order bits must be 0 for a floating-point type, or 0 for an
1092  // integer type with Signedness of 0, or sign extended when Signedness
1093  // is 1."
1094  case 32:
1095  case 16:
1096  case 8: {
1097  uint32_t word = 0;
1098  if (isSigned) {
1099  word = static_cast<int32_t>(value.getSExtValue());
1100  } else {
1101  word = static_cast<uint32_t>(value.getZExtValue());
1102  }
1103  encodeInstructionInto(typesGlobalValues, opcode, {typeID, resultID, word});
1104  } break;
1105  // According to SPIR-V spec: "When the type's bit width is larger than one
1106  // word, the literals low-order words appear first."
1107  case 64: {
1108  struct DoubleWord {
1109  uint32_t word1;
1110  uint32_t word2;
1111  } words;
1112  if (isSigned) {
1113  words = llvm::bit_cast<DoubleWord>(value.getSExtValue());
1114  } else {
1115  words = llvm::bit_cast<DoubleWord>(value.getZExtValue());
1116  }
1117  encodeInstructionInto(typesGlobalValues, opcode,
1118  {typeID, resultID, words.word1, words.word2});
1119  } break;
1120  default: {
1121  std::string valueStr;
1122  llvm::raw_string_ostream rss(valueStr);
1123  value.print(rss, /*isSigned=*/false);
1124 
1125  emitError(loc, "cannot serialize ")
1126  << bitwidth << "-bit integer literal: " << valueStr;
1127  return 0;
1128  }
1129  }
1130 
1131  if (!isSpec) {
1132  constIDMap[intAttr] = resultID;
1133  }
1134  return resultID;
1135 }

1264 Serializer::processBlock(Block *block, bool omitLabel,
1265  function_ref<LogicalResult()> emitMerge) {

1267  LLVM_DEBUG(block->print(llvm::dbgs()));
1268  LLVM_DEBUG(llvm::dbgs() << '\n');

1270  uint32_t blockID = getOrCreateBlockID(block);
1271  LLVM_DEBUG(printBlock(block, llvm::dbgs()));
1272 
1273  // Emit OpLabel for this block.
1274  encodeInstructionInto(functionBody, spirv::Opcode::OpLabel, {blockID});
1275  }
1276 
1277  // Emit OpPhi instructions for block arguments, if any.
1278  if (failed(emitPhiForBlockArguments(block)))
1279  return failure();
1280 
1281  // If we need to emit merge instructions, it must happen in this block. Check
1282  // whether we have other structured control flow ops, which will be expanded
1283  // into multiple basic blocks. If that's the case, we need to emit the merge
1284  // right now and then create new blocks for further serialization of the ops
1285  // in this block.
1286  if (emitMerge &&
1287  llvm::any_of(block->getOperations(),
1288  llvm::IsaPred<spirv::LoopOp, spirv::SelectionOp>)) {
1289  if (failed(emitMerge()))
1290  return failure();
1291  emitMerge = nullptr;
1292 
1293  // Start a new block for further serialization.
1294  uint32_t blockID = getNextID();
1295  encodeInstructionInto(functionBody, spirv::Opcode::OpBranch, {blockID});
1296  encodeInstructionInto(functionBody, spirv::Opcode::OpLabel, {blockID});
1297  }
1298 
1299  // Process each op in this block except the terminator.
1300  for (Operation &op : llvm::drop_end(*block)) {
1301  if (failed(processOperation(&op)))
1302  return failure();
1303  }
1304 
1305  // Process the terminator.
1306  if (emitMerge)
1307  if (failed(emitMerge()))
1308  return failure();
1309  if (failed(processOperation(&block->back())))
1310  return failure();
1311 
1312  return success();
1313 }

1315 LogicalResult Serializer::emitPhiForBlockArguments(Block *block) {
1316  // Nothing to do if this block has no arguments or it's the entry block, which
1317  // always has the same arguments as the function signature.
1318  if (block->args_empty() || block->isEntryBlock())
1319  return success();
1320 
1321  LLVM_DEBUG(llvm::dbgs() << "emitting phi instructions..\n");
1322 
1323  // If the block has arguments, we need to create SPIR-V OpPhi instructions.
1324  // A SPIR-V OpPhi instruction is of the syntax:
1325  // OpPhi | result type | result <id> | (value <id>, parent block <id>) pair
1326  // So we need to collect all predecessor blocks and the arguments they send
1327  // to this block.
1328  SmallVector<std::pair<Block *, OperandRange>, 4> predecessors;
1329  for (Block *mlirPredecessor : block->getPredecessors()) {
1330  auto *terminator = mlirPredecessor->getTerminator();
1331  LLVM_DEBUG(llvm::dbgs() << " mlir predecessor ");
1332  LLVM_DEBUG(printBlock(mlirPredecessor, llvm::dbgs()));
1333  LLVM_DEBUG(llvm::dbgs() << " terminator: " << *terminator << "\n");
1334  // The predecessor here is the immediate one according to MLIR's IR
1335  // structure. It does not directly map to the incoming parent block for the
1336  // OpPhi instructions at SPIR-V binary level. This is because structured
1337  // control flow ops are serialized to multiple SPIR-V blocks. If there is a
1338  // spirv.mlir.selection/spirv.mlir.loop op in the MLIR predecessor block,
1339  // the branch op jumping to the OpPhi's block then resides in the previous
1340  // structured control flow op's merge block.
1341  Block *spirvPredecessor = getPhiIncomingBlock(mlirPredecessor);
1342  LLVM_DEBUG(llvm::dbgs() << " spirv predecessor ");
1343  LLVM_DEBUG(printBlock(spirvPredecessor, llvm::dbgs()));
1344  if (auto branchOp = dyn_cast<spirv::BranchOp>(terminator)) {
1345  predecessors.emplace_back(spirvPredecessor, branchOp.getOperands());
1346  } else if (auto branchCondOp =
1347  dyn_cast<spirv::BranchConditionalOp>(terminator)) {
1348  std::optional<OperandRange> blockOperands;
1349  if (branchCondOp.getTrueTarget() == block) {
1350  blockOperands = branchCondOp.getTrueTargetOperands();
1351  } else {
1352  assert(branchCondOp.getFalseTarget() == block);
1353  blockOperands = branchCondOp.getFalseTargetOperands();
1354  }
1355 
1356  assert(!blockOperands->empty() &&
1357  "expected non-empty block operand range");
1358  predecessors.emplace_back(spirvPredecessor, *blockOperands);
1359  } else {
1360  return terminator->emitError("unimplemented terminator for Phi creation");
1361  }
1362  LLVM_DEBUG({
1363  llvm::dbgs() << " block arguments:\n";
1364  for (Value v : predecessors.back().second)
1365  llvm::dbgs() << " " << v << "\n";
1366  });
1367  }
1368 
1369  // Then create OpPhi instruction for each of the block argument.
1370  for (auto argIndex : llvm::seq<unsigned>(0, block->getNumArguments())) {
1371  BlockArgument arg = block->getArgument(argIndex);
1372 
1373  // Get the type <id> and result <id> for this OpPhi instruction.
1374  uint32_t phiTypeID = 0;
1375  if (failed(processType(arg.getLoc(), arg.getType(), phiTypeID)))
1376  return failure();
1377  uint32_t phiID = getNextID();
1378 
1379  LLVM_DEBUG(llvm::dbgs() << "[phi] for block argument #" << argIndex << ' '
1380  << arg << " (id = " << phiID << ")\n");
1381 
1382  // Prepare the (value <id>, parent block <id>) pairs.
1383  SmallVector<uint32_t, 8> phiArgs;
1384  phiArgs.push_back(phiTypeID);
1385  phiArgs.push_back(phiID);
1386 
1387  for (auto predIndex : llvm::seq<unsigned>(0, predecessors.size())) {
1388  Value value = predecessors[predIndex].second[argIndex];
1389  uint32_t predBlockId = getOrCreateBlockID(predecessors[predIndex].first);
1390  LLVM_DEBUG(llvm::dbgs() << "[phi] use predecessor (id = " << predBlockId
1391  << ") value " << value << ' ');
1392  // Each pair is a value <id> ...
1393  uint32_t valueId = getValueID(value);
1394  if (valueId == 0) {
1395  // The op generating this value hasn't been visited yet so we don't have
1396  // an <id> assigned yet. Record this to fix up later.
1397  LLVM_DEBUG(llvm::dbgs() << "(need to fix)\n");
1398  deferredPhiValues[value].push_back(functionBody.size() + 1 +
1399  phiArgs.size());
1400  } else {
1401  LLVM_DEBUG(llvm::dbgs() << "(id = " << valueId << ")\n");
1402  }
1403  phiArgs.push_back(valueId);
1404  // ... and a parent block <id>.
1405  phiArgs.push_back(predBlockId);
1406  }
1407 
1408  encodeInstructionInto(functionBody, spirv::Opcode::OpPhi, phiArgs);
1409  valueIDMap[arg] = phiID;
1410  }
1411 
1412  return success();
1413 }

1420  Operation *op, StringRef extensionSetName, uint32_t extensionOpcode,
1421  ArrayRef<uint32_t> operands) {

1430  importOperands);

1443  extInstOperands.append(std::next(operands.begin(), 2), operands.end());
1444  encodeInstructionInto(functionBody, spirv::Opcode::OpExtInst,
1445  extInstOperands);
1446  return success();
