24 WarpExecuteOnLane0Op
25 WarpDistributionPattern::moveRegionToNewWarpOpAndReplaceReturns(
26  RewriterBase &rewriter, WarpExecuteOnLane0Op warpOp,
27  ValueRange newYieldedValues, TypeRange newReturnTypes) const {
28  // Create a new op before the existing one, with the extra operands.
29  OpBuilder::InsertionGuard g(rewriter);
30  rewriter.setInsertionPoint(warpOp);
31  auto newWarpOp = WarpExecuteOnLane0Op::create(
32  rewriter, warpOp.getLoc(), newReturnTypes, warpOp.getLaneid(),
33  warpOp.getWarpSize(), warpOp.getArgs(),
34  warpOp.getBody()->getArgumentTypes());
35 
36  Region &opBody = warpOp.getBodyRegion();
37  Region &newOpBody = newWarpOp.getBodyRegion();
38  Block &newOpFirstBlock = newOpBody.front();
39  rewriter.inlineRegionBefore(opBody, newOpBody, newOpBody.begin());
40  rewriter.eraseBlock(&newOpFirstBlock);
41  assert(newWarpOp.getWarpRegion().hasOneBlock() &&
42  "expected WarpOp with single block");
43 
44  auto yield =
45  cast<gpu::YieldOp>(newOpBody.getBlocks().begin()->getTerminator());
46 
47  rewriter.modifyOpInPlace(
48  yield, [&]() { yield.getValuesMutable().assign(newYieldedValues); });
49  return newWarpOp;
50 }

52 WarpExecuteOnLane0Op
53 WarpDistributionPattern::moveRegionToNewWarpOpAndAppendReturns(
54  RewriterBase &rewriter, WarpExecuteOnLane0Op warpOp,
55  ValueRange newYieldedValues, TypeRange newReturnTypes,
56  SmallVector<size_t> &indices) const {
57  SmallVector<Type> types(warpOp.getResultTypes().begin(),
58  warpOp.getResultTypes().end());
59  gpu::YieldOp yield = warpOp.getTerminator();
60  SmallVector<Value> yieldValues(yield.getOperands().begin(),
61  yield.getOperands().end());
62  llvm::SmallDenseMap<Value, unsigned> indexLookup;
63  // Record the value -> first index mapping for faster lookup.
64  for (auto [i, v] : llvm::enumerate(yieldValues)) {
65  if (!indexLookup.count(v))
66  indexLookup[v] = i;
67  }
68 
69  for (auto [value, type] : llvm::zip_equal(newYieldedValues, newReturnTypes)) {
70  // If the value already exists in the yield, don't create a new output.
71  if (indexLookup.count(value)) {
72  indices.push_back(indexLookup[value]);
73  } else {
74  // If the value is new, add it to the yield and to the types.
75  yieldValues.push_back(value);
76  types.push_back(type);
77  indices.push_back(yieldValues.size() - 1);
78  }
79  }
80 
81  WarpExecuteOnLane0Op newWarpOp = moveRegionToNewWarpOpAndReplaceReturns(
82  rewriter, warpOp, yieldValues, types);
83  rewriter.replaceOp(warpOp,
84  newWarpOp.getResults().take_front(warpOp.getNumResults()));
85  return newWarpOp;
86 }

88 OpOperand *WarpDistributionPattern::getWarpResult(
89  WarpExecuteOnLane0Op warpOp,
90  llvm::function_ref<bool(Operation *)> fn) const {
91  gpu::YieldOp yield = warpOp.getTerminator();
92  for (OpOperand &yieldOperand : yield->getOpOperands()) {
93  Value yieldValues = yieldOperand.get();
94  Operation *definedOp = yieldValues.getDefiningOp();
95  if (definedOp && fn(definedOp)) {
96  if (!warpOp.getResult(yieldOperand.getOperandNumber()).use_empty())
97  return &yieldOperand;
98  }
99  }
100  return nullptr;
101 }

103 bool WarpDistributionPattern::delinearizeLaneId(
104  OpBuilder &builder, Location loc, ArrayRef<int64_t> originalShape,
105  ArrayRef<int64_t> distributedShape, int64_t warpSize, Value laneId,
106  SmallVectorImpl<Value> &delinearizedIds) const {
107  // If the original shape and the distributed shape is the same, we don't
108  // distribute at all--every thread is handling the whole. For such case, we
109  // should not rely on lane IDs later. So just return an empty lane ID vector.
110  if (originalShape == distributedShape) {
111  delinearizedIds.clear();
112  return true;
113  }
114 
115  SmallVector<int64_t> sizes;
116  for (auto [large, small] : llvm::zip_equal(originalShape, distributedShape)) {
117  if (large % small != 0)
118  return false;
119  sizes.push_back(large / small);
120  }
121  if (std::accumulate(sizes.begin(), sizes.end(), 1,
122  std::multiplies<int64_t>()) != warpSize)
123  return false;
124 
125  AffineExpr s0, s1;
126  bindSymbols(builder.getContext(), s0, s1);
127 
128  int64_t usedThreads = 1;
129 
130  Value zero = arith::ConstantIndexOp::create(builder, loc, 0);
131  delinearizedIds.assign(sizes.size(), zero);
132 
133  for (int i = sizes.size() - 1; i >= 0; --i) {
134  usedThreads *= sizes[i];
135  if (usedThreads == warpSize) {
136  // We've used up all available threads. Don't need to perform modulo
137  // anymore. And we can stop the calculation for further dimensions.
138  delinearizedIds[i] = laneId;
139  break;
140  }
141  delinearizedIds[i] =
142  affine::makeComposedAffineApply(builder, loc, s0 % sizes[i], {laneId});
143  laneId = affine::makeComposedAffineApply(
144  builder, loc, s0.floorDiv(usedThreads), {laneId});
145  }
146  return true;
147 }
