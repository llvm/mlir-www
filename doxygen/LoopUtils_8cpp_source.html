37 /// Computes the cleanup loop lower bound of the loop being unrolled with
38 /// the specified unroll factor; this bound will also be upper bound of the main
39 /// part of the unrolled loop. Computes the bound as an AffineMap with its
40 /// operands or a null map when the trip count can't be expressed as an affine
41 /// expression.
42 static void
43 getCleanupLoopLowerBound(AffineForOp forOp, unsigned unrollFactor,
44  AffineMap &cleanupLbMap,
45  SmallVectorImpl<Value> &cleanupLbOperands) {
46  AffineMap tripCountMap;
47  SmallVector<Value, 4> tripCountOperands;
48  getTripCountMapAndOperands(forOp, &tripCountMap, &tripCountOperands);
49  // Trip count can't be computed.
50  if (!tripCountMap) {
51  cleanupLbMap = AffineMap();
52  return;
53  }
54 
55  OpBuilder b(forOp);
56  auto lbMap = forOp.getLowerBoundMap();
57  auto lb = AffineApplyOp::create(b, forOp.getLoc(), lbMap,
58  forOp.getLowerBoundOperands());
59 
60  // For each upper bound expr, get the range.
61  // Eg: affine.for %i = lb to min (ub1, ub2),
62  // where tripCountExprs yield (tr1, tr2), we create affine.apply's:
63  // lb + tr1 - tr1 % ufactor, lb + tr2 - tr2 % ufactor; the results of all
64  // these affine.apply's make up the cleanup loop lower bound.
65  SmallVector<AffineExpr, 4> bumpExprs(tripCountMap.getNumResults());
66  SmallVector<Value, 4> bumpValues(tripCountMap.getNumResults());
67  int64_t step = forOp.getStepAsInt();
68  for (unsigned i = 0, e = tripCountMap.getNumResults(); i < e; i++) {
69  auto tripCountExpr = tripCountMap.getResult(i);
70  bumpExprs[i] = (tripCountExpr - tripCountExpr % unrollFactor) * step;
71  auto bumpMap = AffineMap::get(tripCountMap.getNumDims(),
72  tripCountMap.getNumSymbols(), bumpExprs[i]);
73  bumpValues[i] =
74  AffineApplyOp::create(b, forOp.getLoc(), bumpMap, tripCountOperands);
75  }
76 
77  SmallVector<AffineExpr, 4> newUbExprs(tripCountMap.getNumResults());
78  for (unsigned i = 0, e = bumpExprs.size(); i < e; i++)
79  newUbExprs[i] = b.getAffineDimExpr(0) + b.getAffineDimExpr(i + 1);
80 
81  cleanupLbOperands.clear();
82  cleanupLbOperands.push_back(lb);
83  cleanupLbOperands.append(bumpValues.begin(), bumpValues.end());
84  cleanupLbMap = AffineMap::get(1 + tripCountMap.getNumResults(), 0, newUbExprs,
85  b.getContext());
86  // Simplify the cleanupLbMap + cleanupLbOperands.
87  fullyComposeAffineMapAndOperands(&cleanupLbMap, &cleanupLbOperands);
88  cleanupLbMap = simplifyAffineMap(cleanupLbMap);
89  canonicalizeMapAndOperands(&cleanupLbMap, &cleanupLbOperands);
90  // Remove any affine.apply's that became dead from the simplification above.
91  for (auto v : bumpValues)
92  if (v.use_empty())
93  v.getDefiningOp()->erase();
94 
95  if (lb.use_empty())
96  lb.erase();
97 }

99 /// Helper to replace uses of loop carried values (iter_args) and loop
100 /// yield values while promoting single iteration affine.for ops.
101 static void replaceIterArgsAndYieldResults(AffineForOp forOp) {
102  // Replace uses of iter arguments with iter operands (initial values).
103  auto iterOperands = forOp.getInits();
104  auto iterArgs = forOp.getRegionIterArgs();
105  for (auto e : llvm::zip(iterOperands, iterArgs))
106  std::get<1>(e).replaceAllUsesWith(std::get<0>(e));
107 
108  // Replace uses of loop results with the values yielded by the loop.
109  auto outerResults = forOp.getResults();
110  auto innerResults = forOp.getBody()->getTerminator()->getOperands();
111  for (auto e : llvm::zip(outerResults, innerResults))
112  std::get<0>(e).replaceAllUsesWith(std::get<1>(e));

115 /// Promotes the loop body of a forOp to its containing block if the forOp
116 /// was known to have a single iteration.
117 LogicalResult mlir::affine::promoteIfSingleIteration(AffineForOp forOp) {
118  std::optional<uint64_t> tripCount = getConstantTripCount(forOp);
119  if (!tripCount || *tripCount != 1)
120  return failure();
121 
122  // TODO: extend this for arbitrary affine bounds.
123  if (forOp.getLowerBoundMap().getNumResults() != 1)
124  return failure();
125 
126  // Replaces all IV uses to its single iteration value.
127  auto iv = forOp.getInductionVar();
128  auto *parentBlock = forOp->getBlock();
129  if (!iv.use_empty()) {
130  if (forOp.hasConstantLowerBound()) {
131  auto func = forOp->getParentOfType<FunctionOpInterface>();
132  OpBuilder builder(forOp->getContext());
133  if (func)
134  builder.setInsertionPointToStart(&func.getFunctionBody().front());
135  else
136  builder.setInsertionPoint(forOp);
137  auto constOp = arith::ConstantIndexOp::create(
138  builder, forOp.getLoc(), forOp.getConstantLowerBound());
139  iv.replaceAllUsesWith(constOp);
140  } else {
141  auto lbOperands = forOp.getLowerBoundOperands();
142  auto lbMap = forOp.getLowerBoundMap();
143  OpBuilder builder(forOp);
144  if (lbMap == builder.getDimIdentityMap()) {
145  // No need of generating an affine.apply.
146  iv.replaceAllUsesWith(lbOperands[0]);
147  } else {
148  auto affineApplyOp =
149  AffineApplyOp::create(builder, forOp.getLoc(), lbMap, lbOperands);
150  iv.replaceAllUsesWith(affineApplyOp);
151  }
152  }
153  }
154 
155  replaceIterArgsAndYieldResults(forOp);
156 
157  // Move the loop body operations, except for its terminator, to the loop's
158  // containing block.
159  forOp.getBody()->back().erase();
160  parentBlock->getOperations().splice(Block::iterator(forOp),
161  forOp.getBody()->getOperations());
162  forOp.erase();
163  return success();
164 }

166 /// Generates an affine.for op with the specified lower and upper bounds
167 /// while generating the right IV remappings to realize shifts for operations in
168 /// its body. The operations that go into the loop body are specified in
169 /// opGroupQueue starting from the specified offset, and in that order. The
170 /// first element of the pair specifies the shift applied to that group of
171 /// operations; the shift is multiplied by the loop step before being applied.
172 /// Returns nullptr if the generated loop simplifies to a single iteration one.
173 static AffineForOp generateShiftedLoop(
174  AffineMap lbMap, AffineMap ubMap,
175  const std::vector<std::pair<uint64_t, ArrayRef<Operation *>>> &opGroupQueue,
176  unsigned offset, AffineForOp srcForOp, OpBuilder b) {
177  auto lbOperands = srcForOp.getLowerBoundOperands();
178  auto ubOperands = srcForOp.getUpperBoundOperands();
179 
180  assert(lbMap.getNumInputs() == lbOperands.size());
181  assert(ubMap.getNumInputs() == ubOperands.size());
182 
183  auto loopChunk =
184  AffineForOp::create(b, srcForOp.getLoc(), lbOperands, lbMap, ubOperands,
185  ubMap, srcForOp.getStepAsInt());
186  auto loopChunkIV = loopChunk.getInductionVar();
187  auto srcIV = srcForOp.getInductionVar();
188 
189  IRMapping operandMap;
190 
191  auto bodyBuilder = OpBuilder::atBlockTerminator(loopChunk.getBody());
192  for (const auto &it : llvm::drop_begin(opGroupQueue, offset)) {
193  uint64_t shift = it.first;
194  auto ops = it.second;
195  // All 'same shift' operations get added with their operands being
196  // remapped to results of cloned operations, and their IV used remapped.
197  // Generate the remapping if the shift is not zero: remappedIV = newIV -
198  // shift.
199  if (!srcIV.use_empty() && shift != 0) {
200  auto ivRemap = AffineApplyOp::create(
201  bodyBuilder, srcForOp.getLoc(),
202  bodyBuilder.getSingleDimShiftAffineMap(
203  -static_cast<int64_t>(srcForOp.getStepAsInt() * shift)),
204  loopChunkIV);
205  operandMap.map(srcIV, ivRemap);
206  } else {
207  operandMap.map(srcIV, loopChunkIV);
208  }
209  for (auto *op : ops)
210  bodyBuilder.clone(*op, operandMap);
211  };
212  if (succeeded(promoteIfSingleIteration(loopChunk)))
213  return AffineForOp();
214  return loopChunk;
215 }

217 // The skewing of operations with respect to one another can be used for
218 // example to allow overlap of asynchronous operations (such as DMA
219 // communication) with computation, or just relative shifting of operations
220 // for better register reuse, locality or parallelism. As such, the shifts are
221 // typically expected to be at most of the order of the number of operations.
222 // This method should not be used as a substitute for loop distribution/fission.
223 // This method uses an algorithm// in time linear in the number of operations
224 // in the body of the for loop - (using the 'sweep line' paradigm). This method
225 // asserts preservation of SSA dominance. A check for that as well as that for
226 // memory-based dependence preservation check rests with the users of this
227 // method.
228 LogicalResult mlir::affine::affineForOpBodySkew(AffineForOp forOp,
229  ArrayRef<uint64_t> shifts,
230  bool unrollPrologueEpilogue) {
231  assert(forOp.getBody()->getOperations().size() == shifts.size() &&
232  "too few/many shifts");
233  if (forOp.getBody()->begin() == std::prev(forOp.getBody()->end()))
234  return success();
235 
236  // If the trip counts aren't constant, we would need versioning and
237  // conditional guards (or context information to prevent such versioning). The
238  // better way to pipeline for such loops is to first tile them and extract
239  // constant trip count "full tiles" before applying this.
240  auto mayBeConstTripCount = getConstantTripCount(forOp);
241  if (!mayBeConstTripCount) {
242  LLVM_DEBUG(forOp.emitRemark("non-constant trip count loop not handled"));
243  return success();
244  }
245  uint64_t tripCount = *mayBeConstTripCount;
246 
247  assert(isOpwiseShiftValid(forOp, shifts) &&
248  "shifts will lead to an invalid transformation\n");
249 
250  int64_t step = forOp.getStepAsInt();
251 
252  unsigned numChildOps = shifts.size();
253 
254  // Do a linear time (counting) sort for the shifts.
255  uint64_t maxShift = *llvm::max_element(shifts);
256  if (maxShift >= numChildOps) {
257  // Large shifts are not the typical use case.
258  forOp.emitWarning("not shifting because shifts are unrealistically large");
259  return success();
260  }
261 
262  // An array of operation groups sorted by shift amount; each group has all
263  // operations with the same shift in the order in which they appear in the
264  // body of the 'affine.for' op.
265  std::vector<std::vector<Operation *>> sortedOpGroups(maxShift + 1);
266  unsigned pos = 0;
267  for (auto &op : forOp.getBody()->without_terminator()) {
268  auto shift = shifts[pos++];
269  sortedOpGroups[shift].push_back(&op);
270  }
271 
272  // Unless the shifts have a specific pattern (which actually would be the
273  // common use case), prologue and epilogue are not meaningfully defined.
274  // Nevertheless, if 'unrollPrologueEpilogue' is set, we will treat the first
275  // loop generated as the prologue and the last as epilogue and unroll these
276  // fully.
277  AffineForOp prologue, epilogue;
278 
279  // Do a sweep over the sorted shifts while storing open groups in a
280  // vector, and generating loop portions as necessary during the sweep. A block
281  // of operations is paired with its shift.
282  std::vector<std::pair<uint64_t, ArrayRef<Operation *>>> opGroupQueue;
283 
284  auto origLbMap = forOp.getLowerBoundMap();
285  uint64_t lbShift = 0;
286  OpBuilder b(forOp);
287  for (uint64_t d = 0, e = sortedOpGroups.size(); d < e; ++d) {
288  // If nothing is shifted by d, continue.
289  if (sortedOpGroups[d].empty())
290  continue;
291  if (!opGroupQueue.empty()) {
292  assert(d > 0 &&
293  "Queue expected to be empty when the first block is found");
294  // The interval for which the loop needs to be generated here is:
295  // [lbShift, min(lbShift + tripCount, d)) and the body of the
296  // loop needs to have all operations in opQueue in that order.
297  AffineForOp res;
298  if (lbShift + tripCount * step < d * step) {
299  res = generateShiftedLoop(
300  b.getShiftedAffineMap(origLbMap, lbShift),
301  b.getShiftedAffineMap(origLbMap, lbShift + tripCount * step),
302  opGroupQueue, /*offset=*/0, forOp, b);
303  // Entire loop for the queued op groups generated, empty it.
304  opGroupQueue.clear();
305  lbShift += tripCount * step;
306  } else {
307  res = generateShiftedLoop(b.getShiftedAffineMap(origLbMap, lbShift),
308  b.getShiftedAffineMap(origLbMap, d),
309  opGroupQueue, /*offset=*/0, forOp, b);
310  lbShift = d * step;
311  }
312 
313  if (res) {
314  // Simplify/canonicalize the affine.for.
315  RewritePatternSet patterns(res.getContext());
316  AffineForOp::getCanonicalizationPatterns(patterns, res.getContext());
317  bool erased;
318  (void)applyOpPatternsGreedily(
319  res.getOperation(), std::move(patterns),
320  GreedyRewriteConfig().setStrictness(
321  GreedyRewriteStrictness::ExistingAndNewOps),
322  /*changed=*/nullptr, &erased);
323  if (!erased && !prologue)
324  prologue = res;
325  if (!erased)
326  epilogue = res;
327  }
328  } else {
329  // Start of first interval.
330  lbShift = d * step;
331  }
332  // Augment the list of operations that get into the current open interval.
333  opGroupQueue.emplace_back(d, sortedOpGroups[d]);
334  }
335 
336  // Those operations groups left in the queue now need to be processed (FIFO)
337  // and their loops completed.
338  for (unsigned i = 0, e = opGroupQueue.size(); i < e; ++i) {
339  uint64_t ubShift = (opGroupQueue[i].first + tripCount) * step;
340  epilogue = generateShiftedLoop(b.getShiftedAffineMap(origLbMap, lbShift),
341  b.getShiftedAffineMap(origLbMap, ubShift),
342  opGroupQueue, /*offset=*/i, forOp, b);
343  lbShift = ubShift;
344  if (!prologue)
345  prologue = epilogue;
346  }
347 
348  // Erase the original for op.
349  forOp.erase();
350 
351  if (unrollPrologueEpilogue && prologue)
352  (void)loopUnrollFull(prologue);
353  if (unrollPrologueEpilogue && !epilogue && epilogue != prologue)
354  (void)loopUnrollFull(epilogue);
355 
356  return success();
357 }

360 static LogicalResult
361 checkIfHyperRectangular(MutableArrayRef<AffineForOp> input) {
362  FlatAffineValueConstraints cst;
363  SmallVector<Operation *, 8> ops(input.begin(), input.end());
364  // 0-d or 1-d is trivially hyper-rectangular.
365  if (input.size() <= 1)
366  return success();
367  if (failed(getIndexSet(ops, &cst))) {
368  LLVM_DEBUG(llvm::dbgs() << "Index set computation failed!\n");
369  return failure();
370  }
371  if (!cst.isHyperRectangular(0, input.size())) {
372  LLVM_DEBUG(llvm::dbgs()
373  << "Non-hyperrectangular nests not supported for tiling!\n");
374  return failure();
375  }
376  return success();
377 }

421 /// Constructs tiled loop nest, without setting the loop bounds and move the
422 /// body of the original loop nest to the tiled loop nest.
423 static void constructTiledLoopNest(MutableArrayRef<AffineForOp> origLoops,
424  AffineForOp rootAffineForOp, unsigned width,
425  MutableArrayRef<AffineForOp> tiledLoops) {
426  Location loc = rootAffineForOp.getLoc();
427 
428  // The outermost among the loops as we add more..
429  Operation *topLoop = rootAffineForOp.getOperation();
430  AffineForOp innermostPointLoop;
431 
432  // Add intra-tile (or point) loops.
433  for (unsigned i = 0; i < width; i++) {
434  OpBuilder b(topLoop);
435  // Loop bounds will be set later.
436  AffineForOp pointLoop = AffineForOp::create(b, loc, 0, 0);
437  pointLoop.getBody()->getOperations().splice(
438  pointLoop.getBody()->begin(), topLoop->getBlock()->getOperations(),
439  topLoop);
440  tiledLoops[2 * width - 1 - i] = pointLoop;
441  topLoop = pointLoop.getOperation();
442  if (i == 0)
443  innermostPointLoop = pointLoop;
444  }
445 
446  // Add tile space loops;
447  for (unsigned i = width; i < 2 * width; i++) {
448  OpBuilder b(topLoop);
449  // Loop bounds will be set later.
450  AffineForOp tileSpaceLoop = AffineForOp::create(b, loc, 0, 0);
451  tileSpaceLoop.getBody()->getOperations().splice(
452  tileSpaceLoop.getBody()->begin(), topLoop->getBlock()->getOperations(),
453  topLoop);
454  tiledLoops[2 * width - i - 1] = tileSpaceLoop;
455  topLoop = tileSpaceLoop.getOperation();
456  }
457 
458  // Move the loop body of the original nest to the new one.
459  moveLoopBody(origLoops.back(), innermostPointLoop);
460 }

462 /// Set lower and upper bounds of intra-tile loops for parametric tiling.
463 // TODO: Handle non-constant lower bounds.
464 static void setIntraTileBoundsParametric(OpBuilder &b, AffineForOp origLoop,
465  AffineForOp newInterTileLoop,
466  AffineForOp newIntraTileLoop,
467  Value tileSize) {
468  // The lower bound for the intra-tile loop is represented by an affine map
469  // as (%i, %t0)->((%i - %origlb) * %t0 + %origlb). Similarly, the upper bound
470  // for the intra-tile loop is represented by an affine map as (%i, %t0)->((%i
471  // - %origlb) * %t0) + (%t0 * %origLoopStep) + %origlb), where %i is loop IV
472  // of the corresponding inter-tile loop, %t0 is the corresponding tiling
473  // parameter, %origlb is lower bound and %origLoopStep is the loop step of the
474  // corresponding inter-tile loop.
475 
476  assert(origLoop.hasConstantLowerBound() &&
477  "expected input loops to have constant lower bound.");
478 
479  // Get lower bound of original loop as an affine expression.
480  AffineExpr origLowerBoundExpr;
481  origLowerBoundExpr =
482  b.getAffineConstantExpr(origLoop.getConstantLowerBound());
483 
484  // Add dim operands from original lower/upper bound.
485  SmallVector<Value, 4> lbOperands, ubOperands;
486  AffineBound lb = origLoop.getLowerBound();
487  AffineBound ub = origLoop.getUpperBound();
488  lbOperands.reserve(lb.getNumOperands() + 2);
489  ubOperands.reserve(ub.getNumOperands() + 2);
490  AffineMap origLbMap = lb.getMap();
491  AffineMap origUbMap = ub.getMap();
492  for (unsigned j = 0, e = origLbMap.getNumDims(); j < e; ++j)
493  lbOperands.push_back(lb.getOperand(j));
494  for (unsigned j = 0, e = origUbMap.getNumDims(); j < e; ++j)
495  ubOperands.push_back(ub.getOperand(j));
496 
497  // Add a new dim operand in lb/ubOperands corresponding to the origLoop
498  // IV.
499  lbOperands.push_back(newInterTileLoop.getInductionVar());
500  ubOperands.push_back(newInterTileLoop.getInductionVar());
501 
502  // Get loop IV as an affine expression for lower/upper bound. Size of
503  // lb/ubOperands is guaranteed to be atleast one.
504  AffineExpr lbLoopIvExpr = b.getAffineDimExpr(lbOperands.size() - 1);
505  AffineExpr ubLoopIvExpr = b.getAffineDimExpr(ubOperands.size() - 1);
506 
507  // Add symbol operands from original lower/upper bound.
508  for (unsigned j = 0, e = origLbMap.getNumSymbols(); j < e; ++j)
509  lbOperands.push_back(lb.getOperand(origLbMap.getNumDims() + j));
510  for (unsigned j = 0, e = origUbMap.getNumSymbols(); j < e; ++j)
511  ubOperands.push_back(ub.getOperand(origUbMap.getNumDims() + j));
512 
513  // Add a new symbol operand which is the tile size for this loop.
514  lbOperands.push_back(tileSize);
515  ubOperands.push_back(tileSize);
516 
517  SmallVector<AffineExpr, 4> lbBoundExprs;
518  SmallVector<AffineExpr, 4> ubBoundExprs;
519  lbBoundExprs.reserve(origLbMap.getNumResults());
520  ubBoundExprs.reserve(origUbMap.getNumResults());
521 
522  // Get tiling parameter as an affine expression for lb/ub.
523  AffineExpr lbTileParameter = b.getAffineSymbolExpr(origLbMap.getNumSymbols());
524  AffineExpr ubTileParameter = b.getAffineSymbolExpr(origUbMap.getNumSymbols());
525 
526  // Insert lb as inter-tile ((loop IV - origlb) * tilingParameter) + origlb.
527  lbBoundExprs.push_back(
528  ((lbLoopIvExpr - origLowerBoundExpr) * lbTileParameter) +
529  origLowerBoundExpr);
530 
531  // Get the origLoopStep as an affine expression.
532  AffineExpr origLoopStep = b.getAffineConstantExpr(origLoop.getStepAsInt());
533 
534  // Insert ub as inter-tile ((loop IV - origlb) * tilingParameter) +
535  // (tilingParameter * origLoopStep) + origlb.
536  ubBoundExprs.push_back(
537  ((ubLoopIvExpr - origLowerBoundExpr) * ubTileParameter) +
538  (ubTileParameter * origLoopStep) + origLowerBoundExpr);
539 
540  ubBoundExprs.append(origUbMap.getResults().begin(),
541  origUbMap.getResults().end());
542 
543  AffineMap lbMap =
544  AffineMap::get(origLbMap.getNumDims() + 1, origLbMap.getNumSymbols() + 1,
545  lbBoundExprs, b.getContext());

550  ubBoundExprs, b.getContext());

557 /// Set lower and upper bounds of inter-tile loops for parametric tiling.
558 // TODO: Handle non-constant lower bounds.
559 static void setInterTileBoundsParametric(OpBuilder &b, AffineForOp origLoop,
560  AffineForOp newLoop, Value tileSize) {
561  OperandRange newLbOperands = origLoop.getLowerBoundOperands();
562 
563  // The lower bounds for inter-tile loops are same as the corresponding lower
564  // bounds of original loops.
565  newLoop.setLowerBound(newLbOperands, origLoop.getLowerBoundMap());
566 
567  // The new upper bound map for inter-tile loops, assuming constant lower
568  // bounds, are now originalLowerBound + ceildiv((originalUpperBound -
569  // originalLowerBound), tiling parameter); where tiling parameter is the
570  // respective tile size for that loop. For e.g. if the original ubmap was
571  // ()->(1024), the new map will be
572  // ()[s0]->(ceildiv((1024 -lb) % s0)), where s0 is the tiling parameter.
573  // Therefore a new symbol operand is inserted in the map and the result
574  // expression is overwritten.
575 
576  assert(origLoop.hasConstantLowerBound() &&
577  "expected input loops to have constant lower bound.");
578 
579  // Get lower bound of original loop as an affine expression.
580  AffineExpr origLowerBoundExpr;
581  origLowerBoundExpr =
582  b.getAffineConstantExpr(origLoop.getConstantLowerBound());
583 
584  // Add dim operands from original upper bound.
585  SmallVector<Value, 4> ubOperands;
586  AffineBound ub = origLoop.getUpperBound();
587  ubOperands.reserve(ub.getNumOperands() + 1);
588  AffineMap origUbMap = ub.getMap();
589  for (unsigned j = 0, e = origUbMap.getNumDims(); j < e; ++j)
590  ubOperands.push_back(ub.getOperand(j));
591 
592  // Add symbol operands from original upper bound.
593  for (unsigned j = 0, e = origUbMap.getNumSymbols(); j < e; ++j)
594  ubOperands.push_back(ub.getOperand(origUbMap.getNumDims() + j));
595 
596  // Add a new symbol operand which is the tile size for this loop.
597  ubOperands.push_back(tileSize);
598 
599  // Get tiling parameter as an affine expression.
600  AffineExpr tileParameter = b.getAffineSymbolExpr(origUbMap.getNumSymbols());
601 
602  SmallVector<AffineExpr, 4> boundExprs;
603  boundExprs.reserve(origUbMap.getNumResults());
604  int64_t origUpperBound;
605  AffineExpr origUpperBoundExpr;
606 
607  // If upper bound for the original loop is constant, then the constant can
608  // be obtained as an affine expression straight away.
609  if (origLoop.hasConstantUpperBound()) {
610  origUpperBound = origLoop.getConstantUpperBound();
611 
612  // Get original constant upper bound as an affine expression.
613  origUpperBoundExpr = b.getAffineConstantExpr(origUpperBound);
614 
615  // Insert the bound as originalLowerBoundceildiv((originalUpperBound -
616  // originalLowerBound), tilingParameter).
617  boundExprs.push_back(
618  origLowerBoundExpr +
619  (origUpperBoundExpr - origLowerBoundExpr).ceilDiv(tileParameter));
620  } else {
621  // If upper bound for the original loop is not constant then two cases
622  // are possible, although there handeling is the same, 1.) The result of
623  // ubmap has only one result expression. For e.g.
624  // affine.for %i = 5 to %ub
625  //
626  // A symbol operand is added which represents the tiling parameter. The
627  // new loop bounds here will be like ()[s0, s1] -> ((s0 - 5) ceildiv s1 + 5)
628  // where 's0' is the original upper bound and 's1' is the tiling
629  // parameter. 2.) When ubMap has more than one result expression. For e.g.
630  // #map0 = affine_map<()[s0, s1] -> (s0, s1)
631  // affine.for %i = 5 to min #map0()[%s0, %s1]
632  //
633  // A symbol operand is added which represents the tiling parameter. The
634  // new loop bounds will be like ()[s0, s1, s2] -> ((s0 - 5) ceildiv s2 + 5,
635  // (s1 -5) ceildiv s2 + 5), where s2 is the tiling parameter.
636 
637  // Insert the bounds as originalLowerBound + ceildiv((originalUpperBound -
638  // originalLowerBound), tilingParameter).
639  for (AffineExpr origUpperBoundExpr : origUbMap.getResults())
640  boundExprs.push_back(
641  origLowerBoundExpr +
642  (origUpperBoundExpr - origLowerBoundExpr).ceilDiv(tileParameter));
643  }
644 
645  AffineMap ubMap =
646  AffineMap::get(origUbMap.getNumDims(), origUbMap.getNumSymbols() + 1,
647  boundExprs, b.getContext());
648  newLoop.setUpperBound(ubOperands, ubMap);
649 
650  // Original loop step must be preserved.
651  newLoop.setStep(origLoop.getStepAsInt());
652 }

654 /// Constructs and sets new loop bounds after tiling for the case of
655 /// hyper-rectangular index sets, where the bounds of one dimension do not
656 /// depend on other dimensions and tiling parameters are captured from SSA
657 /// values. Bounds of each dimension can thus be treated independently,
658 /// and deriving the new bounds is much simpler and faster than for the case of
659 /// tiling arbitrary polyhedral shapes.
660 static void constructParametricallyTiledIndexSetHyperRect(

662  MutableArrayRef<AffineForOp> newLoops, ArrayRef<Value> tileSizes) {
663  assert(!origLoops.empty() && "expected atleast one loop in band");

682 /// Constructs and sets new loop bounds after tiling for the case of
683 /// hyper-rectangular index sets, where the bounds of one dimension do not
684 /// depend on other dimensions. Bounds of each dimension can thus be treated
685 /// independently, and deriving the new bounds is much simpler and faster
686 /// than for the case of tiling arbitrary polyhedral shapes.
687 static void
688 constructTiledIndexSetHyperRect(MutableArrayRef<AffineForOp> origLoops,
689  MutableArrayRef<AffineForOp> newLoops,
690  ArrayRef<unsigned> tileSizes) {
691  assert(!origLoops.empty());
692  assert(origLoops.size() == tileSizes.size());
693 
694  OpBuilder b(origLoops[0].getOperation());
695  unsigned width = origLoops.size();
696 
697  // Bounds for tile space loops.
698  for (unsigned i = 0; i < width; i++) {
699  OperandRange newLbOperands = origLoops[i].getLowerBoundOperands();
700  OperandRange newUbOperands = origLoops[i].getUpperBoundOperands();
701  newLoops[i].setLowerBound(newLbOperands, origLoops[i].getLowerBoundMap());
702  newLoops[i].setUpperBound(newUbOperands, origLoops[i].getUpperBoundMap());
703  // If the step size of original loop is x and tileSize is y then after
704  // tiling the tile space loops' step size becomes x*y.
705  newLoops[i].setStep(tileSizes[i] * origLoops[i].getStepAsInt());
706  }
707  // Bounds for intra-tile loops.
708  for (unsigned i = 0; i < width; i++) {
709  int64_t largestDiv = getLargestDivisorOfTripCount(origLoops[i]);
710  std::optional<uint64_t> mayBeConstantCount =
711  getConstantTripCount(origLoops[i]);
712  // The lower bound is just the tile-space loop.
713  AffineMap lbMap = b.getDimIdentityMap();
714  newLoops[width + i].setLowerBound(
715  /*operands=*/newLoops[i].getInductionVar(), lbMap);
716  // The step sizes of intra-tile loops is just the original loops' step size.
717  newLoops[width + i].setStep(origLoops[i].getStepAsInt());
718 
719  // Set the upper bound.
720  if (mayBeConstantCount && *mayBeConstantCount < tileSizes[i]) {
721  // Trip count is less than the tile size: upper bound is lower bound +
722  // trip count * stepSize.
723  AffineMap ubMap = b.getSingleDimShiftAffineMap(
724  *mayBeConstantCount * origLoops[i].getStepAsInt());
725  newLoops[width + i].setUpperBound(
726  /*operands=*/newLoops[i].getInductionVar(), ubMap);
727  } else if (largestDiv % tileSizes[i] != 0) {
728  // Intra-tile loop ii goes from i to min(i + tileSize * stepSize, ub_i).
729  // Construct the upper bound map; the operands are the original operands
730  // with 'i' (tile-space loop) appended to it. The new upper bound map is
731  // the original one with an additional expression i + tileSize * stepSize
732  // appended.
733 
734  // Add dim operands from original upper bound.
735  SmallVector<Value, 4> ubOperands;
736  AffineBound ub = origLoops[i].getUpperBound();
737  ubOperands.reserve(ub.getNumOperands() + 1);
738  AffineMap origUbMap = ub.getMap();
739  for (unsigned j = 0, e = origUbMap.getNumDims(); j < e; ++j)
740  ubOperands.push_back(ub.getOperand(j));
741 
742  // Add dim operand for new loop upper bound.
743  ubOperands.push_back(newLoops[i].getInductionVar());
744 
745  // Add symbol operands from original upper bound.
746  for (unsigned j = 0, e = origUbMap.getNumSymbols(); j < e; ++j)
747  ubOperands.push_back(ub.getOperand(origUbMap.getNumDims() + j));
748 
749  SmallVector<AffineExpr, 4> boundExprs;
750  boundExprs.reserve(1 + origUbMap.getNumResults());
751  AffineExpr dim = b.getAffineDimExpr(origUbMap.getNumDims());
752  // The new upper bound map is the original one with an additional
753  // expression i + tileSize * stepSize (of original loop) appended.
754  boundExprs.push_back(dim + tileSizes[i] * origLoops[i].getStepAsInt());
755  boundExprs.append(origUbMap.getResults().begin(),
756  origUbMap.getResults().end());
757  AffineMap ubMap =
758  AffineMap::get(origUbMap.getNumDims() + 1, origUbMap.getNumSymbols(),
759  boundExprs, b.getContext());
760  newLoops[width + i].setUpperBound(/*operands=*/ubOperands, ubMap);
761  } else {
762  // No need of the min expression.
763  AffineExpr dim = b.getAffineDimExpr(0);
764  AffineMap ubMap = AffineMap::get(
765  1, 0, dim + tileSizes[i] * origLoops[i].getStepAsInt());
766  newLoops[width + i].setUpperBound(newLoops[i].getInductionVar(), ubMap);
767  }
768  }
769 }

895 /// Generates unrolled copies of AffineForOp 'loopBodyBlock', with associated
896 /// 'forOpIV' by 'unrollFactor', calling 'ivRemapFn' to remap 'forOpIV' for each
897 /// unrolled body. If specified, annotates the Ops in each unrolled iteration
898 /// using annotateFn.
899 static void generateUnrolledLoop(
900  Block *loopBodyBlock, Value forOpIV, uint64_t unrollFactor,
901  function_ref<Value(unsigned, Value, OpBuilder)> ivRemapFn,
902  function_ref<void(unsigned, Operation *, OpBuilder)> annotateFn,
903  ValueRange iterArgs, ValueRange yieldedValues) {
904  // Builder to insert unrolled bodies just before the terminator of the body of
905  // 'forOp'.
906  auto builder = OpBuilder::atBlockTerminator(loopBodyBlock);
907 
908  constexpr auto defaultAnnotateFn = [](unsigned, Operation *, OpBuilder) {};
909  if (!annotateFn)
910  annotateFn = defaultAnnotateFn;
911 
912  // Keep a pointer to the last non-terminator operation in the original block
913  // so that we know what to clone (since we are doing this in-place).
914  Block::iterator srcBlockEnd = std::prev(loopBodyBlock->end(), 2);
915 
916  // Unroll the contents of 'forOp' (append unrollFactor - 1 additional copies).
917  SmallVector<Value, 4> lastYielded(yieldedValues);
918 
919  for (unsigned i = 1; i < unrollFactor; i++) {
920  IRMapping operandMap;
921 
922  // Prepare operand map.
923  operandMap.map(iterArgs, lastYielded);
924 
925  // If the induction variable is used, create a remapping to the value for
926  // this unrolled instance.
927  if (!forOpIV.use_empty()) {
928  Value ivUnroll = ivRemapFn(i, forOpIV, builder);
929  operandMap.map(forOpIV, ivUnroll);
930  }
931 
932  // Clone the original body of 'forOp'.
933  for (auto it = loopBodyBlock->begin(); it != std::next(srcBlockEnd); it++) {
934  Operation *clonedOp = builder.clone(*it, operandMap);
935  annotateFn(i, clonedOp, builder);
936  }
937 
938  // Update yielded values. If the yielded value is defined outside the
939  // `loopBodyBlock` or if it is a BlockArgument then it won't be cloned, thus
940  // the `lastYielded` value remains unchanged. Else, update the `lastYielded`
941  // value with the clone corresponding to the yielded value.
942  for (unsigned i = 0, e = lastYielded.size(); i < e; i++) {
943  Operation *defOp = yieldedValues[i].getDefiningOp();
944  if (defOp && defOp->getBlock() == loopBodyBlock)
945  lastYielded[i] = operandMap.lookup(yieldedValues[i]);
946  }
947  }
948 
949  // Make sure we annotate the Ops in the original body. We do this last so that
950  // any annotations are not copied into the cloned Ops above.
951  for (auto it = loopBodyBlock->begin(); it != std::next(srcBlockEnd); it++)
952  annotateFn(0, &*it, builder);
953 
954  // Update operands of the yield statement.
955  loopBodyBlock->getTerminator()->setOperands(lastYielded);
956 }

958 /// Helper to generate cleanup loop for unroll or unroll-and-jam when the trip
959 /// count is not a multiple of `unrollFactor`.
960 static LogicalResult generateCleanupLoopForUnroll(AffineForOp forOp,
961  uint64_t unrollFactor) {
962  // Insert the cleanup loop right after 'forOp'.
963  OpBuilder builder(forOp->getBlock(), std::next(Block::iterator(forOp)));
964  auto cleanupForOp = cast<AffineForOp>(builder.clone(*forOp));
965 
966  // Update uses of `forOp` results. `cleanupForOp` should use `forOp` result
967  // and produce results for the original users of `forOp` results.
968  auto results = forOp.getResults();
969  auto cleanupResults = cleanupForOp.getResults();
970  auto cleanupIterOperands = cleanupForOp.getInits();
971 
972  for (auto e : llvm::zip(results, cleanupResults, cleanupIterOperands)) {
973  std::get<0>(e).replaceAllUsesWith(std::get<1>(e));
974  cleanupForOp->replaceUsesOfWith(std::get<2>(e), std::get<0>(e));
975  }
976 
977  AffineMap cleanupMap;
978  SmallVector<Value, 4> cleanupOperands;
979  getCleanupLoopLowerBound(forOp, unrollFactor, cleanupMap, cleanupOperands);
980  if (!cleanupMap)
981  return failure();
982 
983  cleanupForOp.setLowerBound(cleanupOperands, cleanupMap);
984  // Promote the loop body up if this has turned into a single iteration loop.
985  (void)promoteIfSingleIteration(cleanupForOp);
986 
987  // Adjust upper bound of the original loop; this is the same as the lower
988  // bound of the cleanup loop.
989  forOp.setUpperBound(cleanupOperands, cleanupMap);
990  return success();
991 }

993 /// Unrolls this loop by the specified factor. Returns success if the loop
994 /// is successfully unrolled.
995 LogicalResult mlir::affine::loopUnrollByFactor(
996  AffineForOp forOp, uint64_t unrollFactor,
997  function_ref<void(unsigned, Operation *, OpBuilder)> annotateFn,
998  bool cleanUpUnroll) {
999  assert(unrollFactor > 0 && "unroll factor should be positive");
1000 
1001  std::optional<uint64_t> mayBeConstantTripCount = getConstantTripCount(forOp);
1002  if (unrollFactor == 1) {
1003  if (mayBeConstantTripCount == 1 && failed(promoteIfSingleIteration(forOp)))
1004  return failure();
1005  return success();
1006  }
1007 
1008  // Nothing in the loop body other than the terminator.
1009  if (llvm::hasSingleElement(forOp.getBody()->getOperations()))
1010  return success();
1011 
1012  // If the trip count is lower than the unroll factor, no unrolled body.
1013  if (mayBeConstantTripCount && *mayBeConstantTripCount < unrollFactor) {
1014  if (cleanUpUnroll) {
1015  // Unroll the cleanup loop if cleanUpUnroll is specified.
1016  return loopUnrollFull(forOp);
1017  }
1018 
1019  return failure();
1020  }
1021 
1022  // Generate the cleanup loop if trip count isn't a multiple of unrollFactor.
1023  if (getLargestDivisorOfTripCount(forOp) % unrollFactor != 0) {
1024  // Loops where the lower bound is a max expression or the upper bound is
1025  // a min expression and the trip count doesn't divide the unroll factor
1026  // can't be unrolled since the lower bound of the cleanup loop in such cases
1027  // cannot be expressed as an affine function or a max over affine functions.
1028  if (forOp.getLowerBoundMap().getNumResults() != 1 ||
1029  forOp.getUpperBoundMap().getNumResults() != 1)
1030  return failure();
1031  if (cleanUpUnroll)
1032  // Force unroll including cleanup loop
1033  return loopUnrollFull(forOp);
1034  if (failed(generateCleanupLoopForUnroll(forOp, unrollFactor)))
1035  assert(false && "cleanup loop lower bound map for single result lower "
1036  "and upper bound maps can always be determined");
1037  }
1038 
1039  ValueRange iterArgs(forOp.getRegionIterArgs());
1040  auto yieldedValues = forOp.getBody()->getTerminator()->getOperands();
1041 
1042  // Scale the step of loop being unrolled by unroll factor.
1043  int64_t step = forOp.getStepAsInt();
1044  forOp.setStep(step * unrollFactor);
1045  generateUnrolledLoop(
1046  forOp.getBody(), forOp.getInductionVar(), unrollFactor,
1047  [&](unsigned i, Value iv, OpBuilder b) {
1048  // iv' = iv + i * step
1049  auto d0 = b.getAffineDimExpr(0);
1050  auto bumpMap = AffineMap::get(1, 0, d0 + i * step);
1051  return AffineApplyOp::create(b, forOp.getLoc(), bumpMap, iv);
1052  },
1053  /*annotateFn=*/annotateFn,
1054  /*iterArgs=*/iterArgs, /*yieldedValues=*/yieldedValues);
1055 
1056  // Promote the loop body up if this has turned into a single iteration loop.
1057  (void)promoteIfSingleIteration(forOp);
1058  return success();
1059 }

1075  if (!forOp.isDefinedOutsideOfLoop(controlOperand))
1076  return WalkResult::interrupt();

1083 /// Unrolls and jams this loop by the specified factor.
1084 LogicalResult mlir::affine::loopUnrollJamByFactor(AffineForOp forOp,
1085  uint64_t unrollJamFactor) {
1086  assert(unrollJamFactor > 0 && "unroll jam factor should be positive");
1087 
1088  std::optional<uint64_t> mayBeConstantTripCount = getConstantTripCount(forOp);
1089  if (unrollJamFactor == 1) {
1090  if (mayBeConstantTripCount == 1 && failed(promoteIfSingleIteration(forOp)))
1091  return failure();
1092  return success();
1093  }
1094 
1095  // Nothing in the loop body other than the terminator.
1096  if (llvm::hasSingleElement(forOp.getBody()->getOperations()))
1097  return success();
1098 
1099  // If the trip count is lower than the unroll jam factor, no unroll jam.
1100  if (mayBeConstantTripCount && *mayBeConstantTripCount < unrollJamFactor) {
1101  LLVM_DEBUG(llvm::dbgs() << "[failed] trip count < unroll-jam factor\n");
1102  return failure();
1103  }
1104 
1105  // If any control operand of any inner loop of `forOp` is defined within
1106  // `forOp`, no unroll jam.
1107  if (!areInnerBoundsInvariant(forOp))
1108  return failure();
1109 
1110  // Gather all sub-blocks to jam upon the loop being unrolled.
1111  JamBlockGatherer<AffineForOp> jbg;
1112  jbg.walk(forOp);
1113  auto &subBlocks = jbg.subBlocks;
1114 
1115  // Collect loops with iter_args.
1116  SmallVector<AffineForOp, 4> loopsWithIterArgs;
1117  forOp.walk([&](AffineForOp aForOp) {
1118  if (aForOp.getNumIterOperands() > 0)
1119  loopsWithIterArgs.push_back(aForOp);
1120  });
1121 
1122  // Get supported reductions to be used for creating reduction ops at the end.
1123  SmallVector<LoopReduction> reductions;
1124  if (forOp.getNumIterOperands() > 0)
1125  getSupportedReductions(forOp, reductions);
1126 
1127  // Generate the cleanup loop if trip count isn't a multiple of
1128  // unrollJamFactor.
1129  if (getLargestDivisorOfTripCount(forOp) % unrollJamFactor != 0) {
1130  // Loops where the lower bound is a max expression or the upper bound is
1131  // a min expression and the trip count doesn't divide the unroll factor
1132  // can't be unrolled since the lower bound of the cleanup loop in such cases
1133  // cannot be expressed as an affine function or a max over affine functions.
1134  if (forOp.getLowerBoundMap().getNumResults() != 1 ||
1135  forOp.getUpperBoundMap().getNumResults() != 1)
1136  return failure();
1137  if (failed(generateCleanupLoopForUnroll(forOp, unrollJamFactor)))
1138  assert(false && "cleanup loop lower bound map for single result lower "
1139  "and upper bound maps can always be determined");
1140  }
1141 
1142  // `operandMaps[i - 1]` carries old->new operand mapping for the ith unrolled
1143  // iteration. There are (`unrollJamFactor` - 1) iterations.
1144  SmallVector<IRMapping, 4> operandMaps(unrollJamFactor - 1);
1145 
1146  // For any loop with iter_args, replace it with a new loop that has
1147  // `unrollJamFactor` copies of its iterOperands, iter_args and yield
1148  // operands.
1149  SmallVector<AffineForOp, 4> newLoopsWithIterArgs;
1150  IRRewriter rewriter(forOp.getContext());
1151  for (AffineForOp oldForOp : loopsWithIterArgs) {
1152  SmallVector<Value> dupIterOperands, dupYieldOperands;
1153  ValueRange oldIterOperands = oldForOp.getInits();
1154  ValueRange oldIterArgs = oldForOp.getRegionIterArgs();
1155  ValueRange oldYieldOperands =
1156  cast<AffineYieldOp>(oldForOp.getBody()->getTerminator()).getOperands();
1157  // Get additional iterOperands, iterArgs, and yield operands. We will
1158  // fix iterOperands and yield operands after cloning of sub-blocks.
1159  for (unsigned i = unrollJamFactor - 1; i >= 1; --i) {
1160  dupIterOperands.append(oldIterOperands.begin(), oldIterOperands.end());
1161  dupYieldOperands.append(oldYieldOperands.begin(), oldYieldOperands.end());
1162  }
1163  // Create a new loop with additional iterOperands, iter_args and yield
1164  // operands. This new loop will take the loop body of the original loop.
1165  bool forOpReplaced = oldForOp == forOp;
1166  AffineForOp newForOp =
1167  cast<AffineForOp>(*oldForOp.replaceWithAdditionalYields(
1168  rewriter, dupIterOperands, /*replaceInitOperandUsesInLoop=*/false,
1169  [&](OpBuilder &b, Location loc, ArrayRef<BlockArgument> newBbArgs) {
1170  return dupYieldOperands;
1171  }));
1172  newLoopsWithIterArgs.push_back(newForOp);
1173  // `forOp` has been replaced with a new loop.
1174  if (forOpReplaced)
1175  forOp = newForOp;
1176  // Update `operandMaps` for `newForOp` iterArgs and results.
1177  ValueRange newIterArgs = newForOp.getRegionIterArgs();
1178  unsigned oldNumIterArgs = oldIterArgs.size();
1179  ValueRange newResults = newForOp.getResults();
1180  unsigned oldNumResults = newResults.size() / unrollJamFactor;
1181  assert(oldNumIterArgs == oldNumResults &&
1182  "oldNumIterArgs must be the same as oldNumResults");
1183  for (unsigned i = unrollJamFactor - 1; i >= 1; --i) {
1184  for (unsigned j = 0; j < oldNumIterArgs; ++j) {
1185  // `newForOp` has `unrollJamFactor` - 1 new sets of iterArgs and
1186  // results. Update `operandMaps[i - 1]` to map old iterArgs and results
1187  // to those in the `i`th new set.
1188  operandMaps[i - 1].map(newIterArgs[j],
1189  newIterArgs[i * oldNumIterArgs + j]);
1190  operandMaps[i - 1].map(newResults[j],
1191  newResults[i * oldNumResults + j]);
1192  }
1193  }
1194  }
1195 
1196  // Scale the step of loop being unroll-jammed by the unroll-jam factor.
1197  int64_t step = forOp.getStepAsInt();
1198  forOp.setStep(step * unrollJamFactor);
1199 
1200  auto forOpIV = forOp.getInductionVar();
1201  // Unroll and jam (appends unrollJamFactor - 1 additional copies).
1202  for (unsigned i = unrollJamFactor - 1; i >= 1; --i) {
1203  for (auto &subBlock : subBlocks) {
1204  // Builder to insert unroll-jammed bodies. Insert right at the end of
1205  // sub-block.
1206  OpBuilder builder(subBlock.first->getBlock(), std::next(subBlock.second));
1207 
1208  // If the induction variable is used, create a remapping to the value for
1209  // this unrolled instance.
1210  if (!forOpIV.use_empty()) {
1211  // iv' = iv + i * step, i = 1 to unrollJamFactor-1.
1212  auto d0 = builder.getAffineDimExpr(0);
1213  auto bumpMap = AffineMap::get(1, 0, d0 + i * step);
1214  auto ivUnroll =
1215  AffineApplyOp::create(builder, forOp.getLoc(), bumpMap, forOpIV);
1216  operandMaps[i - 1].map(forOpIV, ivUnroll);
1217  }
1218  // Clone the sub-block being unroll-jammed.
1219  for (auto it = subBlock.first; it != std::next(subBlock.second); ++it)
1220  builder.clone(*it, operandMaps[i - 1]);
1221  }
1222  // Fix iterOperands and yield op operands of newly created loops.
1223  for (auto newForOp : newLoopsWithIterArgs) {
1224  unsigned oldNumIterOperands =
1225  newForOp.getNumIterOperands() / unrollJamFactor;
1226  unsigned numControlOperands = newForOp.getNumControlOperands();
1227  auto yieldOp = cast<AffineYieldOp>(newForOp.getBody()->getTerminator());
1228  unsigned oldNumYieldOperands = yieldOp.getNumOperands() / unrollJamFactor;
1229  assert(oldNumIterOperands == oldNumYieldOperands &&
1230  "oldNumIterOperands must be the same as oldNumYieldOperands");
1231  for (unsigned j = 0; j < oldNumIterOperands; ++j) {
1232  // The `i`th duplication of an old iterOperand or yield op operand
1233  // needs to be replaced with a mapped value from `operandMaps[i - 1]`
1234  // if such mapped value exists.
1235  newForOp.setOperand(numControlOperands + i * oldNumIterOperands + j,
1236  operandMaps[i - 1].lookupOrDefault(
1237  newForOp.getOperand(numControlOperands + j)));
1238  yieldOp.setOperand(
1239  i * oldNumYieldOperands + j,
1240  operandMaps[i - 1].lookupOrDefault(yieldOp.getOperand(j)));
1241  }
1242  }
1243  }
1244  if (forOp.getNumResults() > 0) {
1245  // Create reduction ops to combine every `unrollJamFactor` related results
1246  // into one value. For example, for %0:2 = affine.for ... and addf, we add
1247  // %1 = arith.addf %0#0, %0#1, and replace the following uses of %0#0 with
1248  // %1.
1249  rewriter.setInsertionPointAfter(forOp);
1250  auto loc = forOp.getLoc();
1251  unsigned oldNumResults = forOp.getNumResults() / unrollJamFactor;
1252  for (LoopReduction &reduction : reductions) {
1253  unsigned pos = reduction.iterArgPosition;
1254  Value lhs = forOp.getResult(pos);
1255  Value rhs;
1256  SmallPtrSet<Operation *, 4> newOps;
1257  for (unsigned i = unrollJamFactor - 1; i >= 1; --i) {
1258  rhs = forOp.getResult(i * oldNumResults + pos);
1259  // Create ops based on reduction type.
1260  lhs = arith::getReductionOp(reduction.kind, rewriter, loc, lhs, rhs);
1261  if (!lhs)
1262  return failure();
1263  Operation *op = lhs.getDefiningOp();
1264  assert(op && "Reduction op should have been created");
1265  newOps.insert(op);
1266  }
1267  // Replace all uses except those in newly created reduction ops.
1268  forOp.getResult(pos).replaceAllUsesExcept(lhs, newOps);
1269  }
1270  }
1271 
1272  // Promote the loop body up if this has turned into a single iteration loop.
1273  (void)promoteIfSingleIteration(forOp);
1274  return success();
1275 }

1277 /// Performs loop interchange on 'forOpA' and 'forOpB', where 'forOpB' is
1278 /// nested within 'forOpA' as the only non-terminator operation in its block.
1279 void mlir::affine::interchangeLoops(AffineForOp forOpA, AffineForOp forOpB) {
1280  assert(&*forOpA.getBody()->begin() == forOpB.getOperation());
1281  auto &forOpABody = forOpA.getBody()->getOperations();
1282  auto &forOpBBody = forOpB.getBody()->getOperations();
1283 
1284  // 1) Splice forOpA's non-terminator operations (which is just forOpB) just
1285  // before forOpA (in ForOpA's parent's block) this should leave 'forOpA's
1286  // body containing only the terminator.
1287  forOpA->getBlock()->getOperations().splice(Block::iterator(forOpA),
1288  forOpABody, forOpABody.begin(),
1289  std::prev(forOpABody.end()));
1290  // 2) Splice forOpB's non-terminator operations into the beginning of forOpA's
1291  // body (this leaves forOpB's body containing only the terminator).
1292  forOpABody.splice(forOpABody.begin(), forOpBBody, forOpBBody.begin(),
1293  std::prev(forOpBBody.end()));
1294  // 3) Splice forOpA into the beginning of forOpB's body.
1295  forOpBBody.splice(forOpBBody.begin(), forOpA->getBlock()->getOperations(),
1296  Block::iterator(forOpA));
1297 }

1299 // Checks each dependence component against the permutation to see if the
1300 // desired loop interchange would violate dependences by making the
1301 // dependence component lexicographically negative.
1302 static bool checkLoopInterchangeDependences(
1303  const std::vector<SmallVector<DependenceComponent, 2>> &depCompsVec,
1304  ArrayRef<AffineForOp> loops, ArrayRef<unsigned> loopPermMap) {
1305  // Invert permutation map.
1306  unsigned maxLoopDepth = loops.size();
1307  SmallVector<unsigned, 4> loopPermMapInv;
1308  loopPermMapInv.resize(maxLoopDepth);
1309  for (unsigned i = 0; i < maxLoopDepth; ++i)
1310  loopPermMapInv[loopPermMap[i]] = i;
1311 
1312  // Check each dependence component against the permutation to see if the
1313  // desired loop interchange permutation would make the dependence vectors
1314  // lexicographically negative.
1315  // Example 1: [-1, 1][0, 0]
1316  // Example 2: [0, 0][-1, 1]
1317  for (const auto &depComps : depCompsVec) {
1318  assert(depComps.size() >= maxLoopDepth);
1319  // Check if the first non-zero dependence component is positive.
1320  // This iterates through loops in the desired order.
1321  for (unsigned j = 0; j < maxLoopDepth; ++j) {
1322  unsigned permIndex = loopPermMapInv[j];
1323  assert(depComps[permIndex].lb);
1324  int64_t depCompLb = *depComps[permIndex].lb;
1325  if (depCompLb > 0)
1326  break;
1327  if (depCompLb < 0)
1328  return false;
1329  }
1330  }
1331  return true;
1332 }

1349 /// Returns true if `loops` is a perfectly nested loop nest, where loops appear
1350 /// in it from outermost to innermost.
1351 bool LLVM_ATTRIBUTE_UNUSED
1352 mlir::affine::isPerfectlyNested(ArrayRef<AffineForOp> loops) {
1353  assert(!loops.empty() && "no loops provided");
1354 
1355  // We already know that the block can't be empty.
1356  auto hasTwoElements = [](Block *block) {
1357  auto secondOpIt = std::next(block->begin());
1358  return secondOpIt != block->end() && &*secondOpIt == &block->back();
1359  };
1360 
1361  auto enclosingLoop = loops.front();
1362  for (auto loop : loops.drop_front()) {
1363  auto parentForOp = dyn_cast<AffineForOp>(loop->getParentOp());
1364  // parentForOp's body should be just this loop and the terminator.
1365  if (parentForOp != enclosingLoop || !hasTwoElements(parentForOp.getBody()))
1366  return false;
1367  enclosingLoop = loop;

1372 // input[i] should move from position i -> permMap[i]. Returns the position in
1373 // `input` that becomes the new outermost loop.
1374 unsigned mlir::affine::permuteLoops(ArrayRef<AffineForOp> input,
1375  ArrayRef<unsigned> permMap) {
1376  assert(input.size() == permMap.size() && "invalid permutation map size");
1377  // Check whether the permutation spec is valid. This is a small vector - we'll
1378  // just sort and check if it's iota.
1379  SmallVector<unsigned, 4> checkPermMap(permMap);
1380  llvm::sort(checkPermMap);
1381  if (llvm::any_of(llvm::enumerate(checkPermMap),
1382  [](const auto &en) { return en.value() != en.index(); }))
1383  assert(false && "invalid permutation map");
1384 
1385  // Nothing to do.
1386  if (input.size() < 2)
1387  return 0;
1388 
1389  assert(isPerfectlyNested(input) && "input not perfectly nested");
1390 
1391  // Compute the inverse mapping, invPermMap: since input[i] goes to position
1392  // permMap[i], position i of the permuted nest is at input[invPermMap[i]].
1393  SmallVector<std::pair<unsigned, unsigned>, 4> invPermMap;
1394  for (unsigned i = 0, e = input.size(); i < e; ++i)
1395  invPermMap.push_back({permMap[i], i});
1396  llvm::sort(invPermMap);
1397 
1398  // Move the innermost loop body to the loop that would be the innermost in the
1399  // permuted nest (only if the innermost loop is going to change).
1400  if (permMap.back() != input.size() - 1) {
1401  Block *destBody = ((AffineForOp)input[invPermMap.back().second]).getBody();
1402  Block *srcBody = ((AffineForOp)input.back()).getBody();
1403  destBody->getOperations().splice(destBody->begin(),
1404  srcBody->getOperations(), srcBody->begin(),
1405  std::prev(srcBody->end()));
1406  }
1407 
1408  // We'll move each loop in `input` in the reverse order so that its body is
1409  // empty when we are moving it; this incurs zero copies and no erasing.
1410  for (int i = input.size() - 1; i >= 0; --i) {
1411  // If this has to become the outermost loop after permutation, add it to the
1412  // parent block of the original root.
1413  if (permMap[i] == 0) {
1414  // If the root remains the same, nothing to do.
1415  if (i == 0)
1416  continue;
1417  // Make input[i] the new outermost loop moving it into parentBlock.
1418  auto *parentBlock = input[0]->getBlock();
1419  parentBlock->getOperations().splice(Block::iterator(input[0]),
1420  input[i]->getBlock()->getOperations(),
1421  Block::iterator(input[i]));
1422  continue;
1423  }
1424 
1425  // If the parent in the permuted order is the same as in the original,
1426  // nothing to do.
1427  unsigned parentPosInInput = invPermMap[permMap[i] - 1].second;
1428  if (i > 0 && static_cast<unsigned>(i - 1) == parentPosInInput)
1429  continue;
1430 
1431  // Move input[i] to its surrounding loop in the transformed nest.
1432  auto *destBody = ((AffineForOp)input[parentPosInInput]).getBody();
1433  destBody->getOperations().splice(destBody->begin(),
1434  input[i]->getBlock()->getOperations(),
1435  Block::iterator(input[i]));
1436  }
1437 
1438  return invPermMap[0].second;
1439 }

1441 // Sinks all sequential loops to the innermost levels (while preserving
1442 // relative order among them) and moves all parallel loops to the
1443 // outermost (while again preserving relative order among them).
1444 AffineForOp mlir::affine::sinkSequentialLoops(AffineForOp forOp) {
1445  SmallVector<AffineForOp, 4> loops;
1446  getPerfectlyNestedLoops(loops, forOp);
1447  if (loops.size() < 2)
1448  return forOp;
1449 
1450  // Gather dependence components for dependences between all ops in loop nest
1451  // rooted at 'loops[0]', at loop depths in range [1, maxLoopDepth].
1452  unsigned maxLoopDepth = loops.size();
1453  std::vector<SmallVector<DependenceComponent, 2>> depCompsVec;
1454  getDependenceComponents(loops[0], maxLoopDepth, &depCompsVec);
1455 
1456  // Mark loops as either parallel or sequential.
1457  SmallVector<bool, 8> isParallelLoop(maxLoopDepth, true);
1458  for (auto &depComps : depCompsVec) {
1459  assert(depComps.size() >= maxLoopDepth);
1460  for (unsigned j = 0; j < maxLoopDepth; ++j) {
1461  DependenceComponent &depComp = depComps[j];
1462  assert(depComp.lb.has_value() && depComp.ub.has_value());
1463  if (*depComp.lb != 0 || *depComp.ub != 0)
1464  isParallelLoop[j] = false;
1465  }
1466  }
1467 
1468  unsigned numParallelLoops = llvm::count(isParallelLoop, true);
1469 
1470  // Compute permutation of loops that sinks sequential loops (and thus raises
1471  // parallel loops) while preserving relative order.
1472  SmallVector<unsigned, 4> loopPermMap(maxLoopDepth);
1473  unsigned nextSequentialLoop = numParallelLoops;
1474  unsigned nextParallelLoop = 0;
1475  for (unsigned i = 0; i < maxLoopDepth; ++i) {
1476  if (isParallelLoop[i]) {
1477  loopPermMap[i] = nextParallelLoop++;
1478  } else {
1479  loopPermMap[i] = nextSequentialLoop++;
1480  }
1481  }
1482 
1483  // Check if permutation 'loopPermMap' would violate dependences.
1484  if (!checkLoopInterchangeDependences(depCompsVec, loops, loopPermMap))
1485  return forOp;
1486  // Perform loop interchange according to permutation 'loopPermMap'.
1487  unsigned loopNestRootIndex = permuteLoops(loops, loopPermMap);
1488  return loops[loopNestRootIndex];
1489 }

1511 // Stripmines `forOp` by `factor` and sinks it under each of the `targets`.
1512 // Stripmine-sink is a primitive building block for generalized tiling of
1513 // imperfectly nested loops.
1514 // This transformation is purely mechanical and does not check legality,
1515 // profitability or even structural correctness. It is the user's
1516 // responsibility to specify `targets` that are dominated by `forOp`.
1517 // Returns the new AffineForOps, one per `targets`, nested immediately under
1518 // each of the `targets`.
1519 static SmallVector<AffineForOp, 8>
1520 stripmineSink(AffineForOp forOp, uint64_t factor,
1521  ArrayRef<AffineForOp> targets) {
1522  auto originalStep = forOp.getStepAsInt();
1523  auto scaledStep = originalStep * factor;
1524  forOp.setStep(scaledStep);
1525 
1526  OpBuilder b(forOp->getBlock(), std::next(Block::iterator(forOp)));
1527 
1528  // Lower-bound map creation.
1529  auto lbMap = forOp.getLowerBoundMap();
1530  SmallVector<Value, 4> lbOperands(forOp.getLowerBoundOperands());
1531  augmentMapAndBounds(b, forOp.getInductionVar(), &lbMap, &lbOperands);
1532 
1533  // Upper-bound map creation.
1534  auto ubMap = forOp.getUpperBoundMap();
1535  SmallVector<Value, 4> ubOperands(forOp.getUpperBoundOperands());
1536  augmentMapAndBounds(b, forOp.getInductionVar(), &ubMap, &ubOperands,
1537  /*offset=*/scaledStep);
1538 
1539  auto iv = forOp.getInductionVar();
1540  SmallVector<AffineForOp, 8> innerLoops;
1541  for (auto t : targets) {
1542  // Insert newForOp before the terminator of `t`.
1543  auto b = OpBuilder::atBlockTerminator(t.getBody());
1544  auto newForOp = AffineForOp::create(b, t.getLoc(), lbOperands, lbMap,
1545  ubOperands, ubMap, originalStep);
1546  auto begin = t.getBody()->begin();
1547  // Skip terminator and `newForOp` which is just before the terminator.
1548  auto nOps = t.getBody()->getOperations().size() - 2;
1549  newForOp.getBody()->getOperations().splice(
1550  newForOp.getBody()->getOperations().begin(),
1551  t.getBody()->getOperations(), begin, std::next(begin, nOps));
1552  replaceAllUsesInRegionWith(iv, newForOp.getInductionVar(),
1553  newForOp.getRegion());
1554  innerLoops.push_back(newForOp);
1555  }
1556 
1557  return innerLoops;
1558 }

1596 LogicalResult mlir::affine::coalesceLoops(MutableArrayRef<AffineForOp> loops) {
1597  if (loops.size() < 2)
1598  return success();
1599 
1600  AffineForOp innermost = loops.back();
1601  AffineForOp outermost = loops.front();
1602  AffineBound ub = outermost.getUpperBound();
1603  AffineMap origUbMap = ub.getMap();
1604  Location loc = outermost.getLoc();
1605  OpBuilder builder(outermost);
1606  for (AffineForOp loop : loops) {
1607  // We only work on normalized loops.
1608  if (loop.getStepAsInt() != 1 || !loop.hasConstantLowerBound() ||
1609  loop.getConstantLowerBound() != 0)
1610  return failure();
1611  }
1612  SmallVector<Value, 4> upperBoundSymbols;
1613  SmallVector<Value, 4> ubOperands(ub.getOperands().begin(),
1614  ub.getOperands().end());
1615 
1616  // 1. Store the upper bound of the outermost loop in a variable.
1617  Value prev;
1618  if (!llvm::hasSingleElement(origUbMap.getResults()))
1619  prev = AffineMinOp::create(builder, loc, origUbMap, ubOperands);
1620  else
1621  prev = AffineApplyOp::create(builder, loc, origUbMap, ubOperands);
1622  upperBoundSymbols.push_back(prev);
1623 
1624  // 2. Emit code computing the upper bound of the coalesced loop as product of
1625  // the number of iterations of all loops.
1626  for (AffineForOp loop : loops.drop_front()) {
1627  ub = loop.getUpperBound();
1628  origUbMap = ub.getMap();
1629  ubOperands = ub.getOperands();
1630  Value upperBound;
1631  // If upper bound map has more than one result, take their minimum.
1632  if (!llvm::hasSingleElement(origUbMap.getResults()))
1633  upperBound = AffineMinOp::create(builder, loc, origUbMap, ubOperands);
1634  else
1635  upperBound = AffineApplyOp::create(builder, loc, origUbMap, ubOperands);
1636  upperBoundSymbols.push_back(upperBound);
1637  SmallVector<Value, 4> operands;
1638  operands.push_back(prev);
1639  operands.push_back(upperBound);
1640  // Maintain running product of loop upper bounds.
1641  prev = AffineApplyOp::create(
1642  builder, loc,
1643  AffineMap::get(/*dimCount=*/1,
1644  /*symbolCount=*/1,
1645  builder.getAffineDimExpr(0) *
1646  builder.getAffineSymbolExpr(0)),
1647  operands);
1648  }
1649  // Set upper bound of the coalesced loop.
1650  AffineMap newUbMap = AffineMap::get(
1651  /*dimCount=*/0,
1652  /*symbolCount=*/1, builder.getAffineSymbolExpr(0), builder.getContext());
1653  outermost.setUpperBound(prev, newUbMap);
1654 
1655  builder.setInsertionPointToStart(outermost.getBody());
1656 
1657  // 3. Remap induction variables. For each original loop, the value of the
1658  // induction variable can be obtained by dividing the induction variable of
1659  // the linearized loop by the total number of iterations of the loops nested
1660  // in it modulo the number of iterations in this loop (remove the values
1661  // related to the outer loops):
1662  // iv_i = floordiv(iv_linear, product-of-loop-ranges-until-i) mod range_i.
1663  // Compute these iteratively from the innermost loop by creating a "running
1664  // quotient" of division by the range.
1665  Value previous = outermost.getInductionVar();
1666  for (unsigned idx = loops.size(); idx > 0; --idx) {
1667  if (idx != loops.size()) {
1668  SmallVector<Value, 4> operands;
1669  operands.push_back(previous);
1670  operands.push_back(upperBoundSymbols[idx]);
1671  previous = AffineApplyOp::create(builder, loc,
1672  AffineMap::get(
1673  /*dimCount=*/1, /*symbolCount=*/1,
1674  builder.getAffineDimExpr(0).floorDiv(
1675  builder.getAffineSymbolExpr(0))),
1676  operands);
1677  }
1678  // Modified value of the induction variables of the nested loops after
1679  // coalescing.
1680  Value inductionVariable;
1681  if (idx == 1) {
1682  inductionVariable = previous;
1683  } else {
1684  SmallVector<Value, 4> applyOperands;
1685  applyOperands.push_back(previous);
1686  applyOperands.push_back(upperBoundSymbols[idx - 1]);
1687  inductionVariable = AffineApplyOp::create(
1688  builder, loc,
1689  AffineMap::get(
1690  /*dimCount=*/1, /*symbolCount=*/1,
1691  builder.getAffineDimExpr(0) % builder.getAffineSymbolExpr(0)),
1692  applyOperands);
1693  }
1694  replaceAllUsesInRegionWith(loops[idx - 1].getInductionVar(),
1695  inductionVariable, loops.back().getRegion());
1696  }
1697 
1698  // 4. Move the operations from the innermost just above the second-outermost
1699  // loop, delete the extra terminator and the second-outermost loop.
1700  AffineForOp secondOutermostLoop = loops[1];
1701  innermost.getBody()->back().erase();
1702  outermost.getBody()->getOperations().splice(
1703  Block::iterator(secondOutermostLoop.getOperation()),
1704  innermost.getBody()->getOperations());
1705  secondOutermostLoop.erase();
1706  return success();
1707 }

1828 /// Generates a point-wise copy from/to a non-zero ranked `memref' to/from
1829 /// `fastMemRef' and returns the outermost AffineForOp of the copy loop nest.
1830 /// `lbMaps` and `ubMaps` along with `lbOperands` and `ubOperands` hold the
1831 /// lower and upper bound information for the copy loop nest. `fastBufOffsets`
1832 /// contain the expressions to be subtracted out from the respective copy loop
1833 /// iterators in order to index the fast buffer. If `copyOut' is true, generates
1834 /// a copy-out; otherwise a copy-in. Builder `b` should be set to the point the
1835 /// copy nest is inserted.
1836 //
1837 /// The copy-in nest is generated as follows as an example for a 2-d region:
1838 /// for x = ...
1839 /// for y = ...
1840 /// fast_buf[x - offset_x][y - offset_y] = memref[x][y]
1841 ///
1842 static AffineForOp
1843 generatePointWiseCopy(Location loc, Value memref, Value fastMemRef,
1844  ArrayRef<AffineMap> lbMaps, ArrayRef<Value> lbOperands,
1845  ArrayRef<AffineMap> ubMaps, ArrayRef<Value> ubOperands,
1846  ArrayRef<AffineExpr> fastBufOffsets, bool isCopyOut,
1847  OpBuilder b) {
1848  assert(llvm::all_of(lbMaps, [&](AffineMap lbMap) {
1849  return lbMap.getNumInputs() == lbOperands.size();
1850  }));
1851  assert(llvm::all_of(ubMaps, [&](AffineMap ubMap) {
1852  return ubMap.getNumInputs() == ubOperands.size();
1853  }));
1854 
1855  unsigned rank = cast<MemRefType>(memref.getType()).getRank();
1856  // A copy nest can't be generated for 0-ranked memrefs.
1857  assert(rank != 0 && "non-zero rank memref expected");
1858  assert(lbMaps.size() == rank && "wrong number of lb maps");
1859  assert(ubMaps.size() == rank && "wrong number of ub maps");
1860

1862  SmallVector<AffineExpr, 4> fastBufExprs;
1863  SmallVector<Value, 4> fastBufMapOperands;
1864  AffineForOp copyNestRoot;
1865  SmallVector<AffineApplyOp, 4> mayBeDeadApplys;
1866  for (unsigned d = 0; d < rank; ++d) {
1867  auto forOp = createCanonicalizedAffineForOp(b, loc, lbOperands, lbMaps[d],
1868  ubOperands, ubMaps[d]);
1869  if (d == 0)
1870  copyNestRoot = forOp;
1871 
1872  b = OpBuilder::atBlockTerminator(forOp.getBody());
1873 
1874  auto fastBufOffsetMap =
1875  AffineMap::get(lbOperands.size(), 0, fastBufOffsets[d]);
1876  auto offset = AffineApplyOp::create(b, loc, fastBufOffsetMap, lbOperands);
1877

1879  // x - offset_x.
1880  fastBufExprs.push_back(b.getAffineDimExpr(2 * d + 1) -
1881  b.getAffineDimExpr(2 * d));
1882  fastBufMapOperands.push_back(offset);
1883  fastBufMapOperands.push_back(forOp.getInductionVar());
1884  mayBeDeadApplys.push_back(offset);
1885 
1886  // Subscript for the slow memref being copied.
1887  memIndices.push_back(forOp.getInductionVar());
1888  }

1891  AffineMap::get(2 * rank, /*symbolCount=*/0, fastBufExprs, b.getContext());
1892  fullyComposeAffineMapAndOperands(&fastBufMap, &fastBufMapOperands);
1893  fastBufMap = simplifyAffineMap(fastBufMap);
1894  canonicalizeMapAndOperands(&fastBufMap, &fastBufMapOperands);
1895 
1896  // Drop any dead affine.applys.
1897  for (auto applyOp : mayBeDeadApplys)
1898  if (applyOp.use_empty())
1899  applyOp.erase();

1901  if (!isCopyOut) {
1902  // Copy in.
1903  auto load = AffineLoadOp::create(b, loc, memref, memIndices);
1904  AffineStoreOp::create(b, loc, load, fastMemRef, fastBufMap,
1905  fastBufMapOperands);
1906  return copyNestRoot;
1907  }
1908

1911  AffineLoadOp::create(b, loc, fastMemRef, fastBufMap, fastBufMapOperands);
1912  AffineStoreOp::create(b, loc, load, memref, memIndices);
1913  return copyNestRoot;

1922 /// (memref has to be non-zero ranked); generates a copy from the lower memory
1923 /// space to this one, and replaces all loads/stores in the block range
1924 /// [`begin', `end') of `block' to load/store from that buffer. Returns failure
1925 /// if copies could not be generated due to yet unimplemented cases.
1926 /// `copyInPlacementStart` and `copyOutPlacementStart` in copyPlacementBlock
1927 /// specify the insertion points where the incoming copies and outgoing copies,
1928 /// respectively, should be inserted (the insertion happens right before the
1929 /// insertion point). Since `begin` can itself be invalidated due to the memref
1930 /// rewriting done from this method, the output argument `nBegin` is set to its
1931 /// replacement (set to `begin` if no invalidation happens). Since outgoing
1932 /// copies could have been inserted at `end`, the output argument `nEnd` is set
1933 /// to the new end. `sizeInBytes` is set to the size of the fast buffer
1934 /// allocated.
1935 static LogicalResult generateCopy(

1937  Block::iterator end, Block *copyPlacementBlock,
1938  Block::iterator copyInPlacementStart, Block::iterator copyOutPlacementStart,
1939  const AffineCopyOptions &copyOptions, DenseMap<Value, Value> &fastBufferMap,

1946  OpBuilder topBuilder(f.getFunctionBody());

1948

1955  // generation. We later do the memref replacement only in [begin, lastCopyOp]
1956  // so that the original memref's used in the data movement code themselves
1957  // don't get replaced.

1961  // explicit copying.

1967  OpBuilder epilogue(copyPlacementBlock, copyOutPlacementStart);

1976  auto memref = region.memref;
1977  auto memRefType = cast<MemRefType>(memref.getType());

1986  SmallVector<Value, 4> memIndices;
1987  // Indices for the faster buffer being copied into/from.
1988  SmallVector<Value, 4> bufIndices;

2005  return failure();

2010  LLVM_DEBUG(llvm::dbgs()
2011  << "Max lower bound for memref region start not supported\n");
2012  return failure();

2016  LLVM_DEBUG(llvm::dbgs() << "Nothing to copy\n");
2017  return success();

2022  region.getLowerAndUpperBound(i, lbMaps[i], ubMaps[i]);

2024  LLVM_DEBUG(llvm::dbgs()

2034  // copy generation is being done or other valid symbols in MLIR.
2035  SmallVector<Value, 8> regionSymbols;
2036  cst->getValues(rank, cst->getNumVars(), &regionSymbols);

2039  // expression for a particular dimension of the fast buffer is obtained by
2040  // subtracting out the lower bound on the original memref's data region
2041  // along the corresponding dimension.
2042

2044  SmallVector<AffineExpr, 4> fastBufOffsets;
2045  fastBufOffsets.reserve(rank);
2046  for (unsigned d = 0; d < rank; d++) {
2047  assert(lbs[d].getNumSymbols() == cst->getNumCols() - rank - 1 &&
2048  "incorrect bound size");

2056  } else {

2058  arith::ConstantIndexOp::create(top, loc, indexVal).getResult());

2060  } else {

2062  // corresponding dimension on the memory region (stored in 'offset').
2063  // Remap all inputs of the map to dimensions uniformly since in the
2064  // generate IR we need valid affine symbols as opposed to "symbols" for
2065  // the purpose of the memref region.
2066  SmallVector<AffineExpr> symReplacements(lbs[d].getNumSymbols());

2079  // the original memref further below.
2080  fastBufOffsets.push_back(lbs[d].getResult(0));

2659 static LogicalResult
2660 createFullTiles(MutableArrayRef<AffineForOp> inputNest,
2661  SmallVectorImpl<AffineForOp> &fullTileLoops, OpBuilder b) {
2662  fullTileLoops.reserve(inputNest.size());
2663 
2664  // For each loop in the original nest identify a lower/upper bound pair such
2665  // that their difference is a constant.
2666  FlatAffineValueConstraints cst;
2667  for (auto loop : inputNest) {
2668  // TODO: straightforward to generalize to a non-unit stride.
2669  if (loop.getStepAsInt() != 1) {
2670  LLVM_DEBUG(llvm::dbgs()
2671  << "[tile separation] non-unit stride not implemented\n");
2672  return failure();
2673  }
2674  SmallVector<Operation *, 1> loopOp{loop.getOperation()};
2675  (void)getIndexSet(loopOp, &cst);
2676  // We will mark everything other than this loop IV as symbol for getting a
2677  // pair of <lb, ub> with a constant difference.
2678  cst.setDimSymbolSeparation(cst.getNumDimAndSymbolVars() - 1);
2679  unsigned lbPos, ubPos;
2680  if (!((IntegerRelation)cst)
2681  .getConstantBoundOnDimSize(/*pos=*/0, /*lb=*/nullptr,
2682  /*boundFloorDivisor=*/nullptr,
2683  /*ub=*/nullptr, &lbPos, &ubPos) ||
2684  lbPos == ubPos) {
2685  LLVM_DEBUG(llvm::dbgs() << "[tile separation] Can't get constant diff / "
2686  "equalities not yet handled\n");
2687  return failure();
2688  }
2689 
2690  // Set all variables as dimensions uniformly since some of those marked as
2691  // symbols above could be outer loop IVs (corresponding tile space IVs).
2692  cst.setDimSymbolSeparation(/*newSymbolCount=*/0);
2693 
2694  AffineValueMap lbVmap, ubVmap;
2695  cst.getIneqAsAffineValueMap(/*pos=*/0, lbPos, lbVmap, b.getContext());
2696  cst.getIneqAsAffineValueMap(/*pos=*/0, ubPos, ubVmap, b.getContext());

2698  b, loop.getLoc(), lbVmap.getOperands(), lbVmap.getAffineMap(),
2699  ubVmap.getOperands(), ubVmap.getAffineMap());
2700  b = OpBuilder::atBlockTerminator(fullTileLoop.getBody());
2701  fullTileLoops.push_back(fullTileLoop);
2702  }
2703 
2704  // Add the body for the full tile loop nest.
2705  IRMapping operandMap;
2706  for (const auto &loopEn : llvm::enumerate(inputNest))
2707  operandMap.map(loopEn.value().getInductionVar(),
2708  fullTileLoops[loopEn.index()].getInductionVar());
2709  b = OpBuilder::atBlockTerminator(fullTileLoops.back().getBody());
2710  for (auto &op : inputNest.back().getBody()->without_terminator())
2711  b.clone(op, operandMap);
2712  return success();
2713 }

2715 LogicalResult
2716 mlir::affine::separateFullTiles(MutableArrayRef<AffineForOp> inputNest,
2717  SmallVectorImpl<AffineForOp> *fullTileNest) {
2718  if (inputNest.empty())
2719  return success();
2720 
2721  auto firstLoop = inputNest[0];
2722 
2723  // Each successive for op has to be nested in the other.
2724  auto prevLoop = firstLoop;
2725  for (auto loop : inputNest.drop_front(1)) {
2726  assert(loop->getParentOp() == prevLoop && "input not contiguously nested");
2727  prevLoop = loop;
2728  }
2729 
2730  // Create the full tile loop nest.
2731  SmallVector<AffineForOp, 4> fullTileLoops;
2732  OpBuilder b(firstLoop);
2733  if (failed(createFullTiles(inputNest, fullTileLoops, b))) {
2734  if (!fullTileLoops.empty())
2735  fullTileLoops.front().erase();
2736  return failure();
2737  }
2738 
2739  // Create and insert the version select right before the root of the nest.
2740  b = OpBuilder(firstLoop);
2741  AffineIfOp ifOp = createSeparationCondition(inputNest, b);
2742  if (!ifOp) {
2743  fullTileLoops.front().erase();
2744  LLVM_DEBUG(llvm::dbgs() << "All tiles are full tiles, or failure creating "
2745  "separation condition\n");
2746  return failure();
2747  }
2748 
2749  // Move the full tile into the then block.
2750  Block *thenBlock = ifOp.getThenBlock();
2751  AffineForOp outermostFullTileLoop = fullTileLoops[0];
2752  thenBlock->getOperations().splice(
2753  std::prev(thenBlock->end()),
2754  outermostFullTileLoop->getBlock()->getOperations(),
2755  Block::iterator(outermostFullTileLoop));
2756 
2757  // Move the partial tile into the else block. The partial tile is the same as
2758  // the original loop nest.
2759  Block *elseBlock = ifOp.getElseBlock();
2760  elseBlock->getOperations().splice(std::prev(elseBlock->end()),
2761  firstLoop->getBlock()->getOperations(),
2762  Block::iterator(firstLoop));
2763 
2764  if (fullTileNest)
2765  *fullTileNest = std::move(fullTileLoops);
2766 
2767  return success();
2768 }

2770 LogicalResult affine::coalescePerfectlyNestedAffineLoops(AffineForOp op) {
2771  LogicalResult result(failure());
2772  SmallVector<AffineForOp> loops;
2773  getPerfectlyNestedLoops(loops, op);
2774  if (loops.size() <= 1)
2775  return success();
2776 
2777  // Look for a band of loops that can be coalesced, i.e. perfectly nested
2778  // loops with bounds defined above some loop.
2779  // 1. For each loop, find above which parent loop its operands are
2780  // defined.
2781  SmallVector<unsigned> operandsDefinedAbove(loops.size());
2782  for (unsigned i = 0, e = loops.size(); i < e; ++i) {
2783  operandsDefinedAbove[i] = i;
2784  for (unsigned j = 0; j < i; ++j) {
2785  if (areValuesDefinedAbove(loops[i].getOperands(), loops[j].getRegion())) {
2786  operandsDefinedAbove[i] = j;
2787  break;
2788  }
2789  }
2790  }
2791 
2792  // 2. Identify bands of loops such that the operands of all of them are
2793  // defined above the first loop in the band. Traverse the nest bottom-up
2794  // so that modifications don't invalidate the inner loops.
2795  for (unsigned end = loops.size(); end > 0; --end) {
2796  unsigned start = 0;
2797  for (; start < end - 1; ++start) {
2798  auto maxPos =
2799  *std::max_element(std::next(operandsDefinedAbove.begin(), start),
2800  std::next(operandsDefinedAbove.begin(), end));
2801  if (maxPos > start)
2802  continue;
2803  assert(maxPos == start &&
2804  "expected loop bounds to be known at the start of the band");
2805  auto band = llvm::MutableArrayRef(loops.data() + start, end - start);
2806  if (succeeded(coalesceLoops(band)))
2807  result = success();
2808  break;
2809  }
2810  // If a band was found and transformed, keep looking at the loops above
2811  // the outermost transformed loop.
2812  if (start != end - 1)
2813  end = start + 1;
2814  }
2815  return result;
2816 }
