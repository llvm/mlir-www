34 /// Dialects are groups of MLIR operations, types and attributes, as well as
35 /// behavior associated with the entire group. For example, hooks into other
36 /// systems for constant folding, interfaces, default named types for asm
37 /// printing, etc.
38 ///
39 /// Instances of the dialect object are loaded in a specific MLIRContext.
40 ///

276 /// The DialectRegistry maps a dialect namespace to a constructor for the
277 /// matching dialect.
278 /// This allows for decoupling the list of dialects "available" from the
279 /// dialects loaded in the Context. The parser in particular will lazily load
280 /// dialects in the Context as operations are encountered.
281 class DialectRegistry {
282  /// Lists of interfaces that need to be registered when the dialect is loaded.
283  struct DelayedInterfaces {
284  /// Dialect interfaces.
285  SmallVector<std::pair<TypeID, DialectInterfaceAllocatorFunction>, 2>
286  dialectInterfaces;
287  /// Attribute/Operation/Type interfaces.
288  SmallVector<std::pair<TypeID, ObjectInterfaceAllocatorFunction>, 2>
289  objectInterfaces;
290  };
291 
292  using MapTy =
293  std::map<std::string, std::pair<TypeID, DialectAllocatorFunction>>;
294  using InterfaceMapTy = DenseMap<TypeID, DelayedInterfaces>;
295 
296 public:
297  explicit DialectRegistry() {}
298 
299  template <typename ConcreteDialect>
300  void insert() {
301  insert(TypeID::get<ConcreteDialect>(),
302  ConcreteDialect::getDialectNamespace(),
303  static_cast<DialectAllocatorFunction>(([](MLIRContext *ctx) {
304  // Just allocate the dialect, the context
305  // takes ownership of it.
306  return ctx->getOrLoadDialect<ConcreteDialect>();
307  })));
308  }
309 
310  template <typename ConcreteDialect, typename OtherDialect,
311  typename... MoreDialects>
312  void insert() {
313  insert<ConcreteDialect>();
314  insert<OtherDialect, MoreDialects...>();
315  }

319  /// ownership of the dialect and for delayed interface registration to happen.
320  void insert(TypeID typeID, StringRef name, DialectAllocatorFunction ctor);

324  DialectAllocatorFunctionRef getDialectAllocator(StringRef name) const;

326  // Register all dialects available in the current registry with the registry
327  // in the provided context.
328  void appendTo(DialectRegistry &destination) const {
329  for (const auto &nameAndRegistrationIt : registry)
330  destination.insert(nameAndRegistrationIt.second.first,
331  nameAndRegistrationIt.first,
332  nameAndRegistrationIt.second.second);
333  destination.interfaces.insert(interfaces.begin(), interfaces.end());
334  }
