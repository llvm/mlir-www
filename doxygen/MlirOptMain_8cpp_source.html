251 LogicalResult loadIRDLDialects(StringRef irdlFile, MLIRContext &ctx) {
252  DialectRegistry registry;
253  registry.insert<irdl::IRDLDialect>();
254  ctx.appendDialectRegistry(registry);
255 
256  // Set up the input file.
257  std::string errorMessage;
258  std::unique_ptr<MemoryBuffer> file = openInputFile(irdlFile, &errorMessage);
259  if (!file) {
260  emitError(UnknownLoc::get(&ctx)) << errorMessage;
261  return failure();
262  }
263 
264  // Give the buffer to the source manager.
265  // This will be picked up by the parser.
266  SourceMgr sourceMgr;
267  sourceMgr.AddNewSourceBuffer(std::move(file), SMLoc());
268 
269  SourceMgrDiagnosticHandler sourceMgrHandler(sourceMgr, &ctx);
270 
271  // Parse the input file.
272  OwningOpRef<ModuleOp> module(parseSourceFile<ModuleOp>(sourceMgr, &ctx));
273  if (!module)
274  return failure();
275 
276  // Load IRDL dialects.
277  return irdl::loadDialects(module.get());

280 // Return success if the module can correctly round-trip. This intended to test
281 // that the custom printers/parsers are complete.
282 static LogicalResult doVerifyRoundTrip(Operation *op,
283  const MlirOptMainConfig &config,
284  bool useBytecode) {
285  // We use a new context to avoid resource handle renaming issue in the diff.
286  MLIRContext roundtripContext;
287  OwningOpRef<Operation *> roundtripModule;
288  roundtripContext.appendDialectRegistry(
289  op->getContext()->getDialectRegistry());
290  if (op->getContext()->allowsUnregisteredDialects())
291  roundtripContext.allowUnregisteredDialects();
292  StringRef irdlFile = config.getIrdlFile();
293  if (!irdlFile.empty() && failed(loadIRDLDialects(irdlFile, roundtripContext)))
294  return failure();
295 
296  std::string testType = (useBytecode) ? "bytecode" : "textual";
297  // Print a first time with custom format (or bytecode) and parse it back to
298  // the roundtripModule.
299  {
300  std::string buffer;
301  llvm::raw_string_ostream ostream(buffer);
302  if (useBytecode) {
303  if (failed(writeBytecodeToFile(op, ostream))) {
304  op->emitOpError()
305  << "failed to write bytecode, cannot verify round-trip.\n";
306  return failure();
307  }
308  } else {
309  op->print(ostream,
310  OpPrintingFlags().printGenericOpForm().enableDebugInfo());
311  }
312  FallbackAsmResourceMap fallbackResourceMap;
313  ParserConfig parseConfig(&roundtripContext, /*verifyAfterParse=*/true,
314  &fallbackResourceMap);
315  roundtripModule = parseSourceString<Operation *>(buffer, parseConfig);
316  if (!roundtripModule) {
317  op->emitOpError() << "failed to parse " << testType
318  << " content back, cannot verify round-trip.\n";
319  return failure();
320  }
321  }
322 
323  // Print in the generic form for the reference module and the round-tripped
324  // one and compare the outputs.
325  std::string reference, roundtrip;
326  {
327  llvm::raw_string_ostream ostreamref(reference);
328  op->print(ostreamref,
329  OpPrintingFlags().printGenericOpForm().enableDebugInfo());
330  llvm::raw_string_ostream ostreamrndtrip(roundtrip);
331  roundtripModule.get()->print(
332  ostreamrndtrip,
333  OpPrintingFlags().printGenericOpForm().enableDebugInfo());
334  }
335  if (reference != roundtrip) {
336  // TODO implement a diff.
337  return op->emitOpError()
338  << testType
339  << " roundTrip testing roundtripped module differs "
340  "from reference:\n<<<<<<Reference\n"
341  << reference << "\n=====\n"
342  << roundtrip << "\n>>>>>roundtripped\n";
343  }
344 
345  return success();
346 }

348 static LogicalResult doVerifyRoundTrip(Operation *op,
349  const MlirOptMainConfig &config) {
350  auto txtStatus = doVerifyRoundTrip(op, config, /*useBytecode=*/false);
351  auto bcStatus = doVerifyRoundTrip(op, config, /*useBytecode=*/true);
352  return success(succeeded(txtStatus) && succeeded(bcStatus));
353 }

355 /// Perform the actions on the input file indicated by the command line flags
356 /// within the specified context.
357 ///
358 /// This typically parses the main source file, runs zero or more optimization
359 /// passes, then prints the output.
360 ///
361 static LogicalResult
362 performActions(raw_ostream &os,
363  const std::shared_ptr<llvm::SourceMgr> &sourceMgr,
364  MLIRContext *context, const MlirOptMainConfig &config) {
365  DefaultTimingManager tm;
366  applyDefaultTimingManagerCLOptions(tm);
367  TimingScope timing = tm.getRootScope();
368 
369  // Disable multi-threading when parsing the input file. This removes the
370  // unnecessary/costly context synchronization when parsing.
371  bool wasThreadingEnabled = context->isMultithreadingEnabled();
372  context->disableMultithreading();
373 
374  // Prepare the parser config, and attach any useful/necessary resource
375  // handlers. Unhandled external resources are treated as passthrough, i.e.
376  // they are not processed and will be emitted directly to the output
377  // untouched.
378  PassReproducerOptions reproOptions;
379  FallbackAsmResourceMap fallbackResourceMap;
380  ParserConfig parseConfig(context, /*verifyAfterParse=*/true,
381  &fallbackResourceMap);
382  if (config.shouldRunReproducer())
383  reproOptions.attachResourceParser(parseConfig);
384 
385  // Parse the input file and reset the context threading state.
386  TimingScope parserTiming = timing.nest("Parser");
387  OwningOpRef<Operation *> op = parseSourceFileForTool(
388  sourceMgr, parseConfig, !config.shouldUseExplicitModule());
389  parserTiming.stop();
390  if (!op)
391  return failure();
392 
393  // Perform round-trip verification if requested
394  if (config.shouldVerifyRoundtrip() &&
395  failed(doVerifyRoundTrip(op.get(), config)))
396  return failure();
397 
398  context->enableMultithreading(wasThreadingEnabled);
399 
400  // Prepare the pass manager, applying command-line and reproducer options.
401  PassManager pm(op.get()->getName(), PassManager::Nesting::Implicit);
402  pm.enableVerifier(config.shouldVerifyPasses());
403  if (failed(applyPassManagerCLOptions(pm)))
404  return failure();
405  pm.enableTiming(timing);
406  if (config.shouldRunReproducer() && failed(reproOptions.apply(pm)))
407  return failure();
408  if (failed(config.setupPassPipeline(pm)))
409  return failure();
410 
411  // Run the pipeline.
412  if (failed(pm.run(*op)))
413  return failure();
414 
415  // Generate reproducers if requested
416  if (!config.getReproducerFilename().empty()) {
417  StringRef anchorName = pm.getAnyOpAnchorName();
418  const auto &passes = pm.getPasses();
419  makeReproducer(anchorName, passes, op.get(),
420  config.getReproducerFilename());
421  }
422 
423  // Print the output.
424  TimingScope outputTiming = timing.nest("Output");
425  if (config.shouldEmitBytecode()) {
426  BytecodeWriterConfig writerConfig(fallbackResourceMap);
427  if (auto v = config.bytecodeVersionToEmit())
428  writerConfig.setDesiredBytecodeVersion(*v);
429  if (config.shouldElideResourceDataFromBytecode())
430  writerConfig.setElideResourceDataFlag();
431  return writeBytecodeToFile(op.get(), os, writerConfig);
432  }
433 
434  if (config.bytecodeVersionToEmit().has_value())
435  return emitError(UnknownLoc::get(pm.getContext()))
436  << "bytecode version while not emitting bytecode";
437  AsmState asmState(op.get(), OpPrintingFlags(), /*locationMap=*/nullptr,
438  &fallbackResourceMap);
439  op.get()->print(os, asmState);
440  os << '\n';
441  return success();
442 }

444 /// Parses the memory buffer. If successfully, run a series of passes against
445 /// it and print the result.
446 static LogicalResult processBuffer(raw_ostream &os,
447  std::unique_ptr<MemoryBuffer> ownedBuffer,
448  const MlirOptMainConfig &config,
449  DialectRegistry &registry,
450  llvm::ThreadPoolInterface *threadPool) {
451  // Tell sourceMgr about this buffer, which is what the parser will pick up.
452  auto sourceMgr = std::make_shared<SourceMgr>();
453  sourceMgr->AddNewSourceBuffer(std::move(ownedBuffer), SMLoc());
454 
455  // Create a context just for the current buffer. Disable threading on creation
456  // since we'll inject the thread-pool separately.
457  MLIRContext context(registry, MLIRContext::Threading::DISABLED);
458  if (threadPool)
459  context.setThreadPool(*threadPool);
460 
461  StringRef irdlFile = config.getIrdlFile();
462  if (!irdlFile.empty() && failed(loadIRDLDialects(irdlFile, context)))
463  return failure();
464 
465  // Parse the input file.
466  context.allowUnregisteredDialects(config.shouldAllowUnregisteredDialects());
467  if (config.shouldVerifyDiagnostics())
468  context.printOpOnDiagnostic(false);
469 
470  tracing::InstallDebugHandler installDebugHandler(context,
471  config.getDebugConfig());
472 
473  // If we are in verify diagnostics mode then we have a lot of work to do,
474  // otherwise just perform the actions without worrying about it.
475  if (!config.shouldVerifyDiagnostics()) {
476  SourceMgrDiagnosticHandler sourceMgrHandler(*sourceMgr, &context);
477  return performActions(os, sourceMgr, &context, config);
478  }
479 
480  SourceMgrDiagnosticVerifierHandler sourceMgrHandler(*sourceMgr, &context);
481 
482  // Do any processing requested by command line flags. We don't care whether
483  // these actions succeed or fail, we only care what diagnostics they produce
484  // and whether they match our expectations.
485  (void)performActions(os, sourceMgr, &context, config);
486 
487  // Verify the diagnostic handler to make sure that each of the diagnostics
488  // matched.
489  return sourceMgrHandler.verify();
490 }

534 LogicalResult mlir::MlirOptMain(llvm::raw_ostream &outputStream,
535  std::unique_ptr<llvm::MemoryBuffer> buffer,
536  DialectRegistry &registry,
537  const MlirOptMainConfig &config) {
538  if (config.shouldShowDialects())
539  return printRegisteredDialects(registry);
540 
541  if (config.shouldListPasses())
542  return printRegisteredPassesAndReturn();
543 
544  // The split-input-file mode is a very specific mode that slices the file
545  // up into small pieces and checks each independently.
546  // We use an explicit threadpool to avoid creating and joining/destroying
547  // threads for each of the split.
548  ThreadPoolInterface *threadPool = nullptr;
549 
550  // Create a temporary context for the sake of checking if
551  // --mlir-disable-threading was passed on the command line.
552  // We use the thread-pool this context is creating, and avoid
553  // creating any thread when disabled.
554  MLIRContext threadPoolCtx;
555  if (threadPoolCtx.isMultithreadingEnabled())
556  threadPool = &threadPoolCtx.getThreadPool();
557 
558  auto chunkFn = [&](std::unique_ptr<MemoryBuffer> chunkBuffer,
559  raw_ostream &os) {
560  return processBuffer(os, std::move(chunkBuffer), config, registry,
561  threadPool);
562  };
563  return splitAndProcessBuffer(std::move(buffer), chunkFn, outputStream,
564  config.inputSplitMarker(),
565  config.outputSplitMarker());
566 }

568 LogicalResult mlir::MlirOptMain(int argc, char **argv,
569  llvm::StringRef inputFilename,
570  llvm::StringRef outputFilename,
571  DialectRegistry &registry) {
572 
573  InitLLVM y(argc, argv);
574 
575  MlirOptMainConfig config = MlirOptMainConfig::createFromCLOptions();
576 
577  if (config.shouldShowDialects())
578  return printRegisteredDialects(registry);
579 
580  if (config.shouldListPasses())
581  return printRegisteredPassesAndReturn();
582 
583  // When reading from stdin and the input is a tty, it is often a user mistake
584  // and the process "appears to be stuck". Print a message to let the user know
585  // about it!
586  if (inputFilename == "-" &&
587  sys::Process::FileDescriptorIsDisplayed(fileno(stdin)))
588  llvm::errs() << "(processing input from stdin now, hit ctrl-c/ctrl-d to "
589  "interrupt)\n";
590 
591  // Set up the input file.
592  std::string errorMessage;
593  auto file = openInputFile(inputFilename, &errorMessage);
594  if (!file) {
595  llvm::errs() << errorMessage << "\n";
596  return failure();
597  }
598 
599  auto output = openOutputFile(outputFilename, &errorMessage);
600  if (!output) {
601  llvm::errs() << errorMessage << "\n";
602  return failure();
603  }
604  if (failed(MlirOptMain(output->os(), std::move(file), registry, config)))
605  return failure();
606
