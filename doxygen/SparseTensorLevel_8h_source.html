18 /// The base class for all types of sparse tensor levels. It provides interfaces
19 /// to query the loop range (see `peekRangeAt`) and look up the coordinates (see
20 /// `peekCrdAt`).
21 class SparseTensorLevel {
22  SparseTensorLevel(SparseTensorLevel &&) = delete;
23  SparseTensorLevel(const SparseTensorLevel &) = delete;
24  SparseTensorLevel &operator=(SparseTensorLevel &&) = delete;
25  SparseTensorLevel &operator=(const SparseTensorLevel &) = delete;

28  virtual ~SparseTensorLevel() = default;

30  std::string toString() const {
31  return std::string(toMLIRString(lt)) + "[" + std::to_string(tid) + "," +
32  std::to_string(lvl) + "]";

35  virtual Value peekCrdAt(OpBuilder &b, Location l, ValueRange batchPrefix,
36  Value iv) const = 0;
37 
38  /// Peeks the lower and upper bound to *fully* traverse the level with
39  /// the given position `p` that the immediate parent level is current at.
40  /// Returns a pair of values for *posLo* and *loopHi* respectively.
41  ///
42  /// For a dense level, the *posLo* is the linearized position at beginning,
43  /// while *loopHi* is the largest *coordinate*, it also implies that the
44  /// smallest *coordinate* to start the loop is 0.
45  ///
46  /// For a sparse level, [posLo, loopHi) specifies the range of index pointer
47  /// to load coordinate from the coordinate buffer.
48  ///
49  /// `bound` is only used when the level is `non-unique` and deduplication is
50  /// required. It specifies the max upper bound of the non-unique segment.
51  virtual std::pair<Value, Value> peekRangeAt(OpBuilder &b, Location l,
52  ValueRange batchPrefix, Value p,
53  Value segHi = Value()) const = 0;
54 
55  Level getLevel() const { return lvl; }
56  LevelType getLT() const { return lt; }
57  Value getSize() const { return lvlSize; }
58

104  unsigned extraCursorCnt = 0)

106  extraCursorCnt + wrap.cursorValsCnt,

129  // Sets the iterate to the specified position.
130  void seek(ValueRange vals) {
131  assert(vals.size() == cursorValsCnt);
132  std::copy(vals.begin(), vals.end(), cursorValsStorageRef.begin());
133  // Now that the iterator is re-positioned, the coordinate becomes invalid.
134  crd = nullptr;
135  }

214  // Returns a pair of values for *upper*, *lower* bound respectively.
215  virtual std::pair<Value, Value> genForCond(OpBuilder &b, Location l) {
216  assert(randomAccessible());
217  // Random-access iterator is traversed by coordinate, i.e., [curCrd, UB).
218  return {getCrd(), upperBound(b, l)};
219  }
220 
221  // Generates a bool value for scf::ConditionOp.
222  std::pair<Value, ValueRange> genWhileCond(OpBuilder &b, Location l,
223  ValueRange vs) {
224  ValueRange rem = linkNewScope(vs);
225  return std::make_pair(genNotEnd(b, l), rem);
226  }

228  // Generate a conditional it.next() in the following form
229  //
230  // if (cond)
231  // yield it.next
232  // else
233  // yield it
234  //
235  // The function is virtual to allow alternative implementation. For example,
236  // if it.next() is trivial to compute, we can use a select operation instead.
237  // E.g.,
238  //
239  // it = select cond ? it+1 : it
240  virtual ValueRange forwardIf(OpBuilder &b, Location l, Value cond);
241 
242  // Update the SSA value for the iterator after entering a new scope.
243  ValueRange linkNewScope(ValueRange pos) {
244  assert(!randomAccessible() && "random accessible iterators are traversed "
245  "by coordinate, call locate() instead.");
246  seek(pos.take_front(cursorValsCnt));
247  return pos.drop_front(cursorValsCnt);
248  };

273  // iterator. Only loop variants should be included.
274  //
275  // For trivial iterators, it is the position; for dedup iterators, it consists
276  // of the positon and the segment high, for non-empty subsection iterator, it
277  // is the metadata that specifies the subsection.
278  // Note that the wrapped iterator shares the same storage to maintain itVals

281  const unsigned cursorValsCnt;
