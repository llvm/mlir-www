49 static std::optional<SmallVector<int64_t>> getTargetShape(VectorType vecType) {
50  LLVM_DEBUG(llvm::dbgs() << "Get target shape\n");
51  if (vecType.isScalable()) {
52  LLVM_DEBUG(llvm::dbgs()
53  << "--scalable vectors are not supported -> BAIL\n");
54  return std::nullopt;
55  }
56  SmallVector<int64_t> unrollShape = llvm::to_vector<4>(vecType.getShape());
57  std::optional<SmallVector<int64_t>> targetShape = SmallVector<int64_t>(
58  1, mlir::spirv::getComputeVectorSize(vecType.getShape().back()));
59  if (!targetShape) {
60  LLVM_DEBUG(llvm::dbgs() << "--no unrolling target shape defined\n");
61  return std::nullopt;
62  }
63  auto maybeShapeRatio = computeShapeRatio(unrollShape, *targetShape);
64  if (!maybeShapeRatio) {
65  LLVM_DEBUG(llvm::dbgs()
66  << "--could not compute integral shape ratio -> BAIL\n");
67  return std::nullopt;
68  }
69  if (llvm::all_of(*maybeShapeRatio, [](int64_t v) { return v == 1; })) {
70  LLVM_DEBUG(llvm::dbgs() << "--no unrolling needed -> SKIP\n");
71  return std::nullopt;
72  }
73  LLVM_DEBUG(llvm::dbgs()
74  << "--found an integral shape ratio to unroll to -> SUCCESS\n");
75  return targetShape;
76 }

78 /// Checks that `candidates` extension requirements are possible to be satisfied
79 /// with the given `targetEnv`.
80 ///
81 /// `candidates` is a vector of vector for extension requirements following
82 /// ((Extension::A OR Extension::B) AND (Extension::C OR Extension::D))
83 /// convention.
84 template <typename LabelT>
85 static LogicalResult checkExtensionRequirements(
86  LabelT label, const spirv::TargetEnv &targetEnv,
87  const spirv::SPIRVType::ExtensionArrayRefVector &candidates) {
88  for (const auto &ors : candidates) {
89  if (targetEnv.allows(ors))
90  continue;
91 
92  LLVM_DEBUG({
93  SmallVector<StringRef> extStrings;
94  for (spirv::Extension ext : ors)
95  extStrings.push_back(spirv::stringifyExtension(ext));
96 
97  llvm::dbgs() << label << " illegal: requires at least one extension in ["
98  << llvm::join(extStrings, ", ")
99  << "] but none allowed in target environment\n";
100  });
101  return failure();
102  }
103  return success();
104 }

107 /// with the given `isAllowedFn`.
108 ///
109 /// `candidates` is a vector of vector for capability requirements following
110 /// ((Capability::A OR Capability::B) AND (Capability::C OR Capability::D))
111 /// convention.
112 template <typename LabelT>
113 static LogicalResult checkCapabilityRequirements(
114  LabelT label, const spirv::TargetEnv &targetEnv,
115  const spirv::SPIRVType::CapabilityArrayRefVector &candidates) {
116  for (const auto &ors : candidates) {
117  if (targetEnv.allows(ors))

119 
120  LLVM_DEBUG({
121  SmallVector<StringRef> capStrings;
122  for (spirv::Capability cap : ors)

124 
125  llvm::dbgs() << label << " illegal: requires at least one capability in ["
126  << llvm::join(capStrings, ", ")
127  << "] but none allowed in target environment\n";
128  });
129  return failure();
130  }
131  return success();
132 }

168 // TODO: This is a utility function that should probably be exposed by the
169 // SPIR-V dialect. Keeping it local till the use case arises.
170 static std::optional<int64_t>
171 getTypeNumBytes(const SPIRVConversionOptions &options, Type type) {
172  if (isa<spirv::ScalarType>(type)) {
173  auto bitWidth = type.getIntOrFloatBitWidth();
174  // According to the SPIR-V spec:
175  // "There is no physical size or bit pattern defined for values with boolean
176  // type. If they are stored (in conjunction with OpVariable), they can only
177  // be used with logical addressing operations, not physical, and only with
178  // non-externally visible shader Storage Classes: Workgroup, CrossWorkgroup,
179  // Private, Function, Input, and Output."
180  if (bitWidth == 1)
181  return std::nullopt;
182  return bitWidth / 8;
183  }
184 
185  // Handle 8-bit floats.
186  if (options.emulateUnsupportedFloatTypes && isa<FloatType>(type)) {
187  auto bitWidth = type.getIntOrFloatBitWidth();
188  if (bitWidth == 8)
189  return bitWidth / 8;
190  return std::nullopt;
191  }
192 
193  if (auto complexType = dyn_cast<ComplexType>(type)) {
194  auto elementSize = getTypeNumBytes(options, complexType.getElementType());
195  if (!elementSize)
196  return std::nullopt;
197  return 2 * *elementSize;
198  }
199 
200  if (auto vecType = dyn_cast<VectorType>(type)) {
201  auto elementSize = getTypeNumBytes(options, vecType.getElementType());
202  if (!elementSize)
203  return std::nullopt;
204  return vecType.getNumElements() * *elementSize;
205  }
206 
207  if (auto memRefType = dyn_cast<MemRefType>(type)) {
208  // TODO: Layout should also be controlled by the ABI attributes. For now
209  // using the layout from MemRef.
210  int64_t offset;
211  SmallVector<int64_t, 4> strides;
212  if (!memRefType.hasStaticShape() ||
213  failed(memRefType.getStridesAndOffset(strides, offset)))
214  return std::nullopt;
215 
216  // To get the size of the memref object in memory, the total size is the
217  // max(stride * dimension-size) computed for all dimensions times the size
218  // of the element.
219  auto elementSize = getTypeNumBytes(options, memRefType.getElementType());
220  if (!elementSize)
221  return std::nullopt;
222 
223  if (memRefType.getRank() == 0)
224  return elementSize;
225 
226  auto dims = memRefType.getShape();
227  if (llvm::is_contained(dims, ShapedType::kDynamic) ||
228  ShapedType::isDynamic(offset) ||
229  llvm::is_contained(strides, ShapedType::kDynamic))
230  return std::nullopt;
231 
232  int64_t memrefSize = -1;
233  for (const auto &shape : enumerate(dims))
234  memrefSize = std::max(memrefSize, shape.value() * strides[shape.index()]);
235 
236  return (offset + memrefSize) * *elementSize;
237  }
238 
239  if (auto tensorType = dyn_cast<TensorType>(type)) {
240  if (!tensorType.hasStaticShape())
241  return std::nullopt;
242 
243  auto elementSize = getTypeNumBytes(options, tensorType.getElementType());
244  if (!elementSize)
245  return std::nullopt;
246 
247  int64_t size = *elementSize;
248  for (auto shape : tensorType.getShape())
249  size *= shape;
250 
251  return size;
252  }
253 
254  // TODO: Add size computation for other types.
255  return std::nullopt;
256 }

258 /// Converts a scalar `type` to a suitable type under the given `targetEnv`.
259 static Type
260 convertScalarType(const spirv::TargetEnv &targetEnv,
261  const SPIRVConversionOptions &options, spirv::ScalarType type,
262  std::optional<spirv::StorageClass> storageClass = {}) {
263  // Get extension and capability requirements for the given type.
264  SmallVector<ArrayRef<spirv::Extension>, 1> extensions;
265  SmallVector<ArrayRef<spirv::Capability>, 2> capabilities;
266  type.getExtensions(extensions, storageClass);
267  type.getCapabilities(capabilities, storageClass);
268 
269  // If all requirements are met, then we can accept this type as-is.
270  if (succeeded(checkCapabilityRequirements(type, targetEnv, capabilities)) &&
271  succeeded(checkExtensionRequirements(type, targetEnv, extensions)))
272  return type;
273 
274  // Otherwise we need to adjust the type, which really means adjusting the
275  // bitwidth given this is a scalar type.
276  if (!options.emulateLT32BitScalarTypes)
277  return nullptr;
278 
279  // We only emulate narrower scalar types here and do not truncate results.
280  if (type.getIntOrFloatBitWidth() > 32) {
281  LLVM_DEBUG(llvm::dbgs()
282  << type
283  << " not converted to 32-bit for SPIR-V to avoid truncation\n");
284  return nullptr;
285  }
286 
287  if (auto floatType = dyn_cast<FloatType>(type)) {
288  LLVM_DEBUG(llvm::dbgs() << type << " converted to 32-bit for SPIR-V\n");
289  return Builder(targetEnv.getContext()).getF32Type();
290  }
291 
292  auto intType = cast<IntegerType>(type);
293  LLVM_DEBUG(llvm::dbgs() << type << " converted to 32-bit for SPIR-V\n");
294  return IntegerType::get(targetEnv.getContext(), /*width=*/32,
295  intType.getSignedness());
296 }

298 /// Converts a sub-byte integer `type` to i32 regardless of target environment.
299 /// Returns a nullptr for unsupported integer types, including non sub-byte
300 /// types.
301 ///
302 /// Note that we don't recognize sub-byte types in `spirv::ScalarType` and use
303 /// the above given that these sub-byte types are not supported at all in
304 /// SPIR-V; there are no compute/storage capability for them like other
305 /// supported integer types.
306 static Type convertSubByteIntegerType(const SPIRVConversionOptions &options,
307  IntegerType type) {
308  if (type.getWidth() > 8) {
309  LLVM_DEBUG(llvm::dbgs() << "not a subbyte type\n");
310  return nullptr;
311  }
312  if (options.subByteTypeStorage != SPIRVSubByteTypeStorage::Packed) {
313  LLVM_DEBUG(llvm::dbgs() << "unsupported sub-byte storage kind\n");
314  return nullptr;
315  }
316 
317  if (!llvm::isPowerOf2_32(type.getWidth())) {
318  LLVM_DEBUG(llvm::dbgs()
319  << "unsupported non-power-of-two bitwidth in sub-byte" << type
320  << "\n");
321  return nullptr;
322  }
323 
324  LLVM_DEBUG(llvm::dbgs() << type << " converted to 32-bit for SPIR-V\n");
325  return IntegerType::get(type.getContext(), /*width=*/32,
326  type.getSignedness());
327 }

329 /// Converts 8-bit float types to integer types with the same bit width.
330 /// Returns a nullptr for unsupported 8-bit float types.
331 static Type convert8BitFloatType(const SPIRVConversionOptions &options,
332  FloatType type) {
333  if (!options.emulateUnsupportedFloatTypes)
334  return nullptr;
335  // F8 types are converted to integer types with the same bit width.
336  if (isa<Float8E5M2Type, Float8E4M3Type, Float8E4M3FNType, Float8E5M2FNUZType,
337  Float8E4M3FNUZType, Float8E4M3B11FNUZType, Float8E3M4Type,
338  Float8E8M0FNUType>(type))
339  return IntegerType::get(type.getContext(), type.getWidth());
340  LLVM_DEBUG(llvm::dbgs() << "unsupported 8-bit float type: " << type << "\n");
341  return nullptr;
342 }

344 /// Returns a type with the same shape but with any 8-bit float element type
345 /// converted to the same bit width integer type. This is a noop when the
346 /// element type is not the 8-bit float type or emulation flag is set to false.
347 static ShapedType
348 convertShaped8BitFloatType(ShapedType type,
349  const SPIRVConversionOptions &options) {
350  if (!options.emulateUnsupportedFloatTypes)
351  return type;
352  Type srcElementType = type.getElementType();
353  Type convertedElementType = nullptr;
354  // F8 types are converted to integer types with the same bit width.
355  if (isa<Float8E5M2Type, Float8E4M3Type, Float8E4M3FNType, Float8E5M2FNUZType,
356  Float8E4M3FNUZType, Float8E4M3B11FNUZType, Float8E3M4Type,
357  Float8E8M0FNUType>(srcElementType))
358  convertedElementType = IntegerType::get(
359  type.getContext(), srcElementType.getIntOrFloatBitWidth());
360 
361  if (!convertedElementType)
362  return type;
363 
364  return type.clone(convertedElementType);
365 }

380 /// Converts a vector `type` to a suitable type under the given `targetEnv`.
381 static Type
382 convertVectorType(const spirv::TargetEnv &targetEnv,
383  const SPIRVConversionOptions &options, VectorType type,
384  std::optional<spirv::StorageClass> storageClass = {}) {
385  type = cast<VectorType>(convertIndexElementType(type, options));
386  type = cast<VectorType>(convertShaped8BitFloatType(type, options));
387  auto scalarType = dyn_cast_or_null<spirv::ScalarType>(type.getElementType());
388  if (!scalarType) {
389  // If this is not a spec allowed scalar type, try to handle sub-byte integer
390  // types.
391  auto intType = dyn_cast<IntegerType>(type.getElementType());
392  if (!intType) {
393  LLVM_DEBUG(llvm::dbgs()
394  << type
395  << " illegal: cannot convert non-scalar element type\n");
396  return nullptr;
397  }
398 
399  Type elementType = convertSubByteIntegerType(options, intType);
400  if (!elementType)
401  return nullptr;
402 
403  if (type.getRank() <= 1 && type.getNumElements() == 1)
404  return elementType;
405 
406  if (type.getNumElements() > 4) {
407  LLVM_DEBUG(llvm::dbgs()
408  << type << " illegal: > 4-element unimplemented\n");
409  return nullptr;
410  }
411 
412  return VectorType::get(type.getShape(), elementType);
413  }
414 
415  if (type.getRank() <= 1 && type.getNumElements() == 1)
416  return convertScalarType(targetEnv, options, scalarType, storageClass);
417 
418  if (!spirv::CompositeType::isValid(type)) {
419  LLVM_DEBUG(llvm::dbgs()
420  << type << " illegal: not a valid composite type\n");
421  return nullptr;
422  }
423 
424  // Get extension and capability requirements for the given type.
425  SmallVector<ArrayRef<spirv::Extension>, 1> extensions;
426  SmallVector<ArrayRef<spirv::Capability>, 2> capabilities;
427  cast<spirv::CompositeType>(type).getExtensions(extensions, storageClass);
428  cast<spirv::CompositeType>(type).getCapabilities(capabilities, storageClass);
429 
430  // If all requirements are met, then we can accept this type as-is.
431  if (succeeded(checkCapabilityRequirements(type, targetEnv, capabilities)) &&
432  succeeded(checkExtensionRequirements(type, targetEnv, extensions)))
433  return type;
434 
435  auto elementType =
436  convertScalarType(targetEnv, options, scalarType, storageClass);
437  if (elementType)
438  return VectorType::get(type.getShape(), elementType);
439  return nullptr;
440 }

466 /// Converts a tensor `type` to a suitable type under the given `targetEnv`.
467 ///
468 /// Note that this is mainly for lowering constant tensors. In SPIR-V one can
469 /// create composite constants with OpConstantComposite to embed relative large
470 /// constant values and use OpCompositeExtract and OpCompositeInsert to
471 /// manipulate, like what we do for vectors.
472 static Type convertTensorType(const spirv::TargetEnv &targetEnv,
473  const SPIRVConversionOptions &options,
474  TensorType type) {
475  // TODO: Handle dynamic shapes.
476  if (!type.hasStaticShape()) {
477  LLVM_DEBUG(llvm::dbgs()
478  << type << " illegal: dynamic shape unimplemented\n");
479  return nullptr;
480  }
481 
482  type = cast<TensorType>(convertIndexElementType(type, options));
483  type = cast<TensorType>(convertShaped8BitFloatType(type, options));
484  auto scalarType = dyn_cast_or_null<spirv::ScalarType>(type.getElementType());
485  if (!scalarType) {
486  LLVM_DEBUG(llvm::dbgs()
487  << type << " illegal: cannot convert non-scalar element type\n");
488  return nullptr;
489  }
490 
491  std::optional<int64_t> scalarSize = getTypeNumBytes(options, scalarType);
492  std::optional<int64_t> tensorSize = getTypeNumBytes(options, type);
493  if (!scalarSize || !tensorSize) {
494  LLVM_DEBUG(llvm::dbgs()
495  << type << " illegal: cannot deduce element count\n");
496  return nullptr;
497  }
498 
499  int64_t arrayElemCount = *tensorSize / *scalarSize;
500  if (arrayElemCount == 0) {
501  LLVM_DEBUG(llvm::dbgs()
502  << type << " illegal: cannot handle zero-element tensors\n");
503  return nullptr;
504  }
505 
506  Type arrayElemType = convertScalarType(targetEnv, options, scalarType);
507  if (!arrayElemType)
508  return nullptr;
509  std::optional<int64_t> arrayElemSize =
510  getTypeNumBytes(options, arrayElemType);
511  if (!arrayElemSize) {
512  LLVM_DEBUG(llvm::dbgs()
513  << type << " illegal: cannot deduce converted element size\n");
514  return nullptr;
515  }
516 
517  return spirv::ArrayType::get(arrayElemType, arrayElemCount);
518 }

520 static Type convertBoolMemrefType(const spirv::TargetEnv &targetEnv,
521  const SPIRVConversionOptions &options,
522  MemRefType type,
523  spirv::StorageClass storageClass) {
524  unsigned numBoolBits = options.boolNumBits;
525  if (numBoolBits != 8) {
526  LLVM_DEBUG(llvm::dbgs()
527  << "using non-8-bit storage for bool types unimplemented");
528  return nullptr;
529  }
530  auto elementType = dyn_cast<spirv::ScalarType>(
531  IntegerType::get(type.getContext(), numBoolBits));
532  if (!elementType)
533  return nullptr;
534  Type arrayElemType =
535  convertScalarType(targetEnv, options, elementType, storageClass);
536  if (!arrayElemType)
537  return nullptr;
538  std::optional<int64_t> arrayElemSize =
539  getTypeNumBytes(options, arrayElemType);
540  if (!arrayElemSize) {
541  LLVM_DEBUG(llvm::dbgs()
542  << type << " illegal: cannot deduce converted element size\n");
543  return nullptr;
544  }
545 
546  if (!type.hasStaticShape()) {
547  // For OpenCL Kernel, dynamic shaped memrefs convert into a pointer pointing
548  // to the element.
549  if (targetEnv.allows(spirv::Capability::Kernel))
550  return spirv::PointerType::get(arrayElemType, storageClass);
551  int64_t stride = needsExplicitLayout(storageClass) ? *arrayElemSize : 0;
552  auto arrayType = spirv::RuntimeArrayType::get(arrayElemType, stride);
553  // For Vulkan we need extra wrapping struct and array to satisfy interface
554  // needs.
555  return wrapInStructAndGetPointer(arrayType, storageClass);
556  }
557 
558  if (type.getNumElements() == 0) {
559  LLVM_DEBUG(llvm::dbgs()
560  << type << " illegal: zero-element memrefs are not supported\n");
561  return nullptr;
562  }
563 
564  int64_t memrefSize = llvm::divideCeil(type.getNumElements() * numBoolBits, 8);
565  int64_t arrayElemCount = llvm::divideCeil(memrefSize, *arrayElemSize);
566  int64_t stride = needsExplicitLayout(storageClass) ? *arrayElemSize : 0;
567  auto arrayType = spirv::ArrayType::get(arrayElemType, arrayElemCount, stride);
568  if (targetEnv.allows(spirv::Capability::Kernel))
569  return spirv::PointerType::get(arrayType, storageClass);
570  return wrapInStructAndGetPointer(arrayType, storageClass);
571 }

573 static Type convertSubByteMemrefType(const spirv::TargetEnv &targetEnv,
574  const SPIRVConversionOptions &options,
575  MemRefType type,
576  spirv::StorageClass storageClass) {
577  IntegerType elementType = cast<IntegerType>(type.getElementType());
578  Type arrayElemType = convertSubByteIntegerType(options, elementType);
579  if (!arrayElemType)
580  return nullptr;
581  int64_t arrayElemSize = *getTypeNumBytes(options, arrayElemType);
582 
583  if (!type.hasStaticShape()) {
584  // For OpenCL Kernel, dynamic shaped memrefs convert into a pointer pointing
585  // to the element.
586  if (targetEnv.allows(spirv::Capability::Kernel))
587  return spirv::PointerType::get(arrayElemType, storageClass);
588  int64_t stride = needsExplicitLayout(storageClass) ? arrayElemSize : 0;
589  auto arrayType = spirv::RuntimeArrayType::get(arrayElemType, stride);
590  // For Vulkan we need extra wrapping struct and array to satisfy interface
591  // needs.
592  return wrapInStructAndGetPointer(arrayType, storageClass);
593  }
594 
595  if (type.getNumElements() == 0) {
596  LLVM_DEBUG(llvm::dbgs()
597  << type << " illegal: zero-element memrefs are not supported\n");
598  return nullptr;
599  }
600 
601  int64_t memrefSize =
602  llvm::divideCeil(type.getNumElements() * elementType.getWidth(), 8);
603  int64_t arrayElemCount = llvm::divideCeil(memrefSize, arrayElemSize);
604  int64_t stride = needsExplicitLayout(storageClass) ? arrayElemSize : 0;
605  auto arrayType = spirv::ArrayType::get(arrayElemType, arrayElemCount, stride);
606  if (targetEnv.allows(spirv::Capability::Kernel))
607  return spirv::PointerType::get(arrayType, storageClass);
608  return wrapInStructAndGetPointer(arrayType, storageClass);
609 }

611 static Type convertMemrefType(const spirv::TargetEnv &targetEnv,
612  const SPIRVConversionOptions &options,
613  MemRefType type) {
614  auto attr = dyn_cast_or_null<spirv::StorageClassAttr>(type.getMemorySpace());
615  if (!attr) {
616  LLVM_DEBUG(
617  llvm::dbgs()
618  << type
619  << " illegal: expected memory space to be a SPIR-V storage class "
620  "attribute; please use MemorySpaceToStorageClassConverter to map "
621  "numeric memory spaces beforehand\n");
622  return nullptr;
623  }
624  spirv::StorageClass storageClass = attr.getValue();
625 
626  if (isa<IntegerType>(type.getElementType())) {
627  if (type.getElementTypeBitWidth() == 1)
628  return convertBoolMemrefType(targetEnv, options, type, storageClass);
629  if (type.getElementTypeBitWidth() < 8)
630  return convertSubByteMemrefType(targetEnv, options, type, storageClass);
631  }
632 
633  Type arrayElemType;
634  Type elementType = type.getElementType();
635  if (auto vecType = dyn_cast<VectorType>(elementType)) {
636  arrayElemType =
637  convertVectorType(targetEnv, options, vecType, storageClass);
638  } else if (auto complexType = dyn_cast<ComplexType>(elementType)) {
639  arrayElemType =
640  convertComplexType(targetEnv, options, complexType, storageClass);
641  } else if (auto scalarType = dyn_cast<spirv::ScalarType>(elementType)) {
642  arrayElemType =
643  convertScalarType(targetEnv, options, scalarType, storageClass);
644  } else if (auto indexType = dyn_cast<IndexType>(elementType)) {
645  type = cast<MemRefType>(convertIndexElementType(type, options));
646  arrayElemType = type.getElementType();
647  } else if (auto floatType = dyn_cast<FloatType>(elementType)) {
648  // Hnadle 8 bit float types.
649  type = cast<MemRefType>(convertShaped8BitFloatType(type, options));
650  arrayElemType = type.getElementType();
651  } else {
652  LLVM_DEBUG(
653  llvm::dbgs()
654  << type
655  << " unhandled: can only convert scalar or vector element type\n");
656  return nullptr;
657  }
658  if (!arrayElemType)
659  return nullptr;
660 
661  std::optional<int64_t> arrayElemSize =
662  getTypeNumBytes(options, arrayElemType);
663  if (!arrayElemSize) {
664  LLVM_DEBUG(llvm::dbgs()
665  << type << " illegal: cannot deduce converted element size\n");
666  return nullptr;
667  }
668 
669  if (!type.hasStaticShape()) {
670  // For OpenCL Kernel, dynamic shaped memrefs convert into a pointer pointing
671  // to the element.
672  if (targetEnv.allows(spirv::Capability::Kernel))
673  return spirv::PointerType::get(arrayElemType, storageClass);
674  int64_t stride = needsExplicitLayout(storageClass) ? *arrayElemSize : 0;
675  auto arrayType = spirv::RuntimeArrayType::get(arrayElemType, stride);
676  // For Vulkan we need extra wrapping struct and array to satisfy interface
677  // needs.
678  return wrapInStructAndGetPointer(arrayType, storageClass);
679  }
680 
681  std::optional<int64_t> memrefSize = getTypeNumBytes(options, type);
682  if (!memrefSize) {
683  LLVM_DEBUG(llvm::dbgs()
684  << type << " illegal: cannot deduce element count\n");
685  return nullptr;
686  }
687 
688  if (*memrefSize == 0) {
689  LLVM_DEBUG(llvm::dbgs()
690  << type << " illegal: zero-element memrefs are not supported\n");
691  return nullptr;
692  }
693 
694  int64_t arrayElemCount = llvm::divideCeil(*memrefSize, *arrayElemSize);
695  int64_t stride = needsExplicitLayout(storageClass) ? *arrayElemSize : 0;
696  auto arrayType = spirv::ArrayType::get(arrayElemType, arrayElemCount, stride);
697  if (targetEnv.allows(spirv::Capability::Kernel))
698  return spirv::PointerType::get(arrayType, storageClass);
699  return wrapInStructAndGetPointer(arrayType, storageClass);
700 }

706 /// Converts the given `inputs` to the original source `type` considering the
707 /// `targetEnv`'s capabilities.
708 ///
709 /// This function is meant to be used for source materialization in type
710 /// converters. When the type converter needs to materialize a cast op back
711 /// to some original source type, we need to check whether the original source
712 /// type is supported in the target environment. If so, we can insert legal
713 /// SPIR-V cast ops accordingly.
714 ///
715 /// Note that in SPIR-V the capabilities for storage and compute are separate.
716 /// This function is meant to handle the **compute** side; so it does not
717 /// involve storage classes in its logic. The storage side is expected to be
718 /// handled by MemRef conversion logic.
719 static Value castToSourceType(const spirv::TargetEnv &targetEnv,
720  OpBuilder &builder, Type type, ValueRange inputs,
721  Location loc) {
722  // We can only cast one value in SPIR-V.
723  if (inputs.size() != 1) {
724  auto castOp =
725  UnrealizedConversionCastOp::create(builder, loc, type, inputs);
726  return castOp.getResult(0);
727  }
728  Value input = inputs.front();
729 
730  // Only support integer types for now. Floating point types to be implemented.
731  if (!isa<IntegerType>(type)) {
732  auto castOp =
733  UnrealizedConversionCastOp::create(builder, loc, type, inputs);
734  return castOp.getResult(0);
735  }
736  auto inputType = cast<IntegerType>(input.getType());
737 
738  auto scalarType = dyn_cast<spirv::ScalarType>(type);
739  if (!scalarType) {
740  auto castOp =
741  UnrealizedConversionCastOp::create(builder, loc, type, inputs);
742  return castOp.getResult(0);
743  }
744 
745  // Only support source type with a smaller bitwidth. This would mean we are
746  // truncating to go back so we don't need to worry about the signedness.
747  // For extension, we cannot have enough signal here to decide which op to use.
748  if (inputType.getIntOrFloatBitWidth() < scalarType.getIntOrFloatBitWidth()) {
749  auto castOp =
750  UnrealizedConversionCastOp::create(builder, loc, type, inputs);
751  return castOp.getResult(0);
752  }
753 
754  // Boolean values would need to use different ops than normal integer values.
755  if (type.isInteger(1)) {
756  Value one = spirv::ConstantOp::getOne(inputType, loc, builder);
757  return spirv::IEqualOp::create(builder, loc, input, one);
758  }
759 
760  // Check that the source integer type is supported by the environment.
761  SmallVector<ArrayRef<spirv::Extension>, 1> exts;
762  SmallVector<ArrayRef<spirv::Capability>, 2> caps;
763  scalarType.getExtensions(exts);
764  scalarType.getCapabilities(caps);
765  if (failed(checkCapabilityRequirements(type, targetEnv, caps)) ||
766  failed(checkExtensionRequirements(type, targetEnv, exts))) {
767  auto castOp =
768  UnrealizedConversionCastOp::create(builder, loc, type, inputs);
769  return castOp.getResult(0);
770  }
771 
772  // We've already made sure this is truncating previously, so we don't need to
773  // care about signedness here. Still try to use a corresponding op for better
774  // consistency though.
775  if (type.isSignedInteger()) {
776  return spirv::SConvertOp::create(builder, loc, type, input);
777  }
778  return spirv::UConvertOp::create(builder, loc, type, input);
779 }

910 /// A pattern for rewriting function signature to convert arguments of functions
911 /// to be of valid SPIR-V types.
912 struct FuncOpConversion final : OpConversionPattern<func::FuncOp> {
913  using OpConversionPattern<func::FuncOp>::OpConversionPattern;
914 
915  LogicalResult
916  matchAndRewrite(func::FuncOp funcOp, OpAdaptor adaptor,
917  ConversionPatternRewriter &rewriter) const override {
918  FunctionType fnType = funcOp.getFunctionType();
919  if (fnType.getNumResults() > 1)
920  return failure();
921 
922  TypeConverter::SignatureConversion signatureConverter(
923  fnType.getNumInputs());
924  for (const auto &argType : enumerate(fnType.getInputs())) {
925  auto convertedType = getTypeConverter()->convertType(argType.value());
926  if (!convertedType)
927  return failure();
928  signatureConverter.addInputs(argType.index(), convertedType);
929  }
930 
931  Type resultType;
932  if (fnType.getNumResults() == 1) {
933  resultType = getTypeConverter()->convertType(fnType.getResult(0));
934  if (!resultType)
935  return failure();
936  }
937 
938  // Create the converted spirv.func op.
939  auto newFuncOp = spirv::FuncOp::create(
940  rewriter, funcOp.getLoc(), funcOp.getName(),
941  rewriter.getFunctionType(signatureConverter.getConvertedTypes(),
942  resultType ? TypeRange(resultType)
943  : TypeRange()));
944 
945  // Copy over all attributes other than the function name and type.
946  for (const auto &namedAttr : funcOp->getAttrs()) {
947  if (namedAttr.getName() != funcOp.getFunctionTypeAttrName() &&
948  namedAttr.getName() != SymbolTable::getSymbolAttrName())
949  newFuncOp->setAttr(namedAttr.getName(), namedAttr.getValue());
950  }
951 
952  rewriter.inlineRegionBefore(funcOp.getBody(), newFuncOp.getBody(),
953  newFuncOp.end());
954  if (failed(rewriter.convertRegionTypes(
955  &newFuncOp.getBody(), *getTypeConverter(), &signatureConverter)))
956  return failure();
957  rewriter.eraseOp(funcOp);
958  return success();
959  }
960 };

962 /// A pattern for rewriting function signature to convert vector arguments of
963 /// functions to be of valid types
964 struct FuncOpVectorUnroll final : OpRewritePattern<func::FuncOp> {
965  using OpRewritePattern::OpRewritePattern;
966 
967  LogicalResult matchAndRewrite(func::FuncOp funcOp,
968  PatternRewriter &rewriter) const override {
969  FunctionType fnType = funcOp.getFunctionType();
970 
971  // TODO: Handle declarations.
972  if (funcOp.isDeclaration()) {
973  LLVM_DEBUG(llvm::dbgs()
974  << fnType << " illegal: declarations are unsupported\n");
975  return failure();
976  }
977 
978  // Create a new func op with the original type and copy the function body.
979  auto newFuncOp = func::FuncOp::create(rewriter, funcOp.getLoc(),
980  funcOp.getName(), fnType);
981  rewriter.inlineRegionBefore(funcOp.getBody(), newFuncOp.getBody(),
982  newFuncOp.end());
983 
984  Location loc = newFuncOp.getBody().getLoc();
985 
986  Block &entryBlock = newFuncOp.getBlocks().front();
987  OpBuilder::InsertionGuard guard(rewriter);
988  rewriter.setInsertionPointToStart(&entryBlock);
989 
990  TypeConverter::SignatureConversion oneToNTypeMapping(
991  fnType.getInputs().size());
992 
993  // For arguments that are of illegal types and require unrolling.
994  // `unrolledInputNums` stores the indices of arguments that result from
995  // unrolling in the new function signature. `newInputNo` is a counter.
996  SmallVector<size_t> unrolledInputNums;
997  size_t newInputNo = 0;
998 
999  // For arguments that are of legal types and do not require unrolling.
1000  // `tmpOps` stores a mapping from temporary operations that serve as
1001  // placeholders for new arguments that will be added later. These operations
1002  // will be erased once the entry block's argument list is updated.
1003  llvm::SmallDenseMap<Operation *, size_t> tmpOps;
1004 
1005  // This counts the number of new operations created.
1006  size_t newOpCount = 0;
1007 
1008  // Enumerate through the arguments.
1009  for (auto [origInputNo, origType] : enumerate(fnType.getInputs())) {
1010  // Check whether the argument is of vector type.
1011  auto origVecType = dyn_cast<VectorType>(origType);
1012  if (!origVecType) {
1013  // We need a placeholder for the old argument that will be erased later.
1014  Value result = arith::ConstantOp::create(
1015  rewriter, loc, origType, rewriter.getZeroAttr(origType));
1016  rewriter.replaceAllUsesWith(newFuncOp.getArgument(origInputNo), result);
1017  tmpOps.insert({result.getDefiningOp(), newInputNo});
1018  oneToNTypeMapping.addInputs(origInputNo, origType);
1019  ++newInputNo;
1020  ++newOpCount;
1021  continue;
1022  }
1023  // Check whether the vector needs unrolling.
1024  auto targetShape = getTargetShape(origVecType);
1025  if (!targetShape) {
1026  // We need a placeholder for the old argument that will be erased later.
1027  Value result = arith::ConstantOp::create(
1028  rewriter, loc, origType, rewriter.getZeroAttr(origType));
1029  rewriter.replaceAllUsesWith(newFuncOp.getArgument(origInputNo), result);
1030  tmpOps.insert({result.getDefiningOp(), newInputNo});
1031  oneToNTypeMapping.addInputs(origInputNo, origType);
1032  ++newInputNo;
1033  ++newOpCount;
1034  continue;
1035  }
1036  VectorType unrolledType =
1037  VectorType::get(*targetShape, origVecType.getElementType());
1038  auto originalShape =
1039  llvm::to_vector_of<int64_t, 4>(origVecType.getShape());
1040 
1041  // Prepare the result vector.
1042  Value result = arith::ConstantOp::create(
1043  rewriter, loc, origVecType, rewriter.getZeroAttr(origVecType));
1044  ++newOpCount;
1045  // Prepare the placeholder for the new arguments that will be added later.
1046  Value dummy = arith::ConstantOp::create(
1047  rewriter, loc, unrolledType, rewriter.getZeroAttr(unrolledType));
1048  ++newOpCount;
1049 
1050  // Create the `vector.insert_strided_slice` ops.
1051  SmallVector<int64_t> strides(targetShape->size(), 1);
1052  SmallVector<Type> newTypes;
1053  for (SmallVector<int64_t> offsets :
1054  StaticTileOffsetRange(originalShape, *targetShape)) {
1055  result = vector::InsertStridedSliceOp::create(rewriter, loc, dummy,
1056  result, offsets, strides);
1057  newTypes.push_back(unrolledType);
1058  unrolledInputNums.push_back(newInputNo);
1059  ++newInputNo;
1060  ++newOpCount;
1061  }
1062  rewriter.replaceAllUsesWith(newFuncOp.getArgument(origInputNo), result);
1063  oneToNTypeMapping.addInputs(origInputNo, newTypes);
1064  }
1065 
1066  // Change the function signature.
1067  auto convertedTypes = oneToNTypeMapping.getConvertedTypes();
1068  auto newFnType = fnType.clone(convertedTypes, fnType.getResults());
1069  rewriter.modifyOpInPlace(newFuncOp,
1070  [&] { newFuncOp.setFunctionType(newFnType); });
1071 
1072  // Update the arguments in the entry block.
1073  entryBlock.eraseArguments(0, fnType.getNumInputs());
1074  SmallVector<Location> locs(convertedTypes.size(), newFuncOp.getLoc());
1075  entryBlock.addArguments(convertedTypes, locs);
1076 
1077  // Replace all uses of placeholders for initially legal arguments with their
1078  // original function arguments (that were added to `newFuncOp`).
1079  for (auto &[placeholderOp, argIdx] : tmpOps) {
1080  if (!placeholderOp)
1081  continue;
1082  Value replacement = newFuncOp.getArgument(argIdx);
1083  rewriter.replaceAllUsesWith(placeholderOp->getResult(0), replacement);
1084  }
1085 
1086  // Replace dummy operands of new `vector.insert_strided_slice` ops with
1087  // their corresponding new function arguments. The new
1088  // `vector.insert_strided_slice` ops are inserted only into the entry block,
1089  // so iterating over that block is sufficient.
1090  size_t unrolledInputIdx = 0;
1091  for (auto [count, op] : enumerate(entryBlock.getOperations())) {
1092  Operation &curOp = op;
1093  // Since all newly created operations are in the beginning, reaching the
1094  // end of them means that any later `vector.insert_strided_slice` should
1095  // not be touched.
1096  if (count >= newOpCount)
1097  continue;
1098  if (auto vecOp = dyn_cast<vector::InsertStridedSliceOp>(op)) {
1099  size_t unrolledInputNo = unrolledInputNums[unrolledInputIdx];
1100  rewriter.modifyOpInPlace(&curOp, [&] {
1101  curOp.setOperand(0, newFuncOp.getArgument(unrolledInputNo));
1102  });
1103  ++unrolledInputIdx;
1104  }
1105  }
1106 
1107  // Erase the original funcOp. The `tmpOps` do not need to be erased since
1108  // they have no uses and will be handled by dead-code elimination.
1109  rewriter.eraseOp(funcOp);
1110  return success();
1111  }
1112 };

1118 /// A pattern for rewriting function signature and the return op to convert
1119 /// vectors to be of valid types.
1120 struct ReturnOpVectorUnroll final : OpRewritePattern<func::ReturnOp> {
1121  using OpRewritePattern::OpRewritePattern;
1122 
1123  LogicalResult matchAndRewrite(func::ReturnOp returnOp,
1124  PatternRewriter &rewriter) const override {
1125  // Check whether the parent funcOp is valid.
1126  auto funcOp = dyn_cast<func::FuncOp>(returnOp->getParentOp());
1127  if (!funcOp)
1128  return failure();
1129 
1130  FunctionType fnType = funcOp.getFunctionType();
1131  TypeConverter::SignatureConversion oneToNTypeMapping(
1132  fnType.getResults().size());
1133  Location loc = returnOp.getLoc();
1134 
1135  // For the new return op.
1136  SmallVector<Value> newOperands;
1137 
1138  // Enumerate through the results.
1139  for (auto [origResultNo, origType] : enumerate(fnType.getResults())) {
1140  // Check whether the argument is of vector type.
1141  auto origVecType = dyn_cast<VectorType>(origType);
1142  if (!origVecType) {
1143  oneToNTypeMapping.addInputs(origResultNo, origType);
1144  newOperands.push_back(returnOp.getOperand(origResultNo));
1145  continue;
1146  }
1147  // Check whether the vector needs unrolling.
1148  auto targetShape = getTargetShape(origVecType);
1149  if (!targetShape) {
1150  // The original argument can be used.
1151  oneToNTypeMapping.addInputs(origResultNo, origType);
1152  newOperands.push_back(returnOp.getOperand(origResultNo));
1153  continue;
1154  }
1155  VectorType unrolledType =
1156  VectorType::get(*targetShape, origVecType.getElementType());
1157 
1158  // Create `vector.extract_strided_slice` ops to form legal vectors from
1159  // the original operand of illegal type.
1160  auto originalShape =
1161  llvm::to_vector_of<int64_t, 4>(origVecType.getShape());
1162  SmallVector<int64_t> strides(originalShape.size(), 1);
1163  SmallVector<int64_t> extractShape(originalShape.size(), 1);
1164  extractShape.back() = targetShape->back();
1165  SmallVector<Type> newTypes;
1166  Value returnValue = returnOp.getOperand(origResultNo);
1167  for (SmallVector<int64_t> offsets :
1168  StaticTileOffsetRange(originalShape, *targetShape)) {
1169  Value result = vector::ExtractStridedSliceOp::create(
1170  rewriter, loc, returnValue, offsets, extractShape, strides);
1171  if (originalShape.size() > 1) {
1172  SmallVector<int64_t> extractIndices(originalShape.size() - 1, 0);
1173  result =
1174  vector::ExtractOp::create(rewriter, loc, result, extractIndices);
1175  }
1176  newOperands.push_back(result);
1177  newTypes.push_back(unrolledType);
1178  }
1179  oneToNTypeMapping.addInputs(origResultNo, newTypes);
1180  }
1181 
1182  // Change the function signature.

1566  // QueryMinVersionInterface/QueryMaxVersionInterface are available to all
1567  // SPIR-V versions.

1573  << spirv::stringifyVersion(*minVersion) << "\n");

1575  }

1577  if (auto maxVersionIfx = dyn_cast<spirv::QueryMaxVersionInterface>(op)) {
1578  std::optional<spirv::Version> maxVersion = maxVersionIfx.getMaxVersion();
1579  if (maxVersion && *maxVersion < this->targetEnv.getVersion()) {
1580  LLVM_DEBUG(llvm::dbgs()
1581  << op->getName() << " illegal: requiring max version "
1582  << spirv::stringifyVersion(*maxVersion) << "\n");
1583  return false;
1584  }
1585  }

1588  // implementing QueryExtensionInterface do not require extensions to be
1589  // available.
1590  if (auto extensions = dyn_cast<spirv::QueryExtensionInterface>(op))
1591  if (failed(checkExtensionRequirements(op->getName(), this->targetEnv,
1592  extensions.getExtensions())))

1596  // implementing QueryCapabilityInterface do not require capabilities to be
1597  // available.
1598  if (auto capabilities = dyn_cast<spirv::QueryCapabilityInterface>(op))
1599  if (failed(checkCapabilityRequirements(op->getName(), this->targetEnv,
1600  capabilities.getCapabilities())))
1601  return false;
1602

1604  valueTypes.append(op->operand_type_begin(), op->operand_type_end());
