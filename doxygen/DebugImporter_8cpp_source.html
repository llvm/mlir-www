28 Location DebugImporter::translateFuncLocation(llvm::Function *func) {
29  llvm::DISubprogram *subprogram = func->getSubprogram();
30  if (!subprogram)
31  return UnknownLoc::get(context);
32 
33  // Add a fused location to link the subprogram information.
34  StringAttr funcName = StringAttr::get(context, subprogram->getName());
35  StringAttr fileName = StringAttr::get(context, subprogram->getFilename());
36  return FusedLocWith<DISubprogramAttr>::get(
37  {NameLoc::get(funcName),
38  FileLineColLoc::get(fileName, subprogram->getLine(), /*column=*/0)},
39  translate(subprogram), context);
40 }

60 DICompositeTypeAttr DebugImporter::translateImpl(llvm::DICompositeType *node) {
61  std::optional<DIFlags> flags = symbolizeDIFlags(node->getFlags());
62  SmallVector<DINodeAttr> elements;
63  for (llvm::DINode *element : node->getElements()) {
64  assert(element && "expected a non-null element type");
65  elements.push_back(translate(element));
66  }
67  // Drop the elements parameter if any of the elements are invalid.
68  if (llvm::is_contained(elements, nullptr))
69  elements.clear();
70  DITypeAttr baseType = translate(node->getBaseType());
71  // Arrays require a base type, otherwise the debug metadata is considered to
72  // be malformed.
73  if (node->getTag() == llvm::dwarf::DW_TAG_array_type && !baseType)
74  return nullptr;
75  return DICompositeTypeAttr::get(
76  context, node->getTag(), /*recId=*/{},
77  getStringAttrOrNull(node->getRawName()), translate(node->getFile()),
78  node->getLine(), translate(node->getScope()), baseType,
79  flags.value_or(DIFlags::Zero), node->getSizeInBits(),
80  node->getAlignInBits(), elements);
81 }

177 DISubprogramAttr DebugImporter::translateImpl(llvm::DISubprogram *node) {
178  // Only definitions require a distinct identifier.
179  mlir::DistinctAttr id;
180  if (node->isDistinct())
181  id = getOrCreateDistinctID(node);
182  // Return nullptr if the scope or type is invalid.
183  DIScopeAttr scope = translate(node->getScope());
184  if (node->getScope() && !scope)
185  return nullptr;
186  std::optional<DISubprogramFlags> subprogramFlags =
187  symbolizeDISubprogramFlags(node->getSubprogram()->getSPFlags());
188  assert(subprogramFlags && "expected valid subprogram flags");
189  DISubroutineTypeAttr type = translate(node->getType());
190  if (node->getType() && !type)
191  return nullptr;
192  return DISubprogramAttr::get(context, id, translate(node->getUnit()), scope,
193  getStringAttrOrNull(node->getRawName()),
194  getStringAttrOrNull(node->getRawLinkageName()),
195  translate(node->getFile()), node->getLine(),
196  node->getScopeLine(), *subprogramFlags, type);
197 }

199 DISubrangeAttr DebugImporter::translateImpl(llvm::DISubrange *node) {
200  auto getIntegerAttrOrNull = [&](llvm::DISubrange::BoundType data) {
201  if (auto *constInt = llvm::dyn_cast_or_null<llvm::ConstantInt *>(data))
202  return IntegerAttr::get(IntegerType::get(context, 64),
203  constInt->getSExtValue());
204  return IntegerAttr();
205  };
206  IntegerAttr count = getIntegerAttrOrNull(node->getCount());
207  IntegerAttr upperBound = getIntegerAttrOrNull(node->getUpperBound());
208  // Either count or the upper bound needs to be present. Otherwise, the
209  // metadata is invalid. The conversion might fail due to unsupported DI nodes.
210  if (!count && !upperBound)
211  return {};
212  return DISubrangeAttr::get(
213  context, count, getIntegerAttrOrNull(node->getLowerBound()), upperBound,
214  getIntegerAttrOrNull(node->getStride()));
215 }

217 DISubroutineTypeAttr
218 DebugImporter::translateImpl(llvm::DISubroutineType *node) {
219  SmallVector<DITypeAttr> types;
220  for (llvm::DIType *type : node->getTypeArray()) {
221  if (!type) {
222  // A nullptr entry may appear at the beginning or the end of the
223  // subroutine types list modeling either a void result type or the type of
224  // a variadic argument. Translate the nullptr to an explicit
225  // DINullTypeAttr since the attribute list cannot contain a nullptr entry.
226  types.push_back(DINullTypeAttr::get(context));
227  continue;
228  }
229  types.push_back(translate(type));
230  }
231  // Return nullptr if any of the types is invalid.
232  if (llvm::is_contained(types, nullptr))
233  return nullptr;
234  return DISubroutineTypeAttr::get(context, node->getCC(), types);
235 }

241 DINodeAttr DebugImporter::translate(llvm::DINode *node) {
242  if (!node)
243  return nullptr;
244 
245  // Check for a cached instance.
246  if (DINodeAttr attr = nodeToAttr.lookup(node))
247  return attr;
248 
249  // If the node type is capable of being recursive, check if it's seen before.
250  auto recSelfCtor = getRecSelfConstructor(node);
251  if (recSelfCtor) {
252  // If a cyclic dependency is detected since the same node is being traversed
253  // twice, emit a recursive self type, and mark the duplicate node on the
254  // translationStack so it can emit a recursive decl type.
255  auto [iter, inserted] = translationStack.try_emplace(node, nullptr);
256  if (!inserted) {
257  // The original node may have already been assigned a recursive ID from
258  // a different self-reference. Use that if possible.
259  DistinctAttr recId = iter->second;
260  if (!recId) {
261  recId = DistinctAttr::create(UnitAttr::get(context));
262  iter->second = recId;
263  }
264  unboundRecursiveSelfRefs.back().insert(recId);
265  return cast<DINodeAttr>(recSelfCtor(recId));
266  }
267  }
268 
269  unboundRecursiveSelfRefs.emplace_back();
270 
271  auto guard = llvm::make_scope_exit([&]() {
272  if (recSelfCtor)
273  translationStack.pop_back();
274 
275  // Copy unboundRecursiveSelfRefs down to the previous level.
276  if (unboundRecursiveSelfRefs.size() == 1)
277  assert(unboundRecursiveSelfRefs.back().empty() &&
278  "internal error: unbound recursive self reference at top level.");
279  else
280  unboundRecursiveSelfRefs[unboundRecursiveSelfRefs.size() - 2].insert(
281  unboundRecursiveSelfRefs.back().begin(),
282  unboundRecursiveSelfRefs.back().end());
283  unboundRecursiveSelfRefs.pop_back();
284  });
285 
286  // Convert the debug metadata if possible.
287  auto translateNode = [this](llvm::DINode *node) -> DINodeAttr {
288  if (auto *casted = dyn_cast<llvm::DIBasicType>(node))
289  return translateImpl(casted);
290  if (auto *casted = dyn_cast<llvm::DICompileUnit>(node))
291  return translateImpl(casted);
292  if (auto *casted = dyn_cast<llvm::DICompositeType>(node))
293  return translateImpl(casted);
294  if (auto *casted = dyn_cast<llvm::DIDerivedType>(node))
295  return translateImpl(casted);
296  if (auto *casted = dyn_cast<llvm::DIFile>(node))
297  return translateImpl(casted);
298  if (auto *casted = dyn_cast<llvm::DIGlobalVariable>(node))
299  return translateImpl(casted);
300  if (auto *casted = dyn_cast<llvm::DILabel>(node))
301  return translateImpl(casted);
302  if (auto *casted = dyn_cast<llvm::DILexicalBlock>(node))
303  return translateImpl(casted);
304  if (auto *casted = dyn_cast<llvm::DILexicalBlockFile>(node))
305  return translateImpl(casted);
306  if (auto *casted = dyn_cast<llvm::DILocalVariable>(node))
307  return translateImpl(casted);
308  if (auto *casted = dyn_cast<llvm::DIModule>(node))
309  return translateImpl(casted);
310  if (auto *casted = dyn_cast<llvm::DINamespace>(node))
311  return translateImpl(casted);
312  if (auto *casted = dyn_cast<llvm::DISubprogram>(node))
313  return translateImpl(casted);
314  if (auto *casted = dyn_cast<llvm::DISubrange>(node))
315  return translateImpl(casted);
316  if (auto *casted = dyn_cast<llvm::DISubroutineType>(node))
317  return translateImpl(casted);
318  return nullptr;
319  };
320  if (DINodeAttr attr = translateNode(node)) {
321  // If this node was marked as recursive, set its recId.
322  if (auto recType = dyn_cast<DIRecursiveTypeAttrInterface>(attr)) {
323  if (DistinctAttr recId = translationStack.lookup(node)) {
324  attr = cast<DINodeAttr>(recType.withRecId(recId));
325  // Remove the unbound recursive ID from the set of unbound self
326  // references in the translation stack.
327  unboundRecursiveSelfRefs.back().erase(recId);
328  }
329  }
330 
331  // Only cache fully self-contained nodes.
332  if (unboundRecursiveSelfRefs.back().empty())
333  nodeToAttr.try_emplace(node, attr);
334  return attr;
335  }
336  return nullptr;
337 }

343 Location DebugImporter::translateLoc(llvm::DILocation *loc) {
344  if (!loc)
345  return UnknownLoc::get(context);
346 
347  // Get the file location of the instruction.
348  Location result = FileLineColLoc::get(context, loc->getFilename(),
349  loc->getLine(), loc->getColumn());
350 
351  // Add scope information.
352  assert(loc->getScope() && "expected non-null scope");
353  result = FusedLocWith<DIScopeAttr>::get({result}, translate(loc->getScope()),
354  context);
355 
356  // Add call site information, if available.
357  if (llvm::DILocation *inlinedAt = loc->getInlinedAt())
358  result = CallSiteLoc::get(result, translateLoc(inlinedAt));
359 
360  return result;
361 }

363 DIExpressionAttr DebugImporter::translateExpression(llvm::DIExpression *node) {
364  SmallVector<DIExpressionElemAttr> ops;
365 
366  // Begin processing the operations.
367  for (const llvm::DIExpression::ExprOperand &op : node->expr_ops()) {
368  SmallVector<uint64_t> operands;
369  operands.reserve(op.getNumArgs());
370  for (const auto &i : llvm::seq(op.getNumArgs()))
371  operands.push_back(op.getArg(i));
372  const auto attr = DIExpressionElemAttr::get(context, op.getOp(), operands);
373  ops.push_back(attr);
374  }
375  return DIExpressionAttr::get(context, ops);
376 }
