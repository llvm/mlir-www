63/// Sets OpenMP offload module interface attributes on a ModuleOp, shared
64/// between Flang and Clang (CIR) frontends.
65[[maybe_unused]] static void
66setOffloadModuleInterfaceAttributes(ModuleOp module, OffloadModuleOpts opts) {
67 if (auto offloadMod =
68 llvm::dyn_cast<OffloadModuleInterface>(module.getOperation())) {
69 offloadMod.setIsTargetDevice(opts.OpenMPIsTargetDevice);
70 offloadMod.setIsGPU(opts.OpenMPIsGPU);
71 if (opts.OpenMPForceUSM)
72 offloadMod.setRequires(ClauseRequires::unified_shared_memory);
73 offloadMod.setFlags(opts.OpenMPTargetDebug, opts.OpenMPTeamSubscription,
74 opts.OpenMPThreadSubscription, opts.OpenMPNoThreadState,
75 opts.OpenMPNoNestedParallelism, opts.OpenMPVersion,
76 opts.NoGPULib);
77 if (opts.OpenMPIsTargetDevice && !opts.OMPHostIRFile.empty())
78 offloadMod.setHostIRFilePath(opts.OMPHostIRFile);
79
80 auto strTriples = llvm::to_vector(
81 llvm::map_range(opts.OMPTargetTriples, [](llvm::Triple triple) {
82 return triple.normalize();
83 }));
84 offloadMod.setTargetTriples(strTriples);
85 }
86}

88[[maybe_unused]] static void setOpenMPVersionAttribute(ModuleOp module,
89 int64_t version) {
90 module.getOperation()->setAttr(
91 StringAttr::get(module.getContext(), llvm::Twine{"omp.version"}),
92 VersionAttr::get(module.getContext(), version));
93}
94
95[[maybe_unused]] static int64_t
96getOpenMPVersionAttribute(ModuleOp module, int64_t fallback = -1) {
97 if (Attribute verAttr = module->getAttr("omp.version"))
98 return llvm::cast<VersionAttr>(verAttr).getVersion();
99 return fallback;
100}
