37 /// Replace `loop` with a new loop that has a different init operand at
38 /// position `index`. The body of this loop is moved over to the new loop.
39 ///
40 /// `newInitOperands` specifies the replacement "init" operands.
41 /// `newYieldValue` is the replacement yield value of the loop at position
42 /// `index`.
43 static scf::ForOp replaceWithDifferentYield(RewriterBase &rewriter,
44  scf::ForOp loop,
45  Value newInitOperand,
46  unsigned index,
47  Value newYieldValue) {
48  OpBuilder::InsertionGuard g(rewriter);
49  rewriter.setInsertionPoint(loop.getOperation());
50  auto inits = llvm::to_vector(loop.getInits());
51 
52  // Replace the init value with the new operand.
53  assert(index < inits.size());
54  inits[index] = newInitOperand;
55 
56  scf::ForOp newLoop = scf::ForOp::create(
57  rewriter, loop.getLoc(), loop.getLowerBound(), loop.getUpperBound(),
58  loop.getStep(), inits, [](OpBuilder &, Location, Value, ValueRange) {});
59 
60  // Generate the new yield with the replaced operand.
61  auto yieldOp = cast<scf::YieldOp>(loop.getBody()->getTerminator());
62  yieldOp.setOperand(index, newYieldValue);
63 
64  // Move the loop body to the new op.
65  rewriter.mergeBlocks(loop.getBody(), newLoop.getBody(),
66  newLoop.getBody()->getArguments());
67 
68  // Replace the old loop.
69  rewriter.replaceOp(loop.getOperation(), newLoop->getResults());
70  return newLoop;
71 }

73 // Hoist out a pair of corresponding vector.extract+vector.broadcast
74 // operations. This function transforms a loop like this:
75 // %res = scf.for _ = _ to _ step _ iter_args(%iarg = %v) -> (t1) {
76 // %e = vector.extract %iarg : t1 to t2
77 // %u = "some_use"(%e) : (t2) -> t2
78 // %b = vector.broadcast %u : t2 to t1
79 // scf.yield %b : t1
80 // }
81 // into the following:
82 // %e = vector.extract %v: t1 to t2
83 // %res' = scf.for _ = _ to _ step _ iter_args(%iarg = %e) -> (t2) {
84 // %u' = "some_use"(%iarg) : (t2) -> t2
85 // scf.yield %u' : t2
86 // }
87 // %res = vector.broadcast %res' : t2 to t1
88 void mlir::linalg::hoistRedundantVectorBroadcasts(RewriterBase &rewriter,
89  Operation *root) {
90  bool changed = true;
91  while (changed) {
92  changed = false;
93  // First move loop invariant ops outside of their loop. This needs to be
94  // done before as we cannot move ops without interrupting the function walk.
95  root->walk(
96  [&](LoopLikeOpInterface loopLike) { moveLoopInvariantCode(loopLike); });
97 
98  root->walk([&](vector::ExtractOp extractOp) {
99  LLVM_DEBUG(DBGS() << "Candidate for hoisting: "
100  << *extractOp.getOperation() << "\n");
101 
102  auto loop = dyn_cast<scf::ForOp>(extractOp->getParentOp());
103  if (!loop)
104  return WalkResult::advance();
105 
106  // Check that the vector to extract from is a BlockArgument.
107  auto blockArg = dyn_cast<BlockArgument>(extractOp.getVector());
108  if (!blockArg)
109  return WalkResult::advance();
110 
111  // Check that the blockArg is an iter_arg of the loop.
112  OpOperand *initArg = loop.getTiedLoopInit(blockArg);
113  if (!initArg)
114  return WalkResult::advance();
115 
116  // If the iter_arg does not have only one use, it won't be possible to
117  // hoist the extractOp out.
118  if (!blockArg.hasOneUse())
119  return WalkResult::advance();
120 
121  unsigned index = blockArg.getArgNumber() - loop.getNumInductionVars();
122 
123  // Check that the loop yields a broadcast that has just one use.
124  Operation *yieldedVal =
125  loop.getTiedLoopYieldedValue(blockArg)->get().getDefiningOp();
126  auto broadcast = dyn_cast<vector::BroadcastOp>(yieldedVal);
127  if (!broadcast || !broadcast.getResult().hasOneUse())
128  return WalkResult::advance();
129 
130  LLVM_DEBUG(DBGS() << "Candidate broadcast: " << broadcast << "\n");
131 
132  Type broadcastInputType = broadcast.getSourceType();
133  if (broadcastInputType != extractOp.getType())
134  return WalkResult::advance();
135 
136  // The position of the extract must be defined outside of the loop if
137  // it is dynamic.
138  for (auto operand : extractOp.getDynamicPosition())
139  if (!loop.isDefinedOutsideOfLoop(operand))
140  return WalkResult::advance();
141 
142  rewriter.modifyOpInPlace(broadcast, [&] {
143  extractOp.getVectorMutable().assign(initArg->get());
144  });
145  loop.moveOutOfLoop(extractOp);
146  rewriter.moveOpAfter(broadcast, loop);
147 
148  scf::ForOp newLoop = replaceWithDifferentYield(
149  rewriter, loop, extractOp.getResult(), index, broadcast.getSource());
150 
151  LLVM_DEBUG(DBGS() << "New loop: " << newLoop << "\n");
152 
153  rewriter.replaceAllUsesWith(newLoop.getResult(index), broadcast);
154  rewriter.modifyOpInPlace(
155  broadcast, [&] { broadcast.setOperand(newLoop.getResult(index)); });
156 
157  changed = true;
158  return WalkResult::interrupt();
159  });
160  }
161 }

163 static bool noAliasingUseInLoop(vector::TransferReadOp transferRead,
164  LoopLikeOpInterface loop) {
165  Value source = transferRead.getBase();
166 
167  // Skip view-like Ops and retrive the actual soruce Operation
168  while (auto srcOp = source.getDefiningOp<ViewLikeOpInterface>())
169  source = srcOp.getViewSource();
170 
171  llvm::SmallVector<Operation *, 32> users(source.getUsers().begin(),
172  source.getUsers().end());
173  llvm::SmallDenseSet<Operation *, 32> processed;
174  while (!users.empty()) {
175  Operation *user = users.pop_back_val();
176  // If the user has already been processed skip.
177  if (!processed.insert(user).second)
178  continue;
179  if (auto viewLike = dyn_cast<ViewLikeOpInterface>(user)) {
180  users.append(viewLike->getUsers().begin(), viewLike->getUsers().end());
181  continue;
182  }
183  if (isMemoryEffectFree(user) || isa<vector::TransferReadOp>(user))
184  continue;
185  if (!loop->isAncestor(user))
186  continue;
187  return false;
188  }
189  return true;
190 }

192 void mlir::linalg::hoistRedundantVectorTransfers(Operation *root,
193  bool verifyNonZeroTrip) {
194  bool changed = true;
195  while (changed) {
196  changed = false;
197  // First move loop invariant ops outside of their loop. This needs to be
198  // done before as we cannot move ops without interrupting the function walk.
199  root->walk(
200  [&](LoopLikeOpInterface loopLike) { moveLoopInvariantCode(loopLike); });
201 
202  // Find all loops that are certain to have non zero trip count. Any loops
203  // that are not part of this set cannot be hoisted from, since hoisting from
204  // a potentially zero trip count loop may cause a vector transfer to be
205  // executed when it shouldn't be.
206  llvm::DenseSet<LoopLikeOpInterface> definiteNonZeroTripCountLoops;
207  if (verifyNonZeroTrip) {
208  root->walk([&](LoopLikeOpInterface loopLike) {
209  std::optional<SmallVector<OpFoldResult>> lbs =
210  loopLike.getLoopLowerBounds();
211  std::optional<SmallVector<OpFoldResult>> ubs =
212  loopLike.getLoopUpperBounds();
213  // If loop bounds cannot be found, assume possibly zero trip count.
214  if (!lbs || !ubs)
215  return;
216 
217  // Otherwise, use ValueBounds to find the maximum lower bound and
218  // minimum upper bound. If the bounds are found, and maxLb is less
219  // than the minUb, then the loop will not have zero trip count.
220  for (auto [lb, ub] : llvm::zip_equal(lbs.value(), ubs.value())) {
221  FailureOr<int64_t> maxLb =
222  ValueBoundsConstraintSet::computeConstantBound(
223  presburger::BoundType::UB, lb,
224  /*stopCondition=*/nullptr, /*closedUB=*/true);
225  if (failed(maxLb))
226  return;
227  FailureOr<int64_t> minUb =
228  ValueBoundsConstraintSet::computeConstantBound(
229  presburger::BoundType::LB, ub);
230  if (failed(minUb))
231  return;
232  if (minUb.value() <= maxLb.value())
233  return;
234  definiteNonZeroTripCountLoops.insert(loopLike);
235  }
236  });
237  }
238 
239  root->walk([&](vector::TransferReadOp transferRead) {
240  if (!isa<MemRefType>(transferRead.getShapedType()))
241  return WalkResult::advance();
242 
243  LLVM_DEBUG(DBGS() << "Candidate for hoisting: "
244  << *transferRead.getOperation() << "\n");
245  auto loop = dyn_cast<LoopLikeOpInterface>(transferRead->getParentOp());
246  LLVM_DEBUG(DBGS() << "Parent op: " << *transferRead->getParentOp()
247  << "\n");
248  if (!isa_and_nonnull<scf::ForOp, affine::AffineForOp>(loop))
249  return WalkResult::advance();
250 
251  if (verifyNonZeroTrip && !definiteNonZeroTripCountLoops.contains(loop)) {
252  LLVM_DEBUG(DBGS() << "Loop may have zero trip count: " << *loop
253  << "\n");
254  return WalkResult::advance();
255  }
256 
257  LLVM_DEBUG(DBGS() << "Candidate read: " << *transferRead.getOperation()
258  << "\n");
259 
260  SetVector<Operation *> forwardSlice;
261  getForwardSlice(transferRead.getOperation(), &forwardSlice);
262 
263  // Look for the last TransferWriteOp in the forwardSlice of
264  // `transferRead` that operates on the same memref.
265  vector::TransferWriteOp transferWrite;
266  for (auto *sliceOp : llvm::reverse(forwardSlice)) {
267  auto candidateWrite = dyn_cast<vector::TransferWriteOp>(sliceOp);
268  if (!candidateWrite ||
269  candidateWrite.getBase() != transferRead.getBase())
270  continue;
271  transferWrite = candidateWrite;
272  }
273 
274  // All operands of the TransferRead must be defined outside of the loop.
275  for (auto operand : transferRead.getOperands())
276  if (!loop.isDefinedOutsideOfLoop(operand))
277  return WalkResult::advance();
278 
279  // Only hoist transfer_read / transfer_write pairs and singleton
280  // transfer_reads for now.
281  if (!transferWrite) {
282  // Make sure there are no other accesses to the memref before
283  // hoisting transfer_read.
284  if (noAliasingUseInLoop(transferRead, loop))
285  loop.moveOutOfLoop(transferRead);
286  return WalkResult::advance();
287  }
288 
289  LLVM_DEBUG(DBGS() << "Candidate: " << *transferWrite.getOperation()
290  << "\n");
291 
292  // Approximate aliasing by checking that:
293  // 1. indices, vector type and permutation map are the same (i.e., the
294  // transfer_read/transfer_write ops are matching),
295  // 2. source operands for transfer.{read|write} do not originate from
296  // nor have users that are Ops implementing ViewLikeOpInterface.
297  // 3. no other operations in the loop access the same memref except
298  // for transfer_read/transfer_write accessing statically disjoint
299  // slices.
300 
301  // Check 1.
302  if (transferRead.getIndices() != transferWrite.getIndices() ||
303  transferRead.getVectorType() != transferWrite.getVectorType() ||
304  transferRead.getPermutationMap() != transferWrite.getPermutationMap())
305  return WalkResult::advance();
306 
307  // Check 2. Note, since both xfer Ops share the source, we only need to
308  // look at one of them.
309  auto base = transferRead.getBase();
310  auto *source = base.getDefiningOp();
311  if (source) {
312  // NOTE: We treat `memref.assume_alignment` as a special case.
313  //
314  // The idea is that it is safe to look past AssumeAlignmemtOp (i.e.
315  // MemRef _before_ alignment) iff:
316  // 1. It has exactly two uses (these have to be the xfer Ops
317  // being looked at).
318  // 2. The original MemRef has only one use (i.e.
319  // AssumeAlignmentOp).
320  //
321  // Relaxing these conditions will most likely require proper alias
322  // analysis.
323  if (auto assume = dyn_cast<memref::AssumeAlignmentOp>(source)) {
324  Value memPreAlignment = assume.getMemref();
325  auto numInLoopUses =
326  llvm::count_if(base.getUses(), [&loop](OpOperand &use) {
327  return loop->isAncestor(use.getOwner());
328  });
329 
330  if (numInLoopUses && memPreAlignment.hasOneUse())
331  source = memPreAlignment.getDefiningOp();
332  }
333  if (isa_and_nonnull<ViewLikeOpInterface>(source))
334  return WalkResult::advance();
335  }
336 
337  if (llvm::any_of(base.getUsers(), llvm::IsaPred<ViewLikeOpInterface>))
338  return WalkResult::advance();
339 
340  // Check 3.
341  // TODO: may want to memoize this information for performance but it
342  // likely gets invalidated often.
343  DominanceInfo dom(loop);
344  if (!dom.properlyDominates(transferRead.getOperation(), transferWrite))
345  return WalkResult::advance();
346  for (auto &use : transferRead.getBase().getUses()) {
347  if (!loop->isAncestor(use.getOwner()))
348  continue;
349  if (use.getOwner() == transferRead.getOperation() ||
350  use.getOwner() == transferWrite.getOperation())
351  continue;
352  if (auto transferWriteUse =
353  dyn_cast<vector::TransferWriteOp>(use.getOwner())) {
354  if (!vector::isDisjointTransferSet(
355  cast<VectorTransferOpInterface>(*transferWrite),
356  cast<VectorTransferOpInterface>(*transferWriteUse),
357  /*testDynamicValueUsingBounds=*/true))
358  return WalkResult::advance();
359  } else if (auto transferReadUse =
360  dyn_cast<vector::TransferReadOp>(use.getOwner())) {
361  if (!vector::isDisjointTransferSet(
362  cast<VectorTransferOpInterface>(*transferWrite),
363  cast<VectorTransferOpInterface>(*transferReadUse),
364  /*testDynamicValueUsingBounds=*/true))
365  return WalkResult::advance();
366  } else {
367  // Unknown use, we cannot prove that it doesn't alias with the
368  // transferRead/transferWrite operations.
369  return WalkResult::advance();
370  }
371  }
372 
373  // Hoist read before.
374  loop.moveOutOfLoop(transferRead);
375 
376  // Hoist write after.
377  transferWrite->moveAfter(loop);
378 
379  // Rewrite `loop` with new yields by cloning and erase the original
380  // loop.
381  IRRewriter rewriter(transferRead.getContext());
382  NewYieldValuesFn yieldFn = [&](OpBuilder &b, Location loc,
383  ArrayRef<BlockArgument> newBBArgs) {
384  return SmallVector<Value>{transferWrite.getVector()};
385  };
386 
387  auto maybeNewLoop = loop.replaceWithAdditionalYields(
388  rewriter, transferRead.getVector(),
389  /*replaceInitOperandUsesInLoop=*/true, yieldFn);
390  if (failed(maybeNewLoop))
391  return WalkResult::interrupt();
392 
393  transferWrite.getValueToStoreMutable().assign(
394  maybeNewLoop->getOperation()->getResults().back());
395  changed = true;
396  // Need to interrupt and restart because erasing the loop messes up
397  // the walk.
398  return WalkResult::interrupt();
399  });
400  }
