25 /// Converts `atomic_rmw` that cannot be lowered to a simple atomic op with
26 /// AtomicRMWOpLowering pattern, e.g. with "minf" or "maxf" attributes, to
27 /// `generic_atomic_rmw` with the expanded code.
28 ///
29 /// %x = atomic_rmw "maxf" %fval, %F[%i] : (f32, memref<10xf32>) -> f32
30 ///
31 /// will be lowered to
32 ///
33 /// %x = std.generic_atomic_rmw %F[%i] : memref<10xf32> {
34 /// ^bb0(%current: f32):
35 /// %cmp = cmpf "ogt", %current, %fval : f32
36 /// %new_value = select %cmp, %current, %fval : f32
37 /// atomic_yield %new_value : f32
38 /// }
39 struct AtomicRMWOpConverter : public OpRewritePattern<AtomicRMWOp> {
40 public:
41  using OpRewritePattern::OpRewritePattern;
42 
43  LogicalResult matchAndRewrite(AtomicRMWOp op,
44  PatternRewriter &rewriter) const final {
45  CmpFPredicate predicate;
46  switch (op.kind()) {
47  case AtomicRMWKind::maxf:
48  predicate = CmpFPredicate::OGT;
49  break;
50  case AtomicRMWKind::minf:
51  predicate = CmpFPredicate::OLT;
52  break;
53  default:
54  return failure();
55  }
56 
57  auto loc = op.getLoc();
58  auto genericOp =
59  rewriter.create<GenericAtomicRMWOp>(loc, op.memref(), op.indices());
60  OpBuilder bodyBuilder =
61  OpBuilder::atBlockEnd(genericOp.getBody(), rewriter.getListener());
62 
63  Value lhs = genericOp.getCurrentValue();
64  Value rhs = op.value();
65  Value cmp = bodyBuilder.create<CmpFOp>(loc, predicate, lhs, rhs);
66  Value select = bodyBuilder.create<SelectOp>(loc, cmp, lhs, rhs);
67  bodyBuilder.create<AtomicYieldOp>(loc, select);
68 
69  rewriter.replaceOp(op, genericOp.getResult());
70  return success();
71  }
72 };

74 /// Converts `memref.reshape` that has a target shape of a statically-known
75 /// size to `memref.reinterpret_cast`.
76 struct MemRefReshapeOpConverter : public OpRewritePattern<memref::ReshapeOp> {
77 public:
78  using OpRewritePattern::OpRewritePattern;
79 
80  LogicalResult matchAndRewrite(memref::ReshapeOp op,
81  PatternRewriter &rewriter) const final {
82  auto shapeType = op.shape().getType().cast<MemRefType>();
83  if (!shapeType.hasStaticShape())
84  return failure();
85 
86  int64_t rank = shapeType.cast<MemRefType>().getDimSize(0);
87  SmallVector<OpFoldResult, 4> sizes, strides;
88  sizes.resize(rank);
89  strides.resize(rank);
90 
91  Location loc = op.getLoc();
92  Value stride = rewriter.create<ConstantIndexOp>(loc, 1);
93  for (int i = rank - 1; i >= 0; --i) {
94  Value size;
95  // Load dynamic sizes from the shape input, use constants for static dims.
96  if (op.getType().isDynamicDim(i)) {
97  Value index = rewriter.create<ConstantIndexOp>(loc, i);
98  size = rewriter.create<memref::LoadOp>(loc, op.shape(), index);
99  if (!size.getType().isa<IndexType>())
100  size =
101  rewriter.create<IndexCastOp>(loc, size, rewriter.getIndexType());
102  sizes[i] = size;
103  } else {
104  sizes[i] = rewriter.getIndexAttr(op.getType().getDimSize(i));
105  size = rewriter.create<ConstantOp>(loc, sizes[i].get<Attribute>());
106  }
107  strides[i] = stride;
108  if (i > 0)
109  stride = rewriter.create<MulIOp>(loc, stride, size);
110  }
111  rewriter.replaceOpWithNewOp<memref::ReinterpretCastOp>(
112  op, op.getType(), op.source(), /*offset=*/rewriter.getIndexAttr(0),
113  sizes, strides);
114  return success();
115  }
116 };

118 /// Expands SignedCeilDivIOP (n, m) into
119 /// 1) x = (m > 0) ? -1 : 1
120 /// 2) (n*m>0) ? ((n+x) / m) + 1 : - (-n / m)
121 struct SignedCeilDivIOpConverter : public OpRewritePattern<SignedCeilDivIOp> {
122 public:
123  using OpRewritePattern::OpRewritePattern;
124  LogicalResult matchAndRewrite(SignedCeilDivIOp op,
125  PatternRewriter &rewriter) const final {
126  Location loc = op.getLoc();
127  SignedCeilDivIOp signedCeilDivIOp = cast<SignedCeilDivIOp>(op);
128  Type type = signedCeilDivIOp.getType();
129  Value a = signedCeilDivIOp.lhs();
130  Value b = signedCeilDivIOp.rhs();
131  Value plusOne =
132  rewriter.create<ConstantOp>(loc, rewriter.getIntegerAttr(type, 1));
133  Value zero =
134  rewriter.create<ConstantOp>(loc, rewriter.getIntegerAttr(type, 0));
135  Value minusOne =
136  rewriter.create<ConstantOp>(loc, rewriter.getIntegerAttr(type, -1));
137  // Compute x = (b>0) ? -1 : 1.
138  Value compare = rewriter.create<CmpIOp>(loc, CmpIPredicate::sgt, b, zero);
139  Value x = rewriter.create<SelectOp>(loc, compare, minusOne, plusOne);
140  // Compute positive res: 1 + ((x+a)/b).
141  Value xPlusA = rewriter.create<AddIOp>(loc, x, a);
142  Value xPlusADivB = rewriter.create<SignedDivIOp>(loc, xPlusA, b);
143  Value posRes = rewriter.create<AddIOp>(loc, plusOne, xPlusADivB);
144  // Compute negative res: - ((-a)/b).
145  Value minusA = rewriter.create<SubIOp>(loc, zero, a);
146  Value minusADivB = rewriter.create<SignedDivIOp>(loc, minusA, b);
147  Value negRes = rewriter.create<SubIOp>(loc, zero, minusADivB);
148  // Result is (a*b>0) ? pos result : neg result.
149  // Note, we want to avoid using a*b because of possible overflow.
150  // The case that matters are a>0, a==0, a<0, b>0 and b<0. We do
151  // not particuliarly care if a*b<0 is true or false when b is zero
152  // as this will result in an illegal divide. So `a*b<0` can be reformulated
153  // as `(a<0 && b<0) || (a>0 && b>0)' or `(a<0 && b<0) || (a>0 && b>=0)'.
154  // We pick the first expression here.
155  Value aNeg = rewriter.create<CmpIOp>(loc, CmpIPredicate::slt, a, zero);
156  Value aPos = rewriter.create<CmpIOp>(loc, CmpIPredicate::sgt, a, zero);
157  Value bNeg = rewriter.create<CmpIOp>(loc, CmpIPredicate::slt, b, zero);
158  Value bPos = rewriter.create<CmpIOp>(loc, CmpIPredicate::sgt, b, zero);
159  Value firstTerm = rewriter.create<AndOp>(loc, aNeg, bNeg);
160  Value secondTerm = rewriter.create<AndOp>(loc, aPos, bPos);
161  Value compareRes = rewriter.create<OrOp>(loc, firstTerm, secondTerm);
162  Value res = rewriter.create<SelectOp>(loc, compareRes, posRes, negRes);
163  // Perform substitution and return success.
164  rewriter.replaceOp(op, {res});
165  return success();
166  }
167 };

169 /// Expands SignedFloorDivIOP (n, m) into
170 /// 1) x = (m<0) ? 1 : -1
171 /// 2) return (n*m<0) ? - ((-n+x) / m) -1 : n / m
172 struct SignedFloorDivIOpConverter : public OpRewritePattern<SignedFloorDivIOp> {
173 public:
174  using OpRewritePattern::OpRewritePattern;
175  LogicalResult matchAndRewrite(SignedFloorDivIOp op,
176  PatternRewriter &rewriter) const final {
177  Location loc = op.getLoc();
178  SignedFloorDivIOp signedFloorDivIOp = cast<SignedFloorDivIOp>(op);
179  Type type = signedFloorDivIOp.getType();
180  Value a = signedFloorDivIOp.lhs();
181  Value b = signedFloorDivIOp.rhs();
182  Value plusOne =
183  rewriter.create<ConstantOp>(loc, rewriter.getIntegerAttr(type, 1));
184  Value zero =
185  rewriter.create<ConstantOp>(loc, rewriter.getIntegerAttr(type, 0));
186  Value minusOne =
187  rewriter.create<ConstantOp>(loc, rewriter.getIntegerAttr(type, -1));
188  // Compute x = (b<0) ? 1 : -1.
189  Value compare = rewriter.create<CmpIOp>(loc, CmpIPredicate::slt, b, zero);
190  Value x = rewriter.create<SelectOp>(loc, compare, plusOne, minusOne);
191  // Compute negative res: -1 - ((x-a)/b).
192  Value xMinusA = rewriter.create<SubIOp>(loc, x, a);
193  Value xMinusADivB = rewriter.create<SignedDivIOp>(loc, xMinusA, b);
194  Value negRes = rewriter.create<SubIOp>(loc, minusOne, xMinusADivB);
195  // Compute positive res: a/b.
196  Value posRes = rewriter.create<SignedDivIOp>(loc, a, b);
197  // Result is (a*b<0) ? negative result : positive result.
198  // Note, we want to avoid using a*b because of possible overflow.
199  // The case that matters are a>0, a==0, a<0, b>0 and b<0. We do
200  // not particuliarly care if a*b<0 is true or false when b is zero
201  // as this will result in an illegal divide. So `a*b<0` can be reformulated
202  // as `(a>0 && b<0) || (a>0 && b<0)' or `(a>0 && b<0) || (a>0 && b<=0)'.
203  // We pick the first expression here.
204  Value aNeg = rewriter.create<CmpIOp>(loc, CmpIPredicate::slt, a, zero);
205  Value aPos = rewriter.create<CmpIOp>(loc, CmpIPredicate::sgt, a, zero);
206  Value bNeg = rewriter.create<CmpIOp>(loc, CmpIPredicate::slt, b, zero);
207  Value bPos = rewriter.create<CmpIOp>(loc, CmpIPredicate::sgt, b, zero);
208  Value firstTerm = rewriter.create<AndOp>(loc, aNeg, bPos);
209  Value secondTerm = rewriter.create<AndOp>(loc, aPos, bNeg);
210  Value compareRes = rewriter.create<OrOp>(loc, firstTerm, secondTerm);
211  Value res = rewriter.create<SelectOp>(loc, compareRes, negRes, posRes);
212  // Perform substitution and return success.
213  rewriter.replaceOp(op, {res});
214  return success();
215  }
216 };

218 struct StdExpandOpsPass : public StdExpandOpsBase<StdExpandOpsPass> {
219  void runOnFunction() override {
220  MLIRContext &ctx = getContext();
221 
222  RewritePatternSet patterns(&ctx);
223  populateStdExpandOpsPatterns(patterns);
224 
225  ConversionTarget target(getContext());
226 
227  target.addLegalDialect<memref::MemRefDialect, StandardOpsDialect>();
228  target.addDynamicallyLegalOp<AtomicRMWOp>([](AtomicRMWOp op) {
229  return op.kind() != AtomicRMWKind::maxf &&
230  op.kind() != AtomicRMWKind::minf;
231  });
232  target.addDynamicallyLegalOp<memref::ReshapeOp>([](memref::ReshapeOp op) {
233  return !op.shape().getType().cast<MemRefType>().hasStaticShape();
234  });
235  target.addIllegalOp<SignedCeilDivIOp>();
236  target.addIllegalOp<SignedFloorDivIOp>();
237  if (failed(
238  applyPartialConversion(getFunction(), target, std::move(patterns))))
239  signalPassFailure();
240  }
241 };
