58 // This converts the TOSA ApplyScale operator to a set of arithmetic ops,
59 // using 64-bit operations to perform the necessary multiply, bias, and shift.
60 class ApplyScaleGenericOpConverter
61  : public OpRewritePattern<tosa::ApplyScaleOp> {
62 public:
63  using OpRewritePattern<tosa::ApplyScaleOp>::OpRewritePattern;
64 
65  LogicalResult matchAndRewrite(tosa::ApplyScaleOp op,
66  PatternRewriter &rewriter) const final {
67  StringRef roundingMode = op.getRoundingMode();
68  if (roundingMode != "DOUBLE_ROUND" && roundingMode != "SINGLE_ROUND") {
69  return failure();
70  }
71 
72  Location loc = op.getLoc();
73  Value value = op.getValue();
74  Value multiplier32 = op.getMultiplier();
75 
76  Type resultTy = op.getType();
77  Type valueTy = value.getType();
78  Type i32Ty = matchContainerType(rewriter.getI32Type(), resultTy);
79  Type i64Ty = matchContainerType(rewriter.getI64Type(), resultTy);
80 
81  Value zero = getConstantValue(loc, valueTy, 0, rewriter);
82  Value one64 = getConstantValue(loc, i64Ty, 1, rewriter);
83  Value thirtyOne32 = getConstantValue(loc, i32Ty, 31, rewriter);
84 
85  Value shift32 = arith::ExtUIOp::create(rewriter, loc, i32Ty, op.getShift());
86 
87  // Compute the multiplication in 64-bits then select the high / low parts.
88  Value value64 = value;
89  if (getElementTypeOrSelf(valueTy) != rewriter.getI64Type())
90  value64 = arith::ExtSIOp::create(rewriter, loc, i64Ty, value);
91  Value multiplier64 =
92  arith::ExtSIOp::create(rewriter, loc, i64Ty, multiplier32);
93  Value multiply64 =
94  arith::MulIOp::create(rewriter, loc, value64, multiplier64);
95 
96  // Apply normal rounding.
97  Value shift64 = arith::ExtUIOp::create(rewriter, loc, i64Ty, shift32);
98  Value round = arith::ShLIOp::create(rewriter, loc, one64, shift64);
99  round = arith::ShRUIOp::create(rewriter, loc, round, one64);
100  multiply64 = arith::AddIOp::create(rewriter, loc, multiply64, round);
101 
102  // Apply double rounding if necessary.
103  if (op.getRoundingMode() == "DOUBLE_ROUND") {
104  int64_t roundInt = 1 << 30;
105  Value roundUp = getConstantValue(loc, i64Ty, roundInt, rewriter);
106  Value roundDown = getConstantValue(loc, i64Ty, -roundInt, rewriter);
107  Value positive = arith::CmpIOp::create(
108  rewriter, loc, arith::CmpIPredicate::sge, value, zero);
109  Value dir =
110  arith::SelectOp::create(rewriter, loc, positive, roundUp, roundDown);
111  Value val = arith::AddIOp::create(rewriter, loc, dir, multiply64);
112  Value valid = arith::CmpIOp::create(
113  rewriter, loc, arith::CmpIPredicate::sgt, shift32, thirtyOne32);
114  multiply64 =
115  arith::SelectOp::create(rewriter, loc, valid, val, multiply64);
116  }
117 
118  Value result64 = arith::ShRSIOp::create(rewriter, loc, multiply64, shift64);
119  Value result32 = arith::TruncIOp::create(rewriter, loc, i32Ty, result64);
120 
121  rewriter.replaceOp(op, result32);
122  return success();
123  }
124 };

126 class ApplyScale32BitOpConverter : public OpRewritePattern<tosa::ApplyScaleOp> {
127 public:
128  using OpRewritePattern<tosa::ApplyScaleOp>::OpRewritePattern;
129 
130  LogicalResult matchAndRewrite(tosa::ApplyScaleOp op,
131  PatternRewriter &rewriter) const final {
132  StringRef roundingMode = op.getRoundingMode();
133  if (roundingMode != "DOUBLE_ROUND" && roundingMode != "SINGLE_ROUND") {
134  return failure();
135  }
136 
137  Location loc = op.getLoc();
138 
139  Type resultTy = op.getType();
140  Type i32Ty = matchContainerType(rewriter.getI32Type(), resultTy);
141 
142  Value value = op.getValue();
143  if (getElementTypeOrSelf(value.getType()).getIntOrFloatBitWidth() > 32) {
144  return failure();
145  }
146 
147  Value value32 = op.getValue();
148  Value multiplier32 = op.getMultiplier();
149  Value shift32 = arith::ExtUIOp::create(rewriter, loc, i32Ty, op.getShift());
150 
151  // Constants used during the scaling operation.
152  Value zero32 = getConstantValue(loc, i32Ty, 0, rewriter);
153  Value one32 = getConstantValue(loc, i32Ty, 1, rewriter);
154  Value two32 = getConstantValue(loc, i32Ty, 2, rewriter);
155  Value thirty32 = getConstantValue(loc, i32Ty, 30, rewriter);
156  Value thirtyTwo32 = getConstantValue(loc, i32Ty, 32, rewriter);
157 
158  // Compute the multiplication in 64-bits then select the high / low parts.
159  // Grab out the high/low of the computation
160  auto value64 =
161  arith::MulSIExtendedOp::create(rewriter, loc, value32, multiplier32);
162  Value low32 = value64.getLow();
163  Value high32 = value64.getHigh();
164 
165  // Determine the direction and amount to shift the high bits.
166  Value shiftOver32 = arith::CmpIOp::create(
167  rewriter, loc, arith::CmpIPredicate::sge, shift32, thirtyTwo32);
168  Value roundHighBits = arith::CmpIOp::create(
169  rewriter, loc, arith::CmpIPredicate::sgt, shift32, thirtyTwo32);
170 
171  Value shiftHighL =
172  arith::SubIOp::create(rewriter, loc, thirtyTwo32, shift32);
173  Value shiftHighR =
174  arith::SubIOp::create(rewriter, loc, shift32, thirtyTwo32);
175 
176  shiftHighL =
177  arith::SelectOp::create(rewriter, loc, shiftOver32, zero32, shiftHighL);
178  shiftHighR =
179  arith::SelectOp::create(rewriter, loc, shiftOver32, shiftHighR, zero32);
180 
181  // Conditionally perform our double round.
182  if (op.getRoundingMode() == "DOUBLE_ROUND") {
183  Value negOne32 = getConstantValue(loc, i32Ty, -1, rewriter);
184  Value valuePositive = arith::CmpIOp::create(
185  rewriter, loc, arith::CmpIPredicate::sge, value32, zero32);
186 
187  Value roundDir = arith::SelectOp::create(rewriter, loc, valuePositive,
188  one32, negOne32);
189  roundDir =
190  arith::SelectOp::create(rewriter, loc, shiftOver32, roundDir, zero32);
191 
192  Value shiftLow = arith::ShRUIOp::create(rewriter, loc, low32, thirty32);
193  Value rounded = arith::AddIOp::create(rewriter, loc, shiftLow, roundDir);
194  Value carry = arith::ShRSIOp::create(rewriter, loc, rounded, two32);
195 
196  Value shiftRound =
197  arith::ShLIOp::create(rewriter, loc, roundDir, thirty32);
198 
199  low32 = arith::AddIOp::create(rewriter, loc, low32, shiftRound);
200  high32 = arith::AddIOp::create(rewriter, loc, high32, carry);
201  }
202 
203  // Conditionally apply rounding in the low bits.
204  {
205  Value shiftSubOne = arith::SubIOp::create(rewriter, loc, shift32, one32);
206  Value roundBit = arith::ShLIOp::create(rewriter, loc, one32, shiftSubOne);
207  roundBit = arith::SelectOp::create(rewriter, loc, roundHighBits, zero32,
208  roundBit);
209 
210  Value newLow32 = arith::AddIOp::create(rewriter, loc, low32, roundBit);
211  Value wasRounded = arith::CmpIOp::create(
212  rewriter, loc, arith::CmpIPredicate::ugt, low32, newLow32);
213  low32 = newLow32;
214 
215  Value rounded32 =
216  arith::ExtUIOp::create(rewriter, loc, i32Ty, wasRounded);
217  high32 = arith::AddIOp::create(rewriter, loc, high32, rounded32);
218  }
219 
220  // Conditionally apply rounding in the high bits.
221  {
222  Value shiftSubOne =
223  arith::SubIOp::create(rewriter, loc, shiftHighR, one32);
224  Value roundBit = arith::ShLIOp::create(rewriter, loc, one32, shiftSubOne);
225  roundBit = arith::SelectOp::create(rewriter, loc, roundHighBits, roundBit,
226  zero32);
227  high32 = arith::AddIOp::create(rewriter, loc, high32, roundBit);
228  }
229 
230  // Combine the correct high/low bits into the final rescale result.
231  high32 = arith::ShLIOp::create(rewriter, loc, high32, shiftHighL);
232  high32 = arith::ShRSIOp::create(rewriter, loc, high32, shiftHighR);
233  low32 = arith::ShRUIOp::create(rewriter, loc, low32, shift32);
234  low32 = arith::SelectOp::create(rewriter, loc, shiftOver32, zero32, low32);
235 
236  // Apply the rounding behavior and shift to the final alignment.
237  Value result = arith::AddIOp::create(rewriter, loc, low32, high32);
238 
239  // Truncate if necessary.
240  if (!getElementTypeOrSelf(resultTy).isInteger(32)) {
241  result = arith::TruncIOp::create(rewriter, loc, resultTy, result);
242  }
243 
244  rewriter.replaceOp(op, result);
245  return success();
246  }
247 };
