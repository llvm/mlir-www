27  LogicalResult matchAndRewrite(ExpandShapeOp expandShapeOp,
28  PatternRewriter &rewriter) const override {
29  RankedTensorType resultType = expandShapeOp.getResultType();

49  expandShapeOp, extractSliceOp.getSource(), mixedOffsets, mixedSizes,

51  return success();
52  }

55 /// Fold collapse_shape which only removes static dimensions of size `1`
56 /// into extract_slice.
57 struct FoldUnPaddingCollapseIntoExtract
58  : public OpRewritePattern<tensor::CollapseShapeOp> {
59  using OpRewritePattern<tensor::CollapseShapeOp>::OpRewritePattern;
60 
61  LogicalResult matchAndRewrite(tensor::CollapseShapeOp collapseShapeOp,
62  PatternRewriter &rewriter) const override {
63  auto extractSliceOp =
64  collapseShapeOp.getSrc().getDefiningOp<tensor::ExtractSliceOp>();
65  // Collapse cannot be folded away with multiple users of the extract slice
66  // and it is not necessarily beneficial to only convert the collapse into
67  // another extract slice.
68  if (!extractSliceOp || !extractSliceOp->hasOneUse())
69  return failure();
70 
71  // Only fold away simple collapse where all removed dimensions have static
72  // size `1`.
73  SliceVerificationResult res = isRankReducedType(
74  collapseShapeOp.getSrcType(), collapseShapeOp.getResultType());
75  if (res != SliceVerificationResult::Success)
76  return rewriter.notifyMatchFailure(collapseShapeOp,
77  "expected unpadding collapse");
78 
79  Value unPaddedExtractSlice = tensor::ExtractSliceOp::create(
80  rewriter, extractSliceOp.getLoc(), collapseShapeOp.getResultType(),
81  extractSliceOp.getSource(), extractSliceOp.getMixedOffsets(),
82  extractSliceOp.getMixedSizes(), extractSliceOp.getMixedStrides());
83  rewriter.replaceOp(collapseShapeOp, unPaddedExtractSlice);
84  return success();
85  }

120 /// Fold expand_shape which only adds static dimensions of size `1`
121 /// into insert_slice.

123 struct FoldPaddingExpandIntoInsert : public OpRewritePattern<OpTy> {
124  using OpRewritePattern<OpTy>::OpRewritePattern;

126  LogicalResult matchAndRewrite(OpTy insertSliceOp,
127  PatternRewriter &rewriter) const override {
128  auto expandShapeOp = insertSliceOp.getSource()
129  .template getDefiningOp<tensor::ExpandShapeOp>();
130  if (!expandShapeOp)
131  return failure();
132

148 /// Pattern to bubble up a tensor.expand_shape op through a producer
149 /// tensor.collapse_shape op that has non intersecting reassociations.
150 struct BubbleUpExpandThroughParallelCollapse
151  : public OpRewritePattern<tensor::ExpandShapeOp> {
152  using OpRewritePattern<tensor::ExpandShapeOp>::OpRewritePattern;
153 
154  LogicalResult matchAndRewrite(tensor::ExpandShapeOp expandOp,
155  PatternRewriter &rewriter) const override {
156  auto collapseOp =
157  expandOp.getSrc().getDefiningOp<tensor::CollapseShapeOp>();
158  if (!collapseOp)
159  return failure();
160  auto expandReInds = expandOp.getReassociationIndices();
161  auto collapseReInds = collapseOp.getReassociationIndices();
162 
163  // Special case where the collapsed tensor to expand is a 0-D tensor,
164  // then the reassociation maps will be empty and not produce valid results.
165  if (expandReInds.size() == 0) {
166  return failure();
167  }
168 
169  // Reshapes are parallel to each other (by construction the number of
170  // reassociations specified in the collapse and expand are the same), if at
171  // any position
172  // 1. either the reassociation indices are of the same size, or
173  // 2. either the reassociation in the collapse or the expand is of size 1.
174  ArrayRef<int64_t> staticSourceSize = collapseOp.getSrcType().getShape();
175  ArrayRef<int64_t> staticResultSize = expandOp.getStaticOutputShape();
176  for (auto [expandReassociation, collapseReassociation] :
177  llvm::zip_equal(expandReInds, collapseReInds)) {
178  if (collapseReassociation.size() == expandReassociation.size()) {
179  // Even if the reassociations are the same, the collapse/expand should
180  // result in the same dimensions. i.e 4x8x2 into 64 should be expanded
181  // into 4x8x2 again. In presense of dynamic dimensions one can only
182  // verify "equality" when there is only one dynamic dimension present,
183  // and all other static dimensions are equal.
184  ArrayRef<int64_t> collapsedStaticShapes = staticSourceSize.slice(
185  collapseReassociation.front(), collapseReassociation.size());
186  int64_t numCollapsedDynamic =
187  llvm::count_if(collapsedStaticShapes, ShapedType::isDynamic);
188  ArrayRef<int64_t> expandedStaticShapes = staticResultSize.slice(
189  expandReassociation.front(), expandReassociation.size());
190  int64_t numExpandedDynamic =
191  llvm::count_if(expandedStaticShapes, ShapedType::isDynamic);
192  if (numCollapsedDynamic > 1 || numExpandedDynamic > 1 ||
193  collapsedStaticShapes != expandedStaticShapes) {
194  return failure();
195  }
196  continue;
197  }
198  // If the reassociations are not same, one or the other needs to be of
199  // size one.
200  if (collapseReassociation.size() != 1 && expandReassociation.size() != 1)
201  return failure();
202  }
203 
204  // Compute new reassociation indices and expanded/collaped shapes.
205  SmallVector<ReassociationIndices> newExpandReInds, newCollapseReInds;
206  Location loc = expandOp->getLoc();
207  SmallVector<OpFoldResult> sourceSizes =
208  tensor::getMixedSizes(rewriter, loc, collapseOp.getSrc());
209  SmallVector<OpFoldResult> resultSizes = expandOp.getMixedOutputShape();
210  SmallVector<OpFoldResult> newExpandSizes;
211 
212  int64_t newExpandIndex = 0, newCollapseIndex = 0, sourceSizeIndex = 0,
213  resultSizeIndex = 0;
214 
215  for (size_t idx = 0, idxEnd = collapseReInds.size(); idx < idxEnd; idx++) {
216  auto &collapseReassociation = collapseReInds[idx];
217  auto &expandReassociation = expandReInds[idx];
218 
219  // Case 1. The reassociations are same in the collapse producer
220  // and expand consumer. In the swapped expand, each of the final
221  // dimensions are kept as is in the expand and the collapse. So,
222  // for every element in the `ReassocationIndices` vector add a new
223  // `ReassociationIndices` vector for the swapped expand and collapse
224  // (of size 1).
225  if (collapseReassociation.size() == expandReassociation.size()) {
226  for (size_t i = 0; i < collapseReassociation.size(); ++i) {
227  newCollapseReInds.push_back({newCollapseIndex++});
228  newExpandReInds.push_back({newExpandIndex++});
229  newExpandSizes.push_back(resultSizes[resultSizeIndex++]);
230  sourceSizeIndex++;
231  }
232  continue;
233  }
234 
235  // Case 2. The `ReassociationIndices` in the collapse is of size > 1 (and
236  // in the expand is of size == 1). In this case, the original dimensions
237  // are preserved on expansion and collapsed subsequently.
238  if (collapseReassociation.size() != 1) {
239  ReassociationIndices newCollapseReassociation;
240  for (size_t i = 0; i < collapseReassociation.size(); ++i) {
241  newCollapseReassociation.push_back(newCollapseIndex++);
242  newExpandReInds.push_back({newExpandIndex++});
243  newExpandSizes.push_back(sourceSizes[sourceSizeIndex++]);
244  }
245  resultSizeIndex++;
246  newCollapseReInds.push_back(newCollapseReassociation);
247  continue;
248  }
249 
250  // Case 3. The `ReassociationIndices` in the expand is of size > 1 (and
251  // in the collapse is of size == 1). In this case, the expansion happens
252  // first and the expanded dimensions are preserved on collapse.
253  ReassociationIndices newExpandReassociation;
254  for (size_t i = 0; i < expandReassociation.size(); ++i) {
255  newExpandReassociation.push_back(newExpandIndex++);
256  newCollapseReInds.push_back({newCollapseIndex++});
257  newExpandSizes.push_back(resultSizes[resultSizeIndex++]);
258  }
259  newExpandReInds.push_back(newExpandReassociation);
260  sourceSizeIndex++;
261  }
262 
263  // Swap reshape order.
264  SmallVector<Value> dynamicSizes;
265  SmallVector<int64_t> staticSizes;
266  dispatchIndexOpFoldResults(newExpandSizes, dynamicSizes, staticSizes);
267  auto expandResultType = expandOp.getResultType().clone(staticSizes);
268  Value newCollapseSrc = collapseOp.getSrc();
269  // If the number of reassociation indices in the new `expand_shape` op
270  // matches the number of dimensions of the result, then the expand_shape
271  // is a no-op.
272  if (newExpandReInds.size() != newExpandSizes.size()) {
273  newCollapseSrc = tensor::ExpandShapeOp::create(
274  rewriter, loc, expandResultType, newCollapseSrc, newExpandReInds,
275  newExpandSizes);
276  }
277 
278  // If the number of reassociation indices in the new `collapse_shape` op
279  // matches the number of dimensions of the source, then the collapse_shape
280  // is a no-op.
281  Value replacement = newCollapseSrc;
282  if (newCollapseReInds.size() != newExpandSizes.size()) {
283  replacement = tensor::CollapseShapeOp::create(
284  rewriter, loc, newCollapseSrc, newCollapseReInds);
285  }
286  rewriter.replaceOp(expandOp, replacement);
287  return success();
288  }
289 };

291 /// Converts `tensor.extract_slice(tensor.expand_shape)` to
292 /// `tensor.expand_shape(tensor.extract_slice)`.
293 ///
294 /// For this transformation to be possible, the slice must be fully contiguous
295 /// within each reassociation group of the expand_shape. A slice is defined as
296 /// fully contiguous within a reassociation group if after flattening the
297 /// reassociation group to a single 1D range, then the slice taken out of the
298 /// group could be defined as a single contiguous subrange within that range.
299 ///
300 /// Rank reducing slices are not supported.
301 ///
302 /// Example:
303 /// The transformation is possible because each reassociation group has a
304 /// contiguous slice (i.e., [2x4->2x4], [2x8->1x5], [4x2x4->1x1x4]).
305 /// ```
306 /// BEFORE:
307 /// %reshape = tensor.expand_shape %in [[0, 1], [2, 3], [4, 5, 6]]
308 /// tensor<8x16x32xf32> to tensor<2x4x2x8x4x2x4xf32>
309 /// %slice = tensor.extract_slice %reshape ...
310 /// tensor<2x4x2x8x4x2x4xf32> to tensor<2x4x1x5x1x1x4xf32>
311 ///
312 /// AFTER:
313 /// %slice = tensor.extract_slice %in ...
314 /// tensor<8x16x32xf32> to tensor<8x5x4xf32>
315 /// %reshape = tensor.expand_shape %slice [[0, 1], [2, 3], [4, 5, 6]]
316 /// tensor<8x5x4xf32> to tensor<2x4x1x5x1x1x4xf32>
317 /// ```
318 ///
319 /// Note - this pattern could be extended to be a swap pattern between
320 /// `tensor.expand_shape` and `tensor.extract_slice`, but is currently
321 /// implemented only as a bubble up pattern for `tensor.extract_slice`.
322 struct BubbleUpExpandShapeThroughExtractSlice
323  : public OpRewritePattern<tensor::ExtractSliceOp> {
324  using OpRewritePattern<tensor::ExtractSliceOp>::OpRewritePattern;
325 
326  LogicalResult matchAndRewrite(tensor::ExtractSliceOp sliceOp,
327  PatternRewriter &rewriter) const override {
328  auto expandShapeOp =
329  sliceOp.getSource().getDefiningOp<tensor::ExpandShapeOp>();
330 
331  if (checkPreconditionForBubbleUpExtractSlice(sliceOp, expandShapeOp,
332  rewriter)
333  .failed())
334  return failure();
335 
336  // The tensor.extract_slice before applying the pattern works on the result
337  // of the tensor.expand_shape, so variables (i.e. inputs for ExtractSliceOp)
338  // referring to the state before applying the pattern are named with the
339  // prefix "expanded", and ones referring to the state after applying the
340  // pattern are named with the prefix "collapsed".
341  SmallVector<OpFoldResult> expandedOffsets = sliceOp.getMixedOffsets();
342  SmallVector<OpFoldResult> expandedSizes = sliceOp.getMixedSizes();
343  SmallVector<OpFoldResult> expandedShape =
344  getMixedValues(expandShapeOp.getStaticOutputShape(),
345  expandShapeOp.getOutputShape(), rewriter);
346 
347  // Helper variables and function for accumulating the size values.
348  Location loc = expandShapeOp->getLoc();
349  AffineExpr d0, d1, d2;
350  bindDims(rewriter.getContext(), d0, d1, d2);
351  // Multiply two integers.
352  auto mul = [&](OpFoldResult v1, OpFoldResult v2) {
353  auto mulMap = AffineMap::get(2, 0, {d0 * d1});
354  return affine::makeComposedFoldedAffineApply(rewriter, loc, mulMap,
355  {v1, v2});
356  };
357 
358  // Compute new offsets, sizes, and strides for tensor.extract_slice.
359  // The new tensor.extract_slice will work on a tensor that has has a rank of
360  // ReassociationIndices.size(). In the loop a single offset, size, and
361  // stride value is computed per reassociation group.
362  SmallVector<OpFoldResult> collapsedOffsets, collapsedSizes,
363  collapsedStrides;
364  for (const ReassociationIndices &indices :
365  expandShapeOp.getReassociationIndices()) {
366  // collapsedSize will hold the size of the single dim that represents the
367  // reassociation group in the non expanded tensor.
368  OpFoldResult collapsedSize = rewriter.getIndexAttr(1);
369  // The reassocGroupSizes and reassocGroupOffsets are used to create an
370  // affine.linearize_index op to linearize the single offset value required
371  // for this reassociation group.
372  SmallVector<OpFoldResult> reassocGroupSizes, reassocGroupOffsets;
373 
374  for (long expandedDim : indices) {
375  // reassocGroupSizes and reassocGroupOffsets can be obtained directly
376  // from the expanded state, but the collapsed size requires calculation
377  // as it did not previously exist.
378  reassocGroupSizes.push_back(expandedShape[expandedDim]);
379  reassocGroupOffsets.push_back(expandedOffsets[expandedDim]);
380  collapsedSize = mul(collapsedSize, expandedSizes[expandedDim]);
381  }
382 
383  SmallVector<Value> offsetVals =
384  llvm::map_to_vector(reassocGroupOffsets, [&](OpFoldResult ofr) {
385  return getValueOrCreateConstantIndexOp(rewriter, loc, ofr);
386  });
387  OpFoldResult collapsedOffset =
388  affine::AffineLinearizeIndexOp::create(rewriter, loc, offsetVals,
389  reassocGroupSizes,
390  /*disjoint=*/true)
391  .getResult();
392  collapsedOffsets.push_back(collapsedOffset);
393  collapsedSizes.push_back(collapsedSize);
394 
395  // Only unit stride is supported.
396  collapsedStrides.push_back(rewriter.getIndexAttr(1));
397  }
398 
399  // The shape of the result can be obtained from the sizes passed in.
400  SmallVector<Value> dynDims;
401  SmallVector<int64_t> shape;
402  dispatchIndexOpFoldResults(expandedSizes, dynDims, shape);
403  RankedTensorType resultType = RankedTensorType::get(
404  shape, expandShapeOp.getResultType().getElementType());
405 
406  // Create a new ExtractSliceOp and ExpandShapeOp.
407  Value newSliceOp = tensor::ExtractSliceOp::create(
408  rewriter, loc, expandShapeOp.getSrc(), collapsedOffsets, collapsedSizes,
409  collapsedStrides);
410  rewriter.replaceOpWithNewOp<tensor::ExpandShapeOp>(
411  sliceOp, resultType, newSliceOp,
412  expandShapeOp.getReassociationIndices(), expandedSizes);
413  return success();
414  }
415 
416  // Helper function to check if all the required conditions for the
417  // tensor.extract_slice to be bubbled up through the tensor.expand_shape are
418  // met.
419  LogicalResult
420  checkPreconditionForBubbleUpExtractSlice(tensor::ExtractSliceOp sliceOp,
421  tensor::ExpandShapeOp expandShapeOp,
422  PatternRewriter &rewriter) const {
423 
424  if (!expandShapeOp) {
425  return rewriter.notifyMatchFailure(
426  sliceOp, "tensor.extract_slice source not produced by expand_shape");
427  }
428 
429  if (!sliceOp.hasUnitStride()) {
430  return rewriter.notifyMatchFailure(
431  sliceOp, "unsupported: non-unit stride. Only contiguous slices can "
432  "be supported in this transformation.");
433  }
434 
435  SmallVector<OpFoldResult> offsets = sliceOp.getMixedOffsets();
436  SmallVector<OpFoldResult> sizes = sliceOp.getMixedSizes();
437 
438  if (static_cast<size_t>(sliceOp.getResultType().getRank()) !=
439  sizes.size()) {
440  return rewriter.notifyMatchFailure(sliceOp,
441  "unimplemented: rank reducing slice");
442  }
443 
444  SmallVector<OpFoldResult> outputShape =
445  getMixedValues(expandShapeOp.getStaticOutputShape(),
446  expandShapeOp.getOutputShape(), rewriter);
447 
448  std::function<bool(OpFoldResult, OpFoldResult, OpFoldResult)>
449  isZeroOffsetAndFullSize =
450  [](OpFoldResult offset, OpFoldResult sliceSize, OpFoldResult size) {
451  if (!isZeroInteger(offset))
452  return false;
453  FailureOr<bool> maybeEqual =
454  ValueBoundsConstraintSet::areEqual(sliceSize, size);
455  return llvm::succeeded(maybeEqual) && maybeEqual.value();
456  };
457 
458  // Check that the slice is contiguous within each reassociation group.
459  // The slice is contiguous only if after the first dimension where a non
460  // unit slice is taken, the slice size on all subsequent dimensions of the
461  // group is equal to the entire size of the dimension.
462  // Examples of contiguous slices:
463  // full sizes: [8, 8, 10] slice offsets: [0, 0, 0] slice sizes: [1, 1, 10]
464  // full sizes: [5, 10] slice offsets: [3, 0] slice sizes: [2, 10]
465  // Examples of non contiguous slices:
466  // full sizes: [8, 8, 10] slice offsets: [0, 0, 0] slice sizes: [1, 2, 5]
467  // full sizes: [5, 10] slice offsets: [0, 4] slice sizes: [2, 5]
468  for (const ReassociationIndices &indices :
469  expandShapeOp.getReassociationIndices()) {
470  int64_t i = 0;
471  int64_t e = indices.size();
472  // Find the first expanded dim after the first dim with non-unit extracted
473  // size.
474  for (; i < e; ++i) {
475  if (!isOneInteger(sizes[indices[i]])) {
476  // +1 to skip the first non-unit size dim.
477  i++;
478  break;
479  }
480  }
481 
482  // Verify that all subsequent dimensions extract the full size of the
483  // source tensor.
484  for (; i < e; ++i) {
485  int64_t expandedDim = indices[i];
486  if (!isZeroOffsetAndFullSize(offsets[expandedDim], sizes[expandedDim],
487  outputShape[expandedDim])) {
488  return rewriter.notifyMatchFailure(
489  sliceOp, "Not a contiguous slice of the expanded tensor.");
490  }
491  }
492  }
493 
494  return success();
495  }
496 };

498 /// Converts `tensor.extract_slice(tensor.collapse_shape)` to
499 /// `tensor.collapse_shape(tensor.extract_slice)`.
500 ///
501 /// For this transformation to be possible - after bubbling up, the extraction
502 /// of the contiguous slice must be representable as a single slice obtained via
503 /// tensor.extract_slice within each reassociation group of the src.
504 ///
505 /// In case the size and offset extracted are static then this is possible if
506 /// the following conditions are met within each reassociation group:
507 /// Let T be a tensor of shape [A0, A1, ..., An] (these are the sizes of the
508 /// dimensions in the reassociation group), and let S = [S0, S1, ..., Sn] be the
509 /// shape of a desired slice. A slice of shape S can be extracted as a
510 /// contiguous span of elements if and only if there exists an index k in {0, 1,
511 /// ..., n} such that:
512 /// S_i = 1 for all i < k (that is, all leading dimensions are singleton),
513 /// 1 <= S_k <= A_k (that is, non trivial slicing occurs along exactly
514 /// one dimension),
515 /// S_i = A_i for all i > k (that is, all trailing dimensions are preserved
516 /// in full).
517 /// In other words, the slice shape S must be of the form:
518 /// [ 1, 1, ..., 1, Sk, Ak + 1, Ak + 2, ...,An ]
519 ///
520 /// In case the size and/or offset extracted are dynamic then this is possible
521 /// only if there is single dimension in the reassociation group that has a size
522 /// not equal to 1.
523 /// In other words, the tensor shape must be of the form:
524 /// [ 1, 1, ..., 1, A, 1, ...,1 ]
525 /// Note - it might be possible to enable this pattern for more cases when the
526 /// size/offset are dynamic via performing an analysis of the possible values
527 /// that could be given to the size/offset.
528 ///
529 /// Example:
530 /// The transformation is possible because each reassociation group can be
531 /// represented as a contiguous slice (i.e., [8x16->2x16], [1x7->1x?],
532 /// [20->10]).
533 /// ```
534 /// BEFORE:
535 /// %collapse = tensor.collapse_shape %src [[0, 1], [2, 3], [4]] ...
536 /// tensor<8x16x1x7x20f32> to tensor<128x7x20xf32>
537 /// %slice = tensor.extract_slice %slice [0, 0, 0][32, %size, 10][1, 1, 1]
538 /// tensor<128x7x20xf32> to tensor<32x?x10xf32>
539 ///
540 /// AFTER:
541 /// %slice = tensor.extract_slice %src [0, 0, 0, 0, 0][2, 16, 1, %size, 10]
542 // [1, 1, 1, 1, 1] : tensor<8x16x1x7x20f32> to tensor<2x16x1x?x10xf32>
543 /// %collapse = tensor.collapse_shape %slice [[0, 1], [2, 3], [4]] ...
544 /// tensor<2x16x1x?x10xf32> to tensor<32x?x10xf32>
545 /// ```
546 ///
547 /// Negative example:
548 /// The transformation is not possible because we cannot use a single slice to
549 /// represent the reassociation group [2x3x10->???]. If we would want the
550 /// collapse to be after the extraction, we would need to extract multiple
551 /// slices and concat them together.
552 /// ```
553 /// %collapse = tensor.collapse_shape %src [[0, 1, 2]] : tensor<2x3x10xf32> into
554 /// tensor<60xf32> %extract = tensor.extract_slice %collapse[0][15][1] :
555 /// tensor<60xf32> to tensor<15xf32>
556 /// ```
557 /// If we would want the collapse to be after the extraction, a possible
558 /// alternate transformation could be to extract multiple slices and concat them
559 /// together:
560 /// ```
561 /// %extract_1 = tensor.extract_slice %src[0, 0, 0][1, 1, 10] :
562 /// tensor<2x3x10xf32> to tensor <1x1x10xf32>
563 /// %extract_2 = tensor.extract_slice %src[0, 1, 0][1, 1, 5] :
564 /// tensor<2x3x10xf32> to tensor <1x1x5xf32>
565 /// %concat = tosa.concat %extract_1, %extract_2 {axis = 0 : i32} :
566 /// (<1x1x10xf32>, <1x1x5xf32>) -> <1x1x15xf32>
567 /// %collapse = tensor.collapse_shape %concat [[0, 1, 2]] : tensor<1x1x15xf32>
568 /// to tensor<15xf32>
569 /// ```
570 /// But this is not the intended purpose of the transformation.
571 struct BubbleUpCollapseShapeThroughExtractSlice
572  : public OpRewritePattern<tensor::ExtractSliceOp> {
573  using OpRewritePattern<tensor::ExtractSliceOp>::OpRewritePattern;
574 
575  LogicalResult matchAndRewrite(tensor::ExtractSliceOp sliceOp,
576  PatternRewriter &rewriter) const override {
577  auto collapseShapeOp =
578  sliceOp.getSource().getDefiningOp<tensor::CollapseShapeOp>();
579  if (!collapseShapeOp) {
580  return rewriter.notifyMatchFailure(
581  sliceOp,
582  "tensor.extract_slice source not produced by tensor.collapse_shape");
583  }
584 
585  if (!sliceOp.hasUnitStride()) {
586  return rewriter.notifyMatchFailure(
587  sliceOp, "unsupported: non-unit stride. Only contiguous slices can "
588  "be supported in this transformation.");
589  }
590 
591  // The tensor.extract_slice before applying the pattern works on the result
592  // of the tensor.collapse_shape, so variables (i.e. inputs for
593  // ExtractSliceOp) referring to the state before applying the pattern are
594  // named with the prefix "collapsed", and ones referring to the state after
595  // applying the pattern are named with the prefix "expanded".
596  SmallVector<OpFoldResult> collapsedOffsets = sliceOp.getMixedOffsets();
597  SmallVector<OpFoldResult> collapsedSizes = sliceOp.getMixedSizes();
598 
599  if (static_cast<size_t>(sliceOp.getResultType().getRank()) !=
600  collapsedSizes.size()) {
601  return rewriter.notifyMatchFailure(sliceOp,
602  "unimplemented: rank reducing slice");
603  }
604 
605  ArrayRef<int64_t> srcShape = collapseShapeOp.getSrcType().getShape();
606  SmallVector<ReassociationIndices, 4> reassociationIndices =
607  collapseShapeOp.getReassociationIndices();
608 
609  // Compute new offsets, sizes, and strides for tensor.extract_slice.
610  // The new tensor.extract_slice will work on a tensor that has has a rank
611  // equal to the rank of the src of the collapse_shape. In each iteration of
612  // the loop, the offsets and sizes will be computed per reassociation group.
613  SmallVector<OpFoldResult> expandedOffsets, expandedSizes;
614  SmallVector<OpFoldResult> expandedStrides(srcShape.size(),
615  rewriter.getIndexAttr(1));
616 
617  for (auto [collapsedSize, collapsedOffset, reassocIndices] :
618  llvm::zip_equal(collapsedSizes, collapsedOffsets,
619  collapseShapeOp.getReassociationIndices())) {
620  // CASE #1 - size and/or offset are dynamic.
621  // In this case, the slice can be represented as a contiguous slice only
622  // if there is a single dimension in the reassociation group that has a
623  // size not equal to 1.
624  if (isa<Value>(collapsedSize) || isa<Value>(collapsedOffset)) {
625  int nonUnitSizeCount = 0;
626  for (int64_t expandedShapeIdx : reassocIndices) {
627  if (srcShape[expandedShapeIdx] != 1) {
628  nonUnitSizeCount++;
629  expandedSizes.push_back(collapsedSize);
630  expandedOffsets.push_back(collapsedOffset);
631  continue;
632  }
633 
634  expandedSizes.push_back(rewriter.getIndexAttr(1));
635  expandedOffsets.push_back(rewriter.getIndexAttr(0));
636  }
637 
638  if (nonUnitSizeCount != 1) {
639  return rewriter.notifyMatchFailure(
640  sliceOp,
641  "unsupported: slice cannot be verified to be contiguous");
642  }
643  continue;
644  }
645 
646  // CASE #2 = size and offset are static.
647  // Verify that the slice can be represented as a contiguous slice of the
648  // src of the collapse_shape.
649  // Checking this is done on order of most internal dimensions first,
650  // so traversal is done in reverse order of the reassociation group.
651  // If the expected slice shape is [1, 1, ..., 1, Sk, Ak + 1, Ak + 2,
652  // ...,An] then we first find the size and offset for n...k+1 then for k
653  // and then for k-1...0.
654 
655  // currentCollapsedsize and currentCollapsedOffset are initialized with
656  // the original collapsed size and offset and divided by the expanded
657  // shape size in each dimension as we go along the reassociation group.
658  // In essence we are spreading the original collapsed size and offset over
659  // the various expanded slice dimensions.
660  // The variables are used both to check the validity of the slice and to
661  // compute the expanded sizes and offsets.
662  int64_t currentCollapsedsize = getConstantIntValue(collapsedSize).value();
663  int64_t currentCollapsedOffset =
664  getConstantIntValue(collapsedOffset).value();
665 
666  SmallVector<OpFoldResult> groupExpandedSizes, groupExpandedOffsets;
667 
668  ReassociationIndices reversedReassocIndices(reassocIndices.rbegin(),
669  reassocIndices.rend());
670  int64_t idx = 0;
671  int64_t reassocGroupSize = reassocIndices.size();
672 
673  // First handle the trailing dimensions where the slice size should be
674  // equal to the tensor shape and the offset should be 0 (n...k+1).
675  for (; idx < reassocGroupSize; ++idx) {
676  int64_t expandedShapeSize = srcShape[reversedReassocIndices[idx]];
677 
678  if (currentCollapsedsize < expandedShapeSize)
679  break;
680 
681  // We need to make sure that the slice size can be set to the shape size
682  // and the offset to 0.
683  if ((currentCollapsedsize % expandedShapeSize) != 0 ||
684  (currentCollapsedOffset % expandedShapeSize) != 0) {
685  return rewriter.notifyMatchFailure(
686  sliceOp, "unsupported: cannot be extracted as a contiguous slice "
687  "of the src of the collapse_shape");
688  }
689 
690  groupExpandedSizes.push_back(rewriter.getIndexAttr(expandedShapeSize));
691  groupExpandedOffsets.push_back(rewriter.getIndexAttr(0));
692 
693  currentCollapsedsize /= expandedShapeSize;
694  currentCollapsedOffset /= expandedShapeSize;
695  }
696 
697  // Now handle the first dim where slicing occurs on (k).
698  if (idx < reassocGroupSize) {
699  int64_t expandedShapeSize = srcShape[reversedReassocIndices[idx]];
700  int64_t offsetInDim = currentCollapsedOffset % expandedShapeSize;
701  // We need to make sure that the slice size in this dim + offset will
702  // not exceed the shape size.
703  if ((currentCollapsedsize + offsetInDim) >= expandedShapeSize) {
704  return rewriter.notifyMatchFailure(
705  sliceOp, "unsupported: slice cannot be extracted as a contiguous "
706  "slice of the src of the collapse_shape");
707  }
708 
709  groupExpandedSizes.push_back(
710  rewriter.getIndexAttr(currentCollapsedsize));
711  groupExpandedOffsets.push_back(rewriter.getIndexAttr(offsetInDim));
712 
713  currentCollapsedOffset /= expandedShapeSize;
714  }
715 
716  // Now handle the leading dimensions where the slice size is equal to 1
717  // (k-1...0).
718  // The size for these dimensions must be 1 because of how we constructed
719  // the slice size of the expanded shape. We spread the original collapsed
720  // size over the expanded shape sizes until we reached dimension k where
721  // the remaining size was smaller than the expanded shape size, and spread
722  // the remaining size on it. So, now we are left with only 1s.
723  for (idx++; idx < reassocGroupSize; ++idx) {
724  int64_t expandedShapeSize = srcShape[reversedReassocIndices[idx]];
725  int64_t offsetInDim = currentCollapsedOffset % expandedShapeSize;
726  groupExpandedSizes.push_back(rewriter.getIndexAttr(1));
727  groupExpandedOffsets.push_back(rewriter.getIndexAttr(offsetInDim));
728  currentCollapsedOffset /= expandedShapeSize;
729  }
730 
731  expandedSizes.append(groupExpandedSizes.rbegin(),
732  groupExpandedSizes.rend());
733  expandedOffsets.append(groupExpandedOffsets.rbegin(),
734  groupExpandedOffsets.rend());
735  }
736 
737  Value newSliceOp = tensor::ExtractSliceOp::create(
738  rewriter, collapseShapeOp->getLoc(), collapseShapeOp.getSrc(),
739  expandedOffsets, expandedSizes, expandedStrides);
740  rewriter.replaceOpWithNewOp<tensor::CollapseShapeOp>(
741  sliceOp, sliceOp.getResultType(), newSliceOp,
742  collapseShapeOp.getReassociationIndices());
743 
744  return success();
745  }
746 };
