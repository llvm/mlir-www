58 /// Helper function that computes an insertion point where the given value is
59 /// defined and can be used without a dominance violation.
60 static OpBuilder::InsertPoint computeInsertPoint(Value value) {
61  Block *insertBlock = value.getParentBlock();
62  Block::iterator insertPt = insertBlock->begin();
63  if (OpResult inputRes = dyn_cast<OpResult>(value))
64  insertPt = ++inputRes.getOwner()->getIterator();
65  return OpBuilder::InsertPoint(insertBlock, insertPt);
66 }

68 /// Helper function that computes an insertion point where the given values are
69 /// defined and can be used without a dominance violation.
70 static OpBuilder::InsertPoint computeInsertPoint(ArrayRef<Value> vals) {
71  assert(!vals.empty() && "expected at least one value");
72  DominanceInfo domInfo;
73  OpBuilder::InsertPoint pt = computeInsertPoint(vals.front());
74  for (Value v : vals.drop_front()) {
75  // Choose the "later" insertion point.
76  OpBuilder::InsertPoint nextPt = computeInsertPoint(v);
77  if (domInfo.dominates(pt.getBlock(), pt.getPoint(), nextPt.getBlock(),
78  nextPt.getPoint())) {
79  // pt is before nextPt => choose nextPt.
80  pt = nextPt;
81  } else {
82 #ifndef NDEBUG
83  // nextPt should be before pt => choose pt.
84  // If pt, nextPt are no dominance relationship, then there is no valid
85  // insertion point at which all given values are defined.
86  bool dom = domInfo.dominates(nextPt.getBlock(), nextPt.getPoint(),
87  pt.getBlock(), pt.getPoint());
88  assert(dom && "unable to find valid insertion point");
89 #endif // NDEBUG
90  }
91  }
92  return pt;
93 }

117 /// This class wraps a IRMapping to provide recursive lookup
118 /// functionality, i.e. we will traverse if the mapped value also has a mapping.
119 struct ConversionValueMapping {
120  /// Return "true" if an SSA value is mapped to the given value. May return
121  /// false positives.
122  bool isMappedTo(Value value) const { return mappedTo.contains(value); }
123 
124  /// Lookup the most recently mapped values with the desired types in the
125  /// mapping.
126  ///
127  /// Special cases:
128  /// - If the desired type range is empty, simply return the most recently
129  /// mapped values.
130  /// - If there is no mapping to the desired types, also return the most
131  /// recently mapped values.
132  /// - If there is no mapping for the given values at all, return the given
133  /// value.
134  ValueVector lookupOrDefault(Value from, TypeRange desiredTypes = {}) const;

136  template <typename T>
137  struct IsValueVector : std::is_same<std::decay_t<T>, ValueVector> {};
138 
139  /// Map a value vector to the one provided.
140  template <typename OldVal, typename NewVal>
141  std::enable_if_t<IsValueVector<OldVal>::value && IsValueVector<NewVal>::value>
142  map(OldVal &&oldVal, NewVal &&newVal) {
143  LLVM_DEBUG({
144  ValueVector next(newVal);
145  while (true) {
146  assert(next != oldVal && "inserting cyclic mapping");
147  auto it = mapping.find(next);
148  if (it == mapping.end())
149  break;
150  next = it->second;
151  }
152  });
153  mappedTo.insert_range(newVal);
154 
155  mapping[std::forward<OldVal>(oldVal)] = std::forward<NewVal>(newVal);
156  }
157 
158  /// Map a value vector or single value to the one provided.
159  template <typename OldVal, typename NewVal>
160  std::enable_if_t<!IsValueVector<OldVal>::value ||
161  !IsValueVector<NewVal>::value>
162  map(OldVal &&oldVal, NewVal &&newVal) {
163  if constexpr (IsValueVector<OldVal>{}) {
164  map(std::forward<OldVal>(oldVal), ValueVector{newVal});
165  } else if constexpr (IsValueVector<NewVal>{}) {
166  map(ValueVector{oldVal}, std::forward<NewVal>(newVal));
167  } else {
168  map(ValueVector{oldVal}, ValueVector{newVal});
169  }
170  }
171 
172  void map(Value oldVal, SmallVector<Value> &&newVal) {
173  map(ValueVector{oldVal}, ValueVector(std::move(newVal)));
174  }
175 
176  /// Drop the last mapping for the given values.
177  void erase(const ValueVector &value) { mapping.erase(value); }
178 
179 private:
180  /// Current value mappings.
181  DenseMap<ValueVector, ValueVector, ValueVectorMapInfo> mapping;
182 
183  /// All SSA values that are mapped to. May contain false positives.
184  DenseSet<Value> mappedTo;

188 ValueVector
189 ConversionValueMapping::lookupOrDefault(Value from,
190  TypeRange desiredTypes) const {
191  // Try to find the deepest values that have the desired types. If there is no
192  // such mapping, simply return the deepest values.
193  ValueVector desiredValue;
194  ValueVector current{from};
195  do {
196  // Store the current value if the types match.
197  if (TypeRange(ValueRange(current)) == desiredTypes)
198  desiredValue = current;
199 
200  // If possible, Replace each value with (one or multiple) mapped values.
201  ValueVector next;
202  for (Value v : current) {
203  auto it = mapping.find({v});
204  if (it != mapping.end()) {
205  llvm::append_range(next, it->second);
206  } else {
207  next.push_back(v);
208  }
209  }
210  if (next != current) {
211  // If at least one value was replaced, continue the lookup from there.
212  current = std::move(next);
213  continue;
214  }
215 
216  // Otherwise: Check if there is a mapping for the entire vector. Such
217  // mappings are materializations. (N:M mapping are not supported for value
218  // replacements.)
219  //
220  // Note: From a correctness point of view, materializations do not have to
221  // be stored (and looked up) in the mapping. But for performance reasons,
222  // we choose to reuse existing IR (when possible) instead of creating it
223  // multiple times.
224  auto it = mapping.find(current);
225  if (it == mapping.end()) {
226  // No mapping found: The lookup stops here.
227  break;
228  }
229  current = it->second;
230  } while (true);
231 
232  // If the desired values were found use them, otherwise default to the leaf
233  // values.
234  // Note: If `desiredTypes` is empty, this function always returns `current`.
235  return !desiredValue.empty() ? std::move(desiredValue) : std::move(current);
236 }

284 /// An IR rewrite that can be committed (upon success) or rolled back (upon
285 /// failure).
286 ///
287 /// The dialect conversion keeps track of IR modifications (requested by the
288 /// user through the rewriter API) in `IRRewrite` objects. Some kind of rewrites
289 /// are directly applied to the IR as the rewriter API is used, some are applied
290 /// partially, and some are delayed until the `IRRewrite` objects are committed.
291 class IRRewrite {
292 public:
293  /// The kind of the rewrite. Rewrites can be undone if the conversion fails.
294  /// Enum values are ordered, so that they can be used in `classof`: first all
295  /// block rewrites, then all operation rewrites.
296  enum class Kind {
297  // Block rewrites
298  CreateBlock,
299  EraseBlock,
300  InlineBlock,
301  MoveBlock,
302  BlockTypeConversion,
303  ReplaceBlockArg,
304  // Operation rewrites
305  MoveOperation,
306  ModifyOperation,
307  ReplaceOperation,
308  CreateOperation,
309  UnresolvedMaterialization
310  };
311 
312  virtual ~IRRewrite() = default;
313 
314  /// Roll back the rewrite. Operations may be erased during rollback.
315  virtual void rollback() = 0;
316 
317  /// Commit the rewrite. At this point, it is certain that the dialect
318  /// conversion will succeed. All IR modifications, except for operation/block
319  /// erasure, must be performed through the given rewriter.
320  ///
321  /// Instead of erasing operations/blocks, they should merely be unlinked
322  /// commit phase and finally be erased during the cleanup phase. This is
323  /// because internal dialect conversion state (such as `mapping`) may still
324  /// be using them.
325  ///
326  /// Any IR modification that was already performed before the commit phase
327  /// (e.g., insertion of an op) must be communicated to the listener that may
328  /// be attached to the given rewriter.
329  virtual void commit(RewriterBase &rewriter) {}
330 
331  /// Cleanup operations/blocks. Cleanup is called after commit.
332  virtual void cleanup(RewriterBase &rewriter) {}
333 
334  Kind getKind() const { return kind; }
335 
336  static bool classof(const IRRewrite *rewrite) { return true; }
337 
338 protected:
339  IRRewrite(Kind kind, ConversionPatternRewriterImpl &rewriterImpl)
340  : kind(kind), rewriterImpl(rewriterImpl) {}
341 
342  const ConversionConfig &getConfig() const;
343 
344  const Kind kind;
345  ConversionPatternRewriterImpl &rewriterImpl;
346 };

692  dyn_cast_or_null<RewriterBase::Listener>(rewriter.getListener()))
693  listener->notifyOperationModified(op);
694 
695  if (propertiesStorage) {
696  OpaqueProperties propCopy(propertiesStorage);
697  // Note: The operation may have been erased in the mean time, so
698  // OperationName must be stored in this object.
699  name.destroyOpProperties(propCopy);
700  operator delete(propertiesStorage);
701  propertiesStorage = nullptr;
702  }
703  }
704 
705  void rollback() override {
706  op->setLoc(loc);
707  op->setAttrs(attrs);
708  op->setOperands(operands);
709  for (const auto &it : llvm::enumerate(successors))
710  op->setSuccessor(it.value(), it.index());
711  if (propertiesStorage) {
712  OpaqueProperties propCopy(propertiesStorage);
713  op->copyProperties(propCopy);
714  name.destroyOpProperties(propCopy);
715  operator delete(propertiesStorage);
716  propertiesStorage = nullptr;
717  }
718  }

1326  if (!currentTypeConverter) {
1327  // The current pattern does not have a type converter. I.e., it does not
1328  // distinguish between legal and illegal types. For each operand, simply
1329  // pass through the most recently mapped values.
1330  remapped.push_back(lookupOrDefault(operand));
1331  continue;
1332  }
1333 
1334  // If there is no legal conversion, fail to match this pattern.
1335  SmallVector<Type, 1> legalTypes;
1336  if (failed(currentTypeConverter->convertType(origType, legalTypes))) {
1337  notifyMatchFailure(operandLoc, [=](Diagnostic &diag) {
1338  diag << "unable to convert type for " << valueDiagTag << " #"
1339  << it.index() << ", type was " << origType;
1340  });
1341  return failure();
1342  }
1343  // If a type is converted to 0 types, there is nothing to do.
1344  if (legalTypes.empty()) {
1345  remapped.push_back({});
1346  continue;
1347  }
1348 
1349  ValueVector repl = lookupOrDefault(operand, legalTypes);
1350  if (!repl.empty() && TypeRange(ValueRange(repl)) == legalTypes) {
1351  // Mapped values have the correct type or there is an existing
1352  // materialization. Or the operand is not mapped at all and has the
1353  // correct type.
1354  remapped.push_back(std::move(repl));
1355  continue;
1356  }
1357 
1358  // Create a materialization for the most recently mapped values.
1359  repl = lookupOrDefault(operand);
1360  ValueRange castValues = buildUnresolvedMaterialization(
1361  MaterializationKind::Target, computeInsertPoint(repl), operandLoc,
1362  /*valuesToMap=*/repl, /*inputs=*/repl, /*outputTypes=*/legalTypes,
1363  /*originalType=*/origType, currentTypeConverter);
1364  remapped.push_back(castValues);
1365  }
1366  return success();
1367 }

1416 Block *ConversionPatternRewriterImpl::applySignatureConversion(
1417  ConversionPatternRewriter &rewriter, Block *block,
1418  const TypeConverter *converter,
1419  TypeConverter::SignatureConversion &signatureConversion) {
1420 #if MLIR_ENABLE_EXPENSIVE_PATTERN_API_CHECKS
1421  // A block cannot be converted multiple times.
1422  if (hasRewrite<BlockTypeConversionRewrite>(rewrites, block))
1423  llvm::report_fatal_error("block was already converted");
1424 #endif // MLIR_ENABLE_EXPENSIVE_PATTERN_API_CHECKS
1425 
1426  OpBuilder::InsertionGuard g(rewriter);
1427 
1428  // If no arguments are being changed or added, there is nothing to do.
1429  unsigned origArgCount = block->getNumArguments();
1430  auto convertedTypes = signatureConversion.getConvertedTypes();
1431  if (llvm::equal(block->getArgumentTypes(), convertedTypes))
1432  return block;
1433 
1434  // Compute the locations of all block arguments in the new block.
1435  SmallVector<Location> newLocs(convertedTypes.size(),
1436  rewriter.getUnknownLoc());
1437  for (unsigned i = 0; i < origArgCount; ++i) {
1438  auto inputMap = signatureConversion.getInputMapping(i);
1439  if (!inputMap || inputMap->replacedWithValues())
1440  continue;
1441  Location origLoc = block->getArgument(i).getLoc();
1442  for (unsigned j = 0; j < inputMap->size; ++j)
1443  newLocs[inputMap->inputNo + j] = origLoc;
1444  }
1445 
1446  // Insert a new block with the converted block argument types and move all ops
1447  // from the old block to the new block.
1448  Block *newBlock =
1449  rewriter.createBlock(block->getParent(), std::next(block->getIterator()),
1450  convertedTypes, newLocs);
1451 
1452  // If a listener is attached to the dialect conversion, ops cannot be moved
1453  // to the destination block in bulk ("fast path"). This is because at the time
1454  // the notifications are sent, it is unknown which ops were moved. Instead,
1455  // ops should be moved one-by-one ("slow path"), so that a separate
1456  // `MoveOperationRewrite` is enqueued for each moved op. Moving ops in bulk is
1457  // a bit more efficient, so we try to do that when possible.
1458  bool fastPath = !config.listener;
1459  if (fastPath) {
1460  appendRewrite<InlineBlockRewrite>(newBlock, block, newBlock->end());
1461  newBlock->getOperations().splice(newBlock->end(), block->getOperations());
1462  } else {
1463  while (!block->empty())
1464  rewriter.moveOpBefore(&block->front(), newBlock, newBlock->end());
1465  }
1466 
1467  // Replace all uses of the old block with the new block.
1468  block->replaceAllUsesWith(newBlock);
1469 
1470  for (unsigned i = 0; i != origArgCount; ++i) {
1471  BlockArgument origArg = block->getArgument(i);
1472  Type origArgType = origArg.getType();
1473 
1474  std::optional<TypeConverter::SignatureConversion::InputMapping> inputMap =
1475  signatureConversion.getInputMapping(i);
1476  if (!inputMap) {
1477  // This block argument was dropped and no replacement value was provided.
1478  // Materialize a replacement value "out of thin air".
1479  Value mat =
1480  buildUnresolvedMaterialization(
1481  MaterializationKind::Source,
1482  OpBuilder::InsertPoint(newBlock, newBlock->begin()),
1483  origArg.getLoc(),
1484  /*valuesToMap=*/{}, /*inputs=*/ValueRange(),
1485  /*outputTypes=*/origArgType, /*originalType=*/Type(), converter)
1486  .front();
1487  replaceUsesOfBlockArgument(origArg, mat, converter);
1488  continue;
1489  }
1490 
1491  if (inputMap->replacedWithValues()) {
1492  // This block argument was dropped and replacement values were provided.
1493  assert(inputMap->size == 0 &&
1494  "invalid to provide a replacement value when the argument isn't "
1495  "dropped");
1496  replaceUsesOfBlockArgument(origArg, inputMap->replacementValues,
1497  converter);
1498  continue;
1499  }
1500 
1501  // This is a 1->1+ mapping.
1502  auto replArgs =
1503  newBlock->getArguments().slice(inputMap->inputNo, inputMap->size);
1504  replaceUsesOfBlockArgument(origArg, replArgs, converter);
1505  }
1506 
1507  appendRewrite<BlockTypeConversionRewrite>(/*origBlock=*/block, newBlock);
1508 
1509  // Erase the old block. (It is just unlinked for now and will be erased during
1510  // cleanup.)
1511  rewriter.eraseBlock(block);
1512 
1513  return newBlock;
1514 }

1520 /// Build an unresolved materialization operation given an output type and set
1521 /// of input operands.
1522 ValueRange ConversionPatternRewriterImpl::buildUnresolvedMaterialization(
1523  MaterializationKind kind, OpBuilder::InsertPoint ip, Location loc,
1524  ValueVector valuesToMap, ValueRange inputs, TypeRange outputTypes,
1525  Type originalType, const TypeConverter *converter,
1526  UnrealizedConversionCastOp *castOp) {
1527  assert((!originalType || kind == MaterializationKind::Target) &&
1528  "original type is valid only for target materializations");
1529  assert(TypeRange(inputs) != outputTypes &&
1530  "materialization is not necessary");
1531 
1532  // Create an unresolved materialization. We use a new OpBuilder to avoid
1533  // tracking the materialization like we do for other operations.
1534  OpBuilder builder(outputTypes.front().getContext());
1535  builder.setInsertionPoint(ip.getBlock(), ip.getPoint());
1536  auto convertOp =
1537  UnrealizedConversionCastOp::create(builder, loc, outputTypes, inputs);
1538  if (!valuesToMap.empty())
1539  mapping.map(valuesToMap, convertOp.getResults());
1540  if (castOp)
1541  *castOp = convertOp;
1542  unresolvedMaterializations[convertOp] =
1543  UnresolvedMaterializationInfo(converter, kind, originalType);
1544  appendRewrite<UnresolvedMaterializationRewrite>(convertOp,
1545  std::move(valuesToMap));
1546  return convertOp.getResults();
1547 }

1549 Value ConversionPatternRewriterImpl::findOrBuildReplacementValue(
1550  Value value, const TypeConverter *converter) {
1551  // Try to find a replacement value with the same type in the conversion value
1552  // mapping. This includes cached materializations. We try to reuse those
1553  // instead of generating duplicate IR.
1554  ValueVector repl = lookupOrNull(value, value.getType());
1555  if (!repl.empty())
1556  return repl.front();
1557 
1558  // Check if the value is dead. No replacement value is needed in that case.
1559  // This is an approximate check that may have false negatives but does not
1560  // require computing and traversing an inverse mapping. (We may end up
1561  // building source materializations that are never used and that fold away.)
1562  if (llvm::all_of(value.getUsers(),
1563  [&](Operation *op) { return replacedOps.contains(op); }) &&
1564  !mapping.isMappedTo(value))
1565  return Value();
1566 
1567  // No replacement value was found. Get the latest replacement value
1568  // (regardless of the type) and build a source materialization to the
1569  // original type.
1570  repl = lookupOrNull(value);
1571  if (repl.empty()) {
1572  // No replacement value is registered in the mapping. This means that the
1573  // value is dropped and no longer needed. (If the value were still needed,
1574  // a source materialization producing a replacement value "out of thin air"
1575  // would have already been created during `replaceOp` or
1576  // `applySignatureConversion`.)
1577  return Value();
1578  }
1579 
1580  // Note: `computeInsertPoint` computes the "earliest" insertion point at
1581  // which all values in `repl` are defined. It is important to emit the
1582  // materialization at that location because the same materialization may be
1583  // reused in a different context. (That's because materializations are cached
1584  // in the conversion value mapping.) The insertion point of the
1585  // materialization must be valid for all future users that may be created
1586  // later in the conversion process.
1587  Value castValue =
1588  buildUnresolvedMaterialization(MaterializationKind::Source,
1589  computeInsertPoint(repl), value.getLoc(),
1590  /*valuesToMap=*/repl, /*inputs=*/repl,
1591  /*outputTypes=*/value.getType(),
1592  /*originalType=*/Type(), converter)
1593  .front();
1594  return castValue;
1595 }

1629 void ConversionPatternRewriterImpl::replaceOp(
1630  Operation *op, SmallVector<SmallVector<Value>> &&newValues) {
1631  assert(newValues.size() == op->getNumResults());
1632  assert(!ignoredOps.contains(op) && "operation was already replaced");
1633 
1634  // Check if replaced op is an unresolved materialization, i.e., an
1635  // unrealized_conversion_cast op that was created by the conversion driver.
1636  if (auto castOp = dyn_cast<UnrealizedConversionCastOp>(op)) {
1637  // Make sure that the user does not mess with unresolved materializations
1638  // that were inserted by the conversion driver. We keep track of these
1639  // ops in internal data structures.
1640  assert(!unresolvedMaterializations.contains(castOp) &&
1641  "attempting to replace/erase an unresolved materialization");
1642  }
1643 
1644  // Create mappings for each of the new result values.
1645  for (auto [repl, result] : llvm::zip_equal(newValues, op->getResults())) {
1646  if (repl.empty()) {
1647  // This result was dropped and no replacement value was provided.
1648  // Materialize a replacement value "out of thin air".
1649  buildUnresolvedMaterialization(
1650  MaterializationKind::Source, computeInsertPoint(result),
1651  result.getLoc(), /*valuesToMap=*/{result}, /*inputs=*/ValueRange(),
1652  /*outputTypes=*/result.getType(), /*originalType=*/Type(),
1653  currentTypeConverter);
1654  continue;
1655  }
1656 
1657  // Remap result to replacement value.
1658  if (repl.empty())
1659  continue;
1660  mapping.map(static_cast<Value>(result), std::move(repl));
1661  }
1662 
1663  appendRewrite<ReplaceOperationRewrite>(op, currentTypeConverter);
1664  // Mark this operation and all nested ops as replaced.
1665  op->walk([&](Operation *op) { replacedOps.insert(op); });
1666 }

1875 void ConversionPatternRewriter::inlineBlockBefore(Block *source, Block *dest,
1876  Block::iterator before,
1877  ValueRange argValues) {
1878 #ifndef NDEBUG
1879  assert(argValues.size() == source->getNumArguments() &&
1880  "incorrect # of argument replacement values");
1881  assert(!impl->wasOpReplaced(source->getParentOp()) &&
1882  "attempting to inline a block from a replaced/erased op");
1883  assert(!impl->wasOpReplaced(dest->getParentOp()) &&
1884  "attempting to inline a block into a replaced/erased op");
1885  auto opIgnored = [&](Operation *op) { return impl->isOpIgnored(op); };
1886  // The source block will be deleted, so it should not have any users (i.e.,
1887  // there should be no predecessors).
1888  assert(llvm::all_of(source->getUsers(), opIgnored) &&
1889  "expected 'source' to have no predecessors");
1890 #endif // NDEBUG
1891 
1892  // If a listener is attached to the dialect conversion, ops cannot be moved
1893  // to the destination block in bulk ("fast path"). This is because at the time
1894  // the notifications are sent, it is unknown which ops were moved. Instead,
1895  // ops should be moved one-by-one ("slow path"), so that a separate
1896  // `MoveOperationRewrite` is enqueued for each moved op. Moving ops in bulk is
1897  // a bit more efficient, so we try to do that when possible.
1898  bool fastPath = !impl->config.listener;
1899 
1900  if (fastPath)
1901  impl->inlineBlockBefore(source, dest, before);
1902 
1903  // Replace all uses of block arguments.
1904  for (auto it : llvm::zip(source->getArguments(), argValues))

2539 void OperationLegalizer::buildLegalizationGraph(
2540  LegalizationPatterns &anyOpLegalizerPatterns,
2541  DenseMap<OperationName, LegalizationPatterns> &legalizerPatterns) {
2542  // A mapping between an operation and a set of operations that can be used to
2543  // generate it.
2544  DenseMap<OperationName, SmallPtrSet<OperationName, 2>> parentOps;
2545  // A mapping between an operation and any currently invalid patterns it has.
2546  DenseMap<OperationName, SmallPtrSet<const Pattern *, 2>> invalidPatterns;
2547  // A worklist of patterns to consider for legality.
2548  SetVector<const Pattern *> patternWorklist;
2549 
2550  // Build the mapping from operations to the parent ops that may generate them.
2551  applicator.walkAllPatterns([&](const Pattern &pattern) {
2552  std::optional<OperationName> root = pattern.getRootKind();
2553 
2554  // If the pattern has no specific root, we can't analyze the relationship
2555  // between the root op and generated operations. Given that, add all such
2556  // patterns to the legalization set.
2557  if (!root) {
2558  anyOpLegalizerPatterns.push_back(&pattern);
2559  return;
2560  }
2561 
2562  // Skip operations that are always known to be legal.
2563  if (target.getOpAction(*root) == LegalizationAction::Legal)
2564  return;
2565 
2566  // Add this pattern to the invalid set for the root op and record this root
2567  // as a parent for any generated operations.
2568  invalidPatterns[*root].insert(&pattern);
2569  for (auto op : pattern.getGeneratedOps())
2570  parentOps[op].insert(*root);
2571 
2572  // Add this pattern to the worklist.
2573  patternWorklist.insert(&pattern);
2574  });
2575 
2576  // If there are any patterns that don't have a specific root kind, we can't
2577  // make direct assumptions about what operations will never be legalized.
2578  // Note: Technically we could, but it would require an analysis that may
2579  // recurse into itself. It would be better to perform this kind of filtering
2580  // at a higher level than here anyways.
2581  if (!anyOpLegalizerPatterns.empty()) {
2582  for (const Pattern *pattern : patternWorklist)
2583  legalizerPatterns[*pattern->getRootKind()].push_back(pattern);
2584  return;
2585  }
2586 
2587  while (!patternWorklist.empty()) {
2588  auto *pattern = patternWorklist.pop_back_val();
2589 
2590  // Check to see if any of the generated operations are invalid.
2591  if (llvm::any_of(pattern->getGeneratedOps(), [&](OperationName op) {
2592  std::optional<LegalizationAction> action = target.getOpAction(op);
2593  return !legalizerPatterns.count(op) &&
2594  (!action || action == LegalizationAction::Illegal);
2595  }))
2596  continue;
2597 
2598  // Otherwise, if all of the generated operation are valid, this op is now
2599  // legal so add all of the child patterns to the worklist.
2600  legalizerPatterns[*pattern->getRootKind()].push_back(pattern);
2601  invalidPatterns[*pattern->getRootKind()].erase(pattern);
2602 
2603  // Add any invalid patterns of the parent operations to see if they have now
2604  // become legal.
2605  for (auto op : parentOps[*pattern->getRootKind()])
2606  patternWorklist.set_union(invalidPatterns[op]);
2607  }
2608 }

2610 void OperationLegalizer::computeLegalizationGraphBenefit(
2611  LegalizationPatterns &anyOpLegalizerPatterns,
2612  DenseMap<OperationName, LegalizationPatterns> &legalizerPatterns) {
2613  // The smallest pattern depth, when legalizing an operation.
2614  DenseMap<OperationName, unsigned> minOpPatternDepth;
2615 
2616  // For each operation that is transitively legal, compute a cost for it.
2617  for (auto &opIt : legalizerPatterns)
2618  if (!minOpPatternDepth.count(opIt.first))
2619  computeOpLegalizationDepth(opIt.first, minOpPatternDepth,
2620  legalizerPatterns);
2621 
2622  // Apply the cost model to the patterns that can match any operation. Those
2623  // with a specific operation type are already resolved when computing the op
2624  // legalization depth.
2625  if (!anyOpLegalizerPatterns.empty())
2626  applyCostModelToPatterns(anyOpLegalizerPatterns, minOpPatternDepth,
2627  legalizerPatterns);
2628 
2629  // Apply a cost model to the pattern applicator. We order patterns first by
2630  // depth then benefit. `legalizerPatterns` contains per-op patterns by
2631  // decreasing benefit.
2632  applicator.applyCostModel([&](const Pattern &pattern) {
2633  ArrayRef<const Pattern *> orderedPatternList;
2634  if (std::optional<OperationName> rootName = pattern.getRootKind())
2635  orderedPatternList = legalizerPatterns[*rootName];
2636  else
2637  orderedPatternList = anyOpLegalizerPatterns;
2638 
2639  // If the pattern is not found, then it was removed and cannot be matched.
2640  auto *it = llvm::find(orderedPatternList, &pattern);
2641  if (it == orderedPatternList.end())
2642  return PatternBenefit::impossibleToMatch();
2643 
2644  // Patterns found earlier in the list have higher benefit.
2645  return PatternBenefit(std::distance(it, orderedPatternList.end()));
2646  });

2649 unsigned OperationLegalizer::computeOpLegalizationDepth(
2650  OperationName op, DenseMap<OperationName, unsigned> &minOpPatternDepth,
2651  DenseMap<OperationName, LegalizationPatterns> &legalizerPatterns) {
2652  // Check for existing depth.
2653  auto depthIt = minOpPatternDepth.find(op);
2654  if (depthIt != minOpPatternDepth.end())
2655  return depthIt->second;
2656 
2657  // If a mapping for this operation does not exist, then this operation
2658  // is always legal. Return 0 as the depth for a directly legal operation.
2659  auto opPatternsIt = legalizerPatterns.find(op);
2660  if (opPatternsIt == legalizerPatterns.end() || opPatternsIt->second.empty())
2661  return 0u;
2662 
2663  // Record this initial depth in case we encounter this op again when
2664  // recursively computing the depth.
2665  minOpPatternDepth.try_emplace(op, std::numeric_limits<unsigned>::max());
2666 
2667  // Apply the cost model to the operation patterns, and update the minimum
2668  // depth.
2669  unsigned minDepth = applyCostModelToPatterns(
2670  opPatternsIt->second, minOpPatternDepth, legalizerPatterns);
2671  minOpPatternDepth[op] = minDepth;
2672  return minDepth;
2673 }
2674 
2675 unsigned OperationLegalizer::applyCostModelToPatterns(
2676  LegalizationPatterns &patterns,
2677  DenseMap<OperationName, unsigned> &minOpPatternDepth,
2678  DenseMap<OperationName, LegalizationPatterns> &legalizerPatterns) {
2679  unsigned minDepth = std::numeric_limits<unsigned>::max();
2680 
2681  // Compute the depth for each pattern within the set.
2682  SmallVector<std::pair<const Pattern *, unsigned>, 4> patternsByDepth;
2683  patternsByDepth.reserve(patterns.size());
2684  for (const Pattern *pattern : patterns) {
2685  unsigned depth = 1;
2686  for (auto generatedOp : pattern->getGeneratedOps()) {
2687  unsigned generatedOpDepth = computeOpLegalizationDepth(
2688  generatedOp, minOpPatternDepth, legalizerPatterns);
2689  depth = std::max(depth, generatedOpDepth + 1);
2690  }
2691  patternsByDepth.emplace_back(pattern, depth);
2692 
2693  // Update the minimum depth of the pattern list.
2694  minDepth = std::min(minDepth, depth);
2695  }
2696 
2697  // If the operation only has one legalization pattern, there is no need to
2698  // sort them.
2699  if (patternsByDepth.size() == 1)
2700  return minDepth;
2701 
2702  // Sort the patterns by those likely to be the most beneficial.
2703  llvm::stable_sort(patternsByDepth,
2704  [](const std::pair<const Pattern *, unsigned> &lhs,
2705  const std::pair<const Pattern *, unsigned> &rhs) {
2706  // First sort by the smaller pattern legalization
2707  // depth.
2708  if (lhs.second != rhs.second)
2709  return lhs.second < rhs.second;
2710 
2711  // Then sort by the larger pattern benefit.
2712  auto lhsBenefit = lhs.first->getBenefit();
2713  auto rhsBenefit = rhs.first->getBenefit();
2714  return lhsBenefit > rhsBenefit;
2715  });
2716 
2717  // Update the legalization pattern to use the new sorted list.
2718  patterns.clear();
2719  for (auto &patternIt : patternsByDepth)
2720  patterns.push_back(patternIt.first);
2721  return minDepth;
2722 }

2883  rewriterImpl.applyRewrites();

2924 void mlir::reconcileUnrealizedCasts(
2925  ArrayRef<UnrealizedConversionCastOp> castOps,
2926  SmallVectorImpl<UnrealizedConversionCastOp> *remainingCastOps) {
2927  SetVector<UnrealizedConversionCastOp> worklist(llvm::from_range, castOps);
2928  // This set is maintained only if `remainingCastOps` is provided.
2929  DenseSet<Operation *> erasedOps;
2930 
2931  // Helper function that adds all operands to the worklist that are an
2932  // unrealized_conversion_cast op result.
2933  auto enqueueOperands = [&](UnrealizedConversionCastOp castOp) {
2934  for (Value v : castOp.getInputs())
2935  if (auto inputCastOp = v.getDefiningOp<UnrealizedConversionCastOp>())
2936  worklist.insert(inputCastOp);
2937  };
2938 
2939  // Helper function that return the unrealized_conversion_cast op that
2940  // defines all inputs of the given op (in the same order). Return "nullptr"
2941  // if there is no such op.
2942  auto getInputCast =
2943  [](UnrealizedConversionCastOp castOp) -> UnrealizedConversionCastOp {
2944  if (castOp.getInputs().empty())
2945  return {};
2946  auto inputCastOp =
2947  castOp.getInputs().front().getDefiningOp<UnrealizedConversionCastOp>();
2948  if (!inputCastOp)
2949  return {};
2950  if (inputCastOp.getOutputs() != castOp.getInputs())
2951  return {};
2952  return inputCastOp;
2953  };
2954 
2955  // Process ops in the worklist bottom-to-top.
2956  while (!worklist.empty()) {
2957  UnrealizedConversionCastOp castOp = worklist.pop_back_val();
2958  if (castOp->use_empty()) {
2959  // DCE: If the op has no users, erase it. Add the operands to the
2960  // worklist to find additional DCE opportunities.
2961  enqueueOperands(castOp);
2962  if (remainingCastOps)
2963  erasedOps.insert(castOp.getOperation());
2964  castOp->erase();
2965  continue;
2966  }
2967 
2968  // Traverse the chain of input cast ops to see if an op with the same
2969  // input types can be found.
2970  UnrealizedConversionCastOp nextCast = castOp;
2971  while (nextCast) {
2972  if (nextCast.getInputs().getTypes() == castOp.getResultTypes()) {
2973  // Found a cast where the input types match the output types of the
2974  // matched op. We can directly use those inputs and the matched op can
2975  // be removed.
2976  enqueueOperands(castOp);
2977  castOp.replaceAllUsesWith(nextCast.getInputs());
2978  if (remainingCastOps)
2979  erasedOps.insert(castOp.getOperation());
2980  castOp->erase();
2981  break;
2982  }
2983  nextCast = getInputCast(nextCast);
2984  }
2985  }
2986 
2987  if (remainingCastOps)
2988  for (UnrealizedConversionCastOp op : castOps)
2989  if (!erasedOps.contains(op.getOperation()))
2990  remainingCastOps->push_back(op);
2991 }
