89 void ValueBoundsConstraintSet::addBound(BoundType type, int64_t pos,
90  AffineExpr expr) {
91  LogicalResult status = cstr.addBound(
92  type, pos,
93  AffineMap::get(cstr.getNumDimVars(), cstr.getNumSymbolVars(), expr));
94  if (failed(status)) {
95  // Non-pure (e.g., semi-affine) expressions are not yet supported by
96  // FlatLinearConstraints. However, we can just ignore such failures here.
97  // Even without this bound, there may be enough information in the
98  // constraint system to compute the requested bound. In case this bound is
99  // actually needed, `computeBound` will return `failure`.
100  LLVM_DEBUG(llvm::dbgs() << "Failed to add bound: " << expr << "\n");
101  }

104 AffineExpr ValueBoundsConstraintSet::getExpr(Value value,
105  std::optional<int64_t> dim) {
106 #ifndef NDEBUG
107  assertValidValueDim(value, dim);
108 #endif // NDEBUG
109 
110  auto shapedType = dyn_cast<ShapedType>(value.getType());
111  if (shapedType) {
112  // Static dimension: return constant directly.
113  if (shapedType.hasRank() && !shapedType.isDynamicDim(*dim))
114  return builder.getAffineConstantExpr(shapedType.getDimSize(*dim));
115  } else {
116  // Constant index value: return directly.
117  if (auto constInt = ::getConstantIntValue(value))
118  return builder.getAffineConstantExpr(*constInt);
119  }
120 
121  // Dynamic value: add to constraint set.
122  ValueDim valueDim = std::make_pair(value, dim.value_or(kIndexValue));
123  if (!valueDimToPosition.contains(valueDim))
124  (void)insert(value, dim);
125  int64_t pos = getPos(value, dim);
126  return pos < cstr.getNumDimVars()
127  ? builder.getAffineDimExpr(pos)
128  : builder.getAffineSymbolExpr(pos - cstr.getNumDimVars());
129 }

143 int64_t ValueBoundsConstraintSet::insert(Value value,
144  std::optional<int64_t> dim,
145  bool isSymbol) {
146 #ifndef NDEBUG
147  assertValidValueDim(value, dim);
148 #endif // NDEBUG
149 
150  ValueDim valueDim = std::make_pair(value, dim.value_or(kIndexValue));
151  assert(!valueDimToPosition.contains(valueDim) && "already mapped");
152  int64_t pos = isSymbol ? cstr.appendVar(VarKind::Symbol)
153  : cstr.appendVar(VarKind::SetDim);
154  positionToValueDim.insert(positionToValueDim.begin() + pos, valueDim);
155  // Update reverse mapping.
156  for (int64_t i = pos, e = positionToValueDim.size(); i < e; ++i)
157  if (positionToValueDim[i].has_value())
158  valueDimToPosition[*positionToValueDim[i]] = i;
159 
160  worklist.push(pos);
161  return pos;
162 }

196 void ValueBoundsConstraintSet::processWorklist(StopConditionFn stopCondition) {
197  while (!worklist.empty()) {
198  int64_t pos = worklist.front();
199  worklist.pop();
200  assert(positionToValueDim[pos].has_value() &&
201  "did not expect std::nullopt on worklist");
202  ValueDim valueDim = *positionToValueDim[pos];
203  Value value = valueDim.first;
204  int64_t dim = valueDim.second;
205 
206  // Check for static dim size.
207  if (dim != kIndexValue) {
208  auto shapedType = cast<ShapedType>(value.getType());
209  if (shapedType.hasRank() && !shapedType.isDynamicDim(dim)) {
210  bound(value)[dim] == getExpr(shapedType.getDimSize(dim));
211  continue;
212  }
213  }
214 
215  // Do not process any further if the stop condition is met.
216  auto maybeDim = dim == kIndexValue ? std::nullopt : std::make_optional(dim);
217  if (stopCondition(value, maybeDim))
218  continue;
219 
220  // Query `ValueBoundsOpInterface` for constraints. New items may be added to
221  // the worklist.
222  auto valueBoundsOp =
223  dyn_cast<ValueBoundsOpInterface>(getOwnerOfValue(value));
224  if (valueBoundsOp) {
225  if (dim == kIndexValue) {
226  valueBoundsOp.populateBoundsForIndexValue(value, *this);
227  } else {
228  valueBoundsOp.populateBoundsForShapedValueDim(value, dim, *this);
229  }
230  continue;
231  }
232 
233  // If the op does not implement `ValueBoundsOpInterface`, check if it
234  // implements the `DestinationStyleOpInterface`. OpResults of such ops are
235  // tied to OpOperands. Tied values have the same shape.
236  auto dstOp = value.getDefiningOp<DestinationStyleOpInterface>();
237  if (!dstOp || dim == kIndexValue)
238  continue;
239  Value tiedOperand = dstOp.getTiedOpOperand(cast<OpResult>(value))->get();
240  bound(value)[dim] == getExpr(tiedOperand, dim);
241  }
242 }

244 void ValueBoundsConstraintSet::projectOut(int64_t pos) {
245  assert(pos >= 0 && pos < static_cast<int64_t>(positionToValueDim.size()) &&
246  "invalid position");
247  cstr.projectOut(pos);
248  if (positionToValueDim[pos].has_value()) {
249  bool erased = valueDimToPosition.erase(*positionToValueDim[pos]);
250  (void)erased;
251  assert(erased && "inconsistent reverse mapping");
252  }
253  positionToValueDim.erase(positionToValueDim.begin() + pos);
254  // Update reverse mapping.
255  for (int64_t i = pos, e = positionToValueDim.size(); i < e; ++i)
256  if (positionToValueDim[i].has_value())
257  valueDimToPosition[*positionToValueDim[i]] = i;
258 }

260 void ValueBoundsConstraintSet::projectOut(
261  function_ref<bool(ValueDim)> condition) {
262  int64_t nextPos = 0;
263  while (nextPos < static_cast<int64_t>(positionToValueDim.size())) {
264  if (positionToValueDim[nextPos].has_value() &&
265  condition(*positionToValueDim[nextPos])) {
266  projectOut(nextPos);
267  // The column was projected out so another column is now at that position.
268  // Do not increase the counter.
269  } else {
270  ++nextPos;
271  }
272  }

275 LogicalResult ValueBoundsConstraintSet::computeBound(
276  AffineMap &resultMap, ValueDimList &mapOperands, presburger::BoundType type,
277  Value value, std::optional<int64_t> dim, StopConditionFn stopCondition,
278  bool closedUB) {
279 #ifndef NDEBUG
280  assertValidValueDim(value, dim);
281  assert(!stopCondition(value, dim) &&
282  "stop condition should not be satisfied for starting point");
283 #endif // NDEBUG
284 
285  int64_t ubAdjustment = closedUB ? 0 : 1;
286  Builder b(value.getContext());
287  mapOperands.clear();
288 
289  if (stopCondition(value, dim)) {
290  // Special case: If the stop condition is satisfied for the input
291  // value/dimension, directly return it.
292  mapOperands.push_back(std::make_pair(value, dim));
293  AffineExpr bound = b.getAffineDimExpr(0);
294  if (type == BoundType::UB)
295  bound = bound + ubAdjustment;
296  resultMap = AffineMap::get(/*dimCount=*/1, /*symbolCount=*/0,
297  b.getAffineDimExpr(0));
298  return success();
299  }
300 
301  // Process the backward slice of `value` (i.e., reverse use-def chain) until
302  // `stopCondition` is met.
303  ValueDim valueDim = std::make_pair(value, dim.value_or(kIndexValue));
304  ValueBoundsConstraintSet cstr(value.getContext());
305  int64_t pos = cstr.insert(value, dim, /*isSymbol=*/false);
306  cstr.processWorklist(stopCondition);
307 
308  // Project out all variables (apart from `valueDim`) that do not match the
309  // stop condition.
310  cstr.projectOut([&](ValueDim p) {
311  // Do not project out `valueDim`.
312  if (valueDim == p)
313  return false;
314  auto maybeDim =
315  p.second == kIndexValue ? std::nullopt : std::make_optional(p.second);
316  return !stopCondition(p.first, maybeDim);
317  });
318 
319  // Compute lower and upper bounds for `valueDim`.
320  SmallVector<AffineMap> lb(1), ub(1);
321  cstr.cstr.getSliceBounds(pos, 1, value.getContext(), &lb, &ub,
322  /*closedUB=*/true);
323 
324  // Note: There are TODOs in the implementation of `getSliceBounds`. In such a
325  // case, no lower/upper bound can be computed at the moment.
326  // EQ, UB bounds: upper bound is needed.
327  if ((type != BoundType::LB) &&
328  (ub.empty() || !ub[0] || ub[0].getNumResults() == 0))
329  return failure();
330  // EQ, LB bounds: lower bound is needed.
331  if ((type != BoundType::UB) &&
332  (lb.empty() || !lb[0] || lb[0].getNumResults() == 0))
333  return failure();
334 
335  // TODO: Generate an affine map with multiple results.
336  if (type != BoundType::LB)
337  assert(ub.size() == 1 && ub[0].getNumResults() == 1 &&
338  "multiple bounds not supported");
339  if (type != BoundType::UB)
340  assert(lb.size() == 1 && lb[0].getNumResults() == 1 &&
341  "multiple bounds not supported");
342 
343  // EQ bound: lower and upper bound must match.
344  if (type == BoundType::EQ && ub[0] != lb[0])
345  return failure();
346 
347  AffineMap bound;
348  if (type == BoundType::EQ || type == BoundType::LB) {
349  bound = lb[0];
350  } else {
351  // Computed UB is a closed bound.
352  bound = AffineMap::get(ub[0].getNumDims(), ub[0].getNumSymbols(),
353  ub[0].getResult(0) + ubAdjustment);
354  }
355 
356  // Gather all SSA values that are used in the computed bound.
357  assert(cstr.cstr.getNumDimAndSymbolVars() == cstr.positionToValueDim.size() &&
358  "inconsistent mapping state");
359  SmallVector<AffineExpr> replacementDims, replacementSymbols;
360  int64_t numDims = 0, numSymbols = 0;
361  for (int64_t i = 0; i < cstr.cstr.getNumDimAndSymbolVars(); ++i) {
362  // Skip `value`.
363  if (i == pos)
364  continue;
365  // Check if the position `i` is used in the generated bound. If so, it must
366  // be included in the generated affine.apply op.
367  bool used = false;
368  bool isDim = i < cstr.cstr.getNumDimVars();
369  if (isDim) {
370  if (bound.isFunctionOfDim(i))
371  used = true;
372  } else {
373  if (bound.isFunctionOfSymbol(i - cstr.cstr.getNumDimVars()))
374  used = true;
375  }
376 
377  if (!used) {
378  // Not used: Remove dim/symbol from the result.
379  if (isDim) {
380  replacementDims.push_back(b.getAffineConstantExpr(0));
381  } else {
382  replacementSymbols.push_back(b.getAffineConstantExpr(0));
383  }
384  continue;
385  }
386 
387  if (isDim) {
388  replacementDims.push_back(b.getAffineDimExpr(numDims++));
389  } else {
390  replacementSymbols.push_back(b.getAffineSymbolExpr(numSymbols++));
391  }
392 
393  assert(cstr.positionToValueDim[i].has_value() &&
394  "cannot build affine map in terms of anonymous column");
395  ValueBoundsConstraintSet::ValueDim valueDim = *cstr.positionToValueDim[i];
396  Value value = valueDim.first;
397  int64_t dim = valueDim.second;
398  if (dim == ValueBoundsConstraintSet::kIndexValue) {
399  // An index-type value is used: can be used directly in the affine.apply
400  // op.
401  assert(value.getType().isIndex() && "expected index type");
402  mapOperands.push_back(std::make_pair(value, std::nullopt));
403  continue;
404  }
405 
406  assert(cast<ShapedType>(value.getType()).isDynamicDim(dim) &&
407  "expected dynamic dim");
408  mapOperands.push_back(std::make_pair(value, dim));
409  }
410 
411  resultMap = bound.replaceDimsAndSymbols(replacementDims, replacementSymbols,
412  numDims, numSymbols);
413  return success();
414 }

428 LogicalResult ValueBoundsConstraintSet::computeIndependentBound(
429  AffineMap &resultMap, ValueDimList &mapOperands, presburger::BoundType type,
430  Value value, std::optional<int64_t> dim, ValueRange independencies,
431  bool closedUB) {
432  // Return "true" if the given value is independent of all values in
433  // `independencies`. I.e., neither the value itself nor any value in the
434  // backward slice (reverse use-def chain) is contained in `independencies`.
435  auto isIndependent = [&](Value v) {
436  SmallVector<Value> worklist;
437  DenseSet<Value> visited;
438  worklist.push_back(v);
439  while (!worklist.empty()) {
440  Value next = worklist.pop_back_val();
441  if (visited.contains(next))
442  continue;
443  visited.insert(next);
444  if (llvm::is_contained(independencies, next))
445  return false;
446  // TODO: DominanceInfo could be used to stop the traversal early.
447  Operation *op = next.getDefiningOp();
448  if (!op)
449  continue;
450  worklist.append(op->getOperands().begin(), op->getOperands().end());
451  }
452  return true;
453  };
454 
455  // Reify bounds in terms of any independent values.
456  return computeBound(
457  resultMap, mapOperands, type, value, dim,
458  [&](Value v, std::optional<int64_t> d) { return isIndependent(v); },
459  closedUB);
460 }

487 FailureOr<int64_t> ValueBoundsConstraintSet::computeConstantBound(
488  presburger::BoundType type, AffineMap map, ValueDimList operands,
489  StopConditionFn stopCondition, bool closedUB) {
490  assert(map.getNumResults() == 1 && "expected affine map with one result");
491  ValueBoundsConstraintSet cstr(map.getContext());
492 
493  int64_t pos = 0;
494  if (stopCondition) {
495  cstr.populateConstraintsSet(map, operands, stopCondition, &pos);
496  } else {
497  // No stop condition specified: Keep adding constraints until a bound could
498  // be computed.
499  cstr.populateConstraintsSet(
500  map, operands,
501  [&](Value v, std::optional<int64_t> dim) {
502  return cstr.cstr.getConstantBound64(type, pos).has_value();
503  },
504  &pos);
505  }
506  // Compute constant bound for `valueDim`.
507  int64_t ubAdjustment = closedUB ? 0 : 1;
508  if (auto bound = cstr.cstr.getConstantBound64(type, pos))
509  return type == BoundType::UB ? *bound + ubAdjustment : *bound;
510  return failure();
511 }

525 int64_t ValueBoundsConstraintSet::populateConstraintsSet(
526  AffineMap map, ValueDimList operands, StopConditionFn stopCondition,
527  int64_t *posOut) {
528  assert(map.getNumResults() == 1 && "expected affine map with one result");
529  int64_t pos = insert(/*isSymbol=*/false);
530  if (posOut)
531  *posOut = pos;
532 
533  // Add map and operands to the constraint set. Dimensions are converted to
534  // symbols. All operands are added to the worklist.
535  auto mapper = [&](std::pair<Value, std::optional<int64_t>> v) {
536  return getExpr(v.first, v.second);
537  };
538  SmallVector<AffineExpr> dimReplacements = llvm::to_vector(
539  llvm::map_range(ArrayRef(operands).take_front(map.getNumDims()), mapper));
540  SmallVector<AffineExpr> symReplacements = llvm::to_vector(
541  llvm::map_range(ArrayRef(operands).drop_front(map.getNumDims()), mapper));
542  addBound(
543  presburger::BoundType::EQ, pos,
544  map.getResult(0).replaceDimsAndSymbols(dimReplacements, symReplacements));
545 
546  // Process the backward slice of `operands` (i.e., reverse use-def chain)
547  // until `stopCondition` is met.
548  if (stopCondition) {
549  processWorklist(stopCondition);
550  } else {
551  // No stop condition specified: Keep adding constraints until the worklist
552  // is empty.
553  processWorklist([](Value v, std::optional<int64_t> dim) { return false; });
554  }
555 
556  return pos;
557 }

590  AffineMap map =
591  AffineMap::get(/*dimCount=*/0, /*symbolCount=*/2,
592  b.getAffineSymbolExpr(0) - b.getAffineSymbolExpr(1));
593  SmallVector<OpFoldResult> ofrOperands;
594  ofrOperands.push_back(ofr1);
595  ofrOperands.push_back(ofr2);
596  SmallVector<Value> valueOperands;
597  AffineMap foldedMap =
598  foldAttributesIntoMap(b, map, ofrOperands, valueOperands);
599  ValueDimList valueDims;
600  for (Value v : valueOperands) {
601  assert(v.getType().isIndex() && "expected index type");
602  valueDims.emplace_back(v, std::nullopt);
603  }
604  FailureOr<int64_t> delta =
605  computeConstantBound(presburger::BoundType::EQ, foldedMap, valueDims);
606  if (failed(delta))
607  return failure();
608  return *delta == 0;
609 }

611 FailureOr<bool>
612 ValueBoundsConstraintSet::areOverlappingSlices(MLIRContext *ctx,
613  HyperrectangularSlice slice1,
614  HyperrectangularSlice slice2) {
615  assert(slice1.getMixedOffsets().size() == slice1.getMixedOffsets().size() &&
616  "expected slices of same rank");
617  assert(slice1.getMixedSizes().size() == slice1.getMixedSizes().size() &&
618  "expected slices of same rank");
619  assert(slice1.getMixedStrides().size() == slice1.getMixedStrides().size() &&
620  "expected slices of same rank");
621 
622  Builder b(ctx);
623  bool foundUnknownBound = false;
624  for (int64_t i = 0, e = slice1.getMixedOffsets().size(); i < e; ++i) {
625  AffineMap map =
626  AffineMap::get(/*dimCount=*/0, /*symbolCount=*/4,
627  b.getAffineSymbolExpr(0) +
628  b.getAffineSymbolExpr(1) * b.getAffineSymbolExpr(2) -
629  b.getAffineSymbolExpr(3));
630  {
631  // Case 1: Slices are guaranteed to be non-overlapping if
632  // offset1 + size1 * stride1 <= offset2 (for at least one dimension).
633  SmallVector<OpFoldResult> ofrOperands;
634  ofrOperands.push_back(slice1.getMixedOffsets()[i]);
635  ofrOperands.push_back(slice1.getMixedSizes()[i]);
636  ofrOperands.push_back(slice1.getMixedStrides()[i]);
637  ofrOperands.push_back(slice2.getMixedOffsets()[i]);
638  SmallVector<Value> valueOperands;
639  AffineMap foldedMap =
640  foldAttributesIntoMap(b, map, ofrOperands, valueOperands);
641  FailureOr<int64_t> constBound = computeConstantBound(
642  presburger::BoundType::EQ, foldedMap, valueOperands);
643  foundUnknownBound |= failed(constBound);
644  if (succeeded(constBound) && *constBound <= 0)
645  return false;
646  }
647  {
648  // Case 2: Slices are guaranteed to be non-overlapping if
649  // offset2 + size2 * stride2 <= offset1 (for at least one dimension).
650  SmallVector<OpFoldResult> ofrOperands;
651  ofrOperands.push_back(slice2.getMixedOffsets()[i]);
652  ofrOperands.push_back(slice2.getMixedSizes()[i]);
653  ofrOperands.push_back(slice2.getMixedStrides()[i]);
654  ofrOperands.push_back(slice1.getMixedOffsets()[i]);
655  SmallVector<Value> valueOperands;
656  AffineMap foldedMap =
657  foldAttributesIntoMap(b, map, ofrOperands, valueOperands);
658  FailureOr<int64_t> constBound = computeConstantBound(
659  presburger::BoundType::EQ, foldedMap, valueOperands);
660  foundUnknownBound |= failed(constBound);
661  if (succeeded(constBound) && *constBound <= 0)
662  return false;
663  }
664  }
665 
666  // If at least one bound could not be computed, we cannot be certain that the
667  // slices are really overlapping.
668  if (foundUnknownBound)
669  return failure();
670 
671  // All bounds could be computed and none of the above cases applied.
672  // Therefore, the slices are guaranteed to overlap.
673  return true;
674 }

676 FailureOr<bool>
677 ValueBoundsConstraintSet::areEquivalentSlices(MLIRContext *ctx,
678  HyperrectangularSlice slice1,
679  HyperrectangularSlice slice2) {
680  assert(slice1.getMixedOffsets().size() == slice1.getMixedOffsets().size() &&
681  "expected slices of same rank");
682  assert(slice1.getMixedSizes().size() == slice1.getMixedSizes().size() &&
683  "expected slices of same rank");
684  assert(slice1.getMixedStrides().size() == slice1.getMixedStrides().size() &&
685  "expected slices of same rank");
686 
687  // The two slices are equivalent if all of their offsets, sizes and strides
688  // are equal. If equality cannot be determined for at least one of those
689  // values, equivalence cannot be determined and this function returns
690  // "failure".
691  for (auto [offset1, offset2] :
692  llvm::zip_equal(slice1.getMixedOffsets(), slice2.getMixedOffsets())) {
693  FailureOr<bool> equal = areEqual(offset1, offset2);
694  if (failed(equal))
695  return failure();
696  if (!equal.value())
697  return false;
698  }
699  for (auto [size1, size2] :
700  llvm::zip_equal(slice1.getMixedSizes(), slice2.getMixedSizes())) {
701  FailureOr<bool> equal = areEqual(size1, size2);
702  if (failed(equal))
703  return failure();
704  if (!equal.value())
705  return false;
706  }
707  for (auto [stride1, stride2] :
708  llvm::zip_equal(slice1.getMixedStrides(), slice2.getMixedStrides())) {
709  FailureOr<bool> equal = areEqual(stride1, stride2);
710  if (failed(equal))
711  return failure();
712  if (!equal.value())
713  return false;
714  }
715  return true;
716 }
