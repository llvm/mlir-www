37 // Helper function to materialize the semantically correct compare and select
38 // operations given a binary operation with a specific NaN propagation mode.
39 //
40 // In the case of "PROPAGATE" semantics no compare and selection is required and
41 // this function does nothing.
42 //
43 // In the case of "IGNORE" semantics this function materializes a comparison of
44 // the current operands to the op which will return true for any NaN
45 // argument and then selects between the non-NaN operation argument and the
46 // calculated result based on whether the lhs or rhs is NaN or not. In pseudo
47 // code:
48 //
49 // In the case that the op is operating on non floating point types we ignore
50 // the attribute completely, this is consistent with the TOSA spec which has
51 // the following wording: "This attribute is ignored by non floating-point
52 // types."
53 //
54 // binary<op>(lhs, rhs):
55 // result = op(lhs, rhs)
56 // if lhs == NaN return rhs
57 // if rhs == NaN return lhs
58 // return result
59 template <typename OpTy>
60 static Value
61 materializeBinaryNanCheckIfRequired(OpTy op, PatternRewriter &rewriter,
62  Value lhs, Value rhs, Value result) {
63  // NaN propagation has no meaning for non floating point types.
64  if (!isa<FloatType>(getElementTypeOrSelf(lhs)))
65  return result;
66 
67  auto nanMode = op.getNanMode();
68  if (nanMode == "PROPAGATE")
69  return result;
70 
71  // Unordered comparison of NaN against itself will always return true.
72  Value lhsIsNaN = arith::CmpFOp::create(rewriter, op.getLoc(),
73  arith::CmpFPredicate::UNO, lhs, lhs);
74  Value rhsIsNaN = arith::CmpFOp::create(rewriter, op.getLoc(),
75  arith::CmpFPredicate::UNO, rhs, rhs);
76  Value rhsOrResult =
77  arith::SelectOp::create(rewriter, op.getLoc(), lhsIsNaN, rhs, result);
78  return arith::SelectOp::create(rewriter, op.getLoc(), rhsIsNaN, lhs,
79  rhsOrResult);
80 }

82 static Value createLinalgBodyCalculationForElementwiseOp(
83  Operation *op, ValueRange args, ArrayRef<Type> resultTypes,
84  ConversionPatternRewriter &rewriter) {
85  Location loc = op->getLoc();
86  auto elementTy =
87  cast<ShapedType>(op->getOperand(0).getType()).getElementType();
88 
89  // tosa::AbsOp
90  if (isa<tosa::AbsOp>(op) && isa<FloatType>(elementTy))
91  return math::AbsFOp::create(rewriter, loc, resultTypes, args);
92 
93  if (isa<tosa::AbsOp>(op) && isa<IntegerType>(elementTy)) {
94  auto zero = arith::ConstantOp::create(rewriter, loc,
95  rewriter.getZeroAttr(elementTy));
96  auto neg = arith::SubIOp::create(rewriter, loc, zero, args[0]);
97  return arith::MaxSIOp::create(rewriter, loc, args[0], neg);
98  }
99 
100  // tosa::AddOp
101  if (isa<tosa::AddOp>(op) && isa<FloatType>(elementTy))
102  return arith::AddFOp::create(rewriter, loc, resultTypes, args);
103 
104  if (isa<tosa::AddOp>(op) && isa<IntegerType>(elementTy))
105  return arith::AddIOp::create(rewriter, loc, resultTypes, args);
106 
107  // tosa::SubOp
108  if (isa<tosa::SubOp>(op) && isa<FloatType>(elementTy))
109  return arith::SubFOp::create(rewriter, loc, resultTypes, args);
110 
111  if (isa<tosa::SubOp>(op) && isa<IntegerType>(elementTy))
112  return arith::SubIOp::create(rewriter, loc, resultTypes, args);
113 
114  // tosa::IntDivOp
115  if (isa<tosa::IntDivOp>(op) && isa<IntegerType>(elementTy))
116  return arith::DivSIOp::create(rewriter, loc, resultTypes, args);
117 
118  // tosa::ReciprocalOp
119  if (isa<tosa::ReciprocalOp>(op) && isa<FloatType>(elementTy)) {
120  auto one =
121  arith::ConstantOp::create(rewriter, loc, FloatAttr::get(elementTy, 1));
122  return arith::DivFOp::create(rewriter, loc, resultTypes, one, args[0]);
123  }
124 
125  // tosa::MulOp
126  if (isa<tosa::MulOp>(op)) {
127  auto shiftVal = cast<tosa::MulOp>(op).getShift();
128  DenseElementsAttr shiftElem;
129  if (!matchPattern(shiftVal, m_Constant(&shiftElem))) {
130  (void)rewriter.notifyMatchFailure(op, "shift value of mul not found");
131  return nullptr;
132  }
133 
134  int32_t shift = shiftElem.getValues<IntegerAttr>()[0].getInt();
135 
136  if (isa<FloatType>(elementTy)) {
137  if (shift != 0) {
138  (void)rewriter.notifyMatchFailure(op,
139  "Cannot have shift value for float");
140  return nullptr;
141  }
142  return arith::MulFOp::create(rewriter, loc, resultTypes, args[0],
143  args[1]);
144  }
145 
146  if (isa<IntegerType>(elementTy)) {
147  Value a = args[0];
148  Value b = args[1];
149 
150  if (shift > 0) {
151  auto shiftConst =
152  arith::ConstantIntOp::create(rewriter, loc, shift, /*bitwidth=*/8);
153  if (!a.getType().isInteger(32))
154  a = arith::ExtSIOp::create(rewriter, loc, rewriter.getI32Type(), a);
155 
156  if (!b.getType().isInteger(32))
157  b = arith::ExtSIOp::create(rewriter, loc, rewriter.getI32Type(), b);
158 
159  auto result = tosa::ApplyScaleOp::create(
160  rewriter, loc, rewriter.getI32Type(), a, b, shiftConst,
161  rewriter.getStringAttr("SINGLE_ROUND"));
162 
163  if (elementTy.isInteger(32))
164  return result;
165 
166  return arith::TruncIOp::create(rewriter, loc, elementTy, result);
167  }
168 
169  int aWidth = a.getType().getIntOrFloatBitWidth();
170  int bWidth = b.getType().getIntOrFloatBitWidth();
171  int cWidth = resultTypes[0].getIntOrFloatBitWidth();
172 
173  if (aWidth < cWidth)
174  a = arith::ExtSIOp::create(rewriter, loc, resultTypes[0], a);
175  if (bWidth < cWidth)
176  b = arith::ExtSIOp::create(rewriter, loc, resultTypes[0], b);
177 
178  return arith::MulIOp::create(rewriter, loc, resultTypes, a, b);
179  }
180  }
181 
182  // tosa::NegateOp
183  if (isa<tosa::NegateOp>(op)) {
184  auto negate = cast<tosa::NegateOp>(op);
185 
186  FailureOr<int64_t> maybeInZp = negate.getInput1ZeroPoint();
187  if (failed(maybeInZp)) {
188  (void)rewriter.notifyMatchFailure(
189  op, "input1 zero point cannot be statically determined");
190  return nullptr;
191  }
192 
193  FailureOr<int64_t> maybeOutZp = negate.getOutputZeroPoint();
194  if (failed(maybeOutZp)) {
195  (void)rewriter.notifyMatchFailure(
196  op, "output zero point cannot be statically determined");
197  return nullptr;
198  }
199 
200  int64_t inZp = *maybeInZp;
201  int64_t outZp = *maybeOutZp;
202 
203  if (isa<FloatType>(elementTy))
204  return arith::NegFOp::create(rewriter, loc, resultTypes, args[0]);
205 
206  if (isa<IntegerType>(elementTy)) {
207  if (!inZp && !outZp) {
208  auto constant = arith::ConstantOp::create(
209  rewriter, loc, IntegerAttr::get(elementTy, 0));
210  return arith::SubIOp::create(rewriter, loc, resultTypes, constant,
211  args[0]);
212  }
213 
214  // Compute the maximum value that can occur in the intermediate buffer.
215  const int32_t inputBitWidth = elementTy.getIntOrFloatBitWidth();
216  const int64_t zpAdd = inZp + outZp;
217  const int64_t maxValue =
218  APInt::getSignedMaxValue(inputBitWidth).getSExtValue() +
219  std::abs(zpAdd) + 1;
220 
221  // Convert that maximum value into the maximum bitwidth needed to
222  // represent it. We assume 48-bit numbers may be supported further in
223  // the pipeline.
224  int intermediateBitWidth = 64;
225  if (maxValue <= APInt::getSignedMaxValue(16).getSExtValue()) {
226  intermediateBitWidth = 16;
227  } else if (maxValue <= APInt::getSignedMaxValue(32).getSExtValue()) {
228  intermediateBitWidth = 32;
229  } else if (maxValue <= APInt::getSignedMaxValue(48).getSExtValue()) {
230  intermediateBitWidth = 48;
231  }
232 
233  Type intermediateType = rewriter.getIntegerType(intermediateBitWidth);
234  Value zpAddValue = arith::ConstantOp::create(
235  rewriter, loc, rewriter.getIntegerAttr(intermediateType, zpAdd));
236 
237  // The negation can be applied by doing:
238  // outputValue = inZp + outZp - inputValue
239  auto ext =
240  arith::ExtSIOp::create(rewriter, loc, intermediateType, args[0]);
241  auto sub = arith::SubIOp::create(rewriter, loc, zpAddValue, ext);
242 
243  // Clamp to the negation range.
244  Value min = arith::ConstantIntOp::create(
245  rewriter, loc, intermediateType,
246  APInt::getSignedMinValue(inputBitWidth).getSExtValue());
247  Value max = arith::ConstantIntOp::create(
248  rewriter, loc, intermediateType,
249  APInt::getSignedMaxValue(inputBitWidth).getSExtValue());
250  auto clamp = clampIntHelper(loc, sub, min, max, rewriter, false);
251 
252  // Truncate to the final value.
253  return arith::TruncIOp::create(rewriter, loc, elementTy, clamp);
254  }
255  }

290  return result;

295  IntegerAttr::get(elementTy, 1));
296  auto zero = arith::ConstantOp::create(rewriter, loc,

341  return arith::OrIOp::create(rewriter, loc, resultTypes, args);

447  APFloat maxApf = cast<FloatAttr>(op->getAttr("max_val")).getValue();
448  minApf.convert(cast<FloatType>(elementTy).getFloatSemantics(),

453  rewriter, loc, elementTy, rewriter.getFloatAttr(elementTy, minApf));
454  auto max = arith::ConstantOp::create(
455  rewriter, loc, elementTy, rewriter.getFloatAttr(elementTy, maxApf));

458  auto clampOp = llvm::cast<tosa::ClampOp>(op);
459  const auto nanMode = clampOp.getNanMode();
460 
461  // NaN propagation has no meaning for non floating point types.
462  if (!isa<FloatType>(elementTy))
463  return result;
464 
465  // In the case of "PROPAGATE" semantics no compare and selection is
466  // required.
467  if (nanMode == "PROPAGATE")
468  return result;
469

471  // of the current operand to the reduction which will return true for a NaN
472  // argument and then selects between the initial reduction value and the
473  // calculated result based on whether the argument is NaN or not. In pseudo
474  // code:
475  //
476  // reduce<op>(x, init):
477  // result = op(init, x)
478  // return init if x == NaN else result

482  rewriter, op->getLoc(), arith::CmpFPredicate::UNO, args[0], args[0]);
483  // TOSA specifies that in "ignore" NaN mode the result is "min" if the input
484  // is NaN.

486  }

527  if (isa<tosa::SigmoidOp>(op) && isa<FloatType>(elementTy)) {
528  auto one =
529  arith::ConstantOp::create(rewriter, loc, FloatAttr::get(elementTy, 1));
530  auto negate = arith::NegFOp::create(rewriter, loc, resultTypes, args[0]);
531  auto exp = mlir::math::ExpOp::create(rewriter, loc, resultTypes, negate);
532  auto added = arith::AddFOp::create(rewriter, loc, resultTypes, exp, one);
533  return arith::DivFOp::create(rewriter, loc, resultTypes, one, added);
534  }
535 
536  // tosa::CastOp
537  if (isa<tosa::CastOp>(op)) {
538  Type srcTy = elementTy;
539  Type dstTy = resultTypes.front();
540  if (!srcTy.isIntOrFloat() || !dstTy.isIntOrFloat()) {
541  (void)rewriter.notifyMatchFailure(op, "unsupported type");
542  return nullptr;
543  }
544 
545  bool bitExtend =
546  srcTy.getIntOrFloatBitWidth() < dstTy.getIntOrFloatBitWidth();
547 
548  if (srcTy == dstTy)
549  return args.front();
550 
551  if (isa<FloatType>(srcTy) && isa<FloatType>(dstTy) && bitExtend)
552  return arith::ExtFOp::create(rewriter, loc, resultTypes, args,
553  ArrayRef<NamedAttribute>());
554 
555  if (isa<FloatType>(srcTy) && isa<FloatType>(dstTy) && !bitExtend)
556  return arith::TruncFOp::create(rewriter, loc, resultTypes, args,
557  ArrayRef<NamedAttribute>());
558 
559  // 1-bit integers need to be treated as signless.
560  if (srcTy.isInteger(1) && arith::UIToFPOp::areCastCompatible(srcTy, dstTy))
561  return arith::UIToFPOp::create(rewriter, loc, resultTypes, args,
562  ArrayRef<NamedAttribute>());
563 
564  if (srcTy.isInteger(1) && isa<IntegerType>(dstTy) && bitExtend)
565  return arith::ExtUIOp::create(rewriter, loc, resultTypes, args,
566  ArrayRef<NamedAttribute>());
567 
568  // Unsigned integers need an unrealized cast so that they can be passed
569  // to UIToFP.
570  if (srcTy.isUnsignedInteger() && isa<FloatType>(dstTy)) {
571  auto unrealizedCast =
572  UnrealizedConversionCastOp::create(
573  rewriter, loc,
574  rewriter.getIntegerType(srcTy.getIntOrFloatBitWidth()), args[0])
575  .getResult(0);
576  return arith::UIToFPOp::create(rewriter, loc, resultTypes[0],
577  unrealizedCast);
578  }
579 
580  // All other si-to-fp conversions should be handled by SIToFP.
581  if (arith::SIToFPOp::areCastCompatible(srcTy, dstTy))
582  return arith::SIToFPOp::create(rewriter, loc, resultTypes, args,
583  ArrayRef<NamedAttribute>());
584 
585  // Casting to boolean, floats need to only be checked as not-equal to zero.
586  if (isa<FloatType>(srcTy) && dstTy.isInteger(1)) {
587  Value zero = arith::ConstantOp::create(rewriter, loc,
588  rewriter.getFloatAttr(srcTy, 0.0));
589  return arith::CmpFOp::create(rewriter, loc, arith::CmpFPredicate::UNE,
590  args.front(), zero);
591  }
592 
593  if (arith::FPToSIOp::areCastCompatible(srcTy, dstTy)) {
594  auto rounded = math::RoundEvenOp::create(rewriter, loc, args[0]);
595 
596  const auto &fltSemantics = cast<FloatType>(srcTy).getFloatSemantics();
597  // Check whether neither int min nor int max can be represented in the
598  // input floating-point type due to too short exponent range.
599  if (static_cast<int>(dstTy.getIntOrFloatBitWidth()) - 1 >
600  APFloat::semanticsMaxExponent(fltSemantics)) {
601  // Use cmp + select to replace infinites by int min / int max. Other
602  // integral values can be represented in the integer space.
603  auto conv = arith::FPToSIOp::create(rewriter, loc, dstTy, rounded);
604  auto posInf = arith::ConstantOp::create(
605  rewriter, loc,
606  rewriter.getFloatAttr(getElementTypeOrSelf(srcTy),
607  APFloat::getInf(fltSemantics)));
608  auto negInf = arith::ConstantOp::create(
609  rewriter, loc,
610  rewriter.getFloatAttr(
611  getElementTypeOrSelf(srcTy),
612  APFloat::getInf(fltSemantics, /*Negative=*/true)));
613  auto overflow = arith::CmpFOp::create(
614  rewriter, loc, arith::CmpFPredicate::UEQ, rounded, posInf);
615  auto underflow = arith::CmpFOp::create(
616  rewriter, loc, arith::CmpFPredicate::UEQ, rounded, negInf);
617  auto intMin = arith::ConstantOp::create(
618  rewriter, loc,
619  rewriter.getIntegerAttr(
620  getElementTypeOrSelf(dstTy),
621  APInt::getSignedMinValue(dstTy.getIntOrFloatBitWidth())));
622  auto intMax = arith::ConstantOp::create(
623  rewriter, loc,
624  rewriter.getIntegerAttr(
625  getElementTypeOrSelf(dstTy),
626  APInt::getSignedMaxValue(dstTy.getIntOrFloatBitWidth())));
627  auto maxClamped =
628  arith::SelectOp::create(rewriter, loc, overflow, intMax, conv);
629  return arith::SelectOp::create(rewriter, loc, underflow, intMin,
630  maxClamped);
631  }
632 
633  auto intMinFP = arith::ConstantOp::create(
634  rewriter, loc,
635  rewriter.getFloatAttr(
636  getElementTypeOrSelf(srcTy),
637  APInt::getSignedMinValue(dstTy.getIntOrFloatBitWidth())
638  .getSExtValue()));
639 
640  // Check whether the mantissa has enough bits to represent int max.
641  if (cast<FloatType>(srcTy).getFPMantissaWidth() >=
642  dstTy.getIntOrFloatBitWidth() - 1) {
643  // Int min can also be represented since it is a power of two and thus
644  // consists of a single leading bit. Therefore we can clamp the input
645  // in the floating-point domain.
646 
647  auto intMaxFP = arith::ConstantOp::create(
648  rewriter, loc,
649  rewriter.getFloatAttr(
650  getElementTypeOrSelf(srcTy),
651  APInt::getSignedMaxValue(dstTy.getIntOrFloatBitWidth())
652  .getSExtValue()));
653 
654  Value clamped =
655  clampFloatHelper(loc, rounded, intMinFP, intMaxFP, rewriter);
656  return arith::FPToSIOp::create(rewriter, loc, dstTy, clamped);
657  }
658 
659  // Due to earlier check we know exponant range is big enough to represent
660  // int min. We can therefore rely on int max + 1 being representable as
661  // well because it's just int min with a positive sign. So clamp the min
662  // value and compare against that to select the max int value if needed.
663  auto intMaxPlusOneFP = arith::ConstantOp::create(
664  rewriter, loc,
665  rewriter.getFloatAttr(
666  getElementTypeOrSelf(srcTy),
667  static_cast<double>(
668  APInt::getSignedMaxValue(dstTy.getIntOrFloatBitWidth())
669  .getSExtValue()) +
670  1.0f));
671 
672  auto intMax = arith::ConstantOp::create(
673  rewriter, loc,
674  rewriter.getIntegerAttr(
675  getElementTypeOrSelf(dstTy),
676  APInt::getSignedMaxValue(dstTy.getIntOrFloatBitWidth())));
677  auto minClampedFP =
678  arith::MaximumFOp::create(rewriter, loc, rounded, intMinFP);
679  auto minClamped =
680  arith::FPToSIOp::create(rewriter, loc, dstTy, minClampedFP);
681  auto overflow = arith::CmpFOp::create(
682  rewriter, loc, arith::CmpFPredicate::UGE, rounded, intMaxPlusOneFP);
683  return arith::SelectOp::create(rewriter, loc, overflow, intMax,
684  minClamped);
685  }
686 
687  // Casting to boolean, integers need to only be checked as not-equal to
688  // zero.
689  if (isa<IntegerType>(srcTy) && dstTy.isInteger(1)) {
690  Value zero = arith::ConstantIntOp::create(rewriter, loc, 0,
691  srcTy.getIntOrFloatBitWidth());
692  return arith::CmpIOp::create(rewriter, loc, arith::CmpIPredicate::ne,
693  args.front(), zero);
694  }
695 
696  if (isa<IntegerType>(srcTy) && isa<IntegerType>(dstTy) && bitExtend)
697  return arith::ExtSIOp::create(rewriter, loc, resultTypes, args,
698  ArrayRef<NamedAttribute>());
699 
700  if (isa<IntegerType>(srcTy) && isa<IntegerType>(dstTy) && !bitExtend) {
701  return arith::TruncIOp::create(rewriter, loc, dstTy, args[0]);
702  }
703  }
704 
705  (void)rewriter.notifyMatchFailure(
706  op, "unhandled op for linalg body calculation for elementwise op");
707  return nullptr;
708 }

749 // Compute the runtime dimension size for dimension 'dim' of the output by
750 // inspecting input 'operands', all of which are expected to have the same rank.
751 // This function returns a pair {targetSize, masterOperand}.
752 //
753 // The runtime size of the output dimension is returned either as a statically
754 // computed attribute or as a runtime SSA value.
755 //
756 // If the target size was inferred directly from one dominating operand, that
757 // operand is returned in 'masterOperand'. If the target size is inferred from
758 // multiple operands, 'masterOperand' is set to nullptr.
759 static std::pair<OpFoldResult, Value>
760 computeTargetSize(PatternRewriter &rewriter, Location loc, IndexPool &indexPool,
761  ValueRange operands, int64_t dim) {
762  // If any input operand contains a static size greater than 1 for this
763  // dimension, that is the target size. An occurrence of an additional static
764  // dimension greater than 1 with a different value is undefined behavior.
765  for (auto operand : operands) {
766  auto size = cast<RankedTensorType>(operand.getType()).getDimSize(dim);
767  if (ShapedType::isStatic(size) && size > 1)
768  return {rewriter.getIndexAttr(size), operand};
769  }
770 
771  // Filter operands with dynamic dimension
772  auto operandsWithDynamicDim =
773  llvm::filter_to_vector(operands, [&](Value operand) {
774  return cast<RankedTensorType>(operand.getType()).isDynamicDim(dim);
775  });
776 
777  // If no operand has a dynamic dimension, it means all sizes were 1
778  if (operandsWithDynamicDim.empty())
779  return {rewriter.getIndexAttr(1), operands.front()};
780 
781  // Emit code that computes the runtime size for this dimension. If there is
782  // only one operand with a dynamic dimension, it is considered the master
783  // operand that determines the runtime size of the output dimension.
784  auto targetSize =
785  getTensorDim(rewriter, loc, indexPool, operandsWithDynamicDim[0], dim);
786  if (operandsWithDynamicDim.size() == 1)
787  return {targetSize, operandsWithDynamicDim[0]};
788 
789  // Calculate maximum size among all dynamic dimensions
790  for (size_t i = 1; i < operandsWithDynamicDim.size(); i++) {
791  auto nextSize =
792  getTensorDim(rewriter, loc, indexPool, operandsWithDynamicDim[i], dim);
793  targetSize = arith::MaxUIOp::create(rewriter, loc, targetSize, nextSize);
794  }
795  return {targetSize, nullptr};
796 }

798 // Compute the runtime output size for all dimensions. This function returns
799 // a pair {targetShape, masterOperands}.
800 static std::pair<SmallVector<OpFoldResult>, SmallVector<Value>>
801 computeTargetShape(PatternRewriter &rewriter, Location loc,
802  IndexPool &indexPool, ValueRange operands) {
803  assert(!operands.empty());
804  auto rank = cast<RankedTensorType>(operands.front().getType()).getRank();
805  SmallVector<OpFoldResult> targetShape;
806  SmallVector<Value> masterOperands;
807  for (auto dim : llvm::seq<int64_t>(0, rank)) {
808  auto [targetSize, masterOperand] =
809  computeTargetSize(rewriter, loc, indexPool, operands, dim);
810  targetShape.push_back(targetSize);
811  masterOperands.push_back(masterOperand);
812  }
813  return {targetShape, masterOperands};
814 }

816 static Value broadcastDynamicDimension(PatternRewriter &rewriter, Location loc,
817  IndexPool &indexPool, Value operand,
818  int64_t dim, OpFoldResult targetSize,
819  Value masterOperand) {
820  // Nothing to do if this is a static dimension
821  auto rankedTensorType = cast<RankedTensorType>(operand.getType());
822  if (!rankedTensorType.isDynamicDim(dim))
823  return operand;
824 
825  // If the target size for this dimension was directly inferred by only taking
826  // this operand into account, there is no need to broadcast. This is an
827  // optimization that will prevent redundant control flow, and constitutes the
828  // main motivation for tracking "master operands".
829  if (operand == masterOperand)
830  return operand;
831 
832  // Affine maps for 'linalg.generic' op
833  auto rank = rankedTensorType.getRank();
834  SmallVector<AffineExpr> affineExprs;
835  for (auto index : llvm::seq<int64_t>(0, rank)) {
836  auto affineExpr = index == dim ? rewriter.getAffineConstantExpr(0)
837  : rewriter.getAffineDimExpr(index);
838  affineExprs.push_back(affineExpr);
839  }
840  auto broadcastAffineMap =
841  AffineMap::get(rank, 0, affineExprs, rewriter.getContext());
842  auto identityAffineMap = rewriter.getMultiDimIdentityMap(rank);
843  SmallVector<AffineMap> affineMaps = {broadcastAffineMap, identityAffineMap};
844 
845  // Check if broadcast is necessary
846  auto one = createIndex(rewriter, loc, indexPool, 1);
847  auto runtimeSize = getTensorDim(rewriter, loc, indexPool, operand, dim);
848  auto broadcastNecessary = arith::CmpIOp::create(
849  rewriter, loc, arith::CmpIPredicate::eq, runtimeSize, one);
850 
851  // Emit 'then' region of 'scf.if'
852  auto emitThenRegion = [&](OpBuilder &opBuilder, Location loc) {
853  // It is not safe to cache constants across regions.
854  // New constants could potentially violate dominance requirements.
855  IndexPool localPool;
856 
857  // Emit 'tensor.empty' op
858  SmallVector<OpFoldResult> outputTensorShape;
859  for (auto index : llvm::seq<int64_t>(0, rank)) {
860  auto size = index == dim ? targetSize
861  : getOrFoldTensorDim(rewriter, loc, localPool,
862  operand, index);
863  outputTensorShape.push_back(size);
864  }
865  Value outputTensor = tensor::EmptyOp::create(
866  opBuilder, loc, outputTensorShape, rankedTensorType.getElementType());
867 
868  // Emit 'linalg.generic' op
869  auto resultTensor =
870  linalg::GenericOp::create(
871  opBuilder, loc, outputTensor.getType(), operand, outputTensor,
872  affineMaps, getNParallelLoopsAttrs(rank),
873  [&](OpBuilder &opBuilder, Location loc, ValueRange blockArgs) {
874  // Emit 'linalg.yield' op
875  linalg::YieldOp::create(opBuilder, loc, blockArgs.front());
876  })
877  .getResult(0);
878 
879  // Cast to original operand type if necessary
880  auto castResultTensor = rewriter.createOrFold<tensor::CastOp>(
881  loc, operand.getType(), resultTensor);
882 
883  // Emit 'scf.yield' op
884  scf::YieldOp::create(opBuilder, loc, castResultTensor);
885  };
886 
887  // Emit 'else' region of 'scf.if'
888  auto emitElseRegion = [&](OpBuilder &opBuilder, Location loc) {
889  scf::YieldOp::create(opBuilder, loc, operand);
890  };
891 
892  // Emit 'scf.if' op
893  auto ifOp = scf::IfOp::create(rewriter, loc, broadcastNecessary,
894  emitThenRegion, emitElseRegion);
895  return ifOp.getResult(0);
896 }

898 static Value broadcastDynamicDimensions(PatternRewriter &rewriter, Location loc,
899  IndexPool &indexPool, Value operand,
900  ArrayRef<OpFoldResult> targetShape,
901  ArrayRef<Value> masterOperands) {
902  int64_t rank = cast<RankedTensorType>(operand.getType()).getRank();
903  assert((int64_t)targetShape.size() == rank);
904  assert((int64_t)masterOperands.size() == rank);
905  for (auto index : llvm::seq<int64_t>(0, rank))
906  operand =
907  broadcastDynamicDimension(rewriter, loc, indexPool, operand, index,
908  targetShape[index], masterOperands[index]);
909  return operand;
910 }

912 static SmallVector<Value>
913 broadcastDynamicDimensions(PatternRewriter &rewriter, Location loc,
914  IndexPool &indexPool, ValueRange operands,
915  ArrayRef<OpFoldResult> targetShape,
916  ArrayRef<Value> masterOperands) {

919  return operands;

928 static LogicalResult
929 emitElementwiseComputation(ConversionPatternRewriter &rewriter, Location loc,
930  Operation *operation, ValueRange operands,
931  ArrayRef<OpFoldResult> targetShape,
932  const TypeConverter &converter) {
933  // Generate output tensor
934  auto resultType = cast_or_null<RankedTensorType>(
935  converter.convertType(operation->getResultTypes().front()));
936  if (!resultType) {
937  return rewriter.notifyMatchFailure(operation, "failed to convert type");
938  }
939  Value outputTensor = tensor::EmptyOp::create(rewriter, loc, targetShape,
940  resultType.getElementType());
941 
942  // Create affine maps. Input affine maps broadcast static dimensions of size
943  // 1. The output affine map is an identity map.
944  //
945  auto rank = resultType.getRank();
946  auto affineMaps = llvm::map_to_vector(operands, [&](Value operand) {
947  auto shape = cast<ShapedType>(operand.getType()).getShape();
948  SmallVector<AffineExpr> affineExprs;
949  for (auto it : llvm::enumerate(shape)) {
950  // Prefer producting identity maps whenever possible (i.e. no broadcasting
951  // needed) because some transforms (like reshape folding)
952  // do not support affine constant exprs.
953  bool requiresBroadcast =
954  (it.value() == 1 && resultType.getDimSize(it.index()) != 1);
955  auto affineExpr = requiresBroadcast
956  ? rewriter.getAffineConstantExpr(0)
957  : rewriter.getAffineDimExpr(it.index());
958  affineExprs.push_back(affineExpr);
959  }
960  return AffineMap::get(rank, 0, affineExprs, rewriter.getContext());
961  });
962  affineMaps.push_back(rewriter.getMultiDimIdentityMap(rank));
963 
964  // Emit 'linalg.generic' op
965  bool encounteredError = false;
966  auto linalgOp = linalg::GenericOp::create(
967  rewriter, loc, outputTensor.getType(), operands, outputTensor, affineMaps,
968  getNParallelLoopsAttrs(rank),
969  [&](OpBuilder &opBuilder, Location loc, ValueRange blockArgs) {
970  Value opResult = createLinalgBodyCalculationForElementwiseOp(
971  operation, blockArgs.take_front(operation->getNumOperands()),
972  {resultType.getElementType()}, rewriter);
973  if (!opResult) {
974  encounteredError = true;
975  return;
976  }
977  linalg::YieldOp::create(opBuilder, loc, opResult);
978  });
979  if (encounteredError)
980  return rewriter.notifyMatchFailure(
981  operation, "unable to create linalg.generic body for elementwise op");
982 
983  // Cast 'linalg.generic' result into original result type if needed
984  auto castResult = rewriter.createOrFold<tensor::CastOp>(
985  loc, resultType, linalgOp->getResult(0));
986  rewriter.replaceOp(operation, castResult);
987  return success();
988 }

1002 elementwiseMatchAndRewriteHelper(Operation *operation, ValueRange operands,
1003  ConversionPatternRewriter &rewriter,
1004  const TypeConverter &converter) {
1005 
1006  // Collect op properties
1007  assert(operation->getNumResults() == 1 && "elementwise op expects 1 result");
1008  assert(operation->getNumOperands() >= 1 &&

1021  broadcastDynamicDimensions(rewriter, loc, indexPool, operandsToBroadcast,
1022  targetShape, masterOperands);

1062  return rewriter.getIntegerAttr(elementTy, APInt::getAllOnes(1));

1073  return rewriter.getIntegerAttr(
1074  elementTy, APInt::getSignedMinValue(elementTy.getIntOrFloatBitWidth()));

1076  return {};

1079 // Creates the body calculation for a reduction. The operations vary depending
1080 // on the input type.

1082  ValueRange args,
1083  Type elementTy,
1084  PatternRewriter &rewriter) {

1087  return arith::AddFOp::create(rewriter, loc, args);
1088  }
1089 
1090  if (isa<tosa::ReduceSumOp>(op) && isa<IntegerType>(elementTy)) {
1091  return arith::AddIOp::create(rewriter, loc, args);
1092  }
1093 
1094  if (isa<tosa::ReduceProductOp>(op) && isa<FloatType>(elementTy)) {
1095  return arith::MulFOp::create(rewriter, loc, args);
1096  }
1097 
1098  if (isa<tosa::ReduceProductOp>(op) && isa<IntegerType>(elementTy)) {
1099  return arith::MulIOp::create(rewriter, loc, args);
1100  }
1101 
1102  if (isa<tosa::ReduceMinOp>(op) && isa<FloatType>(elementTy)) {
1103  return arith::MinimumFOp::create(rewriter, loc, args[0], args[1]);
1104  }
1105 
1106  if (isa<tosa::ReduceMinOp>(op) && isa<IntegerType>(elementTy)) {
1107  return arith::MinSIOp::create(rewriter, loc, args[0], args[1]);
1108  }
1109

1111  return arith::MaximumFOp::create(rewriter, loc, args[0], args[1]);
1112  }
1113 
1114  if (isa<tosa::ReduceMaxOp>(op) && isa<IntegerType>(elementTy)) {
1115  return arith::MaxSIOp::create(rewriter, loc, args[0], args[1]);
1116  }
1117 
1118  if (isa<tosa::ReduceAllOp>(op) && elementTy.isInteger(1))
1119  return arith::AndIOp::create(rewriter, loc, args);
1120 
1121  if (isa<tosa::ReduceAnyOp>(op) && elementTy.isInteger(1))
1122  return arith::OrIOp::create(rewriter, loc, args);
1123 
1124  return {};

1127 // Performs the match and rewrite for reduction operations. This includes
1128 // declaring a correctly sized initial value, and the linalg.generic operation
1129 // that reduces across the specified axis.
1130 template <typename OpTy>
1131 static LogicalResult reduceMatchAndRewriteHelper(OpTy op, uint64_t axis,
1132  PatternRewriter &rewriter) {
1133  auto loc = op->getLoc();
1134  auto inputTy = dyn_cast<RankedTensorType>(op->getOperand(0).getType());
1135  auto resultTy = dyn_cast<RankedTensorType>(op->getResult(0).getType());
1136  if (!inputTy || !resultTy)
1137  return rewriter.notifyMatchFailure(op, "unranked tensors not supported");
1138 
1139  auto elementTy = resultTy.getElementType();
1140  Value input = op->getOperand(0);
1141 
1142  SmallVector<int64_t> reduceShape;
1143  SmallVector<Value> dynDims;
1144  for (unsigned i = 0; i < inputTy.getRank(); i++) {
1145  if (axis != i) {
1146  reduceShape.push_back(inputTy.getDimSize(i));
1147  if (inputTy.isDynamicDim(i))
1148  dynDims.push_back(tensor::DimOp::create(rewriter, loc, input, i));
1149  }
1150  }
1151 
1152  SmallVector<Value> inputs, outputs;
1153  inputs.push_back(input);
1154 
1155  // First fill the output buffer with the init value.
1156  auto emptyTensor = tensor::EmptyOp::create(rewriter, loc, reduceShape,
1157  resultTy.getElementType(), dynDims)
1158  .getResult();
1159 
1160  auto fillValueAttr = createInitialValueForReduceOp(op, elementTy, rewriter);
1161  if (!fillValueAttr)
1162  return rewriter.notifyMatchFailure(
1163  op, "No initial value found for reduction operation");
1164 
1165  auto fillValue = arith::ConstantOp::create(rewriter, loc, fillValueAttr);
1166  auto filledTensor =
1167  linalg::FillOp::create(rewriter, loc, ValueRange{fillValue},
1168  ValueRange{emptyTensor})
1169  .result();
1170  outputs.push_back(filledTensor);
1171 
1172  bool isNanIgnoreMode = false;
1173  if constexpr (std::is_same_v<OpTy, tosa::ReduceMinOp> ||
1174  std::is_same_v<OpTy, tosa::ReduceMaxOp>) {
1175  // NaN propagation has no meaning for non floating point types.
1176  if (isa<FloatType>(elementTy) && op.getNanMode() == "IGNORE") {
1177  isNanIgnoreMode = true;
1178  // Because the TOSA spec requires the result be NaN iff all elements in
1179  // the reduction are NaN we can't simply perform a compare and select.
1180  // Additionally we have to keep track of whether we've seen any non-NaN
1181  // values and then do a final select based on this predicate.
1182  auto trueAttr = rewriter.getBoolAttr(true);
1183  auto trueValue = arith::ConstantOp::create(rewriter, loc, trueAttr);
1184  auto emptyBoolTensor =
1185  tensor::EmptyOp::create(rewriter, loc, reduceShape,
1186  trueValue.getType(), dynDims)
1187  .getResult();
1188  auto allResultsNaNTensor =
1189  linalg::FillOp::create(rewriter, loc, ValueRange{trueValue},
1190  ValueRange{emptyBoolTensor})
1191  .result();
1192  // Note that because the linalg::ReduceOp has two variadic arguments
1193  // (inputs and outputs) and it has the SameVariadicOperandSize trait we
1194  // need to have the same number of inputs and outputs.
1195  //
1196  // The second input isn't actually used anywhere since the value used to
1197  // update the NaN flag is calculated inside the body of the reduction and
1198  // then used to update an out value.
1199  // In order to satisfy type constraints we just pass another copy of the
1200  // input here.
1201  inputs.push_back(input);
1202  outputs.push_back(allResultsNaNTensor);
1203  }
1204  }
1205 
1206  bool didEncounterError = false;
1207  linalg::LinalgOp linalgOp = linalg::ReduceOp::create(
1208  rewriter, loc, inputs, outputs, axis,
1209  [&](OpBuilder &nestedBuilder, Location nestedLoc, ValueRange blockArgs) {
1210  std::array<Value, 2> binaryArgs{
1211  blockArgs[0], isNanIgnoreMode ? blockArgs[2] : blockArgs[1]};
1212  auto result = createLinalgBodyCalculationForReduceOp(
1213  op, binaryArgs, elementTy, rewriter);
1214  if (result)
1215  didEncounterError = true;
1216 
1217  SmallVector<Value> resultsToYield;
1218  if (isNanIgnoreMode) {
1219  auto inputValue = blockArgs[0];
1220  auto initialValue = blockArgs[2];
1221  auto oldAllResultsNanFlagValue = blockArgs[3];
1222 
1223  // Unordered comparison of NaN against itself will always return true.
1224  Value isNaN = arith::CmpFOp::create(nestedBuilder, op->getLoc(),
1225  arith::CmpFPredicate::UNO,
1226  inputValue, inputValue);
1227  // If we've encountered a NaN, take the non-NaN value.
1228  auto selectOp = arith::SelectOp::create(nestedBuilder, op->getLoc(),
1229  isNaN, initialValue, result);
1230  // Update the flag which keeps track of whether we have seen a non-NaN
1231  // value.
1232  auto newAllResultsNanFlagValue = arith::AndIOp::create(
1233  nestedBuilder, op->getLoc(), oldAllResultsNanFlagValue, isNaN);
1234  resultsToYield.push_back(selectOp);
1235  resultsToYield.push_back(newAllResultsNanFlagValue);
1236  } else {
1237  resultsToYield.push_back(result);
1238  }
1239  linalg::YieldOp::create(nestedBuilder, loc, resultsToYield);
1240  });
1241 
1242  if (!didEncounterError)
1243  return rewriter.notifyMatchFailure(
1244  op, "unable to create linalg.generic body for reduce op");
1245 
1246  if (isNanIgnoreMode) {
1247  // Materialize a check to see whether we encountered any non-NaN values, if
1248  // we didn't we need to select a tensor of NaNs since the result will just
1249  // be the initial identity value propagated through all the compares and
1250  // selects inside the reduction.
1251 
1252  // Create a tensor full of NaNs.
1253  auto nanValueAttr = rewriter.getFloatAttr(
1254  elementTy,
1255  APFloat::getNaN(cast<FloatType>(elementTy).getFloatSemantics(), false));
1256  auto nanValue = arith::ConstantOp::create(rewriter, loc, nanValueAttr);
1257  auto emptyNanTensor =
1258  tensor::EmptyOp::create(rewriter, loc, reduceShape,
1259  resultTy.getElementType(), dynDims)
1260  .getResult();
1261  auto nanFilledTensor =
1262  linalg::FillOp::create(rewriter, loc, ValueRange{nanValue},
1263  ValueRange{emptyNanTensor})
1264  .result();
1265 
1266  // Create an empty tensor, non need to fill this since it will be
1267  // overwritten by the select.
1268  auto finalEmptyTensor =
1269  tensor::EmptyOp::create(rewriter, loc, reduceShape,
1270  resultTy.getElementType(), dynDims)
1271  .getResult();
1272 
1273  // Do a selection between the tensors akin to:
1274  // result = NaN if "all results NaN" else result.
1275  SmallVector<Value> ins, outs;
1276  ins.push_back(linalgOp->getOpResult(1));
1277  ins.push_back(nanFilledTensor);
1278  ins.push_back(linalgOp->getResult(0));
1279  outs.push_back(finalEmptyTensor);
1280  auto linalgSelect =
1281  linalg::SelectOp::create(rewriter, op->getLoc(), ins, outs);
1282  linalgOp = linalgSelect;
1283  }
1284 
1285  SmallVector<ReassociationExprs, 4> reassociationMap;
1286  uint64_t expandInputRank =
1287  cast<ShapedType>(linalgOp->getResults()[0].getType()).getRank();
1288  reassociationMap.resize(expandInputRank);
1289 
1290  for (uint64_t i = 0; i < expandInputRank; i++) {
1291  int32_t dimToPush = i > axis ? i + 1 : i;
1292  reassociationMap[i].push_back(rewriter.getAffineDimExpr(dimToPush));
1293  }
1294 
1295  if (expandInputRank != 0) {
1296  int32_t expandedDim = axis < expandInputRank ? axis : expandInputRank - 1;
1297  reassociationMap[expandedDim].push_back(
1298  rewriter.getAffineDimExpr(expandedDim + 1));
1299  }
1300 
1301  // Lower directly to `tensor::ExpandShapeOp` instead of `tosa::ReshapeOp`,
1302  // since here we know which dimension to expand, and `tosa::ReshapeOp` would
1303  // not have access to such information. This matters when handling dynamically
1304  // sized tensors.
1305  rewriter.replaceOpWithNewOp<tensor::ExpandShapeOp>(
1306  op, resultTy, linalgOp->getResults()[0], reassociationMap);
1307  return success();
1308 }

1327 public:
1328  using OpRewritePattern<tosa::RescaleOp>::OpRewritePattern;
1329 
1330  LogicalResult matchAndRewrite(tosa::RescaleOp op,
1331  PatternRewriter &rewriter) const final {
1332  auto loc = op.getLoc();
1333  auto input = op.getInput();
1334  auto inputTy = cast<ShapedType>(op.getInput().getType());
1335  auto outputTy = cast<ShapedType>(op.getOutput().getType());
1336  unsigned rank = inputTy.getRank();
1337 
1338  // This is an illegal configuration. terminate and log an error
1339  if (op.getRoundingMode() == "INEXACT_ROUND")
1340  return rewriter.notifyMatchFailure(
1341  op, "tosa.rescale with rounding mode = 'INEXACT_ROUND' is not "
1342  "currently supported");
1343  if (op.getRoundingMode() == "DOUBLE_ROUND" && !op.getScale32())
1344  return rewriter.notifyMatchFailure(
1345  op, "tosa.rescale requires scale32 for double_round to be true");
1346 
1347  if (!isa<IntegerType>(inputTy.getElementType()))
1348  return rewriter.notifyMatchFailure(op, "only support integer type");
1349 
1350  SmallVector<Value> dynDims;
1351  for (int i = 0; i < outputTy.getRank(); i++) {
1352  if (outputTy.isDynamicDim(i)) {
1353  dynDims.push_back(tensor::DimOp::create(rewriter, loc, input, i));
1354  }
1355  }
1356 
1357  // The shift and multiplier values.
1358  DenseElementsAttr shiftElems;
1359  if (!matchPattern(op.getShift(), m_Constant(&shiftElems)))
1360  return rewriter.notifyMatchFailure(
1361  op, "tosa.rescale requires constant shift input values");
1362 
1363  DenseElementsAttr multiplierElems;
1364  if (!matchPattern(op.getMultiplier(), m_Constant(&multiplierElems)))
1365  return rewriter.notifyMatchFailure(
1366  op, "tosa.rescale requires constant multiplier input values");
1367

1369  llvm::to_vector(shiftElems.getValues<int8_t>());
1370  // explicit cast is required here
1371  llvm::SmallVector<int32_t> multiplierValues = llvm::to_vector(
1372  llvm::map_range(multiplierElems.getValues<IntegerAttr>(),
1373  [](IntegerAttr attr) -> int32_t {
1374  return static_cast<int32_t>(attr.getInt());
1375  }));
1376 
1377  // If we shift by more than the bitwidth, this just sets to 0.
1378  for (int i = 0, s = multiplierValues.size(); i < s; i++) {
1379  if (shiftValues[i] > 63) {
1380  shiftValues[i] = 0;
1381  multiplierValues[i] = 0;
1382  }
1383  }
1384 
1385  // Double round only occurs if shift is greater than 31, check that this
1386  // is ever true.
1387 
1388  bool doubleRound =
1389  op.getRoundingMode() == "DOUBLE_ROUND" &&
1390  llvm::any_of(shiftValues, [](int32_t v) { return v > 31; });
1391  StringAttr roundingMode = doubleRound
1392  ? rewriter.getStringAttr("DOUBLE_ROUND")
1393  : rewriter.getStringAttr("SINGLE_ROUND");
1394 
1395  SmallVector<AffineMap> indexingMaps = {
1396  rewriter.getMultiDimIdentityMap(rank)};
1397  SmallVector<Value, 4> genericInputs = {input};
1398 
1399  // If we are rescaling per-channel then we need to store the multiplier
1400  // values in a buffer.
1401  Value multiplierConstant;
1402  int64_t multiplierArg = 0;
1403  if (multiplierValues.size() == 1) {
1404  multiplierConstant = arith::ConstantOp::create(
1405  rewriter, loc, rewriter.getI32IntegerAttr(multiplierValues.front()));
1406  } else {
1407  SmallVector<AffineExpr, 2> multiplierExprs{
1408  rewriter.getAffineDimExpr(rank - 1)};
1409  auto multiplierType =
1410  RankedTensorType::get({static_cast<int64_t>(multiplierValues.size())},
1411  rewriter.getI32Type());
1412  genericInputs.push_back(arith::ConstantOp::create(
1413  rewriter, loc,
1414  DenseIntElementsAttr::get(multiplierType, multiplierValues)));
1415 
1416  indexingMaps.push_back(AffineMap::get(/*dimCount=*/rank,
1417  /*symbolCount=*/0, multiplierExprs,
1418  rewriter.getContext()));
1419 
1420  multiplierArg = indexingMaps.size() - 1;
1421  }
1422 
1423  // If we are rescaling per-channel then we need to store the shift
1424  // values in a buffer.
1425  Value shiftConstant;
1426  int64_t shiftArg = 0;
1427  if (shiftValues.size() == 1) {
1428  shiftConstant = arith::ConstantOp::create(
1429  rewriter, loc, rewriter.getI8IntegerAttr(shiftValues.front()));
1430  } else {
1431  SmallVector<AffineExpr, 2> shiftExprs = {
1432  rewriter.getAffineDimExpr(rank - 1)};
1433  auto shiftType =
1434  RankedTensorType::get({static_cast<int64_t>(shiftValues.size())},
1435  rewriter.getIntegerType(8));
1436  genericInputs.push_back(arith::ConstantOp::create(
1437  rewriter, loc, DenseIntElementsAttr::get(shiftType, shiftValues)));
1438  indexingMaps.push_back(AffineMap::get(/*dimCount=*/rank,
1439  /*symbolCount=*/0, shiftExprs,
1440  rewriter.getContext()));
1441  shiftArg = indexingMaps.size() - 1;
1442  }
1443 
1444  // Indexing maps for output values.
1445  indexingMaps.push_back(rewriter.getMultiDimIdentityMap(rank));
1446 
1447  // Construct the indexing maps needed for linalg.generic ops.
1448  Value emptyTensor = tensor::EmptyOp::create(
1449  rewriter, loc, outputTy.getShape(), outputTy.getElementType(),
1450  ArrayRef<Value>({dynDims}));
1451 
1452  auto linalgOp = linalg::GenericOp::create(
1453  rewriter, loc, outputTy, genericInputs, ValueRange{emptyTensor},
1454  indexingMaps, getNParallelLoopsAttrs(rank),
1455  [&](OpBuilder &nestedBuilder, Location nestedLoc,
1456  ValueRange blockArgs) {
1457  Value value = blockArgs[0];
1458  Type valueTy = value.getType();
1459 
1460  FailureOr<int64_t> maybeIZp = op.getInputZeroPoint();
1461  if (failed(maybeIZp)) {
1462  (void)rewriter.notifyMatchFailure(
1463  op, "input zero point cannot be statically determined");
1464  return;
1465  }
1466 
1467  const int32_t inBitwidth = valueTy.getIntOrFloatBitWidth();
1468  // Extend zeropoint for sub-32bits widths.
1469  const int32_t inAttrBitwidth = inBitwidth > 32 ? inBitwidth : 32;
1470  auto inputZp = arith::ConstantOp::create(
1471  nestedBuilder, loc,
1472  IntegerAttr::get(rewriter.getIntegerType(inAttrBitwidth),
1473  *maybeIZp));
1474 
1475  FailureOr<int64_t> maybeOZp = op.getOutputZeroPoint();
1476  if (failed(maybeOZp)) {
1477  (void)rewriter.notifyMatchFailure(
1478  op, "output zero point cannot be statically determined");
1479  return;
1480  };
1481 
1482  IntegerType outIntType =
1483  cast<IntegerType>(blockArgs.back().getType());
1484  unsigned outBitWidth = outIntType.getWidth();
1485  const int32_t outAttrBitwidth = 32;
1486  assert(outBitWidth <= 32 && "Unexpected output zeropoint bitwidth");
1487  auto outputZp = arith::ConstantOp::create(
1488  nestedBuilder, loc,
1489  IntegerAttr::get(rewriter.getIntegerType(outAttrBitwidth),
1490  *maybeOZp));
1491 
1492  Value multiplier = multiplierConstant ? multiplierConstant
1493  : blockArgs[multiplierArg];
1494  Value shift = shiftConstant ? shiftConstant : blockArgs[shiftArg];
1495 
1496  if (valueTy.isUnsignedInteger()) {
1497  value = UnrealizedConversionCastOp::create(
1498  nestedBuilder, nestedLoc,
1499  nestedBuilder.getIntegerType(
1500  valueTy.getIntOrFloatBitWidth()),
1501  value)
1502  .getResult(0);
1503  }
1504  if (valueTy.getIntOrFloatBitWidth() < 32) {
1505  if (op.getInputUnsigned()) {
1506  value = arith::ExtUIOp::create(nestedBuilder, nestedLoc,
1507  nestedBuilder.getI32Type(), value);
1508  } else {
1509  value = arith::ExtSIOp::create(nestedBuilder, nestedLoc,
1510  nestedBuilder.getI32Type(), value);
1511  }
1512  }
1513 
1514  value =
1515  arith::SubIOp::create(nestedBuilder, nestedLoc, value, inputZp);
1516 
1517  value = tosa::ApplyScaleOp::create(nestedBuilder, loc,
1518  nestedBuilder.getI32Type(), value,
1519  multiplier, shift, roundingMode);
1520 
1521  // Move to the new zero-point.
1522  value =
1523  arith::AddIOp::create(nestedBuilder, nestedLoc, value, outputZp);
1524 
1525  // Saturate to the output size.
1526  int32_t intMin = APInt::getSignedMinValue(outBitWidth).getSExtValue();
1527  int32_t intMax = APInt::getSignedMaxValue(outBitWidth).getSExtValue();
1528 
1529  // Unsigned integers have a difference output value.
1530  if (op.getOutputUnsigned()) {
1531  intMin = 0;
1532  intMax = APInt::getMaxValue(outBitWidth).getZExtValue();
1533  }
1534 
1535  auto intMinVal = arith::ConstantOp::create(
1536  nestedBuilder, loc, nestedBuilder.getI32IntegerAttr(intMin));
1537  auto intMaxVal = arith::ConstantOp::create(
1538  nestedBuilder, loc, nestedBuilder.getI32IntegerAttr(intMax));
1539 
1540  value = clampIntHelper(nestedLoc, value, intMinVal, intMaxVal,
1541  nestedBuilder, /*isUnsigned=*/false);
1542 
1543  if (outIntType.getWidth() < 32) {
1544  value = arith::TruncIOp::create(
1545  nestedBuilder, nestedLoc,
1546  rewriter.getIntegerType(outIntType.getWidth()), value);
1547  }
1548 
1549  if (outIntType.isUnsignedInteger()) {
1550  value = UnrealizedConversionCastOp::create(nestedBuilder, nestedLoc,
1551  outIntType, value)
1552  .getResult(0);
1553  }
1554  linalg::YieldOp::create(nestedBuilder, loc, value);
1555  });
1556 
1557  rewriter.replaceOp(op, linalgOp->getResults());
1558  return success();
1559  }
