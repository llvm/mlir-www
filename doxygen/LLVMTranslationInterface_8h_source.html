30 /// Base class for dialect interfaces providing translation to LLVM IR.
31 /// Dialects that can be translated should provide an implementation of this
32 /// interface for the supported operations. The interface may be implemented in
33 /// a separate library to avoid the "main" dialect library depending on LLVM IR.
34 /// The interface can be attached using the delayed registration mechanism
35 /// available in DialectRegistry.
36 class LLVMTranslationDialectInterface
37  : public DialectInterface::Base<LLVMTranslationDialectInterface> {
38 public:
39  LLVMTranslationDialectInterface(Dialect *dialect) : Base(dialect) {}
40 
41  /// Hook for derived dialect interface to provide translation of the
42  /// operations to LLVM IR.
43  virtual LogicalResult
44  convertOperation(Operation *op, llvm::IRBuilderBase &builder,
45  LLVM::ModuleTranslation &moduleTranslation) const {
46  return failure();
47  }
48 
49  /// Hook for derived dialect interface to act on an operation that has dialect
50  /// attributes from the derived dialect (the operation itself may be from a
51  /// different dialect). This gets called after the operation has been
52  /// translated. The hook is expected to use moduleTranslation to look up the
53  /// translation results and amend the corresponding IR constructs. Does
54  /// nothing and succeeds by default.
55  virtual LogicalResult
56  amendOperation(Operation *op, NamedAttribute attribute,
57  LLVM::ModuleTranslation &moduleTranslation) const {
58  return success();
59  }
60 };

62 /// Interface collection for translation to LLVM IR, dispatches to a concrete
63 /// interface implementation based on the dialect to which the given op belongs.
64 class LLVMTranslationInterface
65  : public DialectInterfaceCollection<LLVMTranslationDialectInterface> {
66 public:
67  using Base::Base;
68 
69  /// Translates the given operation to LLVM IR using the interface implemented
70  /// by the op's dialect.
71  virtual LogicalResult
72  convertOperation(Operation *op, llvm::IRBuilderBase &builder,
73  LLVM::ModuleTranslation &moduleTranslation) const {
74  if (const LLVMTranslationDialectInterface *iface = getInterfaceFor(op))
75  return iface->convertOperation(op, builder, moduleTranslation);
76  return failure();
77  }
78 
79  /// Acts on the given operation using the interface implemented by the dialect
80  /// of one of the operation's dialect attributes.
81  virtual LogicalResult
82  amendOperation(Operation *op, NamedAttribute attribute,
83  LLVM::ModuleTranslation &moduleTranslation) const {
84  if (const LLVMTranslationDialectInterface *iface =
85  getInterfaceFor(attribute.first.getDialect())) {
86  return iface->amendOperation(op, attribute, moduleTranslation);
87  }
88  return success();
89  }
90 };
