36 FailureOr<VectorType>
37 mlir::xegpu::getDistributedVectorType(xegpu::TensorDescType tdescTy) {
38  auto layout = llvm::dyn_cast_if_present<LayoutAttr>(tdescTy.getLayout());
39  // It only works for subgroup level layout, which only has lane_layout
40  // and lane_data, and is to distribute a SIMD code into SIMT code.
41  if (!layout || !layout.isSgLayout())
42  return failure();
43 
44  SmallVector<int64_t> laneData(layout.getLaneData().asArrayRef());
45  SmallVector<int64_t> laneLayout(layout.getLaneLayout().asArrayRef());
46  auto tdescShape = tdescTy.getShape();
47  auto elementType = tdescTy.getElementType();
48 
49  // compute sgSize by multiply elements of laneLayout
50  // e.g. for 2D layout, sgSize = laneLayout[0] * laneLayout[1]
51  // e.g. for 1D layout, sgSize = laneLayout[0]
52  auto sgSize = std::accumulate(laneLayout.begin(), laneLayout.end(), 1,
53  std::multiplies<int64_t>());
54 
55  // Case 1: regular loads/stores
56  auto scatterAttr = tdescTy.getEncodingOfType<ScatterTensorDescAttr>();
57  if (scatterAttr) {
58  auto chunkSize = scatterAttr.getChunkSize().getInt();
59  // Verify if the first dimension of the tensor descriptor shape is
60  // distributable.
61  assert(tdescShape[0] == laneLayout[0] &&
62  "tensor descriptor shape is not distributable");
63  return VectorType::get({chunkSize}, elementType);
64  }
65 
66  // Case 2: block loads/stores
67  // Check if the tensor descriptor shape is distributable.
68  int64_t tensorSize = 1;
69  for (auto [tdescDim, laneDim, laneDataDim] :
70  llvm::zip_equal(tdescShape, laneLayout, laneData)) {
71  assert((tdescDim % (laneDim * laneDataDim) == 0) &&
72  "tensor descriptor shape is not distributable");
73  tensorSize *= tdescDim;
74  }
75  // tensorSize must be adjusted for array_length.
76  tensorSize *= tdescTy.getArrayLength();
77 
78  return VectorType::get({tensorSize / sgSize}, elementType);
79 }

81 FailureOr<VectorType>
82 mlir::xegpu::getDistributedVectorType(VectorType originalType,
83  xegpu::LayoutAttr layout) {
84  int64_t rank = originalType.getRank();
85  // Distributed vector type is only supported for 1D, 2D and 3D vectors.
86  if (rank < 1 || rank > 3)
87  return failure();
88  ArrayRef<int64_t> shape = originalType.getShape();
89  // arrayLength is 1 for 1D and 2D vectors, and equal to the first dimension
90  // of the 3D vector.
91  int arrayLength = 1;
92  if (rank == 3) {
93  arrayLength = shape[0];
94  shape = shape.drop_front();
95  }
96  auto helperTdescTy = xegpu::TensorDescType::get(
97  shape, originalType.getElementType(), arrayLength,
98  /*boundary_check=*/true,
99  /*memory_space=*/xegpu::MemorySpace::Global, layout);
100  return xegpu::getDistributedVectorType(helperTdescTy);
101 }

114 xegpu::LayoutAttr xegpu::getLayoutAttr(const Value value) {
115  if (!value)
116  return nullptr;
117 
118  if (auto tdescTy =
119  dyn_cast_if_present<xegpu::TensorDescType>(value.getType()))
120  return tdescTy.getLayoutAttr();
121 
122  if (auto result = dyn_cast<OpResult>(value)) {
123  Operation *defOp = result.getDefiningOp();
124  assert(defOp && "result must have a defining op");
125 
126  // For ConvertLayoutOp, the layout is stored in the targetLayoutAttr
127  if (auto convertOp = dyn_cast<xegpu::ConvertLayoutOp>(defOp))
128  return convertOp.getTargetLayoutAttr();
129 
130  // for LoadNdOp, the layout is stored in the tensor descriptor
131  if (auto loadNd = dyn_cast<xegpu::LoadNdOp>(defOp))
132  return getLayoutAttr(loadNd.getTensorDesc());
133 
134  std::string layoutName = getLayoutName(result);
135  if (defOp->hasAttr(layoutName))
136  return defOp->getAttrOfType<xegpu::LayoutAttr>(layoutName);
137  }
138 
139  if (auto arg = dyn_cast<BlockArgument>(value)) {
140  auto parentOp = arg.getOwner()->getParentOp();
141  if (auto loop = dyn_cast<LoopLikeOpInterface>(parentOp)) {
142  OpOperand *tiedInit = loop.getTiedLoopInit(arg);
143  if (tiedInit)
144  return getLayoutAttr(tiedInit->get());
145  }
146  }
147 
148  return nullptr;
149 }

217 xegpu::extractVectorsWithShapeFromValue(OpBuilder &builder, Location loc,
218  Value value, ArrayRef<int64_t> shape) {
219  auto vecTy = dyn_cast<VectorType>(value.getType());
220  if (!vecTy)
221  return {value};
222 
223  ArrayRef<int64_t> srcShape = vecTy.getShape();
224  if (!computeShapeRatio(srcShape, shape))
225  return {value};
226 
227  SmallVector<Value> result;
228  for (SmallVector<int64_t> offsets : StaticTileOffsetRange(srcShape, shape)) {
229  SmallVector<int64_t> staticStrides(offsets.size(), 1);
230  result.push_back(vector::ExtractStridedSliceOp::create(
231  builder, loc, value, offsets, shape, staticStrides));
232  }
233 
234  return result;
235 }

237 Value xegpu::createVectorWithShapeFromValues(OpBuilder &builder, Location loc,
238  ValueRange values,
239  ArrayRef<int64_t> shape) {
240  VectorType inputTy = dyn_cast<VectorType>(values[0].getType());
241  assert(llvm::all_of(values.getTypes(),
242  [&](Type type) { return type == inputTy; }) &&
243  "values must be of the same VectorType");
244 
245  Type elemTy = inputTy.getElementType();
246  ArrayRef<int64_t> tileShape = inputTy.getShape();
247 
248  VectorType resultTy = VectorType::get(shape, elemTy);
249  auto zeroAttr = builder.getZeroAttr(elemTy);
250  Value result = arith::ConstantOp::create(
251  builder, loc, resultTy, DenseElementsAttr::get(resultTy, zeroAttr));
252 
253  for (auto [src, offsets] :
254  llvm::zip_equal(values, StaticTileOffsetRange(shape, tileShape))) {
255  SmallVector<int64_t> staticStrides(offsets.size(), 1);
256  result = vector::InsertStridedSliceOp::create(builder, loc, src, result,
257  offsets, staticStrides);
258  }
259  return result;
260 }

262 void xegpu::doSCFStructuralTypeConversionWithTensorType(
263  Operation *op, TypeConverter converter) {
264  MLIRContext *context = op->getContext();
265 
266  auto materializeCast = [](OpBuilder &builder, Type type, ValueRange inputs,
267  Location loc) -> Value {
268  return UnrealizedConversionCastOp::create(builder, loc, type, inputs)
269  .getResult(0);
270  };
271 
272  { // convert VectorType to RankedTensorType for SCF Structural ops
273  TypeConverter converter;
274  converter.addConversion([](Type type) -> Type { return type; });
275  converter.addConversion([](VectorType type) -> Type {
276  return RankedTensorType::get(type.getShape(), type.getElementType());
277  });
278  converter.addSourceMaterialization(materializeCast);
279  converter.addTargetMaterialization(materializeCast);
280 
281  mlir::ConversionTarget target(*context);
282  target.addLegalOp<UnrealizedConversionCastOp>();
283 
284  mlir::RewritePatternSet patterns(context);
285  scf::populateSCFStructuralTypeConversionsAndLegality(converter, patterns,
286  target);
287  (void)mlir::applyPartialConversion(op, target, std::move(patterns));
288  }
289 
290  { // propagate the layout attribute to RankedTensorType by checking
291  // BuiltInUnrealizedCastOps
292  // for VectorType to RankedTensorType cast.
293  op->walk([](UnrealizedConversionCastOp castOp) {
294  if (castOp.getNumOperands() != 1 || castOp.getNumResults() != 1)
295  return WalkResult::skip();
296 
297  Value input = castOp.getInputs()[0];
298  Value result = castOp.getResults()[0];
299  auto inputTy = dyn_cast<VectorType>(input.getType());
300  auto resultTy = dyn_cast<RankedTensorType>(result.getType());
301 
302  // Only look at ops casting from VectorType to RankedTensorType
303  if (!inputTy || !resultTy)
304  return WalkResult::skip();
305 
306  xegpu::LayoutAttr layout = xegpu::getLayoutAttr(input);
307  if (!layout)
308  return WalkResult::skip();
309 
310  RankedTensorType newTy = resultTy.cloneWithEncoding(layout);
311  result.setType(newTy);
312 
313  // update the arguments if user is a LoopLike op.
314  for (OpOperand &use : result.getUses()) {
315  if (auto loop = dyn_cast<LoopLikeOpInterface>(use.getOwner())) {
316  BlockArgument arg = loop.getTiedLoopRegionIterArg(&use);
317  arg.setType(newTy);
318  }
319  // whileOp has two regions, the BlockArgument of the after region
320  // is not exposed by LoopLikeOpInterface
321  if (auto whileOp = dyn_cast<scf::WhileOp>(use.getOwner())) {
322  unsigned idx = use.getOperandNumber();
323  BlockArgument arg = whileOp.getAfterArguments()[idx];
324  arg.setType(newTy);
325  }
326  }
327  return WalkResult::advance();
328  });
329 
330  // using yieldOp as anchor to update the result type of its ParentOp
331  op->walk([](scf::YieldOp yieldOp) {
332  Operation *parentOp = yieldOp->getParentOp();
333  for (OpResult r : parentOp->getOpResults()) {
334  unsigned idx = r.getResultNumber();
335  Type resultTy = r.getType();
336  Type yieldTy = yieldOp.getResults()[idx].getType();
337  if (isa<RankedTensorType>(resultTy) && yieldTy != resultTy)
338  r.setType(yieldTy);
339  }
340  });
341  }
342 
343  { // perform the conversion from RankedTensorType to VectorType based on the
344  // LayoutAttr
345 
346  // Handle the UnrealizedConversionCastOp introduced by the first step.
347  // For vector->RankedTensorType, it will simply forward the inputs.
348  // For RankedTensorType->vector, it will update the inputs with the
349  // one from the adaptor.
350  class UnrealizedConversionCastOpPattern
351  : public OpConversionPattern<mlir::UnrealizedConversionCastOp> {
352  using OpConversionPattern<
353  mlir::UnrealizedConversionCastOp>::OpConversionPattern;
354 
355  mlir::LogicalResult
356  matchAndRewrite(mlir::UnrealizedConversionCastOp op,
357  OneToNOpAdaptor adaptor,
358  ConversionPatternRewriter &rewriter) const override {
359  auto inputs = op.getOperands();
360  auto outputs = op.getOutputs();
361 
362  if (inputs.size() != 1 || outputs.size() != 1)
363  return failure();
364 
365  auto inputTy = inputs[0].getType();
366  auto outputTy = outputs[0].getType();
367 
368  if (isa<VectorType>(inputTy) && isa<RankedTensorType>(outputTy)) {
369  rewriter.replaceOpWithMultiple(op, adaptor.getInputs());
370  return success();
371  }
372 
373  if (isa<RankedTensorType>(inputTy) && isa<VectorType>(outputTy)) {
374  SmallVector<Value> values = xegpu::flattenValues(adaptor.getInputs());
375  auto newOp = UnrealizedConversionCastOp::create(rewriter, op.getLoc(),
376  outputTy, values);
377  rewriter.replaceOp(op, newOp);
378  return success();
379  }
380  return failure();
381  }
382  };
383 
384  converter.addSourceMaterialization(materializeCast);
385  converter.addTargetMaterialization([&](OpBuilder &builder, TypeRange type,
386  ValueRange inputs, Location loc) {
387  return UnrealizedConversionCastOp::create(builder, loc, type, inputs)
388  .getResults();
389  });
390 
391  mlir::ConversionTarget target(*context);
392  target.addDynamicallyLegalOp<UnrealizedConversionCastOp>(
393  [](UnrealizedConversionCastOp op) {
394  auto isTensorTy = [](Type type) {
395  return isa<RankedTensorType>(type);
396  };
397  return llvm::none_of(op->getOperandTypes(), isTensorTy) &&
398  llvm::none_of(op->getResultTypes(), isTensorTy);
399  });
400  mlir::RewritePatternSet patterns(context);
401  patterns.insert<UnrealizedConversionCastOpPattern>(context);
402  scf::populateSCFStructuralTypeConversionsAndLegality(converter, patterns,
403  target);
404  (void)mlir::applyPartialConversion(op, target, std::move(patterns));
405  }
406 }
