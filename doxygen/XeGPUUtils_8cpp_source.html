20 FailureOr<VectorType>
21 mlir::xegpu::getDistributedVectorType(xegpu::TensorDescType tdescTy) {
22  auto layout = llvm::dyn_cast_if_present<LayoutAttr>(tdescTy.getLayout());
23  // It only works for subgroup level layout, which only has lane_layout
24  // and lane_data, and is to distribute a SIMD code into SIMT code.
25  if (!layout || !layout.isSgLayout())
26  return failure();
27 
28  SmallVector<int64_t> laneData(layout.getLaneData().asArrayRef());
29  SmallVector<int64_t> laneLayout(layout.getLaneLayout().asArrayRef());
30  auto tdescShape = tdescTy.getShape();
31  auto elementType = tdescTy.getElementType();
32 
33  // compute sgSize by multiply elements of laneLayout
34  // e.g. for 2D layout, sgSize = laneLayout[0] * laneLayout[1]
35  // e.g. for 1D layout, sgSize = laneLayout[0]
36  auto sgSize = std::accumulate(laneLayout.begin(), laneLayout.end(), 1,
37  std::multiplies<int64_t>());
38 
39  // Case 1: regular loads/stores
40  auto scatterAttr = tdescTy.getEncodingAsScatterTensorDescAttr();
41  if (scatterAttr) {
42  auto chunkSize = scatterAttr.getChunkSize().getInt();
43  // Verify if the first dimension of the tensor descriptor shape is
44  // distributable.
45  assert(tdescShape[0] == laneLayout[0] &&
46  "tensor descriptor shape is not distributable");
47  return VectorType::get({chunkSize}, elementType);
48  }
49 
50  // Case 2: block loads/stores
51  // Check if the tensor descriptor shape is distributable.
52  int64_t tensorSize = 1;
53  for (auto [tdescDim, laneDim, laneDataDim] :
54  llvm::zip_equal(tdescShape, laneLayout, laneData)) {
55  assert((tdescDim % (laneDim * laneDataDim) == 0) &&
56  "tensor descriptor shape is not distributable");
57  tensorSize *= tdescDim;
58  }
59  // tensorSize must be adjusted for array_length.
60  tensorSize *= tdescTy.getArrayLength();
61 
62  return VectorType::get({tensorSize / sgSize}, elementType);
63 }

65 FailureOr<VectorType>
66 mlir::xegpu::getDistributedVectorType(VectorType originalType,
67  xegpu::LayoutAttr layout) {
68  int64_t rank = originalType.getRank();
69  // Distributed vector type is only supported for 1D, 2D and 3D vectors.
70  if (rank < 1 || rank > 3)
71  return failure();
72  ArrayRef<int64_t> shape = originalType.getShape();
73  // arrayLength is 1 for 1D and 2D vectors, and equal to the first dimension
74  // of the 3D vector.
75  int arrayLength = 1;
76  if (rank == 3) {
77  arrayLength = shape[0];
78  shape = shape.drop_front();
79  }
80  auto helperTdescTy = xegpu::TensorDescType::get(
81  shape, originalType.getElementType(), arrayLength,
82  /*boundary_check=*/true,
83  /*memory_space=*/xegpu::MemorySpace::Global, layout);
84  return xegpu::getDistributedVectorType(helperTdescTy);
85 }
