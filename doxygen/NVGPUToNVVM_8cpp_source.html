57 /// Returns the type for the intrinsic given the vectorResultType of the
58 /// `gpu.mma.sync` operation.
59 static Type inferIntrinsicResultType(Type vectorResultType) {
60  MLIRContext *ctx = vectorResultType.getContext();
61  auto a = cast<LLVM::LLVMArrayType>(vectorResultType);
62  auto f16x2Ty = VectorType::get(2, Float16Type::get(ctx));
63  auto i32Ty = IntegerType::get(ctx, 32);
64  auto i32x2Ty = VectorType::get(2, i32Ty);
65  Type f64Ty = Float64Type::get(ctx);
66  Type f64x2Ty = VectorType::get(2, f64Ty);
67  Type f32Ty = Float32Type::get(ctx);
68  Type f32x2Ty = VectorType::get(2, f32Ty);
69  if (a.getElementType() == f16x2Ty) {
70  return LLVM::LLVMStructType::getLiteral(
71  ctx, SmallVector<Type>(a.getNumElements(), f16x2Ty));
72  }
73  if (a.getElementType() == i32x2Ty) {
74  return LLVM::LLVMStructType::getLiteral(
75  ctx,
76  SmallVector<Type>(static_cast<size_t>(a.getNumElements()) * 2, i32Ty));
77  }
78  if (a.getElementType() == f64x2Ty) {
79  return LLVM::LLVMStructType::getLiteral(ctx, {f64Ty, f64Ty});
80  }
81  if (a.getElementType() == f32x2Ty) {
82  return LLVM::LLVMStructType::getLiteral(
83  ctx,
84  SmallVector<Type>(static_cast<size_t>(a.getNumElements()) * 2, f32Ty));
85  }
86  if (a.getElementType() == VectorType::get(1, f32Ty)) {
87  return LLVM::LLVMStructType::getLiteral(
88  ctx, SmallVector<Type>(static_cast<size_t>(a.getNumElements()), f32Ty));
89  }
90  return vectorResultType;
91 }

93 /// Convert the SSA result of the NVVM intrinsic `nvvm.mma.sync` (which is
94 /// always an LLVM struct) into a fragment that is compatible with the vector
95 /// type of this operation. This involves extracting elements from the struct
96 /// and inserting them into an LLVM array. These extra data-movement
97 /// operations should be canonicalized away by the LLVM backend.
98 static Value convertIntrinsicResult(Location loc, Type intrinsicResultType,
99  Type resultType, Value intrinsicResult,
100  RewriterBase &rewriter) {
101  MLIRContext *ctx = rewriter.getContext();
102  auto structType = dyn_cast<LLVM::LLVMStructType>(intrinsicResultType);
103  auto arrayType = dyn_cast<LLVM::LLVMArrayType>(resultType);
104  Type i32Ty = rewriter.getI32Type();
105  Type f32Ty = rewriter.getF32Type();
106  Type f64Ty = rewriter.getF64Type();
107  Type f16x2Ty = VectorType::get(2, rewriter.getF16Type());
108  Type i32x2Ty = VectorType::get(2, i32Ty);
109  Type f64x2Ty = VectorType::get(2, f64Ty);
110  Type f32x2Ty = VectorType::get(2, f32Ty);
111  Type f32x1Ty = VectorType::get(1, f32Ty);
112 
113  auto makeConst = [&](int32_t index) -> Value {
114  return LLVM::ConstantOp::create(rewriter, loc, IntegerType::get(ctx, 32),
115  rewriter.getI32IntegerAttr(index));
116  };
117 
118  if (arrayType) {
119  SmallVector<Value, 4> elements;
120 
121  // The intrinsic returns 32-bit wide elements in a form which can be
122  // directly bitcasted and inserted into the result vector.
123  if (arrayType.getElementType() == f16x2Ty ||
124  arrayType.getElementType() == f32x1Ty) {
125  for (unsigned i = 0; i < structType.getBody().size(); i++) {
126  Value el =
127  LLVM::ExtractValueOp::create(rewriter, loc, intrinsicResult, i);
128  el = rewriter.createOrFold<LLVM::BitcastOp>(
129  loc, arrayType.getElementType(), el);
130  elements.push_back(el);
131  }
132  }
133 
134  // The intrinsic returns i32, f64, and f32 values as individual scalars,
135  // even when the result is notionally a 64-bit wide element (e.g. f32x2). We
136  // need to extract them from the struct and pack them into the 64-bit wide
137  // rows of the vector result.
138  if (arrayType.getElementType() == i32x2Ty ||
139  arrayType.getElementType() == f64x2Ty ||
140  arrayType.getElementType() == f32x2Ty) {
141 
142  for (unsigned i = 0, e = structType.getBody().size() / 2; i < e; i++) {
143  Value vec =
144  LLVM::PoisonOp::create(rewriter, loc, arrayType.getElementType());
145  Value x1 =
146  LLVM::ExtractValueOp::create(rewriter, loc, intrinsicResult, i * 2);
147  Value x2 = LLVM::ExtractValueOp::create(rewriter, loc, intrinsicResult,
148  i * 2 + 1);
149  vec = LLVM::InsertElementOp::create(rewriter, loc, vec.getType(), vec,
150  x1, makeConst(0));
151  vec = LLVM::InsertElementOp::create(rewriter, loc, vec.getType(), vec,
152  x2, makeConst(1));
153  elements.push_back(vec);
154  }
155  }
156 
157  // Create the final vectorized result.
158  Value result = LLVM::PoisonOp::create(rewriter, loc, arrayType);
159  for (const auto &el : llvm::enumerate(elements)) {
160  result = LLVM::InsertValueOp::create(rewriter, loc, result, el.value(),
161  el.index());
162  }
163  return result;
164  }
165 
166  return intrinsicResult;
167 }

169 /// The `gpu.mma.sync` converter below expects matrix fragment operands to be
170 /// given as 2D `vectors` where the rows are 32b or 64b wide. The
171 /// `nvvm.mma.sync` op expects these argments to be a given in a long list of
172 /// scalars of certain types. This function helps unpack the `vector` arguments
173 /// and cast them to the types expected by `nvvm.mma.sync`.
174 static SmallVector<Value> unpackOperandVector(ImplicitLocOpBuilder &b,
175  Value operand,
176  NVVM::MMATypes operandPtxType) {
177  SmallVector<Value> result;
178  Type i32Ty = b.getI32Type();
179  Type f64Ty = b.getF64Type();
180  Type f32Ty = b.getF32Type();
181  Type i64Ty = b.getI64Type();
182  Type i8x4Ty = VectorType::get(4, b.getI8Type());
183  Type i4x8Ty = VectorType::get(8, b.getIntegerType(4));
184  Type f32x1Ty = VectorType::get(1, f32Ty);
185  auto arrayTy = cast<LLVM::LLVMArrayType>(operand.getType());
186 
187  for (unsigned i = 0, e = arrayTy.getNumElements(); i < e; ++i) {
188  Value toUse = LLVM::ExtractValueOp::create(b, operand, i);
189 
190  // For 4xi8 vectors, the intrinsic expects these to be provided as i32
191  // scalar types.
192  if (arrayTy.getElementType() == i8x4Ty ||
193  arrayTy.getElementType() == i4x8Ty ||
194  (arrayTy.getElementType() == f32x1Ty &&
195  operandPtxType == NVVM::MMATypes::tf32)) {
196  result.push_back(LLVM::BitcastOp::create(b, i32Ty, toUse));
197  continue;
198  }
199 
200  // For some element types (i32, f32, f64), we need to unpack the inner
201  // vector/array type as well because the intrinsic expects individual
202  // scalars to be provided.
203  VectorType innerArrayTy = dyn_cast<VectorType>(arrayTy.getElementType());
204  if (innerArrayTy && (innerArrayTy.getElementType() == i32Ty ||
205  innerArrayTy.getElementType() == f64Ty ||
206  innerArrayTy.getElementType() == f32Ty)) {
207  for (unsigned idx = 0, innerSize = innerArrayTy.getNumElements();
208  idx < innerSize; idx++) {
209  result.push_back(LLVM::ExtractElementOp::create(
210  b, toUse,
211  LLVM::ConstantOp::create(b, i64Ty, b.getI64IntegerAttr(idx))));
212  }
213  continue;
214  }
215  result.push_back(toUse);
216  }
217  return result;
218 }

250 struct MmaLdMatrixOpToNVVM : public ConvertOpToLLVMPattern<nvgpu::LdMatrixOp> {
251  using ConvertOpToLLVMPattern<nvgpu::LdMatrixOp>::ConvertOpToLLVMPattern;
252 
253  LogicalResult
254  matchAndRewrite(nvgpu::LdMatrixOp op, OpAdaptor adaptor,
255  ConversionPatternRewriter &rewriter) const override {
256  MLIRContext *ctx = getContext();
257  ImplicitLocOpBuilder b(op.getLoc(), rewriter);
258 
259  // The result type of ldmatrix will always be a struct of 32bit integer
260  // registers if more than one 32bit value is returned. Otherwise, the result
261  // is a single i32. The result type of the GPU operation is always a vector
262  // of shape (NumRegisters, VectorRegister) where VectorRegister is the
263  // vector type of the result and always 32 bits long. We bitcast the result
264  // of the NVVM::LdMatrix to this vector type.
265  auto vectorResultType = dyn_cast<VectorType>(op->getResultTypes()[0]);
266  if (!vectorResultType) {
267  return failure();
268  }
269  Type innerVectorType = VectorType::get(vectorResultType.getDimSize(1),
270  vectorResultType.getElementType());
271 
272  int64_t num32BitRegs = vectorResultType.getDimSize(0);
273 
274  Type ldMatrixResultType;
275  if (num32BitRegs > 1) {
276  ldMatrixResultType = LLVM::LLVMStructType::getLiteral(
277  ctx, SmallVector<Type>(num32BitRegs, rewriter.getI32Type()));
278  } else {
279  ldMatrixResultType = rewriter.getI32Type();
280  }
281 
282  auto srcMemrefType = cast<MemRefType>(op.getSrcMemref().getType());
283  Value srcPtr =
284  getStridedElementPtr(rewriter, b.getLoc(), srcMemrefType,
285  adaptor.getSrcMemref(), adaptor.getIndices());
286  Value ldMatrixResult = NVVM::LdMatrixOp::create(
287  b, ldMatrixResultType, srcPtr,
288  /*num=*/op.getNumTiles(),
289  /*layout=*/op.getTranspose() ? NVVM::MMALayout::col
290  : NVVM::MMALayout::row);
291 
292  // The ldmatrix operation returns either a single i32 value or a struct of
293  // i32 values. Here we unpack those values and cast them back to their
294  // actual vector type (still of width 32b) and repack them into a result
295  // struct.
296  Type finalResultType = typeConverter->convertType(vectorResultType);
297  Value result = LLVM::PoisonOp::create(b, finalResultType);
298  for (int64_t i = 0, e = vectorResultType.getDimSize(0); i < e; i++) {
299  Value i32Register =
300  num32BitRegs > 1 ? LLVM::ExtractValueOp::create(b, ldMatrixResult, i)
301  : ldMatrixResult;
302  Value casted = LLVM::BitcastOp::create(b, innerVectorType, i32Register);
303  result = LLVM::InsertValueOp::create(b, result, casted, i);
304  }
305 
306  rewriter.replaceOp(op, result);
307  return success();
308  }

311 /// Convert the given type into the corresponding PTX type (NVVM::MMATypes
312 /// enum).
313 static FailureOr<NVVM::MMATypes> getNvvmMmaType(Type t) {
314  Type elType = getElementTypeOrSelf(t);
315  if (elType.isInteger(8))
316  return NVVM::MMATypes::s8;
317  if (elType.isInteger(4))
318  return NVVM::MMATypes::s4;
319  if (elType.isF16())
320  return NVVM::MMATypes::f16;
321  if (elType.isF64())
322  return NVVM::MMATypes::f64;
323  if (elType.isF32())
324  return NVVM::MMATypes::tf32;
325  return failure();
326 }

328 struct MmaSyncOptoNVVM : public ConvertOpToLLVMPattern<nvgpu::MmaSyncOp> {
329  using ConvertOpToLLVMPattern<nvgpu::MmaSyncOp>::ConvertOpToLLVMPattern;
330 
331  LogicalResult
332  matchAndRewrite(nvgpu::MmaSyncOp op, OpAdaptor adaptor,
333  ConversionPatternRewriter &rewriter) const override {
334  ImplicitLocOpBuilder b(op.getLoc(), rewriter);
335  // Get the shapes of the MMAMatrix type being used. The shapes will
336  // choose which intrinsic this op will be lowered to.
337  VectorType aType = op.getMatrixA().getType();
338  VectorType bType = op.getMatrixA().getType();
339  VectorType cType = op.getMatrixC().getType();
340 
341  std::array<int64_t, 3> gemmShape = op.getMmaShapeAsArray();
342 
343  // Tensor Cores (mma.sync) on F32 works only with TensorFloat32 (TF32).
344  bool tf32Enabled = op->hasAttr(op.getTf32EnabledAttrName());
345  if (aType.getElementType().isF32() && !tf32Enabled)
346  return failure();
347 
348  FailureOr<NVVM::MMATypes> ptxTypeA = getNvvmMmaType(aType);
349  if (failed(ptxTypeA))
350  return op->emitOpError("failed to deduce operand PTX types");
351  FailureOr<NVVM::MMATypes> ptxTypeB = getNvvmMmaType(bType);
352  if (failed(ptxTypeB))
353  return op->emitOpError("failed to deduce operand PTX types");
354  std::optional<NVVM::MMATypes> ptxTypeC =
355  NVVM::MmaOp::inferOperandMMAType(cType.getElementType(),
356  /*isAccumulator=*/true);
357  if (!ptxTypeC)
358  return op->emitError(
359  "could not infer the PTX type for the accumulator/result");
360 
361  // TODO: add an attribute to the op to customize this behavior.
362  std::optional<NVVM::MMAIntOverflow> overflow(std::nullopt);
363  if (isa<IntegerType>(aType.getElementType()))
364  overflow = NVVM::MMAIntOverflow::satfinite;
365 
366  SmallVector<Value> matA =
367  unpackOperandVector(b, adaptor.getMatrixA(), *ptxTypeA);
368  SmallVector<Value> matB =
369  unpackOperandVector(b, adaptor.getMatrixB(), *ptxTypeB);
370  SmallVector<Value> matC =
371  unpackOperandVector(b, adaptor.getMatrixC(), *ptxTypeC);
372 
373  Type desiredRetTy = typeConverter->convertType(op->getResultTypes()[0]);
374  Type intrinsicResTy = inferIntrinsicResultType(
375  typeConverter->convertType(op->getResultTypes()[0]));
376  Value intrinsicResult =
377  NVVM::MmaOp::create(b, intrinsicResTy, matA, matB, matC,
378  /*shape=*/gemmShape,
379  /*b1Op=*/std::nullopt,
380  /*intOverflow=*/overflow,
381  /*multiplicandPtxTypes=*/
382  std::array<NVVM::MMATypes, 2>{*ptxTypeA, *ptxTypeB},
383  /*multiplicandLayouts=*/
384  std::array<NVVM::MMALayout, 2>{
385  NVVM::MMALayout::row, NVVM::MMALayout::col});
386  rewriter.replaceOp(op, convertIntrinsicResult(op.getLoc(), intrinsicResTy,
387  desiredRetTy, intrinsicResult,
388  rewriter));
389  return success();
390  }

402  void runOnOperation() override {
403  LowerToLLVMOptions options(&getContext());
404  RewritePatternSet patterns(&getContext());
405  LLVMTypeConverter converter(&getContext(), options);
406  IRRewriter rewriter(&getContext());
407  populateGpuMemorySpaceAttributeConversions(
408  converter, [](gpu::AddressSpace space) -> unsigned {
409  switch (space) {
410  case gpu::AddressSpace::Global:
411  return static_cast<unsigned>(
412  NVVM::NVVMMemorySpace::kGlobalMemorySpace);
413  case gpu::AddressSpace::Workgroup:
414  return static_cast<unsigned>(
415  NVVM::NVVMMemorySpace::kSharedMemorySpace);
416  case gpu::AddressSpace::Private:
417  return 0;
418  }
419  llvm_unreachable("unknown address space enum value");
420  return 0;
421  });
422  /// device-side async tokens cannot be materialized in nvvm. We just
423  /// convert them to a dummy i32 type in order to easily drop them during
424  /// conversion.
425  converter.addConversion([&](nvgpu::DeviceAsyncTokenType type) -> Type {
426  return converter.convertType(IntegerType::get(type.getContext(), 32));
427  });
428  converter.addConversion([&](nvgpu::WarpgroupAccumulatorType type) -> Type {
429  Type elemType = type.getFragmented().getElementType();
430  int64_t sizeM = type.getFragmented().getDimSize(0);
431  int64_t sizeN = type.getFragmented().getDimSize(1);
432 
433  unsigned numMembers;
434  if (elemType.isF32() || elemType.isInteger(32))
435  numMembers = sizeN / 2;
436  else if (elemType.isF16())
437  numMembers = sizeN / 4;
438  else
439  llvm_unreachable("unsupported type for warpgroup accumulator");
440 
441  SmallVector<Type> innerStructBody;
442  for (unsigned i = 0; i < numMembers; i++)
443  innerStructBody.push_back(elemType);
444  auto innerStructType =
445  LLVM::LLVMStructType::getLiteral(type.getContext(), innerStructBody);
446 
447  SmallVector<Type> structBody;
448  for (int i = 0; i < sizeM; i += kWgmmaSizeM)
449  structBody.push_back(innerStructType);
450 
451  auto convertedType =
452  LLVM::LLVMStructType::getLiteral(type.getContext(), structBody);
453  return converter.convertType(convertedType);
454  });
455  converter.addConversion([&](nvgpu::MBarrierTokenType type) -> Type {
456  return converter.convertType(IntegerType::get(type.getContext(), 64));
457  });
458  converter.addConversion(
459  [&](nvgpu::WarpgroupMatrixDescriptorType type) -> Type {
460  return converter.convertType(IntegerType::get(type.getContext(), 64));
461  });
462  converter.addConversion([&](nvgpu::MBarrierGroupType type) -> Type {
463  return converter.convertType(
464  nvgpu::getMBarrierMemrefType(rewriter.getContext(), type));
465  });
466  converter.addConversion([&](nvgpu::TensorMapDescriptorType type) -> Type {
467  return LLVM::LLVMPointerType::get(type.getContext());
468  });
469  populateNVGPUToNVVMConversionPatterns(converter, patterns);
470  LLVMConversionTarget target(getContext());
471  target.addLegalDialect<::mlir::LLVM::LLVMDialect>();
472  target.addLegalDialect<::mlir::arith::ArithDialect>();
473  target.addLegalDialect<::mlir::memref::MemRefDialect>();
474  target.addLegalDialect<::mlir::NVVM::NVVMDialect>();
475  mlir::scf::populateSCFStructuralTypeConversionsAndLegality(
476  converter, patterns, target);
477  if (failed(applyPartialConversion(getOperation(), target,
478  std::move(patterns))))
479  signalPassFailure();
480  }

503 static std::string buildMmaSparseAsmString(

539 /// sparse sync operation.
540 static FailureOr<LLVM::InlineAsmOp> emitMmaSparseSyncOpAsm(
541  ImplicitLocOpBuilder &b, NVVM::MMATypes ptxTypeA, NVVM::MMATypes ptxTypeB,
542  NVVM::MMATypes ptxTypeC, NVVM::MMATypes ptxTypeD,
543  std::optional<NVVM::MMAIntOverflow> overflow, ArrayRef<Value> unpackedAData,
544  ArrayRef<Value> unpackedB, ArrayRef<Value> unpackedC, Value indexData,
545  int64_t metadataSelector, const std::array<int64_t, 3> &shape,
546  Type intrinsicResultType) {

548  LLVM::AsmDialectAttr::get(b.getContext(), LLVM::AsmDialect::AD_ATT);
549 
550  const unsigned matASize = unpackedAData.size();
551  const unsigned matBSize = unpackedB.size();
552  const unsigned matCSize = unpackedC.size();

554  std::string asmStr = buildMmaSparseAsmString(
555  shape, matASize, matBSize, matCSize, ptxTypeA, ptxTypeB, ptxTypeC,
556  ptxTypeD, overflow, metadataSelector);
557  std::string constraintStr =
558  buildMmaSparseAsmConstraintString(matASize, matBSize, matCSize);
559 
560  SmallVector<Value> asmVals;
561  asmVals.reserve(matASize + matBSize + matCSize + 1);
562  for (ArrayRef<Value> args : {unpackedAData, unpackedB, unpackedC})
563  llvm::append_range(asmVals, args);
564  asmVals.push_back(indexData);
565 
566  return LLVM::InlineAsmOp::create(b,
567  /*resultTypes=*/intrinsicResultType,
568  /*operands=*/asmVals,
569  /*asm_string=*/asmStr,
570  /*constraints=*/constraintStr,
571  /*has_side_effects=*/true,
572  /*is_align_stack=*/false,
573  LLVM::TailCallKind::None,
574  /*asm_dialect=*/asmDialectAttr,
575  /*operand_attrs=*/ArrayAttr());
576 }

579 struct NVGPUMmaSparseSyncLowering
580  : public ConvertOpToLLVMPattern<nvgpu::MmaSparseSyncOp> {
581  using ConvertOpToLLVMPattern<nvgpu::MmaSparseSyncOp>::ConvertOpToLLVMPattern;
582 
583  LogicalResult
584  matchAndRewrite(nvgpu::MmaSparseSyncOp op, OpAdaptor adaptor,
585  ConversionPatternRewriter &rewriter) const override {
586  ImplicitLocOpBuilder b(op.getLoc(), rewriter);
587  // Get the shapes of the MMAMatrix type being used. The shapes will
588  // choose which intrinsic this op will be lowered to.
589  VectorType aType = op.getMatrixA().getType();
590  VectorType bType = op.getMatrixB().getType();
591  VectorType cType = op.getMatrixC().getType();
592 
593  FailureOr<NVVM::MMATypes> ptxTypeA = getNvvmMmaType(aType);
594  if (failed(ptxTypeA))
595  return op->emitOpError("failed to deduce operand PTX types");
596  FailureOr<NVVM::MMATypes> ptxTypeB = getNvvmMmaType(bType);
597  if (failed(ptxTypeB))
598  return op->emitOpError("failed to deduce operand PTX types");
599  std::optional<NVVM::MMATypes> ptxTypeC =
600  NVVM::MmaOp::inferOperandMMAType(cType.getElementType(),
601  /*isAccumulator=*/true);
602  if (!ptxTypeC)
603  return op->emitError(
604  "could not infer the PTX type for the accumulator/result");
605 
606  // Same as `mma.sync`, F32 works only with TensorFloat32 (TF32).
607  bool tf32Enabled = op->hasAttr(op.getTf32EnabledAttrName());
608  if (aType.getElementType().isF32() && !tf32Enabled)
609  return failure();
610 
611  // TODO: add an attribute to the op to customize this behavior.
612  std::optional<NVVM::MMAIntOverflow> overflow(std::nullopt);
613  if (isa<IntegerType>(aType.getElementType()))
614  overflow = NVVM::MMAIntOverflow::satfinite;
615 
616  SmallVector<Value> matA =
617  unpackOperandVector(b, adaptor.getMatrixA(), *ptxTypeA);
618  SmallVector<Value> matB =
619  unpackOperandVector(b, adaptor.getMatrixB(), *ptxTypeB);
620  SmallVector<Value> matC =
621  unpackOperandVector(b, adaptor.getMatrixC(), *ptxTypeC);
622 
623  Type desiredRetTy = typeConverter->convertType(op->getResultTypes()[0]);
624  Type intrinsicResTy = inferIntrinsicResultType(
625  typeConverter->convertType(op->getResultTypes()[0]));
626 
627  // Bitcast the sparse metadata from vector<2xf16> to an i32.
628  Value sparseMetadata = adaptor.getSparseMetadata();
629  if (sparseMetadata.getType() != VectorType::get(2, rewriter.getI16Type()))
630  return op->emitOpError() << "Expected metadata type to be LLVM "
631  "VectorType of 2 i16 elements";
632  sparseMetadata =
633  LLVM::BitcastOp::create(b, rewriter.getI32Type(), sparseMetadata);
634 
635  FailureOr<LLVM::InlineAsmOp> intrinsicResult = emitMmaSparseSyncOpAsm(
636  b, *ptxTypeA, *ptxTypeB, *ptxTypeC, *ptxTypeC, overflow, matA, matB,
637  matC, sparseMetadata, op.getSparsitySelector(), op.getMmaShapeAsArray(),
638  intrinsicResTy);
639  if (failed(intrinsicResult))
640  return failure();
641 
642  assert((*intrinsicResult).getNumResults() == 1 &&
643  "expected inline asm op returns a single LLVM struct type");
644  rewriter.replaceOp(
645  op, convertIntrinsicResult(op.getLoc(), intrinsicResTy, desiredRetTy,
646  (*intrinsicResult)->getResult(0), rewriter));
647  return success();
648  }

657  matchAndRewrite(nvgpu::DeviceAsyncCopyOp op, OpAdaptor adaptor,
658  ConversionPatternRewriter &rewriter) const override {
659  ImplicitLocOpBuilder b(op.getLoc(), rewriter);
660  Location loc = op.getLoc();
661  auto dstMemrefType = cast<MemRefType>(op.getDst().getType());
662  Value dstPtr =
663  getStridedElementPtr(rewriter, b.getLoc(), dstMemrefType,
664  adaptor.getDst(), adaptor.getDstIndices());

666  getTypeConverter()->getMemRefAddressSpace(dstMemrefType);
667  if (failed(dstAddressSpace))

669  loc, "destination memref address space not convertible to integer");

672  FailureOr<unsigned> srcAddressSpace =
673  getTypeConverter()->getMemRefAddressSpace(srcMemrefType);
674  if (failed(srcAddressSpace))
675  return rewriter.notifyMatchFailure(
676  loc, "source memref address space not convertible to integer");
677 
678  Value scrPtr =
679  getStridedElementPtr(rewriter, loc, srcMemrefType, adaptor.getSrc(),
680  adaptor.getSrcIndices());
681  // Intrinsics takes a global pointer so we need an address space cast.
682  auto srcPointerGlobalType = LLVM::LLVMPointerType::get(
683  op->getContext(), NVVM::NVVMMemorySpace::kGlobalMemorySpace);
684  scrPtr = LLVM::AddrSpaceCastOp::create(b, srcPointerGlobalType, scrPtr);
685  int64_t dstElements = adaptor.getDstElements().getZExtValue();
686  int64_t sizeInBytes =
687  (dstMemrefType.getElementTypeBitWidth() * dstElements) / 8;
688  // When the optional SrcElements argument is *not* present, the regular
689  // CpAsyncOp is generated. CopyAsyncOp reads bytes from source (global
690  // memory) to fill DstElements number of elements in the destination
691  // (shared memory).
692  Value srcBytes = adaptor.getSrcElements();
693  if (srcBytes) {
694  // When the optional SrcElements argument is present, the source (global
695  // memory) of CpAsyncOp is read only for SrcElements number of elements.
696  // The rest of the DstElements in the destination (shared memory) are
697  // filled with zeros.
698  Value c3I32 =
699  LLVM::ConstantOp::create(b, b.getI32Type(), b.getI32IntegerAttr(3));
700  Value bitwidth = LLVM::ConstantOp::create(
701  b, b.getI32Type(),
702  b.getI32IntegerAttr(srcMemrefType.getElementTypeBitWidth()));
703  Value srcElementsI32 = LLVM::TruncOp::create(b, b.getI32Type(), srcBytes);
704  srcBytes = LLVM::LShrOp::create(
705  b, LLVM::MulOp::create(b, bitwidth, srcElementsI32), c3I32);
706  }
707  // Cache global (.cg) for 16 dst bytes, Cache all (.ca) for sizes other than
708  // 16 dst bytes.
709  NVVM::LoadCacheModifierKind cacheModifier =
710  (op.getBypassL1().value_or(false) && sizeInBytes == 16)
711  ? NVVM::LoadCacheModifierKind::CG
712  : NVVM::LoadCacheModifierKind::CA;
713 
714  NVVM::CpAsyncOp::create(
715  b, dstPtr, scrPtr, rewriter.getI32IntegerAttr(sizeInBytes),
716  NVVM::LoadCacheModifierKindAttr::get(op->getContext(), cacheModifier),
717  srcBytes);
718 
719  // Drop the result token.
720  Value zero =
721  LLVM::ConstantOp::create(b, IntegerType::get(op.getContext(), 32),
722  rewriter.getI32IntegerAttr(0));
723  rewriter.replaceOp(op, zero);
724  return success();
725  }

1098  // // [49,52) base_offset
1099  dsc = insertBit(dsc, makeConst(offsetVal), startOffsetBit);
1100  // // [32,46) stride
1101  dsc = insertBit(dsc, strideDim, startStrideBit);
1102  // // [16,30) leading dimension
1103  dsc = insertBit(dsc, leadDim, startLeadBit);
1104  // // [0,14) start_address
