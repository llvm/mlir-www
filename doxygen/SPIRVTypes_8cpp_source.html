29 // Helper function to collect extensions implied by a type by visiting all its
30 // subtypes. Maintains a set of `seen` types to avoid recursion in structs.
31 //
32 // Serves as the source-of-truth for type extension information. All extension
33 // logic should be added to this class, while the
34 // `SPIRVType::getExtensions` function should not handle extension-related logic
35 // directly and only invoke `TypeExtensionVisitor::add(Type *)`.
36 class TypeExtensionVisitor {
37 public:
38  TypeExtensionVisitor(SPIRVType::ExtensionArrayRefVector &extensions,
39  std::optional<StorageClass> storage)
40  : extensions(extensions), storage(storage) {}
41 
42  // Main visitor entry point. Adds all extensions to the vector. Saves `type`
43  // as seen and dispatches to the right concrete `.add` function.
44  void add(SPIRVType type) {
45  if (auto [_it, inserted] = seen.insert({type, storage}); !inserted)
46  return;
47 
48  TypeSwitch<SPIRVType>(type)
49  .Case<CooperativeMatrixType, PointerType, ScalarType, TensorArmType>(
50  [this](auto concreteType) { addConcrete(concreteType); })
51  .Case<ArrayType, ImageType, MatrixType, RuntimeArrayType, VectorType>(
52  [this](auto concreteType) { add(concreteType.getElementType()); })
53  .Case<SampledImageType>([this](SampledImageType concreteType) {
54  add(concreteType.getImageType());
55  })
56  .Case<StructType>([this](StructType concreteType) {
57  for (Type elementType : concreteType.getElementTypes())
58  add(elementType);
59  })
60  .Default([](SPIRVType) { llvm_unreachable("Unhandled type"); });
61  }
62 
63  void add(Type type) { add(cast<SPIRVType>(type)); }
64 
65 private:
66  // Types that add unique extensions.
67  void addConcrete(CooperativeMatrixType type);
68  void addConcrete(PointerType type);
69  void addConcrete(ScalarType type);
70  void addConcrete(TensorArmType type);
71 
72  SPIRVType::ExtensionArrayRefVector &extensions;
73  std::optional<StorageClass> storage;
74  llvm::SmallDenseSet<std::pair<Type, std::optional<StorageClass>>> seen;
75 };

233 struct spirv::detail::CooperativeMatrixTypeStorage final : TypeStorage {
234  // In the specification dimensions of the Cooperative Matrix are 32-bit
235  // integers --- the initial implementation kept those values as such. However,
236  // the `ShapedType` expects the shape to be `int64_t`. We could keep the shape
237  // as 32-bits and expose it as int64_t through `getShape`, however, this
238  // method returns an `ArrayRef`, so returning `ArrayRef<int64_t>` having two
239  // 32-bits integers would require an extra logic and storage. So, we diverge
240  // from the spec and internally represent the dimensions as 64-bit integers,
241  // so we can easily return an `ArrayRef` from `getShape` without any extra
242  // logic. Alternatively, we could store both rows and columns (both 32-bits)
243  // and shape (64-bits), assigning rows and columns to shape whenever
244  // `getShape` is called. This would be at the cost of extra logic and storage.
245  // Note: Because `ArrayRef` is returned we cannot construct an object in
246  // `getShape` on the fly.
247  using KeyTy =
248  std::tuple<Type, int64_t, int64_t, Scope, CooperativeMatrixUseKHR>;
249 
250  static CooperativeMatrixTypeStorage *
251  construct(TypeStorageAllocator &allocator, const KeyTy &key) {
252  return new (allocator.allocate<CooperativeMatrixTypeStorage>())
253  CooperativeMatrixTypeStorage(key);
254  }
255 
256  bool operator==(const KeyTy &key) const {
257  return key == KeyTy(elementType, shape[0], shape[1], scope, use);
258  }
259 
260  CooperativeMatrixTypeStorage(const KeyTy &key)
261  : elementType(std::get<0>(key)),
262  shape({std::get<1>(key), std::get<2>(key)}), scope(std::get<3>(key)),
263  use(std::get<4>(key)) {}
264 
265  Type elementType;
266  // [#rows, #columns]
267  std::array<int64_t, 2> shape;
268  Scope scope;
269  CooperativeMatrixUseKHR use;
270 };

585 void TypeCapabilityVisitor::addConcrete(ScalarType type) {
586  unsigned bitwidth = type.getIntOrFloatBitWidth();
587 
588  // 8- or 16-bit integer/floating-point numbers will require extra capabilities
589  // to appear in interface storage classes. See SPV_KHR_16bit_storage and
590  // SPV_KHR_8bit_storage for more details.
591 
592 #define STORAGE_CASE(storage, cap8, cap16) \
593  case StorageClass::storage: { \
594  if (bitwidth == 8) { \
595  static constexpr auto cap = Capability::cap8; \
596  capabilities.push_back(cap); \
597  return; \
598  } \
599  if (bitwidth == 16) { \
600  static constexpr auto cap = Capability::cap16; \
601  capabilities.push_back(cap); \
602  return; \
603  } \
604  /* For 64-bit integers/floats, Int64/Float64 enables support for all */ \
605  /* storage classes. Fall through to the next section. */ \
606  } break
607 
608  // This part only handles the cases where special bitwidths appearing in
609  // interface storage classes.
610  if (storage) {
611  switch (*storage) {
612  STORAGE_CASE(PushConstant, StoragePushConstant8, StoragePushConstant16);
613  STORAGE_CASE(StorageBuffer, StorageBuffer8BitAccess,
614  StorageBuffer16BitAccess);
615  STORAGE_CASE(Uniform, UniformAndStorageBuffer8BitAccess,
616  StorageUniform16);
617  case StorageClass::Input:
618  case StorageClass::Output: {
619  if (bitwidth == 16) {
620  static constexpr auto cap = Capability::StorageInputOutput16;
621  capabilities.push_back(cap);
622  return;
623  }
624  break;
625  }
626  default:
627  break;
628  }
629  }
630 #undef STORAGE_CASE
631 
632  // For other non-interface storage classes, require a different set of
633  // capabilities for special bitwidths.
634 
635 #define WIDTH_CASE(type, width) \
636  case width: { \
637  static constexpr auto cap = Capability::type##width; \
638  capabilities.push_back(cap); \
639  } break
640 
641  if (auto intType = dyn_cast<IntegerType>(type)) {
642  switch (bitwidth) {
643  WIDTH_CASE(Int, 8);
644  WIDTH_CASE(Int, 16);
645  WIDTH_CASE(Int, 64);
646  case 1:
647  case 32:
648  break;
649  default:
650  llvm_unreachable("invalid bitwidth to getCapabilities");
651  }
652  } else {
653  assert(isa<FloatType>(type));
654  switch (bitwidth) {
655  case 16: {
656  if (isa<BFloat16Type>(type)) {
657  static constexpr auto cap = Capability::BFloat16TypeKHR;
658  capabilities.push_back(cap);
659  } else {

708 std::optional<int64_t> SPIRVType::getSizeInBytes() {
709  return TypeSwitch<SPIRVType, std::optional<int64_t>>(*this)
710  .Case<ScalarType>([](ScalarType type) -> std::optional<int64_t> {
711  // According to the SPIR-V spec:
712  // "There is no physical size or bit pattern defined for values with
713  // boolean type. If they are stored (in conjunction with OpVariable),
714  // they can only be used with logical addressing operations, not
715  // physical, and only with non-externally visible shader Storage
716  // Classes: Workgroup, CrossWorkgroup, Private, Function, Input, and
717  // Output."
718  int64_t bitWidth = type.getIntOrFloatBitWidth();
719  if (bitWidth == 1)
720  return std::nullopt;
721  return bitWidth / 8;
722  })
723  .Case<ArrayType>([](ArrayType type) -> std::optional<int64_t> {
724  // Since array type may have an explicit stride declaration (in bytes),
725  // we also include it in the calculation.
726  auto elementType = cast<SPIRVType>(type.getElementType());
727  if (std::optional<int64_t> size = elementType.getSizeInBytes())
728  return (*size + type.getArrayStride()) * type.getNumElements();
729  return std::nullopt;
730  })
731  .Case<VectorType, TensorArmType>([](auto type) -> std::optional<int64_t> {
732  if (std::optional<int64_t> elementSize =
733  cast<ScalarType>(type.getElementType()).getSizeInBytes())
734  return *elementSize * type.getNumElements();
735  return std::nullopt;
736  })
737  .Default(std::optional<int64_t>());
738 }

791 /// Type storage for SPIR-V structure types:
792 ///
793 /// Structures are uniqued using:
794 /// - for identified structs:
795 /// - a string identifier;
796 /// - for literal structs:
797 /// - a list of member types;
798 /// - a list of member offset info;
799 /// - a list of member decoration info;
800 /// - a list of struct decoration info.
801 ///
802 /// Identified structures only have a mutable component consisting of:
803 /// - a list of member types;
804 /// - a list of member offset info;
805 /// - a list of member decoration info;
806 /// - a list of struct decoration info.
807 struct spirv::detail::StructTypeStorage : public TypeStorage {
808  /// Construct a storage object for an identified struct type. A struct type
809  /// associated with such storage must call StructType::trySetBody(...) later
810  /// in order to mutate the storage object providing the actual content.
811  StructTypeStorage(StringRef identifier)
812  : memberTypesAndIsBodySet(nullptr, false), offsetInfo(nullptr),
813  numMembers(0), numMemberDecorations(0), memberDecorationsInfo(nullptr),
814  numStructDecorations(0), structDecorationsInfo(nullptr),
815  identifier(identifier) {}
816 
817  /// Construct a storage object for a literal struct type. A struct type
818  /// associated with such storage is immutable.
819  StructTypeStorage(
820  unsigned numMembers, Type const *memberTypes,
821  StructType::OffsetInfo const *layoutInfo, unsigned numMemberDecorations,
822  StructType::MemberDecorationInfo const *memberDecorationsInfo,
823  unsigned numStructDecorations,
824  StructType::StructDecorationInfo const *structDecorationsInfo)
825  : memberTypesAndIsBodySet(memberTypes, false), offsetInfo(layoutInfo),
826  numMembers(numMembers), numMemberDecorations(numMemberDecorations),
827  memberDecorationsInfo(memberDecorationsInfo),
828  numStructDecorations(numStructDecorations),
829  structDecorationsInfo(structDecorationsInfo) {}
830 
831  /// A storage key is divided into 2 parts:
832  /// - for identified structs:
833  /// - a StringRef representing the struct identifier;
834  /// - for literal structs:
835  /// - an ArrayRef<Type> for member types;
836  /// - an ArrayRef<StructType::OffsetInfo> for member offset info;
837  /// - an ArrayRef<StructType::MemberDecorationInfo> for member decoration
838  /// info;
839  /// - an ArrayRef<StructType::StructDecorationInfo> for struct decoration
840  /// info.
841  ///
842  /// An identified struct type is uniqued only by the first part (field 0)
843  /// of the key.
844  ///
845  /// A literal struct type is uniqued only by the second part (fields 1, 2, 3
846  /// and 4) of the key. The identifier field (field 0) must be empty.
847  using KeyTy =
848  std::tuple<StringRef, ArrayRef<Type>, ArrayRef<StructType::OffsetInfo>,
849  ArrayRef<StructType::MemberDecorationInfo>,
850  ArrayRef<StructType::StructDecorationInfo>>;
851 
852  /// For identified structs, return true if the given key contains the same
853  /// identifier.
854  ///
855  /// For literal structs, return true if the given key contains a matching list
856  /// of member types + offset info + decoration info.
857  bool operator==(const KeyTy &key) const {
858  if (isIdentified()) {
859  // Identified types are uniqued by their identifier.
860  return getIdentifier() == std::get<0>(key);
861  }
862 
863  return key == KeyTy(StringRef(), getMemberTypes(), getOffsetInfo(),
864  getMemberDecorationsInfo(), getStructDecorationsInfo());
865  }
866 
867  /// If the given key contains a non-empty identifier, this method constructs
868  /// an identified struct and leaves the rest of the struct type data to be set
869  /// through a later call to StructType::trySetBody(...).
870  ///
871  /// If, on the other hand, the key contains an empty identifier, a literal
872  /// struct is constructed using the other fields of the key.
873  static StructTypeStorage *construct(TypeStorageAllocator &allocator,
874  const KeyTy &key) {
875  StringRef keyIdentifier = std::get<0>(key);
876 
877  if (!keyIdentifier.empty()) {
878  StringRef identifier = allocator.copyInto(keyIdentifier);
879 
880  // Identified StructType body/members will be set through trySetBody(...)
881  // later.
882  return new (allocator.allocate<StructTypeStorage>())
883  StructTypeStorage(identifier);
884  }
885 
886  ArrayRef<Type> keyTypes = std::get<1>(key);
887 
888  // Copy the member type and layout information into the bump pointer
889  const Type *typesList = nullptr;
890  if (!keyTypes.empty()) {
891  typesList = allocator.copyInto(keyTypes).data();
892  }
893 
894  const StructType::OffsetInfo *offsetInfoList = nullptr;
895  if (!std::get<2>(key).empty()) {
896  ArrayRef<StructType::OffsetInfo> keyOffsetInfo = std::get<2>(key);
897  assert(keyOffsetInfo.size() == keyTypes.size() &&
898  "size of offset information must be same as the size of number of "
899  "elements");
900  offsetInfoList = allocator.copyInto(keyOffsetInfo).data();
901  }
902 
903  const StructType::MemberDecorationInfo *memberDecorationList = nullptr;
904  unsigned numMemberDecorations = 0;
905  if (!std::get<3>(key).empty()) {
906  auto keyMemberDecorations = std::get<3>(key);
907  numMemberDecorations = keyMemberDecorations.size();
908  memberDecorationList = allocator.copyInto(keyMemberDecorations).data();
909  }
910 
911  const StructType::StructDecorationInfo *structDecorationList = nullptr;
912  unsigned numStructDecorations = 0;
913  if (!std::get<4>(key).empty()) {
914  auto keyStructDecorations = std::get<4>(key);
915  numStructDecorations = keyStructDecorations.size();
916  structDecorationList = allocator.copyInto(keyStructDecorations).data();
917  }
918 
919  return new (allocator.allocate<StructTypeStorage>()) StructTypeStorage(
920  keyTypes.size(), typesList, offsetInfoList, numMemberDecorations,
921  memberDecorationList, numStructDecorations, structDecorationList);
922  }

951 
952  bool isIdentified() const { return !identifier.empty(); }
953 
954  /// Sets the struct type content for identified structs. Calling this method
955  /// is only valid for identified structs.
956  ///
957  /// Fails under the following conditions:
958  /// - If called for a literal struct;
959  /// - If called for an identified struct whose body was set before (through a
960  /// call to this method) but with different contents from the passed
961  /// arguments.
962  LogicalResult
963  mutate(TypeStorageAllocator &allocator, ArrayRef<Type> structMemberTypes,
964  ArrayRef<StructType::OffsetInfo> structOffsetInfo,
965  ArrayRef<StructType::MemberDecorationInfo> structMemberDecorationInfo,
966  ArrayRef<StructType::StructDecorationInfo> structDecorationInfo) {
967  if (!isIdentified())
968  return failure();
969 
970  if (memberTypesAndIsBodySet.getInt() &&
971  (getMemberTypes() != structMemberTypes ||
972  getOffsetInfo() != structOffsetInfo ||
973  getMemberDecorationsInfo() != structMemberDecorationInfo ||
974  getStructDecorationsInfo() != structDecorationInfo))
975  return failure();
976 
977  memberTypesAndIsBodySet.setInt(true);
978  numMembers = structMemberTypes.size();
979 
980  // Copy the member type and layout information into the bump pointer.
981  if (!structMemberTypes.empty())
982  memberTypesAndIsBodySet.setPointer(
983  allocator.copyInto(structMemberTypes).data());
984 
985  if (!structOffsetInfo.empty()) {
986  assert(structOffsetInfo.size() == structMemberTypes.size() &&
987  "size of offset information must be same as the size of number of "
988  "elements");
989  offsetInfo = allocator.copyInto(structOffsetInfo).data();
990  }
991 
992  if (!structMemberDecorationInfo.empty()) {
993  numMemberDecorations = structMemberDecorationInfo.size();
994  memberDecorationsInfo =
995  allocator.copyInto(structMemberDecorationInfo).data();
996  }
997 
998  if (!structDecorationInfo.empty()) {
999  numStructDecorations = structDecorationInfo.size();
1000  structDecorationsInfo = allocator.copyInto(structDecorationInfo).data();
1001  }
1002 
1003  return success();
1004  }

1007  StructType::OffsetInfo const *offsetInfo;
1008  unsigned numMembers;
1009  unsigned numMemberDecorations;
1010  StructType::MemberDecorationInfo const *memberDecorationsInfo;
1011  unsigned numStructDecorations;
1012  StructType::StructDecorationInfo const *structDecorationsInfo;
1013  StringRef identifier;
1014 };
