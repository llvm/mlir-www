451  /// Parse an optional integer value from the stream.
452  virtual OptionalParseResult parseOptionalInteger(APInt &result) = 0;

454  template <typename IntT>
455  OptionalParseResult parseOptionalInteger(IntT &result) {
456  auto loc = getCurrentLocation();
457 
458  // Parse the unsigned variant.
459  APInt uintResult;
460  OptionalParseResult parseResult = parseOptionalInteger(uintResult);
461  if (!parseResult.hasValue() || failed(*parseResult))
462  return parseResult;
463 
464  // Try to convert to the provided integer type. sextOrTrunc is correct even
465  // for unsigned types because parseOptionalInteger ensures the sign bit is
466  // zero for non-negated integers.
467  result =
468  (IntT)uintResult.sextOrTrunc(sizeof(IntT) * CHAR_BIT).getLimitedValue();
469  if (APInt(uintResult.getBitWidth(), result) != uintResult)
470  return emitError(loc, "integer value too large");
471  return success();
472  }
