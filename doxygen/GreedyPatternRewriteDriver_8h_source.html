21 /// This class allows control over how the GreedyPatternRewriteDriver works.
22 class GreedyRewriteConfig {
23 public:
24  /// This specifies the order of initial traversal that populates the rewriters
25  /// worklist. When set to true, it walks the operations top-down, which is
26  /// generally more efficient in compile time. When set to false, its initial
27  /// traversal of the region tree is bottom up on each block, which may match
28  /// larger patterns when given an ambiguous pattern set.
29  bool useTopDownTraversal = false;
30 
31  // Perform control flow optimizations to the region tree after applying all
32  // patterns.
33  bool enableRegionSimplification = true;
34 
35  /// This specifies the maximum number of times the rewriter will iterate
36  /// between applying patterns and simplifying regions.
37  unsigned maxIterations = 10;
38 };

44 /// Rewrite the regions of the specified operation, which must be isolated from
45 /// above, by repeatedly applying the highest benefit patterns in a greedy
46 /// work-list driven manner.
47 ///
48 /// This variant may stop after a predefined number of iterations, see the
49 /// alternative below to provide a specific number of iterations before stopping
50 /// in absence of convergence.
51 ///
52 /// Return success if the iterative process converged and no more patterns can
53 /// be matched in the result operation regions.
54 ///
55 /// Note: This does not apply patterns to the top-level operation itself.
56 /// These methods also perform folding and simple dead-code elimination
57 /// before attempting to match any of the provided patterns.
58 ///
59 /// You may configure several aspects of this with GreedyRewriteConfig.
60 LogicalResult applyPatternsAndFoldGreedily(
61  MutableArrayRef<Region> regions, const FrozenRewritePatternSet &patterns,
62  GreedyRewriteConfig config = GreedyRewriteConfig());
63 
64 /// Rewrite the given regions, which must be isolated from above.
65 inline LogicalResult applyPatternsAndFoldGreedily(
66  Operation *op, const FrozenRewritePatternSet &patterns,
67  GreedyRewriteConfig config = GreedyRewriteConfig()) {
68  return applyPatternsAndFoldGreedily(op->getRegions(), patterns, config);
69 }
