309  /// Iterator over the integer elements of a dense array.
310  class PyDenseArrayIterator {
311  public:
312  PyDenseArrayIterator(PyAttribute attr) : attr(std::move(attr)) {}
313 
314  /// Return a copy of the iterator.
315  PyDenseArrayIterator dunderIter() { return *this; }
316 
317  /// Return the next element.
318  EltTy dunderNext() {
319  // Throw if the index has reached the end.
320  if (nextIndex >= mlirDenseArrayGetNumElements(attr.get()))
321  throw nb::stop_iteration();

323  }

325  /// Bind the iterator class.
326  static void bind(nb::module_ &m) {
327  nb::class_<PyDenseArrayIterator>(m, DerivedT::pyIteratorName)
328  .def("__iter__", &PyDenseArrayIterator::dunderIter)
329  .def("__next__", &PyDenseArrayIterator::dunderNext);
330  }
331 
332  private:
333  /// The referenced dense array attribute.
334  PyAttribute attr;
335  /// The next index to read.
336  int nextIndex = 0;
337  };

339  /// Get the element at the given index.
340  EltTy getItem(intptr_t i) { return DerivedT::getElement(*this, i); }
341 
342  /// Bind the attribute class.
343  static void bindDerived(typename PyConcreteAttribute<DerivedT>::ClassTy &c) {
344  // Bind the constructor.
345  if constexpr (std::is_same_v<EltTy, bool>) {
346  c.def_static(
347  "get",
348  [](const nb::sequence &py_values, DefaultingPyMlirContext ctx) {
349  std::vector<bool> values;
350  for (nb::handle py_value : py_values) {
351  int is_true = PyObject_IsTrue(py_value.ptr());
352  if (is_true < 0) {
353  throw nb::python_error();
354  }
355  values.push_back(is_true);
356  }
357  return getAttribute(values, ctx->getRef());
358  },
359  nb::arg("values"), nb::arg("context") = nb::none(),
360  "Gets a uniqued dense array attribute");
361  } else {
362  c.def_static(
363  "get",
364  [](const std::vector<EltTy> &values, DefaultingPyMlirContext ctx) {
365  return getAttribute(values, ctx->getRef());
366  },
367  nb::arg("values"), nb::arg("context") = nb::none(),
368  "Gets a uniqued dense array attribute");
369  }
370  // Bind the array methods.
371  c.def("__getitem__", [](DerivedT &arr, intptr_t i) {
372  if (i >= mlirDenseArrayGetNumElements(arr))
373  throw nb::index_error("DenseArray index out of range");
374  return arr.getItem(i);
375  });
376  c.def("__len__", [](const DerivedT &arr) {
377  return mlirDenseArrayGetNumElements(arr);
378  });
379  c.def("__iter__",
380  [](const DerivedT &arr) { return PyDenseArrayIterator(arr); });
381  c.def("__add__", [](DerivedT &arr, const nb::list &extras) {
382  std::vector<EltTy> values;
383  intptr_t numOldElements = mlirDenseArrayGetNumElements(arr);
384  values.reserve(numOldElements + nb::len(extras));
385  for (intptr_t i = 0; i < numOldElements; ++i)
386  values.push_back(arr.getItem(i));
387  for (nb::handle attr : extras)
388  values.push_back(pyTryCast<EltTy>(attr));
389  return getAttribute(values, arr.getContext());
390  });
391  }
392 
393 private:
394  static DerivedT getAttribute(const std::vector<EltTy> &values,
395  PyMlirContextRef ctx) {
396  if constexpr (std::is_same_v<EltTy, bool>) {
397  std::vector<int> intValues(values.begin(), values.end());
398  MlirAttribute attr = DerivedT::getAttribute(ctx->get(), intValues.size(),
399  intValues.data());
400  return DerivedT(ctx, attr);
401  } else {
402  MlirAttribute attr =
403  DerivedT::getAttribute(ctx->get(), values.size(), values.data());
404  return DerivedT(ctx, attr);
405  }
406  }

1071  static MlirAttribute getAttributeFromBuffer(
1072  Py_buffer &view, bool signless, std::optional<PyType> explicitType,
1073  const std::optional<std::vector<int64_t>> &explicitShape,
1074  MlirContext &context) {
1075  // Detect format codes that are suitable for bulk loading. This includes
1076  // all byte aligned integer and floating point types up to 8 bytes.
1077  // Notably, this excludes exotics types which do not have a direct
1078  // representation in the buffer protocol (i.e. complex, etc).
1079  std::optional<MlirType> bulkLoadElementType;
1080  if (explicitType) {
1081  bulkLoadElementType = *explicitType;
1082  } else {
1083  std::string_view format(view.format);
1084  if (format == "f") {
1085  // f32
1086  assert(view.itemsize == 4 && "mismatched array itemsize");
1087  bulkLoadElementType = mlirF32TypeGet(context);
1088  } else if (format == "d") {
1089  // f64
1090  assert(view.itemsize == 8 && "mismatched array itemsize");
1091  bulkLoadElementType = mlirF64TypeGet(context);
1092  } else if (format == "e") {
1093  // f16
1094  assert(view.itemsize == 2 && "mismatched array itemsize");
1095  bulkLoadElementType = mlirF16TypeGet(context);
1096  } else if (format == "?") {
1097  // i1
1098  // The i1 type needs to be bit-packed, so we will handle it separately
1099  return getBitpackedAttributeFromBooleanBuffer(view, explicitShape,
1100  context);
1101  } else if (isSignedIntegerFormat(format)) {
1102  if (view.itemsize == 4) {
1103  // i32
1104  bulkLoadElementType = signless
1105  ? mlirIntegerTypeGet(context, 32)
1106  : mlirIntegerTypeSignedGet(context, 32);
1107  } else if (view.itemsize == 8) {
1108  // i64
1109  bulkLoadElementType = signless
1110  ? mlirIntegerTypeGet(context, 64)
1111  : mlirIntegerTypeSignedGet(context, 64);
1112  } else if (view.itemsize == 1) {
1113  // i8
1114  bulkLoadElementType = signless ? mlirIntegerTypeGet(context, 8)
1115  : mlirIntegerTypeSignedGet(context, 8);
1116  } else if (view.itemsize == 2) {
1117  // i16
1118  bulkLoadElementType = signless
1119  ? mlirIntegerTypeGet(context, 16)
1120  : mlirIntegerTypeSignedGet(context, 16);
1121  }
1122  } else if (isUnsignedIntegerFormat(format)) {
1123  if (view.itemsize == 4) {
1124  // unsigned i32
1125  bulkLoadElementType = signless
1126  ? mlirIntegerTypeGet(context, 32)
1127  : mlirIntegerTypeUnsignedGet(context, 32);
1128  } else if (view.itemsize == 8) {
1129  // unsigned i64
1130  bulkLoadElementType = signless
1131  ? mlirIntegerTypeGet(context, 64)
1132  : mlirIntegerTypeUnsignedGet(context, 64);
1133  } else if (view.itemsize == 1) {
1134  // i8
1135  bulkLoadElementType = signless
1136  ? mlirIntegerTypeGet(context, 8)
1137  : mlirIntegerTypeUnsignedGet(context, 8);
1138  } else if (view.itemsize == 2) {
1139  // i16
1140  bulkLoadElementType = signless
1141  ? mlirIntegerTypeGet(context, 16)
1142  : mlirIntegerTypeUnsignedGet(context, 16);
1143  }
1144  }
1145  if (!bulkLoadElementType) {
1146  throw std::invalid_argument(
1147  std::string("unimplemented array format conversion from format: ") +
1148  std::string(format));
1149  }
1150  }
1151 
1152  MlirType type = getShapedType(bulkLoadElementType, explicitShape, view);
1153  return mlirDenseElementsAttrRawBufferGet(type, view.len, view.buf);
1154  }

1158  // booleans per byte.

1160  Py_buffer &view, std::optional<std::vector<int64_t>> explicitShape,

1241  // Buffer is configured for read-only access below.

1256  for (intptr_t j = i; j < rank; ++j)
1257  strideFactor *= mlirShapedTypeGetDimSize(shapedType, j);
1258  strides.push_back(sizeof(Type) * strideFactor);
1259  }
1260  strides.push_back(sizeof(Type));
1261  }
1262  const char *format;
1263  if (explicitFormat) {
1264  format = explicitFormat;
1265  } else {
1266  format = nb_format_descriptor<Type>::format();
1267  }
1268  return std::make_unique<nb_buffer_info>(
1269  data, sizeof(Type), format, rank, std::move(shape), std::move(strides),
