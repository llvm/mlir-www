19LogicalResult mlir::DiagnosedSilenceableFailure::checkAndReport() {
20#if LLVM_ENABLE_ABI_BREAKING_CHECKS

23#endif // LLVM_ENABLE_ABI_BREAKING_CHECKS
24 if (!diagnostics.empty()) {
