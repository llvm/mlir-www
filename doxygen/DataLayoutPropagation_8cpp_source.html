40 // The struct contains the infomation about mapping packing information to
41 // the iteration domain of Linalg ops.
42 struct PackInfo {
43  int64_t getNumTiledLoops() const { return tileToPointMapping.size(); };
44  // InnerDimsPos on iteration domain, which follows the order in pack ops.
45  SmallVector<int64_t> tiledDimsPos;
46  // The sizes of tiling data dimensions on iteration domain.
47  llvm::DenseMap<int64_t, OpFoldResult> domainDimAndTileMapping;
48  // The mapping from a dimension of iteration domain to the corresponding inner
49  // tiling dimension on iteration domain.
50  llvm::DenseMap<int64_t, int64_t> tileToPointMapping;
51  // The permutation of outer dims (on domain).
52  SmallVector<int64_t> outerDimsOnDomainPerm;
53 };

55 template <typename OpTy>
56 static FailureOr<PackInfo>
57 getPackingInfoFromOperand(OpOperand *opOperand, linalg::GenericOp genericOp,
58  OpTy packOrUnPackOp) {
59  static_assert(llvm::is_one_of<OpTy, linalg::PackOp, linalg::UnPackOp>::value,
60  "applies to only pack or unpack operations");
61  LLVM_DEBUG(
62  { llvm::dbgs() << "--- Construct PackInfo From an operand ---\n"; });
63 
64  AffineMap indexingMap = genericOp.getMatchingIndexingMap(opOperand);
65  SmallVector<AffineMap> indexingMaps = genericOp.getIndexingMapsArray();
66  SmallVector<utils::IteratorType> iterators =
67  genericOp.getIteratorTypesArray();
68 
69  PackInfo packInfo;
70  int64_t origNumDims = indexingMap.getNumDims();
71  SmallVector<AffineExpr> exprs(indexingMap.getResults());
72  ArrayRef<int64_t> innerDimsPos = packOrUnPackOp.getInnerDimsPos();
73  for (auto [index, innerDimPos, tileSize] :
74  llvm::zip_equal(llvm::seq<unsigned>(0, innerDimsPos.size()),
75  innerDimsPos, packOrUnPackOp.getMixedTiles())) {
76  auto expr = exprs[innerDimPos];
77  if (!isa<AffineDimExpr>(expr))
78  return failure();
79  int64_t domainDimPos =
80  cast<AffineDimExpr>(exprs[innerDimPos]).getPosition();
81  if (!isParallelIterator(iterators[domainDimPos]))
82  return failure();
83  packInfo.tiledDimsPos.push_back(domainDimPos);
84  packInfo.domainDimAndTileMapping[domainDimPos] = tileSize;
85  packInfo.tileToPointMapping[domainDimPos] = origNumDims + index;
86  LLVM_DEBUG({
87  llvm::dbgs() << "map innerDimPos=" << innerDimPos
88  << " to iteration dimension (d" << domainDimPos << ", d"
89  << packInfo.tileToPointMapping[domainDimPos]
90  << "), which has size=("
91  << packInfo.domainDimAndTileMapping[domainDimPos] << ")\n";
92  });
93  }

95  // Bail out if a tiled dimension is present in a map but not as an affine dim
96  // expression.
97  auto areAllAffineDimExpr = [&](int dim) {
98  for (AffineMap map : indexingMaps) {
99  if (llvm::any_of(map.getResults(), [dim](AffineExpr expr) {
100  return expr.isFunctionOfDim(dim) && !isa<AffineDimExpr>(expr);
101  })) {
102  return false;
103  }
104  }
105  return true;
106  };
107  for (int64_t i : packInfo.tiledDimsPos)
108  if (!areAllAffineDimExpr(i))
109  return failure();
110 
111  // Get the outer dims perm on the iteration domain. Start by identifying the
112  // set of domain dims affected by the outer permutation along with the
113  // permuted ordering for those dims. Then the full outer dims permutation can
114  // be constructed by replacing the affected dims with the permuted result in a
115  // numLoops-rank identity. e.g.
116  // outerDimsPerm = [1, 2, 0]
117  // indexingMap = (d0, d1, d2, d3, d4) -> (d1, d4, d3)
118  //
119  // permutedOuterDims = [4, 3, 1]
120  // outerDimsOnDomainPerm = [0, 4, 2, 3, 1]
121  //
122  // Non-affine dim expressions must not be permuted by the outer dims
123  // permutation.
124  SmallVector<int64_t> permutedOuterDims;
125  for (auto [index, dim] : llvm::enumerate(packOrUnPackOp.getOuterDimsPerm())) {
126  auto permutedExpr = indexingMap.getResult(dim);
127  if (auto dimExpr = dyn_cast<AffineDimExpr>(permutedExpr)) {
128  permutedOuterDims.push_back(dimExpr.getPosition());
129  continue;
130  }
131 
132  // TODO: Allow propagation with transposes on non affine dim expressions,
133  // e.g. d0 + d1 which implies transposing both dims simultaneously while
134  // maintaining the relative position between them.
135  if (static_cast<int64_t>(index) != dim)
136  return failure();
137  }
138  if (!permutedOuterDims.empty()) {
139  int64_t outerDimIndex = 0;
140  llvm::DenseSet<int64_t> permutedDomainDims(permutedOuterDims.begin(),
141  permutedOuterDims.end());
142  for (int i = 0, e = indexingMap.getNumDims(); i < e; i++)
143  packInfo.outerDimsOnDomainPerm.push_back(
144  permutedDomainDims.contains(i) ? permutedOuterDims[outerDimIndex++]
145  : i);
146  LLVM_DEBUG({
147  llvm::dbgs() << "map outer dimsDimsPerm to ";
148  for (auto dim : packInfo.outerDimsOnDomainPerm)
149  llvm::dbgs() << dim << " ";
150  llvm::dbgs() << "\n";
151  });
152  }
153 
154  return packInfo;
155 }

157 static SmallVector<int64_t> computeOuterDims(ArrayRef<int64_t> perm,
158  ArrayRef<AffineExpr> exprs) {
159  // Compute `outer_dims_perm`. See example:
160  // current exprs : (d0, d1, d2, d3) -> (d2, d3)
161  // perm : [0, 3, 1, 2]
162  // First map d2, d3 with their position in the array as:
163  // currentPositionTileLoops: dim | pos
164  // d2 | 0
165  // d3 | 1
166  // then scan `perm` in order and get the `outer_dims_perm`
167  // to be used, here it would be [1, 0].
168  assert(!perm.empty() && "expect perm not to be empty");
169  assert(!exprs.empty() && "expect exprs not to be empty");
170  if (exprs.size() == 1)
171  return {};
172  SmallVector<int64_t> outerDimsPerm;
173  DenseMap<int64_t, int64_t> currentPositionTileLoops;
174  for (auto [pos, expr] : llvm::enumerate(exprs)) {
175  // Here we rely on the assumption that the outer dims permutation
176  // when propagating currently requires that non-affine dim expressions
177  // are not permuted, thus allowing the identity assignment below.
178  if (auto dimExpr = dyn_cast<AffineDimExpr>(expr))
179  currentPositionTileLoops[dimExpr.getPosition()] = pos;
180  else
181  currentPositionTileLoops[pos] = pos;
182  }
183  for (int64_t loopIdx : perm) {
184  if (currentPositionTileLoops.count(loopIdx))
185  outerDimsPerm.push_back(currentPositionTileLoops.lookup(loopIdx));
186  }
187  return outerDimsPerm;
188 }

190 /// Returns a tuple for packed operand and indexing_map with the assumptions:
191 /// 1) The generic op is the producer of the pack op.
192 /// 2) The generic op has only one result.
193 /// If the operand is a scalar or packing dimensions are all irrelevant to the
194 /// operand, the operand and the updated indexing map will be returned.
195 /// Otherwise, it returns the packed operand and the updated indexing map. E.g.,
196 ///
197 /// #map0 = affine_map<(d0, d1) -> (d0, d1)>
198 /// #map1 = affine_map<(d0, d1) -> (d0)>
199 /// #map2 = affine_map<(d0, d1) -> (d1)>
200 /// %0 = linalg.generic {indexing_maps = [#map1, #map2, #map0],
201 /// iterator_types = ["parallel", "parallel"]}
202 /// ins(%arg0, %arg1 : tensor<?xf32>, tensor<?xf32>)
203 /// outs(%init : tensor<?x?xf32>) {
204 /// ^bb0(%arg3: f32, %arg4: f32, %arg5: f32):
205 /// %4 = arith.addf %arg3, %arg4 : f32
206 /// linalg.yield %4 : f32
207 /// } -> tensor<?x?xf32>
208 /// %1 = linalg.pack %0
209 /// inner_dims_pos = [0, 1]
210 /// inner_tiles = [8, 2]
211 /// into %dest : tensor<?x?xf32> -> tensor<?x?x8x2xf32>
212 ///
213 /// Taking the first input operand as an example, the inner tile size of d1 is
214 /// 8. Thus, the below operation and `affine_map<(d0, d1, d2, d3)> ->
215 /// affine_map<(d1, d3)>` will be returned.
216 ///
217 /// %pack = linalg.pack %arg0
218 /// inner_dims_pos = [0]
219 /// inner_tiles = [8]
220 /// into %init : tensor<?xf32> -> tensor<?x8xf32>
221 static std::tuple<Value, AffineMap>
222 getOrCreatePackedViewOfOperand(OpBuilder &b, Location loc, PackInfo packInfo,
223  GenericOp genericOp, OpOperand *opOperand) {
224  int64_t numOrigLoops = genericOp.getNumLoops();
225  int64_t numInnerLoops = packInfo.getNumTiledLoops();
226  int64_t numLoops = numOrigLoops + numInnerLoops;
227  AffineMap origIndexingMap = genericOp.getMatchingIndexingMap(opOperand);
228  llvm::DenseMap<int64_t, int64_t> domainDimToOperandDim;
229  SmallVector<AffineExpr> exprs(origIndexingMap.getResults());
230 
231  // If the OpOperand is a scalar or a zero-rank tensor, no need to pack.
232  if (genericOp.isScalar(opOperand) || exprs.empty())
233  return std::make_tuple(opOperand->get(),
234  AffineMap::get(numLoops, 0, exprs, b.getContext()));
235 
236  // Step 1. Construct the information of packing data dimensions; append inner
237  // dimensions to the indexing maps for the operand.
238  for (auto [index, expr] : llvm::enumerate(exprs)) {
239  if (auto dimExpr = dyn_cast<AffineDimExpr>(expr)) {
240  int64_t dimPos = dimExpr.getPosition();
241  domainDimToOperandDim[dimPos] = index;
242  continue;
243  }
244  }
245  SmallVector<int64_t> innerDimsPos;
246  SmallVector<OpFoldResult> innerTileSizes;
247  for (auto dimPos : packInfo.tiledDimsPos) {
248  if (!domainDimToOperandDim.count(dimPos))
249  continue;
250  int64_t index = domainDimToOperandDim[dimPos];
251  innerTileSizes.push_back(packInfo.domainDimAndTileMapping[dimPos]);
252  innerDimsPos.push_back(index);
253  exprs.push_back(b.getAffineDimExpr(packInfo.tileToPointMapping[dimPos]));
254  }
255 
256  // Step 2. Handle outer dim permutations.
257  SmallVector<int64_t> outerDimsPerm;
258  if (!packInfo.outerDimsOnDomainPerm.empty()) {
259  outerDimsPerm = computeOuterDims(packInfo.outerDimsOnDomainPerm, exprs);
260 
261  // Step 2.1: Fold transpose into the linalg.generic.
262  SmallVector<int64_t> inversedOuterPerm =
263  invertPermutationVector(packInfo.outerDimsOnDomainPerm);
264  for (auto i : llvm::seq<unsigned>(0, origIndexingMap.getNumResults())) {
265  if (auto dimExpr = dyn_cast<AffineDimExpr>(exprs[i])) {
266  int64_t dimPos = dimExpr.getPosition();
267  exprs[i] = b.getAffineDimExpr(inversedOuterPerm[dimPos]);
268  continue;
269  }
270  assert(isa<AffineConstantExpr>(exprs[i]) &&
271  "Attempted to permute non-constant and non-affine dim expression");
272  }
273  // Step 2.2: Undo the transposition on `exprs` and propagate the
274  // transposition on the pack using outerDimsPerm.
275  if (!outerDimsPerm.empty()) {
276  SmallVector<AffineExpr> auxVec = exprs;
277  for (const auto &en : enumerate(outerDimsPerm))
278  auxVec[en.index()] = exprs[en.value()];
279  exprs = auxVec;
280  }
281  }
282  auto indexingMap = AffineMap::get(numLoops, 0, exprs, b.getContext());
283 
284  // The operand does not have dimensions that relates to pack op.
285  if (innerDimsPos.empty() && outerDimsPerm.empty())
286  return std::make_tuple(opOperand->get(), indexingMap);
287 
288  auto empty = linalg::PackOp::createDestinationTensor(
289  b, loc, opOperand->get(), innerTileSizes, innerDimsPos, outerDimsPerm);
290  auto packedOperand = linalg::PackOp::create(
291  b, loc, opOperand->get(), empty, innerDimsPos, innerTileSizes,
292  /*padding=*/std::nullopt, outerDimsPerm);
293  return std::make_tuple(packedOperand, indexingMap);
294 }

296 /// This function is a helper subroutine to pack a genericOp and return it. It
297 /// will create a new generic op with the packed operand and the packed output
298 /// according to packInfo when we attempt to push down unpack or bubble up pack
299 /// around it. Implicitly this will only work when a packInfo can be obtained.
300 /// This make sure that we are only using this function on parallel permuted
301 /// dimensions.
302 static GenericOp packGenericOp(RewriterBase &rewriter, GenericOp genericOp,
303  Value dest, AffineMap packedOutIndexingMap,
304  const PackInfo &packInfo,
305  bool isFoldableUnpackPack) {
306  Location loc = genericOp.getLoc();
307  SmallVector<Value> inputOperands;
308  SmallVector<Value> inputOperandsFromUnpackedSource;
309  SmallVector<AffineMap> indexingMaps;
310  auto hasEquivalentTiles = [](PackOp packOp, UnPackOp unPackOp) {
311  return packOp.getOuterDimsPerm() == unPackOp.getOuterDimsPerm() &&
312  packOp.getInnerDimsPos() == unPackOp.getInnerDimsPos() &&
313  llvm::equal(packOp.getMixedTiles(), unPackOp.getMixedTiles());
314  };
315  for (OpOperand *inputOperand : genericOp.getDpsInputOperands()) {
316  auto [packedOperand, packedIndexingMap] = getOrCreatePackedViewOfOperand(
317  rewriter, loc, packInfo, genericOp, inputOperand);
318  auto unpackOp = inputOperand->get().getDefiningOp<linalg::UnPackOp>();
319  auto packOp = packedOperand.getDefiningOp<linalg::PackOp>();
320  if (packOp && unpackOp && hasEquivalentTiles(packOp, unpackOp)) {
321  inputOperandsFromUnpackedSource.push_back(unpackOp.getSource());
322  } else {
323  inputOperandsFromUnpackedSource.push_back(packedOperand);
324  }
325  inputOperands.push_back(packedOperand);
326  indexingMaps.push_back(packedIndexingMap);
327  }
328 
329  // If the unpack->pack sequences can be folded, replace use the sources of
330  // the unpack ops in any unpack->pack chains on the generic op operands.
331  if (isFoldableUnpackPack) {
332  inputOperands = inputOperandsFromUnpackedSource;
333  if (auto destPack = dest.getDefiningOp<linalg::PackOp>()) {
334  auto destUnPack = destPack.getSource().getDefiningOp<linalg::UnPackOp>();
335  if (destUnPack && hasEquivalentTiles(destPack, destUnPack)) {
336  dest = destUnPack.getSource();
337  }
338  }
339  }
340 
341  int64_t numInnerLoops = packInfo.getNumTiledLoops();
342  SmallVector<utils::IteratorType> iterTypes =
343  genericOp.getIteratorTypesArray();
344  iterTypes.append(numInnerLoops, utils::IteratorType::parallel);
345 
346  indexingMaps.push_back(packedOutIndexingMap);
347 
348  auto newGenericOp = linalg::GenericOp::create(
349  rewriter, loc, dest.getType(), inputOperands, dest, indexingMaps,
350  iterTypes,
351  /*bodyBuild=*/nullptr, linalg::getPrunedAttributeList(genericOp));
352  rewriter.cloneRegionBefore(genericOp.getRegion(), newGenericOp.getRegion(),
353  newGenericOp.getRegion().begin());
354  return newGenericOp;
355 }

363 /// Bubbles up linalg.pack op through a producer generic op. This
364 /// swap pack(generic) to generic(pack). The new generic op works on packed
365 /// domain; pack ops are created for input and output operands. E.g.,
366 ///
367 /// #map0 = affine_map<(d0, d1) -> (d0, d1)>
368 /// %0 = tensor.dim %arg0, %c0 : tensor<?x?xf32>
369 /// %1 = tensor.dim %arg0, %c1 : tensor<?x?xf32>
370 /// %2 = tensor.empty(%0, %1) : tensor<?x?xf32>
371 /// %3 = linalg.generic {indexing_maps = [#map0, #map0],
372 /// iterator_types = ["parallel", "parallel"]}
373 /// ins(%arg0 : tensor<?x?xf32>)
374 /// outs(%2 : tensor<?x?xf32>) {
375 /// ^bb0(%arg3: f32, %arg4: f32):
376 /// %4 = arith.addf %arg3, %arg3 : f32
377 /// linalg.yield %4 : f32
378 /// } -> tensor<?x?xf32>
379 /// %4 = linalg.pack %3
380 /// inner_dims_pos = [0, 1]
381 /// inner_tiles = [8, 2]
382 /// into %dest : tensor<?x?xf32> -> tensor<?x?x8x2xf32>
383 ///
384 /// will be converted to
385 ///
386 /// #map = affine_map<()[s0] -> (s0 ceildiv 8)>
387 /// #map1 = affine_map<()[s0] -> (s0 ceildiv 2)>
388 /// #map2 = affine_map<(d0, d1, d2, d3) -> (d0, d1, d2, d3)>
389 /// %dim = tensor.dim %arg0, %c0 : tensor<?x?xf32>
390 /// %dim_0 = tensor.dim %arg0, %c1 : tensor<?x?xf32>
391 /// %0 = affine.apply #map()[%dim]
392 /// %1 = affine.apply #map1()[%dim_0]
393 /// %2 = tensor.empty(%0, %1) : tensor<?x?x8x2xf32>
394 /// %pack = linalg.pack %arg0
395 /// inner_dims_pos = [0, 1]
396 /// inner_tiles = [8, 2]
397 /// into %2 : tensor<?x?xf32> -> tensor<?x?x8x2xf32>
398 /// %3 = linalg.generic {indexing_maps = [#map2, #map2],
399 /// iterator_types = ["parallel", "parallel", "parallel", "parallel"]}
400 /// ins(%pack : tensor<?x?x8x2xf32>)
401 /// outs(%arg1 : tensor<?x?x8x2xf32>) {
402 /// ^bb0(%in: f32, %out: f32):
403 /// %4 = arith.addf %in, %in : f32
404 /// linalg.yield %4 : f32
405 /// } -> tensor<?x?x8x2xf32>
406 static FailureOr<GenericOp>
407 bubbleUpPackOpThroughGenericOp(RewriterBase &rewriter, linalg::PackOp packOp,
408  const ControlPropagationFn &controlFn) {
409  auto genericOp = packOp.getSource().getDefiningOp<GenericOp>();
410  if (!genericOp)
411  return failure();
412 
413  // User controlled propagation function.
414  if (!controlFn(&packOp.getSourceMutable()))
415  return failure();
416 
417  // TODO: Enable propagation in the presence of linalg.index and
418  // tensor.extract, likely as a separate pattern as the pack information and
419  // propagation decision needs to be inferred from the region of the generic.
420  if (hasGatherSemantics(genericOp))
421  return failure();
422 
423  // TODO: Relax the restriction. We are able to bubble up the pack op through
424  // multi-result generic op. It just needs more work.
425  if (genericOp.getNumResults() != 1)
426  return failure();
427 
428  // Bail-out if the result of the generic has multiple uses, as bubbling up
429  // creates recomputation if the generic has multiple users.
430  // TODO: Enable the case where every use is an identical pack op as no
431  // recomputation is needed in that case.
432  if (!genericOp->getResult(0).hasOneUse())
433  return failure();
434 
435  // TODO: Add an option for allowing padding values. It could introduce
436  // undefined behavior if we unconditionally propagate pack op through all
437  // the ops. E.g., if the padding value is zero and there are division ops in
438  // a generic op. Some values of padding area could be NaN (0/0).
439  if (packOp.getPaddingValue())
440  return failure();
441 
442  OpOperand *opOperand = genericOp.getDpsInitOperand(0);
443  auto packInfo = getPackingInfoFromOperand(opOperand, genericOp, packOp);
444  if (failed(packInfo))
445  return failure();
446 
447  // We want to move the pack not the generic.
448  OpBuilder::InsertionGuard guard(rewriter);
449  rewriter.setInsertionPoint(genericOp);
450 
451  // We need to handle two cases:
452  // 1) The linalg.pack destination is a tensor.empty. If this is the case, we
453  // create a new tensor.empty to avoid breaking dominance, as we are moving the
454  // linalg.pack above the linalg.generic.
455  // 2) The destination is not a tensor.empty. In this case we can replace only
456  // if the destination of the linalg.pack dominates the linalg.generic.
457  Value packOpDest = packOp.getDest();
458  if (!packOpDest.hasOneUse())
459  return failure();
460  if (auto emptyOp = packOpDest.getDefiningOp<tensor::EmptyOp>()) {
461  packOpDest = tensor::EmptyOp::create(rewriter, genericOp->getLoc(),
462  emptyOp.getMixedSizes(),
463  emptyOp.getType().getElementType());
464  } else {
465  DominanceInfo dom(genericOp);
466  if (!dom.properlyDominates(packOpDest, genericOp))
467  return failure();
468  }
469 
470  // Rebuild the indexing map for the corresponding init operand.
471  auto [packedOutOperand, packedOutIndexingMap] =
472  getOrCreatePackedViewOfOperand(rewriter, genericOp.getLoc(), *packInfo,
473  genericOp, opOperand);
474 
475  // Forward the new tensor.empty as a destination if it is one of the following
476  // situations:
477  // 1) The dps init operand is a tensor.empty.
478  // 2) The dps init is a write-only operand, i.e., it is not used in the
479  // genericOp
480  Value dest = packedOutOperand;
481  auto initTensor =
482  genericOp.getDpsInitOperand(0)->get().getDefiningOp<tensor::EmptyOp>();
483  if (initTensor || isGenericOutsNotUsed(genericOp)) {
484  dest = packOpDest;
485  }
486  // pack(unpack) isn't naively foldable because the unpack op can be from
487  // an arbitrary domain so we need to keep both.
488  return packGenericOp(rewriter, genericOp, dest, packedOutIndexingMap,
489  *packInfo, /*isFoldableUnpackPack=*/false);
490 }

514 /// Propagate a linalg.pack operation up through a tensor.pad. The idea is to
515 /// add as many zero padding dimensions in `high` and `low` based on the number
516 /// of point loops.
517 class BubbleUpPackThroughPadOp final : public OpRewritePattern<linalg::PackOp> {
518 public:
519  BubbleUpPackThroughPadOp(MLIRContext *context, ControlPropagationFn fun)
520  : OpRewritePattern<linalg::PackOp>(context), controlFn(std::move(fun)) {}
521 
522  LogicalResult matchAndRewrite(linalg::PackOp packOp,
523  PatternRewriter &rewriter) const override {
524  auto padOp = packOp.getSource().getDefiningOp<tensor::PadOp>();
525  if (!padOp)
526  return failure();
527 
528  // User controlled propagation function.
529  if (!controlFn(&packOp.getSourceMutable()))
530  return failure();
531 
532  // TODO: Enable padding when the padding values are the same.
533  if (packOp.getPaddingValue())
534  return failure();
535 
536  // Fail for non-constant padding values. The body of the pad could
537  // depend on the padding indices and/or properties of the padded
538  // tensor so for now we fail.
539  // TODO: Support non-constant padding values.
540  Value paddingVal = padOp.getConstantPaddingValue();
541  if (!paddingVal)
542  return failure();
543 
544  if (!packOp.getDest().getDefiningOp<tensor::EmptyOp>())
545  return failure();
546 
547  ArrayRef<int64_t> innerDimsPos = packOp.getInnerDimsPos();
548 
549  // Bail out if one of the padded dimension is a tiled one.
550  llvm::SmallBitVector paddedDims = padOp.getPaddedDims();
551  llvm::SmallBitVector innerDims(paddedDims.size());
552  for (int64_t dim : innerDimsPos)
553  innerDims.flip(dim);
554  if (paddedDims.anyCommon(innerDims))
555  return failure();
556 
557  Location loc = padOp->getLoc();
558  OpBuilder::InsertionGuard guard(rewriter);
559  rewriter.setInsertionPoint(padOp);
560 
561  ArrayRef<int64_t> outerDimsPerm = packOp.getOuterDimsPerm();
562  SmallVector<OpFoldResult> mixedTiles = packOp.getMixedTiles();
563  auto empty = linalg::PackOp::createDestinationTensor(
564  rewriter, loc, padOp.getSource(), mixedTiles, innerDimsPos,
565  outerDimsPerm);
566  auto sourcePack = linalg::PackOp::create(
567  rewriter, loc, padOp.getSource(), empty, innerDimsPos, mixedTiles,
568  /*padding=*/std::nullopt, outerDimsPerm);
569 
570  // If we have `outer_dims_perms` we need to adjust the padded dimensions.
571  SmallVector<OpFoldResult> lowPad = padOp.getMixedLowPad();
572  SmallVector<OpFoldResult> highPad = padOp.getMixedHighPad();
573  if (!outerDimsPerm.empty()) {
574  applyPermutationToVector<OpFoldResult>(lowPad, outerDimsPerm);
575  applyPermutationToVector<OpFoldResult>(highPad, outerDimsPerm);
576  }
577  // The tiled dimensions were verified to be unpadded above, so here we
578  // just append 0 for the inner tile dimensions.
579  size_t pointLoopsSize = innerDimsPos.size();
580  lowPad.append(pointLoopsSize, rewriter.getIndexAttr(0));
581  highPad.append(pointLoopsSize, rewriter.getIndexAttr(0));
582 
583  auto newPadOp =
584  tensor::PadOp::create(rewriter, loc, /*result=*/Type(), sourcePack,
585  lowPad, highPad, paddingVal, padOp.getNofold());
586 
587  // If the pad has more than one user, create an unpack on the new pad to
588  // replace the other uses.
589  if (!padOp->hasOneUse()) {
590  auto unpackEmpty = linalg::UnPackOp::createDestinationTensor(
591  rewriter, loc, newPadOp, mixedTiles, innerDimsPos, outerDimsPerm);
592  Value unpackedPad =
593  linalg::UnPackOp::create(rewriter, loc, newPadOp, unpackEmpty,
594  innerDimsPos, mixedTiles, outerDimsPerm);
595  rewriter.replaceAllUsesExcept(padOp, unpackedPad, sourcePack);
596  }
597 
598  // Replace the pack with the new pad.
599  rewriter.replaceOp(packOp, newPadOp.getResult());
600 
601  return success();
602  }
603 
604 private:
605  ControlPropagationFn controlFn;
606 };

608 /// Project dimsPos to the inner-most non-unit dim pos with reassocIndices.
609 ///
610 /// For example, given dimsPos [0, 2], reassocIndices [[0, 1], [2, 3]], and
611 /// targetShape [16, 16, 32, 1], it returns [1, 2]. Because for pos 0, the
612 /// inner-most projected dim in pos [0, 1] is 1. And for pos 2, the inner-most
613 /// non-unit projected dims in pos [2, 3] is 2.
614 ///
615 /// If all candidates in a reassociation are unit dims, it chooses the
616 /// inner-most dim pos.
617 static SmallVector<int64_t>
618 projectToInnerMostNonUnitDimsPos(ArrayRef<int64_t> dimsPos,
619  ArrayRef<ReassociationIndices> reassocIndices,
620  ArrayRef<int64_t> targetShape) {
621  SmallVector<int64_t> projectedDimsPos;
622  for (auto pos : dimsPos) {
623  // In the case all dims are unit, this will return the inner-most one.
624  int64_t projectedPos = reassocIndices[pos].back();
625  for (auto i : llvm::reverse(reassocIndices[pos])) {
626  int64_t dim = targetShape[i];
627  if (dim > 1 || ShapedType::isDynamic(dim)) {
628  projectedPos = i;
629  break;
630  }
631  }
632  projectedDimsPos.push_back(projectedPos);
633  }
634  return projectedDimsPos;
635 }

637 /// Check if all dims in dimsPos are divisible by the corresponding tile sizes.
638 static bool isDimsDivisibleByTileSizes(ArrayRef<int64_t> dimsPos,
639  ArrayRef<int64_t> shape,
640  ArrayRef<int64_t> tileSizes) {
641  for (auto [pos, tileSize] : llvm::zip_equal(dimsPos, tileSizes)) {
642  int64_t dim = shape[pos];
643  if (ShapedType::isDynamic(dim) || (dim % tileSize) != 0)
644  return false;
645  }
646  return true;
647 }

649 /// Permutate the reassociation indices and reindex them in the sequence order.
650 /// Returns the next dim pos in the sequence.
651 ///
652 /// For example, given reassocIndices [[0, 1], [2]] and permutation [1, 0], it
653 /// applies the permutation to get [[2], [0, 1]] and reindexes the indices into
654 /// [[0], [1, 2]].
655 static int64_t applyPermutationAndReindexReassoc(
656  SmallVector<ReassociationIndices> &reassocIndices,
657  ArrayRef<int64_t> permutation) {
658  if (!permutation.empty())
659  applyPermutationToVector<ReassociationIndices>(reassocIndices, permutation);
660  int64_t nextPos = 0;
661  for (ReassociationIndices &indices : reassocIndices) {
662  for (auto &index : indices) {
663  index = nextPos;
664  nextPos += 1;
665  }
666  }
667  return nextPos;
668 }

670 /// Bubble up pack op through collapse shape op when the packed dims can be
671 /// projected to the dims before collapsing. This is possible when the inner
672 /// tile sizes can divide the projected dims.
673 ///
674 /// For example:
675 ///
676 /// %collapsed = tensor.collapse_shape %in [[0, 1], 2]
677 /// : tensor<?x16x4xf32> into tensor<?x4xf32>
678 /// %pack = linalg.pack %collapsed outer_dims_perm = [0, 1]
679 /// inner_dims_pos = [0, 1] inner_tiles = [8, 1] into %empty
680 /// : tensor<?x4xf32> -> tensor<?x4x8x1xf32>
681 ///
682 /// can be transformed into:
683 ///
684 /// %pack = linalg.pack %in outer_dims_perm = [1, 2]
685 /// inner_dims_pos = [1, 2] inner_tiles = [8, 1] into %empty
686 /// : tensor<?x16x4xf32> -> tensor<?x2x4x8x1xf32>
687 /// %collapsed = tensor.collapse_shape %pack [[0, 1], 2, 3, 4]
688 /// : tensor<?x2x4x8x1xf32> into tensor<?x4x8x1>
689 static LogicalResult
690 bubbleUpPackOpThroughCollapseShape(tensor::CollapseShapeOp collapseOp,
691  linalg::PackOp packOp,
692  PatternRewriter &rewriter) {
693  SmallVector<int64_t> innerTileSizes = packOp.getStaticTiles();
694  ArrayRef<int64_t> innerDimsPos = packOp.getInnerDimsPos();
695  ArrayRef<int64_t> outerDimsPerm = packOp.getOuterDimsPerm();
696 
697  ArrayRef<int64_t> srcShape = collapseOp.getSrcType().getShape();
698  SmallVector<ReassociationIndices> reassocIndices =
699  collapseOp.getReassociationIndices();
700  // Project inner tile pos to the dim pos before collapsing. For example, if
701  // dims [x, y] is collapsed into [z], packing on dim z can be projected back
702  // to pack on dim y.
703  //
704  // Project to inner-most non-unit dims to increase the chance that they can be
705  // divided by the inner tile sizes. This is correct because for [..., x, 1],
706  // packing on dim 1 is equivalent to packing on dim x.
707  SmallVector<int64_t> projectedInnerDimsPos =
708  projectToInnerMostNonUnitDimsPos(innerDimsPos, reassocIndices, srcShape);
709 
710  if (!isDimsDivisibleByTileSizes(projectedInnerDimsPos, srcShape,
711  innerTileSizes)) {
712  return failure();
713  }
714  // Expand the outer dims permutation with the associated source dims for the
715  // new permutation after bubbling. This is because moving a collapsed dim is
716  // equivalent to moving the associated source dims together.
717  SmallVector<int64_t> newOuterDimsPerm;
718  for (auto outerPos : outerDimsPerm)
719  llvm::append_range(newOuterDimsPerm, reassocIndices[outerPos]);
720 
721  auto emptyOp = linalg::PackOp::createDestinationTensor(
722  rewriter, packOp.getLoc(), collapseOp.getSrc(), packOp.getMixedTiles(),
723  projectedInnerDimsPos, newOuterDimsPerm);
724  auto newPackOp = linalg::PackOp::create(
725  rewriter, packOp.getLoc(), collapseOp.getSrc(), emptyOp,
726  projectedInnerDimsPos, packOp.getMixedTiles(), packOp.getPaddingValue(),
727  newOuterDimsPerm);
728 
729  SmallVector<ReassociationIndices> newReassocIndices = reassocIndices;
730  // First apply the permutation on the reassociations of the outer dims.
731  // For example given the permutation [1, 0], the reassociations [[0, 1], [2]]
732  // -> [[0], [1, 2]]
733  int64_t nextPos =
734  applyPermutationAndReindexReassoc(newReassocIndices, outerDimsPerm);
735  // Then add direct mapping for the inner tile dims.
736  for (size_t i = 0; i < innerDimsPos.size(); ++i) {
737  newReassocIndices.push_back({nextPos});
738  nextPos += 1;
739  }
740 
741  auto newCollapseOp = tensor::CollapseShapeOp::create(
742  rewriter, collapseOp.getLoc(), packOp.getType(), newPackOp,
743  newReassocIndices);
744  rewriter.replaceOp(packOp, newCollapseOp);
745 
746  return success();
747 }

749 /// Project dimsPos to their collapsed positions in the reassocIndices.
750 ///
751 /// For example, given dimsPos [0, 1, 2, 4], and matching reassocIndices
752 /// [[0], [1, 2], [3], [4]], it returns [0, 1, 1, 3]. Because for pos 0,
753 /// the reassoc dim [0] is 0. For pos 1 and 2, the reassoc dim in pos
754 /// [1, 2] is 1. And for pos 4, the reassoc dim [4] is 3.
755 static SmallVector<int64_t>
756 projectDimsPosIntoReassocPos(ArrayRef<int64_t> dimsPos,
757  ArrayRef<ReassociationIndices> reassocIndices) {
758  SmallVector<int64_t> projectedPos;
759 
760  // Map each dimension to the position of corresponding reassociation index.
761  for (auto pos : dimsPos) {
762  for (auto [idx, indices] : llvm::enumerate(reassocIndices)) {
763  // If the dimension is present in the current indices group, the group
764  // position within the reassociation map is the desired projected
765  // dimension position.
766  if (llvm::is_contained(indices, pos)) {
767  projectedPos.push_back(idx);
768  break;
769  }
770  }
771  }
772  assert(projectedPos.size() == dimsPos.size() && "Invalid dim pos projection");
773 
774  return projectedPos;
775 }

777 /// Bubble up pack op through expand shape op.
778 ///
779 /// For example:
780 ///
781 /// %expand = tensor.expand_shape %in [[0], [1, 2]]
782 /// : tensor<?x64xf32> into tensor<?x4x16xf32>
783 /// %pack = linalg.pack %expand outer_dims_perm = [0, 1]
784 /// inner_dims_pos = [2] inner_tiles = [8] into %empty
785 /// : tensor<?x4x16xf32> -> tensor<?x4x2x8xf32>
786 ///
787 /// can be transformed into:
788 ///
789 /// %pack = linalg.pack %in outer_dims_perm = [1, 2]
790 /// inner_dims_pos = [1] inner_tiles = [8] into %empty
791 /// : tensor<?x64xf32> -> tensor<?x8x8xf32>
792 /// %expand = tensor.expand_shape %pack [[0], [1, 2], [3]]
793 /// : tensor<?x8x8xf32> into tensor<?x4x2x8xf32>
794 static LogicalResult
795 bubbleUpPackOpThroughExpandShape(tensor::ExpandShapeOp expandOp,
796  linalg::PackOp packOp,
797  PatternRewriter &rewriter) {
798  // Outer dimensions permutation is not supported currently.
799  // TODO: Handle outer_dims_perm variants.
800  ArrayRef<int64_t> outerDimsPerm = packOp.getOuterDimsPerm();
801  if (!outerDimsPerm.empty() && !isIdentityPermutation(outerDimsPerm)) {
802  return rewriter.notifyMatchFailure(packOp,
803  "non-identity outer dims perm NYI");
804  }
805 
806  // Validate dimensions' relations between shape expansion and packing.
807  SmallVector<ReassociationIndices, 4> reassoc =
808  expandOp.getReassociationIndices();
809  ArrayRef<int64_t> packInnerDims = packOp.getInnerDimsPos();
810  llvm::SetVector<int64_t> packDimsPos(llvm::from_range, packInnerDims);
811 
812  for (auto [idx, indices] : llvm::enumerate(reassoc)) {
813  // For each expand_shape reassociation, figure out which dimensions get
814  // packed if any.
815  llvm::SetVector<int64_t> expandDimPos(llvm::from_range, indices);
816  llvm::SetVector<int64_t> packedDims =
817  llvm::set_intersection(packDimsPos, expandDimPos);
818 
819  // The expanded dimension is not packed so, it does not affect moving pack
820  // before shape expansion - simply continue.
821  if (packedDims.empty())
822  continue;
823  // Shape expansion cannot be propagated when multiple expanded dimension are
824  // packed - in this case operation reordering would affect final element
825  // positions and/or shapes can no longer be projected.
826  if (packedDims.size() != 1)
827  return rewriter.notifyMatchFailure(
828  packOp, "only one of the expanded dimensions can be packed");
829  // Only the inner-most expanded dimension should be packed. Otherwise,
830  // elements order will be affected after operation reordering.
831  if (packedDims.front() != indices.back())
832  return rewriter.notifyMatchFailure(
833  packOp, "can only pack the inner-most expanded dimension");
834  }
835 
836  // Project pack.inner_dims_pos to positions before shape expansion.
837  SmallVector<int64_t> projectedInnerDimsPos =
838  projectDimsPosIntoReassocPos(packInnerDims, reassoc);
839 
840  // Project the shape expansion to new packed shape.
841  // The pack.outer_dims_perm is restricted to identity so, the permutation can
842  // be omitted for simplicity.
843  // TODO: Account for outer dimensions permutation.
844  //
845  // If reassociation is not possible, then reordering cannot happen.
846  // This can be caused by pack padding affecting previously expanded
847  // dimensions or packing extending dimensions.
848  RankedTensorType newPackType = linalg::PackOp::inferPackedType(
849  expandOp.getSrcType(), packOp.getStaticInnerTiles(),
850  projectedInnerDimsPos, /*outerDimsPerm=*/SmallVector<int64_t>{});
851  auto reassocExpand =
852  getReassociationIndicesForReshape(newPackType, packOp.getDestType());
853  if (!reassocExpand)
854  return rewriter.notifyMatchFailure(
855  packOp, "could not reassociate dims after bubbling up");
856 
857  Value destTensor = linalg::PackOp::createDestinationTensor(
858  rewriter, packOp.getLoc(), expandOp.getSrc(), packOp.getMixedTiles(),
859  projectedInnerDimsPos, /*outerDimsPerm=*/SmallVector<int64_t>{});
860  Value packedVal = linalg::PackOp::create(
861  rewriter, packOp.getLoc(), expandOp.getSrc(), destTensor,
862  projectedInnerDimsPos, packOp.getMixedTiles(), packOp.getPaddingValue(),
863  /*outerDimsPerm=*/SmallVector<int64_t>{});
864 
865  Value newExpandOp = tensor::ExpandShapeOp::create(rewriter, packOp.getLoc(),
866  packOp.getDestType(),
867  packedVal, *reassocExpand);
868  rewriter.replaceOp(packOp, newExpandOp);
869 
870  return success();
871 }

1046 /// The new generic op works on packed domain; pack ops are created for input
1047 /// and output operands. A linalg.unpack op is inserted right after the packed
1048 /// generic. E.g.
1049 ///
1050 /// #map = affine_map<(d0, d1, d2, d3) -> (d0, d1, d2, d3)>
1051 ///
1052 /// %arg0 = tensor<12x2x56x56x32xf32> // packed arg.
1053 ///
1054 /// %0 = tensor.empty() : tensor<12x56x56x64xf32>
1055 /// %1 = linalg.unpack %arg0 outer_dims_perm = [0, 3, 1, 2]
1056 /// inner_dims_pos = [3] inner_tiles = [32] into %0
1057 /// %2 = linalg.generic {indexing_maps = [#map],
1058 /// iterator_types = ["parallel", "parallel", "parallel", "parallel"]}
1059 /// outs(%1 : tensor<12x56x56x64xf32>) {
1060 /// ^bb0(%out : f32):
1061 /// linalg.yield %out : f32
1062 /// } -> tensor<12x56x56x64xf32>
1063 ///
1064 /// will be converted to
1065 ///
1066 /// #map = affine_map<(d0, d1, d2, d3, d4) -> (d0, d1, d2, d3, d4)>
1067 ///
1068 /// %0 = tensor.empty() : tensor<12x56x56x64xf32>
1069 /// %1 = linalg.generic {indexing_maps = [#map],
1070 /// iterator_types = ["parallel", "parallel", "parallel",
1071 /// "parallel", "parallel"]}
1072 /// outs(%arg0 : tensor<12x2x56x56x32xf32>) {
1073 /// ^bb0(%out : f32):
1074 /// linalg.yield %out : f32
1075 /// } -> tensor<12x2x56x56x32xf32>
1076 /// %2 = linalg.unpack %1 outer_dims_perm = [0, 3, 1, 2]
1077 /// inner_dims_pos = [3] inner_tiles = [32] into %0
1078 ///
1079 static FailureOr<std::tuple<GenericOp, Value>>
1080 pushDownUnPackOpThroughGenericOp(RewriterBase &rewriter, GenericOp genericOp,
1081  ControlPropagationFn controlFn) {
1082  if (genericOp.getNumResults() != 1)
1083  return failure();
1084 
1085  if (hasGatherSemantics(genericOp))
1086  return failure();
1087 
1088  // Collect the unPacked operand, if present.
1089  auto maybeUnPackedOperand = getUnPackedOperand(genericOp);
1090  if (failed(maybeUnPackedOperand))
1091  return failure();
1092  OpOperand *unPackedOperand = *(maybeUnPackedOperand);
1093 
1094  // Extract packing information.
1095  linalg::UnPackOp producerUnPackOp =
1096  unPackedOperand->get().getDefiningOp<linalg::UnPackOp>();
1097  assert(producerUnPackOp && "expect a valid UnPackOp");
1098 
1099  if (!controlFn(unPackedOperand))
1100  return failure();
1101 
1102  auto packInfo =
1103  getPackingInfoFromOperand(unPackedOperand, genericOp, producerUnPackOp);
1104  if (failed(packInfo))
1105  return failure();
1106 
1107  // Rebuild the indexing map for the corresponding init operand.
1108  auto [packedOutOperand, packedOutIndexingMap] =
1109  getOrCreatePackedViewOfOperand(rewriter, genericOp.getLoc(), *packInfo,
1110  genericOp, genericOp.getDpsInitOperand(0));
1111  auto destPack = packedOutOperand.getDefiningOp<linalg::PackOp>();
1112 
1113  // Forward the new tensor.empty as a destination if it is one of the following
1114  // situations:
1115  // 1) The dps init operand is a tensor.empty.
1116  // 2) The dps init is a write-only operand, i.e., it is not used in the
1117  // genericOp
1118  Value dest = packedOutOperand;
1119  auto initTensor =
1120  genericOp.getDpsInitOperand(0)->get().getDefiningOp<tensor::EmptyOp>();
1121  if (initTensor || isGenericOutsNotUsed(genericOp)) {
1122  if (destPack)
1123  dest = destPack.getDest();
1124  }
1125 
1126  // Pack the genericOp.
1127  // pack(unpack) is foldable in this case. This is because in pushing down the
1128  // unpack, by default we will populate an additional pack op after the unpack.
1129  // This guarantees them to be foldable.
1130  GenericOp newGenericOp =
1131  packGenericOp(rewriter, genericOp, dest, packedOutIndexingMap, *packInfo,
1132  /*isFoldableUnpackPack=*/true);
1133  Value newResult =
1134  newGenericOp.getTiedOpResult(newGenericOp.getDpsInitOperand(0));
1135 
1136  // If the output is unaffected, no need to unpack.
1137  if (!destPack)
1138  return std::make_tuple(newGenericOp, newResult);
1139 
1140  auto mixedTiles = destPack.getMixedTiles();
1141  auto innerDimsPos = destPack.getInnerDimsPos();
1142  auto outerDimsPerm = destPack.getOuterDimsPerm();
1143 
1144  // Insert an unPackOp right after the packed generic.
1145  Value unPackOpRes =
1146  linalg::UnPackOp::create(rewriter, genericOp.getLoc(), newResult,
1147  destPack.getSource(), innerDimsPos, mixedTiles,
1148  outerDimsPerm)
1149  .getResult();
1150 
1151  return std::make_tuple(newGenericOp, unPackOpRes);
1152 }

1176 /// add as many zero padding dimensions in `high` and `low` based on the number
1177 /// of point loops.

1183  PatternRewriter &rewriter) const override {

1185  padOp.getSource().getDefiningOp<linalg::UnPackOp>();

1187  return failure();
1188 
1189  if (!controlFn(&padOp.getSourceMutable()))
1190  return failure();

1193  // Bail out if one of the padded dimension is a tiled one.
1194  llvm::SmallBitVector paddedDims = padOp.getPaddedDims();
1195  ArrayRef<int64_t> innerDimsPos = unpackOp.getInnerDimsPos();
1196  llvm::SmallBitVector innerDims(paddedDims.size());

1207  ArrayRef<int64_t> outerDimsPerm = unpackOp.getOuterDimsPerm();

1212  applyPermutationToVector<OpFoldResult>(highPad, outerDimsPerm);
1213  }

1215  size_t pointLoopsSize = innerDimsPos.size();

1217  highPad.append(pointLoopsSize, rewriter.getIndexAttr(0));

1220  unpackOp.getSource(), lowPad, highPad,
1221  paddingVal, padOp.getNofold());

1224  Value outputUnPack =
1225  tensor::EmptyOp::create(rewriter, loc, padOp.getResultType().getShape(),
1226  padOp.getResultType().getElementType());

1229  rewriter, loc, newPadOp.getResult(), outputUnPack, innerDimsPos,
1230  unpackOp.getMixedTiles(), outerDimsPerm);
1231  rewriter.replaceOp(padOp, replacement);
1232  return success();
