41 struct MemRefPointerLikeModel
42  : public PointerLikeType::ExternalModel<MemRefPointerLikeModel,
43  MemRefType> {
44  Type getElementType(Type pointer) const {
45  return cast<MemRefType>(pointer).getElementType();
46  }
47  mlir::acc::VariableTypeCategory
48  getPointeeTypeCategory(Type pointer, TypedValue<PointerLikeType> varPtr,
49  Type varType) const {
50  if (auto mappableTy = dyn_cast<MappableType>(varType)) {
51  return mappableTy.getTypeCategory(varPtr);
52  }
53  auto memrefTy = cast<MemRefType>(pointer);
54  if (!memrefTy.hasRank()) {
55  // This memref is unranked - aka it could have any rank, including a
56  // rank of 0 which could mean scalar. For now, return uncategorized.
57  return mlir::acc::VariableTypeCategory::uncategorized;
58  }
59 
60  if (memrefTy.getRank() == 0) {
61  if (isScalarLikeType(memrefTy.getElementType())) {
62  return mlir::acc::VariableTypeCategory::scalar;
63  }
64  // Zero-rank non-scalar - need further analysis to determine the type
65  // category. For now, return uncategorized.
66  return mlir::acc::VariableTypeCategory::uncategorized;
67  }
68 
69  // It has a rank - must be an array.
70  assert(memrefTy.getRank() > 0 && "rank expected to be positive");
71  return mlir::acc::VariableTypeCategory::array;
72  }
73 };

102 /// Helper function for any of the times we need to add operands that are
103 /// affected by a device type list. Returns a new ArrayAttr with all of the
104 /// existingDeviceTypes, plus the effective new ones (or an added none, if the
105 /// new list is empty). Additionally, adds the arguments to the argCollection
106 /// the correct number of times. This will also update a 'segments' array, even
107 /// if it won't be used.
108 mlir::ArrayAttr addDeviceTypeAffectedOperandHelper(
109  MLIRContext *context, mlir::ArrayAttr existingDeviceTypes,
110  llvm::ArrayRef<acc::DeviceType> newDeviceTypes, mlir::ValueRange arguments,
111  mlir::MutableOperandRange argCollection,
112  llvm::SmallVector<int32_t> &segments) {
113  llvm::SmallVector<mlir::Attribute> deviceTypes;
114  if (existingDeviceTypes)
115  llvm::copy(existingDeviceTypes, std::back_inserter(deviceTypes));
116 
117  if (newDeviceTypes.empty()) {
118  argCollection.append(arguments);
119  segments.push_back(arguments.size());
120  deviceTypes.push_back(
121  acc::DeviceTypeAttr::get(context, acc::DeviceType::None));
122  }
123 
124  for (DeviceType DT : newDeviceTypes) {
125  argCollection.append(arguments);
126  segments.push_back(arguments.size());
127  deviceTypes.push_back(acc::DeviceTypeAttr::get(context, DT));
128  }
129 
130  return mlir::ArrayAttr::get(context, deviceTypes);
131 }

218 static mlir::Operation::operand_range
219 getValuesFromSegments(std::optional<mlir::ArrayAttr> arrayAttr,
220  mlir::Operation::operand_range range,
221  std::optional<llvm::ArrayRef<int32_t>> segments,
222  mlir::acc::DeviceType deviceType) {
223  if (!arrayAttr)
224  return range.take_front(0);
225  if (auto pos = findSegment(*arrayAttr, deviceType)) {
226  int32_t nbOperandsBefore = 0;
227  for (unsigned i = 0; i < *pos; ++i)
228  nbOperandsBefore += (*segments)[i];
229  return range.drop_front(nbOperandsBefore).take_front((*segments)[*pos]);
230  }
231  return range.take_front(0);
232 }

271 static LogicalResult checkWaitAndAsyncConflict(Op op) {
272  for (uint32_t dtypeInt = 0; dtypeInt != acc::getMaxEnumValForDeviceType();
273  ++dtypeInt) {
274  auto dtype = static_cast<acc::DeviceType>(dtypeInt);
275 
276  // The asyncOnly attribute represent the async clause without value.
277  // Therefore the attribute and operand cannot appear at the same time.
278  if (hasDeviceType(op.getAsyncOperandsDeviceType(), dtype) &&
279  op.hasAsyncOnly(dtype))
280  return op.emitError(
281  "asyncOnly attribute cannot appear with asyncOperand");
282 
283  // The wait attribute represent the wait clause without values. Therefore
284  // the attribute and operands cannot appear at the same time.
285  if (hasDeviceType(op.getWaitOperandsDeviceType(), dtype) &&
286  op.hasWaitOnly(dtype))
287  return op.emitError("wait attribute cannot appear with waitOperands");
288  }
289  return success();
290 }

292 template <typename Op>
293 static LogicalResult checkVarAndVarType(Op op) {
294  if (!op.getVar())
295  return op.emitError("must have var operand");
296 
297  // A variable must have a type that is either pointer-like or mappable.
298  if (!mlir::isa<mlir::acc::PointerLikeType>(op.getVar().getType()) &&
299  !mlir::isa<mlir::acc::MappableType>(op.getVar().getType()))
300  return op.emitError("var must be mappable or pointer-like");
301 
302  // When it is a pointer-like type, the varType must capture the target type.
303  if (mlir::isa<mlir::acc::PointerLikeType>(op.getVar().getType()) &&
304  op.getVarType() == op.getVar().getType())
305  return op.emitError("varType must capture the element type of var");
306 
307  return success();
308 }

394 static ParseResult parseVarPtrType(mlir::OpAsmParser &parser,
395  mlir::Type &varPtrType,
396  mlir::TypeAttr &varTypeAttr) {
397  if (failed(parser.parseType(varPtrType)))
398  return failure();
399  if (failed(parser.parseRParen()))
400  return failure();
401 
402  if (succeeded(parser.parseOptionalKeyword("varType"))) {
403  if (failed(parser.parseLParen()))
404  return failure();
405  mlir::Type varType;
406  if (failed(parser.parseType(varType)))
407  return failure();
408  varTypeAttr = mlir::TypeAttr::get(varType);
409  if (failed(parser.parseRParen()))
410  return failure();
411  } else {
412  // Set `varType` from the element type of the type of `varPtr`.
413  if (mlir::isa<mlir::acc::PointerLikeType>(varPtrType))
414  varTypeAttr = mlir::TypeAttr::get(
415  mlir::cast<mlir::acc::PointerLikeType>(varPtrType).getElementType());
416  else
417  varTypeAttr = mlir::TypeAttr::get(varPtrType);
418  }
419 
420  return success();
421 }

832 /// Pattern to remove operation without region that have constant false `ifCond`
833 /// and remove the condition from the operation if the `ifCond` is a true
834 /// constant.
835 template <typename OpTy>
836 struct RemoveConstantIfCondition : public OpRewritePattern<OpTy> {
837  using OpRewritePattern<OpTy>::OpRewritePattern;
838 
839  LogicalResult matchAndRewrite(OpTy op,
840  PatternRewriter &rewriter) const override {
841  // Early return if there is no condition.
842  Value ifCond = op.getIfCond();
843  if (!ifCond)
844  return failure();
845 
846  IntegerAttr constAttr;
847  if (!matchPattern(ifCond, m_Constant(&constAttr)))
848  return failure();
849  if (constAttr.getInt())
850  rewriter.modifyOpInPlace(op, [&]() { op.getIfCondMutable().erase(0); });
851  else
852  rewriter.eraseOp(op);
853 
854  return success();
855  }
856 };

858 /// Replaces the given op with the contents of the given single-block region,
859 /// using the operands of the block terminator to replace operation results.
860 static void replaceOpWithRegion(PatternRewriter &rewriter, Operation *op,
861  Region &region, ValueRange blockArgs = {}) {
862  assert(region.hasOneBlock() && "expected single-block region");
863  Block *block = &region.front();
864  Operation *terminator = block->getTerminator();
865  ValueRange results = terminator->getOperands();
866  rewriter.inlineBlockBefore(block, op, blockArgs);
867  rewriter.replaceOp(op, results);
868  rewriter.eraseOp(terminator);
869 }

1057 template <typename Op>
1058 static LogicalResult
1059 checkSymOperandList(Operation *op, std::optional<mlir::ArrayAttr> attributes,
1060  mlir::OperandRange operands, llvm::StringRef operandName,
1061  llvm::StringRef symbolName, bool checkOperandType = true) {
1062  if (!operands.empty()) {
1063  if (!attributes || attributes->size() != operands.size())
1064  return op->emitOpError()
1065  << "expected as many " << symbolName << " symbol reference as "
1066  << operandName << " operands";
1067  } else {
1068  if (attributes)
1069  return op->emitOpError()
1070  << "unexpected " << symbolName << " symbol reference";
1071  return success();
1072  }
1073 
1074  llvm::DenseSet<Value> set;
1075  for (auto args : llvm::zip(operands, *attributes)) {
1076  mlir::Value operand = std::get<0>(args);
1077 
1078  if (!set.insert(operand).second)
1079  return op->emitOpError()
1080  << operandName << " operand appears more than once";
1081 
1082  mlir::Type varType = operand.getType();
1083  auto symbolRef = llvm::cast<SymbolRefAttr>(std::get<1>(args));
1084  auto decl = SymbolTable::lookupNearestSymbolFrom<Op>(op, symbolRef);
1085  if (!decl)
1086  return op->emitOpError()
1087  << "expected symbol reference " << symbolRef << " to point to a "
1088  << operandName << " declaration";
1089 
1090  if (checkOperandType && decl.getType() && decl.getType() != varType)
1091  return op->emitOpError() << "expected " << operandName << " (" << varType
1092  << ") to be the same type as " << operandName
1093  << " declaration (" << decl.getType() << ")";
1094  }
1095 
1096  return success();
1097 }
