38 /// Helper to convert a presumed MLIR API object to a capsule, accepting either
39 /// an explicit Capsule (which can happen when two C APIs are communicating
40 /// directly via Python) or indirectly by querying the MLIR_PYTHON_CAPI_PTR_ATTR
41 /// attribute (through which supported MLIR Python API objects export their
42 /// contained API pointer as a capsule). Throws a type error if the object is
43 /// neither. This is intended to be used from type casters, which are invoked
44 /// with a raw handle (unowned). The returned object's lifetime may not extend
45 /// beyond the apiObject handle without explicitly having its refcount increased
46 /// (i.e. on return).
47 static std::optional<nanobind::object>
48 mlirApiObjectToCapsule(nanobind::handle apiObject) {
49  if (PyCapsule_CheckExact(apiObject.ptr()))
50  return nanobind::borrow<nanobind::object>(apiObject);
51  nanobind::object api =
52  nanobind::getattr(apiObject, MLIR_PYTHON_CAPI_PTR_ATTR, nanobind::none());
53  if (api.is_none())
54  return {};
55  return api;
56 }

362 /// Provides a facility like nanobind::class_ for defining a new class in a
363 /// scope, but this allows extension of an arbitrary Python class, defining
364 /// methods on it is a similar way. Classes defined in this way are very similar
365 /// to if defined in Python in the usual way but use nanobind machinery to
366 /// do it. These are not "real" nanobind classes but pure Python classes
367 /// with no relation to a concrete C++ class.
368 ///
369 /// Derived from a discussion upstream:
370 /// https://github.com/pybind/pybind11/issues/1193
371 /// (plus a fair amount of extra curricular poking)
372 /// TODO: If this proves useful, see about including it in nanobind.
373 class pure_subclass {
374 public:
375  pure_subclass(nanobind::handle scope, const char *derivedClassName,
376  const nanobind::object &superClass) {
377  nanobind::object pyType =
378  nanobind::borrow<nanobind::object>((PyObject *)&PyType_Type);
379  nanobind::object metaclass = pyType(superClass);
380  nanobind::dict attributes;
381 
382  thisClass = metaclass(derivedClassName, nanobind::make_tuple(superClass),
383  attributes);
384  scope.attr(derivedClassName) = thisClass;
385  thisClass.attr("__module__") = scope.attr("__name__");
386  }
387 
388  template <typename Func, typename... Extra>
389  pure_subclass &def(const char *name, Func &&f, const Extra &...extra) {
390  nanobind::object cf = nanobind::cpp_function(
391  std::forward<Func>(f), nanobind::name(name), nanobind::is_method(),
392  nanobind::scope(thisClass), extra...);
393  thisClass.attr(name) = cf;
394  return *this;
395  }
396 
397  template <typename Func, typename... Extra>
398  pure_subclass &def_property_readonly(const char *name, Func &&f,
399  const Extra &...extra) {
400  nanobind::object cf = nanobind::cpp_function(
401  std::forward<Func>(f), nanobind::name(name), nanobind::is_method(),
402  nanobind::scope(thisClass), extra...);
403  auto builtinProperty =
404  nanobind::borrow<nanobind::object>((PyObject *)&PyProperty_Type);
405  thisClass.attr(name) = builtinProperty(cf);
406  return *this;
407  }
408 
409  template <typename Func, typename... Extra>
410  pure_subclass &def_staticmethod(const char *name, Func &&f,
411  const Extra &...extra) {
412  static_assert(!std::is_member_function_pointer<Func>::value,
413  "def_staticmethod(...) called with a non-static member "
414  "function pointer");
415  nanobind::object cf = nanobind::cpp_function(
416  std::forward<Func>(f),
417  nanobind::name(name), // nanobind::scope(thisClass),
418  extra...);
419  thisClass.attr(name) = cf;
420  return *this;
421  }
422 
423  template <typename Func, typename... Extra>
424  pure_subclass &def_classmethod(const char *name, Func &&f,
425  const Extra &...extra) {
426  static_assert(!std::is_member_function_pointer<Func>::value,
427  "def_classmethod(...) called with a non-static member "
428  "function pointer");
429  nanobind::object cf = nanobind::cpp_function(
430  std::forward<Func>(f),
431  nanobind::name(name), // nanobind::scope(thisClass),
432  extra...);
433  thisClass.attr(name) =
434  nanobind::borrow<nanobind::object>(PyClassMethod_New(cf.ptr()));
435  return *this;
436  }
437 
438  nanobind::object get_class() const { return thisClass; }
439 
440 protected:
441  nanobind::object superClass;
442  nanobind::object thisClass;
443 };

445 /// Creates a custom subclass of mlir.ir.Attribute, implementing a casting
446 /// constructor and type checking methods.
447 class mlir_attribute_subclass : public pure_subclass {
448 public:
449  using IsAFunctionTy = bool (*)(MlirAttribute);
450  using GetTypeIDFunctionTy = MlirTypeID (*)();
451 
452  /// Subclasses by looking up the super-class dynamically.
453  mlir_attribute_subclass(nanobind::handle scope, const char *attrClassName,
454  IsAFunctionTy isaFunction,
455  GetTypeIDFunctionTy getTypeIDFunction = nullptr)
456  : mlir_attribute_subclass(
457  scope, attrClassName, isaFunction,
458  nanobind::module_::import_(MAKE_MLIR_PYTHON_QUALNAME("ir"))
459  .attr("Attribute"),
460  getTypeIDFunction) {}
461 
462  /// Subclasses with a provided mlir.ir.Attribute super-class. This must
463  /// be used if the subclass is being defined in the same extension module
464  /// as the mlir.ir class (otherwise, it will trigger a recursive
465  /// initialization).
466  mlir_attribute_subclass(nanobind::handle scope, const char *typeClassName,
467  IsAFunctionTy isaFunction,
468  const nanobind::object &superCls,
469  GetTypeIDFunctionTy getTypeIDFunction = nullptr)
470  : pure_subclass(scope, typeClassName, superCls) {
471  // Casting constructor. Note that it is hard, if not impossible, to properly
472  // call chain to parent `__init__` in nanobind due to its special handling
473  // for init functions that don't have a fully constructed self-reference,
474  // which makes it impossible to forward it to `__init__` of a superclass.
475  // Instead, provide a custom `__new__` and call that of a superclass, which
476  // eventually calls `__init__` of the superclass. Since attribute subclasses
477  // have no additional members, we can just return the instance thus created
478  // without amending it.
479  std::string captureTypeName(
480  typeClassName); // As string in case if typeClassName is not static.
481  nanobind::object newCf = nanobind::cpp_function(
482  [superCls, isaFunction, captureTypeName](
483  nanobind::object cls, nanobind::object otherAttribute) {
484  MlirAttribute rawAttribute;
485  if (!nanobind::try_cast<MlirAttribute>(otherAttribute,
486  rawAttribute) ||
487  !isaFunction(rawAttribute)) {
488  auto origRepr =
489  nanobind::cast<std::string>(nanobind::repr(otherAttribute));
490  throw std::invalid_argument(
491  (llvm::Twine("Cannot cast attribute to ") + captureTypeName +
492  " (from " + origRepr + ")")
493  .str());
494  }
495  nanobind::object self = superCls.attr("__new__")(cls, otherAttribute);
496  return self;
497  },
498  nanobind::name("__new__"), nanobind::arg("cls"),
499  nanobind::arg("cast_from_attr"));
500  thisClass.attr("__new__") = newCf;
501 
502  // 'isinstance' method.
503  static const char kIsinstanceSig[] =
504  "def isinstance(other_attribute: " MAKE_MLIR_PYTHON_QUALNAME(
505  "ir") ".Attribute) -> bool";
506  def_staticmethod(
507  "isinstance",
508  [isaFunction](MlirAttribute other) { return isaFunction(other); },
509  nanobind::arg("other_attribute"), nanobind::sig(kIsinstanceSig));
510  def("__repr__", [superCls, captureTypeName](nanobind::object self) {
511  return nanobind::repr(superCls(self))
512  .attr("replace")(superCls.attr("__name__"), captureTypeName);
513  });
514  if (getTypeIDFunction) {
515  def_staticmethod("get_static_typeid",
516  [getTypeIDFunction]() { return getTypeIDFunction(); });
517  nanobind::module_::import_(MAKE_MLIR_PYTHON_QUALNAME("ir"))
518  .attr(MLIR_PYTHON_CAPI_TYPE_CASTER_REGISTER_ATTR)(
519  getTypeIDFunction())(nanobind::cpp_function(
520  [thisClass = thisClass](const nanobind::object &mlirAttribute) {
521  return thisClass(mlirAttribute);
522  }));
523  }
524  }
525 };

556  // which makes it impossible to forward it to `__init__` of a superclass.
557  // Instead, provide a custom `__new__` and call that of a superclass, which
558  // eventually calls `__init__` of the superclass. Since attribute subclasses
559  // have no additional members, we can just return the instance thus created
560  // without amending it.
561  std::string captureTypeName(
562  typeClassName); // As string in case if typeClassName is not static.
563  nanobind::object newCf = nanobind::cpp_function(
564  [superCls, isaFunction, captureTypeName](nanobind::object cls,
565  nanobind::object otherType) {
566  MlirType rawType;
567  if (!nanobind::try_cast<MlirType>(otherType, rawType) ||
568  !isaFunction(rawType)) {
569  auto origRepr =
570  nanobind::cast<std::string>(nanobind::repr(otherType));
571  throw std::invalid_argument((llvm::Twine("Cannot cast type to ") +
572  captureTypeName + " (from " +
573  origRepr + ")")
574  .str());
575  }
576  nanobind::object self = superCls.attr("__new__")(cls, otherType);
577  return self;
578  },
579  nanobind::name("__new__"), nanobind::arg("cls"),
580  nanobind::arg("cast_from_type"));
581  thisClass.attr("__new__") = newCf;
582 
583  // 'isinstance' method.
584  static const char kIsinstanceSig[] =
585  "def isinstance(other_type: " MAKE_MLIR_PYTHON_QUALNAME(
586  "ir") ".Type) -> bool";
587  def_staticmethod(
588  "isinstance",
589  [isaFunction](MlirType other) { return isaFunction(other); },
590  nanobind::arg("other_type"), nanobind::sig(kIsinstanceSig));
591  def("__repr__", [superCls, captureTypeName](nanobind::object self) {
592  return nanobind::cast<std::string>(
593  nanobind::repr(superCls(self))
594  .attr("replace")(superCls.attr("__name__"), captureTypeName));
595  });
596  if (getTypeIDFunction) {
597  // 'get_static_typeid' method.
598  // This is modeled as a static method instead of a static property because
599  // `def_property_readonly_static` is not available in `pure_subclass` and
600  // we do not want to introduce the complexity that pybind uses to
601  // implement it.
602  def_staticmethod("get_static_typeid",
603  [getTypeIDFunction]() { return getTypeIDFunction(); });
604  nanobind::module_::import_(MAKE_MLIR_PYTHON_QUALNAME("ir"))
605  .attr(MLIR_PYTHON_CAPI_TYPE_CASTER_REGISTER_ATTR)(
606  getTypeIDFunction())(nanobind::cpp_function(
607  [thisClass = thisClass](const nanobind::object &mlirType) {
608  return thisClass(mlirType);
609  }));
610  }
611  }
612 };
