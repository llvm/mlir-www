156 /// An interface to be implemented by dialects that can have identifiers in the
157 /// data layout specification entries. Provides hooks for verifying the entry
158 /// validity and combining two entries.
159 class DataLayoutDialectInterface
160  : public DialectInterface::Base<DataLayoutDialectInterface> {
161 public:
162  DataLayoutDialectInterface(Dialect *dialect) : Base(dialect) {}
163 
164  /// Checks whether the given data layout entry is valid and reports any errors
165  /// at the provided location. Derived classes should override this.
166  virtual LogicalResult verifyEntry(DataLayoutEntryInterface entry,
167  Location loc) const {
168  return success();
169  }
170 
171  /// Checks whether the given data layout entry is valid and reports any errors
172  /// at the provided location. Derived classes should override this.
173  virtual LogicalResult verifyEntry(TargetDeviceSpecInterface entry,
174  Location loc) const {
175  return success();
176  }
177 
178  /// Default implementation of entry combination that combines identical
179  /// entries and returns null otherwise.
180  static DataLayoutEntryInterface
181  defaultCombine(DataLayoutEntryInterface outer,
182  DataLayoutEntryInterface inner) {
183  if (!outer || outer == inner)
184  return inner;
185  return {};
186  }
187 
188  /// Combines two entries with identifiers that belong to this dialect. Returns
189  /// the combined entry or null if the entries are not compatible. Derived
190  /// classes likely need to reimplement this.
191  virtual DataLayoutEntryInterface
192  combine(DataLayoutEntryInterface outer,
193  DataLayoutEntryInterface inner) const {
194  return defaultCombine(outer, inner);
195  }
196 };

202 /// The main mechanism for performing data layout queries. Instances of this
203 /// class can be created for an operation implementing DataLayoutOpInterface.
204 /// Upon construction, a layout spec combining that of the given operation with
205 /// all its ancestors will be computed and used to handle further requests. For
206 /// efficiency, results to all requests will be cached in this object.
207 /// Therefore, if the data layout spec for the scoping operation, or any of the
208 /// enclosing operations, changes, the cache is no longer valid. The user is
209 /// responsible creating a new DataLayout object after any spec change. In debug
210 /// mode, the cache validity is being checked in every request.
211 class DataLayout {
212 public:
213  explicit DataLayout();
214  explicit DataLayout(DataLayoutOpInterface op);
215  explicit DataLayout(ModuleOp op);
216 
217  /// Returns the layout of the closest parent operation carrying layout info.
218  static DataLayout closest(Operation *op);
219 
220  /// Returns the size of the given type in the current scope.
221  llvm::TypeSize getTypeSize(Type t) const;
222 
223  /// Returns the size in bits of the given type in the current scope.
224  llvm::TypeSize getTypeSizeInBits(Type t) const;
225 
226  /// Returns the required alignment of the given type in the current scope.
227  uint64_t getTypeABIAlignment(Type t) const;
228 
229  /// Returns the preferred of the given type in the current scope.
230  uint64_t getTypePreferredAlignment(Type t) const;
231 
232  /// Returns the bitwidth that should be used when performing index
233  /// computations for the given pointer-like type in the current scope. If the
234  /// type is not a pointer-like type, it returns std::nullopt.
235  std::optional<uint64_t> getTypeIndexBitwidth(Type t) const;
236 
237  /// Returns the specified endianness.
238  Attribute getEndianness() const;
239 
240  /// Returns the default memory space used for memory operations.
241  Attribute getDefaultMemorySpace() const;
242 
243  /// Returns the memory space used for AllocaOps.
244  Attribute getAllocaMemorySpace() const;
245 
246  /// Returns the mangling mode.
247  Attribute getManglingMode() const;

255  /// Returns the natural alignment of the stack in bits. Alignment promotion of
