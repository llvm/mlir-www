40 // Check that the zero point of the tensor and padding operations are aligned.
41 bool checkMatchingPadConstAndZp(Value padConst, Value zp) {
42  // Check that padConst is a constant value and a scalar tensor
43  DenseElementsAttr padConstAttr;
44  if (!matchPattern(padConst, m_Constant(&padConstAttr)) ||
45  (padConstAttr.size() != 1)) {
46  return false;
47  }
48 
49  // Check that floating point pad is zero
50  if (auto padConstFpAttr = mlir::dyn_cast<DenseFPElementsAttr>(padConstAttr)) {
51  float padConstVal = (*padConstFpAttr.begin()).convertToFloat();
52  return padConstVal == 0.0f;
53  }
54 
55  // Check that the zp and padConst align for the integer (quantized) case
56  if (auto padConstIntAttr =
57  mlir::dyn_cast<DenseIntElementsAttr>(padConstAttr)) {
58  DenseIntElementsAttr zpAttr;
59  // Check that zp is a constant value and a scalar tensor
60  if (!matchPattern(zp, m_Constant(&zpAttr)) || (padConstAttr.size() != 1)) {
61  return false;
62  }
63 
64  // Check equality
65  int64_t zpVal = (*zpAttr.begin()).getSExtValue();
66  int64_t padConstVal = (*padConstIntAttr.begin()).getSExtValue();
67  return zpVal == padConstVal;
68  }
69 
70  // Bail-out on unsupported type
71  return false;
72 }

164 // Pattern attempts to fold a `tosa.pad` operator to a following tensor
165 // operation like `tosa.conv2d` by merging the padding associated with the
166 // pad operator directly to the implicit padding of the tensor operation.
167 // This helps eliminate the explicit padding operator if unused.
168 template <typename OpTy, typename AdaptorTy>
169 struct FoldPadToTensorOp : public OpRewritePattern<OpTy> {
170  using OpRewritePattern<OpTy>::OpRewritePattern;
171 
172  LogicalResult matchAndRewrite(OpTy tensorOp,
173  PatternRewriter &rewriter) const override {
174  // Check producer is a tosa::PadOp
175  auto padOp = tensorOp.getInput().template getDefiningOp<tosa::PadOp>();
176  if (!padOp)
177  return rewriter.notifyMatchFailure(tensorOp,
178  "Producer must be a tosa::PadOp.");
179 
180  // Validate that tensor operation has sane padding
181  const std::vector<int64_t> &tensorOpPad = tensorOp.getPad().vec();
182  if (tensorOpPad.size() != 4) // pad_top, pad_bottom, pad_left, pad_right
183  return rewriter.notifyMatchFailure(
184  tensorOp, "Tensor operation padding shall have 4 elements.");
185 
186  // Validate tosa::PadOp padding
187  DenseIntElementsAttr padOpPadding;
188  if (!matchPattern(padOp.getPadding(), m_Constant(&padOpPadding))) {
189  return rewriter.notifyMatchFailure(
190  tensorOp,
191  "The `padding` input specified on the tosa::PadOp must be constant.");
192  }
193  // N_before, N_after, H_before, H_after, W_before, W_after, C_before,
194  // C_after
195  if (padOpPadding.size() != 8)
196  return rewriter.notifyMatchFailure(tensorOp,
197  "Pad padding should have 8 elements.");
198  int64_t padNBefore = (*(padOpPadding.begin() + 0)).getLimitedValue();
199  int64_t padNAfter = (*(padOpPadding.begin() + 1)).getLimitedValue();
200  int64_t padHBefore = (*(padOpPadding.begin() + 2)).getLimitedValue();
201  int64_t padHAfter = (*(padOpPadding.begin() + 3)).getLimitedValue();
202  int64_t padWBefore = (*(padOpPadding.begin() + 4)).getLimitedValue();
203  int64_t padWAfter = (*(padOpPadding.begin() + 5)).getLimitedValue();
204  int64_t padCBefore = (*(padOpPadding.begin() + 6)).getLimitedValue();
205  int64_t padCAfter = (*(padOpPadding.begin() + 7)).getLimitedValue();
206 
207  if (padNBefore != 0 || padNAfter != 0 || padCBefore != 0 || padCAfter != 0)
208  return rewriter.notifyMatchFailure(
209  tensorOp, "Folding padding in N or C dimensions is not supported.");
210 
211  // Fold padding from Pad into the tensor operation
212  // 4 elements - pad_top, pad_bottom, pad_left, pad_right
213  SmallVector<int64_t> foldedPad(tensorOpPad.size());
214  foldedPad[0] = padHBefore + tensorOpPad[0];
215  foldedPad[1] = padHAfter + tensorOpPad[1];
216  foldedPad[2] = padWBefore + tensorOpPad[2];
217  foldedPad[3] = padWAfter + tensorOpPad[3];
218 
219  // Check kernel related restrictions
220  if (!AdaptorTy::checkKernelCompliance(tensorOp, foldedPad)) {
221  return rewriter.notifyMatchFailure(
222  tensorOp, "Padding size not aligned with kernel restrictions.");
223  }
224 
225  // Check padding constant restrictions
226  if (!AdaptorTy::checkPadConstCompliance(tensorOp, padOp.getPadConst())) {
227  return rewriter.notifyMatchFailure(
228  tensorOp,
229  "Padding constant is not aligned with operator zero-point.");
230  }
231 
232  // Check that padding doesn't grow more than 8K level (8192) for now
233  if (llvm::any_of(foldedPad, [](int64_t padVal) { return padVal > 8192; })) {
234  return rewriter.notifyMatchFailure(
235  tensorOp, "Padding size more than the 8K level limit.");
236  }
237 
238  // Create operator
239  AdaptorTy::replaceOpWithNewPad(rewriter, tensorOp, padOp.getInput1(),
240  foldedPad);
241 
242  return success();
243  }
244 };

346 struct ConsolidateTransposeOptimization
347  : public OpRewritePattern<tosa::TransposeOp> {
348  using OpRewritePattern::OpRewritePattern;
349 
350  LogicalResult matchAndRewrite(tosa::TransposeOp transposeOp,
351  PatternRewriter &rewriter) const override {
352  // Input is also TransposeOp - transpose(transpose(A)).
353  auto innerTranspose =
354  transposeOp.getInput1().getDefiningOp<tosa::TransposeOp>();
355  if (!innerTranspose)
356  return rewriter.notifyMatchFailure(transposeOp,
357  "input must be transpose operation");
358 
359  const llvm::ArrayRef<int32_t> transposePerms = transposeOp.getPerms();
360  const llvm::ArrayRef<int32_t> innerTransposePerms =
361  innerTranspose.getPerms();
362 
363  if (transposePerms.size() != innerTransposePerms.size())
364  return rewriter.notifyMatchFailure(
365  transposeOp,
366  "transpose and inner transpose perms sizes must be equal");
367  if (transposePerms.empty())
368  return rewriter.notifyMatchFailure(
369  transposeOp, "transpose perms sizes must be positive");
370 
371  // Consolidate transposes into one transpose.
372  SmallVector<int32_t> perms(transposePerms.size());
373  for (int i = 0, s = transposePerms.size(); i < s; ++i)
374  perms[i] = innerTransposePerms[transposePerms[i]];
375 
376  rewriter.replaceOpWithNewOp<tosa::TransposeOp>(
377  transposeOp, transposeOp.getResult().getType(),
378  innerTranspose.getInput1(), rewriter.getDenseI32ArrayAttr(perms));
379 
380  return success();
381  }

384 // Determines the case when tosa.transpose is a tosa.reshape operation.
385 struct TransposeIsReshape : public OpRewritePattern<tosa::TransposeOp> {
386  using OpRewritePattern::OpRewritePattern;
387 
388  LogicalResult matchAndRewrite(tosa::TransposeOp op,
389  PatternRewriter &rewriter) const override {
390  if (op.getInput1().getDefiningOp<tosa::TransposeOp>())
391  return rewriter.notifyMatchFailure(
392  op, "Src is from transpose, can compose transposes");
393 
394  Value result = op.getResult();
395  for (Operation *subop : result.getUsers()) {
396  if (isa_and_nonnull<tosa::TransposeOp>(subop))
397  return rewriter.notifyMatchFailure(
398  op, "Dest is used by transpose, can compose transposes");
399  }
400 
401  auto input = op.getInput1();
402  auto inputTy = llvm::cast<ShapedType>(input.getType());
403  if (!inputTy.hasRank())
404  return rewriter.notifyMatchFailure(op, "Unranked input.");
405 
406  int64_t numDynDims = 0;
407  for (int i = 0; i < inputTy.getRank(); ++i)
408  if (inputTy.isDynamicDim(i))
409  numDynDims++;
410 
411  if (numDynDims > 1)
412  return rewriter.notifyMatchFailure(op, "Has more than one dynamic dim.");
413 
414  const llvm::ArrayRef<int32_t> permValues = op.getPerms();
415 
416  SmallVector<int64_t> nonZeroPerms;
417  nonZeroPerms.reserve(permValues.size());
418  for (auto idx : permValues) {
419  auto sz = inputTy.getDimSize(idx);
420  if (sz != 1)
421  nonZeroPerms.push_back(idx);
422  }
423 
424  for (int i = 1, s = nonZeroPerms.size(); i < s; ++i)
425  if (nonZeroPerms[i - 1] > nonZeroPerms[i])
426  return rewriter.notifyMatchFailure(op,
427  "Transpose changes memory layout.");
428 
429  SmallVector<int64_t> newShape;
430  newShape.reserve(inputTy.getRank());
431  for (int i = 0, s = inputTy.getRank(); i < s; ++i)
432  newShape.push_back(inputTy.getDimSize(permValues[i]));
433 
434  rewriter.replaceOpWithNewOp<tosa::ReshapeOp>(
435  op, op.getType(), op.getInput1(),
436  getTosaConstShape(rewriter, op.getLoc(), newShape));
437  return success();
438  }

513 // Attempts the following transformation:
514 //
515 // For integers a, b, a', and b' such that [a, b] ∩ [a', b'] ≠ ∅ and input
516 // tensor X the following identity holds:
517 //
518 // CLAMP(CLAMP(X, a, b), a', b') = CLAMP(X, max(a, a'), min(b, b'))
519 //
520 // subject to the following valid NaN propagation semantics:
521 // --------------------------------------------
522 // | OUTER CLAMP | INNER CLAMP | RESULT MODE |
523 // |-------------|--------------|-------------|
524 // | PROPAGATE | PROPAGATE | PROPAGATE |
525 // | PROPAGATE | IGNORE | IGNORE |
526 // | IGNORE | PROPAGATE | INVALID |
527 // | IGNORE | IGNORE | IGNORE |
528 // |------------------------------------------|
529 
530 struct ClampClampOptimization : public OpRewritePattern<tosa::ClampOp> {
531  using OpRewritePattern<tosa::ClampOp>::OpRewritePattern;
532 
533  // Helper structure to describe the range of a clamp operation.
534  template <typename T>
535  struct ClampRange {
536  ClampRange(const T &start, const T &end) : start(start), end(end) {}
537  T start;
538  T end;
539 
540  // Helper function to determine if two Clamp ranges intersect.
541  bool intersects(const ClampRange<T> &otherRange) {
542  return start < otherRange.end && otherRange.start < end;
543  }
544  };
545 
546  LogicalResult matchAndRewrite(tosa::ClampOp op,
547  PatternRewriter &rewriter) const override {
548  Value input = op.getInput();
549 
550  // Check the input to the CLAMP op is itself a CLAMP.
551  auto clampOp = input.getDefiningOp<tosa::ClampOp>();
552  if (!clampOp)
553  return failure();
554 
555  // Check we have a valid NaN propagation combination.
556  const auto opNanMode = op.getNanMode();
557  const auto clampNanMode = clampOp.getNanMode();
558  if (opNanMode == "IGNORE" && clampNanMode == "PROPAGATE")
559  return failure();
560 
561  auto maxValAttr = op.getMaxValAttr();
562  auto minValAttr = op.getMinValAttr();
563  auto clampOpMaxValAttr = clampOp.getMaxValAttr();
564  auto clampOpMinValAttr = clampOp.getMinValAttr();
565 
566  auto inputEType = llvm::cast<ShapedType>(input.getType()).getElementType();
567  if (auto quantType =
568  llvm::dyn_cast<mlir::quant::UniformQuantizedType>(inputEType)) {
569  inputEType = quantType.getStorageType();
570  }
571 
572  Attribute newMinValAttr, newMaxValAttr;
573  if (mlir::isa<FloatType>(inputEType)) {
574  auto floatMaxValAttr = cast<mlir::FloatAttr>(maxValAttr);
575  auto floatMinValAttr = cast<mlir::FloatAttr>(minValAttr);
576  auto clampOpFloatMaxValAttr = cast<mlir::FloatAttr>(clampOpMaxValAttr);
577  auto clampOpFloatMinValAttr = cast<mlir::FloatAttr>(clampOpMinValAttr);
578 
579  // Check we have intersecting ranges.
580  const auto opMinFloat = floatMinValAttr.getValue();
581  const auto opMaxFloat = floatMaxValAttr.getValue();
582  const auto clampOpMinFloat = clampOpFloatMinValAttr.getValue();
583  const auto clampOpMaxFloat = clampOpFloatMaxValAttr.getValue();
584  ClampRange<APFloat> opRangeFloatRange(opMinFloat, opMaxFloat);
585  ClampRange<APFloat> clampRangeFloatRange(clampOpMinFloat,
586  clampOpMaxFloat);
587  if (!opRangeFloatRange.intersects(clampRangeFloatRange))

722  using OpRewritePattern<tosa::SliceOp>::OpRewritePattern;
723 
724  LogicalResult matchAndRewrite(tosa::SliceOp sliceOp,
725  PatternRewriter &rewriter) const override {
726  Value sliceInput = sliceOp.getInput1();
727 
728  // Check if producer is a PadOp
729  auto padOp = sliceInput.getDefiningOp<tosa::PadOp>();
730  if (!padOp)
731  return rewriter.notifyMatchFailure(sliceOp,
732  "slice input must be a pad operation");
733 
734  // Check PadOp has a single consumer
735  if (!padOp->hasOneUse())
736  return rewriter.notifyMatchFailure(sliceOp,
737  "pad shall have a single consumer");
738 
739  // Check input is statically ranked
740  auto inputTy = dyn_cast<RankedTensorType>(padOp.getInput1().getType());
741  auto padTy = dyn_cast<RankedTensorType>(padOp.getType());
742  if (!inputTy || !padTy || !inputTy.hasRank())
743  return rewriter.notifyMatchFailure(sliceOp,
744  "slice input must be a ranked tensor");
745 
746  // Validate and extract tosa::PadOp padding
747  DenseIntElementsAttr paddingElems;
748  if (!matchPattern(padOp.getPadding(), m_Constant(&paddingElems))) {
749  return rewriter.notifyMatchFailure(
750  sliceOp,
751  "`padding` input specified on the tosa::PadOp must be constant.");
752  }
753  llvm::SmallVector<int64_t> padPaddings =
754  llvm::to_vector(paddingElems.getValues<int64_t>());
755 
756  // Extract slice parameters
757  DenseElementsAttr startElems;
758  if (!matchPattern(sliceOp.getStart(), m_Constant(&startElems)))
759  return rewriter.notifyMatchFailure(
760  sliceOp, "start of slice must be a static ranked shape");
761  llvm::SmallVector<int64_t> sliceStarts =
762  llvm::to_vector(startElems.getValues<int64_t>());
763 
764  DenseElementsAttr sizeElems;
765  if (!matchPattern(sliceOp.getSize(), m_Constant(&sizeElems)))
766  return rewriter.notifyMatchFailure(
767  sliceOp, "size of slice must be a static ranked shape");
768  llvm::SmallVector<int64_t> sliceSizes =
769  llvm::to_vector(sizeElems.getValues<int64_t>());
770 
771  // Check if dynamic dimensions are sliced
772  const int64_t rank = inputTy.getRank();
773  if (llvm::any_of(llvm::seq<int64_t>(0, rank), [&](int64_t i) {
774  const bool isDimDynamic = inputTy.isDynamicDim(i);
775  const bool isDimSliced =
776  (sliceStarts[i] != 0) || (sliceSizes[i] != -1);
777 
778  return isDimDynamic && isDimSliced;
779  })) {
780  return rewriter.notifyMatchFailure(
781  sliceOp, "axis that are sliced shall be statically known.");
782  }
783 
784  // Update the parameters
785  llvm::SmallVector<int64_t> newSliceStarts(rank, 0);
786  llvm::SmallVector<int64_t> newPadPaddings(2 * rank, 0);
787  llvm::SmallVector<int64_t> newPadShape(rank, ShapedType::kDynamic);
788  bool updated = false;
789 
790  for (int64_t i = 0; i < rank; ++i) {
791  const int64_t padLo = padPaddings[i * 2];
792  const int64_t padHi = padPaddings[i * 2 + 1];
793  const int64_t sliceStart = sliceStarts[i];
794  const int64_t sliceSize = sliceSizes[i];
795  const int64_t sliceEnd = sliceStart + sliceSize;
796 
797  // If dimension is dynamic pass-through
798  if (inputTy.isDynamicDim(i)) {
799  newPadPaddings[i * 2] = padLo;
800  newPadPaddings[i * 2 + 1] = padHi;
801  newSliceStarts[i] = sliceStart;
802  continue;
803  }
804 
805  // Handle static dimensions
806  const int64_t dimSize = inputTy.getShape()[i];
807  const int64_t dimTotal = padLo + dimSize + padHi;
808 
809  // Check slice within bounds
810  if (sliceStart < 0 || sliceEnd > dimTotal)
811  return rewriter.notifyMatchFailure(sliceOp, "slice is out-of-bounds");
812 
813  // Compute updated slice start parameter
814  const int64_t newSliceStart = std::max<int64_t>(sliceStart - padLo, 0);
815  newSliceStarts[i] = newSliceStart;
816  updated |= newSliceStart != sliceStart;
817 
818  // Compute updated pad parameters
819  const int64_t newPadLo = std::max<int64_t>(padLo - sliceStart, 0);
820  const int64_t newPadHi =
821  std::max<int64_t>(sliceEnd - (padLo + dimSize), 0);
822  newPadPaddings[i * 2] = newPadLo;
823  newPadPaddings[i * 2 + 1] = newPadHi;
824  updated |= (newPadLo != padLo) || (newPadHi != padHi);
825 
826  // Calculate new pad output shape
827  newPadShape[i] =
828  newPadPaddings[i * 2] + dimSize + newPadPaddings[i * 2 + 1];
829  }
830 
831  // Check that we actually need to proceed with the rewrite
832  if (!updated)
833  return rewriter.notifyMatchFailure(
834  sliceOp, "terminate condition; nothing to rewrite");
835 
836  // Create a PadOp with updated padding
837  auto newPaddingsOp =
838  getTosaConstShape(rewriter, sliceOp.getLoc(), newPadPaddings);
839  auto newPadTy =
840  RankedTensorType::get(newPadShape, inputTy.getElementType());
841  auto newPadOp = tosa::PadOp::create(rewriter, padOp.getLoc(), newPadTy,
842  padOp.getInput1(), newPaddingsOp,
843  padOp.getPadConst());
844 
845  // Update SliceOp and point to new PadOp
846  auto newStartOp =
847  getTosaConstShape(rewriter, sliceOp.getLoc(), newSliceStarts);
848  rewriter.replaceOpWithNewOp<tosa::SliceOp>(sliceOp, sliceOp.getType(),
849  newPadOp.getResult(), newStartOp,
850  sliceOp.getSize());
851 
852  return success();
853  }
