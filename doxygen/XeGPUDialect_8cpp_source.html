36 // Checks if the given shape can be evenly distributed based on the layout
37 // and data factors provided by the LayoutAttr.
38 bool XeGPUDialect::isEvenlyDistributable(llvm::ArrayRef<int64_t> shape,
39  xegpu::LayoutAttr attr) {
40  assert(attr && "Layout attribute is missing.");
41 
42  // Checks whether the given shape can be evenly distributed using the
43  // specified layout and data attributes. If successful, it returns the work
44  // size for each compute unit; otherwise, it returns `std::nullopt`. The work
45  // size per compute unit is calculated as follows:
46  // - If `data` is null: newShape[i] = shape[i] / layout[i]
47  // - If `data` is not null: newShape[i] = data[i]
48  // When round-robin distribution (`rr`) is enabled, `shape[i]` can be
49  // smaller than `layout[i] * data[i]`, allowing multiple compute units to
50  // share the data.
51  auto tryDistribute = [&](llvm::ArrayRef<int64_t> shape,
52  DenseI32ArrayAttr layout, DenseI32ArrayAttr data,
53  bool rr = true) -> optional<SmallVector<int64_t>> {
54  llvm::SmallVector<int64_t> newShape(shape);
55  if (layout) {
56  auto vec = llvm::to_vector_of<int64_t>(layout.asArrayRef());
57  if (vec.size() != shape.size())
58  return std::nullopt;
59  auto ratio = computeShapeRatio(shape, vec);
60  if (!ratio.has_value())
61  return std::nullopt;
62  newShape = ratio.value();
63  }
64 
65  if (data) {
66  auto vec = llvm::to_vector_of<int64_t>(data.asArrayRef());
67  if (vec.size() != shape.size())
68  return std::nullopt;
69  auto ratio = computeShapeRatio(newShape, vec);
70  if (!ratio.has_value() && rr)
71  ratio = computeShapeRatio(vec, newShape);
72  if (!ratio.has_value())
73  return std::nullopt;
74 
75  // if data is not null, we always return it for next phase.
76  newShape = vec;
77  }
78  return newShape;
79  };
80 
81  // check the sgLayout and sgData
82  auto maybeSgShape =
83  tryDistribute(shape, attr.getSgLayout(), attr.getSgData());
84  if (!maybeSgShape)
85  return false;
86  auto sgShape = maybeSgShape.value();
87 
88  // check InstData, it neither have layout nor need round-robin
89  auto maybeInstShape =
90  tryDistribute(sgShape, nullptr, attr.getInstData(), false);
91  if (!maybeInstShape)
92  return false;
93  auto instShape = maybeInstShape.value();
94 
95  // check LaneLayout and LaneData
96  auto maybeLaneShape =
97  tryDistribute(instShape, attr.getLaneLayout(), attr.getLaneData(), false);
98  return maybeLaneShape.has_value();
99 }

145 LogicalResult
146 LayoutAttr::verify(llvm::function_ref<mlir::InFlightDiagnostic()> emitError,
147  DenseI32ArrayAttr sg_layout, DenseI32ArrayAttr sg_data,
148  DenseI32ArrayAttr inst_data, DenseI32ArrayAttr lane_layout,
149  DenseI32ArrayAttr lane_data, DenseI32ArrayAttr order) {
150 
151  // A valid layout must include at least one of sg_layout and lane_layout.
152  // sg_layout is essential for Workgroup layout, while lane_layout is
153  // required for Subgroup layout.
154  if (!sg_layout && !inst_data && !lane_layout) {
155  return emitError()
156  << "expected at least one of sg_layout, inst_data or lane_layout";
157  }
158 
159  // generate code to check sg_laout, inst_data and lane_layout having the same
160  // rank if they are not null.
161 
162  if (sg_layout && inst_data && sg_layout.size() != inst_data.size()) {
163  return emitError()
164  << "expected sg_layout and inst_data to have the same rank";
165  }
166 
167  if (sg_layout && lane_layout && sg_layout.size() != lane_layout.size()) {
168  return emitError()
169  << "expected sg_layout and lane_layout to have the same rank";
170  }
171 
172  if (inst_data && lane_layout && inst_data.size() != lane_layout.size()) {
173  return emitError()
174  << "expected inst_data and lane_layout to have the same rank";
175  }
176 
177  // sg_data is optional for Workgroup layout, but its presence requires
178  // sg_layout.
179  if (sg_data) {
180  if (!sg_layout)
181  return emitError() << "expected sg_layout being used with sg_data";
182  if (sg_data.size() != sg_layout.size())
183  return emitError()
184  << "expected sg_data and sg_layout to have the same rank";
185  }
186 
187  // lane_data is optional for Subgroup layout, but its presence requires
188  // lane_layout.
189  if (lane_data) {
190  if (!lane_layout)
191  return emitError() << "expected lane_layout being used with lane_data";
192  if (lane_data.size() != lane_layout.size())
193  return emitError()
194  << "expected lane_data and lane_layout to have the same rank";
195  }
196 
197  if (order) {
198  if (!sg_layout && !lane_layout)
199  return emitError()
200  << "expected sg_layout/lane_layout being used with order";
201 
202  if (sg_layout && order.size() != sg_layout.size())
203  return emitError()
204  << "expected order and sg_layout to have the same rank";
205 
206  if (lane_layout && order.size() != lane_layout.size())
207  return emitError()
208  << "expected order and lane_layout to have the same rank";
209  }
210 
211  return success();
212 }

233 mlir::Type TensorDescType::parse(::mlir::AsmParser &parser) {
234  llvm::SmallVector<int64_t> shape;
235  mlir::Type elementType;
236  mlir::FailureOr<mlir::Attribute> encoding;
237  mlir::FailureOr<mlir::Attribute> layout;
238 
239  // Parse literal '<'
240  if (parser.parseLess())
241  return {};
242 
243  auto shapeLoc = parser.getCurrentLocation();
244  if (mlir::failed(parser.parseDimensionList(shape))) {
245  parser.emitError(shapeLoc, "failed to parse parameter 'shape'");
246  return {};
247  }
248 
249  auto elemTypeLoc = parser.getCurrentLocation();
250  if (mlir::failed(parser.parseType(elementType))) {
251  parser.emitError(elemTypeLoc, "failed to parse parameter 'elementType'");
252  return {};
253  }
254 
255  // parse optional attributes
256  while (mlir::succeeded(parser.parseOptionalComma())) {
257  mlir::Attribute attr;
258  ParseResult res = parser.parseAttribute(attr);
259  if (mlir::succeeded(res)) {
260  if (mlir::isa<LayoutAttr>(attr)) {
261  layout = attr;
262  continue;
263  }
264  if (mlir::isa<BlockTensorDescAttr, ScatterTensorDescAttr>(attr)) {
265  encoding = attr;
266  continue;
267  }
268  }
269  return {};
270  }
271 
272  // Parse literal '>'
273  if (parser.parseGreater())
274  return {};
275 
276  MLIRContext *ctxt = parser.getContext();
277  return TensorDescType::getChecked(
278  [&]() { return parser.emitError(parser.getNameLoc()); }, ctxt, shape,
279  elementType, encoding.value_or(BlockTensorDescAttr::get(ctxt)),
280  layout.value_or(mlir::Attribute()));
281 }

283 void TensorDescType::print(::mlir::AsmPrinter &printer) const {
284  printer << "<";
285 
286  auto shape = getShape();
287  for (int64_t dim : shape) {
288  if (mlir::ShapedType::isDynamic(dim))
289  printer << '?';
290  else
291  printer << dim;
292  printer << 'x';
293  }
294 
295  printer << getElementType();
296 
297  auto encoding = getEncoding();
298  auto blockAttr = llvm::dyn_cast_if_present<BlockTensorDescAttr>(encoding);
299  if (encoding && (!blockAttr || !blockAttr.hasDefaultsOnly()))
300  printer << ", " << encoding;
301 
302  if (auto layout = getLayout())
303  printer << ", " << layout;
304 
305  printer << ">";
306 }

328 LogicalResult TensorDescType::verify(
329  llvm::function_ref<::mlir::InFlightDiagnostic()> emitError,
330  llvm::ArrayRef<int64_t> shape, mlir::Type elementType,
331  mlir::Attribute encoding, mlir::Attribute layout) {
332  size_t rank = shape.size();
333 
334  if (rank == 0)
335  return emitError() << "expected non-zero rank tensor";
336 
337  auto blockAttr = mlir::dyn_cast_if_present<BlockTensorDescAttr>(encoding);
338  if (blockAttr) {
339  MemorySpaceAttr memorySpaceAttr = blockAttr.getMemorySpace();
340  if (rank > 1 && memorySpaceAttr &&
341  memorySpaceAttr.getValue() == MemorySpace::SLM)
342  return emitError() << "SLM is only supported for 1D block tensor";
343  }
344 
345  // for gather and scatter ops, Low-precision types are packed in 32-bit units.
346  unsigned bitWidth = elementType.getIntOrFloatBitWidth();
347  int chunkAlignmentFactor =
348  bitWidth < targetinfo::packedSizeInBitsForGatherScatter
349  ? targetinfo::packedSizeInBitsForGatherScatter / bitWidth
350  : 1;
351  auto scatterAttr = mlir::dyn_cast_if_present<ScatterTensorDescAttr>(encoding);
352  if (scatterAttr) {
353  int64_t chunkSize = scatterAttr.getChunkSizeAsInt();
354  if (rank == 1 && chunkSize != 1)
355  return emitError() << "expected non-contiguous elements for 1D tensor";
356 
357  // If chunk size > 1, the second dimension of the tensor shape must be
358  // equal to chunk size and it must be a multiple of the
359  // chunkAlignmentFactor.
360  if (chunkSize > 1) {
361  if (shape.back() != chunkSize)
362  return emitError() << "expected last dim of tensor to match chunk size";
363  if (shape.back() % chunkAlignmentFactor != 0)
364  return emitError() << "expected last dim of tensor to be a multiple of "
365  << chunkAlignmentFactor;
366  }
367  }
368 
369  auto layoutAttr = llvm::dyn_cast_if_present<LayoutAttr>(layout);
370  if (layoutAttr) {
371  if (rank != (size_t)layoutAttr.getRank())
372  return emitError() << "expected layout rank to match tensor rank";
373 
374  auto laneData = layoutAttr.getLaneData();
375  if (scatterAttr && laneData) {
376  // Validate subgroup mapping rules for scattered tensors.
377  // if chunkSize > 1, the last dimension of the tensor should
378  // be distributed in the units divisible by chunkAlignmentFactor.
379  int64_t chunkSize = scatterAttr.getChunkSizeAsInt();
380  if (chunkSize > 1 && laneData[rank - 1] % chunkAlignmentFactor)
381  return emitError()
382  << "expected last dim of lane_data to be a multiple of: "
383  << chunkAlignmentFactor;
384  }
385 
386  if (!XeGPUDialect::isEvenlyDistributable(shape, layoutAttr)) {
387  std::string shapeStr;
388  llvm::raw_string_ostream stream(shapeStr);
389  llvm::interleaveComma(shape, stream);
390  return emitError() << "cannot distribute [" << shapeStr << "] using "
391  << layoutAttr;
392  }
393  }
394  return success();
395 }
