40 /// Generates instructions to compute offsets for a subgroup identified by
41 /// its multidimensional indices (sgId), using the specified subgroup layout
42 /// (sgLayout), subgroup data dimensions (sizePerSg), and the overall data
43 /// dimensions (sizePerWg).
44 static SmallVector<SmallVector<Value>>
45 genOffsetsComputingInsts(OpBuilder &builder, Location loc,
46  SmallVector<Value> sgId, ArrayRef<int64_t> sgLayout,
47  ArrayRef<int64_t> sizePerSg,
48  ArrayRef<int64_t> sizePerWg) {
49 
50  SmallVector<SmallVector<Value>> offsets;
51 
52  // nd local offset, localOffset[i] = sgId[i] * sizePerSg[i]
53  SmallVector<Value> localOffsets = llvm::map_to_vector(
54  llvm::zip(sgId, sizePerSg), [&](const auto &t) -> Value {
55  return builder.createOrFold<index::MulOp>(
56  loc, std::get<0>(t),
57  builder.createOrFold<arith::ConstantIndexOp>(loc, std::get<1>(t)));
58  });
59 
60  // distUnit[i] is the minimum value between sizePerWg[i] and
61  // sgLayout[i] * sizePerSg[i]
62  SmallVector<int64_t> distUnit = llvm::map_to_vector(
63  llvm::zip_equal(sizePerWg, computeElementwiseMul(sgLayout, sizePerSg)),
64  [](const auto &t) { return std::min(std::get<0>(t), std::get<1>(t)); });
65 
66  for (SmallVector<int64_t> unitOffs :
67  StaticTileOffsetRange(sizePerWg, distUnit)) {
68  SmallVector<Value> base =
69  llvm::map_to_vector(unitOffs, [&](int64_t d) -> Value {
70  return arith::ConstantIndexOp::create(builder, loc, d);
71  });
72 
73  SmallVector<Value> adds = llvm::map_to_vector(
74  llvm::zip_equal(base, localOffsets), [&](const auto &t) -> Value {
75  return builder.createOrFold<arith::AddIOp>(loc, std::get<0>(t),
76  std::get<1>(t));
77  });
78 
79  SmallVector<Value> mods = llvm::map_to_vector(
80  llvm::zip_equal(adds, sizePerWg), [&](const auto &t) -> Value {
81  return builder.createOrFold<index::RemUOp>(
82  loc, std::get<0>(t),
83  arith::ConstantIndexOp::create(builder, loc, std::get<1>(t)));
84  });
85 
86  offsets.push_back(mods);
87  }
88  return offsets;
89 }

91 // Checks if the given shape can be evenly distributed based on the layout
92 // and data factors provided by the LayoutAttr.
93 bool XeGPUDialect::isEvenlyDistributable(llvm::ArrayRef<int64_t> shape,
94  xegpu::DistributeLayoutAttr attr) {
95  assert(attr && "Layout attribute is missing.");
96 
97  // Checks whether the given shape can be evenly distributed using the
98  // specified layout and data attributes. If successful, it returns the work
99  // size for each compute unit; otherwise, it returns `std::nullopt`. The work
100  // size per compute unit is calculated as follows:
101  // - If `data` is null: newShape[i] = shape[i] / layout[i]
102  // - If `data` is not null: newShape[i] = data[i]
103  // When round-robin distribution (`rr`) is enabled, `shape[i]` can be
104  // smaller than `layout[i] * data[i]`, allowing multiple compute units to
105  // share the data.
106  auto tryDistribute = [&](llvm::ArrayRef<int64_t> shape,
107  SmallVector<int64_t> layout,
108  SmallVector<int64_t> data,
109  bool rr = true) -> optional<SmallVector<int64_t>> {
110  llvm::SmallVector<int64_t> newShape(shape);
111  if (layout.size()) {
112  if (layout.size() != shape.size())
113  return std::nullopt;
114  auto ratio = computeShapeRatio(shape, layout);
115  if (!ratio.has_value())
116  return std::nullopt;
117  newShape = ratio.value();
118  }
119 
120  if (data.size()) {
121  if (data.size() != shape.size())
122  return std::nullopt;
123  auto ratio = computeShapeRatio(newShape, data);
124  if (!ratio.has_value() && rr)
125  ratio = computeShapeRatio(data, newShape);
126  if (!ratio.has_value())
127  return std::nullopt;
128 
129  // if data is not null, we always return it for next phase.
130  newShape = data;
131  }
132  return newShape;
133  };
134 
135  // check the sgLayout and sgData
136  auto maybeSgShape = tryDistribute(shape, attr.getEffectiveSgLayoutAsInt(),
137  attr.getEffectiveSgDataAsInt());
138  if (!maybeSgShape)
139  return false;
140  auto sgShape = maybeSgShape.value();
141 
142  // check InstData, it neither have layout nor need round-robin
143  auto maybeInstShape =
144  tryDistribute(sgShape, {}, attr.getEffectiveInstDataAsInt(), false);
145  if (!maybeInstShape)
146  return false;
147  auto instShape = maybeInstShape.value();
148 
149  // check LaneLayout and LaneData
150  auto maybeLaneShape =
151  tryDistribute(instShape, attr.getEffectiveLaneLayoutAsInt(),
152  attr.getEffectiveLaneDataAsInt(), false);
153  return maybeLaneShape.has_value();
154 }

200 LogicalResult
201 LayoutAttr::verify(llvm::function_ref<mlir::InFlightDiagnostic()> emitError,
202  DenseI32ArrayAttr sg_layout, DenseI32ArrayAttr sg_data,
203  DenseI32ArrayAttr inst_data, DenseI32ArrayAttr lane_layout,
204  DenseI32ArrayAttr lane_data, DenseI32ArrayAttr order) {
205 
206  // A valid layout must include at least one of sg_layout and lane_layout.
207  // sg_layout is essential for Workgroup layout, while lane_layout is
208  // required for Subgroup layout.
209  if (!sg_layout && !inst_data && !lane_layout) {
210  return emitError()
211  << "expected at least one of sg_layout, inst_data or lane_layout";
212  }
213 
214  // generate code to check sg_laout, inst_data and lane_layout having the same
215  // rank if they are not null.
216 
217  if (sg_layout && inst_data && sg_layout.size() != inst_data.size()) {
218  return emitError()
219  << "expected sg_layout and inst_data to have the same rank";
220  }
221 
222  if (sg_layout && lane_layout && sg_layout.size() != lane_layout.size()) {
223  return emitError()
224  << "expected sg_layout and lane_layout to have the same rank";
225  }
226 
227  if (inst_data && lane_layout && inst_data.size() != lane_layout.size()) {
228  return emitError()
229  << "expected inst_data and lane_layout to have the same rank";
230  }
231 
232  // sg_data is optional for Workgroup layout, but its presence requires
233  // sg_layout.
234  if (sg_data) {
235  if (!sg_layout)
236  return emitError() << "expected sg_layout being used with sg_data";
237  if (sg_data.size() != sg_layout.size())
238  return emitError()
239  << "expected sg_data and sg_layout to have the same rank";
240  }
241 
242  // lane_data is optional for Subgroup layout, but its presence requires
243  // lane_layout.
244  if (lane_data) {
245  if (!lane_layout)
246  return emitError() << "expected lane_layout being used with lane_data";
247  if (lane_data.size() != lane_layout.size())
248  return emitError()
249  << "expected lane_data and lane_layout to have the same rank";
250  }
251 
252  if (order) {
253  if (!sg_layout && !lane_layout)
254  return emitError()
255  << "expected sg_layout/lane_layout being used with order";
256 
257  if (sg_layout && order.size() != sg_layout.size())
258  return emitError()
259  << "expected order and sg_layout to have the same rank";
260 
261  if (lane_layout && order.size() != lane_layout.size())
262  return emitError()
263  << "expected order and lane_layout to have the same rank";
264  }
265 
266  return success();
267 }

269 FailureOr<SmallVector<Value>>
270 LayoutAttr::delinearizeSubgroupId(OpBuilder &builder, Location loc,
271  Value linearId) {
272  // delinearizeSubgroupId is only available for
273  // workgroup-level layout attribute
274  if (!isForWorkgroup())
275  return failure();
276 
277  // TODO: handle order attribute
278  auto hasDefaultOrder = [&]() {
279  DenseI32ArrayAttr order = getOrder();
280  return !order || isIdentityPermutation(llvm::to_vector_of<int64_t>(
281  llvm::reverse(order.asArrayRef())));
282  };
283  if (!hasDefaultOrder())
284  return mlir::emitError(loc, "order attribute is currently not supported.");
285 
286  auto dims =
287  llvm::map_to_vector(getEffectiveSgLayoutAsInt(), [&](int64_t d) -> Value {
288  return builder.createOrFold<arith::ConstantIndexOp>(loc, d);
289  });
290 
291  return affine::delinearizeIndex(builder, loc, linearId, dims);
292 }

294 /// Implements DistributeLayoutAttr::getOffsets to generate
295 /// instructions for computing multi-dimensional offsets when distributed by
296 /// LayoutAttr.
297 FailureOr<SmallVector<SmallVector<Value>>>
298 LayoutAttr::getOffsets(OpBuilder &builder, Location loc, Value linearId,
299  ArrayRef<int64_t> shape) {
300  if (!isForWorkgroup())
301  return failure();
302 
303  SmallVector<int64_t> sgLayout = getEffectiveSgLayoutAsInt();
304  SmallVector<int64_t> sgShape = getEffectiveSgDataAsInt();
305  if (sgShape.empty()) {
306  if (auto derivedShape = computeShapeRatio(shape, sgLayout))
307  sgShape = derivedShape.value();
308  else
309  return failure();
310  }
311 
312  // delinearize Ids
313  auto maybeIds = delinearizeSubgroupId(builder, loc, linearId);
314  if (failed(maybeIds))
315  return failure();
316  SmallVector<Value> sgIds = *maybeIds;
317 
318  return genOffsetsComputingInsts(builder, loc, sgIds, sgLayout, sgShape,
319  shape);
320 }

325 LogicalResult
326 SliceAttr::verify(llvm::function_ref<InFlightDiagnostic()> emitError,
327  xegpu::DistributeLayoutAttr parent, DenseI64ArrayAttr dims) {
328  if (!parent || !dims)
329  return emitError() << "expected parent layout and dims attribute";
330 
331  int64_t rank = parent.getRank();
332 
333  // check every element in dims is unique and smaller than rank
334  llvm::SmallDenseSet<int64_t> seen;
335  for (int64_t dim : dims.asArrayRef()) {
336  if (dim < 0 || dim >= rank)
337  return emitError() << "invalid dim (" << dim << ") in slice attribute.";
338  if (!seen.insert(dim).second)
339  return emitError() << "repeated dim (" << dim << ") in slice attribute.";
340  }
341  return success();
342 }

344 SliceAttr SliceAttr::flatten() const {
345  xegpu::DistributeLayoutAttr parent = getParent();
346  SmallVector<DenseI64ArrayAttr> slicedDims({getDims()});
347 
348  while (auto sliceAttr = dyn_cast<xegpu::SliceAttr>(parent)) {
349  parent = sliceAttr.getParent();
350  slicedDims.push_back(sliceAttr.getDims());
351  }
352 
353  auto layoutAttr = dyn_cast<xegpu::LayoutAttr>(parent);
354  SmallVector<int64_t> indices =
355  llvm::to_vector(llvm::seq<int64_t>(0, layoutAttr.getRank()));
356 
357  // get remaining dims (flattend) by applying slice ops with all slicedDims
358  SmallVector<int64_t> remainingDims(indices);
359  for (auto dim : llvm::reverse(slicedDims))
360  remainingDims = XeGPUDialect::slice(llvm::ArrayRef<int64_t>(remainingDims),
361  dim.asArrayRef());
362 
363  // get flattend sliced dims by applying slice ops with the remaining dims
364  SmallVector<int64_t> flattendDims = XeGPUDialect::slice(
365  llvm::ArrayRef<int64_t>(indices), llvm::ArrayRef<int64_t>(remainingDims));
366 
367  return xegpu::SliceAttr::get(
368  getContext(), layoutAttr,
369  DenseI64ArrayAttr::get(getContext(), flattendDims));
370 }

380 /// Implements DistributeLayoutAttr::getOffsets to generate
381 /// instructions for computing multi-dimensional offsets when distributed by
382 /// SliceAttr.
383 FailureOr<SmallVector<SmallVector<Value>>>
384 SliceAttr::getOffsets(OpBuilder &builder, Location loc, Value linearId,
385  ArrayRef<int64_t> shape) {
386  assert(getRank() == static_cast<int64_t>(shape.size()) && "invalid shape.");
387  if (!isForWorkgroup())
388  return failure();
389 
390  SmallVector<int64_t> sgLayout = getEffectiveSgLayoutAsInt();
391  SmallVector<int64_t> sgShape = getEffectiveSgDataAsInt();
392  if (sgShape.empty()) {
393  if (auto derivedShape = computeShapeRatio(shape, sgLayout))
394  sgShape = derivedShape.value();
395  else
396  return failure();
397  }
398 
399  // delinearize Ids
400  auto maybeIds = delinearizeSubgroupId(builder, loc, linearId);
401  if (failed(maybeIds))
402  return failure();
403 
404  // The effective sgIds for offsets computing correspond
405  // to the dims that are not sliced.
406  ArrayRef<int64_t> dims = flatten().getDims().asArrayRef();
407  SmallVector<Value> sgIds =
408  XeGPUDialect::slice(ArrayRef<Value>(*maybeIds), dims);
409 
410  return genOffsetsComputingInsts(builder, loc, sgIds, sgLayout, sgShape,
411  shape);
412 }

414 bool SliceAttr::isSliceOf(const xegpu::DistributeLayoutAttr &other) {
415  auto flattenedThis = flatten();
416  // If other is a LayoutAttr, just compare directly with parent of
417  // flattenedThis.
418  if (auto otherLayout = dyn_cast<xegpu::LayoutAttr>(other))
419  return flattenedThis.getParent() == otherLayout;
420  // If other is a SliceAttr, flatten it first before comparing.
421  auto flattenedOther = dyn_cast<xegpu::SliceAttr>(other).flatten();
422  // Both must have common parent LayoutAttr.
423  if (flattenedThis.getParent() != flattenedOther.getParent())
424  return false;
425  // otherFlattened's sliced dims must be a subset of flattenedThis's sliced
426  // dims.
427  llvm::SmallDenseSet<int64_t> thisDims(
428  flattenedThis.getDims().asArrayRef().begin(),
429  flattenedThis.getDims().asArrayRef().end());
430  return llvm::all_of(flattenedOther.getDims().asArrayRef(),
431  [&](int64_t dim) { return thisDims.contains(dim); });
432 }

453 mlir::Type TensorDescType::parse(AsmParser &parser) {
454  llvm::SmallVector<int64_t> shape;
455  mlir::Type elementType;
456  mlir::FailureOr<mlir::Attribute> encoding;
457  mlir::FailureOr<mlir::Attribute> layout;
458 
459  // Parse literal '<'
460  if (parser.parseLess())
461  return {};
462 
463  auto shapeLoc = parser.getCurrentLocation();
464  if (mlir::failed(parser.parseDimensionList(shape))) {
465  parser.emitError(shapeLoc, "failed to parse parameter 'shape'");
466  return {};
467  }
468 
469  auto elemTypeLoc = parser.getCurrentLocation();
470  if (mlir::failed(parser.parseType(elementType))) {
471  parser.emitError(elemTypeLoc, "failed to parse parameter 'elementType'");
472  return {};
473  }
474 
475  // parse optional attributes
476  while (mlir::succeeded(parser.parseOptionalComma())) {
477  mlir::Attribute attr;
478  ParseResult res = parser.parseAttribute(attr);
479  if (mlir::succeeded(res)) {
480  if (mlir::isa<LayoutAttr>(attr)) {
481  layout = attr;
482  continue;
483  }
484  if (mlir::isa<BlockTensorDescAttr, ScatterTensorDescAttr>(attr)) {
485  encoding = attr;
486  continue;
487  }
488  }
489  return {};
490  }
491 
492  // Parse literal '>'
493  if (parser.parseGreater())
494  return {};
495 
496  MLIRContext *ctxt = parser.getContext();
497  return TensorDescType::getChecked(
498  [&]() { return parser.emitError(parser.getNameLoc()); }, ctxt, shape,
499  elementType, encoding.value_or(BlockTensorDescAttr::get(ctxt)),
500  layout.value_or(mlir::Attribute()));
501 }

548 LogicalResult
549 TensorDescType::verify(llvm::function_ref<InFlightDiagnostic()> emitError,
550  llvm::ArrayRef<int64_t> shape, mlir::Type elementType,
551  mlir::Attribute encoding, mlir::Attribute layout) {
552  size_t rank = shape.size();
553 
554  if (rank == 0)
555  return emitError() << "expected non-zero rank tensor";
556 
557  auto blockAttr = mlir::dyn_cast_if_present<BlockTensorDescAttr>(encoding);
558  if (blockAttr) {
559  MemorySpaceAttr memorySpaceAttr = blockAttr.getMemorySpace();
560  if (rank > 1 && memorySpaceAttr &&
561  memorySpaceAttr.getValue() == MemorySpace::SLM)
562  return emitError() << "SLM is only supported for 1D block tensor";
563  }
564 
565  // for gather and scatter ops, Low-precision types are packed in 32-bit units.
566  unsigned bitWidth = elementType.getIntOrFloatBitWidth();
567  int chunkAlignmentFactor =
568  bitWidth < targetinfo::packedSizeInBitsForGatherScatter
569  ? targetinfo::packedSizeInBitsForGatherScatter / bitWidth
570  : 1;
571  auto scatterAttr = mlir::dyn_cast_if_present<ScatterTensorDescAttr>(encoding);
572  if (scatterAttr) {
573  int64_t chunkSize = scatterAttr.getChunkSizeAsInt();
574  if (rank == 1 && chunkSize != 1)
575  return emitError() << "expected non-contiguous elements for 1D tensor";
576 
577  // If chunk size > 1, the second dimension of the tensor shape must be
578  // equal to chunk size and it must be a multiple of the
579  // chunkAlignmentFactor.
580  if (chunkSize > 1) {
581  if (shape.back() != chunkSize)
582  return emitError() << "expected last dim of tensor to match chunk size";
583  if (shape.back() % chunkAlignmentFactor != 0)
584  return emitError() << "expected last dim of tensor to be a multiple of "
585  << chunkAlignmentFactor;
586  }
587  }
588 
589  auto layoutAttr = llvm::dyn_cast_if_present<LayoutAttr>(layout);
590  if (layoutAttr) {
591  if (rank != (size_t)layoutAttr.getRank())
592  return emitError() << "expected layout rank to match tensor rank";
593 
594  auto laneData = layoutAttr.getLaneData();
595  if (scatterAttr && laneData) {
596  // Validate subgroup mapping rules for scattered tensors.
597  // if chunkSize > 1, the last dimension of the tensor should
598  // be distributed in the units divisible by chunkAlignmentFactor.
599  int64_t chunkSize = scatterAttr.getChunkSizeAsInt();
600  if (chunkSize > 1 && laneData[rank - 1] % chunkAlignmentFactor)
601  return emitError()
602  << "expected last dim of lane_data to be a multiple of: "
603  << chunkAlignmentFactor;
604  }
605 
606  if (!XeGPUDialect::isEvenlyDistributable(shape, layoutAttr)) {
607  std::string shapeStr;
608  llvm::raw_string_ostream stream(shapeStr);
609  llvm::interleaveComma(shape, stream);
610  return emitError() << "cannot distribute [" << shapeStr << "] using "
611  << layoutAttr;
612  }
613  }
614  return success();
615 }

677 Attribute MemLayoutAttr::parse(AsmParser &parser, Type type) {
678 
679  auto context = parser.getContext();
680  llvm::SMLoc loc = parser.getCurrentLocation();
681 
682  llvm::SmallDenseSet<StringRef> seenKeys;
683  SmallVector<NamedAttribute> attributes;
684 
685  auto parseElt = [&]() -> ParseResult {
686  StringRef nameId;
687  if (failed(parser.parseKeyword(&nameId)))
688  return parser.emitError(loc, "expected valid attribute name");
689 
690  if (!seenKeys.insert(nameId).second)
691  return parser.emitError(loc, "duplicate key '")
692  << nameId << " in mem layout attribute";
693 
694  if (failed(parser.parseEqual()))
695  return failure();
696 
697  Attribute attr;
698  if (failed(parser.parseAttribute(attr)))
699  return failure();
700  attributes.emplace_back(nameId, attr);
701  return success();
702  };
703 
704  // Parse literal '<'
705  if (parser.parseLess())
706  return {};
707 
708  if (failed(parser.parseCommaSeparatedList(parseElt)))
709  return {};
710 
711  // Parse literal '>'
712  if (parser.parseGreater())
713  return {};
714 
715  return parser.getChecked<MemLayoutAttr>(
716  loc, context, DictionaryAttr::get(context, attributes));
717 }
