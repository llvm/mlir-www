88 void ConvertToLLVMPattern::getMemRefDescriptorSizes(
89  Location loc, MemRefType memRefType, ValueRange dynamicSizes,
90  ConversionPatternRewriter &rewriter, SmallVectorImpl<Value> &sizes,
91  SmallVectorImpl<Value> &strides, Value &size, bool sizeInBytes) const {
92  assert(isConvertibleAndHasIdentityMaps(memRefType) &&
93  "layout maps must have been normalized away");
94  assert(count(memRefType.getShape(), ShapedType::kDynamic) ==
95  static_cast<ssize_t>(dynamicSizes.size()) &&
96  "dynamicSizes size doesn't match dynamic sizes count in memref shape");
97 
98  sizes.reserve(memRefType.getRank());
99  unsigned dynamicIndex = 0;
100  Type indexType = getIndexType();
101  for (int64_t size : memRefType.getShape()) {
102  sizes.push_back(
103  size == ShapedType::kDynamic
104  ? dynamicSizes[dynamicIndex++]
105  : createIndexAttrConstant(rewriter, loc, indexType, size));
106  }
107 
108  // Strides: iterate sizes in reverse order and multiply.
109  int64_t stride = 1;
110  Value runningStride = createIndexAttrConstant(rewriter, loc, indexType, 1);
111  strides.resize(memRefType.getRank());
112  for (auto i = memRefType.getRank(); i-- > 0;) {
113  strides[i] = runningStride;
114 
115  int64_t staticSize = memRefType.getShape()[i];
116  bool useSizeAsStride = stride == 1;
117  if (staticSize == ShapedType::kDynamic)
118  stride = ShapedType::kDynamic;
119  if (stride != ShapedType::kDynamic)
120  stride *= staticSize;
121 
122  if (useSizeAsStride)
123  runningStride = sizes[i];
124  else if (stride == ShapedType::kDynamic)
125  runningStride =
126  LLVM::MulOp::create(rewriter, loc, runningStride, sizes[i]);
127  else
128  runningStride = createIndexAttrConstant(rewriter, loc, indexType, stride);
129  }
130  if (sizeInBytes) {
131  // Buffer size in bytes.
132  Type elementType = typeConverter->convertType(memRefType.getElementType());
133  auto elementPtrType = LLVM::LLVMPointerType::get(rewriter.getContext());
134  Value nullPtr = LLVM::ZeroOp::create(rewriter, loc, elementPtrType);
135  Value gepPtr = LLVM::GEPOp::create(rewriter, loc, elementPtrType,
136  elementType, nullPtr, runningStride);
137  size = LLVM::PtrToIntOp::create(rewriter, loc, getIndexType(), gepPtr);
138  } else {
139  size = runningStride;
140  }
141 }

143 Value ConvertToLLVMPattern::getSizeInBytes(
144  Location loc, Type type, ConversionPatternRewriter &rewriter) const {
145  // Compute the size of an individual element. This emits the MLIR equivalent
146  // of the following sizeof(...) implementation in LLVM IR:
147  // %0 = getelementptr %elementType* null, %indexType 1
148  // %1 = ptrtoint %elementType* %0 to %indexType
149  // which is a common pattern of getting the size of a type in bytes.
150  Type llvmType = typeConverter->convertType(type);
151  auto convertedPtrType = LLVM::LLVMPointerType::get(rewriter.getContext());
152  auto nullPtr = LLVM::ZeroOp::create(rewriter, loc, convertedPtrType);
153  auto gep = LLVM::GEPOp::create(rewriter, loc, convertedPtrType, llvmType,
154  nullPtr, ArrayRef<LLVM::GEPArg>{1});
155  return LLVM::PtrToIntOp::create(rewriter, loc, getIndexType(), gep);
156 }

158 Value ConvertToLLVMPattern::getNumElements(
159  Location loc, MemRefType memRefType, ValueRange dynamicSizes,
160  ConversionPatternRewriter &rewriter) const {
161  assert(count(memRefType.getShape(), ShapedType::kDynamic) ==
162  static_cast<ssize_t>(dynamicSizes.size()) &&
163  "dynamicSizes size doesn't match dynamic sizes count in memref shape");
164 
165  Type indexType = getIndexType();
166  Value numElements = memRefType.getRank() == 0
167  ? createIndexAttrConstant(rewriter, loc, indexType, 1)
168  : nullptr;
169  unsigned dynamicIndex = 0;
170 
171  // Compute the total number of memref elements.
172  for (int64_t staticSize : memRefType.getShape()) {
173  if (numElements) {
174  Value size =
175  staticSize == ShapedType::kDynamic
176  ? dynamicSizes[dynamicIndex++]
177  : createIndexAttrConstant(rewriter, loc, indexType, staticSize);
178  numElements = LLVM::MulOp::create(rewriter, loc, numElements, size);
179  } else {
180  numElements =
181  staticSize == ShapedType::kDynamic
182  ? dynamicSizes[dynamicIndex++]
183  : createIndexAttrConstant(rewriter, loc, indexType, staticSize);
184  }
185  }
186  return numElements;
187 }

189 /// Creates and populates the memref descriptor struct given all its fields.
190 MemRefDescriptor ConvertToLLVMPattern::createMemRefDescriptor(
191  Location loc, MemRefType memRefType, Value allocatedPtr, Value alignedPtr,
192  ArrayRef<Value> sizes, ArrayRef<Value> strides,
193  ConversionPatternRewriter &rewriter) const {
194  auto structType = typeConverter->convertType(memRefType);
195  auto memRefDescriptor = MemRefDescriptor::poison(rewriter, loc, structType);
196 
197  // Field 1: Allocated pointer, used for malloc/free.
198  memRefDescriptor.setAllocatedPtr(rewriter, loc, allocatedPtr);
199 
200  // Field 2: Actual aligned pointer to payload.
201  memRefDescriptor.setAlignedPtr(rewriter, loc, alignedPtr);
202 
203  // Field 3: Offset in aligned pointer.
204  Type indexType = getIndexType();
205  memRefDescriptor.setOffset(
206  rewriter, loc, createIndexAttrConstant(rewriter, loc, indexType, 0));
207 
208  // Fields 4: Sizes.
209  for (const auto &en : llvm::enumerate(sizes))
210  memRefDescriptor.setSize(rewriter, loc, en.index(), en.value());
211 
212  // Field 5: Strides.
213  for (const auto &en : llvm::enumerate(strides))
214  memRefDescriptor.setStride(rewriter, loc, en.index(), en.value());
215 
216  return memRefDescriptor;
217 }

219 LogicalResult ConvertToLLVMPattern::copyUnrankedDescriptors(
220  OpBuilder &builder, Location loc, TypeRange origTypes,
221  SmallVectorImpl<Value> &operands, bool toDynamic) const {
222  assert(origTypes.size() == operands.size() &&
223  "expected as may original types as operands");
224 
225  // Find operands of unranked memref type and store them.
226  SmallVector<UnrankedMemRefDescriptor> unrankedMemrefs;
227  SmallVector<unsigned> unrankedAddressSpaces;
228  for (unsigned i = 0, e = operands.size(); i < e; ++i) {
229  if (auto memRefType = dyn_cast<UnrankedMemRefType>(origTypes[i])) {
230  unrankedMemrefs.emplace_back(operands[i]);
231  FailureOr<unsigned> addressSpace =
232  getTypeConverter()->getMemRefAddressSpace(memRefType);
233  if (failed(addressSpace))
234  return failure();
235  unrankedAddressSpaces.emplace_back(*addressSpace);
236  }
237  }
238 
239  if (unrankedMemrefs.empty())
240  return success();
241 
242  // Compute allocation sizes.
243  SmallVector<Value> sizes;
244  UnrankedMemRefDescriptor::computeSizes(builder, loc, *getTypeConverter(),
245  unrankedMemrefs, unrankedAddressSpaces,
246  sizes);
247 
248  // Get frequently used types.
249  Type indexType = getTypeConverter()->getIndexType();
250 
251  // Find the malloc and free, or declare them if necessary.
252  auto module = builder.getInsertionPoint()->getParentOfType<ModuleOp>();
253  FailureOr<LLVM::LLVMFuncOp> freeFunc, mallocFunc;
254  if (toDynamic) {
255  mallocFunc = LLVM::lookupOrCreateMallocFn(builder, module, indexType);
256  if (failed(mallocFunc))
257  return failure();
258  }
259  if (!toDynamic) {
260  freeFunc = LLVM::lookupOrCreateFreeFn(builder, module);
261  if (failed(freeFunc))
262  return failure();
263  }
264 
265  unsigned unrankedMemrefPos = 0;
266  for (unsigned i = 0, e = operands.size(); i < e; ++i) {
267  Type type = origTypes[i];
268  if (!isa<UnrankedMemRefType>(type))
269  continue;
270  Value allocationSize = sizes[unrankedMemrefPos++];
271  UnrankedMemRefDescriptor desc(operands[i]);
272 
273  // Allocate memory, copy, and free the source if necessary.
274  Value memory =
275  toDynamic ? LLVM::CallOp::create(builder, loc, mallocFunc.value(),
276  allocationSize)
277  .getResult()
278  : LLVM::AllocaOp::create(builder, loc, getPtrType(),
279  IntegerType::get(getContext(), 8),
280  allocationSize,
281  /*alignment=*/0);
282  Value source = desc.memRefDescPtr(builder, loc);
283  LLVM::MemcpyOp::create(builder, loc, memory, source, allocationSize, false);
284  if (!toDynamic)
285  LLVM::CallOp::create(builder, loc, freeFunc.value(), source);
286 
287  // Create a new descriptor. The same descriptor can be returned multiple
288  // times, attempting to modify its pointer can lead to memory leaks
289  // (allocated twice and overwritten) or double frees (the caller does not
290  // know if the descriptor points to the same memory).
291  Type descriptorType = getTypeConverter()->convertType(type);
292  if (!descriptorType)
293  return failure();
294  auto updatedDesc =
295  UnrankedMemRefDescriptor::poison(builder, loc, descriptorType);
296  Value rank = desc.rank(builder, loc);
297  updatedDesc.setRank(builder, loc, rank);
298  updatedDesc.setMemRefDescPtr(builder, loc, memory);
299 
300  operands[i] = updatedDesc;
301  }
302 
303  return success();
304 }

316 /// Replaces the given operation "op" with a new operation of type "targetOp"
317 /// and given operands.
318 LogicalResult LLVM::detail::oneToOneRewrite(
319  Operation *op, StringRef targetOp, ValueRange operands,
320  ArrayRef<NamedAttribute> targetAttrs,
321  const LLVMTypeConverter &typeConverter, ConversionPatternRewriter &rewriter,
322  IntegerOverflowFlags overflowFlags) {
323  unsigned numResults = op->getNumResults();
324 
325  SmallVector<Type> resultTypes;
326  if (numResults != 0) {
327  resultTypes.push_back(
328  typeConverter.packOperationResults(op->getResultTypes()));
329  if (!resultTypes.back())
330  return failure();
331  }
332 
333  // Create the operation through state since we don't know its C++ type.
334  Operation *newOp =
335  rewriter.create(op->getLoc(), rewriter.getStringAttr(targetOp), operands,
336  resultTypes, targetAttrs);
337 
338  setNativeProperties(newOp, overflowFlags);
339 
340  // If the operation produced 0 or 1 result, return them immediately.
341  if (numResults == 0)
342  return rewriter.eraseOp(op), success();
343  if (numResults == 1)
344  return rewriter.replaceOp(op, newOp->getResult(0)), success();
345 
346  // Otherwise, it had been converted to an operation producing a structure.
347  // Extract individual results from the structure and return them as list.
348  SmallVector<Value, 4> results;
349  results.reserve(numResults);
350  for (unsigned i = 0; i < numResults; ++i) {
351  results.push_back(LLVM::ExtractValueOp::create(rewriter, op->getLoc(),
352  newOp->getResult(0), i));
353  }
354  rewriter.replaceOp(op, results);
355  return success();
356 }

358 LogicalResult LLVM::detail::intrinsicRewrite(
359  Operation *op, StringRef intrinsic, ValueRange operands,
360  const LLVMTypeConverter &typeConverter, RewriterBase &rewriter) {
361  auto loc = op->getLoc();
362 
363  if (!llvm::all_of(operands, [](Value value) {
364  return LLVM::isCompatibleType(value.getType());
365  }))
366  return failure();
367 
368  unsigned numResults = op->getNumResults();
369  Type resType;
370  if (numResults != 0)
371  resType = typeConverter.packOperationResults(op->getResultTypes());
372 
373  auto callIntrOp = LLVM::CallIntrinsicOp::create(
374  rewriter, loc, resType, rewriter.getStringAttr(intrinsic), operands);
375  // Propagate attributes.
376  callIntrOp->setAttrs(op->getAttrDictionary());
377 
378  if (numResults <= 1) {
379  // Directly replace the original op.
380  rewriter.replaceOp(op, callIntrOp);
381  return success();
382  }
383 
384  // Extract individual results from packed structure and use them as
385  // replacements.
386  SmallVector<Value, 4> results;
387  results.reserve(numResults);
388  Value intrRes = callIntrOp.getResults();
389  for (unsigned i = 0; i < numResults; ++i)
390  results.push_back(LLVM::ExtractValueOp::create(rewriter, loc, intrRes, i));
391  rewriter.replaceOp(op, results);
392 
393  return success();
394 }

411 SmallVector<Value> mlir::LLVM::decomposeValue(OpBuilder &builder, Location loc,
412  Value src, Type dstType) {
413  Type srcType = src.getType();
414  if (srcType == dstType)
415  return {src};
416 
417  unsigned srcBitWidth = getBitWidth(srcType);
418  unsigned dstBitWidth = getBitWidth(dstType);
419  if (srcBitWidth == dstBitWidth) {
420  Value cast = LLVM::BitcastOp::create(builder, loc, dstType, src);
421  return {cast};
422  }
423 
424  if (dstBitWidth > srcBitWidth) {
425  auto smallerInt = builder.getIntegerType(srcBitWidth);
426  if (srcType != smallerInt)
427  src = LLVM::BitcastOp::create(builder, loc, smallerInt, src);
428 
429  auto largerInt = builder.getIntegerType(dstBitWidth);
430  Value res = LLVM::ZExtOp::create(builder, loc, largerInt, src);
431  return {res};
432  }
433  assert(srcBitWidth % dstBitWidth == 0 &&
434  "src bit width must be a multiple of dst bit width");
435  int64_t numElements = srcBitWidth / dstBitWidth;
436  auto vecType = VectorType::get(numElements, dstType);
437 
438  src = LLVM::BitcastOp::create(builder, loc, vecType, src);
439 
440  SmallVector<Value> res;
441  for (auto i : llvm::seq(numElements)) {
442  Value idx = createI32Constant(builder, loc, i);
443  Value elem = LLVM::ExtractElementOp::create(builder, loc, src, idx);
444  res.emplace_back(elem);
445  }
446 
447  return res;
448 }

450 Value mlir::LLVM::composeValue(OpBuilder &builder, Location loc, ValueRange src,
451  Type dstType) {
452  assert(!src.empty() && "src range must not be empty");
453  if (src.size() == 1) {
454  Value res = src.front();
455  if (res.getType() == dstType)
456  return res;
457 
458  unsigned srcBitWidth = getBitWidth(res.getType());
459  unsigned dstBitWidth = getBitWidth(dstType);
460  if (dstBitWidth < srcBitWidth) {
461  auto largerInt = builder.getIntegerType(srcBitWidth);
462  if (res.getType() != largerInt)
463  res = LLVM::BitcastOp::create(builder, loc, largerInt, res);
464 
465  auto smallerInt = builder.getIntegerType(dstBitWidth);
466  res = LLVM::TruncOp::create(builder, loc, smallerInt, res);
467  }
468 
469  if (res.getType() != dstType)
470  res = LLVM::BitcastOp::create(builder, loc, dstType, res);
471 
472  return res;
473  }
474 
475  int64_t numElements = src.size();
476  auto srcType = VectorType::get(numElements, src.front().getType());
477  Value res = LLVM::PoisonOp::create(builder, loc, srcType);
478  for (auto &&[i, elem] : llvm::enumerate(src)) {
479  Value idx = createI32Constant(builder, loc, i);
480  res = LLVM::InsertElementOp::create(builder, loc, srcType, res, elem, idx);
481  }
482 
483  if (res.getType() != dstType)
484  res = LLVM::BitcastOp::create(builder, loc, dstType, res);
485 
486  return res;
487 }

489 Value mlir::LLVM::getStridedElementPtr(OpBuilder &builder, Location loc,
490  const LLVMTypeConverter &converter,
491  MemRefType type, Value memRefDesc,
492  ValueRange indices,
493  LLVM::GEPNoWrapFlags noWrapFlags) {
494  auto [strides, offset] = type.getStridesAndOffset();
495 
496  MemRefDescriptor memRefDescriptor(memRefDesc);
497  // Use a canonical representation of the start address so that later
498  // optimizations have a longer sequence of instructions to CSE.
499  // If we don't do that we would sprinkle the memref.offset in various
500  // position of the different address computations.
501  Value base = memRefDescriptor.bufferPtr(builder, loc, converter, type);
502 
503  LLVM::IntegerOverflowFlags intOverflowFlags =
504  LLVM::IntegerOverflowFlags::none;
505  if (LLVM::bitEnumContainsAny(noWrapFlags, LLVM::GEPNoWrapFlags::nusw)) {
506  intOverflowFlags = intOverflowFlags | LLVM::IntegerOverflowFlags::nsw;
507  }
508  if (LLVM::bitEnumContainsAny(noWrapFlags, LLVM::GEPNoWrapFlags::nuw)) {
509  intOverflowFlags = intOverflowFlags | LLVM::IntegerOverflowFlags::nuw;
510  }
511 
512  Type indexType = converter.getIndexType();
513  Value index;
514  for (int i = 0, e = indices.size(); i < e; ++i) {
515  Value increment = indices[i];
516  if (strides[i] != 1) { // Skip if stride is 1.
517  Value stride =
518  ShapedType::isDynamic(strides[i])
519  ? memRefDescriptor.stride(builder, loc, i)
520  : LLVM::ConstantOp::create(builder, loc, indexType,
521  builder.getIndexAttr(strides[i]));
522  increment = LLVM::MulOp::create(builder, loc, increment, stride,
523  intOverflowFlags);
524  }
525  index = index ? LLVM::AddOp::create(builder, loc, index, increment,
526  intOverflowFlags)
527  : increment;
528  }
529 
530  Type elementPtrType = memRefDescriptor.getElementPtrType();
531  return index
532  ? LLVM::GEPOp::create(builder, loc, elementPtrType,
533  converter.convertType(type.getElementType()),
534  base, index, noWrapFlags)
535  : base;
536 }
