38 // LoopNestStateCollector walks loop nests and collects load and store
39 // operations, and whether or not a region holding op other than ForOp and IfOp
40 // was encountered in the loop nest.
41 void LoopNestStateCollector::collect(Operation *opToWalk) {
42  opToWalk->walk([&](Operation *op) {
43  if (auto forOp = dyn_cast<AffineForOp>(op)) {
44  forOps.push_back(forOp);
45  } else if (isa<AffineReadOpInterface>(op)) {
46  loadOpInsts.push_back(op);
47  } else if (isa<AffineWriteOpInterface>(op)) {
48  storeOpInsts.push_back(op);
49  } else {
50  auto memInterface = dyn_cast<MemoryEffectOpInterface>(op);
51  if (!memInterface) {
52  if (op->hasTrait<OpTrait::HasRecursiveMemoryEffects>())
53  // This op itself is memory-effect free.
54  return;
55  // Check operands. Eg. ops like the `call` op are handled here.
56  for (Value v : op->getOperands()) {
57  if (!isa<MemRefType>(v.getType()))
58  continue;
59  // Conservatively, we assume the memref is read and written to.
60  memrefLoads.push_back(op);
61  memrefStores.push_back(op);
62  }
63  } else {
64  // Non-affine loads and stores.
65  if (hasEffect<MemoryEffects::Read>(op))
66  memrefLoads.push_back(op);
67  if (hasEffect<MemoryEffects::Write>(op))
68  memrefStores.push_back(op);
69  if (hasEffect<MemoryEffects::Free>(op))
70  memrefFrees.push_back(op);
71  }
72  }
73  });
74 }

160 /// Returns the values that this op has a memref effect of type `EffectTys` on,
161 /// not considering recursive effects.
162 template <typename... EffectTys>
163 static void getEffectedValues(Operation *op, SmallVectorImpl<Value> &values) {
164  auto memOp = dyn_cast<MemoryEffectOpInterface>(op);
165  if (!memOp) {
166  if (op->hasTrait<OpTrait::HasRecursiveMemoryEffects>())
167  // No effects.
168  return;
169  // Memref operands have to be considered as being affected.
170  for (Value operand : op->getOperands()) {
171  if (isa<MemRefType>(operand.getType()))
172  values.push_back(operand);
173  }
174  return;
175  }
176  SmallVector<SideEffects::EffectInstance<MemoryEffects::Effect>, 4> effects;
177  memOp.getEffects(effects);
178  for (auto &effect : effects) {
179  Value effectVal = effect.getValue();
180  if (isa<EffectTys...>(effect.getEffect()) && effectVal &&
181  isa<MemRefType>(effectVal.getType()))
182  values.push_back(effectVal);
183  };
184 }

186 /// Add `op` to MDG creating a new node and adding its memory accesses (affine
187 /// or non-affine to memrefAccesses (memref -> list of nodes with accesses) map.
188 static Node *
189 addNodeToMDG(Operation *nodeOp, MemRefDependenceGraph &mdg,
190  DenseMap<Value, SetVector<unsigned>> &memrefAccesses) {
191  auto &nodes = mdg.nodes;
192  // Create graph node 'id' to represent top-level 'forOp' and record
193  // all loads and store accesses it contains.
194  LoopNestStateCollector collector;
195  collector.collect(nodeOp);
196  unsigned newNodeId = mdg.nextNodeId++;
197  Node &node = nodes.insert({newNodeId, Node(newNodeId, nodeOp)}).first->second;
198  for (Operation *op : collector.loadOpInsts) {
199  node.loads.push_back(op);
200  auto memref = cast<AffineReadOpInterface>(op).getMemRef();
201  memrefAccesses[memref].insert(node.id);
202  }
203  for (Operation *op : collector.storeOpInsts) {
204  node.stores.push_back(op);
205  auto memref = cast<AffineWriteOpInterface>(op).getMemRef();
206  memrefAccesses[memref].insert(node.id);
207  }
208  for (Operation *op : collector.memrefLoads) {
209  SmallVector<Value> effectedValues;
210  getEffectedValues<MemoryEffects::Read>(op, effectedValues);
211  if (llvm::any_of(((ValueRange)effectedValues).getTypes(),
212  [](Type type) { return !isa<MemRefType>(type); }))
213  // We do not know the interaction here.
214  return nullptr;
215  for (Value memref : effectedValues)
216  memrefAccesses[memref].insert(node.id);
217  node.memrefLoads.push_back(op);
218  }
219  for (Operation *op : collector.memrefStores) {
220  SmallVector<Value> effectedValues;
221  getEffectedValues<MemoryEffects::Write>(op, effectedValues);
222  if (llvm::any_of((ValueRange(effectedValues)).getTypes(),
223  [](Type type) { return !isa<MemRefType>(type); }))
224  return nullptr;
225  for (Value memref : effectedValues)
226  memrefAccesses[memref].insert(node.id);
227  node.memrefStores.push_back(op);
228  }
229  for (Operation *op : collector.memrefFrees) {
230  SmallVector<Value> effectedValues;
231  getEffectedValues<MemoryEffects::Free>(op, effectedValues);
232  if (llvm::any_of((ValueRange(effectedValues)).getTypes(),
233  [](Type type) { return !isa<MemRefType>(type); }))
234  return nullptr;
235  for (Value memref : effectedValues)
236  memrefAccesses[memref].insert(node.id);
237  node.memrefFrees.push_back(op);
238  }
239 
240  return &node;
241 }

243 bool MemRefDependenceGraph::init() {
244  LLVM_DEBUG(llvm::dbgs() << "--- Initializing MDG ---\n");
245  // Map from a memref to the set of ids of the nodes that have ops accessing
246  // the memref.
247  DenseMap<Value, SetVector<unsigned>> memrefAccesses;
248 
249  // Create graph nodes.
250  DenseMap<Operation *, unsigned> forToNodeMap;
251  for (Operation &op : block) {
252  if (auto forOp = dyn_cast<AffineForOp>(op)) {
253  Node *node = addNodeToMDG(&op, *this, memrefAccesses);
254  if (!node)
255  return false;
256  forToNodeMap[&op] = node->id;
257  } else if (isa<AffineReadOpInterface>(op)) {
258  // Create graph node for top-level load op.
259  Node node(nextNodeId++, &op);
260  node.loads.push_back(&op);
261  auto memref = cast<AffineReadOpInterface>(op).getMemRef();
262  memrefAccesses[memref].insert(node.id);
263  nodes.insert({node.id, node});
264  } else if (isa<AffineWriteOpInterface>(op)) {
265  // Create graph node for top-level store op.
266  Node node(nextNodeId++, &op);
267  node.stores.push_back(&op);
268  auto memref = cast<AffineWriteOpInterface>(op).getMemRef();
269  memrefAccesses[memref].insert(node.id);
270  nodes.insert({node.id, node});
271  } else if (op.getNumResults() > 0 && !op.use_empty()) {
272  // Create graph node for top-level producer of SSA values, which
273  // could be used by loop nest nodes.
274  Node *node = addNodeToMDG(&op, *this, memrefAccesses);
275  if (!node)
276  return false;
277  } else if (!isMemoryEffectFree(&op) &&
278  (op.getNumRegions() == 0 || isa<RegionBranchOpInterface>(op))) {
279  // Create graph node for top-level op unless it is known to be
280  // memory-effect free. This covers all unknown/unregistered ops,
281  // non-affine ops with memory effects, and region-holding ops with a
282  // well-defined control flow. During the fusion validity checks, edges
283  // to/from these ops get looked at.
284  Node *node = addNodeToMDG(&op, *this, memrefAccesses);
285  if (!node)
286  return false;
287  } else if (op.getNumRegions() != 0 && !isa<RegionBranchOpInterface>(op)) {
288  // Return false if non-handled/unknown region-holding ops are found. We
289  // won't know what such ops do or what its regions mean; for e.g., it may
290  // not be an imperative op.
291  LLVM_DEBUG(llvm::dbgs()
292  << "MDG init failed; unknown region-holding op found!\n");
293  return false;
294  }
295  // We aren't creating nodes for memory-effect free ops either with no
296  // regions (unless it has results being used) or those with branch op
297  // interface.
298  }
299 
300  LLVM_DEBUG(llvm::dbgs() << "Created " << nodes.size() << " nodes\n");
301 
302  // Add dependence edges between nodes which produce SSA values and their
303  // users. Load ops can be considered as the ones producing SSA values.
304  for (auto &idAndNode : nodes) {
305  const Node &node = idAndNode.second;
306  // Stores don't define SSA values, skip them.
307  if (!node.stores.empty())
308  continue;
309  Operation *opInst = node.op;
310  for (Value value : opInst->getResults()) {
311  for (Operation *user : value.getUsers()) {
312  // Ignore users outside of the block.
313  if (block.getParent()->findAncestorOpInRegion(*user)->getBlock() !=
314  &block)
315  continue;
316  SmallVector<AffineForOp, 4> loops;
317  getAffineForIVs(*user, &loops);
318  // Find the surrounding affine.for nested immediately within the
319  // block.
320  auto *it = llvm::find_if(loops, [&](AffineForOp loop) {
321  return loop->getBlock() == &block;
322  });
323  if (it == loops.end())
324  continue;
325  assert(forToNodeMap.count(*it) > 0 && "missing mapping");
326  unsigned userLoopNestId = forToNodeMap[*it];
327  addEdge(node.id, userLoopNestId, value);
328  }
329  }
330  }
331 
332  // Walk memref access lists and add graph edges between dependent nodes.
333  for (auto &memrefAndList : memrefAccesses) {
334  unsigned n = memrefAndList.second.size();
335  Value srcMemRef = memrefAndList.first;
336  // Add edges between all dependent pairs among the node IDs on this memref.
337  for (unsigned i = 0; i < n; ++i) {
338  unsigned srcId = memrefAndList.second[i];
339  Node *srcNode = getNode(srcId);
340  bool srcHasStoreOrFree =
341  srcNode->hasStore(srcMemRef) || srcNode->hasFree(srcMemRef);
342  for (unsigned j = i + 1; j < n; ++j) {
343  unsigned dstId = memrefAndList.second[j];
344  Node *dstNode = getNode(dstId);
345  bool dstHasStoreOrFree =
346  dstNode->hasStore(srcMemRef) || dstNode->hasFree(srcMemRef);
347  if (srcHasStoreOrFree || dstHasStoreOrFree)
348  addEdge(srcId, dstId, srcMemRef);
349  }
350  }
351  }
352  return true;
353 }

377 // Remove node 'id' (and its associated edges) from graph.
378 void MemRefDependenceGraph::removeNode(unsigned id) {
379  // Remove each edge in 'inEdges[id]'.
380  if (inEdges.count(id) > 0) {
381  SmallVector<Edge, 2> oldInEdges = inEdges[id];
382  for (auto &inEdge : oldInEdges) {
383  removeEdge(inEdge.id, id, inEdge.value);
384  }
385  }
386  // Remove each edge in 'outEdges[id]'.
387  if (outEdges.contains(id)) {
388  SmallVector<Edge, 2> oldOutEdges = outEdges[id];
389  for (auto &outEdge : oldOutEdges) {
390  removeEdge(id, outEdge.id, outEdge.value);
391  }
392  }
393  // Erase remaining node state.
394  inEdges.erase(id);
395  outEdges.erase(id);
396  nodes.erase(id);
397 }

418 // Returns true iff there is an edge from node 'srcId' to node 'dstId' which
419 // is for 'value' if non-null, or for any value otherwise. Returns false
420 // otherwise.
421 bool MemRefDependenceGraph::hasEdge(unsigned srcId, unsigned dstId,
422  Value value) const {
423  if (!outEdges.contains(srcId) || !inEdges.contains(dstId)) {
424  return false;
425  }
426  bool hasOutEdge = llvm::any_of(outEdges.lookup(srcId), [=](const Edge &edge) {
427  return edge.id == dstId && (!value || edge.value == value);
428  });
429  bool hasInEdge = llvm::any_of(inEdges.lookup(dstId), [=](const Edge &edge) {
430  return edge.id == srcId && (!value || edge.value == value);
431  });
432  return hasOutEdge && hasInEdge;
433 }

446 // Removes an edge from node 'srcId' to node 'dstId' for 'value'.
447 void MemRefDependenceGraph::removeEdge(unsigned srcId, unsigned dstId,
448  Value value) {
449  assert(inEdges.count(dstId) > 0);
450  assert(outEdges.count(srcId) > 0);
451  if (isa<MemRefType>(value.getType())) {
452  assert(memrefEdgeCount.count(value) > 0);
453  memrefEdgeCount[value]--;
454  }
455  // Remove 'srcId' from 'inEdges[dstId]'.
456  for (auto *it = inEdges[dstId].begin(); it != inEdges[dstId].end(); ++it) {
457  if ((*it).id == srcId && (*it).value == value) {
458  inEdges[dstId].erase(it);
459  break;
460  }
461  }
462  // Remove 'dstId' from 'outEdges[srcId]'.
463  for (auto *it = outEdges[srcId].begin(); it != outEdges[srcId].end(); ++it) {
464  if ((*it).id == dstId && (*it).value == value) {
465  outEdges[srcId].erase(it);
466  break;
467  }
468  }

471 // Returns true if there is a path in the dependence graph from node 'srcId'
472 // to node 'dstId'. Returns false otherwise. `srcId`, `dstId`, and the
473 // operations that the edges connected are expected to be from the same block.
474 bool MemRefDependenceGraph::hasDependencePath(unsigned srcId,
475  unsigned dstId) const {
476  // Worklist state is: <node-id, next-output-edge-index-to-visit>
477  SmallVector<std::pair<unsigned, unsigned>, 4> worklist;
478  worklist.push_back({srcId, 0});
479  Operation *dstOp = getNode(dstId)->op;
480  // Run DFS traversal to see if 'dstId' is reachable from 'srcId'.
481  while (!worklist.empty()) {
482  auto &idAndIndex = worklist.back();
483  // Return true if we have reached 'dstId'.
484  if (idAndIndex.first == dstId)
485  return true;
486  // Pop and continue if node has no out edges, or if all out edges have
487  // already been visited.
488  if (!outEdges.contains(idAndIndex.first) ||
489  idAndIndex.second == outEdges.lookup(idAndIndex.first).size()) {
490  worklist.pop_back();
491  continue;
492  }
493  // Get graph edge to traverse.
494  const Edge edge = outEdges.lookup(idAndIndex.first)[idAndIndex.second];
495  // Increment next output edge index for 'idAndIndex'.
496  ++idAndIndex.second;
497  // Add node at 'edge.id' to the worklist. We don't need to consider
498  // nodes that are "after" dstId in the containing block; one can't have a
499  // path to `dstId` from any of those nodes.
500  bool afterDst = dstOp->isBeforeInBlock(getNode(edge.id)->op);
501  if (!afterDst && edge.id != idAndIndex.first)
502  worklist.push_back({edge.id, 0});
503  }
504  return false;
505 }

545 // Computes and returns an insertion point operation, before which the
546 // the fused <srcId, dstId> loop nest can be inserted while preserving
547 // dependences. Returns nullptr if no such insertion point is found.
548 Operation *
549 MemRefDependenceGraph::getFusedLoopNestInsertionPoint(unsigned srcId,
550  unsigned dstId) const {
551  if (!outEdges.contains(srcId))
552  return getNode(dstId)->op;
553 
554  // Skip if there is any defining node of 'dstId' that depends on 'srcId'.
555  DenseSet<unsigned> definingNodes;
556  gatherDefiningNodes(dstId, definingNodes);
557  if (llvm::any_of(definingNodes,
558  [&](unsigned id) { return hasDependencePath(srcId, id); })) {
559  LLVM_DEBUG(llvm::dbgs()
560  << "Can't fuse: a defining op with a user in the dst "
561  "loop has dependence from the src loop\n");
562  return nullptr;
563  }
564 
565  // Build set of insts in range (srcId, dstId) which depend on 'srcId'.
566  SmallPtrSet<Operation *, 2> srcDepInsts;
567  for (auto &outEdge : outEdges.lookup(srcId))
568  if (outEdge.id != dstId)
569  srcDepInsts.insert(getNode(outEdge.id)->op);
570 
571  // Build set of insts in range (srcId, dstId) on which 'dstId' depends.
572  SmallPtrSet<Operation *, 2> dstDepInsts;
573  for (auto &inEdge : inEdges.lookup(dstId))
574  if (inEdge.id != srcId)
575  dstDepInsts.insert(getNode(inEdge.id)->op);
576 
577  Operation *srcNodeInst = getNode(srcId)->op;
578  Operation *dstNodeInst = getNode(dstId)->op;
579 
580  // Computing insertion point:
581  // *) Walk all operation positions in Block operation list in the
582  // range (src, dst). For each operation 'op' visited in this search:
583  // *) Store in 'firstSrcDepPos' the first position where 'op' has a
584  // dependence edge from 'srcNode'.
585  // *) Store in 'lastDstDepPost' the last position where 'op' has a
586  // dependence edge to 'dstNode'.
587  // *) Compare 'firstSrcDepPos' and 'lastDstDepPost' to determine the
588  // operation insertion point (or return null pointer if no such
589  // insertion point exists: 'firstSrcDepPos' <= 'lastDstDepPos').
590  SmallVector<Operation *, 2> depInsts;
591  std::optional<unsigned> firstSrcDepPos;
592  std::optional<unsigned> lastDstDepPos;
593  unsigned pos = 0;
594  for (Block::iterator it = std::next(Block::iterator(srcNodeInst));
595  it != Block::iterator(dstNodeInst); ++it) {
596  Operation *op = &(*it);
597  if (srcDepInsts.count(op) > 0 && firstSrcDepPos == std::nullopt)
598  firstSrcDepPos = pos;
599  if (dstDepInsts.count(op) > 0)
600  lastDstDepPos = pos;
601  depInsts.push_back(op);
602  ++pos;
603  }
604 
605  if (firstSrcDepPos.has_value()) {
606  if (lastDstDepPos.has_value()) {
607  if (*firstSrcDepPos <= *lastDstDepPos) {
608  // No valid insertion point exists which preserves dependences.
609  return nullptr;
610  }
611  }
612  // Return the insertion point at 'firstSrcDepPos'.
613  return depInsts[*firstSrcDepPos];
614  }
615  // No dependence targets in range (or only dst deps in range), return
616  // 'dstNodInst' insertion point.
617  return dstNodeInst;
618 }

620 // Updates edge mappings from node 'srcId' to node 'dstId' after fusing them,
621 // taking into account that:
622 // *) if 'removeSrcId' is true, 'srcId' will be removed after fusion,
623 // *) memrefs in 'privateMemRefs' has been replaced in node at 'dstId' by a
624 // private memref.
625 void MemRefDependenceGraph::updateEdges(unsigned srcId, unsigned dstId,
626  const DenseSet<Value> &privateMemRefs,
627  bool removeSrcId) {
628  // For each edge in 'inEdges[srcId]': add new edge remapping to 'dstId'.
629  if (inEdges.count(srcId) > 0) {
630  SmallVector<Edge, 2> oldInEdges = inEdges[srcId];
631  for (auto &inEdge : oldInEdges) {
632  // Add edge from 'inEdge.id' to 'dstId' if it's not a private memref.
633  if (!privateMemRefs.contains(inEdge.value))
634  addEdge(inEdge.id, dstId, inEdge.value);
635  }
636  }
637  // For each edge in 'outEdges[srcId]': remove edge from 'srcId' to 'dstId'.
638  // If 'srcId' is going to be removed, remap all the out edges to 'dstId'.
639  if (outEdges.count(srcId) > 0) {
640  SmallVector<Edge, 2> oldOutEdges = outEdges[srcId];
641  for (auto &outEdge : oldOutEdges) {
642  // Remove any out edges from 'srcId' to 'dstId' across memrefs.
643  if (outEdge.id == dstId)
644  removeEdge(srcId, outEdge.id, outEdge.value);
645  else if (removeSrcId) {
646  addEdge(dstId, outEdge.id, outEdge.value);
647  removeEdge(srcId, outEdge.id, outEdge.value);
648  }
649  }
650  }
651  // Remove any edges in 'inEdges[dstId]' on 'oldMemRef' (which is being
652  // replaced by a private memref). These edges could come from nodes
653  // other than 'srcId' which were removed in the previous step.
654  if (inEdges.count(dstId) > 0 && !privateMemRefs.empty()) {
655  SmallVector<Edge, 2> oldInEdges = inEdges[dstId];
656  for (auto &inEdge : oldInEdges)
657  if (privateMemRefs.count(inEdge.value) > 0)
658  removeEdge(inEdge.id, dstId, inEdge.value);
659  }
660 }

801 // Populates 'cst' with FlatAffineValueConstraints which represent slice bounds.
802 LogicalResult
803 ComputationSliceState::getAsConstraints(FlatAffineValueConstraints *cst) const {
804  assert(!lbOperands.empty());
805  // Adds src 'ivs' as dimension variables in 'cst'.
806  unsigned numDims = ivs.size();
807  // Adds operands (dst ivs and symbols) as symbols in 'cst'.
808  unsigned numSymbols = lbOperands[0].size();
809 
810  SmallVector<Value, 4> values(ivs);
811  // Append 'ivs' then 'operands' to 'values'.
812  values.append(lbOperands[0].begin(), lbOperands[0].end());
813  *cst = FlatAffineValueConstraints(numDims, numSymbols, 0, values);
814 
815  // Add loop bound constraints for values which are loop IVs of the destination
816  // of fusion and equality constraints for symbols which are constants.
817  for (unsigned i = numDims, end = values.size(); i < end; ++i) {
818  Value value = values[i];
819  assert(cst->containsVar(value) && "value expected to be present");
820  if (isValidSymbol(value)) {
821  // Check if the symbol is a constant.
822  if (std::optional<int64_t> cOp = getConstantIntValue(value))
823  cst->addBound(BoundType::EQ, value, cOp.value());
824  } else if (auto loop = getForInductionVarOwner(value)) {
825  if (failed(cst->addAffineForOpDomain(loop)))
826  return failure();
827  }
828  }
829 
830  // Add slices bounds on 'ivs' using maps 'lbs'/'ubs' with 'lbOperands[0]'
831  LogicalResult ret = cst->addSliceBounds(ivs, lbs, ubs, lbOperands[0]);
832  assert(succeeded(ret) &&
833  "should not fail as we never have semi-affine slice maps");
834  (void)ret;
835  return success();
836 }

868 /// Fast check to determine if the computation slice is maximal. Returns true if
869 /// each slice dimension maps to an existing dst dimension and both the src
870 /// and the dst loops for those dimensions have the same bounds. Returns false
871 /// if both the src and the dst loops don't have the same bounds. Returns
872 /// std::nullopt if none of the above can be proven.
873 std::optional<bool> ComputationSliceState::isSliceMaximalFastCheck() const {
874  assert(lbs.size() == ubs.size() && !lbs.empty() && !ivs.empty() &&
875  "Unexpected number of lbs, ubs and ivs in slice");
876 
877  for (unsigned i = 0, end = lbs.size(); i < end; ++i) {
878  AffineMap lbMap = lbs[i];
879  AffineMap ubMap = ubs[i];
880 
881  // Check if this slice is just an equality along this dimension.
882  if (!lbMap || !ubMap || lbMap.getNumResults() != 1 ||
883  ubMap.getNumResults() != 1 ||
884  lbMap.getResult(0) + 1 != ubMap.getResult(0) ||
885  // The condition above will be true for maps describing a single
886  // iteration (e.g., lbMap.getResult(0) = 0, ubMap.getResult(0) = 1).
887  // Make sure we skip those cases by checking that the lb result is not
888  // just a constant.
889  isa<AffineConstantExpr>(lbMap.getResult(0)))
890  return std::nullopt;
891 
892  // Limited support: we expect the lb result to be just a loop dimension for
893  // now.
894  AffineDimExpr result = dyn_cast<AffineDimExpr>(lbMap.getResult(0));
895  if (!result)
896  return std::nullopt;
897 
898  // Retrieve dst loop bounds.
899  AffineForOp dstLoop =
900  getForInductionVarOwner(lbOperands[i][result.getPosition()]);
901  if (!dstLoop)
902  return std::nullopt;
903  AffineMap dstLbMap = dstLoop.getLowerBoundMap();
904  AffineMap dstUbMap = dstLoop.getUpperBoundMap();
905 
906  // Retrieve src loop bounds.
907  AffineForOp srcLoop = getForInductionVarOwner(ivs[i]);
908  assert(srcLoop && "Expected affine for");
909  AffineMap srcLbMap = srcLoop.getLowerBoundMap();
910  AffineMap srcUbMap = srcLoop.getUpperBoundMap();
911 
912  // Limited support: we expect simple src and dst loops with a single
913  // constant component per bound for now.
914  if (srcLbMap.getNumResults() != 1 || srcUbMap.getNumResults() != 1 ||
915  dstLbMap.getNumResults() != 1 || dstUbMap.getNumResults() != 1)
916  return std::nullopt;
917 
918  AffineExpr srcLbResult = srcLbMap.getResult(0);
919  AffineExpr dstLbResult = dstLbMap.getResult(0);
920  AffineExpr srcUbResult = srcUbMap.getResult(0);
921  AffineExpr dstUbResult = dstUbMap.getResult(0);
922  if (!isa<AffineConstantExpr>(srcLbResult) ||
923  !isa<AffineConstantExpr>(srcUbResult) ||
924  !isa<AffineConstantExpr>(dstLbResult) ||
925  !isa<AffineConstantExpr>(dstUbResult))
926  return std::nullopt;
927 
928  // Check if src and dst loop bounds are the same. If not, we can guarantee
929  // that the slice is not maximal.
930  if (srcLbResult != dstLbResult || srcUbResult != dstUbResult ||
931  srcLoop.getStep() != dstLoop.getStep())
932  return false;
933  }
934 
935  return true;
936 }

938 /// Returns true if it is deterministically verified that the original iteration
939 /// space of the slice is contained within the new iteration space that is
940 /// created after fusing 'this' slice into its destination.
941 std::optional<bool> ComputationSliceState::isSliceValid() const {
942  // Fast check to determine if the slice is valid. If the following conditions
943  // are verified to be true, slice is declared valid by the fast check:
944  // 1. Each slice loop is a single iteration loop bound in terms of a single
945  // destination loop IV.
946  // 2. Loop bounds of the destination loop IV (from above) and those of the
947  // source loop IV are exactly the same.
948  // If the fast check is inconclusive or false, we proceed with a more
949  // expensive analysis.
950  // TODO: Store the result of the fast check, as it might be used again in
951  // `canRemoveSrcNodeAfterFusion`.
952  std::optional<bool> isValidFastCheck = isSliceMaximalFastCheck();
953  if (isValidFastCheck && *isValidFastCheck)
954  return true;
955 
956  // Create constraints for the source loop nest using which slice is computed.
957  FlatAffineValueConstraints srcConstraints;
958  // TODO: Store the source's domain to avoid computation at each depth.
959  if (failed(getSourceAsConstraints(srcConstraints))) {
960  LLVM_DEBUG(llvm::dbgs() << "Unable to compute source's domain\n");
961  return std::nullopt;
962  }
963  // As the set difference utility currently cannot handle symbols in its
964  // operands, validity of the slice cannot be determined.
965  if (srcConstraints.getNumSymbolVars() > 0) {
966  LLVM_DEBUG(llvm::dbgs() << "Cannot handle symbols in source domain\n");
967  return std::nullopt;
968  }
969  // TODO: Handle local vars in the source domains while using the 'projectOut'
970  // utility below. Currently, aligning is not done assuming that there will be
971  // no local vars in the source domain.
972  if (srcConstraints.getNumLocalVars() != 0) {
973  LLVM_DEBUG(llvm::dbgs() << "Cannot handle locals in source domain\n");
974  return std::nullopt;
975  }
976 
977  // Create constraints for the slice loop nest that would be created if the
978  // fusion succeeds.
979  FlatAffineValueConstraints sliceConstraints;
980  if (failed(getAsConstraints(&sliceConstraints))) {
981  LLVM_DEBUG(llvm::dbgs() << "Unable to compute slice's domain\n");
982  return std::nullopt;
983  }
984 
985  // Projecting out every dimension other than the 'ivs' to express slice's
986  // domain completely in terms of source's IVs.
987  sliceConstraints.projectOut(ivs.size(),
988  sliceConstraints.getNumVars() - ivs.size());
989 
990  LLVM_DEBUG(llvm::dbgs() << "Domain of the source of the slice:\n");
991  LLVM_DEBUG(srcConstraints.dump());
992  LLVM_DEBUG(llvm::dbgs() << "Domain of the slice if this fusion succeeds "
993  "(expressed in terms of its source's IVs):\n");
994  LLVM_DEBUG(sliceConstraints.dump());
995 
996  // TODO: Store 'srcSet' to avoid recalculating for each depth.
997  PresburgerSet srcSet(srcConstraints);
998  PresburgerSet sliceSet(sliceConstraints);
999  PresburgerSet diffSet = sliceSet.subtract(srcSet);
1000 
1001  if (!diffSet.isIntegerEmpty()) {
1002  LLVM_DEBUG(llvm::dbgs() << "Incorrect slice\n");
1003  return false;
1004  }
1005  return true;
1006 }

1008 /// Returns true if the computation slice encloses all the iterations of the
1009 /// sliced loop nest. Returns false if it does not. Returns std::nullopt if it
1010 /// cannot determine if the slice is maximal or not.
1011 std::optional<bool> ComputationSliceState::isMaximal() const {
1012  // Fast check to determine if the computation slice is maximal. If the result
1013  // is inconclusive, we proceed with a more expensive analysis.
1014  std::optional<bool> isMaximalFastCheck = isSliceMaximalFastCheck();
1015  if (isMaximalFastCheck)
1016  return isMaximalFastCheck;
1017 
1018  // Create constraints for the src loop nest being sliced.
1019  FlatAffineValueConstraints srcConstraints(/*numDims=*/ivs.size(),
1020  /*numSymbols=*/0,
1021  /*numLocals=*/0, ivs);
1022  for (Value iv : ivs) {
1023  AffineForOp loop = getForInductionVarOwner(iv);
1024  assert(loop && "Expected affine for");
1025  if (failed(srcConstraints.addAffineForOpDomain(loop)))
1026  return std::nullopt;
1027  }
1028 
1029  // Create constraints for the slice using the dst loop nest information. We
1030  // retrieve existing dst loops from the lbOperands.
1031  SmallVector<Value> consumerIVs;
1032  for (Value lbOp : lbOperands[0])
1033  if (getForInductionVarOwner(lbOp))
1034  consumerIVs.push_back(lbOp);
1035 
1036  // Add empty IV Values for those new loops that are not equalities and,
1037  // therefore, are not yet materialized in the IR.
1038  for (int i = consumerIVs.size(), end = ivs.size(); i < end; ++i)
1039  consumerIVs.push_back(Value());
1040 
1041  FlatAffineValueConstraints sliceConstraints(/*numDims=*/consumerIVs.size(),
1042  /*numSymbols=*/0,
1043  /*numLocals=*/0, consumerIVs);
1044 
1045  if (failed(sliceConstraints.addDomainFromSliceMaps(lbs, ubs, lbOperands[0])))
1046  return std::nullopt;
1047 
1048  if (srcConstraints.getNumDimVars() != sliceConstraints.getNumDimVars())
1049  // Constraint dims are different. The integer set difference can't be
1050  // computed so we don't know if the slice is maximal.
1051  return std::nullopt;
1052 
1053  // Compute the difference between the src loop nest and the slice integer
1054  // sets.
1055  PresburgerSet srcSet(srcConstraints);
1056  PresburgerSet sliceSet(sliceConstraints);
1057  PresburgerSet diffSet = srcSet.subtract(sliceSet);
1058  return diffSet.isIntegerEmpty();
1059 }

1065 std::optional<int64_t> MemRefRegion::getConstantBoundingSizeAndShape(
1066  SmallVectorImpl<int64_t> *shape, SmallVectorImpl<AffineMap> *lbs) const {
1067  auto memRefType = cast<MemRefType>(memref.getType());
1068  MLIRContext *context = memref.getContext();
1069  unsigned rank = memRefType.getRank();
1070  if (shape)
1071  shape->reserve(rank);
1072 
1073  assert(rank == cst.getNumDimVars() && "inconsistent memref region");
1074 
1075  // Use a copy of the region constraints that has upper/lower bounds for each
1076  // memref dimension with static size added to guard against potential
1077  // over-approximation from projection or union bounding box. We may not add
1078  // this on the region itself since they might just be redundant constraints
1079  // that will need non-trivials means to eliminate.
1080  FlatLinearValueConstraints cstWithShapeBounds(cst);
1081  for (unsigned r = 0; r < rank; r++) {
1082  cstWithShapeBounds.addBound(BoundType::LB, r, 0);
1083  int64_t dimSize = memRefType.getDimSize(r);
1084  if (ShapedType::isDynamic(dimSize))
1085  continue;
1086  cstWithShapeBounds.addBound(BoundType::UB, r, dimSize - 1);
1087  }
1088 
1089  // Find a constant upper bound on the extent of this memref region along
1090  // each dimension.
1091  int64_t numElements = 1;
1092  int64_t diffConstant;
1093  for (unsigned d = 0; d < rank; d++) {
1094  AffineMap lb;
1095  std::optional<int64_t> diff =
1096  cstWithShapeBounds.getConstantBoundOnDimSize(context, d, &lb);
1097  if (diff.has_value()) {
1098  diffConstant = *diff;
1099  assert(diffConstant >= 0 && "dim size bound cannot be negative");
1100  } else {
1101  // If no constant bound is found, then it can always be bound by the
1102  // memref's dim size if the latter has a constant size along this dim.
1103  auto dimSize = memRefType.getDimSize(d);
1104  if (ShapedType::isDynamic(dimSize))
1105  return std::nullopt;
1106  diffConstant = dimSize;
1107  // Lower bound becomes 0.
1108  lb = AffineMap::get(/*dimCount=*/0, cstWithShapeBounds.getNumSymbolVars(),
1109  /*result=*/getAffineConstantExpr(0, context));
1110  }

1112  // Populate outputs if available.
1113  if (lbs)
1114  lbs->push_back(lb);
1115  if (shape)
1116  shape->push_back(diffConstant);
1117  }
1118  return numElements;
1119 }

1145 /// Computes the memory region accessed by this memref with the region
1146 /// represented as constraints symbolic/parametric in 'loopDepth' loops
1147 /// surrounding opInst and any additional Function symbols.
1148 // For example, the memref region for this load operation at loopDepth = 1 will
1149 // be as below:
1150 //
1151 // affine.for %i = 0 to 32 {
1152 // affine.for %ii = %i to (d0) -> (d0 + 8) (%i) {
1153 // load %A[%ii]
1154 // }
1155 // }
1156 //
1157 // region: {memref = %A, write = false, {%i <= m0 <= %i + 7} }
1158 // The last field is a 2-d FlatAffineValueConstraints symbolic in %i.
1159 //
1160 // TODO: extend this to any other memref dereferencing ops
1161 // (dma_start, dma_wait).
1162 LogicalResult MemRefRegion::compute(Operation *op, unsigned loopDepth,
1163  const ComputationSliceState *sliceState,
1164  bool addMemRefDimBounds, bool dropLocalVars,
1165  bool dropOuterIvs) {
1166  assert((isa<AffineReadOpInterface, AffineWriteOpInterface>(op)) &&
1167  "affine read/write op expected");
1168 
1169  MemRefAccess access(op);
1170  memref = access.memref;
1171  write = access.isStore();
1172 
1173  unsigned rank = access.getRank();
1174 
1175  LLVM_DEBUG(llvm::dbgs() << "MemRefRegion::compute: " << *op
1176  << "\ndepth: " << loopDepth << "\n";);
1177 
1178  // 0-d memrefs.
1179  if (rank == 0) {
1180  SmallVector<Value, 4> ivs;
1181  getAffineIVs(*op, ivs);
1182  assert(loopDepth <= ivs.size() && "invalid 'loopDepth'");
1183  // The first 'loopDepth' IVs are symbols for this region.
1184  ivs.resize(loopDepth);
1185  // A 0-d memref has a 0-d region.
1186  cst = FlatAffineValueConstraints(rank, loopDepth, /*numLocals=*/0, ivs);
1187  return success();
1188  }
1189 
1190  // Build the constraints for this region.
1191  AffineValueMap accessValueMap;
1192  access.getAccessMap(&accessValueMap);
1193  AffineMap accessMap = accessValueMap.getAffineMap();
1194 
1195  unsigned numDims = accessMap.getNumDims();
1196  unsigned numSymbols = accessMap.getNumSymbols();
1197  unsigned numOperands = accessValueMap.getNumOperands();
1198  // Merge operands with slice operands.
1199  SmallVector<Value, 4> operands;
1200  operands.resize(numOperands);
1201  for (unsigned i = 0; i < numOperands; ++i)
1202  operands[i] = accessValueMap.getOperand(i);
1203 
1204  if (sliceState != nullptr) {
1205  operands.reserve(operands.size() + sliceState->lbOperands[0].size());
1206  // Append slice operands to 'operands' as symbols.
1207  for (auto extraOperand : sliceState->lbOperands[0]) {
1208  if (!llvm::is_contained(operands, extraOperand)) {
1209  operands.push_back(extraOperand);
1210  numSymbols++;
1211  }
1212  }
1213  }
1214  // We'll first associate the dims and symbols of the access map to the dims
1215  // and symbols resp. of cst. This will change below once cst is
1216  // fully constructed out.
1217  cst = FlatAffineValueConstraints(numDims, numSymbols, 0, operands);
1218 
1219  // Add equality constraints.
1220  // Add inequalities for loop lower/upper bounds.
1221  for (unsigned i = 0; i < numDims + numSymbols; ++i) {
1222  auto operand = operands[i];
1223  if (auto affineFor = getForInductionVarOwner(operand)) {
1224  // Note that cst can now have more dimensions than accessMap if the
1225  // bounds expressions involve outer loops or other symbols.
1226  // TODO: rewrite this to use getInstIndexSet; this way
1227  // conditionals will be handled when the latter supports it.
1228  if (failed(cst.addAffineForOpDomain(affineFor)))
1229  return failure();
1230  } else if (auto parallelOp = getAffineParallelInductionVarOwner(operand)) {
1231  if (failed(cst.addAffineParallelOpDomain(parallelOp)))
1232  return failure();
1233  } else if (isValidSymbol(operand)) {
1234  // Check if the symbol is a constant.
1235  Value symbol = operand;
1236  if (auto constVal = getConstantIntValue(symbol))
1237  cst.addBound(BoundType::EQ, symbol, constVal.value());
1238  } else {
1239  LLVM_DEBUG(llvm::dbgs() << "unknown affine dimensional value");
1240  return failure();
1241  }
1242  }
1243 
1244  // Add lower/upper bounds on loop IVs using bounds from 'sliceState'.
1245  if (sliceState != nullptr) {
1246  // Add dim and symbol slice operands.
1247  for (auto operand : sliceState->lbOperands[0]) {
1248  if (failed(cst.addInductionVarOrTerminalSymbol(operand)))
1249  return failure();
1250  }
1251  // Add upper/lower bounds from 'sliceState' to 'cst'.
1252  LogicalResult ret =
1253  cst.addSliceBounds(sliceState->ivs, sliceState->lbs, sliceState->ubs,
1254  sliceState->lbOperands[0]);
1255  assert(succeeded(ret) &&
1256  "should not fail as we never have semi-affine slice maps");
1257  (void)ret;
1258  }
1259 
1260  // Add access function equalities to connect loop IVs to data dimensions.
1261  if (failed(cst.composeMap(&accessValueMap))) {
1262  op->emitError("getMemRefRegion: compose affine map failed");
1263  LLVM_DEBUG(accessValueMap.getAffineMap().dump());
1264  return failure();
1265  }
1266 
1267  // Set all variables appearing after the first 'rank' variables as
1268  // symbolic variables - so that the ones corresponding to the memref
1269  // dimensions are the dimensional variables for the memref region.
1270  cst.setDimSymbolSeparation(cst.getNumDimAndSymbolVars() - rank);
1271 
1272  // Eliminate any loop IVs other than the outermost 'loopDepth' IVs, on which
1273  // this memref region is symbolic.
1274  SmallVector<Value, 4> enclosingIVs;
1275  getAffineIVs(*op, enclosingIVs);
1276  assert(loopDepth <= enclosingIVs.size() && "invalid loop depth");
1277  enclosingIVs.resize(loopDepth);
1278  SmallVector<Value, 4> vars;
1279  cst.getValues(cst.getNumDimVars(), cst.getNumDimAndSymbolVars(), &vars);
1280  for (auto en : llvm::enumerate(vars)) {
1281  if ((isAffineInductionVar(en.value())) &&
1282  !llvm::is_contained(enclosingIVs, en.value())) {
1283  if (dropOuterIvs) {
1284  cst.projectOut(en.value());
1285  } else {
1286  unsigned varPosition;
1287  cst.findVar(en.value(), &varPosition);
1288  auto varKind = cst.getVarKindAt(varPosition);
1289  varPosition -= cst.getNumDimVars();
1290  cst.convertToLocal(varKind, varPosition, varPosition + 1);
1291  }
1292  }
1293  }
1294 
1295  // Project out any local variables (these would have been added for any
1296  // mod/divs) if specified.
1297  if (dropLocalVars)
1298  cst.projectOut(cst.getNumDimAndSymbolVars(), cst.getNumLocalVars());
1299 
1300  // Constant fold any symbolic variables.
1301  cst.constantFoldVarRange(/*pos=*/cst.getNumDimVars(),
1302  /*num=*/cst.getNumSymbolVars());
1303 
1304  assert(cst.getNumDimVars() == rank && "unexpected MemRefRegion format");
1305 
1306  // Add upper/lower bounds for each memref dimension with static size
1307  // to guard against potential over-approximation from projection.
1308  // TODO: Support dynamic memref dimensions.
1309  if (addMemRefDimBounds) {
1310  auto memRefType = cast<MemRefType>(memref.getType());
1311  for (unsigned r = 0; r < rank; r++) {
1312  cst.addBound(BoundType::LB, /*pos=*/r, /*value=*/0);
1313  if (memRefType.isDynamicDim(r))
1314  continue;
1315  cst.addBound(BoundType::UB, /*pos=*/r, memRefType.getDimSize(r) - 1);
1316  }
1317  }
1318  cst.removeTrivialRedundancy();
1319 
1320  LLVM_DEBUG(llvm::dbgs() << "Memory region:\n");
1321  LLVM_DEBUG(cst.dump());
1322  return success();
1323 }

1536 /// Computes in 'sliceUnion' the union of all slice bounds computed at
1537 /// 'loopDepth' between all dependent pairs of ops in 'opsA' and 'opsB', and
1538 /// then verifies if it is valid. Returns 'SliceComputationResult::Success' if
1539 /// union was computed correctly, an appropriate failure otherwise.
1540 SliceComputationResult
1541 mlir::affine::computeSliceUnion(ArrayRef<Operation *> opsA,
1542  ArrayRef<Operation *> opsB, unsigned loopDepth,
1543  unsigned numCommonLoops, bool isBackwardSlice,
1544  ComputationSliceState *sliceUnion) {
1545  // Compute the union of slice bounds between all pairs in 'opsA' and
1546  // 'opsB' in 'sliceUnionCst'.
1547  FlatAffineValueConstraints sliceUnionCst;
1548  assert(sliceUnionCst.getNumDimAndSymbolVars() == 0);
1549  std::vector<std::pair<Operation *, Operation *>> dependentOpPairs;
1550  MemRefAccess srcAccess;
1551  MemRefAccess dstAccess;
1552  for (Operation *a : opsA) {
1553  srcAccess = MemRefAccess(a);
1554  for (Operation *b : opsB) {
1555  dstAccess = MemRefAccess(b);
1556  if (srcAccess.memref != dstAccess.memref)
1557  continue;
1558  // Check if 'loopDepth' exceeds nesting depth of src/dst ops.
1559  if ((!isBackwardSlice && loopDepth > getNestingDepth(a)) ||
1560  (isBackwardSlice && loopDepth > getNestingDepth(b))) {
1561  LLVM_DEBUG(llvm::dbgs() << "Invalid loop depth\n");
1562  return SliceComputationResult::GenericFailure;
1563  }
1564 
1565  bool readReadAccesses = isa<AffineReadOpInterface>(srcAccess.opInst) &&
1566  isa<AffineReadOpInterface>(dstAccess.opInst);
1567  FlatAffineValueConstraints dependenceConstraints;
1568  // Check dependence between 'srcAccess' and 'dstAccess'.
1569  DependenceResult result = checkMemrefAccessDependence(
1570  srcAccess, dstAccess, /*loopDepth=*/numCommonLoops + 1,
1571  &dependenceConstraints, /*dependenceComponents=*/nullptr,
1572  /*allowRAR=*/readReadAccesses);
1573  if (result.value == DependenceResult::Failure) {
1574  LLVM_DEBUG(llvm::dbgs() << "Dependence check failed\n");
1575  return SliceComputationResult::GenericFailure;
1576  }
1577  if (result.value == DependenceResult::NoDependence)
1578  continue;
1579  dependentOpPairs.emplace_back(a, b);
1580 
1581  // Compute slice bounds for 'srcAccess' and 'dstAccess'.
1582  ComputationSliceState tmpSliceState;
1583  getComputationSliceState(a, b, dependenceConstraints, loopDepth,
1584  isBackwardSlice, &tmpSliceState);
1585 
1586  if (sliceUnionCst.getNumDimAndSymbolVars() == 0) {
1587  // Initialize 'sliceUnionCst' with the bounds computed in previous step.
1588  if (failed(tmpSliceState.getAsConstraints(&sliceUnionCst))) {
1589  LLVM_DEBUG(llvm::dbgs()
1590  << "Unable to compute slice bound constraints\n");
1591  return SliceComputationResult::GenericFailure;
1592  }
1593  assert(sliceUnionCst.getNumDimAndSymbolVars() > 0);
1594  continue;
1595  }
1596 
1597  // Compute constraints for 'tmpSliceState' in 'tmpSliceCst'.
1598  FlatAffineValueConstraints tmpSliceCst;
1599  if (failed(tmpSliceState.getAsConstraints(&tmpSliceCst))) {
1600  LLVM_DEBUG(llvm::dbgs()
1601  << "Unable to compute slice bound constraints\n");
1602  return SliceComputationResult::GenericFailure;
1603  }
1604 
1605  // Align coordinate spaces of 'sliceUnionCst' and 'tmpSliceCst' if needed.
1606  if (!sliceUnionCst.areVarsAlignedWithOther(tmpSliceCst)) {
1607 
1608  // Pre-constraint var alignment: record loop IVs used in each constraint
1609  // system.
1610  SmallPtrSet<Value, 8> sliceUnionIVs;
1611  for (unsigned k = 0, l = sliceUnionCst.getNumDimVars(); k < l; ++k)
1612  sliceUnionIVs.insert(sliceUnionCst.getValue(k));
1613  SmallPtrSet<Value, 8> tmpSliceIVs;
1614  for (unsigned k = 0, l = tmpSliceCst.getNumDimVars(); k < l; ++k)
1615  tmpSliceIVs.insert(tmpSliceCst.getValue(k));
1616 
1617  sliceUnionCst.mergeAndAlignVarsWithOther(/*offset=*/0, &tmpSliceCst);
1618 
1619  // Post-constraint var alignment: add loop IV bounds missing after
1620  // var alignment to constraint systems. This can occur if one constraint
1621  // system uses an loop IV that is not used by the other. The call
1622  // to unionBoundingBox below expects constraints for each Loop IV, even
1623  // if they are the unsliced full loop bounds added here.
1624  if (failed(addMissingLoopIVBounds(sliceUnionIVs, &sliceUnionCst)))
1625  return SliceComputationResult::GenericFailure;
1626  if (failed(addMissingLoopIVBounds(tmpSliceIVs, &tmpSliceCst)))
1627  return SliceComputationResult::GenericFailure;
1628  }
1629  // Compute union bounding box of 'sliceUnionCst' and 'tmpSliceCst'.
1630  if (sliceUnionCst.getNumLocalVars() > 0 ||
1631  tmpSliceCst.getNumLocalVars() > 0 ||
1632  failed(sliceUnionCst.unionBoundingBox(tmpSliceCst))) {
1633  LLVM_DEBUG(llvm::dbgs()
1634  << "Unable to compute union bounding box of slice bounds\n");
1635  return SliceComputationResult::GenericFailure;
1636  }
1637  }
1638  }
1639 
1640  // Empty union.
1641  if (sliceUnionCst.getNumDimAndSymbolVars() == 0) {
1642  LLVM_DEBUG(llvm::dbgs() << "empty slice union - unexpected\n");
1643  return SliceComputationResult::GenericFailure;
1644  }
1645 
1646  // Gather loops surrounding ops from loop nest where slice will be inserted.
1647  SmallVector<Operation *, 4> ops;
1648  for (auto &dep : dependentOpPairs) {
1649  ops.push_back(isBackwardSlice ? dep.second : dep.first);
1650  }

1657  }

1660  unsigned numSliceLoopIVs = sliceUnionCst.getNumDimVars();

1662  // Convert any dst loop IVs which are symbol variables to dim variables.
1663  sliceUnionCst.convertLoopIVSymbolsToDims();

1665  sliceUnion->lbs.resize(numSliceLoopIVs, AffineMap());
1666  sliceUnion->ubs.resize(numSliceLoopIVs, AffineMap());
1667 
1668  // Get slice bounds from slice union constraints 'sliceUnionCst'.
1669  sliceUnionCst.getSliceBounds(/*offset=*/0, numSliceLoopIVs,
1670  opsA[0]->getContext(), &sliceUnion->lbs,
1671  &sliceUnion->ubs);
1672 
1673  // Add slice bound operands of union.
1674  SmallVector<Value, 4> sliceBoundOperands;
1675  sliceUnionCst.getValues(numSliceLoopIVs,
1676  sliceUnionCst.getNumDimAndSymbolVars(),
1677  &sliceBoundOperands);
1678 
1679  // Copy src loop IVs from 'sliceUnionCst' to 'sliceUnion'.
1680  sliceUnion->ivs.clear();
1681  sliceUnionCst.getValues(0, numSliceLoopIVs, &sliceUnion->ivs);
1682

1684  // slices, while at the end for backward slices.
1685  sliceUnion->insertPoint =
1686  isBackwardSlice
1687  ? surroundingLoops[loopDepth - 1].getBody()->begin()
1688  : std::prev(surroundingLoops[loopDepth - 1].getBody()->end());
1689 
1690  // Give each bound its own copy of 'sliceBoundOperands' for subsequent
1691  // canonicalization.
1692  sliceUnion->lbOperands.resize(numSliceLoopIVs, sliceBoundOperands);
1693  sliceUnion->ubOperands.resize(numSliceLoopIVs, sliceBoundOperands);
1694

1696  // that the slice is valid, otherwise return appropriate failure status.
1697  std::optional<bool> isSliceValid = sliceUnion->isSliceValid();
1698  if (!isSliceValid) {
1699  LLVM_DEBUG(llvm::dbgs() << "Cannot determine if the slice is valid\n");
1700  return SliceComputationResult::GenericFailure;
1701  }
1702  if (!*isSliceValid)
1703  return SliceComputationResult::IncorrectSliceFailure;
1704 
1705  return SliceComputationResult::Success;

1725 // Builds a map 'tripCountMap' from AffineForOp to constant trip count for loop
1726 // nest surrounding represented by slice loop bounds in 'slice'. Returns true
1727 // on success, false otherwise (if a non-constant trip count was encountered).
1728 // TODO: Make this work with non-unit step loops.
1729 bool mlir::affine::buildSliceTripCountMap(
1730  const ComputationSliceState &slice,
1731  llvm::SmallDenseMap<Operation *, uint64_t, 8> *tripCountMap) {
1732  unsigned numSrcLoopIVs = slice.ivs.size();
1733  // Populate map from AffineForOp -> trip count
1734  for (unsigned i = 0; i < numSrcLoopIVs; ++i) {
1735  AffineForOp forOp = getForInductionVarOwner(slice.ivs[i]);
1736  auto *op = forOp.getOperation();
1737  AffineMap lbMap = slice.lbs[i];
1738  AffineMap ubMap = slice.ubs[i];
1739  // If lower or upper bound maps are null or provide no results, it implies
1740  // that source loop was not at all sliced, and the entire loop will be a
1741  // part of the slice.
1742  if (!lbMap || lbMap.getNumResults() == 0 || !ubMap ||
1743  ubMap.getNumResults() == 0) {
1744  // The iteration of src loop IV 'i' was not sliced. Use full loop bounds.
1745  if (forOp.hasConstantLowerBound() && forOp.hasConstantUpperBound()) {
1746  (*tripCountMap)[op] =
1747  forOp.getConstantUpperBound() - forOp.getConstantLowerBound();
1748  continue;
1749  }
1750  std::optional<uint64_t> maybeConstTripCount = getConstantTripCount(forOp);
1751  if (maybeConstTripCount.has_value()) {
1752  (*tripCountMap)[op] = *maybeConstTripCount;
1753  continue;
1754  }
1755  return false;
1756  }
1757  std::optional<uint64_t> tripCount = getConstDifference(lbMap, ubMap);
1758  // Slice bounds are created with a constant ub - lb difference.
1759  if (!tripCount.has_value())
1760  return false;
1761  (*tripCountMap)[op] = *tripCount;
1762  }
1763  return true;
1764 }

1777 // Computes slice bounds by projecting out any loop IVs from
1778 // 'dependenceConstraints' at depth greater than 'loopDepth', and computes slice
1779 // bounds in 'sliceState' which represent the one loop nest's IVs in terms of
1780 // the other loop nest's IVs, symbols and constants (using 'isBackwardsSlice').
1781 void mlir::affine::getComputationSliceState(
1782  Operation *depSourceOp, Operation *depSinkOp,
1783  const FlatAffineValueConstraints &dependenceConstraints, unsigned loopDepth,
1784  bool isBackwardSlice, ComputationSliceState *sliceState) {
1785  // Get loop nest surrounding src operation.
1786  SmallVector<AffineForOp, 4> srcLoopIVs;
1787  getAffineForIVs(*depSourceOp, &srcLoopIVs);
1788  unsigned numSrcLoopIVs = srcLoopIVs.size();
1789 
1790  // Get loop nest surrounding dst operation.
1791  SmallVector<AffineForOp, 4> dstLoopIVs;
1792  getAffineForIVs(*depSinkOp, &dstLoopIVs);
1793  unsigned numDstLoopIVs = dstLoopIVs.size();
1794 
1795  assert((!isBackwardSlice && loopDepth <= numSrcLoopIVs) ||
1796  (isBackwardSlice && loopDepth <= numDstLoopIVs));
1797 
1798  // Project out dimensions other than those up to 'loopDepth'.
1799  unsigned pos = isBackwardSlice ? numSrcLoopIVs + loopDepth : loopDepth;
1800  unsigned num =
1801  isBackwardSlice ? numDstLoopIVs - loopDepth : numSrcLoopIVs - loopDepth;
1802  FlatAffineValueConstraints sliceCst(dependenceConstraints);
1803  sliceCst.projectOut(pos, num);
1804 
1805  // Add slice loop IV values to 'sliceState'.
1806  unsigned offset = isBackwardSlice ? 0 : loopDepth;
1807  unsigned numSliceLoopIVs = isBackwardSlice ? numSrcLoopIVs : numDstLoopIVs;
1808  sliceCst.getValues(offset, offset + numSliceLoopIVs, &sliceState->ivs);
1809 
1810  // Set up lower/upper bound affine maps for the slice.
1811  sliceState->lbs.resize(numSliceLoopIVs, AffineMap());
1812  sliceState->ubs.resize(numSliceLoopIVs, AffineMap());
1813 
1814  // Get bounds for slice IVs in terms of other IVs, symbols, and constants.
1815  sliceCst.getSliceBounds(offset, numSliceLoopIVs, depSourceOp->getContext(),
1816  &sliceState->lbs, &sliceState->ubs);
1817 
1818  // Set up bound operands for the slice's lower and upper bounds.
1819  SmallVector<Value, 4> sliceBoundOperands;
1820  unsigned numDimsAndSymbols = sliceCst.getNumDimAndSymbolVars();
1821  for (unsigned i = 0; i < numDimsAndSymbols; ++i) {
1822  if (i < offset || i >= offset + numSliceLoopIVs)
1823  sliceBoundOperands.push_back(sliceCst.getValue(i));
1824  }
1825 
1826  // Give each bound its own copy of 'sliceBoundOperands' for subsequent
1827  // canonicalization.
1828  sliceState->lbOperands.resize(numSliceLoopIVs, sliceBoundOperands);
1829  sliceState->ubOperands.resize(numSliceLoopIVs, sliceBoundOperands);
1830 
1831  // Set destination loop nest insertion point to block start at 'dstLoopDepth'.
1832  sliceState->insertPoint =
1833  isBackwardSlice ? dstLoopIVs[loopDepth - 1].getBody()->begin()
1834  : std::prev(srcLoopIVs[loopDepth - 1].getBody()->end());
1835 
1836  llvm::SmallDenseSet<Value, 8> sequentialLoops;
1837  if (isa<AffineReadOpInterface>(depSourceOp) &&
1838  isa<AffineReadOpInterface>(depSinkOp)) {
1839  // For read-read access pairs, clear any slice bounds on sequential loops.
1840  // Get sequential loops in loop nest rooted at 'srcLoopIVs[0]'.
1841  getSequentialLoops(isBackwardSlice ? srcLoopIVs[0] : dstLoopIVs[0],
1842  &sequentialLoops);
1843  }
1844  auto getSliceLoop = [&](unsigned i) {
1845  return isBackwardSlice ? srcLoopIVs[i] : dstLoopIVs[i];
1846  };
1847  auto isInnermostInsertion = [&]() {
1848  return (isBackwardSlice ? loopDepth >= srcLoopIVs.size()
1849  : loopDepth >= dstLoopIVs.size());
1850  };
1851  llvm::SmallDenseMap<Operation *, uint64_t, 8> sliceTripCountMap;
1852  auto srcIsUnitSlice = [&]() {
1853  return (buildSliceTripCountMap(*sliceState, &sliceTripCountMap) &&
1854  (getSliceIterationCount(sliceTripCountMap) == 1));
1855  };
1856  // Clear all sliced loop bounds beginning at the first sequential loop, or
1857  // first loop with a slice fusion barrier attribute..
1858 
1859  for (unsigned i = 0; i < numSliceLoopIVs; ++i) {
1860  Value iv = getSliceLoop(i).getInductionVar();
1861  if (sequentialLoops.count(iv) == 0 &&
1862  getSliceLoop(i)->getAttr(kSliceFusionBarrierAttrName) == nullptr)
1863  continue;
1864  // Skip reset of bounds of reduction loop inserted in the destination loop
1865  // that meets the following conditions:
1866  // 1. Slice is single trip count.
1867  // 2. Loop bounds of the source and destination match.
1868  // 3. Is being inserted at the innermost insertion point.
1869  std::optional<bool> isMaximal = sliceState->isMaximal();
1870  if (isLoopParallelAndContainsReduction(getSliceLoop(i)) &&
1871  isInnermostInsertion() && srcIsUnitSlice() && isMaximal && *isMaximal)
1872  continue;
1873  for (unsigned j = i; j < numSliceLoopIVs; ++j) {
1874  sliceState->lbs[j] = AffineMap();
1875  sliceState->ubs[j] = AffineMap();
1876  }
1877  break;
1878  }
1879 }

1902  SmallVector<AffineForOp, 4> dstLoopIVs;
1903  getAffineForIVs(*dstOpInst, &dstLoopIVs);
1904  unsigned dstLoopIVsSize = dstLoopIVs.size();

1906  dstOpInst->emitError("invalid destination loop depth");
1907  return AffineForOp();

1911  SmallVector<unsigned, 4> positions;
1912  // TODO: This code is incorrect since srcLoopIVs can be 0-d.
1913  findInstPosition(srcOpInst, srcLoopIVs[0]->getBlock(), &positions);
1914

1916  // of the loop at 'dstLoopDepth' in 'dstLoopIVs'.
1917  auto dstAffineForOp = dstLoopIVs[dstLoopDepth - 1];
1918  OpBuilder b(dstAffineForOp.getBody(), dstAffineForOp.getBody()->begin());

1920  cast<AffineForOp>(b.clone(*srcLoopIVs[0].getOperation()));

1932  unsigned sliceLoopLimit = dstLoopDepth + numSrcLoopIVs;

2158 // bounded by the same lower and upper bound.

2168 // * If `isMin`: r_i >= opBound
2169 // * If `isMax`: r_i <= opBound
2170 // If this is the case, ub(op) == lb(op).
2171 // 4. Replace `op` with `opBound`.
2172 //
2173 // In summary, the following constraints are added throughout this function.
2174 // Note: `invar` are dimensions added by the caller to express the invariants.
2175 // (Showing only the case where `isMin`.)
2176 //
2177 // invar | op | opBound | r_i | extra syms... | const | eq/ineq
2178 // ------+-------+---------+-----+---------------+-------+-------------------
2179 // (various eq./ineq. constraining `invar`, added by the caller)
2180 // ... | 0 | 0 | 0 | 0 | ... | ...
2181 // ------+-------+---------+-----+---------------+-------+-------------------
2182 // (various ineq. constraining `op` in terms of `op` operands (`invar` and
2183 // extra `op` operands "extra syms" that are not in `invar`)).
2184 // ... | -1 | 0 | 0 | ... | ... | >= 0
2185 // ------+-------+---------+-----+---------------+-------+-------------------
2186 // (set `opBound` to `op` upper bound in terms of `invar` and "extra syms")
2187 // ... | 0 | -1 | 0 | ... | ... | = 0
2188 // ------+-------+---------+-----+---------------+-------+-------------------
2189 // (for each `op` map result r_i: set r_i to corresponding map result,
2190 // prove that r_i >= minOpUb via contradiction)
2191 // ... | 0 | 0 | -1 | ... | ... | = 0
2192 // 0 | 0 | 1 | -1 | 0 | -1 | >= 0
2193 //
2194 FailureOr<AffineValueMap> mlir::affine::simplifyConstrainedMinMaxOp(
2195  Operation *op, FlatAffineValueConstraints constraints) {
2196  bool isMin = isa<AffineMinOp>(op);
2197  assert((isMin || isa<AffineMaxOp>(op)) && "expect AffineMin/MaxOp");
2198  MLIRContext *ctx = op->getContext();
2199  Builder builder(ctx);
2200  AffineMap map =
2201  isMin ? cast<AffineMinOp>(op).getMap() : cast<AffineMaxOp>(op).getMap();
2202  ValueRange operands = op->getOperands();
2203  unsigned numResults = map.getNumResults();
2204 
2205  // Add a few extra dimensions.
2206  unsigned dimOp = constraints.appendDimVar(); // `op`
2207  unsigned dimOpBound = constraints.appendDimVar(); // `op` lower/upper bound
2208  unsigned resultDimStart = constraints.appendDimVar(/*num=*/numResults);
2209 
2210  // Add an inequality for each result expr_i of map:
2211  // isMin: op <= expr_i, !isMin: op >= expr_i
2212  auto boundType = isMin ? BoundType::UB : BoundType::LB;
2213  // Upper bounds are exclusive, so add 1. (`affine.min` ops are inclusive.)
2214  AffineMap mapLbUb = isMin ? addConstToResults(map, 1) : map;
2215  if (failed(
2216  alignAndAddBound(constraints, boundType, dimOp, mapLbUb, operands)))
2217  return failure();
2218 
2219  // Try to compute a lower/upper bound for op, expressed in terms of the other
2220  // `dims` and extra symbols.
2221  SmallVector<AffineMap> opLb(1), opUb(1);
2222  constraints.getSliceBounds(dimOp, 1, ctx, &opLb, &opUb);
2223  AffineMap sliceBound = isMin ? opUb[0] : opLb[0];
2224  // TODO: `getSliceBounds` may return multiple bounds at the moment. This is
2225  // a TODO of `getSliceBounds` and not handled here.
2226  if (!sliceBound || sliceBound.getNumResults() != 1)
2227  return failure(); // No or multiple bounds found.
2228  // Recover the inclusive UB in the case of an `affine.min`.
2229  AffineMap boundMap = isMin ? addConstToResults(sliceBound, -1) : sliceBound;
2230 
2231  // Add an equality: Set dimOpBound to computed bound.
2232  // Add back dimension for op. (Was removed by `getSliceBounds`.)
2233  AffineMap alignedBoundMap = boundMap.shiftDims(/*shift=*/1, /*offset=*/dimOp);
2234  if (failed(constraints.addBound(BoundType::EQ, dimOpBound, alignedBoundMap)))
2235  return failure();
2236 
2237  // If the constraint system is empty, there is an inconsistency. (E.g., this
2238  // can happen if loop lb > ub.)
2239  if (constraints.isEmpty())
2240  return failure();
2241 
2242  // In the case of `isMin` (`!isMin` is inversed):
2243  // Prove that each result of `map` has a lower bound that is equal to (or
2244  // greater than) the upper bound of `op` (`dimOpBound`). In that case, `op`
2245  // can be replaced with the bound. I.e., prove that for each result
2246  // expr_i (represented by dimension r_i):
2247  //
2248  // r_i >= opBound
2249  //
2250  // To prove this inequality, add its negation to the constraint set and prove
2251  // that the constraint set is empty.
2252  for (unsigned i = resultDimStart; i < resultDimStart + numResults; ++i) {
2253  FlatAffineValueConstraints newConstr(constraints);
2254 
2255  // Add an equality: r_i = expr_i
2256  // Note: These equalities could have been added earlier and used to express
2257  // minOp <= expr_i. However, then we run the risk that `getSliceBounds`
2258  // computes minOpUb in terms of r_i dims, which is not desired.
2259  if (failed(alignAndAddBound(newConstr, BoundType::EQ, i,
2260  map.getSubMap({i - resultDimStart}), operands)))
2261  return failure();
2262 
2263  // If `isMin`: Add inequality: r_i < opBound
2264  // equiv.: opBound - r_i - 1 >= 0
2265  // If `!isMin`: Add inequality: r_i > opBound
2266  // equiv.: -opBound + r_i - 1 >= 0
2267  SmallVector<int64_t> ineq(newConstr.getNumCols(), 0);
2268  ineq[dimOpBound] = isMin ? 1 : -1;
2269  ineq[i] = isMin ? -1 : 1;
2270  ineq[newConstr.getNumCols() - 1] = -1;
2271  newConstr.addInequality(ineq);
2272  if (!newConstr.isEmpty())
2273  return failure();
2274  }
2275 
2276  // Lower and upper bound of `op` are equal. Replace `minOp` with its bound.
2277  AffineMap newMap = alignedBoundMap;
2278  SmallVector<Value> newOperands;
2279  unpackOptionalValues(constraints.getMaybeValues(), newOperands);
2280  // If dims/symbols have known constant values, use those in order to simplify
2281  // the affine map further.
2282  for (int64_t i = 0, e = constraints.getNumDimAndSymbolVars(); i < e; ++i) {
2283  // Skip unused operands and operands that are already constants.
2284  if (!newOperands[i] || getConstantIntValue(newOperands[i]))
2285  continue;
2286  if (auto bound = constraints.getConstantBound64(BoundType::EQ, i)) {
2287  AffineExpr expr =
2288  i < newMap.getNumDims()
2289  ? builder.getAffineDimExpr(i)
2290  : builder.getAffineSymbolExpr(i - newMap.getNumDims());
2291  newMap = newMap.replace(expr, builder.getAffineConstantExpr(*bound),
2292  newMap.getNumDims(), newMap.getNumSymbols());
2293  }
2294  }
2295  affine::canonicalizeMapAndOperands(&newMap, &newOperands);
2296  return AffineValueMap(newMap, newOperands);
2297 }
