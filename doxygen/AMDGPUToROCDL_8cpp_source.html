69 /// Returns the linear index used to access an element in the memref.
70 static Value getLinearIndexI32(ConversionPatternRewriter &rewriter,
71  Location loc, MemRefDescriptor &memRefDescriptor,
72  ValueRange indices, ArrayRef<int64_t> strides) {
73  IntegerType i32 = rewriter.getI32Type();
74  Value index;
75  for (auto [i, increment, stride] : llvm::enumerate(indices, strides)) {
76  if (stride != 1) { // Skip if stride is 1.
77  Value strideValue =
78  ShapedType::isDynamic(stride)
79  ? convertUnsignedToI32(rewriter, loc,
80  memRefDescriptor.stride(rewriter, loc, i))
81  : LLVM::ConstantOp::create(rewriter, loc, i32, stride);
82  increment = LLVM::MulOp::create(rewriter, loc, increment, strideValue);
83  }
84  index = index ? LLVM::AddOp::create(rewriter, loc, index, increment)
85  : increment;
86  }
87  return index ? index : createI32Constant(rewriter, loc, 0);
88 }

90 /// Compute the contents of the `num_records` field for a given memref
91 /// descriptor - that is, the number of bytes that's one element past the
92 /// greatest possible valid index into the memref.
93 static Value getNumRecords(ConversionPatternRewriter &rewriter, Location loc,
94  MemRefType memrefType,
95  MemRefDescriptor &memrefDescriptor,
96  ArrayRef<int64_t> strides,
97  uint32_t elementByteWidth) {
98  if (memrefType.hasStaticShape() &&
99  !llvm::any_of(strides, ShapedType::isDynamic)) {
100  int64_t size = memrefType.getRank() == 0 ? 1 : 0;
101  ArrayRef<int64_t> shape = memrefType.getShape();
102  for (uint32_t i = 0, e = memrefType.getRank(); i < e; ++i)
103  size = std::max(shape[i] * strides[i], size);
104  size = size * elementByteWidth;
105  assert(size < std::numeric_limits<uint32_t>::max() &&
106  "the memref buffer is too large");
107  return createI32Constant(rewriter, loc, static_cast<int32_t>(size));
108  }
109  Value maxIndex;
110  for (uint32_t i = 0, e = memrefType.getRank(); i < e; ++i) {
111  Value size = memrefDescriptor.size(rewriter, loc, i);
112  Value stride = memrefDescriptor.stride(rewriter, loc, i);
113  Value maxThisDim = LLVM::MulOp::create(rewriter, loc, size, stride);
114  maxIndex = maxIndex
115  ? LLVM::UMaxOp::create(rewriter, loc, maxIndex, maxThisDim)
116  : maxThisDim;
117  }
118  Value maxIndexI32 = convertUnsignedToI32(rewriter, loc, maxIndex);
119  Value byteWidthConst = createI32Constant(rewriter, loc, elementByteWidth);
120  return LLVM::MulOp::create(rewriter, loc, maxIndexI32, byteWidthConst);
121 }

123 static Value makeBufferRsrc(ConversionPatternRewriter &rewriter, Location loc,
124  Value basePointer, Value numRecords,
125  bool boundsCheck, amdgpu::Chipset chipset,
126  Value cacheSwizzleStride = nullptr,
127  unsigned addressSpace = 8) {
128  // The stride value is generally 0. However, on MI-300 and onward, you can
129  // enable a cache swizzling mode by setting bit 14 of the stride field
130  // and setting that stride to a cache stride.
131  Type i16 = rewriter.getI16Type();
132  Value stride;
133  if (chipset.majorVersion == 9 && chipset >= kGfx942 && cacheSwizzleStride) {
134  Value cacheStrideZext =
135  LLVM::ZExtOp::create(rewriter, loc, i16, cacheSwizzleStride);
136  Value swizzleBit = LLVM::ConstantOp::create(
137  rewriter, loc, i16, rewriter.getI16IntegerAttr(1 << 14));
138  stride = LLVM::OrOp::create(rewriter, loc, cacheStrideZext, swizzleBit,
139  /*isDisjoint=*/true);
140  } else {
141  stride = LLVM::ConstantOp::create(rewriter, loc, i16,
142  rewriter.getI16IntegerAttr(0));
143  }
144  // Get the number of elements.
145  // Flag word:
146  // bits 0-11: dst sel, ignored by these intrinsics
147  // bits 12-14: data format (ignored, must be nonzero, 7=float)
148  // bits 15-18: data format (ignored, must be nonzero, 4=32bit)
149  // bit 19: In nested heap (0 here)
150  // bit 20: Behavior on unmap (0 means "return 0 / ignore")
151  // bits 21-22: Index stride for swizzles (N/A)
152  // bit 23: Add thread ID (0)
153  // bit 24: Reserved to 1 (RDNA) or 0 (CDNA)
154  // bits 25-26: Reserved (0)
155  // bit 27: Buffer is non-volatile (CDNA only)
156  // bits 28-29: Out of bounds select (0 = structured, 1 = check index, 2 =
157  // none, 3 = either swizzles or testing against offset field) RDNA only
158  // bits 30-31: Type (must be 0)
159  uint32_t flags = (7 << 12) | (4 << 15);
160  if (chipset.majorVersion >= 10) {
161  flags |= (1 << 24);
162  uint32_t oob = boundsCheck ? 3 : 2;
163  flags |= (oob << 28);
164  }
165  Value flagsConst = createI32Constant(rewriter, loc, flags);
166  Type rsrcType =
167  LLVM::LLVMPointerType::get(rewriter.getContext(), addressSpace);
168  Value resource = rewriter.createOrFold<ROCDL::MakeBufferRsrcOp>(
169  loc, rsrcType, basePointer, stride, numRecords, flagsConst);
170  return resource;
171 }

184  ConversionPatternRewriter &rewriter) const override {
185  Location loc = op.getLoc();
186  Value memRef = adaptor.getSource();
187  Value unconvertedMemref = op.getSource();
188  MemRefType memrefType = cast<MemRefType>(unconvertedMemref.getType());
189  MemRefDescriptor descriptor(memRef);
190 
191  DataLayout dataLayout = DataLayout::closest(op);
192  int64_t elementByteWidth =
193  dataLayout.getTypeSizeInBits(memrefType.getElementType()) / 8;
194 
195  int64_t unusedOffset = 0;

197  if (failed(memrefType.getStridesAndOffset(strideVals, unusedOffset)))

230  chipset, adaptor.getCacheSwizzleStride(), /*addressSpace=*/7);
231 
232  Value result = MemRefDescriptor::poison(
233  rewriter, loc,
234  getTypeConverter()->convertType(op.getResult().getType()));
235  SmallVector<int64_t> pos{kAllocatedPtrPosInMemRefDescriptor};
236  result = LLVM::InsertValueOp::create(rewriter, loc, result, fatPtr, pos);
237  result = LLVM::InsertValueOp::create(rewriter, loc, result, fatPtr,
238  kAlignedPtrPosInMemRefDescriptor);
239  result = LLVM::InsertValueOp::create(rewriter, loc, result, offset,
240  kOffsetPosInMemRefDescriptor);
241  if (hasSizes) {
242  result = LLVM::InsertValueOp::create(rewriter, loc, result, sizes,
243  kSizePosInMemRefDescriptor);
244  result = LLVM::InsertValueOp::create(rewriter, loc, result, strides,
245  kStridePosInMemRefDescriptor);
246  }
247  rewriter.replaceOp(op, result);
248  return success();
249  }

253 template <typename GpuOp, typename Intrinsic>
254 struct RawBufferOpLowering : public ConvertOpToLLVMPattern<GpuOp> {
255  RawBufferOpLowering(const LLVMTypeConverter &converter, Chipset chipset)
256  : ConvertOpToLLVMPattern<GpuOp>(converter), chipset(chipset) {}
257 
258  Chipset chipset;
259  static constexpr uint32_t maxVectorOpWidth = 128;
260 
261  LogicalResult
262  matchAndRewrite(GpuOp gpuOp, typename GpuOp::Adaptor adaptor,
263  ConversionPatternRewriter &rewriter) const override {
264  Location loc = gpuOp.getLoc();
265  Value memref = adaptor.getMemref();
266  Value unconvertedMemref = gpuOp.getMemref();
267  MemRefType memrefType = cast<MemRefType>(unconvertedMemref.getType());
268 
269  if (chipset.majorVersion < 9)
270  return gpuOp.emitOpError("raw buffer ops require GCN or higher");
271 
272  Value storeData = adaptor.getODSOperands(0)[0];
273  if (storeData == memref) // no write component to this op
274  storeData = Value();
275  Type wantedDataType;
276  if (storeData)
277  wantedDataType = storeData.getType();
278  else
279  wantedDataType = gpuOp.getODSResults(0)[0].getType();
280 
281  Value atomicCmpData = Value();
282  // Operand index 1 of a load is the indices, trying to read them can crash.
283  if (storeData) {
284  Value maybeCmpData = adaptor.getODSOperands(1)[0];
285  if (maybeCmpData != memref)
286  atomicCmpData = maybeCmpData;
287  }
288 
289  Type llvmWantedDataType = this->typeConverter->convertType(wantedDataType);
290 
291  Type i32 = rewriter.getI32Type();
292 
293  // Get the type size in bytes.
294  DataLayout dataLayout = DataLayout::closest(gpuOp);
295  int64_t elementByteWidth =
296  dataLayout.getTypeSizeInBits(memrefType.getElementType()) / 8;
297  Value byteWidthConst = createI32Constant(rewriter, loc, elementByteWidth);
298 
299  // If we want to load a vector<NxT> with total size <= 32
300  // bits, use a scalar load and bitcast it. Similarly, if bitsize(T) < 32
301  // and the total load size is >= 32, use a vector load of N / (bitsize(T) /
302  // 32) x i32 and bitcast. Also, the CAS intrinsic requires integer operands,
303  // so bitcast any floats to integers.
304  Type llvmBufferValType = llvmWantedDataType;
305  if (atomicCmpData) {
306  if (auto floatType = dyn_cast<FloatType>(wantedDataType))
307  llvmBufferValType = this->getTypeConverter()->convertType(
308  rewriter.getIntegerType(floatType.getWidth()));
309  }
310  if (auto dataVector = dyn_cast<VectorType>(wantedDataType)) {
311  uint32_t vecLen = dataVector.getNumElements();
312  uint32_t elemBits =
313  dataLayout.getTypeSizeInBits(dataVector.getElementType());
314  uint32_t totalBits = elemBits * vecLen;
315  bool usePackedFp16 =
316  isa_and_present<RawBufferAtomicFaddOp>(*gpuOp) && vecLen == 2;
317  if (totalBits > maxVectorOpWidth)
318  return gpuOp.emitOpError(
319  "Total width of loads or stores must be no more than " +
320  Twine(maxVectorOpWidth) + " bits, but we call for " +
321  Twine(totalBits) +
322  " bits. This should've been caught in validation");
323  if (!usePackedFp16 && elemBits < 32) {
324  if (totalBits > 32) {
325  if (totalBits % 32 != 0)
326  return gpuOp.emitOpError("Load or store of more than 32-bits that "
327  "doesn't fit into words. Can't happen\n");
328  llvmBufferValType = this->typeConverter->convertType(
329  VectorType::get(totalBits / 32, i32));
330  } else {
331  llvmBufferValType = this->typeConverter->convertType(
332  rewriter.getIntegerType(totalBits));
333  }
334  }
335  }
336  if (auto vecType = dyn_cast<VectorType>(llvmBufferValType)) {
337  // Buffer intrinsics doesn't support 1-element vectors, cast them to
338  // scalars.
339  if (vecType.getNumElements() == 1)
340  llvmBufferValType = vecType.getElementType();
341  }
342 
343  SmallVector<Value, 6> args;
344  if (storeData) {
345  if (llvmBufferValType != llvmWantedDataType) {
346  Value castForStore = LLVM::BitcastOp::create(
347  rewriter, loc, llvmBufferValType, storeData);
348  args.push_back(castForStore);
349  } else {
350  args.push_back(storeData);
351  }
352  }
353 
354  if (atomicCmpData) {
355  if (llvmBufferValType != llvmWantedDataType) {
356  Value castForCmp = LLVM::BitcastOp::create(
357  rewriter, loc, llvmBufferValType, atomicCmpData);
358  args.push_back(castForCmp);
359  } else {
360  args.push_back(atomicCmpData);
361  }
362  }
363 
364  // Construct buffer descriptor from memref, attributes
365  int64_t offset = 0;
366  SmallVector<int64_t, 5> strides;
367  if (failed(memrefType.getStridesAndOffset(strides, offset)))
368  return gpuOp.emitOpError("Can't lower non-stride-offset memrefs");
369 
370  MemRefDescriptor memrefDescriptor(memref);
371 
372  Value ptr = memrefDescriptor.bufferPtr(
373  rewriter, loc, *this->getTypeConverter(), memrefType);
374  Value numRecords = getNumRecords(
375  rewriter, loc, memrefType, memrefDescriptor, strides, elementByteWidth);
376  Value resource = makeBufferRsrc(rewriter, loc, ptr, numRecords,
377  adaptor.getBoundsCheck(), chipset);
378  args.push_back(resource);
379 
380  // Indexing (voffset)
381  Value voffset = getLinearIndexI32(rewriter, loc, memrefDescriptor,
382  adaptor.getIndices(), strides);
383  if (std::optional<int32_t> indexOffset = adaptor.getIndexOffset();
384  indexOffset && *indexOffset > 0) {
385  Value extraOffsetConst = createI32Constant(rewriter, loc, *indexOffset);
386  voffset = voffset ? LLVM::AddOp::create(rewriter, loc, voffset,
387  extraOffsetConst)
388  : extraOffsetConst;
389  }
390  voffset = LLVM::MulOp::create(rewriter, loc, voffset, byteWidthConst);
391  args.push_back(voffset);
392 
393  // SGPR offset.
394  Value sgprOffset = adaptor.getSgprOffset();
395  if (!sgprOffset)
396  sgprOffset = createI32Constant(rewriter, loc, 0);
397  sgprOffset = LLVM::MulOp::create(rewriter, loc, sgprOffset, byteWidthConst);
398  args.push_back(sgprOffset);
399 
400  // bit 0: GLC = 0 (atomics drop value, less coherency)
401  // bits 1-2: SLC, DLC = 0 (similarly)
402  // bit 3: swizzled (0 for raw)
403  args.push_back(createI32Constant(rewriter, loc, 0));
404 
405  llvm::SmallVector<Type, 1> resultTypes(gpuOp->getNumResults(),
406  llvmBufferValType);
407  Operation *lowered = Intrinsic::create(rewriter, loc, resultTypes, args,
408  ArrayRef<NamedAttribute>());
409  if (lowered->getNumResults() == 1) {
410  Value replacement = lowered->getResult(0);
411  if (llvmBufferValType != llvmWantedDataType) {
412  replacement = LLVM::BitcastOp::create(rewriter, loc, llvmWantedDataType,
413  replacement);
414  }
415  rewriter.replaceOp(gpuOp, replacement);
416  } else {
417  rewriter.eraseOp(gpuOp);
418  }
419  return success();
420  }

425 /// Vmcnt, Expcnt and Lgkmcnt are decoded as follows:
426 /// Vmcnt = Waitcnt[3:0] (pre-gfx9)
427 /// Vmcnt = Waitcnt[15:14,3:0] (gfx9,10)
428 /// Vmcnt = Waitcnt[15:10] (gfx11)
429 /// Expcnt = Waitcnt[6:4] (pre-gfx11)
430 /// Expcnt = Waitcnt[2:0] (gfx11)
431 /// Lgkmcnt = Waitcnt[11:8] (pre-gfx10)
432 /// Lgkmcnt = Waitcnt[13:8] (gfx10)
433 /// Lgkmcnt = Waitcnt[9:4] (gfx11)
434 static FailureOr<unsigned> encodeWaitcnt(Chipset chipset, unsigned vmcnt,
435  unsigned expcnt, unsigned lgkmcnt) {
436  if (chipset.majorVersion < 9) {
437  vmcnt = std::min(15u, vmcnt);
438  expcnt = std::min(7u, expcnt);
439  lgkmcnt = std::min(15u, lgkmcnt);
440  return vmcnt | (expcnt << 4) | (lgkmcnt << 8);
441  }
442  if (chipset.majorVersion == 9) {
443  vmcnt = std::min(63u, vmcnt);
444  expcnt = std::min(7u, expcnt);
445  lgkmcnt = std::min(15u, lgkmcnt);
446  unsigned lowBits = vmcnt & 0xF;
447  unsigned highBits = (vmcnt >> 4) << 14;
448  unsigned otherCnts = (expcnt << 4) | (lgkmcnt << 8);
449  return lowBits | highBits | otherCnts;
450  }
451  if (chipset.majorVersion == 10) {
452  vmcnt = std::min(63u, vmcnt);
453  expcnt = std::min(7u, expcnt);
454  lgkmcnt = std::min(63u, lgkmcnt);
455  unsigned lowBits = vmcnt & 0xF;
456  unsigned highBits = (vmcnt >> 4) << 14;
457  unsigned otherCnts = (expcnt << 4) | (lgkmcnt << 8);
458  return lowBits | highBits | otherCnts;
459  }
460  if (chipset.majorVersion == 11) {
461  vmcnt = std::min(63u, vmcnt);
462  expcnt = std::min(7u, expcnt);
463  lgkmcnt = std::min(63u, lgkmcnt);
464  return (vmcnt << 10) | expcnt | (lgkmcnt << 4);
465  }
466  return failure();
467 }

549  /*asm_string=*/asmStr, constraints, /*has_side_effects=*/true,
550  /*is_align_stack=*/false, LLVM::TailCallKind::None,
551  /*asm_dialect=*/asmDialectAttr,
552  /*operand_attrs=*/ArrayAttr());
553  return success();
554  }
555  if (chipset.majorVersion < 12) {
556  constexpr int32_t ldsOnlyBitsGfx6789 = ~(0x1f << 8);
557  constexpr int32_t ldsOnlyBitsGfx10 = ~(0x3f << 8);
558  // Left in place in case someone disables the inline ASM path or future
559  // chipsets use the same bit pattern.
560  constexpr int32_t ldsOnlyBitsGfx11 = ~(0x3f << 4);
561 
562  int32_t ldsOnlyBits;
563  if (chipset.majorVersion == 11)
564  ldsOnlyBits = ldsOnlyBitsGfx11;
565  else if (chipset.majorVersion == 10)
566  ldsOnlyBits = ldsOnlyBitsGfx10;
567  else if (chipset.majorVersion <= 9)
568  ldsOnlyBits = ldsOnlyBitsGfx6789;
569  else
570  return op.emitOpError(
571  "don't know how to lower this for chipset major version")
572  << chipset.majorVersion;
573 
574  Location loc = op->getLoc();
575  ROCDL::SWaitcntOp::create(rewriter, loc, ldsOnlyBits);
576  rewriter.replaceOpWithNewOp<ROCDL::SBarrierOp>(op);
577  } else {
578  Location loc = op->getLoc();
579  ROCDL::WaitDscntOp::create(rewriter, loc, 0);
580  ROCDL::BarrierSignalOp::create(rewriter, loc, -1);
581  rewriter.replaceOpWithNewOp<ROCDL::BarrierWaitOp>(op, -1);
582  }
583 
584  return success();
585  }

605 /// Converts a MFMA vector operand from MLIR AMDGPU dialect convention to ROCDL
606 /// and LLVM AMDGPU intrinsics convention.
607 ///
608 /// Specifically:
609 /// 1. If the element type is bfloat16, bitcast it to i16 unless rocdl intrinsic
610 /// allows bf16. Newer MFMAs support bf16 types on operand, check
611 /// IntrinsicsAMDGPU.td file for reference.
612 /// 2. If instead we have a more than 64-bit quantity, use a <N / 4 x i32>
613 /// instead, which is what the f8f6f4 intrinsics use.
614 /// 3. If `input` is a vector of N <= 8 bytes, bitcast it to a (N * 8)-bit
615 /// integer.
616 ///
617 /// Note that the type of `input` has already been LLVM type converted:
618 /// therefore 8-bit and smaller floats are represented as their corresponding
619 /// `iN` integers.
620 static Value convertMFMAVectorOperand(ConversionPatternRewriter &rewriter,
621  Location loc, Value input,
622  bool allowBf16 = true) {
623  Type inputType = input.getType();
624  if (auto vectorType = dyn_cast<VectorType>(inputType)) {
625  if (vectorType.getElementType().isBF16() && !allowBf16)
626  return LLVM::BitcastOp::create(
627  rewriter, loc, vectorType.clone(rewriter.getI16Type()), input);
628  if (vectorType.getElementType().isInteger(8) &&
629  vectorType.getNumElements() <= 8)
630  return LLVM::BitcastOp::create(
631  rewriter, loc,
632  rewriter.getIntegerType(vectorType.getNumElements() * 8), input);
633  if (isa<IntegerType>(vectorType.getElementType()) &&
634  vectorType.getElementTypeBitWidth() <= 8) {
635  int64_t numWords = llvm::divideCeil(
636  vectorType.getNumElements() * vectorType.getElementTypeBitWidth(),
637  32);
638  return LLVM::BitcastOp::create(
639  rewriter, loc, VectorType::get(numWords, rewriter.getI32Type()),
640  input);
641  }
642  }
643  return input;
644 }

665 /// Push an input operand. If it is a float type, nothing to do. If it is
666 /// an integer type, then we need to also push its signdness (1 for signed, 0
667 /// for unsigned) and we need to pack the input 16xi8 vector into a 4xi32
668 /// vector (or the 8xi8 vector into a 2xi32 one for gfx12+).
669 /// We also need to convert bfloat inputs to i16 to account for the bfloat
670 /// intrinsics having been defined before the AMD backend supported bfloat. We
671 /// similarly need to pack 8-bit float types into integers as if they were i8
672 /// (which they are for the backend's purposes).
673 static void wmmaPushInputOperand(ConversionPatternRewriter &rewriter,
674  Location loc,
675  const TypeConverter *typeConverter,
676  bool isUnsigned, Value llvmInput,
677  Value mlirInput,
678  SmallVector<Value, 4> &operands) {
679  Type inputType = llvmInput.getType();
680  auto vectorType = dyn_cast<VectorType>(inputType);
681  if (!vectorType) {
682  operands.push_back(llvmInput);
683  return;
684  }
685  Type elemType = vectorType.getElementType();
686 
687  if (elemType.isBF16())
688  llvmInput = LLVM::BitcastOp::create(
689  rewriter, loc, vectorType.clone(rewriter.getI16Type()), llvmInput);
690  if (elemType.getIntOrFloatBitWidth() > 8) {
691  operands.push_back(llvmInput);
692  return;
693  }
694 
695  // We need to check the type of the input before conversion to properly test
696  // for int8. This is because, in LLVM, fp8 type is converted to int8, so the
697  // fp8/int8 information is lost during the conversion process.
698  auto mlirInputType = cast<VectorType>(mlirInput.getType());
699  bool isInputInteger = mlirInputType.getElementType().isInteger();
700  if (isInputInteger) {
701  // if element type is 8-bit signed or unsigned, ignore the isUnsigned flag
702  bool localIsUnsigned = isUnsigned;
703  if (elemType.isUnsignedInteger()) {
704  localIsUnsigned = true;
705  } else if (elemType.isSignedInteger()) {
706  localIsUnsigned = false;
707  }
708  Value sign = createI1Constant(rewriter, loc, !localIsUnsigned);
709  operands.push_back(sign);
710  }
711 
712  int64_t numBits =
713  vectorType.getNumElements() * elemType.getIntOrFloatBitWidth();
714  Type i32 = rewriter.getI32Type();
715  Type intrinsicInType = numBits <= 32
716  ? (Type)rewriter.getIntegerType(numBits)
717  : (Type)VectorType::get(numBits / 32, i32);
718  auto llvmIntrinsicInType = typeConverter->convertType(intrinsicInType);
719  Value castInput = rewriter.createOrFold<LLVM::BitcastOp>(
720  loc, llvmIntrinsicInType, llvmInput);
721  // The wave64-mode 16x16x16 intrinsics that take 4-bit integers only need
722  // (256 / 64) * 4 = 16 bits of input (on gfx12+) but take i32 arguments.
723  // Add in the zeros here.
724  if (numBits < 32)
725  castInput = LLVM::ZExtOp::create(rewriter, loc, i32, castInput);
726  operands.push_back(castInput);
727 }

729 /// Push the output operand. For many cases this is only pushing the output in
730 /// the operand list. But when we have f16 -> f16 or bf16 -> bf16 intrinsics,
731 /// since the same numbers of VGPRs is used, we need to decide if to store the
732 /// result in the upper 16 bits of the VGPRs or in the lower part. To store the
733 /// result in the lower 16 bits, set subwordOffset to 1, otherwise result will
734 /// be stored it in the upper part. The subwordOffset must not be set for gfx12,
735 /// as the instructions have been changed to return fewer registers instead.
736 static void wmmaPushOutputOperand(ConversionPatternRewriter &rewriter,
737  Location loc,
738  const TypeConverter *typeConverter,
739  Value output, int32_t subwordOffset,
740  bool clamp, SmallVector<Value, 4> &operands) {
741  Type inputType = output.getType();
742  auto vectorType = dyn_cast<VectorType>(inputType);
743  Type elemType = vectorType.getElementType();
744  if (elemType.isBF16())
745  output = LLVM::BitcastOp::create(
746  rewriter, loc, vectorType.clone(rewriter.getI16Type()), output);
747  operands.push_back(output);
748  if (elemType.isF16() || elemType.isBF16() || elemType.isInteger(16)) {
749  operands.push_back(createI1Constant(rewriter, loc, subwordOffset));
750  } else if (elemType.isInteger(32)) {
751  operands.push_back(createI1Constant(rewriter, loc, clamp));
752  }
753 }
